 107     def get_tag_value_label(self, key, value):
 108         label = value
 109 
 110         if key == "sentry:user" and value:
 111             if value.startswith("id:"):
 112                 label = value[len("id:") :]
 113             elif value.startswith("email:"):
 114                 label = value[len("email:") :]
 115             elif value.startswith("username:"):
 116                 label = value[len("username:") :]
 117             elif value.startswith("ip:"):
 118                 label = value[len("ip:") :]
 119 
 120         return label

 287     def get_group_tag_keys_and_top_values(
 288         self,
 289         project_id,
 290         group_id,
 291         environment_ids,
 292         keys=None,
 293         value_limit=TOP_VALUES_DEFAULT_LIMIT,
 294         **kwargs
 295     ):
 296 
 297         # only the snuba backend supports multi env, and that overrides this method
 298         if environment_ids and len(environment_ids) > 1:
 299             environment_ids = environment_ids[:1]
 300 
 301         # If keys is unspecified, we will grab all tag keys for this group.
 302         tag_keys = self.get_group_tag_keys(project_id, group_id, environment_ids, keys=keys)
 303 
 304         environment_id = environment_ids[0] if environment_ids else None
 305         for tk in tag_keys:
 306             tk.top_values = self.get_top_group_tag_values(
 307                 project_id, group_id, environment_id, tk.key, limit=value_limit
 308             )
 309             if tk.count is None:
 310                 tk.count = self.get_group_tag_value_count(
 311                     project_id, group_id, environment_id, tk.key
 312                 )
 313 
 314         return tag_keys
