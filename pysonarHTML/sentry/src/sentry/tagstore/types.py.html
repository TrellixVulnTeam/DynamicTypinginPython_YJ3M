  11 class TagType(object):
  12     def __repr__(self):
  13         return "<%s: %s>" % (
  14             type(self).__name__,
  15             ", ".join("%s=%r" % (name, getattr(self, name)) for name in self.__slots__),
  16         )
  17 
  18     def __hash__(self):
  19         return hash(tuple([getattr(self, name) for name in self.__slots__]))
  20 
  21     def __eq__(self, other):
  22         return type(self) == type(other) and all(
  23             getattr(self, name) == getattr(other, name) for name in self.__slots__
  24         )
  25 
  26     def __getstate__(self):
  27         return {name: getattr(self, name) for name in self.__slots__}
  28 
  29     def __setstate__(self, state):
