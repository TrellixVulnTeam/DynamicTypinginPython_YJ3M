  92     def update_subscription(self, user, list_id=None, create=False, **kwargs):
  93         if not list_id:
  94             list_id = self.get_default_list_id()
  95 
  96         if create:
  97             self._subscriptions[user].setdefault(
  98                 list_id, NewsletterSubscription(user, list_id, subscribed=True)
  99             )
 100         self._subscriptions[user][list_id].update(**kwargs)
 101 
 102         return self._subscriptions[user]

 104     def update_subscriptions(self, user, list_ids=None, create=False, **kwargs):
 105         if not list_ids:
 106             list_ids = self.get_default_list_ids()
 107 
 108         for list_id in list_ids:
 109             self.update_subscription(user, list_id, create, **kwargs)
