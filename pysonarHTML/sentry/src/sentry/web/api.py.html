  99 def allow_cors_options(func):
 100     """
 101     Decorator that adds automatic handling of OPTIONS requests for CORS
 102 
 103     If the request is OPTIONS (i.e. pre flight CORS) construct a OK (200) response
 104     in which we explicitly enable the caller and add the custom headers that we support
 105     For other requests just add the appropriate CORS headers
 106 
 107     :param func: the original request handler
 108     :return: a request handler that shortcuts OPTIONS requests and just returns an OK (CORS allowed)
 109     """
 110 
 111     @wraps(func)
 112     def allow_cors_options_wrapper(self, request, *args, **kwargs):
 113 
 114         if request.method == "OPTIONS":
 115             response = HttpResponse(status=200)
 116             response["Access-Control-Max-Age"] = "3600"  # don't ask for options again for 1 hour
 117         else:
 118             response = func(self, request, *args, **kwargs)
 119 
 120         allow = ", ".join(self._allowed_methods())
 121         response["Allow"] = allow
 122         response["Access-Control-Allow-Methods"] = allow
 123         response["Access-Control-Allow-Headers"] = (
 124             "X-Sentry-Auth, X-Requested-With, Origin, Accept, "
 125             "Content-Type, Authentication, Authorization, Content-Encoding"
 126         )
 127         response["Access-Control-Expose-Headers"] = "X-Sentry-Error, Retry-After"
 128 
 129         if request.META.get("HTTP_ORIGIN") == "null":
 130             origin = "null"  # if ORIGIN header is explicitly specified as 'null' leave it alone
 131         else:
 132             origin = origin_from_request(request)
 133 
 134         if origin is None or origin == "null":
 135             response["Access-Control-Allow-Origin"] = "*"
 136         else:
 137             response["Access-Control-Allow-Origin"] = origin
 138 
 139         return response
 140 
 141     return allow_cors_options_wrapper

 155 def api(func):
 156     @wraps(func)
 157     def wrapped(request, *args, **kwargs):
 158         data = func(request, *args, **kwargs)
 159         if request.is_ajax():
 160             response = HttpResponse(data)
 161             response["Content-Type"] = "application/json"
 162         else:
 163             ref = request.META.get("HTTP_REFERER")
 164             if ref is None or not is_same_domain(ref, request.build_absolute_uri()):
 165                 ref = reverse("sentry")
 166             return HttpResponseRedirect(ref)
 167         return response
 168 
 169     return wrapped

 172 def _get_project_id_from_request(project_id, request, auth_helper_cls, helper):
 173     """
 174     Tries to return the project id (as a string) from the request params or from the auth info
 175 
 176     :param project_id: the project id from the url (or None if not specified)
 177     :param request: the HTTP request
 178     :param auth_helper_cls: Authentication helper class (from APIView)
 179     :param helper: client API helper
 180     :return: the project id (as string) if found raises if not found
 181 
 182     :raises APIUnauthorized if bad Authorization header detected or the key is not usable (e.g. disabled)
 183     """
 184     if project_id is not None:
 185         # we have an explicit project id, just return it
 186         return six.text_type(project_id)
 187     else:  # look in the authentication information for the project id
 188         auth = auth_helper_cls.auth_from_request(request)
 189         return helper.project_id_from_auth(auth)

 192 def process_event(event_manager, project, key, remote_addr, helper, attachments, project_config):
 193     event_received.send_robust(ip=remote_addr, project=project, sender=process_event)
 194 
 195     start_time = time()
 196 
 197     data = event_manager.get_data()
 198     should_filter, filter_reason = event_manager.should_filter()
 199     del event_manager
 200 
 201     event_id = data["event_id"]
 202     data_category = DataCategory.from_event_type(data.get("type"))
 203 
 204     if should_filter:
 205         track_outcome(
 206             project_config.organization_id,
 207             project_config.project_id,
 208             key.id,
 209             Outcome.FILTERED,
 210             filter_reason,
 211             event_id=event_id,
 212             category=data_category,
 213         )
 214         metrics.incr("events.blacklisted", tags={"reason": filter_reason}, skip_internal=False)
 215 
 216         # relay will no longer be able to provide information about filter
 217         # status so to see the impact we're adding a way to turn on relay
 218         # like behavior here.
 219         if options.get("store.lie-about-filter-status"):
 220             return event_id
 221 
 222         raise APIForbidden("Event dropped due to filter: %s" % (filter_reason,))
 223 
 224     # TODO: improve this API (e.g. make RateLimit act on __ne__)
 225     rate_limit = safe_execute(
 226         quotas.is_rate_limited, project=project, key=key, _with_transaction=False
 227     )
 228     if isinstance(rate_limit, bool):
 229         rate_limit = RateLimit(is_limited=rate_limit, retry_after=None)
 230 
 231     # XXX(dcramer): when the rate limiter fails we drop events to ensure
 232     # it cannot cascade
 233     if rate_limit is None or rate_limit.is_limited:
 234         if rate_limit is None:
 235             api_logger.debug("Dropped event due to error with rate limiter")
 236 
 237         reason = rate_limit.reason_code if rate_limit else None
 238         track_outcome(
 239             project_config.organization_id,
 240             project_config.project_id,
 241             key.id,
 242             Outcome.RATE_LIMITED,
 243             reason,
 244             event_id=event_id,
 245             category=data_category,
 246         )
 247         metrics.incr("events.dropped", tags={"reason": reason or "unknown"}, skip_internal=False)
 248 
 249         if rate_limit is not None:
 250             raise APIRateLimited(rate_limit.retry_after)
 251 
 252     # TODO(dcramer): ideally we'd only validate this if the event_id was
 253     # supplied by the user
 254     cache_key = "ev:%s:%s" % (project_config.project_id, event_id)
 255 
 256     # XXX(markus): I believe this code is extremely broken:
 257     #
 258     # * it practically uses memcached in prod which has no consistency
 259     #   guarantees (no idea how we don't run into issues there)
 260     #
 261     # * a TTL of 1h basically doesn't guarantee any deduplication at all. It
 262     #   just guarantees a good error message... for one hour.
 263     if cache.get(cache_key) is not None:
 264         track_outcome(
 265             project_config.organization_id,
 266             project_config.project_id,
 267             key.id,
 268             Outcome.INVALID,
 269             "duplicate",
 270             event_id=event_id,
 271             category=data_category,
 272         )
 273         raise APIForbidden("An event with the same ID already exists (%s)" % (event_id,))
 274 
 275     data = scrub_data(project_config, dict(data))
 276 
 277     # mutates data (strips a lot of context if not queued)
 278     helper.insert_data_to_database(data, start_time=start_time, attachments=attachments)
 279 
 280     cache.set(cache_key, "", 60 * 60)  # Cache for 1 hour
 281 
 282     api_logger.debug("New event received (%s)", event_id)
 283 
 284     event_accepted.send_robust(ip=remote_addr, data=data, project=project, sender=process_event)
 285 
 286     return event_id

 292     def _get_project_from_id(self, project_id):
 293         if not project_id:
 294             return
 295         if not project_id.isdigit():
 296             track_outcome(0, 0, None, Outcome.INVALID, "project_id")
 297             raise APIError("Invalid project_id: %r" % project_id)
 298         try:
 299             project = Project.objects.get_from_cache(id=project_id)
 300         except Project.DoesNotExist:
 301             track_outcome(0, 0, None, Outcome.INVALID, "project_id")
 302             raise APIError("Invalid project_id: %r" % project_id)
 303         else:
 304             if project.status != ObjectStatus.VISIBLE:
 305                 track_outcome(0, 0, None, Outcome.INVALID, "project_id")
 306                 raise APIError("Invalid project_id: %r" % project_id)
 307             return project

 309     def _parse_header(self, request, project_config):
 310         auth = self.auth_helper_cls.auth_from_request(request)
 311 
 312         if auth.version not in PROTOCOL_VERSIONS:
 313             track_outcome(
 314                 project_config.organization_id,
 315                 project_config.project_id,
 316                 None,
 317                 Outcome.INVALID,
 318                 "auth_version",
 319             )
 320             raise APIError(
 321                 "Client using unsupported server protocol version (%r)"
 322                 % six.text_type(auth.version or "")
 323             )
 324 
 325         if not auth.client:
 326             track_outcome(
 327                 project_config.organization_id,
 328                 project_config.project_id,
 329                 None,
 330                 Outcome.INVALID,
 331                 "auth_client",
 332             )
 333             raise APIError("Client did not send 'client' identifier")
 334 
 335         return auth

 337     def _publish_to_kafka(self, request, project_config):
 338         """
 339         Sends raw event data to Kafka for later offline processing.
 340         """
 341         try:
 342             raw_event_sample_rate = options.get("kafka-publisher.raw-event-sample-rate")
 343 
 344             # Early return if sampling is disabled
 345             if raw_event_sample_rate == 0:
 346                 return
 347 
 348             # This may fail when we e.g. send a multipart form. We ignore those errors for now.
 349             data = request.body
 350 
 351             if not data or len(data) > options.get("kafka-publisher.max-event-size"):
 352                 return
 353 
 354             # Sampling
 355             if random.random() >= raw_event_sample_rate:
 356                 return
 357 
 358             # We want to send only serializable items from request.META
 359             meta = {}
 360             for key, value in request.META.items():
 361                 try:
 362                     json.dumps([key, value])
 363                     meta[key] = value
 364                 except (TypeError, ValueError):
 365                     pass
 366 
 367             meta["SENTRY_API_VIEW_NAME"] = self.__class__.__name__
 368 
 369             kafka_publisher.publish(
 370                 channel=getattr(settings, "KAFKA_RAW_EVENTS_PUBLISHER_TOPIC", "raw-store-events"),
 371                 value=json.dumps([meta, base64.b64encode(data), project_config.to_dict()]),
 372             )
 373         except Exception as e:
 374             logger.debug("Cannot publish event to Kafka: {}".format(six.text_type(e)))

 376     @csrf_exempt
 377     @never_cache
 378     @allow_cors_options
 379     def dispatch(self, request, project_id=None, *args, **kwargs):
 380         helper = None
 381         try:
 382             helper = ClientApiHelper(
 383                 agent=request.META.get("HTTP_USER_AGENT"),
 384                 project_id=project_id,
 385                 ip_address=request.META["REMOTE_ADDR"],
 386             )
 387 
 388             # if the project id is not directly specified get it from the authentication information
 389             project_id = _get_project_id_from_request(
 390                 project_id, request, self.auth_helper_cls, helper
 391             )
 392 
 393             project = self._get_project_from_id(six.text_type(project_id))
 394 
 395             # Explicitly bind Organization so we don't implicitly query it later
 396             # this just allows us to comfortably assure that `project.organization` is safe.
 397             # This also allows us to pull the object from cache, instead of being
 398             # implicitly fetched from database.
 399             project.organization = Organization.objects.get_from_cache(id=project.organization_id)
 400 
 401             # XXX: This never returns a disabled project since visibility of the
 402             # project is already verified in `self._get_project_from_id`.
 403             project_config = get_project_config(project)
 404 
 405             helper.context.bind_project(project_config.project)
 406 
 407             if kafka_publisher is not None:
 408                 self._publish_to_kafka(request, project_config)
 409 
 410             origin = self.auth_helper_cls.origin_from_request(request)
 411 
 412             response = self._dispatch(
 413                 request, helper, project_config, origin=origin, *args, **kwargs
 414             )
 415         except APIError as e:
 416             context = {"error": force_bytes(e.msg, errors="replace")}
 417             if e.name:
 418                 context["error_name"] = e.name
 419 
 420             response = HttpResponse(
 421                 json.dumps(context), content_type="application/json", status=e.http_status
 422             )
 423             # Set X-Sentry-Error as in many cases it is easier to inspect the headers
 424             response["X-Sentry-Error"] = context["error"]
 425 
 426             if isinstance(e, APIRateLimited) and e.retry_after is not None:
 427                 response["Retry-After"] = six.text_type(int(math.ceil(e.retry_after)))
 428 
 429         except Exception as e:
 430             # TODO(dcramer): test failures are not outputting the log message
 431             # here
 432             if settings.DEBUG:
 433                 content = traceback.format_exc()
 434             else:
 435                 content = ""
 436             logger.exception(e)
 437             response = HttpResponse(content, content_type="text/plain", status=500)
 438 
 439         # TODO(dcramer): it'd be nice if we had an incr_multi method so
 440         # tsdb could optimize this
 441         metrics.incr("client-api.all-versions.requests", skip_internal=False)
 442         metrics.incr(
 443             "client-api.all-versions.responses.%s" % (response.status_code,), skip_internal=False
 444         )
 445         metrics.incr(
 446             "client-api.all-versions.responses.%sxx" % (six.text_type(response.status_code)[0],),
 447             skip_internal=False,
 448         )
 449 
 450         if helper is not None and helper.context is not None and helper.context.version:
 451             metrics.incr("client-api.v%s.requests" % (helper.context.version,), skip_internal=False)
 452             metrics.incr(
 453                 "client-api.v%s.responses.%s" % (helper.context.version, response.status_code),
 454                 skip_internal=False,
 455             )
 456             metrics.incr(
 457                 "client-api.v%s.responses.%sxx"
 458                 % (helper.context.version, six.text_type(response.status_code)[0]),
 459                 skip_internal=False,
 460             )
 461 
 462         return response

 464     def _dispatch(self, request, helper, project_config, origin=None, *args, **kwargs):
 465         request.user = AnonymousUser()
 466 
 467         project = project_config.project
 468         config = project_config.config
 469         allowed = config.get("allowedDomains")
 470 
 471         if origin is not None:
 472             if not is_valid_origin(origin, allowed=allowed):
 473                 track_outcome(
 474                     project_config.organization_id,
 475                     project_config.project_id,
 476                     None,
 477                     Outcome.INVALID,
 478                     FilterStatKeys.CORS,
 479                 )
 480                 raise APIForbidden("Invalid origin: %s" % (origin,))
 481 
 482         auth = self._parse_header(request, project_config)
 483 
 484         key = helper.project_key_from_auth(auth)
 485 
 486         # Legacy API was /api/store/ and the project ID was only available elsewhere
 487         if six.text_type(key.project_id) != six.text_type(project_config.project_id):
 488             raise APIError("Two different projects were specified")
 489 
 490         helper.context.bind_auth(auth)
 491 
 492         response = super(APIView, self).dispatch(
 493             request=request,
 494             project=project,
 495             auth=auth,
 496             helper=helper,
 497             key=key,
 498             project_config=project_config,
 499             **kwargs
 500         )
 501         return response

 521 class StoreView(APIView):
 522     """
 523     The primary endpoint for storing new events.
 524 
 525     This will validate the client's authentication and data, and if
 526     successful pass on the payload to the internal database handler.
 527 
 528     Authentication works in three flavors:
 529 
 530     1. Explicit signed requests
 531 
 532        These are implemented using the documented signed request protocol, and
 533        require an authentication header which is signed using with the project
 534        member's secret key.
 535 
 536     2. CORS Secured Requests
 537 
 538        Generally used for communications with client-side platforms (such as
 539        JavaScript in the browser), they require a standard header, excluding
 540        the signature and timestamp requirements, and must be listed in the
 541        origins for the given project (or the global origins).
 542 
 543     3. Implicit trusted requests
 544 
 545        Used by the Sentry core, they are only available from same-domain requests
 546        and do not require any authentication information. They only require that
 547        the user be authenticated, and a project_id be sent in the GET variables.
 548 
 549     """
 550 
 551     type_name = "store"
 552 
 553     def post(self, request, **kwargs):
 554         try:
 555             data = request.body
 556         except Exception as e:
 557             logger.exception(e)
 558             # We were unable to read the body.
 559             # This would happen if a request were submitted
 560             # as a multipart form for example, where reading
 561             # body yields an Exception. There's also not a more
 562             # sane exception to catch here. This will ultimately
 563             # bubble up as an APIError.
 564             data = None
 565 
 566         event_id = self.process(request, data=data, **kwargs)
 567         return HttpResponse(json.dumps({"id": event_id}), content_type="application/json")
 568 
 569     def get(self, request, **kwargs):
 570         data = request.GET.get("sentry_data", "")
 571         event_id = self.process(request, data=data, **kwargs)
 572 
 573         # Return a simple 1x1 gif for browser so they don't throw a warning
 574         response = HttpResponse(PIXEL, "image/gif")
 575         response["X-Sentry-ID"] = event_id
 576         return response
 577 
 578     def pre_normalize(self, data, helper):
 579         """Mutate the given EventManager. Hook for subtypes of StoreView (CSP)"""
 580         pass
 581 
 582     def process(
 583         self, request, project, key, auth, helper, data, project_config, attachments=None, **kwargs
 584     ):
 585         disable_transaction_events()
 586         metrics.incr("events.total", skip_internal=False)
 587 
 588         project_id = project_config.project_id
 589         organization_id = project_config.organization_id
 590 
 591         if not data:
 592             track_outcome(organization_id, project_id, key.id, Outcome.INVALID, "no_data")
 593             raise APIError("No JSON data was found")
 594 
 595         remote_addr = request.META["REMOTE_ADDR"]
 596 
 597         event_manager = EventManager(
 598             data,
 599             project=project,
 600             key=key,
 601             auth=auth,
 602             client_ip=remote_addr,
 603             user_agent=helper.context.agent,
 604             version=auth.version,
 605             content_encoding=request.META.get("HTTP_CONTENT_ENCODING", ""),
 606             project_config=project_config,
 607         )
 608         del data
 609 
 610         self.pre_normalize(event_manager, helper)
 611 
 612         try:
 613             event_manager.normalize()
 614         except ProcessingErrorInvalidTransaction as e:
 615             track_outcome(
 616                 organization_id,
 617                 project_id,
 618                 key.id,
 619                 Outcome.INVALID,
 620                 "invalid_transaction",
 621                 category=DataCategory.TRANSACTION,
 622             )
 623             raise APIError(six.text_type(e).split("\n", 1)[0])
 624 
 625         data = event_manager.get_data()
 626         dict_data = dict(data)
 627         data_size = len(json.dumps(dict_data))
 628 
 629         if data_size > 10000000:
 630             metrics.timing("events.size.rejected", data_size)
 631             track_outcome(
 632                 organization_id,
 633                 project_id,
 634                 key.id,
 635                 Outcome.INVALID,
 636                 "too_large",
 637                 event_id=dict_data.get("event_id"),
 638                 category=DataCategory.from_event_type(dict_data.get("type")),
 639             )

 736     def post(self, request, project, project_config, **kwargs):
 737         # Minidump request payloads do not have the same structure as usual
 738         # events from other SDKs. The minidump can either be transmitted as
 739         # request body, or as `upload_file_minidump` in a multipart formdata
 740         # request. Optionally, an event payload can be sent in the `sentry` form
 741         # field, either as JSON or as nested form data.
 742 
 743         request_files = request.FILES or {}
 744         content_type = request.META.get("CONTENT_TYPE")
 745 
 746         # Track these submissions statically as ERROR. Relay infers properly.
 747         data_category = DataCategory.ERROR
 748 
 749         if content_type in self.dump_types:
 750             minidump = io.BytesIO(request.body)
 751             minidump_name = "Minidump"
 752             data = {}
 753         else:
 754             minidump = request_files.get("upload_file_minidump")
 755             minidump_name = minidump and minidump.name or None
 756 
 757             if any(key.startswith("sentry[") for key in request.POST):
 758                 # First, try to parse the nested form syntax `sentry[key][key]`
 759                 # This is required for the Breakpad client library, which only
 760                 # supports string values of up to 64 characters.
 761                 extra = parser.parse(request.POST.urlencode())
 762                 data = extra.pop("sentry", {})
 763             else:
 764                 # Custom clients can submit longer payloads and should JSON
 765                 # encode event data into the optional `sentry` field.
 766                 extra = request.POST.dict()
 767                 json_data = extra.pop("sentry", None)
 768                 try:
 769                     data = json.loads(json_data) if json_data else {}
 770                 except ValueError:
 771                     data = {}
 772 
 773             if not isinstance(data, dict):
 774                 data = {}
 775 
 776             # Merge additional form fields from the request with `extra` data
 777             # from the event payload and set defaults for processing. This is
 778             # sent by clients like Breakpad or Crashpad.
 779             extra.update(data.get("extra") or ())
 780             data["extra"] = extra
 781 
 782         if not minidump:
 783             track_outcome(
 784                 project_config.organization_id,
 785                 project_config.project_id,
 786                 None,
 787                 Outcome.INVALID,
 788                 "missing_minidump_upload",
 789                 category=data_category,
 790             )
 791             raise APIError("Missing minidump upload")
 792 
 793         # Breakpad on linux sometimes stores the entire HTTP request body as
 794         # dump file instead of just the minidump. The Electron SDK then for
 795         # example uploads a multipart formdata body inside the minidump file.
 796         # It needs to be re-parsed, to extract the actual minidump before
 797         # continuing.
 798         minidump.seek(0)
 799         if minidump.read(2) == b"--":
 800             # The remaining bytes of the first line are the form boundary. We
 801             # have already read two bytes, the remainder is the form boundary
 802             # (excluding the initial '--').
 803             boundary = minidump.readline().rstrip()
 804             minidump.seek(0)
 805 
 806             # Next, we have to fake a HTTP request by specifying the form
 807             # boundary and the content length, or otherwise Django will not try
 808             # to parse our form body. Also, we need to supply new upload
 809             # handlers since they cannot be reused from the current request.
 810             meta = {
 811                 "CONTENT_TYPE": b"multipart/form-data; boundary=%s" % boundary,
 812                 "CONTENT_LENGTH": minidump.size,
 813             }
 814             handlers = [
 815                 uploadhandler.load_handler(handler, request)
 816                 for handler in settings.FILE_UPLOAD_HANDLERS
 817             ]
 818 
 819             _, inner_files = MultiPartParser(meta, minidump, handlers).parse()
 820             try:
 821                 minidump = inner_files["upload_file_minidump"]
 822                 minidump_name = minidump.name
 823             except KeyError:
 824                 track_outcome(
 825                     project_config.organization_id,
 826                     project_config.project_id,
 827                     None,
 828                     Outcome.INVALID,
 829                     "missing_minidump_upload",
 830                     category=data_category,
 831                 )
 832                 raise APIError("Missing minidump upload")
 833 
 834         minidump.seek(0)
 835         if minidump.read(4) != "MDMP":
 836             track_outcome(
 837                 project_config.organization_id,
 838                 project_config.project_id,
 839                 None,
 840                 Outcome.INVALID,
 841                 "invalid_minidump",
 842                 category=data_category,
 843             )
 844             raise APIError("Uploaded file was not a minidump")
 845 
 846         # Always store the minidump in attachments so we can access it during
 847         # processing, regardless of the event-attachments feature. This is
 848         # required to process the minidump with debug information.
 849         attachments = []
 850 
 851         # The minidump attachment is special. It has its own attachment type to
 852         # distinguish it from regular attachments for processing. Also, it might
 853         # not be part of `request_files` if it has been uploaded as raw request
 854         # body instead of a multipart formdata request.
 855         minidump.seek(0)
 856         attachments.append(
 857             CachedAttachment(
 858                 name=minidump_name,
 859                 content_type="application/octet-stream",
 860                 data=minidump.read(),
 861                 type=MINIDUMP_ATTACHMENT_TYPE,
 862             )
 863         )
 864 
 865         # Append all other files as generic attachments.
 866         # RaduW 4 Jun 2019 always sent attachments for minidump (does not use
 867         # event-attachments feature)
 868         for name, file in six.iteritems(request_files):
 869             if name == "upload_file_minidump":
 870                 continue
 871 
 872             # Known attachment: msgpack event
 873             if name == "__sentry-event":
 874                 merge_attached_event(file, data)
 875                 continue
 876             if name in ("__sentry-breadcrumb1", "__sentry-breadcrumb2"):
 877                 merge_attached_breadcrumbs(file, data)
 878                 continue
 879 
 880             # Add any other file as attachment
 881             attachments.append(CachedAttachment.from_upload(file))
 882 
 883         # Assign our own UUID so we can track this minidump. We cannot trust
 884         # the uploaded filename, and if reading the minidump fails there is
 885         # no way we can ever retrieve the original UUID from the minidump.
 886         event_id = data.get("event_id") or uuid.uuid4().hex
 887         data["event_id"] = event_id
 888 
 889         # Write a minimal event payload that is required to kick off native
 890         # event processing. It is also used as fallback if processing of the
 891         # minidump fails.
 892         # NB: This occurs after merging attachments to overwrite potentially
 893         # contradicting payloads transmitted in __sentry_event.
 894         write_minidump_placeholder(data)
 895 
 896         event_id = self.process(
 897             request,
 898             attachments=attachments,
 899             data=data,
 900             project=project,
 901             project_config=project_config,
 902             **kwargs
 903         )
 904 
 905         # Return the formatted UUID of the generated event. This is
 906         # expected by the Electron http uploader on Linux and doesn't
 907         # break the default Breakpad client library.
 908         return HttpResponse(six.text_type(uuid.UUID(event_id)), content_type="text/plain")
