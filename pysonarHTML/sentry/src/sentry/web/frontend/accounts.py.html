  42 def recover(request):
  43     from sentry.app import ratelimiter
  44 
  45     extra = {
  46         "ip_address": request.META["REMOTE_ADDR"],
  47         "user_agent": request.META.get("HTTP_USER_AGENT"),
  48     }
  49 
  50     if request.method == "POST" and ratelimiter.is_limited(
  51         u"accounts:recover:{}".format(extra["ip_address"]),
  52         limit=5,
  53         window=60,  # 5 per minute should be enough for anyone
  54     ):
  55         logger.warning("recover.rate-limited", extra=extra)
  56 
  57         return HttpResponse(
  58             "You have made too many password recovery attempts. Please try again later.",
  59             content_type="text/plain",
  60             status=429,
  61         )
  62 
  63     prefill = {"user": request.GET.get("email")}
  64 
  65     form = RecoverPasswordForm(request.POST or None, initial=prefill)
  66     extra["user_recovered"] = form.data.get("user")
  67 
  68     if form.is_valid():
  69         email = form.cleaned_data["user"]
  70         if email:
  71             password_hash = LostPasswordHash.for_user(email)
  72             password_hash.send_email(request)
  73 
  74             extra["passwordhash_id"] = password_hash.id
  75             extra["user_id"] = password_hash.user_id
  76 
  77             logger.info("recover.sent", extra=extra)
  78 
  79         tpl = "sentry/account/recover/sent.html"
  80         context = {"email": email}
  81 
  82         return render_to_response(tpl, context, request)
  83 
  84     if form._errors:
  85         logger.warning("recover.error", extra=extra)
  86 
  87     tpl = "sentry/account/recover/index.html"
  88     context = {"form": form}
  89 
  90     return render_to_response(tpl, context, request)

  93 def recover_confirm(request, user_id, hash, mode="recover"):
  94     try:
  95         password_hash = LostPasswordHash.objects.get(user=user_id, hash=hash)
  96         if not password_hash.is_valid():
  97             password_hash.delete()
  98             raise LostPasswordHash.DoesNotExist
  99         user = password_hash.user
 100 
 101     except LostPasswordHash.DoesNotExist:
 102         return render_to_response(u"sentry/account/{}/{}.html".format(mode, "failure"), {}, request)
 103 
 104     if request.method == "POST":
 105         form = ChangePasswordRecoverForm(request.POST)
 106         if form.is_valid():
 107             with transaction.atomic():
 108                 user.set_password(form.cleaned_data["password"])
 109                 user.refresh_session_nonce(request)
 110                 user.save()
 111 
 112                 # Ugly way of doing this, but Django requires the backend be set
 113                 user = authenticate(username=user.username, password=form.cleaned_data["password"])
 114 
 115                 # Only log the user in if there is no two-factor on the
 116                 # account.
 117                 if not Authenticator.objects.user_has_2fa(user):
 118                     login_user(request, user)
 119 
 120                 password_hash.delete()
 121 
 122                 capture_security_activity(
 123                     account=user,
 124                     type="password-changed",
 125                     actor=request.user,
 126                     ip_address=request.META["REMOTE_ADDR"],
 127                     send_email=True,
 128                 )
 129 
 130             return login_redirect(request)
 131     else:
 132         form = ChangePasswordRecoverForm()
 133 
 134     return render_to_response(
 135         u"sentry/account/{}/{}.html".format(mode, "confirm"), {"form": form}, request
 136     )

 146 def start_confirm_email(request):
 147     from sentry.app import ratelimiter
 148 
 149     if ratelimiter.is_limited(
 150         u"auth:confirm-email:{}".format(request.user.id),
 151         limit=10,
 152         window=60,  # 10 per minute should be enough for anyone
 153     ):
 154         return HttpResponse(
 155             "You have made too many email confirmation requests. Please try again later.",
 156             content_type="text/plain",
 157             status=429,
 158         )
 159 
 160     if "primary-email" in request.POST:
 161         email = request.POST.get("email")
 162         try:
 163             email_to_send = UserEmail.objects.get(user=request.user, email=email)
 164         except UserEmail.DoesNotExist:
 165             msg = _("There was an error confirming your email.")
 166             level = messages.ERROR
 167         else:
 168             request.user.send_confirm_email_singular(email_to_send)
 169             msg = _("A verification email has been sent to %s.") % (email)
 170             level = messages.SUCCESS
 171         messages.add_message(request, level, msg)
 172         return HttpResponseRedirect(reverse("sentry-account-settings"))
 173     elif request.user.has_unverified_emails():
 174         request.user.send_confirm_emails()
 175         unverified_emails = [e.email for e in request.user.get_unverified_emails()]
 176         msg = _("A verification email has been sent to %s.") % (", ").join(unverified_emails)
 177         for email in unverified_emails:
 178             logger.info(
 179                 "user.email.start_confirm",
 180                 extra={
 181                     "user_id": request.user.id,
 182                     "ip_address": request.META["REMOTE_ADDR"],
 183                     "email": email,
 184                 },
 185             )
 186     else:
 187         msg = _("Your email (%s) has already been verified.") % request.user.email
 188     messages.add_message(request, messages.SUCCESS, msg)
 189     return HttpResponseRedirect(reverse("sentry-account-settings-emails"))

 192 def confirm_email(request, user_id, hash):
 193     msg = _("Thanks for confirming your email")
 194     level = messages.SUCCESS
 195     try:
 196         email = UserEmail.objects.get(user=user_id, validation_hash=hash)
 197         if not email.hash_is_valid():
 198             raise UserEmail.DoesNotExist
 199     except UserEmail.DoesNotExist:
 200         if request.user.is_anonymous() or request.user.has_unverified_emails():
 201             msg = _(
 202                 "There was an error confirming your email. Please try again or "
 203                 "visit your Account Settings to resend the verification email."
 204             )
 205             level = messages.ERROR
 206     else:
 207         email.is_verified = True
 208         email.validation_hash = ""
 209         email.save()
 210         email_verified.send(email=email.email, sender=email)
 211         logger.info(
 212             "user.email.confirm",
 213             extra={
 214                 "user_id": user_id,
 215                 "ip_address": request.META["REMOTE_ADDR"],
 216                 "email": email.email,
 217             },
 218         )
 219     messages.add_message(request, level, msg)
 220     return HttpResponseRedirect(reverse("sentry-account-settings-emails"))

 227 def email_unsubscribe_project(request, project_id):
 228     # For now we only support getting here from the signed link.
 229     if not request.user_from_signed_request:
 230         raise Http404()
 231     try:
 232         project = Project.objects.get(pk=project_id)
 233     except Project.DoesNotExist:
 234         raise Http404()
 235 
 236     if request.method == "POST":
 237         if "cancel" not in request.POST:
 238             UserOption.objects.set_value(
 239                 user=request.user, key="mail:alert", value=0, project=project
 240             )
 241         return HttpResponseRedirect(auth.get_login_url())
 242 
 243     context = csrf(request)
 244     context["project"] = project
 245     return render_to_response("sentry/account/email_unsubscribe_project.html", context, request)
