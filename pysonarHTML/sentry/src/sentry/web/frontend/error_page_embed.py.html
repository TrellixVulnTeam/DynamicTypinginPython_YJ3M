  70     def _get_project_key(self, request):

  86     def _smart_response(self, request, context=None, status=200):
  87         json_context = json.dumps(context or {})
  88         accept = request.META.get("HTTP_ACCEPT") or ""
  89         if "text/javascript" in accept:
  90             content_type = "text/javascript"
  91             content = ""
  92         else:
  93             content_type = "application/json"
  94             content = json_context
  95         response = HttpResponse(content, status=status, content_type=content_type)
  96         response["Access-Control-Allow-Origin"] = request.META.get("HTTP_ORIGIN", "")
  97         response["Access-Control-Allow-Methods"] = "GET, POST, OPTIONS"
  98         response["Access-Control-Max-Age"] = "1000"
  99         response["Access-Control-Allow-Headers"] = "Content-Type, Authorization, X-Requested-With"
 100         response["Vary"] = "Accept"
 101         if content == "" and context:
 102             response["X-Sentry-Context"] = json_context
 103         return response

 105     @csrf_exempt
 106     def dispatch(self, request):
 107         try:
 108             event_id = request.GET["eventId"]
 109         except KeyError:
 110             return self._smart_response(
 111                 request, {"eventId": "Missing or invalid parameter."}, status=400
 112             )
 113 
 114         normalized_event_id = normalize_event_id(event_id)
 115         if normalized_event_id:
 116             event_id = normalized_event_id
 117         elif event_id:
 118             return self._smart_response(
 119                 request, {"eventId": "Missing or invalid parameter."}, status=400
 120             )
 121 
 122         key = self._get_project_key(request)
 123         if not key:
 124             return self._smart_response(
 125                 request, {"dsn": "Missing or invalid parameter."}, status=404
 126             )
 127 
 128         origin = self._get_origin(request)
 129         if not is_valid_origin(origin, key.project):
 130             return self._smart_response(request, status=403)
 131 
 132         if request.method == "OPTIONS":
 133             return self._smart_response(request)
 134 
 135         # customization options
 136         options = DEFAULT_OPTIONS.copy()
 137         for name in six.iterkeys(options):
 138             if name in request.GET:
 139                 options[name] = six.text_type(request.GET[name])
 140 
 141         # TODO(dcramer): since we cant use a csrf cookie we should at the very
 142         # least sign the request / add some kind of nonce
 143         initial = {"name": request.GET.get("name"), "email": request.GET.get("email")}
 144 
 145         form = UserReportForm(request.POST if request.method == "POST" else None, initial=initial)
 146         if form.is_valid():
 147             # TODO(dcramer): move this to post to the internal API
 148             report = form.save(commit=False)
 149             report.project = key.project
 150             report.event_id = event_id
 151 
 152             event = eventstore.get_event_by_id(report.project.id, report.event_id)
 153 
 154             if event is not None:
 155                 report.environment = event.get_environment()
 156                 report.group = event.group
 157 
 158             try:
 159                 with transaction.atomic():
 160                     report.save()
 161             except IntegrityError:
 162                 # There was a duplicate, so just overwrite the existing
 163                 # row with the new one. The only way this ever happens is
 164                 # if someone is messing around with the API, or doing
 165                 # something wrong with the SDK, but this behavior is
 166                 # more reasonable than just hard erroring and is more
 167                 # expected.
 168                 UserReport.objects.filter(project=report.project, event_id=report.event_id).update(
 169                     name=report.name,
 170                     email=report.email,
 171                     comments=report.comments,
 172                     date_added=timezone.now(),
 173                 )
 174 
 175             else:
 176                 if report.group:
 177                     report.notify()
 178 
 179             user_feedback_received.send(project=report.project, group=report.group, sender=self)
 180 
 181             return self._smart_response(request)
 182         elif request.method == "POST":
 183             return self._smart_response(request, {"errors": dict(form.errors)}, status=400)
 184 
 185         show_branding = (
 186             ProjectOption.objects.get_value(
 187                 project=key.project, key="feedback:branding", default="1"
 188             )
 189             == "1"
 190         )
 191 
 192         template = render_to_string(
 193             "sentry/error-page-embed.html",
 194             context={
 195                 "form": form,
 196                 "show_branding": show_branding,
 197                 "title": options["title"],
 198                 "subtitle": options["subtitle"],
 199                 "subtitle2": options["subtitle2"],
 200                 "name_label": options["labelName"],
 201                 "email_label": options["labelEmail"],
 202                 "comments_label": options["labelComments"],
 203                 "submit_label": options["labelSubmit"],
 204                 "close_label": options["labelClose"],
 205             },
 206         )
 207 
 208         context = {
 209             "endpoint": mark_safe("*/" + json.dumps(absolute_uri(request.get_full_path())) + ";/*"),
 210             "template": mark_safe("*/" + json.dumps(template) + ";/*"),
 211             "strings": json.dumps_htmlsafe(
 212                 {
 213                     "generic_error": six.text_type(options["errorGeneric"]),
 214                     "form_error": six.text_type(options["errorFormEntry"]),
 215                     "sent_message": six.text_type(options["successMessage"]),
 216                 }
 217             ),
 218         }
 219 
 220         return render_to_response(
 221             "sentry/error-page-embed.js", context, request, content_type="text/javascript"
 222         )
