  34     def get(self, request, *args, **kwargs):
  35         if not request.user.is_authenticated():
  36             configure_uri = u"/extensions/{}/configure/?{}".format(
  37                 self.provider, urlencode(request.GET.dict()),
  38             )
  39 
  40             redirect_uri = u"{}?{}".format(
  41                 reverse("sentry-login"), urlencode({"next": configure_uri})
  42             )
  43 
  44             return self.redirect(redirect_uri)
  45 
  46         # check if we have one org
  47         organization = None
  48         if request.user.get_orgs().count() == 1:
  49             organization = request.user.get_orgs()[0]
  50         # if we have an org slug in the query param, use that org
  51         elif "orgSlug" in request.GET:
  52             organization = Organization.objects.get(slug=request.GET["orgSlug"])
  53 
  54         if organization:
  55             # if org does not have the feature, redirect
  56             if not self.is_enabled_for_org(organization, request.user):
  57                 return self.redirect("/")
  58 
  59             # TODO(steve): we probably should check the user has permissions and show an error page if not
  60             pipeline = self.init_pipeline(request, organization, request.GET.dict())
  61             return pipeline.current_step()
  62 
  63         return self.redirect(
  64             u"/extensions/{}/link/?{}".format(self.provider, urlencode(request.GET.dict()))
  65         )
  66 
  67     def init_pipeline(self, request, organization, params):
  68         pipeline = ExternalIntegrationPipeline(

  71 
  72         pipeline.initialize()
  73         pipeline.bind_state(self.provider, self.map_params_to_state(params))
  74 
  75         return pipeline
