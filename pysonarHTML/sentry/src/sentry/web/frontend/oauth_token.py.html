  18 class OAuthTokenView(View):
  19     @csrf_exempt
  20     @never_cache
  21     def dispatch(self, request, *args, **kwargs):
  22         return super(OAuthTokenView, self).dispatch(request, *args, **kwargs)

  24     # Note: the reason parameter is for internal use only
  25     def error(self, request, name, reason=None, status=400):
  26         client_id = request.POST.get("client_id")
  27         redirect_uri = request.POST.get("redirect_uri")
  28 
  29         logging.error(
  30             "oauth.token-error",
  31             extra={
  32                 "error_name": name,
  33                 "status": status,
  34                 "client_id": client_id,
  35                 "redirect_uri": redirect_uri,
  36                 "reason": reason,
  37             },
  38         )
  39         return HttpResponse(
  40             json.dumps({"error": name}), content_type="application/json", status=status
  41         )
  42 
  43     @never_cache
  44     def post(self, request):
  45         grant_type = request.POST.get("grant_type")
  46 
  47         if grant_type == "authorization_code":
  48             client_id = request.POST.get("client_id")
  49             redirect_uri = request.POST.get("redirect_uri")
  50             code = request.POST.get("code")
  51 
  52             if not client_id:
  53                 return self.error(request, "invalid_client", "missing client_id")
  54 
  55             try:
  56                 application = ApiApplication.objects.get(
  57                     client_id=client_id, status=ApiApplicationStatus.active
  58                 )
  59             except ApiApplication.DoesNotExist:
  60                 return self.error(request, "invalid_client", "invalid client_id")
  61 
  62             try:
  63                 grant = ApiGrant.objects.get(application=application, code=code)
  64             except ApiGrant.DoesNotExist:
  65                 return self.error(request, "invalid_grant", "invalid grant")
  66 
  67             if grant.is_expired():
  68                 return self.error(request, "invalid_grant", "grant expired")
  69 
  70             if not redirect_uri:
  71                 redirect_uri = application.get_default_redirect_uri()
  72             elif grant.redirect_uri != redirect_uri:
  73                 return self.error(request, "invalid_grant", "invalid redirect_uri")
  74 
  75             token = ApiToken.from_grant(grant)
  76         elif grant_type == "refresh_token":
  77             refresh_token = request.POST.get("refresh_token")
  78             scope = request.POST.get("scope")
  79             client_id = request.POST.get("client_id")
  80 
  81             if not refresh_token:
  82                 return self.error(request, "invalid_request")
  83 
  84             # TODO(dcramer): support scope
  85             if scope:
  86                 return self.error(request, "invalid_request")
  87 
  88             if not client_id:
  89                 return self.error(request, "invalid_client", "missing client_id")
  90 
  91             try:
  92                 application = ApiApplication.objects.get(
  93                     client_id=client_id, status=ApiApplicationStatus.active
  94                 )
  95             except ApiApplication.DoesNotExist:
  96                 return self.error(request, "invalid_client", "invalid client_id")
  97 
  98             try:
  99                 token = ApiToken.objects.get(application=application, refresh_token=refresh_token)
 100             except ApiToken.DoesNotExist:
 101                 return self.error(request, "invalid_grant", "invalid token")
 102 
 103             token.refresh()
 104         else:
 105             return self.error(request, "unsupported_grant_type")
 106 
 107         return HttpResponse(
 108             json.dumps(
 109                 {
 110                     "access_token": token.token,
 111                     "refresh_token": token.refresh_token,
 112                     "expires_in": int((token.expires_at - timezone.now()).total_seconds())
 113                     if token.expires_at
 114                     else None,
 115                     "expires_at": token.expires_at,
 116                     "token_type": "bearer",
 117                     "scope": " ".join(token.get_scopes()),  # NOQA
 118                     "user": {
 119                         "id": six.text_type(token.user.id),
 120                         # we might need these to become scope based
 121                         "name": token.user.name,
 122                         "email": token.user.email,
 123                     },
 124                 }
 125             ),
 126             content_type="application/json",
 127         )
