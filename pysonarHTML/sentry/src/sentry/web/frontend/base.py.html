  43 class OrganizationMixin(object):
  44     # TODO(dcramer): move the implicit organization logic into its own class
  45     # as it's only used in a single location and over complicates the rest of
  46     # the code
  47     def get_active_organization(self, request, organization_slug=None):
  48         """
  49         Returns the currently active organization for the request or None
  50         if no organization.
  51         """
  52 
  53         # TODO(dcramer): this is a huge hack, and we should refactor this
  54         # it is currently needed to handle the is_auth_required check on
  55         # OrganizationBase
  56         active_organization = getattr(self, "_active_org", None)
  57         cached_active_org = (
  58             active_organization
  59             and active_organization[0].slug == organization_slug
  60             and active_organization[1] == request.user
  61         )
  62         if cached_active_org:
  63             return active_organization[0]
  64 
  65         active_organization = None
  66 
  67         is_implicit = organization_slug is None
  68 
  69         if is_implicit:
  70             organization_slug = request.session.get("activeorg")
  71 
  72         if organization_slug is not None:
  73             if is_active_superuser(request):
  74                 try:
  75                     active_organization = Organization.objects.get_from_cache(
  76                         slug=organization_slug
  77                     )
  78                     if active_organization.status != OrganizationStatus.VISIBLE:
  79                         raise Organization.DoesNotExist
  80                 except Organization.DoesNotExist:
  81                     logger.info("Active organization [%s] not found", organization_slug)
  82 
  83         if active_organization is None:
  84             organizations = Organization.objects.get_for_user(user=request.user)
  85 
  86         if active_organization is None and organization_slug:
  87             try:
  88                 active_organization = six.next(
  89                     o for o in organizations if o.slug == organization_slug
  90                 )
  91             except StopIteration:
  92                 logger.info("Active organization [%s] not found in scope", organization_slug)
  93                 if is_implicit:
  94                     del request.session["activeorg"]
  95                 active_organization = None
  96 
  97         if active_organization is None:
  98             if not is_implicit:
  99                 return None
 100 
 101             try:
 102                 active_organization = organizations[0]
 103             except IndexError:
 104                 logger.info("User is not a member of any organizations")
 105                 pass
 106 
 107         if active_organization and self._is_org_member(request.user, active_organization):
 108             if active_organization.slug != request.session.get("activeorg"):
 109                 request.session["activeorg"] = active_organization.slug
 110 
 111         self._active_org = (active_organization, request.user)
 112 
 113         return active_organization
 114 
 115     def _is_org_member(self, user, organization):
 116         return OrganizationMember.objects.filter(user=user, organization=organization).exists()

 120             organization.flags.require_2fa

 166 class BaseView(View, OrganizationMixin):
 167     auth_required = True
 168     # TODO(dcramer): change sudo so it can be required only on POST
 169     sudo_required = False
 170 
 171     csrf_protect = True
 172 
 173     def __init__(self, auth_required=None, sudo_required=None, csrf_protect=None, *args, **kwargs):
 174         if auth_required is not None:
 175             self.auth_required = auth_required
 176         if sudo_required is not None:
 177             self.sudo_required = sudo_required
 178         if csrf_protect is not None:
 179             self.csrf_protect = csrf_protect
 180         super(BaseView, self).__init__(*args, **kwargs)
 181 
 182     @csrf_exempt
 183     def dispatch(self, request, *args, **kwargs):
 184         """
 185         A note on the CSRF protection process.
 186 
 187         Because the CSRF decorators don't work well with view subclasses, we
 188         allow them to control whether a CSRF check is done by setting
 189         self.csrf_protect. This has a couple of implications:
 190 
 191         1. We need to mark this method as @csrf_exempt so that when the CSRF
 192            middleware checks it as part of the regular middleware sequence, it
 193            always passes.
 194         2. If self.csrf_protect is set, we will re-run the CSRF check ourselves
 195            using CsrfViewMiddleware().process_view()
 196         3. But first we must remove the csrf_exempt attribute that was set by
 197            the decorator so that the middleware doesn't shortcut and pass the
 198            check unconditionally again.
 199 
 200         """
 201         if self.csrf_protect:
 202             if hasattr(self.dispatch.__func__, "csrf_exempt"):
 203                 delattr(self.dispatch.__func__, "csrf_exempt")
 204             response = self.test_csrf(request)
 205             if response:
 206                 return response
 207 
 208         if self.is_auth_required(request, *args, **kwargs):
 209             return self.handle_auth_required(request, *args, **kwargs)
 210 
 211         if self.is_sudo_required(request, *args, **kwargs):
 212             return self.handle_sudo_required(request, *args, **kwargs)
 213 
 214         args, kwargs = self.convert_args(request, *args, **kwargs)
 215 
 216         request.access = self.get_access(request, *args, **kwargs)
 217 
 218         if not self.has_permission(request, *args, **kwargs):
 219             return self.handle_permission_required(request, *args, **kwargs)
 220 
 221         if "organization" in kwargs and self.is_not_2fa_compliant(request, kwargs["organization"]):
 222             return self.handle_not_2fa_compliant(request, *args, **kwargs)
 223 
 224         self.request = request
 225         self.default_context = self.get_context_data(request, *args, **kwargs)
 226 
 227         return self.handle(request, *args, **kwargs)
 228 
 229     def test_csrf(self, request):
 230         middleware = CsrfViewMiddleware()
 231         return middleware.process_view(request, self.dispatch, [request], {})
 232 
 233     def get_access(self, request, *args, **kwargs):
 234         return access.DEFAULT
 235 
 236     def convert_args(self, request, *args, **kwargs):
 237         return (args, kwargs)
 238 
 239     def handle(self, request, *args, **kwargs):
 240         return super(BaseView, self).dispatch(request, *args, **kwargs)
 241 
 242     def is_auth_required(self, request, *args, **kwargs):
 243         return self.auth_required and not (
 244             request.user.is_authenticated() and request.user.is_active
 245         )
 246 
 247     def handle_auth_required(self, request, *args, **kwargs):
 248         auth.initiate_login(request, next_url=request.get_full_path())
 249         if "organization_slug" in kwargs:
 250             redirect_to = reverse("sentry-auth-organization", args=[kwargs["organization_slug"]])
 251         else:
 252             redirect_to = auth.get_login_url()
 253         return self.redirect(redirect_to, headers={"X-Robots-Tag": "noindex, nofollow"})
 254 
 255     def is_sudo_required(self, request, *args, **kwargs):
 256         return self.sudo_required and not request.is_sudo()
 257 
 258     def handle_sudo_required(self, request, *args, **kwargs):
 259         return redirect_to_sudo(request.get_full_path())
 260 
 261     def has_permission(self, request, *args, **kwargs):
 262         return True
 263 
 264     def handle_permission_required(self, request, *args, **kwargs):
 265         redirect_uri = self.get_no_permission_url(request, *args, **kwargs)
 266         return self.redirect(redirect_uri)
 267 
 268     def handle_not_2fa_compliant(self, request, *args, **kwargs):
 269         redirect_uri = self.get_not_2fa_compliant_url(request, *args, **kwargs)
 270         return self.redirect(redirect_uri)
 271 
 272     def get_no_permission_url(request, *args, **kwargs):
 273         return reverse("sentry-login")
 274 
 275     def get_not_2fa_compliant_url(self, request, *args, **kwargs):
 276         return reverse("sentry-account-settings-security")
 277 
 278     def get_context_data(self, request, **kwargs):
 279         context = csrf(request)
 280         return context
 281 
 282     def respond(self, template, context=None, status=200):
 283         default_context = self.default_context
 284         if context:
 285             default_context.update(context)
 286 
 287         return render_to_response(template, default_context, self.request, status=status)
 288 
 289     def redirect(self, url, headers=None):
 290         res = HttpResponseRedirect(url)
 291         if headers:
 292             for k, v in headers.items():
 293                 res[k] = v
 294         return res
 295 
 296     def get_team_list(self, user, organization):
 297         return Team.objects.get_for_user(organization=organization, user=user, with_projects=True)
 298 
 299     def create_audit_entry(self, request, transaction_id=None, **kwargs):
 300         return create_audit_entry(request, transaction_id, audit_logger, **kwargs)

 303 class OrganizationView(BaseView):
 304     """
 305     Any view acting on behalf of an organization should inherit from this base.
 306 
 307     The 'organization' keyword argument is automatically injected into the
 308     resulting dispatch.
 309     """
 310 
 311     required_scope = None
 312     valid_sso_required = True
 313 
 314     def get_access(self, request, organization, *args, **kwargs):
 315         if organization is None:
 316             return access.DEFAULT
 317         return access.from_request(request, organization)
 318 
 319     def get_context_data(self, request, organization, **kwargs):
 320         context = super(OrganizationView, self).get_context_data(request)
 321         context["organization"] = organization
 322         context["TEAM_LIST"] = self.get_team_list(request.user, organization)
 323         context["ACCESS"] = request.access.to_django_context()
 324         return context
 325 
 326     def has_permission(self, request, organization, *args, **kwargs):
 327         if organization is None:
 328             return False
 329         if self.valid_sso_required:
 330             if request.access.requires_sso and not request.access.sso_is_valid:
 331                 return False
 332             if self.needs_sso(request, organization):
 333                 return False
 334         if self.required_scope and not request.access.has_scope(self.required_scope):
 335             logger.info(
 336                 "User %s does not have %s permission to access organization %s",
 337                 request.user,
 338                 self.required_scope,
 339                 organization,
 340             )
 341             return False
 342         return True
 343 
 344     def is_auth_required(self, request, organization_slug=None, *args, **kwargs):
 345         result = super(OrganizationView, self).is_auth_required(request, *args, **kwargs)
 346         if result:
 347             return result
 348 
 349         # if the user is attempting to access an organization that *may* be
 350         # accessible if they simply re-authenticate, we want to allow that
 351         # this opens up a privacy hole, but the pros outweigh the cons
 352         if not organization_slug:
 353             return False
 354 
 355         active_organization = self.get_active_organization(
 356             request=request, organization_slug=organization_slug
 357         )
 358         if not active_organization:
 359             try:
 360                 Organization.objects.get_from_cache(slug=organization_slug)
 361             except Organization.DoesNotExist:
 362                 pass
 363             else:
 364                 return True
 365         return False
 366 
 367     def handle_permission_required(self, request, organization, *args, **kwargs):
 368         if self.needs_sso(request, organization):
 369             logger.info(
 370                 "access.must-sso",
 371                 extra={"organization_id": organization.id, "user_id": request.user.id},
 372             )
 373             auth.initiate_login(request, next_url=request.get_full_path())
 374             redirect_uri = reverse("sentry-auth-organization", args=[organization.slug])
 375         else:
 376             redirect_uri = self.get_no_permission_url(request, *args, **kwargs)
 377         return self.redirect(redirect_uri)
 378 
 379     def needs_sso(self, request, organization):
 380         if not organization:
 381             return False
 382         # XXX(dcramer): this branch should really never hit
 383         if not request.user.is_authenticated():
 384             return False
 385         if not self.valid_sso_required:
 386             return False
 387         if not request.access.requires_sso:
 388             return False
 389         if not auth.has_completed_sso(request, organization.id):
 390             return True
 391         if not request.access.sso_is_valid:
 392             return True
 393         return False
 394 
 395     def convert_args(self, request, organization_slug=None, *args, **kwargs):
 396         active_organization = self.get_active_organization(
 397             request=request, organization_slug=organization_slug
 398         )
 399 
 400         kwargs["organization"] = active_organization
 401 
 402         return (args, kwargs)
 403 
 404     def get_allowed_roles(self, request, organization, member=None):
 405         can_admin = request.access.has_scope("member:admin")
 406 
 407         allowed_roles = []
 408         if can_admin and not is_active_superuser(request):
 409             acting_member = OrganizationMember.objects.get(
 410                 user=request.user, organization=organization
 411             )
 412             if member and roles.get(acting_member.role).priority < roles.get(member.role).priority:
 413                 can_admin = False
 414             else:
 415                 allowed_roles = [
 416                     r
 417                     for r in roles.get_all()
 418                     if r.priority <= roles.get(acting_member.role).priority
 419                 ]
 420                 can_admin = bool(allowed_roles)
 421         elif is_active_superuser(request):
 422             allowed_roles = roles.get_all()
 423         return (can_admin, allowed_roles)

 426 class TeamView(OrganizationView):
 427     """
 428     Any view acting on behalf of a team should inherit from this base and the
 429     matching URL pattern must pass 'team_slug'.
 430 
 431     Two keyword arguments are added to the resulting dispatch:
 432 
 433     - organization
 434     - team
 435     """
 436 
 437     def get_context_data(self, request, organization, team, **kwargs):
 438         context = super(TeamView, self).get_context_data(request, organization)
 439         context["team"] = team
 440         return context
 441 
 442     def has_permission(self, request, organization, team, *args, **kwargs):
 443         if team is None:
 444             return False
 445         rv = super(TeamView, self).has_permission(request, organization)
 446         if not rv:
 447             return rv
 448         if self.required_scope:
 449             if not request.access.has_team_scope(team, self.required_scope):
 450                 logger.info(
 451                     "User %s does not have %s permission to access team %s",
 452                     request.user,
 453                     self.required_scope,
 454                     team,
 455                 )
 456                 return False
 457         elif not request.access.has_team(team):
 458             logger.info("User %s does not have access to team %s", request.user, team)
 459             return False
 460         return True
 461 
 462     def convert_args(self, request, organization_slug, team_slug, *args, **kwargs):
 463         active_organization = self.get_active_organization(
 464             request=request, organization_slug=organization_slug
 465         )
 466 
 467         if active_organization:
 468             active_team = self.get_active_team(
 469                 request=request, team_slug=team_slug, organization=active_organization
 470             )
 471         else:
 472             active_team = None
 473 
 474         kwargs["organization"] = active_organization
 475         kwargs["team"] = active_team
 476 
 477         return (args, kwargs)

 480 class ProjectView(OrganizationView):
 481     """
 482     Any view acting on behalf of a project should inherit from this base and the
 483     matching URL pattern must pass 'org_slug' as well as 'project_slug'.
 484 
 485     Three keyword arguments are added to the resulting dispatch:
 486 
 487     - organization
 488     - project
 489     """
 490 
 491     def get_context_data(self, request, organization, project, **kwargs):
 492         context = super(ProjectView, self).get_context_data(request, organization)
 493         context["project"] = project
 494         context["processing_issues"] = serialize(project).get("processingIssues", 0)
 495         return context
 496 
 497     def has_permission(self, request, organization, project, *args, **kwargs):
 498         if project is None:
 499             return False
 500         rv = super(ProjectView, self).has_permission(request, organization)
 501         if not rv:
 502             return rv
 503 
 504         teams = list(project.teams.all())
 505 
 506         if self.required_scope:
 507             if not any(request.access.has_team_scope(team, self.required_scope) for team in teams):
 508                 logger.info(
 509                     "User %s does not have %s permission to access project %s",
 510                     request.user,
 511                     self.required_scope,
 512                     project,
 513                 )
 514                 return False
 515         elif not any(request.access.has_team(team) for team in teams):
 516             logger.info("User %s does not have access to project %s", request.user, project)
 517             return False
 518         return True
 519 
 520     def convert_args(self, request, organization_slug, project_slug, *args, **kwargs):
 521         active_organization = self.get_active_organization(
 522             request=request, organization_slug=organization_slug
 523         )
 524 
 525         if active_organization:
 526             active_project = self.get_active_project(
 527                 request=request, organization=active_organization, project_slug=project_slug
 528             )
 529         else:
 530             active_project = None
 531 
 532         kwargs["project"] = active_project
 533         kwargs["organization"] = active_organization
 534 
 535         return (args, kwargs)

 560                 photo_file = avatar.get_cached_photo(size)
 561 
 562         res = HttpResponse(photo_file, content_type="image/png")
 563         res["Cache-Control"] = FOREVER_CACHE
 564         return res
