  51     def _disable_provider(self, request, organization, auth_provider):
  52         self.create_audit_entry(
  53             request,
  54             organization=organization,
  55             target_object=auth_provider.id,
  56             event=AuditLogEntryEvent.SSO_DISABLE,
  57             data=auth_provider.get_audit_log_data(),
  58         )
  59 
  60         OrganizationMember.objects.filter(organization=organization).update(
  61             flags=F("flags")
  62             .bitand(~OrganizationMember.flags["sso:linked"])
  63             .bitand(~OrganizationMember.flags["sso:invalid"])
  64         )
  65 
  66         user_ids = OrganizationMember.objects.filter(organization=organization).values("user")
  67         User.objects.filter(id__in=user_ids).update(is_managed=False)
  68 
  69         email_unlink_notifications.delay(organization.id, request.user.id, auth_provider.provider)
  70         auth_provider.delete()

  72     def handle_existing_provider(self, request, organization, auth_provider):
  73         provider = auth_provider.get_provider()
  74 
  75         if request.method == "POST":
  76             op = request.POST.get("op")
  77             if op == "disable":
  78                 self._disable_provider(request, organization, auth_provider)
  79 
  80                 messages.add_message(request, messages.SUCCESS, OK_PROVIDER_DISABLED)
  81 
  82                 next_uri = u"/settings/{}/auth/".format(organization.slug)
  83                 return self.redirect(next_uri)
  84             elif op == "reinvite":
  85                 email_missing_links.delay(organization.id, request.user.id, provider.key)
  86 
  87                 messages.add_message(request, messages.SUCCESS, OK_REMINDERS_SENT)
  88 
  89                 next_uri = reverse(
  90                     "sentry-organization-auth-provider-settings", args=[organization.slug]
  91                 )
  92                 return self.redirect(next_uri)
  93 
  94         form = AuthProviderSettingsForm(
  95             data=request.POST if request.POST.get("op") == "settings" else None,
  96             initial={
  97                 "require_link": not auth_provider.flags.allow_unlinked,
  98                 "default_role": organization.default_role,
  99             },
 100         )
 101 
 102         if form.is_valid():
 103             auth_provider.flags.allow_unlinked = not form.cleaned_data["require_link"]
 104             auth_provider.save()
 105 
 106             organization.default_role = form.cleaned_data["default_role"]
 107             organization.save()
 108 
 109             if form.initial != form.cleaned_data:
 110                 changed_data = {}
 111                 for key, value in form.cleaned_data.items():
 112                     if form.initial.get(key) != value:
 113                         changed_data[key] = u"to {}".format(value)
 114 
 115                 self.create_audit_entry(
 116                     request,
 117                     organization=organization,
 118                     target_object=auth_provider.id,
 119                     event=AuditLogEntryEvent.SSO_EDIT,
 120                     data=changed_data,
 121                 )
 122 
 123         view = provider.get_configure_view()
 124         response = view(request, organization, auth_provider)
 125         if isinstance(response, HttpResponse):
 126             return response
 127         elif isinstance(response, Response):
 128             response = response.render(
 129                 request,
 130                 {
 131                     "auth_provider": auth_provider,
 132                     "organization": organization,
 133                     "provider": provider,
 134                 },
 135             )
 136 
 137         pending_links_count = OrganizationMember.objects.filter(
 138             organization=organization,
 139             flags=F("flags").bitand(~OrganizationMember.flags["sso:linked"]),
 140         ).count()
 141 
 142         context = {
 143             "form": form,
 144             "pending_links_count": pending_links_count,
 145             "login_url": absolute_uri(organization.get_url()),
 146             "auth_provider": auth_provider,
 147             "provider_name": provider.name,
 148             "content": response,
 149         }
 150 
 151         return self.respond("sentry/organization-auth-provider-settings.html", context)

 153     @transaction.atomic
 154     def handle(self, request, organization):
 155         try:
 156             auth_provider = AuthProvider.objects.get(organization=organization)
 157         except AuthProvider.DoesNotExist:
 158             pass
 159         else:
 160             provider = auth_provider.get_provider()
 161             requires_feature = provider.required_feature
 162 
 163             # Provider is not enabled
 164             # Allow superusers to edit and disable SSO for orgs that
 165             # downgrade plans and can no longer access the feature
 166             if (
 167                 requires_feature
 168                 and not features.has(requires_feature, organization, actor=request.user)
 169                 and not is_active_superuser(request)
 170             ):
 171                 home_url = organization.get_url()
 172                 messages.add_message(request, messages.ERROR, ERR_NO_SSO)
 173 
 174                 return HttpResponseRedirect(home_url)
 175 
 176             return self.handle_existing_provider(
 177                 request=request, organization=organization, auth_provider=auth_provider
 178             )
 179 
 180         if request.method == "POST":
 181             provider_key = request.POST.get("provider")
 182             if not manager.exists(provider_key):
 183                 raise ValueError(u"Provider not found: {}".format(provider_key))
 184 
 185             helper = AuthHelper(
 186                 request=request,
 187                 organization=organization,
 188                 provider_key=provider_key,
 189                 flow=AuthHelper.FLOW_SETUP_PROVIDER,
 190             )
 191 
 192             feature = helper.provider.required_feature
 193             if feature and not features.has(feature, organization, actor=request.user):
 194                 return HttpResponse("Provider is not enabled", status=401)
 195 
 196             if request.POST.get("init"):
 197                 helper.init_pipeline()
 198 
 199             if not helper.pipeline_is_valid():
 200                 return helper.error("Something unexpected happened during authentication.")
 201 
 202             # render first time setup view
 203             return helper.current_step()
 204 
 205         # Otherwise user is in bad state since frontend/react should handle this case
 206         return HttpResponseRedirect(organization.get_url())
