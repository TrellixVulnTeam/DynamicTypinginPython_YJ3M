  24     def redirect_response(self, response_type, redirect_uri, params):
  25         if response_type == "token":
  26             return self.redirect(
  27                 u"{}#{}".format(
  28                     redirect_uri,
  29                     urlencode([(k, v) for k, v in six.iteritems(params) if v is not None]),
  30                 )
  31             )
  32 
  33         parts = list(urlparse(redirect_uri))
  34         query = parse_qsl(parts[4])
  35         for key, value in six.iteritems(params):
  36             if value is not None:
  37                 query.append((key, value))
  38         parts[4] = urlencode(query)
  39         return self.redirect(urlunparse(parts))

  57     def get(self, request, **kwargs):
  58         response_type = request.GET.get("response_type")
  59         client_id = request.GET.get("client_id")
  60         redirect_uri = request.GET.get("redirect_uri")
  61         scopes = request.GET.get("scope")
  62         state = request.GET.get("state")
  63         force_prompt = request.GET.get("force_prompt")
  64 
  65         if not client_id:
  66             logging.error(
  67                 "oauth.authorize-error",
  68                 extra={
  69                     "error_name": "unauthorized_client",
  70                     "response_type": response_type,
  71                     "client_id": client_id,
  72                     "redirect_uri": redirect_uri,
  73                 },
  74             )
  75             return self.respond(
  76                 "sentry/oauth-error.html",
  77                 {"error": mark_safe("Missing or invalid <em>client_id</em> parameter.")},
  78             )
  79 
  80         try:
  81             application = ApiApplication.objects.get(
  82                 client_id=client_id, status=ApiApplicationStatus.active
  83             )
  84         except ApiApplication.DoesNotExist:
  85             logging.error(
  86                 "oauth.authorize-error",
  87                 extra={
  88                     "error_name": "unauthorized_client",
  89                     "response_type": response_type,
  90                     "client_id": client_id,
  91                     "redirect_uri": redirect_uri,
  92                 },
  93             )
  94             return self.respond(
  95                 "sentry/oauth-error.html",
  96                 {"error": mark_safe("Missing or invalid <em>client_id</em> parameter.")},
  97             )
  98 
  99         if not redirect_uri:
 100             redirect_uri = application.get_default_redirect_uri()
 101         elif not application.is_valid_redirect_uri(redirect_uri):
 102             logging.error(
 103                 "oauth.authorize-error",
 104                 extra={
 105                     "error_name": "invalid_request",
 106                     "response_type": response_type,
 107                     "client_id": client_id,
 108                     "redirect_uri": redirect_uri,
 109                 },
 110             )
 111             return self.respond(
 112                 "sentry/oauth-error.html",
 113                 {"error": mark_safe("Missing or invalid <em>redirect_uri</em> parameter.")},
 114             )
 115 
 116         if not application.is_allowed_response_type(response_type):
 117             return self.error(
 118                 request=request,
 119                 client_id=client_id,
 120                 response_type=response_type,
 121                 redirect_uri=redirect_uri,
 122                 name="unsupported_response_type",
 123                 state=state,
 124             )
 125 
 126         if scopes:
 127             scopes = scopes.split(" ")
 128             for scope in scopes:
 129                 if scope not in settings.SENTRY_SCOPES:
 130                     return self.error(
 131                         request=request,
 132                         client_id=client_id,
 133                         response_type=response_type,
 134                         redirect_uri=redirect_uri,
 135                         name="invalid_scope",
 136                         state=state,
 137                     )
 138         else:
 139             scopes = []
 140 
 141         payload = {
 142             "rt": response_type,
 143             "cid": client_id,
 144             "ru": redirect_uri,
 145             "sc": scopes,
 146             "st": state,
 147             "uid": request.user.id if request.user.is_authenticated() else "",
 148         }
 149         request.session["oa2"] = payload
 150 
 151         if not request.user.is_authenticated():
 152             return super(OAuthAuthorizeView, self).get(request, application=application)
 153 
 154         if not force_prompt:
 155             try:
 156                 existing_auth = ApiAuthorization.objects.get(
 157                     user=request.user, application=application
 158                 )
 159             except ApiAuthorization.DoesNotExist:
 160                 pass
 161             else:
 162                 # if we've already approved all of the required scopes
 163                 # we can skip prompting the user
 164                 if all(existing_auth.has_scope(s) for s in scopes):
 165                     return self.approve(
 166                         request=request,
 167                         application=application,
 168                         scopes=scopes,
 169                         response_type=response_type,
 170                         redirect_uri=redirect_uri,
 171                         state=state,
 172                     )
 173 
 174         payload = {
 175             "rt": response_type,
 176             "cid": client_id,
 177             "ru": redirect_uri,
 178             "sc": scopes,
 179             "st": state,
 180             "uid": request.user.id,
 181         }
 182         request.session["oa2"] = payload
 183 
 184         permissions = []
 185         if scopes:
 186             pending_scopes = set(scopes)
 187             matched_sets = set()
 188             for scope_set in settings.SENTRY_SCOPE_SETS:
 189                 for scope, description in scope_set:
 190                     if scope_set in matched_sets and scope in pending_scopes:
 191                         pending_scopes.remove(scope)
 192                     elif scope in pending_scopes:
 193                         permissions.append(description)
 194                         matched_sets.add(scope_set)
 195                         pending_scopes.remove(scope)
 196 
 197             if pending_scopes:
 198                 raise NotImplementedError(
 199                     u"{} scopes did not have descriptions".format(pending_scopes)
 200                 )
 201 
 202         context = {
 203             "user": request.user,
 204             "application": application,
 205             "scopes": scopes,
 206             "permissions": permissions,
 207         }
 208         return self.respond("sentry/oauth-authorize.html", context)

 210     def post(self, request, **kwargs):
 211         try:
 212             payload = request.session["oa2"]
 213         except KeyError:
 214             return self.respond(
 215                 "sentry/oauth-error.html",
 216                 {
 217                     "error": "We were unable to complete your request. Please re-initiate the authorization flow."
 218                 },
 219             )
 220 
 221         try:
 222             application = ApiApplication.objects.get(
 223                 client_id=payload["cid"], status=ApiApplicationStatus.active
 224             )
 225         except ApiApplication.DoesNotExist:
 226             return self.respond(
 227                 "sentry/oauth-error.html",
 228                 {"error": mark_safe("Missing or invalid <em>client_id</em> parameter.")},
 229             )
 230 
 231         if not request.user.is_authenticated():
 232             response = super(OAuthAuthorizeView, self).post(
 233                 request, application=application, **kwargs
 234             )
 235             # once they login, bind their user ID
 236             if request.user.is_authenticated():
 237                 request.session["oa2"]["uid"] = request.user.id
 238                 request.session.modified = True
 239             return response
 240 
 241         if payload["uid"] != request.user.id:
 242             return self.respond(
 243                 "sentry/oauth-error.html",
 244                 {
 245                     "error": "We were unable to complete your request. Please re-initiate the authorization flow."
 246                 },
 247             )
 248 
 249         response_type = payload["rt"]
 250         redirect_uri = payload["ru"]
 251         scopes = payload["sc"]
 252 
 253         op = request.POST.get("op")
 254         if op == "approve":
 255             return self.approve(
 256                 request=request,
 257                 application=application,
 258                 scopes=scopes,
 259                 response_type=response_type,
 260                 redirect_uri=redirect_uri,
 261                 state=payload["st"],
 262             )
 263 
 264         elif op == "deny":
 265             return self.error(
 266                 request=request,
 267                 client_id=payload["cid"],
 268                 response_type=response_type,
 269                 redirect_uri=redirect_uri,
 270                 name="access_denied",
 271                 state=payload["st"],
 272             )
 273         else:
 274             raise NotImplementedError

 276     def approve(self, request, application, **params):
 277         try:
 278             with transaction.atomic():
 279                 ApiAuthorization.objects.create(
 280                     application=application, user=request.user, scope_list=params["scopes"]
 281                 )
 282         except IntegrityError:
 283             if params["scopes"]:
 284                 auth = ApiAuthorization.objects.get(application=application, user=request.user)
 285                 for scope in params["scopes"]:
 286                     if scope not in auth.scope_list:
 287                         auth.scope_list.append(scope)
 288                 auth.save()
 289 
 290         if params["response_type"] == "code":
 291             grant = ApiGrant.objects.create(
 292                 user=request.user,
 293                 application=application,
 294                 redirect_uri=params["redirect_uri"],
 295                 scope_list=params["scopes"],
 296             )
 297             return self.redirect_response(
 298                 params["response_type"],
 299                 params["redirect_uri"],
 300                 {"code": grant.code, "state": params["state"]},
 301             )
 302         elif params["response_type"] == "token":
 303             token = ApiToken.objects.create(
 304                 application=application,
 305                 user=request.user,
 306                 refresh_token=None,
 307                 scope_list=params["scopes"],
 308             )
 309 
 310             return self.redirect_response(
 311                 params["response_type"],
 312                 params["redirect_uri"],
 313                 {
 314                     "access_token": token.token,
 315                     "expires_in": int((timezone.now() - token.expires_at).total_seconds()),
 316                     "expires_at": token.expires_at.strftime("%Y-%m-%dT%H:%M:%S.%fZ"),
 317                     "token_type": "bearer",
 318                     "scope": " ".join(token.get_scopes()),  # NOQA
 319                     "state": params["state"],
 320                 },
 321             )
