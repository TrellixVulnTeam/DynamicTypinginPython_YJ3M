  21 def resolve(path):
  22     # Mostly yanked from Django core and changed to return the path:
  23     # See: https://github.com/django/django/blob/1.6.11/django/contrib/staticfiles/views.py
  24     normalized_path = posixpath.normpath(unquote(path)).lstrip("/")
  25     try:
  26         absolute_path = finders.find(normalized_path)
  27     except Exception:
  28         # trying to access bad paths like, `../../etc/passwd`, etc that
  29         # Django rejects, but respond nicely instead of erroring.
  30         absolute_path = None
  31     if not absolute_path:
  32         raise Http404("'%s' could not be found" % path)
  33     if path[-1] == "/" or os.path.isdir(absolute_path):
  34         raise Http404("Directory indexes are not allowed here.")
  35     return os.path.split(absolute_path)

  38 def static_media(request, **kwargs):
  39     """
  40     Serve static files below a given point in the directory structure.
  41     """
  42     module = kwargs.get("module")
  43     path = kwargs.get("path", "")
  44     version = kwargs.get("version")
  45 
  46     if module:
  47         path = "%s/%s" % (module, path)
  48 
  49     try:
  50         document_root, path = resolve(path)
  51     except Http404:
  52         # Return back a simpler plain-text 404 response, more suitable
  53         # for static files, rather than our full blown HTML.
  54         return HttpResponseNotFound("", content_type="text/plain")
  55 
  56     if (
  57         "gzip" in request.META.get("HTTP_ACCEPT_ENCODING", "")
  58         and not path.endswith(".gz")
  59         and not settings.DEBUG
  60     ):
  61         paths = (path + ".gz", path)
  62     else:
  63         paths = (path,)
  64 
  65     for p in paths:
  66         try:
  67             response = static.serve(request, p, document_root=document_root)
  68             break
  69         except Http404:
  70             # We don't need to handle this since `resolve()` is assuring to us that
  71             # at least the non-gzipped version exists, so in theory, this can
  72             # only happen on the first .gz path
  73             continue
  74 
  75     # Make sure we Vary: Accept-Encoding for gzipped responses
  76     response["Vary"] = "Accept-Encoding"
  77 
  78     # We need CORS for font files
  79     if path.endswith((".js", ".ttf", ".ttc", ".otf", ".eot", ".woff", ".woff2")):
  80         response["Access-Control-Allow-Origin"] = "*"
  81 
  82     # If we have a version and not DEBUG, we can cache it FOREVER
  83     if version is not None and not settings.DEBUG:
  84         response["Cache-Control"] = FOREVER_CACHE
  85     else:
  86         # Otherwise, we explicitly don't want to cache at all
  87         response["Cache-Control"] = NEVER_CACHE
  88 
  89     return response
