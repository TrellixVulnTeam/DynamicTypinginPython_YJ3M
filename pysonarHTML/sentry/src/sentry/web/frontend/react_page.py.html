  12 class ReactMixin(object):
  13     def handle_react(self, request):
  14         context = {"CSRF_COOKIE_NAME": settings.CSRF_COOKIE_NAME}
  15 
  16         # Force a new CSRF token to be generated and set in user's
  17         # Cookie. Alternatively, we could use context_processor +
  18         # template tag, but in this case, we don't need a form on the
  19         # page. So there's no point in rendering a random `<input>` field.
  20         get_csrf_token(request)
  21 
  22         return render_to_response("sentry/bases/react.html", context=context, request=request)

  27 class ReactPageView(OrganizationView, ReactMixin):
  28     def handle_auth_required(self, request, *args, **kwargs):
  29         # If user is a superuser (but not active, because otherwise this method would never be called)
  30         # Then allow client to handle the route and respond to any API request errors
  31         if request.user.is_superuser:
  32             return self.handle_react(request)
  33 
  34         # For normal users, let parent class handle (e.g. redirect to login page)
  35         return super(ReactPageView, self).handle_auth_required(request, *args, **kwargs)
  36 
  37     def handle(self, request, organization, **kwargs):
  38         if "project_id" in kwargs and request.GET.get("onboarding"):
  39             project = Project.objects.filter(
  40                 organization=organization, slug=kwargs["project_id"]
  41             ).first()
  42             first_event_pending.send(project=project, user=request.user, sender=self)
  43         return self.handle_react(request)
