  26 # Stores callbacks that are called to get additional template context data before the login page
  27 # is rendered. Callbacks are called in any order. If an error is encountered in a callback it is
  28 # ignored. This works like HookStore in Javascript.
  29 class AdditionalContext(object):
  30     def __init__(self):
  31         self._callbacks = set()
  32 
  33     def add_callback(self, callback):
  34         """callback should take a request object and return a dict of key-value pairs
  35         to add to the context."""
  36         self._callbacks.add(callback)
  37 
  38     def run_callbacks(self, request):
  39         context = {}
  40         for cb in self._callbacks:
  41             try:
  42                 result = cb(request)
  43                 context.update(result)
  44             except Exception:
  45                 capture_exception()
  46         return context
  47

  52 class AuthLoginView(BaseView):

  55     def get_auth_provider(self, organization_slug):
  56         try:
  57             organization = Organization.objects.get(
  58                 slug=organization_slug, status=OrganizationStatus.VISIBLE
  59             )
  60         except Organization.DoesNotExist:
  61             return None
  62 
  63         try:
  64             auth_provider = AuthProvider.objects.get(organization=organization)
  65         except AuthProvider.DoesNotExist:
  66             return None
  67 
  68         return auth_provider
  69 
  70     def get_login_form(self, request):
  71         op = request.POST.get("op")
  72         return AuthenticationForm(request, request.POST if op == "login" else None)
  73 
  74     def get_register_form(self, request, initial=None):
  75         op = request.POST.get("op")
  76         return RegistrationForm(
  77             request.POST if op == "register" else None,
  78             initial=initial,
  79             # Custom auto_id to avoid ID collision with AuthenticationForm.
  80             auto_id="id_registration_%s",
  81         )

  83     def can_register(self, request):
  84         return bool(auth.has_user_registration() or request.session.get("can_register"))

  86     def get_join_request_link(self, organization):
  87         if not organization:
  88             return None
  89 
  90         if organization.get_option("sentry:join_requests") is False:
  91             return None
  92 
  93         join_request_link_viewed.send_robust(sender=self, organization=organization)
  94 
  95         return reverse("sentry-join-request", args=[organization.slug])

  97     def get_next_uri(self, request):
  98         next_uri_fallback = None
  99         if request.session.get("_next") is not None:
 100             next_uri_fallback = request.session.pop("_next")
 101         return request.GET.get(REDIRECT_FIELD_NAME, next_uri_fallback)

 106     def handle_basic_auth(self, request, **kwargs):
 107         can_register = self.can_register(request)
 108 
 109         op = request.POST.get("op")
 110         organization = kwargs.pop("organization", None)
 111 
 112         if not op:
 113             # Detect that we are on the register page by url /register/ and
 114             # then activate the register tab by default.
 115             if "/register" in request.path_info and can_register:
 116                 op = "register"
 117             elif request.GET.get("op") == "sso":
 118                 op = "sso"
 119 
 120         login_form = self.get_login_form(request)
 121         if can_register:
 122             register_form = self.get_register_form(
 123                 request, initial={"username": request.session.get("invite_email", "")}
 124             )
 125         else:
 126             register_form = None
 127 
 128         if can_register and register_form.is_valid():
 129             user = register_form.save()
 130             user.send_confirm_emails(is_new_user=True)
 131             user_signup.send_robust(
 132                 sender=self, user=user, source="register-form", referrer="in-app"
 133             )
 134 
 135             # HACK: grab whatever the first backend is and assume it works
 136             user.backend = settings.AUTHENTICATION_BACKENDS[0]
 137 
 138             auth.login(request, user, organization_id=organization.id if organization else None)
 139 
 140             # can_register should only allow a single registration
 141             request.session.pop("can_register", None)
 142             request.session.pop("invite_email", None)
 143 
 144             # In single org mode, associate the user to the orgnaization
 145             if settings.SENTRY_SINGLE_ORGANIZATION:
 146                 organization = Organization.get_default()
 147                 OrganizationMember.objects.create(
 148                     organization=organization, role=organization.default_role, user=user
 149                 )
 150 
 151             # Attempt to directly accept any pending invites
 152             invite_helper = ApiInviteHelper.from_cookie(request=request, instance=self)
 153 
 154             if invite_helper and invite_helper.valid_request:
 155                 invite_helper.accept_invite()
 156                 response = self.redirect_to_org(request)
 157                 remove_invite_cookie(request, response)
 158 
 159                 return response
 160 
 161             return self.redirect(auth.get_login_redirect(request))
 162 
 163         elif request.method == "POST":
 164             from sentry.app import ratelimiter
 165             from sentry.utils.hashlib import md5_text
 166 
 167             login_attempt = (
 168                 op == "login" and request.POST.get("username") and request.POST.get("password")
 169             )
 170 
 171             if login_attempt and ratelimiter.is_limited(
 172                 u"auth:login:username:{}".format(
 173                     md5_text(login_form.clean_username(request.POST["username"])).hexdigest()
 174                 ),
 175                 limit=10,
 176                 window=60,  # 10 per minute should be enough for anyone
 177             ):
 178                 login_form.errors["__all__"] = [
 179                     u"You have made too many login attempts. Please try again later."
 180                 ]
 181                 metrics.incr(
 182                     "login.attempt", instance="rate_limited", skip_internal=True, sample_rate=1.0
 183                 )
 184             elif login_form.is_valid():
 185                 user = login_form.get_user()
 186 
 187                 auth.login(request, user, organization_id=organization.id if organization else None)
 188                 metrics.incr(
 189                     "login.attempt", instance="success", skip_internal=True, sample_rate=1.0
 190                 )
 191 
 192                 if not user.is_active:
 193                     return self.redirect(reverse("sentry-reactivate-account"))
 194 
 195                 return self.redirect(auth.get_login_redirect(request))
 196             else:
 197                 metrics.incr(
 198                     "login.attempt", instance="failure", skip_internal=True, sample_rate=1.0
 199                 )
 200 
 201         context = {
 202             "op": op or "login",
 203             "server_hostname": get_server_hostname(),
 204             "login_form": login_form,
 205             "organization": organization,
 206             "register_form": register_form,
 207             "CAN_REGISTER": can_register,
 208             "join_request_link": self.get_join_request_link(organization),
 209         }
 210         context.update(additional_context.run_callbacks(request))
 211 
 212         return self.respond_login(request, context, **kwargs)

 214     def handle_authenticated(self, request):
 215         next_uri = self.get_next_uri(request)
 216         if auth.is_valid_redirect(next_uri, host=request.get_host()):
 217             return self.redirect(next_uri)
 218         return self.redirect_to_org(request)

 225     # XXX(dcramer): OAuth provider hooks this view
 226     def get(self, request, **kwargs):
 227         next_uri = self.get_next_uri(request)
 228         if request.user.is_authenticated():
 229             # if the user is a superuser, but not 'superuser authenticated'
 230             # we allow them to re-authenticate to gain superuser status
 231             if not request.user.is_superuser or is_active_superuser(request):
 232                 return self.handle_authenticated(request)
 233 
 234         request.session.set_test_cookie()
 235 
 236         # we always reset the state on GET so you dont end up at an odd location
 237         auth.initiate_login(request, next_uri)
 238 
 239         # Single org mode -- send them to the org-specific handler
 240         if settings.SENTRY_SINGLE_ORGANIZATION:
 241             org = Organization.get_default()
 242             next_uri = reverse("sentry-auth-organization", args=[org.slug])
 243             return HttpResponseRedirect(next_uri)
 244 
 245         session_expired = "session_expired" in request.COOKIES
 246         if session_expired:
 247             messages.add_message(request, messages.WARNING, WARN_SESSION_EXPIRED)
 248 
 249         response = self.handle_basic_auth(request, **kwargs)
 250 
 251         if session_expired:
 252             response.delete_cookie("session_expired")
 253 
 254         return response

 256     # XXX(dcramer): OAuth provider hooks this view
 257     def post(self, request, **kwargs):
 258         op = request.POST.get("op")
 259         if op == "sso" and request.POST.get("organization"):
 260             auth_provider = self.get_auth_provider(request.POST["organization"])
 261             if auth_provider:
 262                 next_uri = reverse("sentry-auth-organization", args=[request.POST["organization"]])
 263             else:
 264                 next_uri = request.get_full_path()
 265                 messages.add_message(request, messages.ERROR, ERR_NO_SSO)
 266 
 267             return HttpResponseRedirect(next_uri)
 268 
 269         return self.handle_basic_auth(request, **kwargs)
