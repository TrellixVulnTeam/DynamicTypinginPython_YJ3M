  57 def make_message(random, length=None):
  58     if length is None:
  59         length = int(random.weibullvariate(8, 3))
  60     return " ".join(random.choice(loremipsum.words) for _ in range(length))

  63 def make_culprit(random):
  64     def make_module_path_components(min, max):
  65         for _ in range(random.randint(min, max)):
  66             yield "".join(
  67                 random.sample(loremipsum.words, random.randint(1, int(random.paretovariate(2.2))))
  68             )
  69 
  70     return u"{module} in {function}".format(
  71         module=".".join(make_module_path_components(1, 4)), function=random.choice(loremipsum.words)
  72     )

  89 def make_group_generator(random, project):
  90     epoch = to_timestamp(datetime(2016, 6, 1, 0, 0, 0, tzinfo=timezone.utc))
  91     for id in itertools.count(1):
  92         first_seen = epoch + random.randint(0, 60 * 60 * 24 * 30)
  93         last_seen = random.randint(first_seen, first_seen + (60 * 60 * 24 * 30))
  94 
  95         culprit = make_culprit(random)
  96         level = random.choice(LOG_LEVELS.keys())
  97         message = make_message(random)
  98 
  99         group = Group(
 100             id=id,
 101             project=project,
 102             culprit=culprit,
 103             level=level,
 104             message=message,
 105             first_seen=to_datetime(first_seen),
 106             last_seen=to_datetime(last_seen),
 107             status=random.choice((GroupStatus.UNRESOLVED, GroupStatus.RESOLVED)),
 108             data={"type": "default", "metadata": {"title": message}},
 109         )
 110 
 111         if random.random() < 0.8:
 112             group.data = make_group_metadata(random, group)
 113 
 114         yield group

 390     from sentry.tasks import reports
 391 
 392     random = get_random(request)
 393 
 394     duration = 60 * 60 * 24 * 7
 395     timestamp = to_timestamp(
 396         reports.floor_to_utc_day(
 397             to_datetime(
 398                 random.randint(
 399                     to_timestamp(datetime(2015, 6, 1, 0, 0, 0, tzinfo=timezone.utc)),
 400                     to_timestamp(datetime(2016, 7, 1, 0, 0, 0, tzinfo=timezone.utc)),
 401                 )
 402             )
 403         )
 404     )
 405 
 406     start, stop = interval = reports._to_interval(timestamp, duration)
 407 
 408     organization = Organization(id=1, slug="example", name="Example")
 409 
 410     projects = []
 411     for i in xrange(0, random.randint(1, 8)):
 412         name = " ".join(random.sample(loremipsum.words, random.randint(1, 4)))
 413         projects.append(
 414             Project(
 415                 id=i,
 416                 organization=organization,
 417                 slug=slugify(name),
 418                 name=name,
 419                 date_added=start - timedelta(days=random.randint(0, 120)),

 423     def make_release_generator():
 424         id_sequence = itertools.count(1)
 425         while True:
 426             dt = to_datetime(random.randint(timestamp - (30 * 24 * 60 * 60), timestamp))
 427             p = random.choice(projects)
 428             yield Release(
 429                 id=next(id_sequence),
 430                 project=p,
 431                 organization_id=p.organization_id,
 432                 version="".join([random.choice("0123456789abcdef") for _ in range(40)]),
 433                 date_added=dt,
 434             )
 435 
 436     def build_issue_summaries():
 437         summaries = []
 438         for i in range(3):
 439             summaries.append(int(random.weibullvariate(10, 1) * random.paretovariate(0.5)))
 440         return summaries
 441 
 442     def build_usage_summary():
 443         return (
 444             int(random.weibullvariate(3, 1) * random.paretovariate(0.2)),
 445             int(random.weibullvariate(5, 1) * random.paretovariate(0.2)),
 446         )
 447 
 448     def build_calendar_data(project):
 449         start, stop = reports.get_calendar_query_range(interval, 3)
 450         rollup = 60 * 60 * 24
 451         series = []
 452 
 453         weekend = frozenset((5, 6))
 454         value = int(random.weibullvariate(5000, 3))
 455         for timestamp in tsdb.get_optimal_rollup_series(start, stop, rollup)[1]:
 456             damping = random.uniform(0.2, 0.6) if to_datetime(timestamp).weekday in weekend else 1
 457             jitter = random.paretovariate(1.2)
 458             series.append((timestamp, int(value * damping * jitter)))
 459             value = value * random.uniform(0.25, 2)
 460 
 461         return reports.clean_calendar_data(project, series, start, stop, rollup, stop)
 462 
 463     def build_report(project):
 464         daily_maximum = random.randint(1000, 10000)
 465 
 466         rollup = 60 * 60 * 24
 467         series = [
 468             (
 469                 timestamp + (i * rollup),
 470                 (random.randint(0, daily_maximum), random.randint(0, daily_maximum)),
 471             )
 472             for i in xrange(0, 7)
 473         ]
 474 
 475         aggregates = [
 476             random.randint(0, daily_maximum * 7) if random.random() < 0.9 else None
 477             for _ in xrange(0, 4)
 478         ]
 479 
 480         return reports.Report(
 481             series,
 482             aggregates,
 483             build_issue_summaries(),
 484             build_usage_summary(),
 485             build_calendar_data(project),
 486         )
 487 
 488     if random.random() < 0.85:
 489         personal = {"resolved": random.randint(0, 100), "users": int(random.paretovariate(0.2))}
 490     else:
 491         personal = {"resolved": 0, "users": 0}
 492 
 493     return MailPreview(
 494         html_template="sentry/emails/reports/body.html",
 495         text_template="sentry/emails/reports/body.txt",
 496         context={
 497             "duration": reports.durations[duration],
 498             "interval": {"start": reports.date_format(start), "stop": reports.date_format(stop)},
 499             "report": reports.to_context(
 500                 organization, interval, {project: build_report(project) for project in projects}
 501             ),
 502             "organization": organization,
 503             "personal": personal,
 504             "user": request.user,
 505         },
 506     ).render(request)
