  13 class Error500View(View):
  14     def get_embed_config(self, request):
  15         if not hasattr(request, "sentry"):
  16             return
  17 
  18         try:
  19             projectkey = ProjectKey.objects.filter(project=settings.SENTRY_PROJECT)[0]
  20         except Exception:
  21             logging.exception("Unable to fetch ProjectKey for internal project")
  22             return
  23 
  24         result = {"dsn": projectkey.dsn_public, "eventId": request.sentry["id"]}
  25         if hasattr(request, "user") and request.user.is_authenticated():
  26             try:
  27                 result.update({"userName": request.user.name, "userEmail": request.user.email})
  28             except Exception:
  29                 logging.exception("Unable to fetch user information for embed")
  30         return result
  31 
  32     def dispatch(self, request):
  33         context = {}
  34         embed_config = self.get_embed_config(request)
  35         if embed_config:
  36             context["embed_config"] = json.dumps_htmlsafe(embed_config)
  37 
  38         return render_to_response("sentry/500.html", status=500, context=context, request=request)
