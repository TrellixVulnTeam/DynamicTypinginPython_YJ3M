  23 def _get_version_info():
  24     current = sentry.VERSION
  25 
  26     latest = options.get("sentry:latest_version") or current
  27     upgrade_available = parse_version(latest) > parse_version(current)
  28     build = sentry.__build__ or current
  29 
  30     return {
  31         "current": current,
  32         "latest": latest,
  33         "build": build,
  34         "upgradeAvailable": upgrade_available,
  35     }

  38 def _needs_upgrade():
  39     version_configured = options.get("sentry:version-configured")
  40     if not version_configured:
  41         # If we were never previously upgraded (being a new install)
  42         # we want to force an upgrade, even if the values are set.
  43         return True
  44 
  45     smtp_disabled = not is_smtp_enabled()
  46 
  47     # Check all required options to see if they've been set
  48     for key in options.filter(flag=options.FLAG_REQUIRED):
  49         # ignore required flags which can be empty
  50         if key.flags & options.FLAG_ALLOW_EMPTY:
  51             continue
  52         # Ignore mail.* keys if smtp is disabled
  53         if smtp_disabled and key.name[:5] == "mail.":
  54             continue
  55         if not options.isset(key.name):
  56             return True
  57 
  58     if version_configured != sentry.get_version():
  59         # Everything looks good, but version changed, so let's bump it
  60         options.set("sentry:version-configured", sentry.get_version())
  61 
  62     return False

  65 def _get_statuspage():
  66     id = settings.STATUS_PAGE_ID
  67     if id is None:
  68         return None
  69     return {"id": id, "api_host": settings.STATUS_PAGE_API_HOST}

  72 def _get_project_key(project_id):
  73     try:
  74         return ProjectKey.objects.filter(
  75             project=project_id, roles=F("roles").bitor(ProjectKey.roles.store)
  76         )[0]
  77     except IndexError:
  78         return None

  81 def _get_public_dsn():
  82     if settings.SENTRY_FRONTEND_DSN:
  83         return settings.SENTRY_FRONTEND_DSN
  84 
  85     project_id = settings.SENTRY_FRONTEND_PROJECT or settings.SENTRY_PROJECT
  86     cache_key = "dsn:%s" % (project_id,)
  87 
  88     result = cache.get(cache_key)
  89     if result is None:
  90         key = _get_project_key(project_id)
  91         if key:
  92             result = key.dsn_public
  93         else:
  94             result = ""
  95         cache.set(cache_key, result, 60)
  96     return result

 106 def get_client_config(request=None):
 107     """
 108     Provides initial bootstrap data needed to boot the frontend application.
 109     """
 110     if request is not None:
 111         user = getattr(request, "user", None) or AnonymousUser()
 112         messages = get_messages(request)
 113         session = getattr(request, "session", None)
 114         is_superuser = is_active_superuser(request)
 115         language_code = getattr(request, "LANGUAGE_CODE", "en")
 116 
 117         # User identity is used by the sentry SDK
 118         user_identity = {"ip_address": request.META["REMOTE_ADDR"]}
 119         if user and user.is_authenticated():
 120             user_identity.update({"email": user.email, "id": user.id, "isStaff": user.is_staff})
 121             if user.name:
 122                 user_identity["name"] = user.name
 123     else:
 124         user = None
 125         user_identity = {}
 126         messages = []
 127         session = None
 128         is_superuser = False
 129         language_code = "en"
 130 
 131     enabled_features = []
 132     if features.has("organizations:create", actor=user):
 133         enabled_features.append("organizations:create")
 134     if auth.has_user_registration():
 135         enabled_features.append("auth:register")
 136 
 137     version_info = _get_version_info()
 138 
 139     needs_upgrade = False
 140 
 141     if is_superuser:
 142         needs_upgrade = _needs_upgrade()
 143 
 144     public_dsn = _get_public_dsn()
 145 
 146     context = {
 147         "singleOrganization": settings.SENTRY_SINGLE_ORGANIZATION,
 148         "supportEmail": get_support_mail(),
 149         "urlPrefix": options.get("system.url-prefix"),
 150         "version": version_info,
 151         "features": enabled_features,
 152         "distPrefix": get_asset_url("sentry", "dist/"),
 153         "needsUpgrade": needs_upgrade,
 154         "dsn": public_dsn,
 155         "dsn_requests": _get_dsn_requests(),
 156         "statuspage": _get_statuspage(),
 157         "messages": [{"message": msg.message, "level": msg.tags} for msg in messages],
 158         "apmSampling": float(settings.SENTRY_APM_SAMPLING or 0),
 159         "isOnPremise": settings.SENTRY_ONPREMISE,
 160         "invitesEnabled": settings.SENTRY_ENABLE_INVITES,
 161         "gravatarBaseUrl": settings.SENTRY_GRAVATAR_BASE_URL,
 162         "termsUrl": settings.TERMS_URL,
 163         "privacyUrl": settings.PRIVACY_URL,
 164         # Note `lastOrganization` should not be expected to update throughout frontend app lifecycle
 165         # It should only be used on a fresh browser nav to a path where an
 166         # organization is not in context
 167         "lastOrganization": session["activeorg"] if session and "activeorg" in session else None,
 168         "languageCode": language_code,
 169         "userIdentity": user_identity,
 170         "csrfCookieName": settings.CSRF_COOKIE_NAME,
 171         "sentryConfig": {
 172             "dsn": public_dsn,
 173             "release": settings.SENTRY_SDK_CONFIG["release"],
 174             "environment": settings.SENTRY_SDK_CONFIG["environment"],
 175             # By default `ALLOWED_HOSTS` is [*], however the JS SDK does not support globbing
 176             "whitelistUrls": (
 177                 settings.SENTRY_FRONTEND_WHITELIST_URLS
 178                 if settings.SENTRY_FRONTEND_WHITELIST_URLS
 179                 else list("" if settings.ALLOWED_HOSTS == ["*"] else settings.ALLOWED_HOSTS)
 180             ),
 181         },
 182     }
 183     if user and user.is_authenticated():
 184         context.update(
 185             {"isAuthenticated": True, "user": serialize(user, user, DetailedUserSerializer())}
 186         )
 187 
 188         if request.user.is_superuser:
 189             # Note: This intentionally does not use the "active" superuser flag as
 190             # the frontend should only ever use this flag as a hint that the user can be a superuser
 191             # the API will always need to check for active superuser.
 192             #
 193             # This is needed in the case where you access a different org and get denied, but the UI
 194             # can open the sudo dialog if you are an "inactive" superuser
 195             context["user"]["isSuperuser"] = request.user.is_superuser
 196     else:
 197         context.update({"isAuthenticated": False, "user": None})
 198 
 199     return context
