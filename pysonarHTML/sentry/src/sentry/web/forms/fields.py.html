  16 class CustomTypedChoiceField(TypedChoiceField):
  17     # A patched version of TypedChoiceField which correctly validates a 0
  18     # as a real input that may be invalid
  19     # See https://github.com/django/django/pull/3774
  20     def validate(self, value):
  21         """
  22         Validates that the input is in self.choices.
  23         """
  24         super(CustomTypedChoiceField, self).validate(value)
  25         # this will validate itself twice due to the internal ChoiceField
  26         # validation
  27         if value is not None and not self.valid_value(value):
  28             raise ValidationError(
  29                 self.error_messages["invalid_choice"],
  30                 code="invalid_choice",
  31                 params={"value": value},
  32             )

  35 class UserField(CharField):
  36     class widget(TextInput):
  37         def render(self, name, value, attrs=None):
  38             if not attrs:
  39                 attrs = {}
  40             if "placeholder" not in attrs:
  41                 attrs["placeholder"] = "username"
  42             if isinstance(value, six.integer_types):
  43                 value = User.objects.get(id=value).username
  44             return super(UserField.widget, self).render(name, value, attrs)
  45 
  46     def clean(self, value):
  47         value = super(UserField, self).clean(value)
  48         if not value:
  49             return None
  50         try:
  51             return User.objects.get(username=value, is_active=True)
  52         except User.DoesNotExist:
  53             raise ValidationError(_("Invalid username"))

  56 class ReadOnlyTextWidget(Widget):
  57     def render(self, name, value, attrs):
  58         final_attrs = self.build_attrs(attrs)
  59         if not value:
  60             value = mark_safe("<em>%s</em>" % _("Not set"))
  61         return format_html("<div{0}>{1}</div>", flatatt(final_attrs), value)

  77 def email_address_validator(value):
