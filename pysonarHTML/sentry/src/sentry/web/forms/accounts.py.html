  25 def _get_timezone_choices():
  26     results = []
  27     for tz in pytz.common_timezones:
  28         now = datetime.now(pytz.timezone(tz))
  29         offset = now.strftime("%z")
  30         results.append((int(offset), tz, "(UTC%s) %s" % (offset, tz)))
  31     results.sort()
  32 
  33     for i in range(len(results)):
  34         results[i] = results[i][1:]
  35     return results

  99     def _is_ip_rate_limited(self):
 100         limit = options.get("auth.ip-rate-limit")
 101         if not limit:
 102             return False
 103 
 104         ip_address = self.request.META["REMOTE_ADDR"]
 105         return ratelimiter.is_limited(u"auth:ip:{}".format(ip_address), limit)
 106 
 107     def _is_user_rate_limited(self):
 108         limit = options.get("auth.user-rate-limit")
 109         if not limit:
 110             return False
 111 
 112         username = self.cleaned_data.get("username")
 113         if not username:
 114             return False
 115 
 116         return ratelimiter.is_limited(u"auth:username:{}".format(username), limit)

 118     def clean(self):
 119         username = self.cleaned_data.get("username")
 120         password = self.cleaned_data.get("password")
 121 
 122         if not (username and password):
 123             raise forms.ValidationError(
 124                 self.error_messages["invalid_login"]
 125                 % {"username": self.username_field.verbose_name}
 126             )
 127 
 128         if self.is_rate_limited():
 129             logger.info(
 130                 "user.auth.rate-limited",
 131                 extra={"ip_address": self.request.META["REMOTE_ADDR"], "username": username},
 132             )
 133             raise forms.ValidationError(self.error_messages["rate_limited"])
 134 
 135         self.user_cache = authenticate(username=username, password=password)
 136         if self.user_cache is None:
 137             raise forms.ValidationError(
 138                 self.error_messages["invalid_login"]
 139                 % {"username": self.username_field.verbose_name}
 140             )
 141 
 142         self.check_for_test_cookie()
 143         return self.cleaned_data
 144 
 145     def check_for_test_cookie(self):
 146         if self.request and not self.request.session.test_cookie_worked():
 147             raise forms.ValidationError(self.error_messages["no_cookies"])
 148

 150         if self.user_cache:
 151             return self.user_cache.id
 152         return None

 155         return self.user_cache

 250     user = forms.CharField(
 251         label=_("Account"),
 252         max_length=128,

 254     )

 256     def clean_user(self):
 257         value = (self.cleaned_data.get("user") or "").strip()
 258         if not value:
 259             return
 260         users = find_users(value, with_valid_password=False)
 261         if not users:
 262             return
 263 
 264         # If we find more than one user, we likely matched on email address.
 265         # We silently bail here as we emailing the 'wrong' person isn't great.
 266         # They will have to retry with their username which is guaranteed
 267         # to be unique
 268         if len(users) > 1:
 269             return
 270 
 271         users = [u for u in users if not u.is_managed]
 272         if not users:
 273             raise forms.ValidationError(
 274                 _(
 275                     "The account you are trying to recover is managed and does not support password recovery."
 276                 )
 277             )
 278         return users[0]

 344     def __init__(self, user, request, *args, **kwargs):
 345         self.user = user
 346         self.request = request
 347         super(AccountSettingsForm, self).__init__(*args, **kwargs)
 348 
 349         needs_password = user.has_usable_password()
 350 
 351         if self.user.is_managed:
 352             # username and password always managed, email and

 354             for field in ("email", "name", "username"):
 355                 if field == "username" or field in settings.SENTRY_MANAGED_USER_FIELDS:
 356                     self.fields[field] = ReadOnlyTextField(label=self.fields[field].label)

 370     def is_readonly(self):
 371         if self.user.is_managed:
 372             return set(("email", "name")) == set(settings.SENTRY_MANAGED_USER_FIELDS)
 373         return False

 375     def _clean_managed_field(self, field):
 376         if self.user.is_managed and (
 377             field == "username" or field in settings.SENTRY_MANAGED_USER_FIELDS
 378         ):
 379             return getattr(self.user, field)
 380         return self.cleaned_data[field]

 382     def clean_email(self):
 383         value = self._clean_managed_field("email").lower()
 384         if self.user.email.lower() == value:
 385             return value
 386         if (
 387             User.objects.filter(Q(email__iexact=value) | Q(username__iexact=value))
 388             .exclude(id=self.user.id)
 389             .exists()
 390         ):
 391             raise forms.ValidationError(
 392                 _("There was an error adding %s: that email is already in use")
 393                 % self.cleaned_data["email"]
 394             )
 395         return value
 396 
 397     def clean_name(self):
 398         return self._clean_managed_field("name")
 399 
 400     def clean_username(self):
 401         value = self._clean_managed_field("username")
 402         if User.objects.filter(username__iexact=value).exclude(id=self.user.id).exists():
 403             raise forms.ValidationError(_("That username is already in use."))
 404         return value

 406     def clean_password(self):
 407         value = self.cleaned_data.get("password")
 408         if value and not self.user.check_password(value):
 409             raise forms.ValidationError("The password you entered is not correct.")
 410         elif not value and (
 411             self.cleaned_data.get("email", self.user.email) != self.user.email
 412             or self.cleaned_data.get("new_password")
 413         ):
 414             raise forms.ValidationError("You must confirm your current password to make changes.")
 415         return value
 416 
 417     def clean_verify_new_password(self):
 418         new_password = self.cleaned_data.get("new_password")
 419 
 420         if new_password:
 421             verify_new_password = self.cleaned_data.get("verify_new_password")
 422             if verify_new_password is None:
 423                 raise forms.ValidationError("You must verify your new password.")
 424 
 425             if new_password != verify_new_password:
 426                 raise forms.ValidationError("Your new password and verify new password must match.")
 427 
 428             return verify_new_password
 429 
 430     def clean_new_password(self):
 431         new_password = self.cleaned_data.get("new_password")
 432         if new_password:
 433             password_validation.validate_password(new_password)
 434         return new_password
 435 
 436     def save(self, commit=True):
 437         if self.cleaned_data.get("new_password"):
 438             self.user.set_password(self.cleaned_data["new_password"])
 439             self.user.refresh_session_nonce(self.request)
 440 
 441             capture_security_activity(
 442                 account=self.user,
 443                 type="password-changed",
 444                 actor=self.request.user,
 445                 ip_address=self.request.META["REMOTE_ADDR"],
 446                 send_email=True,
 447             )
 448 
 449         self.user.name = self.cleaned_data["name"]
 450 
 451         if self.cleaned_data["email"] != self.user.email:
 452             new_username = self.user.email == self.user.username
 453         else:
 454             new_username = False
 455 
 456         self.user.email = self.cleaned_data["email"]
 457 
 458         if self.cleaned_data.get("username"):
 459             self.user.username = self.cleaned_data["username"]
 460         elif new_username and not User.objects.filter(username__iexact=self.user.email).exists():
 461             self.user.username = self.user.email
 462 
 463         if commit:
 464             self.user.save()
 465 
 466         return self.user

 787     def clean_password(self):
 788         value = self.cleaned_data.get("password")
 789         if value and not self.user.check_password(value):
 790             raise forms.ValidationError(_("The password you entered is not correct."))
 791         elif not value:
 792             raise forms.ValidationError(
 793                 _("You must confirm your current password to make changes.")
 794             )
 795         return value
