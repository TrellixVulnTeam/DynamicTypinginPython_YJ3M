  23 def get_all_languages():
  24     results = []
  25     for path in os.listdir(os.path.join(MODULE_ROOT, "locale")):
  26         if path.startswith("."):
  27             continue
  28         if "_" in path:
  29             pre, post = path.split("_", 1)
  30             path = u"{}-{}".format(pre, post.lower())
  31         results.append(path)
  32     return results

 279 def _load_platform_data():
 280     INTEGRATION_ID_TO_PLATFORM_DATA.clear()
 281     data = load_doc("_platforms")
 282 
 283     if not data:
 284         return
 285 
 286     for platform in data["platforms"]:
 287         integrations = platform.pop("integrations")
 288         if integrations:
 289             for integration in integrations:
 290                 integration_id = integration.pop("id")
 291                 if integration["type"] != "language":
 292                     integration["language"] = platform["id"]
 293                 INTEGRATION_ID_TO_PLATFORM_DATA[integration_id] = integration

 352 # to go from event data to the integration id (in _platforms.json),
 353 # for example an event like:
 354 # {"platform": "java",
 355 #  "sdk": {"name": "sentry-java",
 356 #          "integrations": ["java.util.logging"]}} -> java-logging
 357 def get_integration_id_for_event(platform, sdk_name, integrations):
 358     if integrations:
 359         for integration in integrations:
 360             # check special cases
 361             if (
 362                 platform in PLATFORM_INTEGRATION_TO_INTEGRATION_ID
 363                 and integration in PLATFORM_INTEGRATION_TO_INTEGRATION_ID[platform]
 364             ):
 365                 return PLATFORM_INTEGRATION_TO_INTEGRATION_ID[platform][integration]
 366 
 367             # try <platform>-<integration>, for example "java-log4j"
 368             integration_id = "%s-%s" % (platform, integration)
 369             if integration_id in INTEGRATION_ID_TO_PLATFORM_DATA:
 370                 return integration_id
 371 
 372     # try sdk name, for example "sentry-java" -> "java" or "raven-java:log4j" -> "java-log4j"
 373     sdk_name = sdk_name.lower().replace("sentry-", "").replace("raven-", "").replace(":", "-")
 374     if sdk_name in INTEGRATION_ID_TO_PLATFORM_DATA:
 375         return sdk_name
 376 
 377     # try platform name, for example "java"
 378     if platform in INTEGRATION_ID_TO_PLATFORM_DATA:
 379         return platform

 503 @unique

 512     @classmethod

 514         if event_type == "error":

 517             return DataCategory.TRANSACTION
 518         elif event_type in ("csp", "hpkp", "expectct", "expectstaple"):
 519             return DataCategory.SECURITY
 520         return DataCategory.DEFAULT

 524         return [

 527             DataCategory.TRANSACTION,
 528             DataCategory.SECURITY,

 533         return [DataCategory.DEFAULT, DataCategory.ERROR]
