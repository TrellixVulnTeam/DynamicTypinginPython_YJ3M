  22 class SafeConnectionMixin(object):
  23     """
  24     HACK(mattrobenolt): Most of this is yanked out of core urllib3
  25     to override `_new_conn` with the ability to create our own socket.
  26     """
  27 
  28     # These `host` properties need rebound otherwise `self._dns_host` doesn't
  29     # get set correctly.
  30     @property
  31     def host(self):
  32         """
  33         Getter method to remove any trailing dots that indicate the hostname is an FQDN.
  34         In general, SSL certificates don't include the trailing dot indicating a
  35         fully-qualified domain name, and thus, they don't validate properly when
  36         checked against a domain name that includes the dot. In addition, some
  37         servers may not expect to receive the trailing dot when provided.
  38         However, the hostname with trailing dot is critical to DNS resolution; doing a
  39         lookup with the trailing dot will properly only resolve the appropriate FQDN,
  40         whereas a lookup without a trailing dot will search the system's search domain
  41         list. Thus, it's important to keep the original host around for use only in
  42         those cases where it's appropriate (i.e., when doing DNS lookup to establish the
  43         actual TCP connection across which we're going to send HTTP requests).
  44         """
  45         return self._dns_host.rstrip(".")
  46 
  47     @host.setter
  48     def host(self, value):
  49         """
  50         Setter for the `host` property.
  51         We assume that only urllib3 uses the _dns_host attribute; httplib itself
  52         only uses `host`, and it seems reasonable that other libraries follow suit.
  53         """
  54         self._dns_host = value
  55 
  56     # Mostly yanked from https://github.com/urllib3/urllib3/blob/1.22/urllib3/connection.py#L127
  57     def _new_conn(self):
  58         """ Establish a socket connection and set nodelay settings on it.
  59         :return: New socket connection.
  60         """
  61         extra_kw = {}
  62         if self.source_address:
  63             extra_kw["source_address"] = self.source_address
  64 
  65         if self.socket_options:
  66             extra_kw["socket_options"] = self.socket_options
  67 
  68         try:
  69             # HACK(mattrobenolt): All of this is to replace this one line
  70             # to establish our own connection.
  71             conn = safe_create_connection((self._dns_host, self.port), self.timeout, **extra_kw)
  72 
  73         except SocketTimeout:
  74             raise ConnectTimeoutError(
  75                 self, "Connection to %s timed out. (connect timeout=%s)" % (self.host, self.timeout)
  76             )
  77 
  78         except SocketError as e:
  79             raise NewConnectionError(self, "Failed to establish a new connection: %s" % e)
  80 
  81         return conn
  82

 100 class SafePoolManager(PoolManager):
 101     """
 102     This custom PoolManager is needed to override
 103     pool_classes_by_scheme which allows us to set which
 104     ConnectionPool classes to create.
 105     """
 106 
 107     def __init__(self, *args, **kwargs):
 108         PoolManager.__init__(self, *args, **kwargs)
 109         self.pool_classes_by_scheme = {
 110             "http": SafeHTTPConnectionPool,
 111             "https": SafeHTTPSConnectionPool,

 115 class BlacklistAdapter(HTTPAdapter):
 116     """
 117     We need a custom HTTPAdapter to initialize our custom SafePoolManager
 118     rather than the default PoolManager.
 119     """
 120 
 121     def init_poolmanager(self, connections, maxsize, block=DEFAULT_POOLBLOCK, **pool_kwargs):
 122         self._pool_connections = connections
 123         self._pool_maxsize = maxsize
 124         self._pool_block = block
 125         self.poolmanager = SafePoolManager(
 126             num_pools=connections, maxsize=maxsize, block=block, strict=True, **pool_kwargs

 130 class TimeoutAdapter(HTTPAdapter):
 131     def __init__(self, *args, **kwargs):
 132         timeout = kwargs.pop("timeout", None)
 133         HTTPAdapter.__init__(self, *args, **kwargs)
 134         if timeout is None:
 135             timeout = 10.0
 136         self.default_timeout = timeout
 137 
 138     def send(self, *args, **kwargs):
 139         if kwargs.get("timeout") is None:
 140             kwargs["timeout"] = self.default_timeout
 141         return HTTPAdapter.send(self, *args, **kwargs)

 147 class Session(_Session):
 148     def request(self, *args, **kwargs):
 149         kwargs.setdefault("timeout", 30)
 150         response = _Session.request(self, *args, **kwargs)
 151         # requests' attempts to use chardet internally when no encoding is found
 152         # and we want to avoid that slow behavior
 153         if not response.encoding:
 154             response.encoding = "utf-8"
 155         return response

 158 class SafeSession(Session):
 159     def __init__(self):
 160         Session.__init__(self)
 161         self.headers.update({"User-Agent": USER_AGENT})
 162         adapter = BlacklistAdapter()
 163         self.mount("https://", adapter)
 164         self.mount("http://", adapter)

 167 class UnixHTTPConnection(HTTPConnection):
 168     default_socket_options = []
 169 
 170     def __init__(self, host, **kwargs):
 171         # We're using the `host` as the socket path, but
 172         # urllib3 uses this host as the Host header by default.
 173         # If we send along the socket path as a Host header, this is
 174         # never what you want and would typically be malformed value.
 175         # So we fake this by sending along `localhost` by default as
 176         # other libraries do.
 177         self.socket_path = host
 178         super(UnixHTTPConnection, self).__init__(host="localhost", **kwargs)
 179 
 180     def _new_conn(self):
 181         sock = socket.socket(socket.AF_UNIX, socket.SOCK_STREAM)
 182 
 183         # If provided, set socket level options before connecting.
 184         _set_socket_options(sock, self.socket_options)
 185 
 186         if self.timeout is not socket._GLOBAL_DEFAULT_TIMEOUT:
 187             sock.settimeout(self.timeout)
 188         sock.connect(self.socket_path)
 189         return sock

 199 def connection_from_url(endpoint, **kw):
 200     if endpoint[:1] == "/":
 201         return UnixHTTPConnectionPool(endpoint, **kw)
 202     return _connection_from_url(endpoint, **kw)
