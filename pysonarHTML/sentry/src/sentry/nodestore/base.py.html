  49     def delete_multi(self, id_list):
  50         """
  51         Delete multiple nodes.
  52 
  53         Note: This is not guaranteed to be atomic and may result in a partial
  54         delete.
  55 
  56         >>> delete_multi(['key1', 'key2'])

  59             self.delete(id)

  68     def get_multi(self, id_list):

 101     def _get_cache_item(self, id):
 102         if self.cache:
 103             return self.cache.get(id)
 104 
 105     def _get_cache_items(self, id_list):
 106         if self.cache:
 107             return self.cache.get_many(id_list)
 108         return {}
 109 
 110     def _set_cache_item(self, id, data):
 111         if self.cache and data:
 112             self.cache.set(id, data)
 113 
 114     def _set_cache_items(self, items):
 115         cacheable_items = {k: v for k, v in six.iteritems(items) if v}
 116         if self.cache:
 117             self.cache.set_many(cacheable_items)

 119     def _delete_cache_item(self, id):
