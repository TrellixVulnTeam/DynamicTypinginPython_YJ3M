  35 def get_connection(project, instance, table, options):
  36     key = (project, instance, table)
  37     try:
  38         # Fast check for an existing connection cached
  39         return _connection_cache[key]
  40     except KeyError:
  41         # if missing, we acquire our lock to initialize a new one
  42         with _connection_lock:
  43             try:
  44                 # It's possible that the lock was blocked waiting
  45                 # on someone else who already initialized, so
  46                 # we first check again to make sure this isn't the case.
  47                 return _connection_cache[key]
  48             except KeyError:
  49                 _connection_cache[key] = (
  50                     bigtable.Client(project=project, **options).instance(instance).table(table)
  51                 )
  52     return _connection_cache[key]

 100     def get(self, id):
 101         item_from_cache = self._get_cache_item(id)
 102         if item_from_cache:
 103             return item_from_cache
 104 
 105         data = self.decode_row(self.connection.read_row(id))
 106         self._set_cache_item(id, data)
 107         return data
 108 
 109     def get_multi(self, id_list):
 110         id_list = list(set(id_list))
 111 
 112         if len(id_list) == 1:
 113             return {id_list[0]: self.get(id_list[0])}
 114 
 115         cache_items = self._get_cache_items(id_list)
 116 
 117         if len(cache_items) == len(id_list):
 118             return cache_items
 119 
 120         uncached_ids = [id for id in id_list if id not in cache_items]
 121         rv = {}
 122         rows = RowSet()
 123         for id in uncached_ids:
 124             rows.add_row_key(id)
 125             rv[id] = None
 126 
 127         for row in self.connection.read_rows(row_set=rows):
 128             rv[row.row_key] = self.decode_row(row)
 129         self._set_cache_items(rv)
 130         rv.update(cache_items)
 131         return rv

 133     def decode_row(self, row):
 134         if row is None:
 135             return None
 136 
 137         columns = row.cells[self.column_family]
 138 
 139         try:
 140             cell = columns[self.data_column][0]
 141         except KeyError:
 142             return None
 143 
 144         # Check if a TTL column exists
 145         # for this row. If there is,
 146         # we can use the `timestamp` property of the
 147         # cells to see if we should return the
 148         # row or not.
 149         if self.ttl_column in columns:
 150             # If we needed the actual value, we could unpack it.
 151             # ttl = struct.unpack('<I', columns[self.ttl_column][0].value)[0]
 152             if cell.timestamp < timezone.now():
 153                 return None
 154 
 155         data = cell.value
 156 
 157         # Read our flags
 158         flags = 0
 159         if self.flags_column in columns:
 160             flags = struct.unpack("B", columns[self.flags_column][0].value)[0]
 161 
 162         # Check for a compression flag on, if so
 163         # decompress the data.
 164         if flags & self._FLAG_COMPRESSED:
 165             data = zlib_decompress(data)
 166 
 167         return json_loads(data)
 168 
 169     def set(self, id, data, ttl=None):
 170         row = self.encode_row(id, data, ttl)
 171         row.commit()
 172         self._set_cache_item(id, data)

 174     def encode_row(self, id, data, ttl=None):
 175         data = json_dumps(data)
 176 
 177         row = self.connection.row(id)
 178         # Call to delete is just a state mutation,
 179         # and in this case is just used to clear all columns
 180         # so the entire row will be replaced. Otherwise,
 181         # if an existing row were mutated, and it took up more
 182         # than one column, it'd be possible to overwrite
 183         # beginning columns and still retain the end ones.
 184         row.delete()
 185 
 186         # If we are setting a TTL on this row,
 187         # we want to set the timestamp of the cells
 188         # into the future. This allows our GC policy
 189         # to delete them when the time comes. It also
 190         # allows us to filter the rows on read if
 191         # we are past the timestamp to not return.
 192         # We want to set a ttl column to the ttl
 193         # value in the future if we wanted to bump the timestamp
 194         # and rewrite a row with a new ttl.
 195         ttl = ttl or self.default_ttl
 196         if ttl is None:
 197             ts = None
 198         else:
 199             ts = timezone.now() + ttl
 200             row.set_cell(
 201                 self.column_family,
 202                 self.ttl_column,
 203                 struct.pack("<I", int(ttl.total_seconds())),
 204                 timestamp=ts,
 205             )
 206 
 207         # Track flags for metadata about this row.
 208         # This only flag we're tracking now is whether compression
 209         # is on or not for the data column.
 210         flags = 0
 211         if self.compression:
 212             flags |= self._FLAG_COMPRESSED
 213             data = zlib_compress(data)
 214 
 215         # Only need to write the column at all if any flags
 216         # are enabled. And if so, pack it into a single byte.
 217         if flags:
 218             row.set_cell(
 219                 self.column_family, self.flags_column, struct.pack("B", flags), timestamp=ts
 220             )
 221 
 222         assert len(data) <= self.max_size
 223 
 224         row.set_cell(self.column_family, self.data_column, data, timestamp=ts)
 225         return row

 227     def delete(self, id):
 228         if self.skip_deletes:
 229             return
 230 
 231         row = self.connection.row(id)
 232         row.delete()
 233         row.commit()
 234         self._delete_cache_item(id)
 235 
 236     def delete_multi(self, id_list):
 237         if self.skip_deletes:
 238             return
 239 
 240         if len(id_list) == 1:
 241             self.delete(id_list[0])
 242             return
 243 
 244         rows = []
 245         for id in id_list:
 246             row = self.connection.row(id)
 247             row.delete()
 248             rows.append(row)
 249 
 250         self.connection.mutate_rows(rows)
 251         self._delete_cache_items(id_list)

 256     def bootstrap(self):
 257         table = (
 258             bigtable.Client(project=self.project, admin=True, **self.options)
 259             .instance(self.instance)
 260             .table(self.table)
 261         )
 262         if table.exists():
 263             return
 264 
 265         # With automatic expiry, we set a GC rule to automatically
 266         # delete rows with an age of 0. This sounds odd, but when
 267         # we write rows, we write them with a future timestamp as long
 268         # as a TTL is set during write. By doing this, we are effectively
 269         # writing rows into the future, and they will be deleted due to TTL
 270         # when their timestamp is passed.
 271         if self.automatic_expiry:
 272             from datetime import timedelta
 273 
 274             # NOTE: Bigtable can't actually use 0 TTL, and
 275             # requires a minimum value of 1ms.
 276             # > InvalidArgument desc = Error in field 'Modifications list' : Error in element #0 : max_age must be at least one millisecond
 277             delta = timedelta(milliseconds=1)
 278             gc_rule = bigtable.column_family.MaxAgeGCRule(delta)
 279         else:
 280             gc_rule = None
 281 
 282         from google.api_core import exceptions
 283         from google.api_core import retry
 284 
 285         retry_504 = retry.Retry(retry.if_exception_type(exceptions.DeadlineExceeded))
 286         retry_504(table.create)(column_families={self.column_family: gc_rule})
