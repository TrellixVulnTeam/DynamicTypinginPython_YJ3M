  13 class DjangoNodeStorage(NodeStorage):
  14     def delete(self, id):
  15         Node.objects.filter(id=id).delete()
  16         self._delete_cache_item(id)
  17 
  18     def get(self, id):
  19         item_from_cache = self._get_cache_item(id)
  20         if item_from_cache:
  21             return item_from_cache
  22         try:
  23             data = Node.objects.get(id=id).data
  24             self._set_cache_item(id, data)
  25             return data
  26         except Node.DoesNotExist:
  27             return None
  28 
  29     def get_multi(self, id_list):
  30         cache_items = self._get_cache_items(id_list)
  31         if len(cache_items) == len(id_list):
  32             return cache_items
  33 
  34         uncached_ids = [id for id in id_list if id not in cache_items]
  35         items = {n.id: n.data for n in Node.objects.filter(id__in=uncached_ids)}
  36         self._set_cache_items(items)
  37         items.update(cache_items)
  38         return items
  39 
  40     def delete_multi(self, id_list):
  41         Node.objects.filter(id__in=id_list).delete()
  42         self._delete_cache_items(id_list)
  43 
  44     def set(self, id, data, ttl=None):
  45         create_or_update(Node, id=id, values={"data": data, "timestamp": timezone.now()})
  46         self._set_cache_item(id, data)
  47 
  48     def cleanup(self, cutoff_timestamp):
  49         from sentry.db.deletion import BulkDeleteQuery
  50 
  51         total_seconds = (timezone.now() - cutoff_timestamp).total_seconds()
  52         days = math.floor(total_seconds / 86400)
  53 
  54         BulkDeleteQuery(model=Node, dtfield="timestamp", days=days).execute()
  55         if self.cache:
  56             self.cache.clear()
  57

  59         # Nothing for Django backend to do during bootstrap
  60         pass
