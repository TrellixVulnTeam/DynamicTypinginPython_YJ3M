  24 def sane_repr(*attrs):
  25     if "id" not in attrs and "pk" not in attrs:
  26         attrs = ("id",) + attrs
  27 
  28     def _repr(self):
  29         cls = type(self).__name__
  30 
  31         pairs = ("%s=%s" % (a, repr(getattr(self, a, None))) for a in attrs)
  32 
  33         return u"<%s at 0x%x: %s>" % (cls, id(self), ", ".join(pairs))
  34 
  35     return _repr

  38 class BaseModel(models.Model):
  39     class Meta:
  40         abstract = True
  41 
  42     objects = BaseManager()
  43 
  44     update = update
  45 
  46     def __init__(self, *args, **kwargs):
  47         super(BaseModel, self).__init__(*args, **kwargs)
  48         self._update_tracked_data()
  49 
  50     def __getstate__(self):
  51         d = self.__dict__.copy()
  52         # we cant serialize weakrefs
  53         d.pop("_Model__data", None)
  54         return d
  55 
  56     def __hash__(self):
  57         # Django decided that it shouldn't let us hash objects even though they have
  58         # memory addresses. We need that behavior, so let's revert.
  59         if self.pk:
  60             return models.Model.__hash__(self)
  61         return id(self)
  62 
  63     def __reduce__(self):
  64         (model_unpickle, stuff, _) = super(BaseModel, self).__reduce__()
  65         return (model_unpickle, stuff, self.__getstate__())
  66 
  67     def __setstate__(self, state):
  68         self.__dict__.update(state)
  69         self._update_tracked_data()
  70 
  71     def __get_field_value(self, field):
  72         if isinstance(type(field).__dict__.get(field.attname), DeferredAttribute):
  73             return DEFERRED
  74         if isinstance(field, models.ForeignKey):
  75             return getattr(self, field.column, None)
  76         return getattr(self, field.attname, None)
  77 
  78     def _update_tracked_data(self):
  79         "Updates a local copy of attributes values"
  80         if self.id:
  81             data = {}
  82             for f in self._meta.fields:
  83                 # XXX(dcramer): this is how Django determines this (copypasta from Model)
  84                 if (
  85                     isinstance(type(f).__dict__.get(f.attname), DeferredAttribute)
  86                     or f.column is None
  87                 ):
  88                     continue
  89                 try:
  90                     v = self.__get_field_value(f)
  91                 except AttributeError as e:
  92                     # this case can come up from pickling
  93                     logging.exception(six.text_type(e))
  94                 else:
  95                     if isinstance(v, BitHandler):
  96                         v = copy(v)
  97                     data[f.column] = v
  98             self.__data = data
  99         else:
 100             self.__data = UNSAVED
 101 
 102     def has_changed(self, field_name):
 103         "Returns ``True`` if ``field`` has changed since initialization."
 104         if self.__data is UNSAVED:
 105             return False
 106         field = self._meta.get_field(field_name)
 107         value = self.__get_field_value(field)
 108         if value is DEFERRED:
 109             return False
 110         return self.__data.get(field_name) != value
 111 
 112     def old_value(self, field_name):
 113         "Returns the previous value of ``field``"
 114         if self.__data is UNSAVED:
 115             return None
 116         value = self.__data.get(field_name)
 117         if value is DEFERRED:
 118             return None
 119         return self.__data.get(field_name)

 139 def __model_pre_save(instance, **kwargs):
 140     if not isinstance(instance, DefaultFieldsModel):
 141         return
 142     instance.date_updated = timezone.now()
 143 
 144 
 145 def __model_post_save(instance, **kwargs):
 146     if not isinstance(instance, BaseModel):
 147         return
 148     instance._update_tracked_data()

 151 def __model_class_prepared(sender, **kwargs):
 152     if not issubclass(sender, BaseModel):
 153         return
 154 
 155     if not hasattr(sender, "__core__"):
 156         raise ValueError(u"{!r} model has not defined __core__".format(sender))
