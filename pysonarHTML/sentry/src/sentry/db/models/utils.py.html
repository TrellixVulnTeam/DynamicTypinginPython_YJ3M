  25 def resolve_combined_expression(instance, node):
  26     def _resolve(instance, node):
  27         if isinstance(node, Value):
  28             return node.value
  29         if isinstance(node, F):
  30             return getattr(instance, node.name)
  31         if isinstance(node, CombinedExpression):
  32             return resolve_combined_expression(instance, node)
  33         return node
  34 
  35     if isinstance(node, Value):
  36         return node.value
  37     if not hasattr(node, "connector"):
  38         raise CannotResolveExpression
  39     op = COMBINED_EXPRESSION_CALLBACKS.get(node.connector, None)
  40     if not op:
  41         raise CannotResolveExpression
  42     if hasattr(node, "children"):
  43         children = node.children
  44     else:
  45         children = [node.lhs, node.rhs]
  46     runner = _resolve(instance, children[0])
  47     for n in children[1:]:
  48         runner = op(runner, _resolve(instance, n))
  49     return runner

  57 def slugify_instance(inst, label, reserved=(), max_length=30, field_name="slug", *args, **kwargs):
  58     base_value = slugify(label)[:max_length]
  59 
  60     if base_value is not None:
  61         base_value = base_value.strip()
  62         if base_value in reserved:
  63             base_value = None
  64 
  65     if not base_value:
  66         base_value = uuid4().hex[:12]
  67 
  68     base_qs = type(inst).objects.all()
  69     if inst.id:
  70         base_qs = base_qs.exclude(id=inst.id)
  71     if args or kwargs:
  72         base_qs = base_qs.filter(*args, **kwargs)
  73 
  74     setattr(inst, field_name, base_value)
  75 
  76     # We don't need to further mutate if we're unique at this point
  77     if not base_qs.filter(**{"{}__iexact".format(field_name): base_value}).exists():
  78         return
  79 
  80     # We want to sanely generate the shortest unique slug possible, so
  81     # we try different length endings until we get one that works, or bail.
  82 
  83     # At most, we have 27 attempts here to derive a unique slug
  84     sizes = (
  85         (1, 2),  # (36^2) possibilities, 2 attempts
  86         (5, 3),  # (36^3) possibilities, 3 attempts
  87         (20, 5),  # (36^5) possibilities, 20 attempts
  88         (1, 12),  # (36^12) possibilities, 1 final attempt
  89     )
  90     for attempts, size in sizes:
  91         for i in range(attempts):
  92             end = get_random_string(size, allowed_chars="abcdefghijklmnopqrstuvwxyz0123456790")
  93             value = base_value[: max_length - size - 1] + "-" + end
  94             setattr(inst, field_name, value)
  95             if not base_qs.filter(**{"{}__iexact".format(field_name): value}).exists():
  96                 return
  97 
  98     # If at this point, we've exhausted all possibilities, we'll just end up hitting
  99     # an IntegrityError from database, which is ok, and unlikely to happen

 102 class Creator(object):
 103     """
 104     A descriptor that invokes `to_python` when attributes are set.
 105     This provides backwards compatibility for fields that used to use
 106     SubfieldBase which will be removed in Django1.10
 107     """
 108 
 109     def __init__(self, field):
 110         self.field = field
 111 
 112     def __get__(self, obj, type=None):
 113         if obj is None:
 114             return self
 115         return obj.__dict__[self.field.name]
 116 
 117     def __set__(self, obj, value):
 118         obj.__dict__[self.field.name] = self.field.to_python(value)
