  10 class ParanoidQuerySet(QuerySet):
  11     """
  12     Prevents objects from being hard-deleted. Instead, sets the
  13     ``date_deleted``, effectively soft-deleting the object.
  14     """
  15 
  16     def delete(self):
  17         self.update(date_deleted=timezone.now())
  18 
  19 
  20 class ParanoidManager(BaseManager):
  21     """
  22     Only exposes objects that have NOT been soft-deleted.
  23     """
  24 
  25     def get_queryset(self):
  26         return ParanoidQuerySet(self.model, using=self._db).filter(date_deleted__isnull=True)
