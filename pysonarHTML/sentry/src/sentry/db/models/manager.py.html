  49 def make_key(model, prefix, kwargs):
  50     kwargs_bits = []
  51     for k, v in sorted(six.iteritems(kwargs)):
  52         k = __prep_key(model, k)
  53         v = smart_text(__prep_value(model, k, v))
  54         kwargs_bits.append("%s=%s" % (k, v))
  55     kwargs_bits = ":".join(kwargs_bits)
  56 
  57     return "%s:%s:%s" % (prefix, model.__name__, md5_text(kwargs_bits).hexdigest())

  79     def __init__(self, *args, **kwargs):
  80         #: Model fields for which we should build up a cache to be used with
  81         #: Model.objects.get_from_cache(fieldname=value)`.
  82         #:
  83         #: Note that each field by its own needs to be a potential primary key
  84         #: (uniquely identify a row), so for example organization slug is ok,
  85         #: project slug is not.
  86         self.cache_fields = kwargs.pop("cache_fields", [])
  87         self.cache_ttl = kwargs.pop("cache_ttl", 60 * 5)
  88         self._cache_version = kwargs.pop("cache_version", None)
  89         self.__local_cache = threading.local()
  90         super(BaseManager, self).__init__(*args, **kwargs)

  93     @contextmanager
  94     def local_cache():
  95         """Enables local caching for the entire process."""
  96         global _local_cache_enabled, _local_cache_generation
  97         if _local_cache_enabled:
  98             raise RuntimeError("nested use of process global local cache")
  99         _local_cache_enabled = True
 100         try:
 101             yield
 102         finally:
 103             _local_cache_enabled = False
 104             _local_cache_generation += 1
 105 
 106     def _get_local_cache(self):
 107         if not _local_cache_enabled:
 108             return
 109 
 110         gen = _local_cache_generation
 111         cache_gen = getattr(_local_cache, "generation", None)
 112 
 113         if cache_gen != gen or not hasattr(_local_cache, "cache"):
 114             _local_cache.cache = {}
 115             _local_cache.generation = gen
 116 
 117         return _local_cache.cache

 148     def __class_prepared(self, sender, **kwargs):
 149         """
 150         Given the cache is configured, connects the required signals for invalidation.
 151         """
 152         post_save.connect(self.post_save, sender=sender, weak=False)
 153         post_delete.connect(self.post_delete, sender=sender, weak=False)
 154 
 155         if not self.cache_fields:
 156             return
 157 
 158         post_init.connect(self.__post_init, sender=sender, weak=False)
 159         post_save.connect(self.__post_save, sender=sender, weak=False)
 160         post_delete.connect(self.__post_delete, sender=sender, weak=False)

 162     def __cache_state(self, instance):
 163         """
 164         Updates the tracked state of an instance.
 165         """
 166         if instance.pk:
 167             self.__cache[instance] = {
 168                 f: self.__value_for_field(instance, f) for f in self.cache_fields
 169             }
 170 
 171     def __post_init(self, instance, **kwargs):
 172         """
 173         Stores the initial state of an instance.
 174         """
 175         self.__cache_state(instance)
 176 
 177     def __post_save(self, instance, **kwargs):
 178         """
 179         Pushes changes to an instance into the cache, and removes invalid (changed)
 180         lookup values.
 181         """
 182         pk_name = instance._meta.pk.name
 183         pk_names = ("pk", pk_name)
 184         pk_val = instance.pk
 185         for key in self.cache_fields:
 186             if key in pk_names:
 187                 continue
 188             # store pointers
 189             value = self.__value_for_field(instance, key)
 190             cache.set(
 191                 key=self.__get_lookup_cache_key(**{key: value}),
 192                 value=pk_val,
 193                 timeout=self.cache_ttl,
 194                 version=self.cache_version,
 195             )
 196 
 197         # Ensure we don't serialize the database into the cache
 198         db = instance._state.db
 199         instance._state.db = None
 200         # store actual object
 201         try:
 202             cache.set(
 203                 key=self.__get_lookup_cache_key(**{pk_name: pk_val}),
 204                 value=instance,
 205                 timeout=self.cache_ttl,
 206                 version=self.cache_version,
 207             )
 208         except Exception as e:
 209             logger.error(e, exc_info=True)
 210         instance._state.db = db
 211 
 212         # Kill off any keys which are no longer valid
 213         if instance in self.__cache:
 214             for key in self.cache_fields:
 215                 if key not in self.__cache[instance]:
 216                     continue
 217                 value = self.__cache[instance][key]
 218                 current_value = self.__value_for_field(instance, key)
 219                 if value != current_value:
 220                     cache.delete(
 221                         key=self.__get_lookup_cache_key(**{key: value}), version=self.cache_version
 222                     )
 223 
 224         self.__cache_state(instance)
 225 
 226     def __post_delete(self, instance, **kwargs):
 227         """
 228         Drops instance from all cache storages.
 229         """
 230         pk_name = instance._meta.pk.name
 231         for key in self.cache_fields:
 232             if key in ("pk", pk_name):
 233                 continue
 234             # remove pointers
 235             value = self.__value_for_field(instance, key)
 236             cache.delete(
 237                 key=self.__get_lookup_cache_key(**{key: value}), version=self.cache_version
 238             )
 239         # remove actual object
 240         cache.delete(
 241             key=self.__get_lookup_cache_key(**{pk_name: instance.pk}), version=self.cache_version
 242         )

 244     def __get_lookup_cache_key(self, **kwargs):
 245         return make_key(self.model, "modelcache", kwargs)

 247     def __value_for_field(self, instance, key):
 248         """
 249         Return the cacheable value for a field.
 250 
 251         ForeignKey's will cache via the primary key rather than using an
 252         instance ref. This is needed due to the way lifecycle of models works
 253         as otherwise we end up doing wasteful queries.
 254         """
 255         if key == "pk":
 256             return instance.pk
 257         field = instance._meta.get_field(key)
 258         return getattr(instance, field.attname)

 264     def get_from_cache(self, **kwargs):
 265         """
 266         Wrapper around QuerySet.get which supports caching of the
 267         intermediate value.  Callee is responsible for making sure
 268         the cache key is cleared on save.
 269         """
 270         if not self.cache_fields or len(kwargs) > 1:
 271             raise ValueError("We cannot cache this query. Just hit the database.")
 272 
 273         key, value = next(six.iteritems(kwargs))
 274         pk_name = self.model._meta.pk.name
 275         if key == "pk":
 276             key = pk_name
 277 
 278         # We store everything by key references (vs instances)
 279         if isinstance(value, Model):
 280             value = value.pk
 281 
 282         # Kill __exact since it's the default behavior
 283         if key.endswith("__exact"):
 284             key = key.split("__exact", 1)[0]
 285 
 286         if key in self.cache_fields or key == pk_name:
 287             cache_key = self.__get_lookup_cache_key(**{key: value})
 288             local_cache = self._get_local_cache()
 289             if local_cache is not None:
 290                 result = local_cache.get(cache_key)
 291                 if result is not None:
 292                     return result
 293 
 294             retval = cache.get(cache_key, version=self.cache_version)
 295             if retval is None:
 296                 result = self.get(**kwargs)
 297                 # Ensure we're pushing it into the cache
 298                 self.__post_save(instance=result)
 299                 if local_cache is not None:
 300                     local_cache[cache_key] = result
 301                 return result
 302 
 303             # If we didn't look up by pk we need to hit the reffed
 304             # key
 305             if key != pk_name:
 306                 result = self.get_from_cache(**{pk_name: retval})
 307                 if local_cache is not None:
 308                     local_cache[cache_key] = result
 309                 return result
 310 
 311             if not isinstance(retval, self.model):
 312                 if settings.DEBUG:
 313                     raise ValueError("Unexpected value type returned from cache")
 314                 logger.error("Cache response returned invalid value %r", retval)
 315                 return self.get(**kwargs)
 316 
 317             if key == pk_name and int(value) != retval.pk:
 318                 if settings.DEBUG:
 319                     raise ValueError("Unexpected value returned from cache")
 320                 logger.error("Cache response returned invalid value %r", retval)
 321                 return self.get(**kwargs)
 322 
 323             retval._state.db = router.db_for_read(self.model, **kwargs)
 324 
 325             return retval
 326         else:
 327             raise ValueError("We cannot cache this query. Just hit the database.")

 329     def get_many_from_cache(self, values, key="pk"):
 330         """
 331         Wrapper around `QuerySet.filter(pk__in=values)` which supports caching of
 332         the intermediate value.  Callee is responsible for making sure the
 333         cache key is cleared on save.
 334 
 335         NOTE: We can only query by primary key or some other unique identifier.
 336         It is not possible to e.g. run `Project.objects.get_many_from_cache([1,
 337         2, 3], key="organization_id")` and get back all projects belonging to
 338         those orgs. The length of the return value is bounded by the length of
 339         `values`.
 340 
 341         For most models, if one attempts to use a non-PK value this will just
 342         degrade to a DB query, like with `get_from_cache`.
 343         """
 344 
 345         pk_name = self.model._meta.pk.name
 346 
 347         if key == "pk":
 348             key = pk_name
 349 
 350         # Kill __exact since it's the default behavior
 351         if key.endswith("__exact"):
 352             key = key.split("__exact", 1)[0]
 353 
 354         if key not in self.cache_fields and key != pk_name:
 355             raise ValueError("We cannot cache this query. Just hit the database.")
 356 
 357         final_results = []
 358         cache_lookup_cache_keys = []
 359         cache_lookup_values = []
 360 
 361         local_cache = self._get_local_cache()
 362         for value in values:
 363             cache_key = self.__get_lookup_cache_key(**{key: value})
 364             result = local_cache and local_cache.get(cache_key)
 365             if result is not None:
 366                 final_results.append(result)
 367             else:
 368                 cache_lookup_cache_keys.append(cache_key)
 369                 cache_lookup_values.append(value)
 370 
 371         if not cache_lookup_cache_keys:
 372             return final_results
 373 
 374         cache_results = cache.get_many(cache_lookup_cache_keys, version=self.cache_version)
 375 
 376         db_lookup_cache_keys = []
 377         db_lookup_values = []
 378 
 379         nested_lookup_cache_keys = []
 380         nested_lookup_values = []
 381 
 382         for cache_key, value in zip(cache_lookup_cache_keys, cache_lookup_values):
 383             cache_result = cache_results.get(cache_key)
 384             if cache_result is None:
 385                 db_lookup_cache_keys.append(cache_key)
 386                 db_lookup_values.append(value)
 387                 continue
 388 
 389             # If we didn't look up by pk we need to hit the reffed key
 390             if key != pk_name:
 391                 nested_lookup_cache_keys.append(cache_key)
 392                 nested_lookup_values.append(cache_result)
 393                 continue
 394 
 395             if not isinstance(cache_result, self.model):
 396                 if settings.DEBUG:
 397                     raise ValueError("Unexpected value type returned from cache")
 398                 logger.error("Cache response returned invalid value %r", cache_result)
 399                 db_lookup_cache_keys.append(cache_key)
 400                 db_lookup_values.append(value)
 401                 continue
 402 
 403             if key == pk_name and int(value) != cache_result.pk:
 404                 if settings.DEBUG:
 405                     raise ValueError("Unexpected value returned from cache")
 406                 logger.error("Cache response returned invalid value %r", cache_result)
 407                 db_lookup_cache_keys.append(cache_key)
 408                 db_lookup_values.append(value)
 409                 continue
 410 
 411             final_results.append(cache_result)
 412 
 413         if nested_lookup_values:
 414             nested_results = self.get_many_from_cache(nested_lookup_values, key=pk_name)
 415             final_results.extend(nested_results)
 416             if local_cache is not None:
 417                 for nested_result in nested_results:
 418                     value = getattr(nested_result, key)
 419                     cache_key = self.__get_lookup_cache_key(**{key: value})
 420                     local_cache[cache_key] = nested_result
 421 
 422         if not db_lookup_values:
 423             return final_results
 424 
 425         cache_writes = []
 426 
 427         db_results = {getattr(x, key): x for x in self.filter(**{key + "__in": db_lookup_values})}
 428         for cache_key, value in zip(db_lookup_cache_keys, db_lookup_values):
 429             db_result = db_results.get(value)
 430             if db_result is None:
 431                 continue  # This model ultimately does not exist
 432 
 433             # Ensure we're pushing it into the cache
 434             cache_writes.append(db_result)
 435             if local_cache is not None:
 436                 local_cache[cache_key] = db_result
 437 
 438             final_results.append(db_result)
 439 
 440         # XXX: Should use set_many here, but __post_save code is too complex
 441         for instance in cache_writes:
 442             self.__post_save(instance=instance)
 443 
 444         return final_results

 449     def uncache_object(self, instance_id):
 450         pk_name = self.model._meta.pk.name
 451         cache_key = self.__get_lookup_cache_key(**{pk_name: instance_id})
 452         cache.delete(cache_key, version=self.cache_version)
