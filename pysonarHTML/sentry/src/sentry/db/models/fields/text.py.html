   1 """
   2 Traditional Django model fields, except their type is
   3 explicitly `text` as opposed to `varchar`. All other
   4 validations and behaviors are identical.
   5 
   6 The reason this exists is because in Postgres, `text`
   7 and `varchar` are nearly identical. The issue is when
   8 picking a CharField and needing to size up the length
   9 needing a database migration. Typically the database
  10 validation isn't strictly required, so we can use
  11 `text` type instead to only enforce the length in
  12 python to allow us to size up the lengths without any
  13 migrations.
  14 """
  15 
  16 from __future__ import absolute_import
  17 
  18 from django.db import models
  19 
  20 __all__ = ("CharField", "EmailField")
  21 
  22 
  23 class TextType(object):
  24     def db_type(self, connection):
  25         return "text"
  26 
  27 
  28 class CharField(TextType, models.CharField):
  29     pass

  31
