  19 class EncryptedCharField(CharField):
  20     def contribute_to_class(self, cls, name):
  21         """
  22         Add a descriptor for backwards compatibility
  23         with previous Django behavior.
  24         """
  25         super(EncryptedCharField, self).contribute_to_class(cls, name)
  26         setattr(cls, name, Creator(self))

  49 class EncryptedPickledObjectField(PickledObjectField):
  50     def get_db_prep_value(self, value, *args, **kwargs):
  51         if isinstance(value, six.binary_type):
  52             value = value.decode("utf-8")
  53         value = super(EncryptedPickledObjectField, self).get_db_prep_value(value, *args, **kwargs)
  54         return encrypt(value)
  55 
  56     def to_python(self, value):
  57         if value is not None and isinstance(value, six.string_types):
  58             value = decrypt(value)
  59 
  60         # The following below is a copypaste of PickledObjectField.to_python of
  61         # v1.0.0 with one change: We re-raise any baseexceptions such as
  62         # signals. 1.0.0 has a bare `except:` which causes issues.
  63 
  64         if value is not None:
  65             try:
  66                 value = dbsafe_decode(value, self.compress)
  67             except Exception:
  68                 # If the value is a definite pickle; and an error is raised in
  69                 # de-pickling it should be allowed to propogate.
  70                 if isinstance(value, PickledObject):
  71                     raise
  72             else:
  73                 if isinstance(value, _ObjectWrapper):
  74                     return value._obj
  75 
  76         return value
