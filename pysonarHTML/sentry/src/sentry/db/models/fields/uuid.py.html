  16 class UUIDField(models.Field):
  17     """Field for storing UUIDs."""
  18 
  19     description = "Universally unique identifier."
  20 
  21     def __init__(self, auto_add=False, coerce_to=UUID, **kwargs):
  22         """Instantiate the field."""
  23 
  24         # If the `auto_add` argument is specified as True, substitute an
  25         # appropriate callable which requires no arguments and will return
  26         # a UUID.
  27         if auto_add is True:
  28             auto_add = uuid4
  29 
  30         # If the `auto_add` arguments is specified as a string
  31         # parse out and import the callable.
  32         if isinstance(auto_add, six.text_type):
  33             module_name, member = auto_add.split(":")
  34             module = importlib.import_module(module_name)
  35             auto_add = getattr(module, member)
  36 
  37         # Save the `auto_add` and `coerce_to` rules.
  38         self._auto_add = auto_add
  39         self._coerce_to = coerce_to
  40 
  41         # If `auto_add` is enabled, it should imply that the field
  42         # is not editable, and should not show up in ModelForms.
  43         if auto_add and "editable" not in kwargs:
  44             kwargs["editable"] = False
  45 
  46         # Blank values shall be nulls.
  47         if kwargs.get("blank", False) and not kwargs.get("null", False):
  48             raise AttributeError(
  49                 " ".join(
  50                     (
  51                         "Blank UUIDs are stored as NULL. Therefore, setting",
  52                         "`blank` to True requires `null` to be True.",
  53                     )
  54                 )
  55             )
  56 
  57         # Enforce CHAR(32) for unsupported engines
  58         kwargs["max_length"] = 32
  59 
  60         # Now pass the rest of the work to CharField.
  61         super(UUIDField, self).__init__(**kwargs)
  62 
  63     def db_type(self, connection):
  64         return "uuid"
  65 
  66     def get_internal_type(self):
  67         return "CharField"

  69     def get_prep_value(self, value):
  70         """Return a wrapped, valid UUID value."""
  71 
  72         # If the value is None, return None.
  73         if not value:
  74             if self.null or self._auto_add or (self.default != NOT_PROVIDED):
  75                 return None
  76             raise ValueError(
  77                 "Explicit UUID required unless either `null` is " "True or `auto_add` is given."
  78             )
  79 
  80         # If we already have a UUID, pass it through.
  81         if isinstance(value, UUID):
  82             return value
  83 
  84         # Convert our value to a UUID.
  85         return UUID(value)
  86 
  87     def pre_save(self, instance, add):
  88         """If auto is set, generate a UUID at random."""
  89 
  90         # If the `auto_add` option was set, and there is no value
  91         # on the model instance, then generate a UUID using the given
  92         # callable.
  93         if self._auto_add and add and not getattr(instance, self.attname):
  94             uuid_value = self._auto_add()
  95 
  96             # Save the UUID to the model instance
  97             setattr(instance, self.attname, uuid_value)
  98             return uuid_value
  99 
 100         # This is the standard case; just use the superclass logic.
 101         return super(UUIDField, self).pre_save(instance, add)
 102 
 103     def contribute_to_class(self, cls, name):
 104         super(UUIDField, self).contribute_to_class(cls, name)
 105         setattr(cls, name, Creator(self))

 107     def to_python(self, value):
 108         """Return a UUID object."""
 109         if isinstance(value, self._coerce_to) or not value:
 110             return value
 111         return self._coerce_to(value)
 112 
 113     @property
 114     def _auto_add_str(self):
 115         """Return a dot path, as a string, of the `_auto_add` callable.
 116         If `_auto_add` is a boolean, return it unchanged.
 117         """
 118         if isinstance(self._auto_add, bool):
 119             return self._auto_add
 120         return "%s:%s" % (self._auto_add.__module__, self._auto_add.__name__)

 123 class UUIDAdapter(object):
 124     def __init__(self, value):
 125         if not isinstance(value, UUID):
 126             raise TypeError("UUIDAdapter only understands UUID objects.")
 127         self.value = value
 128 
 129     def getquoted(self):
 130         return ("'%s'" % self.value).encode("utf8")
