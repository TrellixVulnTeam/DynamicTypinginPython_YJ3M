  34 def create_citext_extension(using, **kwargs):
  35     # We always need the citext extension installed for Postgres,
  36     # and for tests, it's not always guaranteed that we will have
  37     # run full migrations which installed it.
  38     cursor = connections[using].cursor()
  39     try:
  40         cursor.execute("CREATE EXTENSION IF NOT EXISTS citext")
  41     except Exception:
  42         pass
