  29 class NodeData(collections.MutableMapping):
  30     """
  31         A wrapper for nodestore data that fetches the underlying data
  32         from nodestore.
  33 
  34         Initializing with:
  35         data=None means, this is a node that needs to be fetched from nodestore.
  36         data={...} means, this is an object that should be saved to nodestore.
  37     """
  38 
  39     def __init__(self, id, data=None, wrapper=None, ref_version=None, ref_func=None):
  40         self.id = id
  41         self.ref = None
  42         # ref version is used to discredit a previous ref
  43         # (this does not mean the Event is mutable, it just removes ref checking
  44         #  in the case of something changing on the data model)
  45         self.ref_version = ref_version
  46         self.ref_func = ref_func
  47         self.wrapper = wrapper
  48         if data is not None and self.wrapper is not None:
  49             data = self.wrapper(data)
  50         self._node_data = data
  51 
  52     def __getstate__(self):
  53         data = dict(self.__dict__)
  54         # downgrade this into a normal dict in case it's a shim dict.
  55         # This is needed as older workers might not know about newer
  56         # collection types.  For instance we have events where this is a
  57         # CanonicalKeyDict
  58         data.pop("data", None)
  59         data["_node_data_CANONICAL"] = isinstance(data["_node_data"], CANONICAL_TYPES)
  60         data["_node_data"] = dict(data["_node_data"].items())
  61         return data
  62 
  63     def __setstate__(self, state):
  64         # If there is a legacy pickled version that used to have data as a
  65         # duplicate, reject it.
  66         state.pop("data", None)
  67         if state.pop("_node_data_CANONICAL", False):
  68             state["_node_data"] = CanonicalKeyDict(state["_node_data"])
  69         self.__dict__ = state
  70 
  71     def __getitem__(self, key):
  72         return self.data[key]
  73 
  74     def __setitem__(self, key, value):
  75         self.data[key] = value
  76 
  77     def __delitem__(self, key):
  78         del self.data[key]
  79 
  80     def __iter__(self):
  81         return iter(self.data)
  82 
  83     def __len__(self):
  84         return len(self.data)
  85 
  86     def __repr__(self):
  87         cls_name = type(self).__name__
  88         if self._node_data:
  89             return "<%s: id=%s data=%r>" % (cls_name, self.id, repr(self._node_data))
  90         return "<%s: id=%s>" % (cls_name, self.id)
  91 
  92     def get_ref(self, instance):
  93         if not self.ref_func:
  94             return
  95         return self.ref_func(instance)
  96 
  97     def copy(self):
  98         return self.data.copy()
  99 
 100     @memoize
 101     def data(self):
 102         """
 103         Get the current data object, fetching from nodestore if necessary.
 104         """
 105 
 106         if self._node_data is not None:
 107             return self._node_data
 108 
 109         elif self.id:
 110             self.bind_data(nodestore.get(self.id) or {})
 111             return self._node_data
 112 
 113         rv = {}
 114         if self.wrapper is not None:
 115             rv = self.wrapper(rv)
 116         return rv
 117 
 118     def bind_data(self, data, ref=None):
 119         self.ref = data.pop("_ref", ref)
 120         ref_version = data.pop("_ref_version", None)
 121         if ref_version == self.ref_version and ref is not None and self.ref != ref:
 122             raise NodeIntegrityFailure(
 123                 "Node reference for %s is invalid: %s != %s" % (self.id, ref, self.ref)
 124             )
 125         if self.wrapper is not None:
 126             data = self.wrapper(data)
 127         self._node_data = data
 128 
 129     def bind_ref(self, instance):
 130         ref = self.get_ref(instance)
 131         if ref:
 132             self.data["_ref"] = ref
 133             self.data["_ref_version"] = self.ref_version
 134 
 135     def save(self):
 136         """
 137         Write current data back to nodestore.
 138         """
 139 
 140         # We never loaded any data for reading or writing, so there
 141         # is nothing to save.
 142         if self._node_data is None:
 143             return
 144 
 145         # We can't put our wrappers into the nodestore, so we need to
 146         # ensure that the data is converted into a plain old dict
 147         to_write = self._node_data
 148         if isinstance(to_write, CANONICAL_TYPES):
 149             to_write = dict(to_write.items())
 150 
 151         nodestore.set(self.id, to_write)

 154 class NodeField(GzippedDictField):
 155     """
 156     Similar to the gzippedictfield except that it stores a reference
 157     to an external node.
 158     """
 159 
 160     def __init__(self, *args, **kwargs):
 161         self.ref_func = kwargs.pop("ref_func", None)
 162         self.ref_version = kwargs.pop("ref_version", None)
 163         self.wrapper = kwargs.pop("wrapper", None)
 164         self.id_func = kwargs.pop("id_func", lambda: b64encode(uuid4().bytes))
 165         super(NodeField, self).__init__(*args, **kwargs)
 166 
 167     def contribute_to_class(self, cls, name):
 168         super(NodeField, self).contribute_to_class(cls, name)
 169         setattr(cls, name, Creator(self))
 170         post_delete.connect(self.on_delete, sender=self.model, weak=False)
 171 
 172     def on_delete(self, instance, **kwargs):
 173         value = getattr(instance, self.name)
 174         if not value.id:
 175             return
 176 
 177         nodestore.delete(value.id)
 178 
 179     def to_python(self, value):
 180         node_id = None
 181         # If value is a string, we assume this is a value we've loaded from the
 182         # database, it should be decompressed/unpickled, and we should end up
 183         # with a dict.
 184         if value and isinstance(value, six.string_types):
 185             try:
 186                 value = pickle.loads(decompress(value))
 187             except Exception as e:
 188                 # TODO this is a bit dangerous as a failure to read/decode the
 189                 # node_id will end up with this record being replaced with an
 190                 # empty value under a new key, potentially orphaning an
 191                 # original value in nodestore. OTOH if we can't decode the info
 192                 # here, the node was already effectively orphaned.
 193                 logger.exception(e)
 194                 value = None
 195 
 196         if value:
 197             if "node_id" in value:
 198                 node_id = value.pop("node_id")
 199                 # If the value is now empty, that means that it only had the
 200                 # node_id in it, which means that we should be looking to *load*
 201                 # the event body from nodestore. If it does have other stuff in
 202                 # it, that means we got an event body with a precomputed id in
 203                 # it, and we want to *save* the rest of the body to nodestore.
 204                 if value == {}:
 205                     value = None
 206         else:
 207             # Either we were passed a null/empty value in the constructor, or
 208             # we failed to decode the value from the database so we have no id
 209             # to load data from, and no data to save.
 210             value = None
 211 
 212         return NodeData(
 213             node_id,
 214             value,
 215             wrapper=self.wrapper,
 216             ref_version=self.ref_version,
 217             ref_func=self.ref_func,
 218         )
 219 
 220     def get_prep_value(self, value):
 221         """
 222             Prepares the NodeData to be written in a Model.save() call.
 223 
 224             Makes sure the event body is written to nodestore and
 225             returns the node_id reference to be written to rowstore.
 226         """
 227         if not value and self.null:
 228             # save ourselves some storage
 229             return None
 230 
 231         if value.id is None:
 232             value.id = self.id_func()
 233 
 234         value.save()
 235         return compress(pickle.dumps({"node_id": value.id}))
