  13     def __init__(self, model, project_id=None, dtfield=None, days=None, order_by=None):
  14         self.model = model
  15         self.project_id = int(project_id) if project_id else None
  16         self.dtfield = dtfield
  17         self.days = int(days) if days is not None else None
  18         self.order_by = order_by
  19         self.using = router.db_for_write(model)

  21     def execute(self, chunk_size=10000):
  22         quote_name = connections[self.using].ops.quote_name
  23 
  24         where = []
  25         if self.dtfield and self.days is not None:
  26             where.append(
  27                 u"{} < '{}'::timestamptz".format(
  28                     quote_name(self.dtfield),
  29                     (timezone.now() - timedelta(days=self.days)).isoformat(),
  30                     self.days,
  31                 )
  32             )
  33         if self.project_id:
  34             where.append(u"project_id = {}".format(self.project_id))
  35 
  36         if where:
  37             where_clause = u"where {}".format(" and ".join(where))
  38         else:
  39             where_clause = ""
  40 
  41         if self.order_by:
  42             if self.order_by[0] == "-":
  43                 direction = "desc"
  44                 order_field = self.order_by[1:]
  45             else:
  46                 direction = "asc"
  47                 order_field = self.order_by
  48             order_clause = u"order by {} {}".format(quote_name(order_field), direction)
  49         else:
  50             order_clause = ""
  51 
  52         query = u"""
  53             delete from {table}
  54             where id = any(array(
  55                 select id
  56                 from {table}
  57                 {where}
  58                 {order}
  59                 limit {chunk_size}
  60             ));
  61         """.format(
  62             table=self.model._meta.db_table,
  63             chunk_size=chunk_size,
  64             where=where_clause,
  65             order=order_clause,
  66         )
  67 
  68         return self._continuous_query(query)

  70     def _continuous_query(self, query):
  71         results = True
  72         cursor = connections[self.using].cursor()
  73         while results:
  74             cursor.execute(query)
  75             results = cursor.rowcount > 0

  77     def iterator(self, chunk_size=100, batch_size=100000):
  78         assert self.days is not None
  79         assert self.dtfield is not None and self.dtfield == self.order_by
  80 
  81         dbc = connections[self.using]
  82         quote_name = dbc.ops.quote_name
  83 
  84         position = None
  85         cutoff = timezone.now() - timedelta(days=self.days)
  86 
  87         with dbc.get_new_connection(dbc.get_connection_params()) as conn:
  88             conn.autocommit = False
  89 
  90             chunk = []
  91 
  92             completed = False
  93             while not completed:
  94                 # We explicitly use a named cursor here so that we can read a
  95                 # large quantity of rows from postgres incrementally, without
  96                 # having to pull all rows into memory at once.
  97                 with conn.cursor(uuid4().hex) as cursor:
  98                     where = [(u"{} < %s".format(quote_name(self.dtfield)), [cutoff])]
  99 
 100                     if self.project_id:
 101                         where.append(("project_id = %s", [self.project_id]))
 102 
 103                     if self.order_by[0] == "-":
 104                         direction = "desc"
 105                         order_field = self.order_by[1:]
 106                         if position is not None:
 107                             where.append((u"{} <= %s".format(quote_name(order_field)), [position]))
 108                     else:
 109                         direction = "asc"
 110                         order_field = self.order_by
 111                         if position is not None:
 112                             where.append((u"{} >= %s".format(quote_name(order_field)), [position]))
 113 
 114                     conditions, parameters = zip(*where)
 115                     parameters = list(itertools.chain.from_iterable(parameters))
 116 
 117                     query = u"""
 118                         select id, {order_field}
 119                         from {table}
 120                         where {conditions}
 121                         order by {order_field} {direction}
 122                         limit {batch_size}
 123                     """.format(
 124                         table=self.model._meta.db_table,
 125                         conditions=" and ".join(conditions),
 126                         order_field=quote_name(order_field),
 127                         direction=direction,
 128                         batch_size=batch_size,
 129                     )
 130 
 131                     cursor.execute(query, parameters)
 132 
 133                     i = 0
 134                     for i, row in enumerate(cursor, 1):
 135                         key, position = row
 136                         chunk.append(key)
 137                         if len(chunk) == chunk_size:
 138                             yield tuple(chunk)
 139                             chunk = []
 140 
 141                     # If we retrieved less rows than the batch size, there are
 142                     # no more rows remaining to delete and we can exit the
 143                     # loop.
 144                     if i < batch_size:
 145                         completed = True
 146 
 147                 conn.commit()
 148 
 149             if chunk:
 150                 yield tuple(chunk)
