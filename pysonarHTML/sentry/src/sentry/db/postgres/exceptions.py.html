  10 class CompositeTraceback(object):
  11     def __init__(self, tb_list):
  12         assert isinstance(tb_list, (list, tuple))
  13         self.__tb_list = tb_list
  14         self.__iterator = iter(self)

  16     def __iter__(self):
  17         for tb in self.__tb_list:
  18             while tb:
  19                 self.__curframe = tb
  20                 tb = tb.tb_next
  21                 yield tb

  23     def tb_frame(self):
