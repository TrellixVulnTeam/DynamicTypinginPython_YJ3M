  23 def remove_null(value):
  24     # In psycopg2 2.7+, behavior was introduced where a
  25     # NULL byte in a parameter would start raising a ValueError.
  26     # psycopg2 chose to do this rather than let Postgres silently
  27     # truncate the data, which is it's behavior when it sees a
  28     # NULL byte. But for us, we'd rather remove the null value so it's
  29     # somewhat legible rather than error. Considering this is better
  30     # behavior than the database truncating, seems good to do this
  31     # rather than attempting to sanitize all data inputs now manually.
  32     return value.replace("\x00", "")

  35 def remove_surrogates(value):
  36     # Another hack.  postgres does not accept lone surrogates
  37     # in utf-8 mode.  If we encounter any lone surrogates in
  38     # our string we need to remove it.
  39     if type(value) is bytes:
  40         try:
  41             return strip_lone_surrogates(value.decode("utf-8")).encode("utf-8")
  42         except UnicodeError:
  43             return value
  44     return strip_lone_surrogates(value)

  47 def clean_bad_params(params):
  48     params = list(params)
  49     for idx, param in enumerate(params):
  50         if isinstance(param, string_types):
  51             params[idx] = remove_null(remove_surrogates(param))
  52     return params

  55 class CursorWrapper(object):
  56     """
  57     A wrapper around the postgresql_psycopg2 backend which handles various events
  58     from cursors, such as auto reconnects and lazy time zone evaluation.
  59     """
  60 
  61     def __init__(self, db, cursor):
  62         self.db = db
  63         self.cursor = cursor
  64 
  65     def __getattr__(self, attr):
  66         return getattr(self.cursor, attr)
  67 
  68     def __iter__(self):
  69         return iter(self.cursor)
  70 
  71     @capture_transaction_exceptions
  72     @auto_reconnect_cursor
  73     @less_shitty_error_messages
  74     def execute(self, sql, params=None):
  75         if params is not None:
  76             return self.cursor.execute(sql, clean_bad_params(params))
  77         return self.cursor.execute(sql)
  78 
  79     @capture_transaction_exceptions
  80     @auto_reconnect_cursor
  81     @less_shitty_error_messages
  82     def executemany(self, sql, paramlist=()):
  83         return self.cursor.executemany(sql, paramlist)

  86 class DatabaseWrapper(DatabaseWrapper):
  87     def __init__(self, *args, **kwargs):
  88         super(DatabaseWrapper, self).__init__(*args, **kwargs)
  89         self.ops = DatabaseOperations(self)
  90 
  91     @auto_reconnect_connection
  92     def _set_isolation_level(self, level):
  93         return super(DatabaseWrapper, self)._set_isolation_level(level)
  94 
  95     @auto_reconnect_connection
  96     def _cursor(self, *args, **kwargs):
  97         return super(DatabaseWrapper, self)._cursor()
  98 
  99     # We're overriding this internal method that's present in Django 1.11+, because
 100     # things were shuffled around since 1.10 resulting in not constructing a django CursorWrapper
 101     # with our CursorWrapper. We need to be passing our wrapped cursor to their wrapped cursor,
 102     # not the other way around since then we'll lose things like __enter__ due to the way this
 103     # wrapper is working (getattr on self.cursor).
 104     def _prepare_cursor(self, cursor):
 105         cursor = super(DatabaseWrapper, self)._prepare_cursor(CursorWrapper(self, cursor))
 106         return cursor

 108     def close(self, reconnect=False):
 109         """
 110         This ensures we dont error if the connection has already been closed.
 111         """
 112         if self.connection is not None:
 113             if not self.connection.closed:
 114                 try:
 115                     self.connection.close()
 116                 except Database.InterfaceError:
 117                     # connection was already closed by something
 118                     # like pgbouncer idle timeout.
 119                     pass
 120             self.connection = None
