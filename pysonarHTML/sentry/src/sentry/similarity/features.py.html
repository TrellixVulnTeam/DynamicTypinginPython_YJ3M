  14 def get_application_chunks(exception):
  15     """\
  16     Filters out system and framework frames from a stacktrace in order to
  17     better align similar logical application paths. This returns a sequence of
  18     application code "chunks": blocks of contiguously called application code.
  19     """
  20     return map(
  21         lambda in_app__frames: list(in_app__frames[1]),
  22         itertools.ifilter(
  23             lambda in_app__frames: in_app__frames[0],
  24             itertools.groupby(exception.stacktrace.frames, key=lambda frame: frame.in_app),
  25         ),
  26     )

  57 class FeatureSet(object):
  58     def __init__(
  59         self,
  60         index,
  61         encoder,
  62         aliases,
  63         features,
  64         expected_extraction_errors,
  65         expected_encoding_errors,
  66     ):
  67         self.index = index
  68         self.encoder = encoder
  69         self.aliases = aliases
  70         self.features = features
  71         self.expected_extraction_errors = expected_extraction_errors
  72         self.expected_encoding_errors = expected_encoding_errors
  73         assert set(self.aliases) == set(self.features)
  74 
  75     def __get_scope(self, project):
  76         return u"{}".format(project.id)
  77 
  78     def __get_key(self, group):
  79         return u"{}".format(group.id)
  80 
  81     def extract(self, event):
  82         results = {}
  83         for label, strategy in self.features.items():
  84             try:
  85                 results[label] = strategy.extract(event)
  86             except Exception as error:
  87                 log = (
  88                     logger.debug
  89                     if isinstance(error, self.expected_extraction_errors)
  90                     else functools.partial(logger.warning, exc_info=True)
  91                 )
  92                 log(
  93                     "Could not extract features from %r for %r due to error: %r",
  94                     event,
  95                     label,
  96                     error,
  97                     exc_info=True,
  98                 )
  99         return results

 101     def record(self, events):
 102         if not events:
 103             return []
 104 
 105         scope = None
 106         key = None
 107 
 108         items = []
 109         for event in events:
 110             if not event.group_id:
 111                 continue
 112             for label, features in self.extract(event).items():
 113                 if scope is None:
 114                     scope = self.__get_scope(event.project)
 115                 else:
 116                     assert (
 117                         self.__get_scope(event.project) == scope
 118                     ), "all events must be associated with the same project"
 119 
 120                 if key is None:
 121                     key = self.__get_key(event.group)
 122                 else:
 123                     assert (
 124                         self.__get_key(event.group) == key
 125                     ), "all events must be associated with the same group"
 126 
 127                 try:
 128                     features = map(self.encoder.dumps, features)
 129                 except Exception as error:
 130                     log = (
 131                         logger.debug
 132                         if isinstance(error, self.expected_encoding_errors)
 133                         else functools.partial(logger.warning, exc_info=True)
 134                     )
 135                     log(
 136                         "Could not encode features from %r for %r due to error: %r",
 137                         event,
 138                         label,
 139                         error,
 140                     )
 141                 else:
 142                     if features:
 143                         items.append((self.aliases[label], features))
 144 
 145         return self.index.record(scope, key, items, timestamp=int(to_timestamp(event.datetime)))

 147     def classify(self, events, limit=None, thresholds=None):
 148         if not events:
 149             return []
 150 
 151         if thresholds is None:
 152             thresholds = {}
 153 
 154         scope = None
 155 
 156         labels = []
 157         items = []
 158         for event in events:
 159             for label, features in self.extract(event).items():
 160                 if scope is None:
 161                     scope = self.__get_scope(event.project)
 162                 else:
 163                     assert (
 164                         self.__get_scope(event.project) == scope
 165                     ), "all events must be associated with the same project"
 166 
 167                 try:
 168                     features = map(self.encoder.dumps, features)
 169                 except Exception as error:
 170                     log = (
 171                         logger.debug
 172                         if isinstance(error, self.expected_encoding_errors)
 173                         else functools.partial(logger.warning, exc_info=True)
 174                     )
 175                     log(
 176                         "Could not encode features from %r for %r due to error: %r",
 177                         event,
 178                         label,
 179                         error,
 180                     )
 181                 else:
 182                     if features:
 183                         items.append((self.aliases[label], thresholds.get(label, 0), features))
 184                         labels.append(label)
 185 
 186         return map(
 187             lambda key__scores: (int(key__scores[0]), dict(zip(labels, key__scores[1]))),
 188             self.index.classify(
 189                 scope, items, limit=limit, timestamp=int(to_timestamp(event.datetime))
 190             ),
 191         )

 193     def compare(self, group, limit=None, thresholds=None):
 194         if thresholds is None:
 195             thresholds = {}
 196 
 197         features = list(self.features.keys())
 198 
 199         items = [(self.aliases[label], thresholds.get(label, 0)) for label in features]
 200 
 201         return map(
 202             lambda key__scores: (int(key__scores[0]), dict(zip(features, key__scores[1]))),
 203             self.index.compare(
 204                 self.__get_scope(group.project), self.__get_key(group), items, limit=limit
 205             ),
 206         )

 208     def merge(self, destination, sources, allow_unsafe=False):
 209         def add_index_aliases_to_key(key):
 210             return [(self.aliases[label], key) for label in self.features.keys()]
 211 
 212         # Collect all of the sources by the scope that they are contained
 213         # within so that we can make the most efficient queries possible and
 214         # reject queries that cross scopes if we haven't explicitly allowed
 215         # unsafe actions.
 216         scopes = {}
 217         for source in sources:
 218             scopes.setdefault(self.__get_scope(source.project), set()).add(source)
 219 
 220         unsafe_scopes = set(scopes.keys()) - set([self.__get_scope(destination.project)])
 221         if unsafe_scopes and not allow_unsafe:
 222             raise ValueError(
 223                 "all groups must belong to same project if unsafe merges are not allowed"
 224             )
 225 
 226         destination_scope = self.__get_scope(destination.project)
 227         destination_key = self.__get_key(destination)
 228 
 229         for source_scope, sources in scopes.items():
 230             items = []
 231             for source in sources:
 232                 items.extend(add_index_aliases_to_key(self.__get_key(source)))
 233 
 234             if source_scope != destination_scope:
 235                 imports = [
 236                     (alias, destination_key, data)
 237                     for (alias, _), data in zip(items, self.index.export(source_scope, items))
 238                 ]
 239                 self.index.delete(source_scope, items)
 240                 self.index.import_(destination_scope, imports)
 241             else:
 242                 self.index.merge(destination_scope, destination_key, items)
 243 
 244     def delete(self, group):
 245         key = self.__get_key(group)
 246         return self.index.delete(
 247             self.__get_scope(group.project),
 248             [(self.aliases[label], key) for label in self.features.keys()],
 249         )
 250 
 251     def flush(self, project):
 252         return self.index.flush(self.__get_scope(project), self.aliases.values())
