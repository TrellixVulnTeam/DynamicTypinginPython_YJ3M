  25 class RedisScriptMinHashIndexBackend(AbstractIndexBackend):
  26     def __init__(
  27         self, cluster, namespace, signature_builder, bands, interval, retention, candidate_set_limit
  28     ):
  29         self.cluster = cluster
  30         self.namespace = namespace
  31         self.signature_builder = signature_builder
  32         self.bands = bands
  33         self.interval = interval
  34         self.retention = retention
  35         self.candidate_set_limit = candidate_set_limit

  37     def _build_signature_arguments(self, features):
  38         if not features:
  39             return [0] * self.bands
  40 
  41         arguments = []
  42         for bucket in band(self.bands, self.signature_builder(features)):
  43             arguments.extend([1, ",".join(map("{}".format, bucket)), 1])
  44         return arguments
  45 
  46     def __index(self, scope, args):
  47         # scope must be passed into the script call as a key to allow the
  48         # cluster client to determine what cluster the script should be
  49         # executed on. The script itself will use the scope as the hashtag for
  50         # all redis operations.
  51         return index(self.cluster, [scope], args)

  53     def _as_search_result(self, results):
  54         score_replacements = {
  55             -1.0: None,  # both items don't have the feature (no comparison)
  56             -2.0: 0,  # one item doesn't have the feature (totally dissimilar)
  57         }
  58 
  59         def decode_search_result(result):
  60             key, scores = result
  61             return (
  62                 key,
  63                 map(lambda score: score_replacements.get(score, score), map(float, scores)),
  64             )
  65 
  66         def get_comparison_key(result):
  67             key, scores = result
  68 
  69             scores = [score for score in scores if score is not None]
  70 
  71             return (
  72                 sum(scores) / len(scores) * -1,  # average score, descending
  73                 len(scores) * -1,  # number of indexes with scores, descending
  74                 key,  # lexicographical sort on key, ascending
  75             )
  76 
  77         return sorted(map(decode_search_result, results), key=get_comparison_key)

  79     def classify(self, scope, items, limit=None, timestamp=None):
  80         if timestamp is None:
  81             timestamp = int(time.time())
  82 
  83         arguments = [
  84             "CLASSIFY",
  85             timestamp,
  86             self.namespace,
  87             self.bands,
  88             self.interval,
  89             self.retention,
  90             self.candidate_set_limit,
  91             scope,
  92             limit if limit is not None else -1,
  93         ]
  94 
  95         for idx, threshold, features in items:
  96             arguments.extend([idx, threshold])
  97             arguments.extend(self._build_signature_arguments(features))
  98 
  99         return self._as_search_result(self.__index(scope, arguments))

 189     def scan(self, scope, indices, batch=1000, timestamp=None):
 190         if timestamp is None:
 191             timestamp = int(time.time())
 192 
 193         arguments = [
 194             "SCAN",
 195             timestamp,
 196             self.namespace,
 197             self.bands,
 198             self.interval,
 199             self.retention,
 200             self.candidate_set_limit,
 201             scope,
 202         ]
 203 
 204         cursors = {idx: 0 for idx in indices}
 205         while cursors:
 206             requests = []
 207             for idx, cursor in cursors.items():
 208                 requests.append([idx, cursor, batch])
 209 
 210             responses = self.__index(scope, arguments + flatten(requests))
 211 
 212             for (idx, _, _), (cursor, chunk) in zip(requests, responses):
 213                 cursor = int(cursor)
 214                 if cursor == 0:
 215                     del cursors[idx]
 216                 else:
 217                     cursors[idx] = cursor
 218 
 219                 yield idx, chunk
 220 
 221     def flush(self, scope, indices, batch=1000, timestamp=None):
 222         for index, chunk in self.scan(scope, indices, batch, timestamp):
 223             if chunk:
 224                 self.cluster.delete(*chunk)
