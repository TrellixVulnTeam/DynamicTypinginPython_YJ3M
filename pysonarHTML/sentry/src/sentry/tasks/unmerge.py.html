  33 def cache(function):
  34     results = {}
  35 
  36     def fetch(*key):
  37         value = results.get(key)
  38         if value is None:
  39             try:
  40                 value = results[key] = (True, function(*key))
  41             except Exception as error:
  42                 value = results[key] = (False, error)
  43 
  44         ok, result = value
  45         if ok:
  46             return result
  47         else:
  48             raise result
  49 
  50     return fetch

  74 def merge_mappings(values):
  75     result = {}
  76     for value in values:
  77         result.update(value)
  78     return result

  81 def _generate_culprit(event):
  82     # XXX(mitsuhiko): workaround: some old events do not have this data yet.
  83     # This should be save delete by end of 2019 even considering slow on-prem
  84     # releases.  Platform was added back to data in december 2018.
  85     data = event.data
  86     if data.get("platform") is None:
  87         data = dict(data.items())
  88         data["platform"] = event.platform
  89     return generate_culprit(data)

  92 initial_fields = {
  93     "culprit": lambda event: _generate_culprit(event),
  94     "data": lambda event: {
  95         "last_received": event.data.get("received") or float(event.datetime.strftime("%s")),
  96         "type": event.data["type"],
  97         "metadata": event.data["metadata"],
  98     },
  99     "last_seen": lambda event: event.datetime,
 100     "level": lambda event: LOG_LEVELS_MAP.get(event.get_tag("level"), logging.ERROR),
 101     "message": lambda event: event.search_message,
 102     "times_seen": lambda event: 0,

 106 backfill_fields = {
 107     "platform": lambda caches, data, event: event.platform,
 108     "logger": lambda caches, data, event: event.get_tag("logger") or DEFAULT_LOGGER_NAME,
 109     "first_seen": lambda caches, data, event: event.datetime,
 110     "active_at": lambda caches, data, event: event.datetime,
 111     "first_release": lambda caches, data, event: caches["Release"](

 116     "times_seen": lambda caches, data, event: data["times_seen"] + 1,
 117     "score": lambda caches, data, event: Group.calculate_score(
 118         data["times_seen"] + 1, data["last_seen"]

 123 def get_group_creation_attributes(caches, events):
 124     latest_event = events[0]
 125     return reduce(
 126         lambda data, event: merge_mappings(
 127             [data, {name: f(caches, data, event) for name, f in backfill_fields.items()}]
 128         ),
 129         events,
 130         {name: f(latest_event) for name, f in initial_fields.items()},
 131     )
 132 
 133 
 134 def get_group_backfill_attributes(caches, group, events):
 135     return {
 136         k: v
 137         for k, v in reduce(
 138             lambda data, event: merge_mappings(
 139                 [data, {name: f(caches, data, event) for name, f in backfill_fields.items()}]
 140             ),
 141             events,
 142             {
 143                 name: getattr(group, name)
 144                 for name in set(initial_fields.keys()) | set(backfill_fields.keys())
 145             },
 146         ).items()
 147         if k in backfill_fields
 148     }

 156 def migrate_events(
 157     caches, project, source_id, destination_id, fingerprints, events, actor_id, eventstream_state
 158 ):
 159     # XXX: This is only actually able to create a destination group and migrate
 160     # the group hashes if there are events that can be migrated. How do we
 161     # handle this if there aren't any events? We can't create a group (there
 162     # isn't any data to derive the aggregates from), so we'd have to mark the
 163     # hash as in limbo somehow...?)
 164     if not events:
 165         return (destination_id, eventstream_state)
 166 
 167     if destination_id is None:
 168         # XXX: There is a race condition here between the (wall clock) time
 169         # that the migration is started by the user and when we actually
 170         # get to this block where the new destination is created and we've
 171         # moved the ``GroupHash`` so that events start being associated
 172         # with it. During this gap, there could have been additional events
 173         # ingested, and if we want to handle this, we'd need to record the
 174         # highest event ID we've seen at the beginning of the migration,
 175         # then scan all events greater than that ID and migrate the ones
 176         # where necessary. (This still isn't even guaranteed to catch all
 177         # of the events due to processing latency, but it's a better shot.)
 178         # Create a new destination group.
 179         destination = Group.objects.create(
 180             project_id=project.id,
 181             short_id=project.next_short_id(),
 182             **get_group_creation_attributes(caches, events)
 183         )
 184 
 185         destination_id = destination.id
 186 
 187         eventstream_state = eventstream.start_unmerge(
 188             project.id, fingerprints, source_id, destination_id
 189         )
 190 
 191         # Move the group hashes to the destination.
 192         GroupHash.objects.filter(project_id=project.id, hash__in=fingerprints).update(
 193             group=destination_id
 194         )
 195 
 196         # Create activity records for the source and destination group.
 197         Activity.objects.create(
 198             project_id=project.id,
 199             group_id=destination_id,
 200             type=Activity.UNMERGE_DESTINATION,
 201             user_id=actor_id,
 202             data={"fingerprints": fingerprints, "source_id": source_id},
 203         )
 204 
 205         Activity.objects.create(
 206             project_id=project.id,
 207             group_id=source_id,
 208             type=Activity.UNMERGE_SOURCE,
 209             user_id=actor_id,
 210             data={"fingerprints": fingerprints, "destination_id": destination_id},
 211         )
 212     else:
 213         # Update the existing destination group.
 214         destination = Group.objects.get(id=destination_id)
 215         destination.update(**get_group_backfill_attributes(caches, destination, events))
 216 
 217     event_id_set = set(event.event_id for event in events)
 218 
 219     for event in events:
 220         event.group = destination
 221 
 222     event_id_set = set(event.event_id for event in events)
 223 
 224     UserReport.objects.filter(project_id=project.id, event_id__in=event_id_set).update(
 225         group=destination_id
 226     )
 227     EventAttachment.objects.filter(project_id=project.id, event_id__in=event_id_set).update(
 228         group_id=destination_id
 229     )
 230 
 231     return (destination.id, eventstream_state)

 234 def truncate_denormalizations(group):
 235     GroupRelease.objects.filter(group_id=group.id).delete()
 236 
 237     # XXX: This can cause a race condition with the ``FirstSeenEventCondition``
 238     # where notifications can be erroneously sent if they occur in this group
 239     # before the reprocessing of the denormalizated data completes, since a new
 240     # ``GroupEnvironment`` will be created.
 241     for instance in GroupEnvironment.objects.filter(group_id=group.id):
 242         instance.delete()
 243 
 244     environment_ids = list(
 245         Environment.objects.filter(projects=group.project).values_list("id", flat=True)
 246     )
 247 
 248     tsdb.delete([tsdb.models.group], [group.id], environment_ids=environment_ids)
 249 
 250     tsdb.delete_distinct_counts(
 251         [tsdb.models.users_affected_by_group], [group.id], environment_ids=environment_ids
 252     )
 253 
 254     tsdb.delete_frequencies(
 255         [tsdb.models.frequent_releases_by_group, tsdb.models.frequent_environments_by_group],
 256         [group.id],
 257     )
 258 
 259     features.delete(group)

 262 def collect_group_environment_data(events):
 263     """\
 264     Find the first release for a each group and environment pair from a
 265     date-descending sorted list of events.
 266     """
 267     results = OrderedDict()
 268     for event in events:
 269         results[(event.group_id, get_environment_name(event))] = event.get_tag("sentry:release")
 270     return results

 273 def repair_group_environment_data(caches, project, events):
 274     for (group_id, env_name), first_release in collect_group_environment_data(events).items():
 275         fields = {}
 276         if first_release:
 277             fields["first_release"] = caches["Release"](project.organization_id, first_release)
 278 
 279         GroupEnvironment.objects.create_or_update(
 280             environment_id=caches["Environment"](project.organization_id, env_name).id,
 281             group_id=group_id,
 282             defaults=fields,
 283             values=fields,
 284         )

 287 def collect_tag_data(events):
 288     results = OrderedDict()
 289 
 290     for event in events:
 291         environment = get_environment_name(event)
 292         tags = results.setdefault((event.group_id, environment), {})
 293 
 294         for key, value in event.tags:
 295             values = tags.setdefault(key, {})
 296 
 297             if value in values:
 298                 times_seen, first_seen, last_seen = values[value]
 299                 values[value] = (times_seen + 1, event.datetime, last_seen)
 300             else:
 301                 values[value] = (1, event.datetime, event.datetime)
 302 
 303     return results

 310 def collect_release_data(caches, project, events):
 311     results = OrderedDict()
 312 
 313     for event in events:
 314         release = event.get_tag("sentry:release")
 315 
 316         if not release:
 317             continue
 318 
 319         key = (
 320             event.group_id,
 321             get_environment_name(event),
 322             caches["Release"](project.organization_id, release).id,
 323         )
 324 
 325         if key in results:
 326             first_seen, last_seen = results[key]
 327             results[key] = (event.datetime, last_seen)
 328         else:
 329             results[key] = (event.datetime, event.datetime)
 330 
 331     return results

 334 def repair_group_release_data(caches, project, events):
 335     attributes = collect_release_data(caches, project, events).items()
 336     for (group_id, environment, release_id), (first_seen, last_seen) in attributes:
 337         instance, created = GroupRelease.objects.get_or_create(
 338             project_id=project.id,
 339             group_id=group_id,
 340             environment=environment,
 341             release_id=release_id,
 342             defaults={"first_seen": first_seen, "last_seen": last_seen},
 343         )
 344 
 345         if not created:
 346             instance.update(first_seen=first_seen)

 358 def collect_tsdb_data(caches, project, events):
 359     counters = defaultdict(lambda: defaultdict(lambda: defaultdict(int)))
 360 
 361     sets = defaultdict(lambda: defaultdict(lambda: defaultdict(set)))
 362 
 363     frequencies = defaultdict(lambda: defaultdict(lambda: defaultdict(lambda: defaultdict(int))))
 364 
 365     for event in events:
 366         environment = caches["Environment"](project.organization_id, get_environment_name(event))
 367 
 368         counters[event.datetime][tsdb.models.group][(event.group_id, environment.id)] += 1
 369 
 370         user = event.data.get("user")
 371         if user:
 372             sets[event.datetime][tsdb.models.users_affected_by_group][
 373                 (event.group_id, environment.id)
 374             ].add(get_event_user_from_interface(user).tag_value)
 375 
 376         frequencies[event.datetime][tsdb.models.frequent_environments_by_group][event.group_id][
 377             environment.id
 378         ] += 1
 379 
 380         release = event.get_tag("sentry:release")
 381         if release:
 382             # TODO: I'm also not sure if "environment" here is correct, see
 383             # similar comment above during creation.
 384             grouprelease = caches["GroupRelease"](
 385                 event.group_id,
 386                 get_environment_name(event),
 387                 caches["Release"](project.organization_id, release).id,
 388             )
 389 
 390             frequencies[event.datetime][tsdb.models.frequent_releases_by_group][event.group_id][
 391                 grouprelease.id
 392             ] += 1
 393 
 394     return counters, sets, frequencies
 395

 397 def repair_tsdb_data(caches, project, events):
 398     counters, sets, frequencies = collect_tsdb_data(caches, project, events)
 399 
 400     for timestamp, data in counters.items():
 401         for model, keys in data.items():
 402             for (key, environment_id), value in keys.items():
 403                 tsdb.incr(model, key, timestamp, value, environment_id=environment_id)
 404 
 405     for timestamp, data in sets.items():
 406         for model, keys in data.items():
 407             for (key, environment_id), values in keys.items():
 408                 # TODO: This should use `record_multi` rather than `record`.
 409                 tsdb.record(model, key, values, timestamp, environment_id=environment_id)
 410 
 411     for timestamp, data in frequencies.items():
 412         tsdb.record_frequency_multi(data.items(), timestamp)

 415 def repair_denormalizations(caches, project, events):
 416     repair_group_environment_data(caches, project, events)
 417     repair_group_release_data(caches, project, events)
 418     repair_tsdb_data(caches, project, events)
 419 
 420     for event in events:
 421         features.record([event])

 424 def lock_hashes(project_id, source_id, fingerprints):
 425     with transaction.atomic():
 426         eligible_hashes = list(
 427             GroupHash.objects.filter(
 428                 project_id=project_id, group_id=source_id, hash__in=fingerprints
 429             )
 430             .exclude(state=GroupHash.State.LOCKED_IN_MIGRATION)
 431             .select_for_update()
 432         )
 433 
 434         GroupHash.objects.filter(id__in=[h.id for h in eligible_hashes]).update(
 435             state=GroupHash.State.LOCKED_IN_MIGRATION
 436         )
 437 
 438     return [h.hash for h in eligible_hashes]

 441 def unlock_hashes(project_id, fingerprints):
 442     GroupHash.objects.filter(
 443         project_id=project_id, hash__in=fingerprints, state=GroupHash.State.LOCKED_IN_MIGRATION
 444     ).update(state=GroupHash.State.UNLOCKED)

 447 @instrumented_task(name="sentry.tasks.unmerge", queue="unmerge")
 448 def unmerge(
 449     project_id,
 450     source_id,
 451     destination_id,
 452     fingerprints,
 453     actor_id,
 454     last_event=None,
 455     batch_size=500,
 456     source_fields_reset=False,
 457     eventstream_state=None,
 458 ):
 459     # XXX: The queryset chunking logic below is awfully similar to
 460     # ``RangeQuerySetWrapper``. Ideally that could be refactored to be able to
 461     # be run without iteration by passing around a state object and we could
 462     # just use that here instead.
 463 
 464     source = Group.objects.get(project_id=project_id, id=source_id)
 465 
 466     # On the first iteration of this loop, we clear out all of the
 467     # denormalizations from the source group so that we can have a clean slate
 468     # for the new, repaired data.
 469     if last_event is None:
 470         fingerprints = lock_hashes(project_id, source_id, fingerprints)
 471         truncate_denormalizations(source)
 472 
 473     caches = get_caches()
 474 
 475     project = caches["Project"](project_id)
 476 
 477     # We process events sorted in descending order by -timestamp, -event_id. We need
 478     # to include event_id as well as timestamp in the ordering criteria since:
 479     #
 480     # - Event timestamps are rounded to the second so multiple events are likely
 481     # to have the same timestamp.
 482     #
 483     # - When sorting by timestamp alone, Snuba may not give us a deterministic
 484     # order for events with the same timestamp.
 485     #
 486     # - We need to ensure that we do not skip any events between batches. If we
 487     # only sorted by timestamp < last_event.timestamp it would be possible to
 488     # have missed an event with the same timestamp as the last item in the
 489     # previous batch.
 490 
 491     conditions = []
 492     if last_event is not None:
 493         conditions.extend(
 494             [
 495                 ["timestamp", "<=", last_event["timestamp"]],
 496                 [
 497                     ["timestamp", "<", last_event["timestamp"]],
 498                     ["event_id", "<", last_event["event_id"]],
 499                 ],
 500             ]
 501         )
 502 
 503     events = eventstore.get_events(
 504         filter=eventstore.Filter(
 505             project_ids=[project_id], group_ids=[source.id], conditions=conditions
 506         ),
 507         limit=batch_size,
 508         referrer="unmerge",
 509         orderby=["-timestamp", "-event_id"],
 510     )
 511 
 512     # If there are no more events to process, we're done with the migration.
 513     if not events:
 514         unlock_hashes(project_id, fingerprints)
 515         logger.warning("Unmerge complete (eventstream state: %s)", eventstream_state)
 516         if eventstream_state:
 517             eventstream.end_unmerge(eventstream_state)
 518 
 519         return destination_id
 520 
 521     source_events = []
 522     destination_events = []
 523 
 524     for event in events:
 525         (destination_events if get_fingerprint(event) in fingerprints else source_events).append(
 526             event
 527         )
 528 
 529     if source_events:
 530         if not source_fields_reset:
 531             source.update(**get_group_creation_attributes(caches, source_events))
 532             source_fields_reset = True
 533         else:
 534             source.update(**get_group_backfill_attributes(caches, source, source_events))
 535 
 536     (destination_id, eventstream_state) = migrate_events(
 537         caches,
 538         project,
 539         source_id,
 540         destination_id,
 541         fingerprints,
 542         destination_events,
 543         actor_id,
 544         eventstream_state,
 545     )
 546 
 547     repair_denormalizations(caches, project, events)
 548 
 549     unmerge.delay(
 550         project_id,
 551         source_id,
 552         destination_id,
 553         fingerprints,
 554         actor_id,
 555         last_event={"timestamp": events[-1].timestamp, "event_id": events[-1].event_id},
 556         batch_size=batch_size,
 557         source_fields_reset=source_fields_reset,
 558         eventstream_state=eventstream_state,
 559     )
