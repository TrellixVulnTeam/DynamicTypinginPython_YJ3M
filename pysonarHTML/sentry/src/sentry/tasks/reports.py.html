  65     """
  66     Calculate the relative change between a value and a reference point.
  67     """
  68     if not reference:  # handle both None and divide by zero case
  69         return None
  70 
  71     return ((value or 0) - reference) / float(reference)

  74 def safe_add(x, y):
  75     """
  76     Adds two values which are either numeric types or None.
  77 
  78     - If both values are numeric, the result is the sum of those values.
  79     - If only one numeric value is provided, that value is returned.
  80     - If both values are None, then None is returned.
  81     """
  82     if x is not None and y is not None:
  83         return x + y
  84     elif x is not None:
  85         return x
  86     elif y is not None:
  87         return y
  88     else:
  89         return None

  92 def month_to_index(year, month):
  93     """
  94     Convert a year and month to a single value: the number of months between
  95     this month and 1 AD.
  96 
  97     This mainly exists to simplify doing month-based arithmetic (e.g. "three
  98     months ago") without having to manually handle wrapping around years, since
  99     timedelta doesn't accept a "months" parameter.
 100     """
 101     assert 12 >= month >= 1
 102     return (year - 1) * 12 + month - 1

 105 def index_to_month(index):
 106     """
 107     The opposite companion to ``month_to_index``. Returns a (year, month)
 108     tuple.
 109     """
 110     return (index // 12) + 1, index % 12 + 1

 113 def clean_series(start, stop, rollup, series):
 114     """
 115     Validate a series, ensuring that it follows the specified rollup and
 116     boundaries. The start bound is inclusive, while the stop bound is
 117     exclusive (similar to the slice operation.)
 118     """
 119     start_timestamp = to_timestamp(start)
 120     stop_timestamp = to_timestamp(stop)
 121 
 122     result = []
 123     for i, (timestamp, value) in enumerate(series):
 124         assert timestamp == start_timestamp + rollup * i
 125         if timestamp >= stop_timestamp:
 126             break
 127 
 128         result.append((timestamp, value))
 129 
 130     return result

 133 def merge_sequences(target, other, function=operator.add):
 134     """
 135     Merge two sequences into a single sequence. The length of the two
 136     sequences must be equal.
 137     """
 138     assert len(target) == len(other), "sequence lengths must match"
 139     return type(target)([function(x, y) for x, y in zip(target, other)])

 142 def merge_mappings(target, other, function=lambda x, y: x + y):
 143     """
 144     Merge two mappings into a single mapping. The set of keys in both
 145     mappings must be equal.
 146     """
 147     assert set(target) == set(other), "keys must match"
 148     return {k: function(v, other[k]) for k, v in target.items()}

 151 def merge_series(target, other, function=operator.add):
 152     """
 153     Merge two series into a single series. Both series must have the same
 154     start and end points as well as the same resolution.
 155     """
 156     missing = object()
 157     results = []
 158     for x, y in zip_longest(target, other, fillvalue=missing):
 159         assert x is not missing and y is not missing, "series must be same length"
 160         assert x[0] == y[0], "series timestamps must match"
 161         results.append((x[0], function(x[1], y[1])))
 162     return results

 174 def prepare_project_series(start__stop, project, rollup=60 * 60 * 24):
 175     start, stop = start__stop
 176     resolution, series = tsdb.get_optimal_rollup_series(start, stop, rollup)
 177     assert resolution == rollup, "resolution does not match requested value"
 178     clean = functools.partial(clean_series, start, stop, rollup)
 179     issue_ids = project.group_set.filter(
 180         status=GroupStatus.RESOLVED, resolved_at__gte=start, resolved_at__lt=stop
 181     ).values_list("id", flat=True)
 182 
 183     tsdb_range = _query_tsdb_chunked(tsdb.get_range, issue_ids, start, stop, rollup)
 184 
 185     return merge_series(
 186         reduce(
 187             merge_series,
 188             map(clean, tsdb_range.values()),
 189             clean([(timestamp, 0) for timestamp in series]),
 190         ),
 191         clean(
 192             tsdb.get_range(tsdb.models.project, [project.id], start, stop, rollup=rollup)[
 193                 project.id
 194             ]
 195         ),
 196         lambda resolved, total: (resolved, total - resolved),  # unresolved
 197     )

 220 def prepare_project_issue_summaries(interval, project):
 221     start, stop = interval
 222 
 223     queryset = project.group_set.exclude(status=GroupStatus.IGNORED)
 224 
 225     # Fetch all new issues.
 226     new_issue_ids = set(
 227         queryset.filter(first_seen__gte=start, first_seen__lt=stop).values_list("id", flat=True)
 228     )
 229 
 230     # Fetch all regressions. This is a little weird, since there's no way to
 231     # tell *when* a group regressed using the Group model. Instead, we query
 232     # all groups that have been seen in the last week and have ever regressed
 233     # and query the Activity model to find out if they regressed within the
 234     # past week. (In theory, the activity table *could* be used to answer this
 235     # query without the subselect, but there's no suitable indexes to make it's
 236     # performance predictable.)
 237     reopened_issue_ids = set(
 238         Activity.objects.filter(
 239             group__in=queryset.filter(
 240                 last_seen__gte=start,
 241                 last_seen__lt=stop,
 242                 resolved_at__isnull=False,  # signals this has *ever* been resolved
 243             ),
 244             type__in=(Activity.SET_REGRESSION, Activity.SET_UNRESOLVED),
 245             datetime__gte=start,
 246             datetime__lt=stop,
 247         )
 248         .distinct()
 249         .values_list("group_id", flat=True)
 250     )
 251 
 252     rollup = 60 * 60 * 24
 253     event_counts = _query_tsdb_chunked(
 254         tsdb.get_sums, new_issue_ids | reopened_issue_ids, start, stop, rollup
 255     )
 256 
 257     new_issue_count = sum(event_counts[id] for id in new_issue_ids)
 258     reopened_issue_count = sum(event_counts[id] for id in reopened_issue_ids)
 259     existing_issue_count = max(
 260         tsdb.get_sums(tsdb.models.project, [project.id], start, stop, rollup=rollup)[project.id]
 261         - new_issue_count
 262         - reopened_issue_count,
 263         0,
 264     )
 265 
 266     return [new_issue_count, reopened_issue_count, existing_issue_count]

 281 def get_calendar_range(ignore__stop_time, months):
 282     _, stop_time = ignore__stop_time
 283     assert (
 284         stop_time.hour,
 285         stop_time.minute,
 286         stop_time.second,
 287         stop_time.microsecond,
 288         stop_time.tzinfo,
 289     ) == (0, 0, 0, 0, pytz.utc)
 290 
 291     last_day = stop_time - timedelta(days=1)
 292 
 293     stop_month_index = month_to_index(last_day.year, last_day.month)
 294 
 295     start_month_index = stop_month_index - months + 1
 296     return start_month_index, stop_month_index

 332 def build(name, fields):
 333     names, prepare_fields, merge_fields = zip(*fields)
 334 
 335     cls = namedtuple(name, names)
 336 
 337     def prepare(*args):
 338         return cls(*[f(*args) for f in prepare_fields])
 339 
 340     def merge(target, other):
 341         return cls(*[f(target[i], other[i]) for i, f in enumerate(merge_fields)])
 342 
 343     return cls, prepare, merge

 397 class RedisReportBackend(ReportBackend):
 398     version = 1
 399 
 400     def __init__(self, cluster, ttl, namespace="r"):
 401         self.cluster = cluster
 402         self.ttl = ttl
 403         self.namespace = namespace
 404 
 405     def __make_key(self, timestamp, duration, organization):
 406         return u"{}:{}:{}:{}:{}".format(
 407             self.namespace, self.version, organization.id, int(timestamp), int(duration)
 408         )
 409 
 410     def __encode(self, report):
 411         return zlib.compress(json.dumps(list(report)))
 412 
 413     def __decode(self, value):
 414         if value is None:
 415             return None
 416 
 417         return Report(*json.loads(zlib.decompress(value)))
 418 
 419     def prepare(self, timestamp, duration, organization):
 420         reports = {}
 421         for project in organization.project_set.all():
 422             reports[project.id] = self.__encode(self.build(timestamp, duration, project))
 423 
 424         if not reports:
 425             # XXX: HMSET requires at least one key/value pair, so we need to
 426             # protect ourselves here against organizations that were created
 427             # but haven't set up any projects yet.
 428             return
 429 
 430         with self.cluster.map() as client:
 431             key = self.__make_key(timestamp, duration, organization)
 432             client.hmset(key, reports)
 433             client.expire(key, self.ttl)
 434 
 435     def fetch(self, timestamp, duration, organization, projects):
 436         with self.cluster.map() as client:
 437             result = client.hmget(
 438                 self.__make_key(timestamp, duration, organization),
 439                 [project.id for project in projects],
 440             )
 441 
 442         return map(self.__decode, result.value)

 645 def build_project_breakdown_series(reports):
 646     Key = namedtuple("Key", "label url color data")
 647 
 648     def get_legend_data(report):
 649         filtered, rate_limited = report.usage_summary
 650         return {
 651             "events": sum(sum(value) for timestamp, value in report.series),
 652             "filtered": filtered,
 653             "rate_limited": rate_limited,
 654         }
 655 
 656     # Find the reports with the most total events. (The number of reports to
 657     # keep is the same as the number of colors available to use in the legend.)
 658     instances = map(
 659         operator.itemgetter(0),
 660         sorted(
 661             reports.items(),
 662             key=lambda instance__report: sum(
 663                 sum(values) for timestamp, values in instance__report[1][0]
 664             ),
 665             reverse=True,
 666         ),
 667     )[: len(colors)]
 668 
 669     # Starting building the list of items to include in the report chart. This
 670     # is a list of [Key, Report] pairs, in *ascending* order of the total sum
 671     # of values in the series. (This is so when we render the series, the
 672     # largest color blocks are at the bottom and it feels appropriately
 673     # weighted.)
 674     selections = map(
 675         lambda instance__color: (
 676             Key(
 677                 instance__color[0].slug,
 678                 instance__color[0].get_absolute_url(),
 679                 instance__color[1],
 680                 get_legend_data(reports[instance__color[0]]),
 681             ),
 682             reports[instance__color[0]],
 683         ),
 684         zip(instances, colors),
 685     )[::-1]
 686 
 687     # Collect any reports that weren't in the selection set, merge them
 688     # together and add it at the top (front) of the stack.
 689     overflow = set(reports) - set(instances)
 690     if overflow:
 691         overflow_report = reduce(merge_reports, [reports[instance] for instance in overflow])
 692         selections.insert(
 693             0, (Key("Other", None, "#f2f0fa", get_legend_data(overflow_report)), overflow_report)
 694         )
 695 
 696     def summarize(key, points):
 697         total = sum(points)
 698         return [(key, total)] if total else []
 699 
 700     # Collect all of the independent series into a single series to make it
 701     # easier to render, resulting in a series where each value is a sequence of
 702     # (key, count) pairs.
 703     series = reduce(
 704         merge_series,
 705         [series_map(functools.partial(summarize, key), report[0]) for key, report in selections],
 706     )
 707 
 708     legend = [key for key, value in reversed(selections)]
 709     return {
 710         "points": [(to_datetime(timestamp), value) for timestamp, value in series],
 711         "maximum": max(sum(count for key, count in value) for timestamp, value in series),
 712         "legend": {
 713             "rows": legend,
 714             "total": Key("Total", None, None, reduce(merge_mappings, [key.data for key in legend])),
 715         },
 716     }

 759 def get_percentile(values, percentile):
 760     # XXX: ``values`` must be sorted.
 761     assert 1 >= percentile > 0
 762     if percentile == 1:
 763         index = -1
 764     else:
 765         index = int(math.ceil(len(values) * percentile)) - 1
 766     return values[index]
 767

 769 def colorize(spectrum, values):
 770     calculate_percentile = functools.partial(get_percentile, sorted(values))
 771 
 772     legend = OrderedDict()
 773     width = 1.0 / len(spectrum)
 774     for i, color in enumerate(spectrum, 1):
 775         legend[color] = calculate_percentile(i * width)
 776 
 777     find_index = functools.partial(bisect.bisect_left, legend.values())
 778 
 779     results = []
 780     for value in values:
 781         results.append((value, spectrum[find_index(value)]))
 782 
 783     return legend, results

 786 def to_calendar(interval, series):
 787     start, stop = get_calendar_range(interval, 3)
 788 
 789     legend, values = colorize(
 790         [
 791             "#fae5cf",
 792             "#f9ddc2",
 793             "#f9d6b6",
 794             "#f9cfaa",
 795             "#f8c79e",
 796             "#f8bf92",
 797             "#f8b786",
 798             "#f9a66d",
 799             "#f99d60",
 800             "#fa9453",
 801             "#fb8034",
 802             "#fc7520",
 803             "#f9600c",
 804             "#f75500",
 805         ],
 806         [value for timestamp, value in series if value is not None],
 807     )
 808 
 809     value_color_map = dict(values)
 810     value_color_map[None] = "#F2F2F2"
 811 
 812     series_value_map = dict(series)
 813 
 814     def get_data_for_date(date):
 815         dt = datetime(date.year, date.month, date.day, tzinfo=pytz.utc)
 816         ts = to_timestamp(dt)
 817         value = series_value_map.get(ts, None)
 818         return (dt, {"value": value, "color": value_color_map[value]})
 819 
 820     calendar = Calendar(6)
 821     sheets = []
 822     for year, month in map(index_to_month, range(start, stop + 1)):
 823         weeks = []
 824 
 825         for week in calendar.monthdatescalendar(year, month):
 826             weeks.append(map(get_data_for_date, week))
 827 
 828         sheets.append((datetime(year, month, 1, tzinfo=pytz.utc), weeks))
 829 
 830     return {"legend": list(legend.keys()), "sheets": sheets}
