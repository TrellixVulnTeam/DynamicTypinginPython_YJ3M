  23 def _get_service_hooks(project_id):
  24     from sentry.models import ServiceHook
  25 
  26     cache_key = u"servicehooks:1:{}".format(project_id)
  27     result = cache.get(cache_key)
  28 
  29     if result is None:
  30         hooks = ServiceHook.objects.filter(servicehookproject__project_id=project_id)
  31         result = [(h.id, h.events) for h in hooks]
  32         cache.set(cache_key, result, 60)
  33     return result

  36 def _should_send_error_created_hooks(project):
  37     from sentry.models import ServiceHook, Organization
  38 
  39     cache_key = u"servicehooks-error-created:1:{}".format(project.id)
  40     result = cache.get(cache_key)
  41 
  42     if result is None:
  43 
  44         org = Organization.objects.get_from_cache(id=project.organization_id)
  45         if not features.has("organizations:integrations-event-hooks", organization=org):
  46             cache.set(cache_key, 0, 60)
  47             return False
  48 
  49         result = (
  50             ServiceHook.objects.filter(organization_id=org.id)
  51             .extra(where=["events @> '{error.created}'"])
  52             .exists()
  53         )
  54 
  55         cache_value = 1 if result else 0
  56         cache.set(cache_key, cache_value, 60)
  57 
  58     return result

  61 def _capture_stats(event, is_new):
  62     # TODO(dcramer): limit platforms to... something?
  63     platform = event.group.platform if event.group else event.platform
  64     if not platform:
  65         return
  66     platform = platform.split("-", 1)[0].split("_", 1)[0]
  67     tags = {"platform": platform}
  68 
  69     if is_new:
  70         metrics.incr("events.unique", tags=tags, skip_internal=False)
  71 
  72     metrics.incr("events.processed", tags=tags, skip_internal=False)
  73     metrics.incr(u"events.processed.{platform}".format(platform=platform), skip_internal=False)
  74     metrics.timing("events.size.data", event.size, tags=tags)
  75 
  76     # This is an experiment to understand whether we have, in production,
  77     # mismatches between event and group before we permanently rely on events
  78     # for the platform. before adding some more verbose logging on this
  79     # case, using a stats will give us a sense of the magnitude of the problem.
  80     if event.group:
  81         if event.group.platform != event.platform:
  82             metrics.incr("events.platform_mismatch", tags=tags)

  85 def check_event_already_post_processed(event):
  86     cluster_key = getattr(settings, "SENTRY_POST_PROCESSING_LOCK_REDIS_CLUSTER", None)
  87     if cluster_key is None:
  88         return
  89 
  90     client = redis_clusters.get(cluster_key)
  91     result = client.set(
  92         u"pp:{}/{}".format(event.project_id, event.event_id),
  93         u"{:.0f}".format(time.time()),
  94         ex=60 * 60,
  95         nx=True,
  96     )
  97 
  98     return not result

 101 def handle_owner_assignment(project, group, event):
 102     from sentry.models import GroupAssignee, ProjectOwnership
 103 
 104     # Is the issue already assigned to a team or user?
 105     key = "assignee_exists:1:%s" % (group.id)
 106     assignee_exists = cache.get(key)
 107     if assignee_exists is None:
 108         assignee_exists = group.assignee_set.exists()
 109         # Cache for an hour if it's assigned. We don't need to move that fast.
 110         cache.set(key, assignee_exists, 3600 if assignee_exists else 60)
 111     if assignee_exists:
 112         return
 113 
 114     owner = ProjectOwnership.get_autoassign_owner(group.project_id, event.data)
 115     if owner is not None:
 116         GroupAssignee.objects.assign(group, owner)

 119 @instrumented_task(name="sentry.tasks.post_process.post_process_group")
 120 def post_process_group(event, is_new, is_regression, is_new_group_environment, **kwargs):
 121     """
 122     Fires post processing hooks for a group.
 123     """
 124     set_current_project(event.project_id)
 125 
 126     from sentry.utils import snuba
 127 
 128     with snuba.options_override({"consistent": True}):
 129         if check_event_already_post_processed(event):
 130             logger.info(
 131                 "post_process.skipped",
 132                 extra={
 133                     "project_id": event.project_id,
 134                     "event_id": event.event_id,
 135                     "reason": "duplicate",
 136                 },
 137             )
 138             return
 139 
 140         # NOTE: we must pass through the full Event object, and not an
 141         # event_id since the Event object may not actually have been stored
 142         # in the database due to sampling.
 143         from sentry.models import Project, Organization, EventDict
 144         from sentry.models.group import get_group_with_redirect
 145         from sentry.rules.processor import RuleProcessor
 146         from sentry.tasks.servicehooks import process_service_hook
 147 
 148         # Re-bind node data to avoid renormalization. We only want to
 149         # renormalize when loading old data from the database.
 150         event.data = EventDict(event.data, skip_renormalization=True)
 151 
 152         if event.group_id:
 153             # Re-bind Group since we're pickling the whole Event object
 154             # which may contain a stale Project.
 155             event.group, _ = get_group_with_redirect(event.group_id)
 156             event.group_id = event.group.id
 157 
 158         # Re-bind Project and Org since we're pickling the whole Event object
 159         # which may contain stale parent models.
 160         event.project = Project.objects.get_from_cache(id=event.project_id)
 161         event.project._organization_cache = Organization.objects.get_from_cache(
 162             id=event.project.organization_id
 163         )
 164 
 165         _capture_stats(event, is_new)
 166 
 167         if event.group_id:
 168             # we process snoozes before rules as it might create a regression
 169             # but not if it's new because you can't immediately snooze a new group
 170             has_reappeared = False if is_new else process_snoozes(event.group)
 171 
 172             handle_owner_assignment(event.project, event.group, event)
 173 
 174             rp = RuleProcessor(
 175                 event, is_new, is_regression, is_new_group_environment, has_reappeared
 176             )
 177             has_alert = False
 178             # TODO(dcramer): ideally this would fanout, but serializing giant
 179             # objects back and forth isn't super efficient
 180             for callback, futures in rp.apply():
 181                 has_alert = True
 182                 with sentry_sdk.start_span(
 183                     Span(op="post_process_group", transaction="rule_processor_apply", sampled=True)
 184                 ):
 185                     safe_execute(callback, event, futures)
 186 
 187             if features.has("projects:servicehooks", project=event.project):
 188                 allowed_events = set(["event.created"])
 189                 if has_alert:
 190                     allowed_events.add("event.alert")
 191 
 192                 if allowed_events:
 193                     for servicehook_id, events in _get_service_hooks(project_id=event.project_id):
 194                         if any(e in allowed_events for e in events):
 195                             process_service_hook.delay(servicehook_id=servicehook_id, event=event)
 196 
 197             from sentry.tasks.sentry_apps import process_resource_change_bound
 198 
 199             if event.get_event_type() == "error" and _should_send_error_created_hooks(
 200                 event.project
 201             ):
 202                 process_resource_change_bound.delay(
 203                     action="created", sender="Error", instance_id=event.event_id, instance=event
 204                 )
 205             if is_new:
 206                 process_resource_change_bound.delay(
 207                     action="created", sender="Group", instance_id=event.group_id
 208                 )
 209 
 210             from sentry.plugins.base import plugins
 211 
 212             for plugin in plugins.for_project(event.project):
 213                 plugin_post_process_group(
 214                     plugin_slug=plugin.slug, event=event, is_new=is_new, is_regresion=is_regression
 215                 )
 216 
 217         event_processed.send_robust(
 218             sender=post_process_group,
 219             project=event.project,
 220             event=event,
 221             primary_hash=kwargs.get("primary_hash"),
 222         )

 225 def process_snoozes(group):
 226     """
 227     Return True if the group is transitioning from "resolved" to "unresolved",
 228     otherwise return False.
 229     """
 230     from sentry.models import GroupSnooze, GroupStatus
 231 
 232     key = GroupSnooze.get_cache_key(group.id)
 233     snooze = cache.get(key)
 234     if snooze is None:
 235         try:
 236             snooze = GroupSnooze.objects.get(group=group)
 237         except GroupSnooze.DoesNotExist:
 238             snooze = False
 239         # This cache is also set in post_save|delete.
 240         cache.set(key, snooze, 3600)
 241     if not snooze:
 242         return False
 243 
 244     if not snooze.is_valid(group, test_rates=True):
 245         snooze.delete()
 246         group.update(status=GroupStatus.UNRESOLVED)
 247         return True
 248 
 249     return False

 256 def plugin_post_process_group(plugin_slug, event, **kwargs):
 257     """
 258     Fires post processing hooks for a group.
 259     """
 260     set_current_project(event.project_id)
 261 
 262     from sentry.plugins.base import plugins
 263 
 264     plugin = plugins.get(plugin_slug)
 265     safe_execute(
 266         plugin.post_process,
 267         event=event,
 268         group=event.group,
 269         expected_errors=(PluginError,),
 270         **kwargs
 271     )
