  47 def should_process(data):
  48     """Quick check if processing is needed at all."""
  49     from sentry.plugins.base import plugins
  50 
  51     if data.get("type") == "transaction":
  52         return False
  53 
  54     for plugin in plugins.all(version=2):
  55         processors = safe_execute(
  56             plugin.get_event_preprocessors, data=data, _with_transaction=False
  57         )
  58         if processors:
  59             return True
  60 
  61         enhancers = safe_execute(plugin.get_event_enhancers, data=data, _with_transaction=False)
  62         if enhancers:
  63             return True
  64 
  65     if should_process_for_stacktraces(data):
  66         return True
  67 
  68     return False

 109 def _do_preprocess_event(cache_key, data, start_time, event_id, process_task, project):
 110     from sentry.lang.native.processing import should_process_with_symbolicator
 111 
 112     if cache_key and data is None:
 113         data = event_processing_store.get(cache_key)
 114 
 115     if data is None:
 116         metrics.incr("events.failed", tags={"reason": "cache", "stage": "pre"}, skip_internal=False)
 117         error_logger.error("preprocess.failed.empty", extra={"cache_key": cache_key})
 118         return
 119 
 120     original_data = data
 121     data = CanonicalKeyDict(data)
 122     project_id = data["project"]
 123     set_current_project(project_id)
 124 
 125     if project is None:
 126         project = Project.objects.get_from_cache(id=project_id)
 127     else:
 128         assert project.id == project_id, (project.id, project_id)
 129 
 130     from_reprocessing = process_task is process_event_from_reprocessing
 131 
 132     if should_process_with_symbolicator(data):
 133         submit_symbolicate(
 134             project, from_reprocessing, cache_key, event_id, start_time, original_data
 135         )
 136         return
 137 
 138     if should_process(data):
 139         submit_process(
 140             project,
 141             from_reprocessing,
 142             cache_key,
 143             event_id,
 144             start_time,
 145             original_data,
 146             data_has_changed=False,
 147         )
 148         return
 149 
 150     submit_save_event(project, cache_key, event_id, start_time, original_data)

 191 def _do_symbolicate_event(cache_key, start_time, event_id, symbolicate_task, data=None):
 192     from sentry.lang.native.processing import get_symbolication_function
 193 
 194     if data is None:
 195         data = event_processing_store.get(cache_key)
 196 
 197     if data is None:
 198         metrics.incr(
 199             "events.failed", tags={"reason": "cache", "stage": "symbolicate"}, skip_internal=False
 200         )
 201         error_logger.error("symbolicate.failed.empty", extra={"cache_key": cache_key})
 202         return
 203 
 204     data = CanonicalKeyDict(data)
 205 
 206     project_id = data["project"]
 207     set_current_project(project_id)
 208 
 209     event_id = data["event_id"]
 210 
 211     symbolication_function = get_symbolication_function(data)
 212 
 213     has_changed = False
 214 
 215     from_reprocessing = symbolicate_task is symbolicate_event_from_reprocessing
 216 
 217     try:
 218         with sentry_sdk.start_span(op="tasks.store.symbolicate_event.symbolication") as span:
 219             span.set_data("symbolicaton_function", symbolication_function.__name__)
 220 
 221             with metrics.timer("tasks.store.symbolicate_event.symbolication"):
 222                 symbolicated_data = symbolication_function(data)
 223 
 224             span.set_data("symbolicated_data", bool(symbolicated_data))
 225             if symbolicated_data:
 226                 data = symbolicated_data
 227                 has_changed = True
 228 
 229     except RetrySymbolication as e:
 230         if start_time and (time() - start_time) > settings.SYMBOLICATOR_PROCESS_EVENT_WARN_TIMEOUT:
 231             error_logger.warning(
 232                 "symbolicate.slow", extra={"project_id": project_id, "event_id": event_id}
 233             )
 234 
 235         if start_time and (time() - start_time) > settings.SYMBOLICATOR_PROCESS_EVENT_HARD_TIMEOUT:
 236             # Do not drop event but actually continue with rest of pipeline
 237             # (persisting unsymbolicated event)
 238             error_logger.exception(
 239                 "symbolicate.failed.infinite_retry",
 240                 extra={"project_id": project_id, "event_id": event_id},
 241             )
 242             data.setdefault("_metrics", {})["flag.processing.error"] = True
 243             data.setdefault("_metrics", {})["flag.processing.fatal"] = True
 244             has_changed = True
 245         else:
 246             # Requeue the task in the "sleep" queue
 247             retry_symbolicate_event.apply_async(
 248                 args=(),
 249                 kwargs={
 250                     "symbolicate_task_name": symbolicate_task.__name__,
 251                     "task_kwargs": {
 252                         "cache_key": cache_key,
 253                         "event_id": event_id,
 254                         "start_time": start_time,
 255                     },
 256                 },
 257                 countdown=e.retry_after,
 258             )
 259             return
 260     except Exception:
 261         error_logger.exception("tasks.store.symbolicate_event.symbolication")
 262         data.setdefault("_metrics", {})["flag.processing.error"] = True
 263         data.setdefault("_metrics", {})["flag.processing.fatal"] = True
 264         has_changed = True
 265 
 266     # We cannot persist canonical types in the cache, so we need to
 267     # downgrade this.
 268     if isinstance(data, CANONICAL_TYPES):
 269         data = dict(data.items())
 270 
 271     if has_changed:
 272         cache_key = event_processing_store.store(data)
 273 
 274     process_task = process_event_from_reprocessing if from_reprocessing else process_event
 275     _do_process_event(
 276         cache_key=cache_key,
 277         start_time=start_time,
 278         event_id=event_id,
 279         process_task=process_task,
 280         data=data,
 281         data_has_changed=has_changed,
 282         from_symbolicate=True,
 283     )

 387 def _do_process_event(
 388     cache_key,
 389     start_time,
 390     event_id,
 391     process_task,
 392     data=None,
 393     data_has_changed=None,
 394     from_symbolicate=False,
 395 ):
 396     from sentry.plugins.base import plugins
 397 
 398     if data is None:
 399         data = event_processing_store.get(cache_key)
 400 
 401     if data is None:
 402         metrics.incr(
 403             "events.failed", tags={"reason": "cache", "stage": "process"}, skip_internal=False
 404         )
 405         error_logger.error("process.failed.empty", extra={"cache_key": cache_key})
 406         return
 407 
 408     data = CanonicalKeyDict(data)
 409 
 410     project_id = data["project"]
 411     set_current_project(project_id)
 412 
 413     event_id = data["event_id"]
 414 
 415     with sentry_sdk.start_span(op="tasks.store.process_event.get_project_from_cache"):
 416         project = Project.objects.get_from_cache(id=project_id)
 417 
 418     with metrics.timer("tasks.store.process_event.organization.get_from_cache"):
 419         project._organization_cache = Organization.objects.get_from_cache(
 420             id=project.organization_id
 421         )
 422 
 423     has_changed = bool(data_has_changed)
 424 
 425     with sentry_sdk.start_span(op="tasks.store.process_event.get_reprocessing_revision"):
 426         # Fetch the reprocessing revision
 427         reprocessing_rev = reprocessing.get_reprocessing_revision(project_id)
 428 
 429     # Stacktrace based event processors.
 430     with sentry_sdk.start_span(op="task.store.process_event.stacktraces"):
 431         with metrics.timer(
 432             "tasks.store.process_event.stacktraces", tags={"from_symbolicate": from_symbolicate}
 433         ):
 434             new_data = process_stacktraces(data)
 435 
 436     if new_data is not None:
 437         has_changed = True
 438         data = new_data
 439 
 440     # Second round of datascrubbing after stacktrace and language-specific
 441     # processing. First round happened as part of ingest.
 442     #
 443     # *Right now* the only sensitive data that is added in stacktrace
 444     # processing are usernames in filepaths, so we run directly after
 445     # stacktrace processors.
 446     #
 447     # We do not yet want to deal with context data produced by plugins like
 448     # sessionstack or fullstory (which are in `get_event_preprocessors`), as
 449     # this data is very unlikely to be sensitive data. This is why scrubbing
 450     # happens somewhere in the middle of the pipeline.
 451     #
 452     # On the other hand, Javascript event error translation is happening after
 453     # this block because it uses `get_event_preprocessors` instead of
 454     # `get_event_enhancers`.
 455     #
 456     # We are fairly confident, however, that this should run *before*
 457     # re-normalization as it is hard to find sensitive data in partially
 458     # trimmed strings.
 459     if (
 460         has_changed
 461         and options.get("processing.can-use-scrubbers")
 462         and features.has("organizations:datascrubbers-v2", project.organization, actor=None)
 463     ):
 464         with sentry_sdk.start_span(op="task.store.datascrubbers.scrub"):
 465             with metrics.timer(
 466                 "tasks.store.datascrubbers.scrub", tags={"from_symbolicate": from_symbolicate}
 467             ):
 468                 project_config = get_project_config(project)
 469 
 470                 new_data = safe_execute(scrub_data, project_config=project_config, event=data.data)
 471 
 472                 # XXX(markus): When datascrubbing is finally "totally stable", we might want
 473                 # to drop the event if it crashes to avoid saving PII
 474                 if new_data is not None:
 475                     data.data = new_data
 476 
 477     # TODO(dcramer): ideally we would know if data changed by default
 478     # Default event processors.
 479     for plugin in plugins.all(version=2):
 480         with sentry_sdk.start_span(op="task.store.process_event.preprocessors") as span:
 481             span.set_data("plugin", plugin.slug)
 482             span.set_data("from_symbolicate", from_symbolicate)
 483             with metrics.timer(
 484                 "tasks.store.process_event.preprocessors",
 485                 tags={"plugin": plugin.slug, "from_symbolicate": from_symbolicate},
 486             ):
 487                 processors = safe_execute(
 488                     plugin.get_event_preprocessors, data=data, _with_transaction=False
 489                 )
 490                 for processor in processors or ():
 491                     try:
 492                         result = processor(data)
 493                     except Exception:
 494                         error_logger.exception("tasks.store.preprocessors.error")
 495                         data.setdefault("_metrics", {})["flag.processing.error"] = True
 496                         has_changed = True
 497                     else:
 498                         if result:
 499                             data = result
 500                             has_changed = True
 501 
 502     assert data["project"] == project_id, "Project cannot be mutated by plugins"
 503 
 504     # We cannot persist canonical types in the cache, so we need to
 505     # downgrade this.
 506     if isinstance(data, CANONICAL_TYPES):
 507         data = dict(data.items())
 508 
 509     if has_changed:
 510         # Run some of normalization again such that we don't:
 511         # - persist e.g. incredibly large stacktraces from minidumps
 512         # - store event timestamps that are older than our retention window
 513         #   (also happening with minidumps)
 514         normalizer = StoreNormalizer(
 515             remove_other=False, is_renormalize=True, **DEFAULT_STORE_NORMALIZER_ARGS
 516         )
 517         data = normalizer.normalize_event(dict(data))
 518 
 519         issues = data.get("processing_issues")
 520 
 521         try:
 522             if issues and create_failed_event(
 523                 cache_key,
 524                 data,
 525                 project_id,
 526                 list(issues.values()),
 527                 event_id=event_id,
 528                 start_time=start_time,
 529                 reprocessing_rev=reprocessing_rev,
 530             ):
 531                 return
 532         except RetryProcessing:
 533             # If `create_failed_event` indicates that we need to retry we
 534             # invoke ourselves again.  This happens when the reprocessing
 535             # revision changed while we were processing.
 536             _do_preprocess_event(cache_key, data, start_time, event_id, process_task, project)
 537             return
 538 
 539         cache_key = event_processing_store.store(data)
 540 
 541     submit_save_event(project, cache_key, event_id, start_time, data)
 542

 602 def delete_raw_event(project_id, event_id, allow_hint_clear=False):
 603     set_current_project(project_id)
 604 
 605     if event_id is None:
 606         error_logger.error("process.failed_delete_raw_event", extra={"project_id": project_id})
 607         return
 608 
 609     from sentry.models import RawEvent, ReprocessingReport
 610 
 611     RawEvent.objects.filter(project_id=project_id, event_id=event_id).delete()
 612     ReprocessingReport.objects.filter(project_id=project_id, event_id=event_id).delete()
 613 
 614     # Clear the sent notification if we reprocessed everything
 615     # successfully and reprocessing is enabled
 616     reprocessing_active = ProjectOption.objects.get_value(
 617         project_id, "sentry:reprocessing_active", REPROCESSING_DEFAULT
 618     )
 619     if reprocessing_active:
 620         sent_notification = ProjectOption.objects.get_value(
 621             project_id, "sentry:sent_failed_event_hint", False
 622         )
 623         if sent_notification:
 624             if ReprocessingReport.objects.filter(project_id=project_id, event_id=event_id).exists():
 625                 project = Project.objects.get_from_cache(id=project_id)
 626                 ProjectOption.objects.set_value(project, "sentry:sent_failed_event_hint", False)

 640         return False
 641 
 642     reprocessing_active = ProjectOption.objects.get_value(
 643         project_id, "sentry:reprocessing_active", REPROCESSING_DEFAULT
 644     )
 645 
 646     # In case there is reprocessing active but the current reprocessing
 647     # revision is already different than when we started, we want to
 648     # immediately retry the event.  This resolves the problem when
 649     # otherwise a concurrent change of debug symbols might leave a
 650     # reprocessing issue stuck in the project forever.
 651     if (
 652         reprocessing_active
 653         and reprocessing.get_reprocessing_revision(project_id, cached=False) != reprocessing_rev
 654     ):
 655         raise RetryProcessing()
 656 
 657     # The first time we encounter a failed event and the hint was cleared
 658     # we send a notification.
 659     sent_notification = ProjectOption.objects.get_value(
 660         project_id, "sentry:sent_failed_event_hint", False
 661     )
 662     if not sent_notification:
 663         project = Project.objects.get_from_cache(id=project_id)
 664         Activity.objects.create(
 665             type=Activity.NEW_PROCESSING_ISSUES,
 666             project=project,
 667             datetime=to_datetime(start_time),
 668             data={"reprocessing_active": reprocessing_active, "issues": issues},
 669         ).send_notification()
 670         ProjectOption.objects.set_value(project, "sentry:sent_failed_event_hint", True)
 671 
 672     # If reprocessing is not active we bail now without creating the
 673     # processing issues
 674     if not reprocessing_active:
 675         return False
 676 
 677     # We need to get the original data here instead of passing the data in
 678     # from the last processing step because we do not want any
 679     # modifications to take place.
 680     delete_raw_event(project_id, event_id)
 681     data = event_processing_store.get(cache_key)
 682     if data is None:
 683         metrics.incr("events.failed", tags={"reason": "cache", "stage": "raw"}, skip_internal=False)
 684         error_logger.error("process.failed_raw.empty", extra={"cache_key": cache_key})
 685         return True
 686 
 687     data = CanonicalKeyDict(data)
 688     from sentry.models import RawEvent, ProcessingIssue
 689 
 690     raw_event = RawEvent.objects.create(
 691         project_id=project_id,
 692         event_id=event_id,
 693         datetime=datetime.utcfromtimestamp(data["timestamp"]).replace(tzinfo=timezone.utc),
 694         data=data,
 695     )
 696 
 697     for issue in issues:
 698         ProcessingIssue.objects.record_processing_issue(
 699             raw_event=raw_event,
 700             scope=issue["scope"],
 701             object=issue["object"],
 702             type=issue["type"],
 703             data=issue["data"],
 704         )
 705 
 706     event_processing_store.delete_by_key(cache_key)
 707 
 708     return True
 709

 711 def _do_save_event(
 712     cache_key=None, data=None, start_time=None, event_id=None, project_id=None, **kwargs
 713 ):
 714     """
 715     Saves an event to the database.
 716     """
 717 
 718     set_current_project(project_id)
 719 
 720     from sentry.event_manager import EventManager, HashDiscarded
 721 
 722     event_type = "none"
 723 
 724     if cache_key and data is None:
 725         with metrics.timer("tasks.store.do_save_event.get_cache") as metric_tags:
 726             data = event_processing_store.get(cache_key)
 727             if data is not None:
 728                 metric_tags["event_type"] = event_type = data.get("type") or "none"
 729 
 730     with metrics.global_tags(event_type=event_type):
 731         if data is not None:
 732             data = CanonicalKeyDict(data)
 733 
 734         if event_id is None and data is not None:
 735             event_id = data["event_id"]
 736 
 737         # only when we come from reprocessing we get a project_id sent into
 738         # the task.
 739         if project_id is None:
 740             project_id = data.pop("project")
 741             set_current_project(project_id)
 742 
 743         # We only need to delete raw events for events that support
 744         # reprocessing.  If the data cannot be found we want to assume
 745         # that we need to delete the raw event.
 746         if not data or reprocessing.event_supports_reprocessing(data):
 747             with metrics.timer("tasks.store.do_save_event.delete_raw_event"):
 748                 delete_raw_event(project_id, event_id, allow_hint_clear=True)
 749 
 750         # This covers two cases: where data is None because we did not manage
 751         # to fetch it from the default cache or the empty dictionary was
 752         # stored in the default cache.  The former happens if the event
 753         # expired while being on the queue, the second happens on reprocessing
 754         # if the raw event was deleted concurrently while we held on to
 755         # it.  This causes the node store to delete the data and we end up
 756         # fetching an empty dict.  We could in theory not invoke `save_event`
 757         # in those cases but it's important that we always clean up the
 758         # reprocessing reports correctly or they will screw up the UI.  So
 759         # to future proof this correctly we just handle this case here.
 760         if not data:
 761             metrics.incr(
 762                 "events.failed", tags={"reason": "cache", "stage": "post"}, skip_internal=False
 763             )
 764             return
 765 
 766         event = None
 767         try:
 768             with metrics.timer("tasks.store.do_save_event.event_manager.save"):
 769                 manager = EventManager(data)
 770                 # event.project.organization is populated after this statement.
 771                 event = manager.save(
 772                     project_id, assume_normalized=True, start_time=start_time, cache_key=cache_key
 773                 )
 774 
 775         except HashDiscarded:
 776             pass
 777 
 778         finally:
 779             if cache_key:
 780                 with metrics.timer("tasks.store.do_save_event.delete_cache"):
 781                     event_processing_store.delete_by_key(cache_key)
 782 
 783                 with metrics.timer("tasks.store.do_save_event.delete_attachment_cache"):
 784                     # For the unlikely case that we did not manage to persist the
 785                     # event we also delete the key always.
 786                     if event is None or features.has(
 787                         "organizations:event-attachments", event.project.organization, actor=None
 788                     ):
 789                         attachment_cache.delete(cache_key)
 790 
 791             if start_time:
 792                 metrics.timing(
 793                     "events.time-to-process", time() - start_time, instance=data["platform"]
 794                 )
 795 
 796 
 797 @instrumented_task(
 798     name="sentry.tasks.store.save_event",
 799     queue="events.save_event",
 800     time_limit=65,
 801     soft_time_limit=60,
 802 )
 803 def save_event(
 804     cache_key=None, data=None, start_time=None, event_id=None, project_id=None, **kwargs
 805 ):
 806     _do_save_event(cache_key, data, start_time, event_id, project_id, **kwargs)
