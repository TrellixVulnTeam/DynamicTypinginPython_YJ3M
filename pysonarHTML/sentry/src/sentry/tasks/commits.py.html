  27 def generate_invalid_identity_email(identity, commit_failure=False):
  28     new_context = {
  29         "identity": identity,
  30         "auth_url": absolute_uri(reverse("socialauth_associate", args=[identity.provider])),
  31         "commit_failure": commit_failure,
  32     }
  33 
  34     return MessageBuilder(
  35         subject="Unable to Fetch Commits" if commit_failure else "Action Required",
  36         context=new_context,
  37         template="sentry/emails/identity-invalid.txt",
  38         html_template="sentry/emails/identity-invalid.html",

  56 def handle_invalid_identity(identity, commit_failure=False):
  57     # email the user
  58     msg = generate_invalid_identity_email(identity, commit_failure)
  59     msg.send_async(to=[identity.user.email])
  60 
  61     # now remove the identity, as its invalid
  62     identity.delete()

  65 @instrumented_task(
  66     name="sentry.tasks.commits.fetch_commits",
  67     queue="commits",
  68     default_retry_delay=60 * 5,
  69     max_retries=5,
  70 )
  71 @retry(exclude=(Release.DoesNotExist, User.DoesNotExist))
  72 def fetch_commits(release_id, user_id, refs, prev_release_id=None, **kwargs):
  73     # TODO(dcramer): this function could use some cleanup/refactoring as it's a bit unwieldy
  74     commit_list = []
  75 
  76     release = Release.objects.get(id=release_id)
  77     user = User.objects.get(id=user_id)
  78     prev_release = None
  79     if prev_release_id is not None:
  80         try:
  81             prev_release = Release.objects.get(id=prev_release_id)
  82         except Release.DoesNotExist:
  83             pass
  84 
  85     for ref in refs:
  86         try:
  87             repo = Repository.objects.get(
  88                 organization_id=release.organization_id, name=ref["repository"]
  89             )
  90         except Repository.DoesNotExist:
  91             logger.info(
  92                 "repository.missing",
  93                 extra={
  94                     "organization_id": release.organization_id,
  95                     "user_id": user_id,
  96                     "repository": ref["repository"],
  97                 },
  98             )
  99             continue
 100 
 101         binding_key = (
 102             "integration-repository.provider"
 103             if is_integration_provider(repo.provider)
 104             else "repository.provider"
 105         )
 106         try:
 107             provider_cls = bindings.get(binding_key).get(repo.provider)
 108         except KeyError:
 109             continue
 110 
 111         # if previous commit isn't provided, try to get from
 112         # previous release otherwise, try to get
 113         # recent commits from provider api
 114         start_sha = None
 115         if ref.get("previousCommit"):
 116             start_sha = ref["previousCommit"]
 117         elif prev_release:
 118             try:
 119                 start_sha = ReleaseHeadCommit.objects.filter(
 120                     organization_id=release.organization_id,
 121                     release=prev_release,
 122                     repository_id=repo.id,
 123                 ).values_list("commit__key", flat=True)[0]
 124             except IndexError:
 125                 pass
 126 
 127         end_sha = ref["commit"]
 128         provider = provider_cls(id=repo.provider)
 129         try:
 130             if is_integration_provider(provider.id):
 131                 repo_commits = provider.compare_commits(repo, start_sha, end_sha)
 132             else:
 133                 repo_commits = provider.compare_commits(repo, start_sha, end_sha, actor=user)
 134         except NotImplementedError:
 135             pass
 136         except Exception as e:
 137             logger.info(
 138                 "fetch_commits.error",
 139                 extra={
 140                     "organization_id": repo.organization_id,
 141                     "user_id": user_id,
 142                     "repository": repo.name,
 143                     "provider": provider.id,
 144                     "error": six.text_type(e),
 145                     "end_sha": end_sha,
 146                     "start_sha": start_sha,
 147                 },
 148             )
 149             if isinstance(e, InvalidIdentity) and getattr(e, "identity", None):
 150                 handle_invalid_identity(identity=e.identity, commit_failure=True)
 151             elif isinstance(e, (PluginError, InvalidIdentity, IntegrationError)):
 152                 msg = generate_fetch_commits_error_email(release, repo, six.text_type(e))
 153                 emails = get_emails_for_user_or_org(user, release.organization_id)
 154                 msg.send_async(to=emails)
 155             else:
 156                 msg = generate_fetch_commits_error_email(
 157                     release, repo, "An internal system error occurred."
 158                 )
 159                 emails = get_emails_for_user_or_org(user, release.organization_id)
 160                 msg.send_async(to=emails)
 161         else:
 162             logger.info(
 163                 "fetch_commits.complete",
 164                 extra={
 165                     "organization_id": repo.organization_id,
 166                     "user_id": user_id,
 167                     "repository": repo.name,
 168                     "end_sha": end_sha,
 169                     "start_sha": start_sha,
 170                     "num_commits": len(repo_commits or []),
 171                 },
 172             )
 173             commit_list.extend(repo_commits)
 174 
 175     if commit_list:
 176         release.set_commits(commit_list)
 177         deploys = Deploy.objects.filter(
 178             organization_id=release.organization_id, release=release, notified=False
 179         ).values_list("id", "environment_id", "date_finished")
 180 
 181         # XXX(dcramer): i dont know why this would have multiple environments, but for
 182         # our sanity lets assume it can
 183         pending_notifications = []
 184         last_deploy_per_environment = {}
 185         for deploy_id, environment_id, date_finished in deploys:
 186             last_deploy_per_environment[environment_id] = (deploy_id, date_finished)
 187             pending_notifications.append(deploy_id)
 188 
 189         repo_queryset = ReleaseHeadCommit.objects.filter(
 190             organization_id=release.organization_id, release=release
 191         ).values_list("repository_id", "commit")
 192 
 193         # for each repo, update (or create if this is the first one) our records
 194         # of the latest commit-associated release in each env
 195         # use deploys as a proxy for ReleaseEnvironment, because they contain
 196         # a timestamp in addition to release and env data
 197         for repository_id, commit_id in repo_queryset:
 198             for environment_id, (deploy_id, date_finished) in six.iteritems(
 199                 last_deploy_per_environment
 200             ):
 201                 # we need to mark LatestRepoReleaseEnvironment, but only if there's not a
 202                 # deploy in the given environment which has completed *after*
 203                 # this deploy (given we might process commits out of order)
 204                 if not Deploy.objects.filter(
 205                     id__in=LatestRepoReleaseEnvironment.objects.filter(
 206                         repository_id=repository_id, environment_id=environment_id
 207                     ).values("deploy_id"),
 208                     date_finished__gt=date_finished,
 209                 ).exists():
 210                     LatestRepoReleaseEnvironment.objects.create_or_update(
 211                         repository_id=repository_id,
 212                         environment_id=environment_id,
 213                         values={
 214                             "release_id": release.id,
 215                             "deploy_id": deploy_id,
 216                             "commit_id": commit_id,
 217                         },
 218                     )
 219 
 220         for deploy_id in pending_notifications:
 221             Deploy.notify_if_ready(deploy_id, fetch_complete=True)

 228 def get_emails_for_user_or_org(user, orgId):
 229     emails = []
 230     if user.is_sentry_app:
 231         members = OrganizationMember.objects.filter(
 232             organization_id=orgId, role="owner", user_id__isnull=False
 233         ).select_related("user")
 234         for m in list(members):
 235             emails.append(m.user.email)
 236     else:
 237         emails = [user.email]
 238 
 239     return emails
