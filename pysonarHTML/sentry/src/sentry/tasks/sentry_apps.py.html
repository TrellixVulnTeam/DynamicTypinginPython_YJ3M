  60 def _webhook_event_data(event, group_id, project_id):
  61     project = Project.objects.get_from_cache(id=project_id)
  62     organization = Organization.objects.get_from_cache(id=project.organization_id)
  63 
  64     event_context = event.as_dict()
  65     event_context["url"] = absolute_uri(
  66         reverse(
  67             "sentry-api-0-project-event-details",
  68             args=[project.organization.slug, project.slug, event.event_id],
  69         )
  70     )
  71 
  72     event_context["web_url"] = absolute_uri(
  73         reverse(
  74             "sentry-organization-event-detail", args=[organization.slug, group_id, event.event_id]
  75         )
  76     )
  77 
  78     # The URL has a regex OR in it ("|") which means `reverse` cannot generate
  79     # a valid URL (it can't know which option to pick). We have to manually
  80     # create this URL for, that reason.
  81     event_context["issue_url"] = absolute_uri("/api/0/issues/{}/".format(group_id))
  82 
  83     return event_context

 127 def _process_resource_change(action, sender, instance_id, retryer=None, *args, **kwargs):
 128     # The class is serialized as a string when enqueueing the class.
 129     model = TYPES[sender]
 130     # The Event model has different hooks for the different event types. The sender
 131     # determines which type eg. Error and therefore the 'name' eg. error
 132     if issubclass(model, Event):
 133         if not kwargs.get("instance"):
 134             extra = {"sender": sender, "action": action, "event_id": instance_id}
 135             logger.info("process_resource_change.event_missing_event", extra=extra)
 136             return
 137 
 138         name = sender.lower()
 139     else:
 140         # Some resources are named differently than their model. eg. Group vs Issue.
 141         # Looks up the human name for the model. Defaults to the model name.
 142         name = RESOURCE_RENAMES.get(model.__name__, model.__name__.lower())
 143 
 144     # By default, use Celery's `current` but allow a value to be passed for the
 145     # bound Task.
 146     retryer = retryer or current
 147 
 148     # We may run into a race condition where this task executes before the
 149     # transaction that creates the Group has committed.
 150     try:
 151         if issubclass(model, Event):
 152             # XXX:(Meredith): Passing through the entire event was an intentional choice
 153             # to avoid having to query NodeStore again for data we had previously in
 154             # post_process. While this is not ideal, changing this will most likely involve
 155             # an overhaul of how we do things in post_process, not just this task alone.
 156             instance = kwargs.get("instance")
 157         else:
 158             instance = model.objects.get(id=instance_id)
 159     except model.DoesNotExist as e:
 160         # Explicitly requeue the task, so we don't report this to Sentry until
 161         # we hit the max number of retries.
 162         return retryer.retry(exc=e)
 163 
 164     event = "{}.{}".format(name, action)
 165 
 166     if event not in VALID_EVENTS:
 167         return
 168 
 169     org = None
 170 
 171     if isinstance(instance, Group) or isinstance(instance, Event):
 172         org = Organization.objects.get_from_cache(
 173             id=Project.objects.get_from_cache(id=instance.project_id).organization_id
 174         )
 175 
 176     installations = filter(
 177         lambda i: event in i.sentry_app.events,
 178         SentryAppInstallation.get_installed_for_org(org.id).select_related("sentry_app"),
 179     )
 180 
 181     for installation in installations:
 182         data = {}
 183         if isinstance(instance, Event):
 184             data[name] = _webhook_event_data(instance, instance.group_id, instance.project_id)
 185             send_webhooks(installation, event, data=data)
 186         else:
 187             data[name] = serialize(instance)
 188             send_webhooks(installation, event, data=data)
 189 
 190         metrics.incr("resource_change.processed", sample_rate=1.0, tags={"change_event": event})
 191

 269 def send_webhooks(installation, event, **kwargs):
 270     try:
 271         servicehook = ServiceHook.objects.get(
 272             organization_id=installation.organization_id, actor_id=installation.id
 273         )
 274     except ServiceHook.DoesNotExist:
 275         return
 276 
 277     if event not in servicehook.events:
 278         return
 279 
 280     # The service hook applies to all projects if there are no
 281     # ServiceHookProject records. Otherwise we want check if
 282     # the event is within the allowed projects.
 283     project_limited = ServiceHookProject.objects.filter(service_hook_id=servicehook.id).exists()
 284 
 285     # TODO(nola): This is disabled for now, because it could potentially affect internal integrations w/ error.created
 286     # # If the event is error.created & the request is going out to the Org that owns the Sentry App,
 287     # # Make sure we don't send the request, to prevent potential infinite loops
 288     # if (
 289     #     event == "error.created"
 290     #     and installation.organization_id == installation.sentry_app.owner_id
 291     # ):
 292     #     # We just want to exclude error.created from the project that the integration lives in
 293     #     # Need to first implement project mapping for integration partners
 294     #     metrics.incr(
 295     #         "webhook_request.dropped",
 296     #         tags={"sentry_app": installation.sentry_app.id, "event": event},
 297     #     )
 298     #     return
 299 
 300     if not project_limited:
 301         resource, action = event.split(".")
 302 
 303         kwargs["resource"] = resource
 304         kwargs["action"] = action
 305         kwargs["install"] = installation
 306 
 307         request_data = AppPlatformEvent(**kwargs)
 308         send_and_save_webhook_request(
 309             servicehook.sentry_app.webhook_url, installation.sentry_app, request_data
 310         )

 313 def ignore_unpublished_app_errors(func):
 314     def wrapper(url, sentry_app, app_platform_event):
 315         try:
 316             return func(url, sentry_app, app_platform_event)
 317         except Exception:
 318             if sentry_app.is_published:
 319                 raise
 320             else:
 321                 raise IgnorableSentryAppError("unpublished or internal app")
 322 
 323     return wrapper

 326 @ignore_unpublished_app_errors
 327 def send_and_save_webhook_request(url, sentry_app, app_platform_event):
 328     buffer = SentryAppWebhookRequestsBuffer(sentry_app)
 329 
 330     org_id = app_platform_event.install.organization_id
 331     event = "{}.{}".format(app_platform_event.resource, app_platform_event.action)
 332     slug = sentry_app.slug_for_metrics
 333 
 334     try:
 335         resp = safe_urlopen(
 336             url=url, data=app_platform_event.body, headers=app_platform_event.headers, timeout=5
 337         )
 338 
 339     except (Timeout, ConnectionError) as e:
 340         track_response_code(e.__class__.__name__.lower(), slug, event)
 341         # Response code of 0 represents timeout
 342         buffer.add_request(response_code=0, org_id=org_id, event=event, url=url)
 343         # Re-raise the exception because some of these tasks might retry on the exception
 344         raise
 345 
 346     else:
 347         track_response_code(resp.status_code, slug, event)
 348         buffer.add_request(
 349             response_code=resp.status_code,
 350             org_id=org_id,
 351             event=event,
 352             url=url,
 353             error_id=resp.headers.get("Sentry-Hook-Error"),
 354             project_id=resp.headers.get("Sentry-Hook-Project"),
 355         )
 356 
 357         if resp.status_code == 503:
 358             raise ApiHostError.from_request(resp.request)
 359 
 360         elif resp.status_code == 504:
 361             raise ApiTimeoutError.from_request(resp.request)
 362 
 363         resp.raise_for_status()
 364 
 365         return resp
