  41 def _get_cache_key(task, scope, checksum):
  42     """Computes the cache key for assemble status.
  43 
  44     ``task`` must be one of the ``AssembleTask`` values. The scope can be the
  45     identifier of any model, such as the organization or project that this task
  46     is performed under.
  47 
  48     ``checksum`` should be the SHA1 hash of the main file that is being
  49     assembled.
  50     """
  51     return (
  52         "assemble-status:%s"
  53         % hashlib.sha1(
  54             b"%s|%s|%s" % (str(scope).encode("ascii"), checksum.encode("ascii"), task)
  55         ).hexdigest()
  56     )

  59 def get_assemble_status(task, scope, checksum):
  60     """
  61     Checks the current status of an assembling task.
  62 
  63     Returns a tuple in the form ``(status, details)``, where ``status`` is the
  64     ChunkFileState, and ``details`` is either None or a string containing a
  65     notice or error message.
  66     """
  67     cache_key = _get_cache_key(task, scope, checksum)
  68     rv = default_cache.get(cache_key)
  69     if rv is None:
  70         return None, None
  71     return tuple(rv)

  74 def set_assemble_status(task, scope, checksum, state, detail=None):
  75     """
  76     Updates the status of an assembling task. It is cached for 10 minutes.
  77     """
  78     cache_key = _get_cache_key(task, scope, checksum)
  79     default_cache.set(cache_key, (state, detail), 600)

  82 @instrumented_task(name="sentry.tasks.assemble.assemble_dif", queue="assemble")
  83 def assemble_dif(project_id, name, checksum, chunks, debug_id=None, **kwargs):
  84     """
  85     Assembles uploaded chunks into a ``ProjectDebugFile``.
  86     """
  87 
  88     from sentry.models import debugfile, Project, BadDif
  89     from sentry.reprocessing import bump_reprocessing_revision
  90 
  91     with configure_scope() as scope:
  92         scope.set_tag("project", project_id)
  93 
  94     project = Project.objects.filter(id=project_id).get()
  95     set_assemble_status(AssembleTask.DIF, project.id, checksum, ChunkFileState.ASSEMBLING)
  96 
  97     # Assemble the chunks into a temporary file
  98     rv = assemble_file(AssembleTask.DIF, project, name, checksum, chunks, file_type="project.dif")
  99 
 100     # If not file has been created this means that the file failed to
 101     # assemble because of bad input data.  Return.
 102     if rv is None:
 103         return
 104 
 105     file, temp_file = rv
 106     delete_file = True
 107     try:
 108         with temp_file:
 109             # We only permit split difs to hit this endpoint.  The
 110             # client is required to split them up first or we error.
 111             try:
 112                 result = debugfile.detect_dif_from_path(
 113                     temp_file.name, name=name, debug_id=debug_id
 114                 )
 115             except BadDif as e:
 116                 set_assemble_status(
 117                     AssembleTask.DIF, project.id, checksum, ChunkFileState.ERROR, detail=e.args[0]
 118                 )
 119                 return
 120 
 121             if len(result) != 1:
 122                 detail = "Object contains %s architectures (1 expected)" % len(result)
 123                 set_assemble_status(
 124                     AssembleTask.DIF, project.id, checksum, ChunkFileState.ERROR, detail=detail
 125                 )
 126                 return
 127 
 128             dif, created = debugfile.create_dif_from_id(project, result[0], file=file)
 129             delete_file = False
 130 
 131             if created:
 132                 # Bump the reprocessing revision since the symbol has changed
 133                 # and might resolve processing issues. If the file was not
 134                 # created, someone else has created it and will bump the
 135                 # revision instead.
 136                 bump_reprocessing_revision(project)
 137     except BaseException:
 138         set_assemble_status(
 139             AssembleTask.DIF,
 140             project.id,
 141             checksum,
 142             ChunkFileState.ERROR,
 143             detail="internal server error",
 144         )
 145         logger.error("failed to assemble dif", exc_info=True)
 146     else:
 147         set_assemble_status(
 148             AssembleTask.DIF, project.id, checksum, ChunkFileState.OK, detail=serialize(dif)
 149         )
 150     finally:
 151         if delete_file:
 152             file.delete()

 159 @instrumented_task(name="sentry.tasks.assemble.assemble_artifacts", queue="assemble")
 160 def assemble_artifacts(org_id, version, checksum, chunks, **kwargs):
 161     """
 162     Creates release files from an uploaded artifact bundle.
 163     """
 164 
 165     import shutil
 166     import tempfile
 167     from sentry.utils.zip import safe_extract_zip
 168     from sentry.models import File, Organization, Release, ReleaseFile
 169 
 170     organization = Organization.objects.get_from_cache(pk=org_id)
 171 
 172     bind_organization_context(organization)
 173 
 174     set_assemble_status(AssembleTask.ARTIFACTS, org_id, checksum, ChunkFileState.ASSEMBLING)
 175 
 176     # Assemble the chunks into a temporary file
 177     rv = assemble_file(
 178         AssembleTask.ARTIFACTS,
 179         organization,
 180         "release-artifacts.zip",
 181         checksum,
 182         chunks,
 183         file_type="release.bundle",
 184     )
 185 
 186     # If not file has been created this means that the file failed to
 187     # assemble because of bad input data.  Return.
 188     if rv is None:
 189         return
 190 
 191     bundle, temp_file = rv
 192     scratchpad = tempfile.mkdtemp()
 193 
 194     # Initially, always delete the bundle file. Later on, we can start to store
 195     # the artifact bundle as a release file.
 196     delete_bundle = True
 197 
 198     try:
 199         try:
 200             safe_extract_zip(temp_file, scratchpad, strip_toplevel=False)
 201         except BaseException:
 202             raise AssembleArtifactsError("failed to extract bundle")
 203 
 204         try:
 205             manifest_path = path.join(scratchpad, "manifest.json")
 206             with open(manifest_path, "rb") as manifest:
 207                 manifest = json.loads(manifest.read())
 208         except BaseException:
 209             raise AssembleArtifactsError("failed to open release manifest")
 210 
 211         org_slug = manifest.get("org")
 212         if organization.slug != org_slug:
 213             raise AssembleArtifactsError("organization does not match uploaded bundle")
 214 
 215         release_name = manifest.get("release")
 216         if release_name != version:
 217             raise AssembleArtifactsError("release does not match uploaded bundle")
 218 
 219         try:
 220             release = Release.objects.get(organization_id=organization.id, version=release_name)
 221         except Release.DoesNotExist:
 222             raise AssembleArtifactsError("release does not exist")
 223 
 224         dist_name = manifest.get("dist")
 225         dist = None
 226         if dist_name:
 227             dist = release.add_dist(dist_name)
 228 
 229         artifacts = manifest.get("files", {})
 230         for rel_path, artifact in six.iteritems(artifacts):
 231             artifact_url = artifact.get("url", rel_path)
 232             artifact_basename = artifact_url.rsplit("/", 1)[-1]
 233 
 234             file = File.objects.create(
 235                 name=artifact_basename, type="release.file", headers=artifact.get("headers", {})
 236             )
 237 
 238             full_path = path.join(scratchpad, rel_path)
 239             with open(full_path, "rb") as fp:
 240                 file.putfile(fp, logger=logger)
 241 
 242             kwargs = {
 243                 "organization_id": organization.id,
 244                 "release": release,
 245                 "name": artifact_url,
 246                 "dist": dist,
 247             }
 248 
 249             # Release files must have unique names within their release
 250             # and dist. If a matching file already exists, replace its
 251             # file with the new one; otherwise create it.
 252             try:
 253                 release_file = ReleaseFile.objects.get(**kwargs)
 254             except ReleaseFile.DoesNotExist:
 255                 try:
 256                     with transaction.atomic():
 257                         ReleaseFile.objects.create(file=file, **kwargs)
 258                 except IntegrityError:
 259                     # NB: This indicates a race, where another assemble task or
 260                     # file upload job has just created a conflicting file. Since
 261                     # we're upserting here anyway, yield to the faster actor and
 262                     # do not try again.
 263                     file.delete()
 264             else:
 265                 old_file = release_file.file
 266                 release_file.update(file=file)
 267                 old_file.delete()
 268 
 269     except AssembleArtifactsError as e:
 270         set_assemble_status(
 271             AssembleTask.ARTIFACTS, org_id, checksum, ChunkFileState.ERROR, detail=six.text_type(e)
 272         )
 273     except BaseException:
 274         logger.error("failed to assemble release bundle", exc_info=True)
 275         set_assemble_status(
 276             AssembleTask.ARTIFACTS,
 277             org_id,
 278             checksum,
 279             ChunkFileState.ERROR,
 280             detail="internal server error",
 281         )
 282     else:
 283         set_assemble_status(AssembleTask.ARTIFACTS, org_id, checksum, ChunkFileState.OK)
 284     finally:
 285         shutil.rmtree(scratchpad)
 286         if delete_bundle:
 287             bundle.delete()

 290 def assemble_file(task, org_or_project, name, checksum, chunks, file_type):
 291     """
 292     Verifies and assembles a file model from chunks.
 293 
 294     This downloads all chunks from blob store to verify their integrity and
 295     associates them with a created file model. Additionally, it assembles the
 296     full file in a temporary location and verifies the complete content hash.
 297 
 298     Returns a tuple ``(File, TempFile)`` on success, or ``None`` on error.
 299     """
 300     from sentry.models import File, AssembleChecksumMismatch, FileBlob, Project
 301 
 302     if isinstance(org_or_project, Project):
 303         organization = org_or_project.organization
 304     else:
 305         organization = org_or_project
 306 
 307     # Load all FileBlobs from db since we can be sure here we already own all
 308     # chunks need to build the file
 309     file_blobs = FileBlob.objects.filter(checksum__in=chunks).values_list("id", "checksum", "size")
 310 
 311     # Reject all files that exceed the maximum allowed size for this
 312     # organization. This value cannot be
 313     file_size = sum(x[2] for x in file_blobs)
 314     if file_size > get_max_file_size(organization):
 315         set_assemble_status(
 316             task,
 317             org_or_project.id,
 318             checksum,
 319             ChunkFileState.ERROR,
 320             detail="File exceeds maximum size",
 321         )
 322         return
 323 
 324     # Sanity check.  In case not all blobs exist at this point we have a
 325     # race condition.
 326     if set(x[1] for x in file_blobs) != set(chunks):
 327         set_assemble_status(
 328             task,
 329             org_or_project.id,
 330             checksum,
 331             ChunkFileState.ERROR,
 332             detail="Not all chunks available for assembling",
 333         )
 334         return
 335 
 336     # Ensure blobs are in the order and duplication in which they were
 337     # transmitted. Otherwise, we would assemble the file in the wrong order.
 338     ids_by_checksum = {chks: id for id, chks, _ in file_blobs}
 339     file_blob_ids = [ids_by_checksum[c] for c in chunks]
 340 
 341     file = File.objects.create(name=name, checksum=checksum, type=file_type)
 342     try:
 343         temp_file = file.assemble_from_file_blob_ids(file_blob_ids, checksum)
 344     except AssembleChecksumMismatch:
 345         file.delete()
 346         set_assemble_status(
 347             task,
 348             org_or_project.id,
 349             checksum,
 350             ChunkFileState.ERROR,
 351             detail="Reported checksum mismatch",
 352         )
 353     else:
 354         file.save()
 355         return file, temp_file
