 194 @retry(exclude=(Integration.DoesNotExist))
 195 @track_group_async_operation
 196 def sync_status_outbound(group_id, external_issue_id, **kwargs):
 197     try:
 198         group = Group.objects.filter(
 199             id=group_id, status__in=[GroupStatus.UNRESOLVED, GroupStatus.RESOLVED]
 200         )[0]
 201     except IndexError:
 202         return False
 203 
 204     has_issue_sync = features.has("organizations:integrations-issue-sync", group.organization)
 205     if not has_issue_sync:
 206         return False
 207 
 208     try:
 209         external_issue = ExternalIssue.objects.get(id=external_issue_id)
 210     except ExternalIssue.DoesNotExist:
 211         # Issue link could have been deleted while sync job was in the queue.
 212         return
 213     integration = Integration.objects.get(id=external_issue.integration_id)
 214     installation = integration.get_installation(organization_id=external_issue.organization_id)
 215     if installation.should_sync("outbound_status"):
 216         installation.sync_status_outbound(
 217             external_issue, group.status == GroupStatus.RESOLVED, group.project_id
 218         )
 219         analytics.record(
 220             "integration.issue.status.synced",
 221             provider=integration.provider,
 222             id=integration.id,
 223             organization_id=external_issue.organization_id,
 224         )

 250     name="sentry.tasks.integrations.migrate_repo",
 251     queue="integrations",
 252     default_retry_delay=60 * 5,
 253     max_retries=5,

 255 @retry(exclude=(Integration.DoesNotExist, Repository.DoesNotExist, Organization.DoesNotExist))
 256 def migrate_repo(repo_id, integration_id, organization_id):
 257     integration = Integration.objects.get(id=integration_id)
 258     installation = integration.get_installation(organization_id=organization_id)
 259     repo = Repository.objects.get(id=repo_id)
 260     if installation.has_repo_access(repo):
 261         # this probably shouldn't happen, but log it just in case
 262         if repo.integration_id is not None and repo.integration_id != integration_id:
 263             logger.info(
 264                 "repo.migration.integration-change",
 265                 extra={
 266                     "integration_id": integration_id,
 267                     "old_integration_id": repo.integration_id,
 268                     "organization_id": organization_id,
 269                     "repo_id": repo.id,
 270                 },
 271             )
 272 
 273         repo.integration_id = integration_id
 274         repo.provider = "integrations:%s" % (integration.provider,)
 275         # check against disabled specifically -- don't want to accidentally un-delete repos
 276         original_status = repo.status
 277         if repo.status == ObjectStatus.DISABLED:
 278             repo.status = ObjectStatus.VISIBLE
 279         repo.save()
 280         logger.info(
 281             "repo.migrated",
 282             extra={
 283                 "integration_id": integration_id,
 284                 "organization_id": organization_id,
 285                 "repo_id": repo.id,
 286                 "original_status": original_status,
 287             },
 288         )
 289 
 290         from sentry.mediators.plugins import Migrator
 291 
 292         Migrator.run(
 293             integration=integration, organization=Organization.objects.get(id=organization_id)
 294         )

 297 @instrumented_task(
 298     name="sentry.tasks.integrations.kickoff_vsts_subscription_check",
 299     queue="integrations",
 300     default_retry_delay=60 * 5,
 301     max_retries=5,
 302 )
 303 @retry()
 304 def kickoff_vsts_subscription_check():
 305     organization_integrations = OrganizationIntegration.objects.filter(
 306         integration__provider="vsts",
 307         integration__status=ObjectStatus.VISIBLE,
 308         status=ObjectStatus.VISIBLE,
 309     ).select_related("integration")
 310     six_hours_ago = time() - timedelta(hours=6).seconds
 311     for org_integration in organization_integrations:
 312         organization_id = org_integration.organization_id
 313         integration = org_integration.integration
 314 
 315         try:
 316             if (
 317                 "subscription" not in integration.metadata
 318                 or integration.metadata["subscription"]["check"] > six_hours_ago
 319             ):
 320                 continue
 321         except KeyError:
 322             pass
 323 
 324         vsts_subscription_check.apply_async(
 325             kwargs={"integration_id": integration.id, "organization_id": organization_id}
 326         )

 329 @instrumented_task(
 330     name="sentry.tasks.integrations.vsts_subscription_check",
 331     queue="integrations",
 332     default_retry_delay=60 * 5,
 333     max_retries=5,
 334 )
 335 @retry(exclude=(ApiError, ApiUnauthorized, Integration.DoesNotExist))
 336 def vsts_subscription_check(integration_id, organization_id, **kwargs):
 337     integration = Integration.objects.get(id=integration_id)
 338     installation = integration.get_installation(organization_id=organization_id)
 339     client = installation.get_client()
 340 
 341     try:
 342         subscription_id = integration.metadata["subscription"]["id"]
 343         subscription = client.get_subscription(
 344             instance=installation.instance, subscription_id=subscription_id
 345         )
 346     except (KeyError, ApiError) as e:
 347         logger.info(
 348             "vsts_subscription_check.failed_to_get_subscription",
 349             extra={
 350                 "integration_id": integration_id,
 351                 "organization_id": organization_id,
 352                 "error": six.text_type(e),
 353             },
 354         )
 355         subscription = None
 356 
 357     # https://docs.microsoft.com/en-us/rest/api/vsts/hooks/subscriptions/replace%20subscription?view=vsts-rest-4.1#subscriptionstatus
 358     if not subscription or subscription["status"] == "disabledBySystem":
 359         # Update subscription does not work for disabled subscriptions
 360         # We instead will try to delete and then create a new one.
 361 
 362         if subscription:
 363             try:
 364                 client.delete_subscription(
 365                     instance=installation.instance, subscription_id=subscription_id
 366                 )
 367             except ApiError as e:
 368                 logger.info(
 369                     "vsts_subscription_check.failed_to_delete_subscription",
 370                     extra={
 371                         "integration_id": integration_id,
 372                         "organization_id": organization_id,
 373                         "subscription_id": subscription_id,
 374                         "error": six.text_type(e),
 375                     },
 376                 )
 377 
 378         try:
 379             secret = generate_token()
 380             subscription = client.create_subscription(
 381                 instance=installation.instance, shared_secret=secret
 382             )
 383         except ApiError as e:
 384             logger.info(
 385                 "vsts_subscription_check.failed_to_create_subscription",
 386                 extra={
 387                     "integration_id": integration_id,
 388                     "organization_id": organization_id,
 389                     "error": six.text_type(e),
 390                 },
 391             )
 392         else:
 393             integration.metadata["subscription"]["id"] = subscription["id"]
 394             integration.metadata["subscription"]["secret"] = secret
 395             logger.info(
 396                 "vsts_subscription_check.updated_disabled_subscription",
 397                 extra={
 398                     "integration_id": integration_id,
 399                     "organization_id": organization_id,
 400                     "subscription_id": subscription_id,
 401                 },
 402             )
 403 
 404         integration.metadata["subscription"]["check"] = time()
 405         integration.save()
