  20 @instrumented_task(
  21     name="sentry.tasks.merge.merge_groups",
  22     queue="merge",
  23     default_retry_delay=60 * 5,
  24     max_retries=None,

  26 @track_group_async_operation
  27 def merge_groups(
  28     from_object_ids=None,
  29     to_object_id=None,
  30     transaction_id=None,
  31     recursed=False,
  32     eventstream_state=None,
  33     **kwargs
  34 ):
  35     # TODO(mattrobenolt): Write tests for all of this
  36     from sentry.models import (
  37         Activity,
  38         Group,
  39         GroupAssignee,
  40         GroupEnvironment,
  41         GroupHash,
  42         GroupRuleStatus,
  43         GroupSubscription,
  44         Environment,
  45         EventAttachment,
  46         UserReport,
  47         GroupRedirect,
  48         GroupMeta,
  49         get_group_with_redirect,
  50     )
  51 
  52     if not (from_object_ids and to_object_id):
  53         logger.error("group.malformed.missing_params", extra={"transaction_id": transaction_id})
  54         return False
  55 
  56     # Operate on one "from" group per task iteration. The task is recursed
  57     # until each group has been merged.
  58     from_object_id = from_object_ids[0]
  59 
  60     try:
  61         new_group, _ = get_group_with_redirect(to_object_id)
  62     except Group.DoesNotExist:
  63         logger.warn(
  64             "group.malformed.invalid_id",
  65             extra={"transaction_id": transaction_id, "old_object_ids": from_object_ids},
  66         )
  67         return False
  68 
  69     if not recursed:
  70         logger.info(
  71             "merge.queued",
  72             extra={
  73                 "transaction_id": transaction_id,
  74                 "new_group_id": new_group.id,
  75                 "old_group_ids": from_object_ids,
  76                 # TODO(jtcunning): figure out why these are full seq scans and/or alternative solution
  77                 # 'new_event_id': getattr(new_group.event_set.order_by('-id').first(), 'id', None),
  78                 # 'old_event_id': getattr(group.event_set.order_by('-id').first(), 'id', None),
  79                 # 'new_hash_id': getattr(new_group.grouphash_set.order_by('-id').first(), 'id', None),
  80                 # 'old_hash_id': getattr(group.grouphash_set.order_by('-id').first(), 'id', None),
  81             },
  82         )
  83 
  84     try:
  85         group = Group.objects.select_related("project").get(id=from_object_id)
  86     except Group.DoesNotExist:
  87         from_object_ids.remove(from_object_id)
  88 
  89         logger.warn(
  90             "group.malformed.invalid_id",
  91             extra={"transaction_id": transaction_id, "old_object_id": from_object_id},
  92         )
  93     else:
  94         model_list = tuple(EXTRA_MERGE_MODELS) + (
  95             Activity,
  96             GroupAssignee,
  97             GroupEnvironment,
  98             GroupHash,
  99             GroupRuleStatus,
 100             GroupSubscription,
 101             EventAttachment,
 102             UserReport,
 103             GroupRedirect,
 104             GroupMeta,
 105         )
 106 
 107         has_more = merge_objects(
 108             model_list, group, new_group, logger=logger, transaction_id=transaction_id
 109         )
 110 
 111         if not has_more:
 112             # There are no more objects to merge for *this* "from" group, remove it
 113             # from the list of "from" groups that are being merged, and finish the
 114             # work for this group.
 115             from_object_ids.remove(from_object_id)
 116 
 117             features.merge(new_group, [group], allow_unsafe=True)
 118 
 119             environment_ids = list(
 120                 Environment.objects.filter(projects=group.project).values_list("id", flat=True)
 121             )
 122 
 123             for model in [tsdb.models.group]:
 124                 tsdb.merge(
 125                     model,
 126                     new_group.id,
 127                     [group.id],
 128                     environment_ids=environment_ids
 129                     if model in tsdb.models_with_environment_support
 130                     else None,
 131                 )
 132 
 133             for model in [tsdb.models.users_affected_by_group]:
 134                 tsdb.merge_distinct_counts(
 135                     model,
 136                     new_group.id,
 137                     [group.id],
 138                     environment_ids=environment_ids
 139                     if model in tsdb.models_with_environment_support
 140                     else None,
 141                 )
 142 
 143             for model in [
 144                 tsdb.models.frequent_releases_by_group,
 145                 tsdb.models.frequent_environments_by_group,
 146             ]:
 147                 tsdb.merge_frequencies(
 148                     model,
 149                     new_group.id,
 150                     [group.id],
 151                     environment_ids=environment_ids
 152                     if model in tsdb.models_with_environment_support
 153                     else None,
 154                 )
 155 
 156             previous_group_id = group.id
 157 
 158             with transaction.atomic():
 159                 GroupRedirect.create_for_group(group, new_group)
 160                 group.delete()
 161             delete_logger.info(
 162                 "object.delete.executed",
 163                 extra={
 164                     "object_id": previous_group_id,
 165                     "transaction_id": transaction_id,
 166                     "model": Group.__name__,
 167                 },
 168             )
 169 
 170             new_group.update(
 171                 # TODO(dcramer): ideally these would be SQL clauses
 172                 first_seen=min(group.first_seen, new_group.first_seen),
 173                 last_seen=max(group.last_seen, new_group.last_seen),
 174             )
 175             try:
 176                 # it's possible to hit an out of range value for counters
 177                 new_group.update(
 178                     times_seen=F("times_seen") + group.times_seen,
 179                     num_comments=F("num_comments") + group.num_comments,
 180                 )
 181             except DataError:
 182                 pass
 183 
 184     if from_object_ids:
 185         # This task is recursed until `from_object_ids` is empty and all
 186         # "from" groups have merged into the `to_group_id`.
 187         merge_groups.delay(
 188             from_object_ids=from_object_ids,
 189             to_object_id=to_object_id,
 190             transaction_id=transaction_id,
 191             recursed=True,
 192             eventstream_state=eventstream_state,
 193         )
 194     elif eventstream_state:
 195         # All `from_object_ids` have been merged!
 196         eventstream.end_merge(eventstream_state)

 199 def _get_event_environment(event, project, cache):
 200     from sentry.models import Environment
 201 
 202     environment_name = event.get_tag("environment")
 203 
 204     if environment_name not in cache:
 205         try:
 206             environment = Environment.get_for_organization_id(
 207                 project.organization_id, environment_name
 208             )
 209         except Environment.DoesNotExist:
 210             logger.warn(
 211                 "event.environment.does_not_exist",
 212                 extra={"project_id": project.id, "environment_name": environment_name},
 213             )
 214             environment = Environment.get_or_create(project, environment_name)
 215 
 216         cache[environment_name] = environment
 217 
 218     return cache[environment_name]

 221 def merge_objects(models, group, new_group, limit=1000, logger=None, transaction_id=None):
 222     has_more = False
 223     for model in models:
 224         all_fields = [f.name for f in model._meta.get_fields()]
 225 
 226         # Not all models have a 'project' or 'project_id' field, but we make a best effort
 227         # to filter on one if it is available.
 228         # Also note that all_fields doesn't contain f.attname
 229         # (django ForeignKeys have only attribute "attname" where "_id" is implicitly appended)
 230         # but we still want to check for "project_id" because some models define a project_id bigint.
 231         has_project = "project_id" in all_fields or "project" in all_fields
 232 
 233         if has_project:
 234             project_qs = model.objects.filter(project_id=group.project_id)
 235         else:
 236             project_qs = model.objects.all()
 237 
 238         has_group = "group" in all_fields
 239         if has_group:
 240             queryset = project_qs.filter(group=group)
 241         else:
 242             queryset = project_qs.filter(group_id=group.id)
 243 
 244         for obj in queryset[:limit]:
 245             try:
 246                 with transaction.atomic(using=router.db_for_write(model)):
 247                     if has_group:
 248                         project_qs.filter(id=obj.id).update(group=new_group)
 249                     else:
 250                         project_qs.filter(id=obj.id).update(group_id=new_group.id)
 251             except IntegrityError:
 252                 delete = True
 253             else:
 254                 delete = False
 255 
 256             if delete:
 257                 # Before deleting, we want to merge in counts
 258                 if hasattr(model, "merge_counts"):
 259                     obj.merge_counts(new_group)
 260 
 261                 obj_id = obj.id
 262                 obj.delete()
 263 
 264                 if logger is not None:
 265                     delete_logger.debug(
 266                         "object.delete.executed",
 267                         extra={
 268                             "object_id": obj_id,
 269                             "transaction_id": transaction_id,
 270                             "model": model.__name__,
 271                         },
 272                     )
 273             has_more = True
 274 
 275         if has_more:
 276             return True
 277     return has_more
