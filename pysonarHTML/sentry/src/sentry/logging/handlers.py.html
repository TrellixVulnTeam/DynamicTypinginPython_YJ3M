  57 class HumanRenderer(object):
  58     def __call__(self, logger, name, event_dict):
  59         level = event_dict.pop("level")
  60         real_level = (
  61             level.upper() if isinstance(level, six.string_types) else logging.getLevelName(level)
  62         )
  63         base = "%s [%s] %s: %s" % (
  64             now().strftime("%H:%M:%S"),
  65             real_level,
  66             event_dict.pop("name", "root"),
  67             event_dict.pop("event", ""),
  68         )
  69         join = " ".join(k + "=" + repr(v) for k, v in six.iteritems(event_dict))
  70         return "%s%s" % (base, (" (%s)" % join if join else ""))

  73 class StructLogHandler(logging.StreamHandler):
  74     def emit(self, record, logger=None):
  75         # If anyone wants to use the 'extra' kwarg to provide context within
  76         # structlog, we have to strip all of the default attributes from
  77         # a record because the RootLogger will take the 'extra' dictionary
  78         # and just turn them into attributes.
  79         if logger is None:
  80             logger = get_logger()
  81 
  82         kwargs = {
  83             k: v for k, v in six.iteritems(vars(record)) if k not in throwaways and v is not None
  84         }
  85         kwargs.update({"level": record.levelno, "event": record.msg})
  86 
  87         if record.args:
  88             # record.args inside of LogRecord.__init__ gets unrolled
  89             # if it's the shape `({},)`, a single item dictionary.
  90             # so we need to check for this, and re-wrap it because
  91             # down the line of structlog, it's expected to be this
  92             # original shape.
  93             if isinstance(record.args, (tuple, list)):
  94                 kwargs["positional_args"] = record.args
  95             else:
  96                 kwargs["positional_args"] = (record.args,)
  97 
  98         logger.log(**kwargs)

 101 class MessageContainsFilter(logging.Filter):
 102     """
 103     A logging filter that allows log records where the message
 104     contains given substring(s).
 105 
 106     contains -- a string or list of strings to match
 107     """
 108 
 109     def __init__(self, contains):
 110         if not isinstance(contains, list):
 111             contains = [contains]
 112         if not all(isinstance(c, six.string_types) for c in contains):
 113             raise TypeError("'contains' must be a string or list of strings")
 114         self.contains = contains
 115 
 116     def filter(self, record):
 117         message = record.getMessage()
 118         return any(c in message for c in self.contains)

 125 class MetricsLogHandler(logging.Handler):
 126     def emit(self, record, logger=None):
 127         """
 128         Turn something like:
 129             > django.request.Forbidden (CSRF cookie not set.): /account
 130         into:
 131             > django.request.forbidden_csrf_cookie_not_set
 132         and track it as an incremented counter.
 133         """
 134         key = record.name + "." + record.getMessage()
 135         key = key.lower()
 136         key = whitespace_re.sub("_", key)
 137         key = metrics_badchars_re.sub("", key)
 138         key = ".".join(key.split(".")[:3])
 139         metrics.incr(key, skip_internal=False)
