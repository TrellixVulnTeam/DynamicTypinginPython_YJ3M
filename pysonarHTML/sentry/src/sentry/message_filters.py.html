  22 def should_filter_event(project_config, data):
  23     """
  24     Checks if an event should be filtered
  25 
  26     :param project_config: relay config for the request (for the project really)
  27     :param data: the event data
  28     :return: an EventFilteredRet explaining if the event should be filtered and, if it should the reason
  29         for filtering
  30     """
  31     for event_filter in get_all_filters():
  32         if _is_filter_enabled(project_config, event_filter) and event_filter(project_config, data):
  33             return EventFilteredRet(should_filter=True, reason=event_filter.spec.id)
  34 
  35     return EventFilteredRet(should_filter=False, reason=None)

  55 def set_filter_state(filter_id, project, state):
  56     flt = _filter_from_filter_id(filter_id)
  57     if flt is None:
  58         raise FilterNotRegistered(filter_id)
  59 
  60     if flt == _legacy_browsers_filter:
  61         if state is None:
  62             state = {}
  63 
  64         option_val = "0"
  65         if "active" in state:
  66             if state["active"]:
  67                 option_val = "1"
  68         elif "subfilters" in state and len(state["subfilters"]) > 0:
  69             option_val = set(state["subfilters"])
  70 
  71         ProjectOption.objects.set_value(
  72             project=project, key=u"filters:{}".format(filter_id), value=option_val
  73         )
  74 
  75         return option_val == "1" if option_val in ("0", "1") else option_val
  76 
  77     else:
  78         # all boolean filters
  79         if state is None:
  80             state = {"active": True}
  81 
  82         ProjectOption.objects.set_value(
  83             project=project,
  84             key=u"filters:{}".format(filter_id),
  85             value="1" if state.get("active", False) else "0",
  86         )
  87 
  88         if state:
  89             inbound_filter_toggled.send(project=project, sender=flt)
  90 
  91         return state.get("active", False)

  94 def get_filter_state(filter_id, project):
  95     """
  96     Returns the filter state
  97 
  98     IMPORTANT: this function accesses the database, it should NEVER be used by the ingestion pipe.
  99     This api is used by the ProjectFilterDetails and ProjectFilters endpoints
 100     :param filter_id: the filter Id
 101     :param project: the project for which we want the filter state
 102     :return: True if the filter is enabled False otherwise
 103     :raises: ValueError if filter id not registered
 104     """
 105     flt = _filter_from_filter_id(filter_id)
 106     if flt is None:
 107         raise FilterNotRegistered(filter_id)
 108 
 109     filter_state = ProjectOption.objects.get_value(
 110         project=project, key=u"filters:{}".format(flt.spec.id)
 111     )
 112 
 113     if filter_state is None:
 114         raise ValueError(
 115             "Could not find filter state for filter {0}."
 116             " You need to register default filter state in projectoptions.defaults.".format(
 117                 filter_id
 118             )
 119         )
 120 
 121     if flt == _legacy_browsers_filter:
 122         # special handling for legacy browser state
 123         if filter_state == "1":
 124             return True
 125         if filter_state == "0":
 126             return False
 127         return filter_state
 128     else:
 129         return filter_state == "1"

 136 def _filter_from_filter_id(filter_id):
 137     """
 138     Returns the corresponding filter for a filter id or None if no filter with the given id found
 139     """
 140     for flt in get_all_filters():
 141         if flt.spec.id == filter_id:
 142             return flt
 143     return None

 194     url = get_path(data, "request", "url") or ""
 195     domain = urlparse(url).hostname

 197     return ip_address in _LOCAL_IPS or domain in _LOCAL_DOMAINS

 284 def _browser_extensions_filter(project_config, data):
 285     if data.get("platform") != "javascript":
 286         return False
 287 
 288     # get exception value
 289     try:
 290         exc_value = data["exception"]["values"][0]["value"]
 291     except (LookupError, TypeError):
 292         exc_value = ""
 293     if exc_value:
 294         if _EXTENSION_EXC_VALUES.search(exc_value):
 295             return True
 296 
 297     # get exception source
 298     try:
 299         exc_source = data["exception"]["values"][0]["stacktrace"]["frames"][-1]["abs_path"]
 300     except (LookupError, TypeError):
 301         exc_source = ""
 302     if exc_source:
 303         if _EXTENSION_EXC_SOURCES.search(exc_source):
 304             return True
 305 
 306     return False

 328 def _legacy_browsers_filter(project_config, data):
 329     def get_user_agent(data):
 330         try:
 331             for key, value in get_path(data, "request", "headers", filter=True) or ():
 332                 if key.lower() == "user-agent":
 333                     return value
 334         except LookupError:
 335             return ""
 336 
 337     if data.get("platform") != "javascript":
 338         return False
 339 
 340     value = get_user_agent(data)
 341     if not value:
 342         return False
 343 
 344     ua = Parse(value)
 345     if not ua:
 346         return False
 347 
 348     browser = ua["user_agent"]
 349 
 350     if not browser["family"]:
 351         return False
 352 
 353     # IE Desktop and IE Mobile use the same engines, therefore we can treat them as one
 354     if browser["family"] == "IE Mobile":
 355         browser["family"] = "IE"
 356 
 357     filter_settings = _get_filter_settings(project_config, _legacy_browsers_filter)
 358 
 359     # handle old style config
 360     if filter_settings is None:
 361         return _filter_default(browser)
 362 
 363     enabled_sub_filters = filter_settings.get("options")
 364     if isinstance(enabled_sub_filters, collections.Sequence):
 365         for sub_filter_name in enabled_sub_filters:
 366             sub_filter = _legacy_browsers_sub_filters.get(sub_filter_name)
 367             if sub_filter is not None and sub_filter(browser):
 368                 return True
 369 
 370     return False

 398 def _filter_default(browser):
 399     """
 400     Legacy filter - new users specify individual filters
 401     """
 402     try:
 403         minimum_version = MIN_VERSIONS[browser["family"]]
 404     except KeyError:
 405         return False
 406 
 407     try:
 408         major_browser_version = int(browser["major"])
 409     except (TypeError, ValueError):
 410         return False
 411 
 412     if minimum_version > major_browser_version:
 413         return True
 414 
 415     return False

 564 def _web_crawlers_filter(project_config, data):
 565     try:
 566         for key, value in get_path(data, "request", "headers", filter=True) or ():
 567             if key.lower() == "user-agent":
 568                 if not value:
 569                     return False
 570                 return bool(_CRAWLERS.search(value))
 571         return False
 572     except LookupError:
 573         return False
