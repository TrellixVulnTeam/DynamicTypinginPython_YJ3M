  20     def __init__(self, **options):

  27     def delivery_callback(self, error, message):
  28         if error is not None:
  29             logger.warning("Could not publish message (error: %s): %r", error, message)

  31     def _send(
  32         self,
  33         project_id,
  34         _type,
  35         extra_data=(),
  36         asynchronous=True,
  37         headers=None,  # Optional[Mapping[str, str]]
  38     ):
  39         if headers is None:
  40             headers = {}
  41 
  42         # Polling the producer is required to ensure callbacks are fired. This
  43         # means that the latency between a message being delivered (or failing
  44         # to be delivered) and the corresponding callback being fired is
  45         # roughly the same as the duration of time that passes between publish
  46         # calls. If this ends up being too high, the publisher should be moved
  47         # into a background thread that can poll more frequently without
  48         # interfering with request handling. (This does `poll` does not act as
  49         # a heartbeat for the purposes of any sort of session expiration.)
  50         # Note that this call to poll() is *only* dealing with earlier
  51         # asynchronous produce() calls from the same process.
  52         self.producer.poll(0.0)
  53 
  54         assert isinstance(extra_data, tuple)
  55         key = six.text_type(project_id)
  56 
  57         try:
  58             self.producer.produce(
  59                 topic=self.topic,
  60                 key=key.encode("utf-8"),
  61                 value=json.dumps((self.EVENT_PROTOCOL_VERSION, _type) + extra_data),
  62                 on_delivery=self.delivery_callback,
  63                 headers=[(k, v.encode("utf-8")) for k, v in headers.items()],
  64             )
  65         except Exception as error:
  66             logger.error("Could not publish message: %s", error, exc_info=True)
  67             return
  68 
  69         if not asynchronous:
  70             # flush() is a convenience method that calls poll() until len() is zero
  71             self.producer.flush()
  72 
  73     def requires_post_process_forwarder(self):
  74         return True

  76     def run_post_process_forwarder(
  77         self,
  78         consumer_group,
  79         commit_log_topic,
  80         synchronize_commit_group,
  81         commit_batch_size=100,
  82         initial_offset_reset="latest",
  83     ):
  84         logger.debug("Starting post-process forwarder...")
  85 
  86         cluster_name = settings.KAFKA_TOPICS[settings.KAFKA_EVENTS]["cluster"]
  87         bootstrap_servers = settings.KAFKA_CLUSTERS[cluster_name]["bootstrap.servers"]
  88 
  89         consumer = SynchronizedConsumer(
  90             bootstrap_servers=bootstrap_servers,
  91             consumer_group=consumer_group,
  92             commit_log_topic=commit_log_topic,
  93             synchronize_commit_group=synchronize_commit_group,
  94             initial_offset_reset=initial_offset_reset,
  95         )
  96 
  97         owned_partition_offsets = {}
  98 
  99         def commit(partitions):
 100             results = consumer.commit(offsets=partitions, asynchronous=False)
 101 
 102             errors = [i for i in results if i.error is not None]
 103             if errors:
 104                 raise Exception(
 105                     "Failed to commit %s/%s partitions: %r" % (len(errors), len(partitions), errors)
 106                 )
 107 
 108             return results
 109 
 110         def on_assign(consumer, partitions):
 111             logger.debug("Received partition assignment: %r", partitions)
 112 
 113             for i in partitions:
 114                 if i.offset == OFFSET_INVALID:
 115                     updated_offset = None
 116                 elif i.offset < 0:
 117                     raise Exception(
 118                         "Received unexpected negative offset during partition assignment: %r" % (i,)
 119                     )
 120                 else:
 121                     updated_offset = i.offset
 122 
 123                 key = (i.topic, i.partition)
 124                 previous_offset = owned_partition_offsets.get(key, None)
 125                 if previous_offset is not None and previous_offset != updated_offset:
 126                     logger.warning(
 127                         "Received new offset for owned partition %r, will overwrite previous stored offset %r with %r.",
 128                         key,
 129                         previous_offset,
 130                         updated_offset,
 131                     )
 132 
 133                 owned_partition_offsets[key] = updated_offset
 134 
 135         def on_revoke(consumer, partitions):
 136             logger.debug("Revoked partition assignment: %r", partitions)
 137 
 138             offsets_to_commit = []
 139 
 140             for i in partitions:
 141                 key = (i.topic, i.partition)
 142 
 143                 try:
 144                     offset = owned_partition_offsets.pop(key)
 145                 except KeyError:
 146                     logger.warning(
 147                         "Received unexpected partition revocation for unowned partition: %r",
 148                         i,
 149                         exc_info=True,
 150                     )
 151                     continue
 152 
 153                 if offset is None:
 154                     logger.debug("Skipping commit of unprocessed partition: %r", i)
 155                     continue
 156 
 157                 offsets_to_commit.append(TopicPartition(i.topic, i.partition, offset))
 158 
 159             if offsets_to_commit:
 160                 logger.debug(
 161                     "Committing offset(s) for %s revoked partition(s): %r",
 162                     len(offsets_to_commit),
 163                     offsets_to_commit,
 164                 )
 165                 commit(offsets_to_commit)
 166 
 167         consumer.subscribe([self.topic], on_assign=on_assign, on_revoke=on_revoke)
 168 
 169         def commit_offsets():
 170             offsets_to_commit = []
 171             for (topic, partition), offset in owned_partition_offsets.items():
 172                 if offset is None:
 173                     logger.debug("Skipping commit of unprocessed partition: %r", (topic, partition))
 174                     continue
 175 
 176                 offsets_to_commit.append(TopicPartition(topic, partition, offset))
 177 
 178             if offsets_to_commit:
 179                 logger.debug(
 180                     "Committing offset(s) for %s owned partition(s): %r",
 181                     len(offsets_to_commit),
 182                     offsets_to_commit,
 183                 )
 184                 commit(offsets_to_commit)
 185 
 186         try:
 187             i = 0
 188             while True:
 189                 message = consumer.poll(0.1)
 190                 if message is None:
 191                     continue
 192 
 193                 error = message.error()
 194                 if error is not None:
 195                     raise Exception(error)
 196 
 197                 key = (message.topic(), message.partition())
 198                 if key not in owned_partition_offsets:
 199                     logger.warning("Skipping message for unowned partition: %r", key)
 200                     continue
 201 
 202                 i = i + 1
 203                 owned_partition_offsets[key] = message.offset() + 1
 204 
 205                 with metrics.timer("eventstream.duration", instance="get_task_kwargs_for_message"):
 206                     task_kwargs = get_task_kwargs_for_message(message.value())
 207 
 208                 if task_kwargs is not None:
 209                     with metrics.timer(
 210                         "eventstream.duration", instance="dispatch_post_process_group_task"
 211                     ):
 212                         self._dispatch_post_process_group_task(**task_kwargs)
 213 
 214                 if i % commit_batch_size == 0:
 215                     commit_offsets()
 216         except KeyboardInterrupt:
 217             pass
 218 
 219         logger.debug("Committing offsets and closing consumer...")
 220         commit_offsets()
 221 
 222         consumer.close()
