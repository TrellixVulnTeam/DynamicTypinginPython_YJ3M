  28 def get_commit_data(message):
  29     topic, partition, group = message.key().decode("utf-8").split(":", 3)
  30     partition = int(partition)
  31     offset = int(message.value().decode("utf-8"))
  32     return group, topic, partition, offset

  41 def run_commit_log_consumer(
  42     bootstrap_servers,
  43     consumer_group,
  44     commit_log_topic,
  45     partition_state_manager,
  46     synchronize_commit_group,
  47     start_event,
  48     stop_request_event,
  49 ):
  50     start_event.set()
  51 
  52     logging.debug("Starting commit log consumer...")
  53 
  54     positions = {}
  55 
  56     # NOTE: The commit log consumer group should not be persisted into the
  57     # ``__consumer_offsets`` topic since no offsets are committed by this
  58     # consumer. The group membership metadata messages will be published
  59     # initially but as long as this group remains a single consumer it will
  60     # be deleted after the consumer is closed.
  61     # It is very important to note that the ``group.id`` **MUST** be unique to
  62     # this consumer process!!! This ensures that it is able to consume from all
  63     # partitions of the commit log topic and get a comprehensive view of the
  64     # state of the consumer groups it is tracking.
  65     consumer = Consumer(
  66         {
  67             "bootstrap.servers": bootstrap_servers,
  68             "group.id": consumer_group,
  69             "enable.auto.commit": "false",
  70             "enable.auto.offset.store": "true",
  71             "enable.partition.eof": "false",
  72             "default.topic.config": {"auto.offset.reset": "error"},
  73         }
  74     )
  75 
  76     def rewind_partitions_on_assignment(consumer, assignment):
  77         # The commit log consumer must start consuming from the beginning of
  78         # the commit log topic to ensure that it has a comprehensive view of
  79         # all active partitions.
  80         consumer.assign(
  81             [
  82                 TopicPartition(
  83                     i.topic, i.partition, positions.get((i.topic, i.partition), OFFSET_BEGINNING)
  84                 )
  85                 for i in assignment
  86             ]
  87         )
  88 
  89     consumer.subscribe([commit_log_topic], on_assign=rewind_partitions_on_assignment)
  90 
  91     while not stop_request_event.is_set():
  92         message = consumer.poll(1)
  93         if message is None:
  94             continue
  95 
  96         error = message.error()
  97         if error is not None:
  98             raise Exception(error)
  99 
 100         positions[(message.topic(), message.partition())] = message.offset() + 1
 101 
 102         group, topic, partition, offset = get_commit_data(message)
 103         if group != synchronize_commit_group:
 104             logger.debug("Received consumer offsets update from %r, ignoring...", group)
 105             continue
 106 
 107         if offset in LOGICAL_OFFSETS:
 108             logger.debug(
 109                 "Skipping invalid logical offset (%r) from %s/%s...", offset, topic, partition
 110             )
 111             continue
 112         elif offset < 0:
 113             logger.warning(
 114                 "Received unexpected negative offset (%r) from %s/%s!", offset, topic, partition
 115             )
 116 
 117         partition_state_manager.set_remote_offset(topic, partition, offset)

 130 class SynchronizedConsumer(object):
 131     """
 132     This class implements the framework for a consumer that is intended to only
 133     consume messages that have already been consumed and committed by members
 134     of another consumer group.
 135 
 136     This works similarly to the Kafka built-in ``__consumer_offsets`` topic.
 137     The consumer group that is being "followed" (the one that must make
 138     progress for our consumer here to make progress, identified by the
 139     ``synchronize_commit_group`` constructor parameter/instance attribute) must
 140     report its offsets to a topic (identified by the ``commit_log_topic``
 141     constructor parameter/instance attribute). This consumer subscribes to both
 142     commit log topic, as well as the topic(s) that we are actually interested
 143     in consuming messages from. The messages received from the commit log topic
 144     control whether or not consumption from partitions belonging to the main
 145     topic is paused, resumed, or allowed to continue in its current state
 146     without changes.
 147 
 148     The furthest point in any partition that this consumer should ever consume
 149     to is the maximum offset that has been recorded to the commit log topic for
 150     that partition. If the offsets recorded to that topic move
 151     non-monotonically (due to an intentional offset rollback, for instance)
 152     this consumer *may* consume up to the highest watermark point. (The
 153     implementation here tries to pause consuming from the partition as soon as
 154     possible, but this makes no explicit guarantees about that behavior.)
 155     """
 156 
 157     initial_offset_reset_strategies = {"earliest": get_earliest_offset, "latest": get_latest_offset}
 158 
 159     def __init__(
 160         self,
 161         bootstrap_servers,
 162         consumer_group,
 163         commit_log_topic,
 164         synchronize_commit_group,
 165         initial_offset_reset="latest",
 166         on_commit=None,
 167     ):
 168         self.bootstrap_servers = bootstrap_servers
 169         self.consumer_group = consumer_group
 170         self.commit_log_topic = commit_log_topic
 171         self.synchronize_commit_group = synchronize_commit_group
 172         self.initial_offset_reset = self.initial_offset_reset_strategies[initial_offset_reset]
 173 
 174         self.__partition_state_manager = SynchronizedPartitionStateManager(
 175             self.__on_partition_state_change
 176         )
 177         (
 178             self.__commit_log_consumer,
 179             self.__commit_log_consumer_stop_request,
 180         ) = self.__start_commit_log_consumer()
 181 
 182         self.__positions = {}
 183 
 184         def commit_callback(error, partitions):
 185             if on_commit is not None:
 186                 return on_commit(error, partitions)
 187 
 188         consumer_configuration = {
 189             "bootstrap.servers": self.bootstrap_servers,
 190             "group.id": self.consumer_group,
 191             "enable.auto.commit": "false",
 192             "enable.auto.offset.store": "true",
 193             "enable.partition.eof": "false",
 194             "default.topic.config": {"auto.offset.reset": "error"},
 195             "on_commit": commit_callback,
 196         }
 197 
 198         self.__consumer = Consumer(consumer_configuration)
 199 
 200     def __start_commit_log_consumer(self, timeout=None):
 201         """
 202         Starts running the commit log consumer.
 203         """
 204         stop_request_event = threading.Event()
 205         start_event = threading.Event()
 206         result = execute(
 207             functools.partial(
 208                 run_commit_log_consumer,
 209                 bootstrap_servers=self.bootstrap_servers,
 210                 consumer_group="{}:sync:{}".format(self.consumer_group, uuid.uuid1().hex),
 211                 commit_log_topic=self.commit_log_topic,
 212                 synchronize_commit_group=self.synchronize_commit_group,
 213                 partition_state_manager=self.__partition_state_manager,
 214                 start_event=start_event,
 215                 stop_request_event=stop_request_event,
 216             )
 217         )
 218         start_event.wait(timeout)
 219         return result, stop_request_event
 220 
 221     def __check_commit_log_consumer_running(self):
 222         if not self.__commit_log_consumer.running():
 223             try:
 224                 result = self.__commit_log_consumer.result(timeout=0)  # noqa
 225             except TimeoutError:
 226                 pass  # not helpful
 227 
 228             raise Exception("Commit log consumer unexpectedly exit!")

 230     def __on_partition_state_change(
 231         self, topic, partition, previous_state_and_offsets, current_state_and_offsets
 232     ):
 233         """
 234         Callback that is invoked when a partition state changes.
 235         """
 236         logger.debug(
 237             "State change for %r: %r to %r",
 238             (topic, partition),
 239             previous_state_and_offsets,
 240             current_state_and_offsets,
 241         )
 242 
 243         current_state, current_offsets = current_state_and_offsets
 244         if current_offsets.local is None:
 245             # It only makes sense to manipulate the consumer if we've got an
 246             # assignment. (This block should only be entered at startup if the
 247             # remote offsets are retrieved from the commit log before the local
 248             # consumer has received its assignment.)
 249             return
 250 
 251         # TODO: This will be called from the commit log consumer thread, so need
 252         # to verify that calling the ``consumer.{pause,resume}`` methods is
 253         # thread safe!
 254         if current_state in (
 255             SynchronizedPartitionState.UNKNOWN,
 256             SynchronizedPartitionState.SYNCHRONIZED,
 257             SynchronizedPartitionState.REMOTE_BEHIND,
 258         ):
 259             self.__consumer.pause([TopicPartition(topic, partition, current_offsets.local)])
 260         elif current_state is SynchronizedPartitionState.LOCAL_BEHIND:
 261             self.__consumer.resume([TopicPartition(topic, partition, current_offsets.local)])
 262         else:
 263             raise NotImplementedError("Unexpected partition state: %s" % (current_state,))
 264 
 265     def subscribe(self, topics, on_assign=None, on_revoke=None):
 266         """
 267         Subscribe to a topic.
 268         """
 269         self.__check_commit_log_consumer_running()
 270 
 271         def assignment_callback(consumer, assignment):
 272             # Since ``auto.offset.reset`` is set to ``error`` to force human
 273             # interaction on an offset reset, we have to explicitly specify the
 274             # starting offset if no offset has been committed for this topic during
 275             # the ``__consumer_offsets`` topic retention period.
 276             assignment = {
 277                 (i.topic, i.partition): self.__positions.get((i.topic, i.partition))
 278                 for i in assignment
 279             }
 280 
 281             for i in self.__consumer.committed(
 282                 [
 283                     TopicPartition(topic, partition)
 284                     for (topic, partition), offset in assignment.items()
 285                     if offset is None
 286                 ]
 287             ):
 288                 k = (i.topic, i.partition)
 289                 if i.offset > -1:
 290                     assignment[k] = i.offset
 291                 else:
 292                     assignment[k] = self.initial_offset_reset(consumer, i.topic, i.partition)
 293 
 294             self.__consumer.assign(
 295                 [
 296                     TopicPartition(topic, partition, offset)
 297                     for (topic, partition), offset in assignment.items()
 298                 ]
 299             )
 300 
 301             for (topic, partition), offset in assignment.items():
 302                 # Setting the local offsets will either cause the partition to be
 303                 # paused (if the remote offset is unknown or the local offset is
 304                 # not trailing the remote offset) or resumed.
 305                 self.__partition_state_manager.set_local_offset(topic, partition, offset)
 306                 self.__positions[(topic, partition)] = offset
 307 
 308             if on_assign is not None:
 309                 on_assign(
 310                     self,
 311                     [TopicPartition(topic, partition) for topic, partition in assignment.keys()],
 312                 )
 313 
 314         def revocation_callback(consumer, assignment):
 315             for item in assignment:
 316                 # TODO: This should probably also be removed from the state manager.
 317                 self.__positions.pop((item.topic, item.partition))
 318 
 319             if on_revoke is not None:
 320                 on_revoke(self, assignment)
 321 
 322         self.__consumer.subscribe(
 323             topics, on_assign=assignment_callback, on_revoke=revocation_callback
 324         )
 325 
 326     def poll(self, timeout):
 327         self.__check_commit_log_consumer_running()
 328 
 329         message = self.__consumer.poll(timeout)
 330         if message is None:
 331             return
 332 
 333         if message.error() is not None:
 334             return message
 335 
 336         self.__partition_state_manager.validate_local_message(
 337             message.topic(), message.partition(), message.offset()
 338         )
 339         self.__partition_state_manager.set_local_offset(
 340             message.topic(), message.partition(), message.offset() + 1
 341         )
 342         self.__positions[(message.topic(), message.partition())] = message.offset() + 1
 343 
 344         return message
 345 
 346     def commit(self, *args, **kwargs):
 347         self.__check_commit_log_consumer_running()
 348 
 349         return self.__consumer.commit(*args, **kwargs)
 350 
 351     def close(self):
 352         self.__check_commit_log_consumer_running()
 353 
 354         self.__commit_log_consumer_stop_request.set()
 355         try:
 356             self.__consumer.close()
 357         finally:
 358             self.__commit_log_consumer.result()
