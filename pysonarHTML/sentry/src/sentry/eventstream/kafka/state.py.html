  26 class SynchronizedPartitionState:
  27     # The ``SYNCHRONIZED`` state represents that the local offset is equal to
  28     # the remote offset. The local consumer should be paused to avoid advancing
  29     # further beyond the remote consumer.
  30     SYNCHRONIZED = "SYNCHRONIZED"
  31 
  32     # The ``LOCAL_BEHIND`` state represents that the remote offset is greater
  33     # than the local offset. The local consumer should be unpaused to avoid
  34     # falling behind the remote consumer.
  35     LOCAL_BEHIND = "LOCAL_BEHIND"
  36 
  37     # The ``REMOTE_BEHIND`` state represents that the local offset is greater
  38     # than the remote offset. The local consumer should be paused to avoid
  39     # advancing further beyond the remote consumer.
  40     REMOTE_BEHIND = "REMOTE_BEHIND"
  41 
  42     # The ``UNKNOWN`` state represents that we haven't received enough data to
  43     # know the current offset state.
  44     UNKNOWN = "UNKNOWN"

  47 class SynchronizedPartitionStateManager(object):
  48     """
  49     This class implements a state machine that can be used to track the
  50     consumption progress of a Kafka partition (the "local" consumer) relative
  51     to a the progress of another consumer (the "remote" consumer.)
  52 
  53     This is intended to be paired with the ``SynchronizedConsumer``.
  54     """
  55 
  56     transitions = {  # from state -> set(to states)
  57         None: frozenset([SynchronizedPartitionState.UNKNOWN]),
  58         SynchronizedPartitionState.UNKNOWN: frozenset(
  59             [
  60                 SynchronizedPartitionState.LOCAL_BEHIND,
  61                 SynchronizedPartitionState.REMOTE_BEHIND,
  62                 SynchronizedPartitionState.SYNCHRONIZED,
  63             ]
  64         ),
  65         SynchronizedPartitionState.REMOTE_BEHIND: frozenset(
  66             [SynchronizedPartitionState.LOCAL_BEHIND, SynchronizedPartitionState.SYNCHRONIZED]
  67         ),
  68         SynchronizedPartitionState.LOCAL_BEHIND: frozenset(
  69             [SynchronizedPartitionState.SYNCHRONIZED, SynchronizedPartitionState.REMOTE_BEHIND]
  70         ),
  71         SynchronizedPartitionState.SYNCHRONIZED: frozenset(
  72             [SynchronizedPartitionState.LOCAL_BEHIND, SynchronizedPartitionState.REMOTE_BEHIND]
  73         ),
  74     }
  75 
  76     def __init__(self, callback):
  77         self.partitions = defaultdict(lambda: (None, Offsets(None, None)))
  78         self.callback = callback
  79         self.__lock = threading.RLock()
  80 
  81     def get_state_from_offsets(self, offsets):
  82         """
  83         Derive the partition state by comparing local and remote offsets.
  84         """
  85         if offsets.local is None or offsets.remote is None:
  86             return SynchronizedPartitionState.UNKNOWN
  87         else:
  88             if offsets.local < offsets.remote:
  89                 return SynchronizedPartitionState.LOCAL_BEHIND
  90             elif offsets.remote < offsets.local:
  91                 return SynchronizedPartitionState.REMOTE_BEHIND
  92             else:  # local == remote
  93                 return SynchronizedPartitionState.SYNCHRONIZED
  94 
  95     def set_local_offset(self, topic, partition, local_offset):
  96         """
  97         Update the local offset for a topic and partition.
  98 
  99         If this update operation results in a state change, the callback
 100         function will be invoked.
 101         """
 102         with self.__lock:
 103             previous_state, previous_offsets = self.partitions[(topic, partition)]
 104             if local_offset < previous_offsets.local:
 105                 logger.info(
 106                     "Local offset for %s/%s has moved backwards (current: %s, previous: %s)",
 107                     topic,
 108                     partition,
 109                     local_offset,
 110                     previous_offsets.local,
 111                 )
 112             updated_offsets = Offsets(local_offset, previous_offsets.remote)
 113             updated_state = self.get_state_from_offsets(updated_offsets)
 114             if (
 115                 previous_state is not updated_state
 116                 and updated_state not in self.transitions[previous_state]
 117             ):
 118                 raise InvalidStateTransition(
 119                     u"Unexpected state transition for {}/{} from {} to {}".format(
 120                         topic, partition, previous_state, updated_state
 121                     )
 122                 )
 123             self.partitions[(topic, partition)] = (updated_state, updated_offsets)
 124             if previous_state is not updated_state:
 125                 if updated_state == SynchronizedPartitionState.REMOTE_BEHIND:
 126                     logger.warning(
 127                         "Current local offset for %s/%s (%s) exceeds remote offset (%s)!",
 128                         topic,
 129                         partition,
 130                         updated_offsets.local,
 131                         updated_offsets.remote,
 132                     )
 133                 self.callback(
 134                     topic,
 135                     partition,
 136                     (previous_state, previous_offsets),
 137                     (updated_state, updated_offsets),
 138                 )
 139 
 140     def set_remote_offset(self, topic, partition, remote_offset):
 141         """
 142         Update the remote offset for a topic and partition.
 143 
 144         If this update operation results in a state change, the callback
 145         function will be invoked.
 146         """
 147         with self.__lock:
 148             previous_state, previous_offsets = self.partitions[(topic, partition)]
 149             if remote_offset < previous_offsets.remote:
 150                 logger.info(
 151                     "Remote offset for %s/%s has moved backwards (current: %s, previous: %s)",
 152                     topic,
 153                     partition,
 154                     remote_offset,
 155                     previous_offsets.remote,
 156                 )
 157             updated_offsets = Offsets(previous_offsets.local, remote_offset)
 158             updated_state = self.get_state_from_offsets(updated_offsets)
 159             if (
 160                 previous_state is not updated_state
 161                 and updated_state not in self.transitions[previous_state]
 162             ):
 163                 raise InvalidStateTransition(
 164                     u"Unexpected state transition for {}/{} from {} to {}".format(
 165                         topic, partition, previous_state, updated_state
 166                     )
 167                 )
 168             self.partitions[(topic, partition)] = (updated_state, updated_offsets)
 169             if previous_state is not updated_state:
 170                 self.callback(
 171                     topic,
 172                     partition,
 173                     (previous_state, previous_offsets),
 174                     (updated_state, updated_offsets),
 175                 )

 177     def validate_local_message(self, topic, partition, offset):
 178         """
 179         Check if a message should be consumed by the local consumer.
 180 
 181         The local consumer should be prevented from consuming messages that
 182         have yet to have been committed by the remote consumer.
 183         """
 184         with self.__lock:
 185             state, offsets = self.partitions[(topic, partition)]
 186             if state is not SynchronizedPartitionState.LOCAL_BEHIND:
 187                 raise InvalidState(
 188                     "Received a message while consumer is not in LOCAL_BEHIND state!"
 189                 )
 190             if offset >= offsets.remote:
 191                 raise MessageNotReady(
 192                     "Received a message that has not been committed by remote consumer"
 193                 )
 194             if offset < offsets.local:
 195                 logger.warning(
 196                     "Received a message prior to local offset (local consumer offset rewound without update?)"
 197                 )
