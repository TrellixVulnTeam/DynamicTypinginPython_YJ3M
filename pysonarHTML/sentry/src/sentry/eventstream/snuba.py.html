  21 # Version 1 format: (1, TYPE, [...REST...])
  22 #   Insert: (1, 'insert', {
  23 #       ...event json...
  24 #   }, {
  25 #       ...state for post-processing...
  26 #   })
  27 #
  28 #   Mutations that *should be ignored*: (1, ('delete_groups'|'unmerge'|'merge'), {...})
  29 #
  30 #   In short, for protocol version 1 only messages starting with (1, 'insert', ...)
  31 #   should be processed.
  32 
  33 # Version 2 format: (2, TYPE, [...REST...])
  34 #   Insert: (2, 'insert', {
  35 #       ...event json...
  36 #   }, {
  37 #       ...state for post-processing...
  38 #   })
  39 #   Delete Groups: (2, '(start_delete_groups|end_delete_groups)', {
  40 #       'transaction_id': uuid,
  41 #       'project_id': id,
  42 #       'group_ids': [id2, id2, id3],
  43 #       'datetime': timestamp,
  44 #   })
  45 #   Merge: (2, '(start_merge|end_merge)', {
  46 #       'transaction_id': uuid,
  47 #       'project_id': id,
  48 #       'previous_group_ids': [id2, id2],
  49 #       'new_group_id': id,
  50 #       'datetime': timestamp,
  51 #   })
  52 #   Unmerge: (2, '(start_unmerge|end_unmerge)', {
  53 #       'transaction_id': uuid,
  54 #       'project_id': id,
  55 #       'previous_group_id': id,
  56 #       'new_group_id': id,
  57 #       'hashes': [hash2, hash2]
  58 #       'datetime': timestamp,
  59 #   })
  60 #   Delete Tag: (2, '(start_delete_tag|end_delete_tag)', {
  61 #       'transaction_id': uuid,
  62 #       'project_id': id,
  63 #       'tag': 'foo',
  64 #       'datetime': timestamp,
  65 #   })

  70     # Beware! Changing this protocol (introducing a new version, or the message
  71     # format/fields themselves) requires consideration of all downstream
  72     # consumers. This includes the post-processing forwarder code!
  73     EVENT_PROTOCOL_VERSION = 2
  74 
  75     # These keys correspond to tags that are typically prefixed with `sentry:`
  76     # and will wreak havok in the UI if both the `sentry:`-prefixed and
  77     # non-prefixed variations occur in a response.
  78     UNEXPECTED_TAG_KEYS = frozenset(["dist", "release", "user"])

  80     def insert(
  81         self,
  82         group,
  83         event,
  84         is_new,
  85         is_regression,
  86         is_new_group_environment,
  87         primary_hash,
  88         received_timestamp,  # type: float
  89         skip_consume=False,
  90     ):
  91         project = event.project
  92         set_current_project(project.id)
  93         retention_days = quotas.get_event_retention(organization=project.organization)
  94 
  95         event_data = event.get_raw_data()
  96 
  97         unexpected_tags = set(
  98             [
  99                 k
 100                 for (k, v) in (get_path(event_data, "tags", filter=True) or [])
 101                 if k in self.UNEXPECTED_TAG_KEYS
 102             ]
 103         )
 104         if unexpected_tags:
 105             logger.error("%r received unexpected tags: %r", self, unexpected_tags)
 106 
 107         self._send(
 108             project.id,
 109             "insert",
 110             extra_data=(
 111                 {
 112                     "group_id": event.group_id,
 113                     "event_id": event.event_id,
 114                     "organization_id": project.organization_id,
 115                     "project_id": event.project_id,
 116                     # TODO(mitsuhiko): We do not want to send this incorrect
 117                     # message but this is what snuba needs at the moment.
 118                     "message": event.search_message,
 119                     "platform": event.platform,
 120                     "datetime": event.datetime,
 121                     "data": event_data,
 122                     "primary_hash": primary_hash,
 123                     "retention_days": retention_days,
 124                 },
 125                 {
 126                     "is_new": is_new,
 127                     "is_regression": is_regression,
 128                     "is_new_group_environment": is_new_group_environment,
 129                     "skip_consume": skip_consume,
 130                 },
 131             ),
 132             headers={"Received-Timestamp": six.text_type(received_timestamp)},
 133         )

 135     def start_delete_groups(self, project_id, group_ids):
 136         if not group_ids:
 137             return
 138 
 139         state = {
 140             "transaction_id": uuid4().hex,
 141             "project_id": project_id,
 142             "group_ids": list(group_ids),
 143             "datetime": datetime.now(tz=pytz.utc),
 144         }
 145 
 146         self._send(project_id, "start_delete_groups", extra_data=(state,), asynchronous=False)
 147 
 148         return state
 149 
 150     def end_delete_groups(self, state):
 151         state = state.copy()
 152         state["datetime"] = datetime.now(tz=pytz.utc)
 153         self._send(
 154             state["project_id"], "end_delete_groups", extra_data=(state,), asynchronous=False
 155         )

 231 class SnubaEventStream(SnubaProtocolEventStream):
 232     def _send(
 233         self,
 234         project_id,
 235         _type,
 236         extra_data=(),
 237         asynchronous=True,
 238         headers=None,  # Optional[Mapping[str, str]]
 239     ):
 240         if headers is None:
 241             headers = {}
 242 
 243         data = (self.EVENT_PROTOCOL_VERSION, _type) + extra_data
 244 
 245         # TODO remove this once the unified dataset is available.
 246         # Inserting into both events and transactions datasets lets us
 247         # simulate what is currently happening via kafka when both the events
 248         # and transactions consumers are running.
 249         datasets = ["events"]
 250         if get_path(extra_data, 0, "data", "type") == "transaction":
 251             datasets.append("transactions")
 252         try:
 253             for dataset in datasets:
 254                 resp = snuba._snuba_pool.urlopen(
 255                     "POST",
 256                     "/tests/{}/eventstream".format(dataset),
 257                     body=json.dumps(data),
 258                     headers={"X-Sentry-{}".format(k): v for k, v in headers.items()},
 259                 )
 260                 if resp.status != 200:
 261                     raise snuba.SnubaError("HTTP %s response from Snuba!" % resp.status)
 262             return resp
 263         except urllib3.exceptions.HTTPError as err:
 264             raise snuba.SnubaError(err)
