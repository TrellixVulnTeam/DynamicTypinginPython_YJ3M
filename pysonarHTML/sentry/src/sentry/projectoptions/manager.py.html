  11 class WellKnownProjectOption(object):
  12     def __init__(self, key, default=None, epoch_defaults=None):
  13         self.key = key
  14         self.default = default
  15         self.epoch_defaults = epoch_defaults
  16         self._epoch_default_list = sorted(epoch_defaults or ())
  17 
  18     def get_default(self, project=None, epoch=None):
  19         if self.epoch_defaults:
  20             if epoch is None:
  21                 if project is None:
  22                     epoch = 1
  23                 else:
  24                     epoch = project.get_option("sentry:option-epoch") or 1
  25             idx = bisect.bisect(self._epoch_default_list, epoch)
  26             if idx > 0:
  27                 return self.epoch_defaults[self._epoch_default_list[idx - 1]]
  28         return self.default

  31 class ProjectOptionsManager(object):
  32     """Project options used to be implemented in a relatively ad-hoc manner
  33     in the past.  The project manager still uses the functionality of the
  34     project model and just dispatches to it.
  35 
  36     Options can be used without declaring defaults, but if defaults are
  37     declared they are returned without having to define a default at the
  38     time of the option lookup.
  39     """
  40 
  41     def __init__(self):
  42         self.registry = {}
  43 
  44     def lookup_well_known_key(self, key):
  45         return self.registry.get(key)
  46 
  47     def freeze_option_epoch(self, project, force=False):
  48         # The options are frozen in a receiver hook for project saves.
  49         # See `sentry.receivers.core.freeze_option_epoch_for_project`
  50         if force or project.get_option("sentry:option-epoch") is None:
  51             from .defaults import LATEST_EPOCH
  52 
  53             project.update_option("sentry:option-epoch", LATEST_EPOCH)
