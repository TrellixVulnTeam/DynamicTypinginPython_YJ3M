  49     """
  50 
  51     def first_lower(s):
  52         return s[:1].lower() + s[1:]
  53 
  54     def first_upper(s):
  55         return s[:1].upper() + s[1:]
  56 
  57     if not isinstance(name, six.string_types):
  58         return name
  59     else:
  60         name = name.strip("_")
  61         pieces = name.split("_")
