  81 def get_project_config(project, org_options=None, full_config=True, project_keys=None):
  82     """
  83     Constructs the ProjectConfig information.
  84 
  85     :param project: The project to load configuration for. Ensure that
  86         organization is bound on this object; otherwise it will be loaded from
  87         the database.
  88     :param org_options: Inject preloaded organization options for faster loading.
  89         If ``None``, options are lazy-loaded from the database.
  90     :param full_config: True if only the full config is required, False
  91         if only the restricted (for external relays) is required
  92         (default True, i.e. full configuration)
  93     :param project_keys: Pre-fetched project keys for performance, similar to
  94         org_options. However, if no project keys are provided it is assumed
  95         that the config does not need to contain auth information (this is the
  96         case when used in python's StoreView)
  97 
  98     :return: a ProjectConfig object for the given project
  99     """
 100     with configure_scope() as scope:
 101         scope.set_tag("project", project.id)
 102 
 103     if project.status != ObjectStatus.VISIBLE:
 104         return ProjectConfig(project, disabled=True)
 105 
 106     public_keys = get_public_key_configs(project, full_config, project_keys=project_keys)
 107 
 108     if org_options is None:
 109         org_options = OrganizationOption.objects.get_all_values(project.organization_id)
 110 
 111     with Hub.current.start_span(op="get_public_config"):
 112         now = datetime.utcnow().replace(tzinfo=utc)
 113         cfg = {
 114             "disabled": False,
 115             "slug": project.slug,
 116             "lastFetch": now,
 117             "lastChange": project.get_option("sentry:relay-rev-lastchange", now),
 118             "rev": project.get_option("sentry:relay-rev", uuid.uuid4().hex),
 119             "publicKeys": public_keys,
 120             "config": {
 121                 "allowedDomains": list(get_origins(project)),
 122                 "trustedRelays": org_options.get("sentry:trusted-relays", []),
 123                 "piiConfig": _get_pii_config(project),
 124                 "datascrubbingSettings": _get_datascrubbing_settings(project, org_options),
 125             },
 126             "organizationId": project.organization_id,
 127             "projectId": project.id,  # XXX: Unused by Relay, required by Python store
 128         }
 129 
 130     if not full_config:
 131         # This is all we need for external Relay processors
 132         return ProjectConfig(project, **cfg)
 133 
 134     with Hub.current.start_span(op="get_filter_settings"):
 135         cfg["config"]["filterSettings"] = get_filter_settings(project)
 136     with Hub.current.start_span(op="get_grouping_config_dict_for_project"):
 137         cfg["config"]["groupingConfig"] = get_grouping_config_dict_for_project(project)
 138     with Hub.current.start_span(op="get_event_retention"):
 139         cfg["config"]["eventRetention"] = quotas.get_event_retention(project.organization)
 140     with Hub.current.start_span(op="get_all_quotas"):
 141         cfg["config"]["quotas"] = get_quotas(project, keys=project_keys)
 142 
 143     return ProjectConfig(project, **cfg)

 146 class _ConfigBase(object):
 147     """
 148     Base class for configuration objects
 149 
 150     Offers a readonly configuration class that can be serialized to json and viewed as a simple dictionary
 151 
 152     >>> x = _ConfigBase( a= 1, b="The b", c= _ConfigBase(x=33, y = _ConfigBase(m=3.14159 , w=[1,2,3], z={'t':1})))
 153     >>> x.a
 154     1
 155     >>> x.b
 156     'The b'
 157     >>> x.something is None # accessing non-existing elements
 158     True
 159     >>> x.c.y.w
 160     [1, 2, 3]
 161 
 162     """
 163 
 164     def __init__(self, **kwargs):
 165         data = {}
 166         object.__setattr__(self, "data", data)
 167         for (key, val) in six.iteritems(kwargs):
 168             if val is not None:
 169                 data[key] = val
 170 
 171     def __setattr__(self, key, value):
 172         raise Exception("Trying to change read only ProjectConfig object")
 173 
 174     def __getattr__(self, name):
 175         data = self.__get_data()
 176         return data.get(to_camel_case_name(name))
 177 
 178     def to_dict(self):
 179         """
 180         Converts the config object into a dictionary
 181 
 182         :return: A dictionary containing the object properties, with config properties also converted in dictionaries
 183 
 184         >>> x = _ConfigBase( a= 1, b="The b", c= _ConfigBase(x=33, y = _ConfigBase(m=3.14159 , w=[1,2,3], z={'t':1})))
 185         >>> x.to_dict() == {'a': 1, 'c': {'y': {'m': 3.14159, 'w': [1, 2, 3], 'z':{'t': 1}}, 'x': 33}, 'b': 'The b'}
 186         True
 187         """
 188         data = self.__get_data()
 189         return {
 190             key: value.to_dict() if isinstance(value, _ConfigBase) else value
 191             for (key, value) in six.iteritems(data)
 192         }
 193 
 194     def to_json_string(self):
 195         """
 196         >>> x = _ConfigBase( a = _ConfigBase(b = _ConfigBase( w=[1,2,3])))
 197         >>> x.to_json_string()
 198         '{"a": {"b": {"w": [1, 2, 3]}}}'
 199 
 200         :return:
 201         """
 202         data = self.to_dict()
 203         return utils.json.dumps(data)
 204 
 205     def get_at_path(self, *args):
 206         """
 207         Gets an element at the specified path returning None if the element or the path doesn't exists
 208 
 209         :param args: the path to follow ( a list of strings)
 210         :return: the element if present at specified path or None otherwise)
 211 
 212         >>> x = _ConfigBase( a= 1, b="The b", c= _ConfigBase(x=33, y = _ConfigBase(m=3.14159 , w=[1,2,3], z={'t':1})))
 213         >>> x.get_at_path('c','y','m')
 214         3.14159
 215         >>> x.get_at_path('bb') is None # property not set
 216         True
 217         >>> x.get_at_path('a', 'something') is None # trying to go past existing Config paths
 218         True
 219         >>> x.get_at_path('c','y','z')
 220         {'t': 1}
 221         >>> x.get_at_path('c','y','z','t') is None # only navigates in ConfigBase does not try to go into normal dicts.
 222         True
 223 
 224         """
 225         if len(args) == 0:
 226             return self
 227 
 228         data = self.__get_data()
 229         val = data.get(args[0])
 230 
 231         if len(args) == 1:
 232             return val
 233 
 234         if isinstance(val, _ConfigBase):
 235             return val.get_at_path(*args[1:])
 236 
 237         return None  # property not set or path goes beyond the Config defined valid path
 238 
 239     def __get_data(self):
 240         return object.__getattribute__(self, "data")
 241 
 242     def __str__(self):
 243         try:
 244             return utils.json.dumps(self.to_dict(), sort_keys=True)
 245         except Exception as e:
 246             return "Content Error:{}".format(e)
 247 
 248     def __repr__(self):
 249         return "({0}){1}".format(self.__class__.__name__, self)

 263 def _get_pii_config(project):
 264     def _decode(value):
 265         if value:
 266             return safe_execute(utils.json.loads, value)
 267 
 268     # Order of merging is important here. We want to apply organization rules
 269     # before project rules. For example:
 270     #
 271     # * Organization rule: remove substrings "mypassword"
 272     # * Project rule: remove substrings "my"
 273     #
 274     # If we were to apply project rules before organization rules, "password"
 275     # would leak. We effectively disabled an organization rule using a project rule.
 276     #
 277     # Of course organization rules can also break project rules the same way,
 278     # but we communicate in the UI that organization options take precedence
 279     # here.
 280     return merge_pii_configs(
 281         [
 282             ("organization:", _decode(project.organization.get_option("sentry:relay_pii_config"))),
 283             ("project:", _decode(project.get_option("sentry:relay_pii_config"))),
 284         ]
 285     )

 288 def _get_datascrubbing_settings(project, org_options):
 289     rv = {}
 290 
 291     exclude_fields_key = "sentry:safe_fields"
 292     rv["excludeFields"] = org_options.get(exclude_fields_key, []) + project.get_option(
 293         exclude_fields_key, []
 294     )
 295 
 296     rv["scrubData"] = org_options.get("sentry:require_scrub_data", False) or project.get_option(
 297         "sentry:scrub_data", True
 298     )
 299 
 300     rv["scrubIpAddresses"] = org_options.get(
 301         "sentry:require_scrub_ip_address", False
 302     ) or project.get_option("sentry:scrub_ip_address", False)
 303 
 304     sensitive_fields_key = "sentry:sensitive_fields"
 305     rv["sensitiveFields"] = org_options.get(sensitive_fields_key, []) + project.get_option(
 306         sensitive_fields_key, []
 307     )
 308 
 309     rv["scrubDefaults"] = org_options.get(
 310         "sentry:require_scrub_defaults", False
 311     ) or project.get_option("sentry:scrub_defaults", True)
 312 
 313     return rv

 316 def _load_filter_settings(flt, project):
 317     """
 318     Returns the filter settings for the specified project
 319 
 320     :param flt: the filter function
 321     :param project: the project for which we want to retrieve the options
 322     :return: a dictionary with the filter options.
 323         If the project does not explicitly specify the filter options then the
 324         default options for the filter will be returned
 325     """
 326     filter_id = flt.spec.id
 327     filter_key = u"filters:{}".format(filter_id)
 328     setting = project.get_option(filter_key)
 329 
 330     return _filter_option_to_config_setting(flt, setting)

 333 def _filter_option_to_config_setting(flt, setting):
 334     """
 335     Encapsulates the logic for associating a filter database option with the filter setting from project_config
 336 
 337     :param flt: the filter
 338     :param setting: the option deserialized from the database
 339     :return: the option as viewed from project_config
 340     """
 341     if setting is None:
 342         raise ValueError(
 343             "Could not find filter state for filter {0}."
 344             " You need to register default filter state in projectoptions.defaults.".format(
 345                 flt.spec.id
 346             )
 347         )
 348 
 349     is_enabled = setting != "0"
 350 
 351     ret_val = {"isEnabled": is_enabled}
 352 
 353     # special case for legacy browser.
 354     # If the number of special cases increases we'll have to factor this functionality somewhere
 355     if flt.spec.id == FilterStatKeys.LEGACY_BROWSER:
 356         if is_enabled:
 357             if setting == "1":
 358                 ret_val["options"] = ["default"]
 359             else:
 360                 # new style filter, per legacy browser type handling
 361                 # ret_val['options'] = setting.split(' ')
 362                 ret_val["options"] = list(setting)
 363     return ret_val
