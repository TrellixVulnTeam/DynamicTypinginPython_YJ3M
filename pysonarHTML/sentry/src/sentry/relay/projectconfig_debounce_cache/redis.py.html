  35     def check_is_debounced(self, project_id, organization_id):
  36         key = _get_redis_key(project_id, organization_id)
  37         client = self.__get_redis_client(key)
  38         if client.get(key):
  39             return True
  40 
  41         client.setex(key, REDIS_CACHE_TIMEOUT, 1)
  42         return False

  44     def mark_task_done(self, project_id, organization_id):
