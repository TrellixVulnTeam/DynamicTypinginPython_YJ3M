  13 class RedisProjectConfigCache(ProjectConfigCache):
  14     def __init__(self, **options):
  15         self.is_redis_cluster, self.cluster, options = get_dynamic_cluster_from_options(
  16             "SENTRY_RELAY_PROJECTCONFIG_CACHE_OPTIONS", options
  17         )
  18         super(RedisProjectConfigCache, self).__init__(**options)
  19 
  20     def validate(self):
  21         validate_dynamic_cluster(self.is_redis_cluster, self.cluster)
  22 
  23     def __get_redis_key(self, project_id):
  24         return "relayconfig:%s" % (project_id,)
  25 
  26     def __get_redis_client(self, routing_key):
  27         if self.is_redis_cluster:
  28             return self.cluster
  29         else:
  30             return self.cluster.get_local_client_for_key(routing_key)
  31 
  32     def set_many(self, configs):
  33         for project_id, config in six.iteritems(configs):
  34             # XXX(markus): Figure out how to do pipelining here. We may have
  35             # multiple routing keys (-> multiple clients).
  36             #
  37             # We cannot route by org, because Relay does not know the org when
  38             # fetching.
  39 
  40             key = self.__get_redis_key(project_id)
  41             client = self.__get_redis_client(key)
  42             client.setex(key, REDIS_CACHE_TIMEOUT, json.dumps(config))
  43 
  44     def delete_many(self, project_ids):
  45         for project_id in project_ids:
  46             # XXX(markus): Figure out how to do pipelining here. We may have
  47             # multiple routing keys (-> multiple clients).
  48             #
  49             # We cannot route by org, because Relay does not know the org when
  50             # fetching.
  51 
  52             key = self.__get_redis_key(project_id)
  53             client = self.__get_redis_client(key)
  54             client.delete(key)
  55 
  56     def get(self, project_id):
  57         key = self.__get_redis_key(project_id)
  58         client = self.__get_redis_client(key)
  59         rv = client.get(key)
  60         if rv is not None:
  61             return json.loads(rv)
  62         return None
