  34 def try_mark_onboarding_complete(organization_id):
  35     if OrganizationOption.objects.filter(
  36         organization_id=organization_id, key="onboarding:complete"
  37     ).exists():
  38         return
  39 
  40     completed = set(
  41         OrganizationOnboardingTask.objects.filter(
  42             Q(organization_id=organization_id)
  43             & (Q(status=OnboardingTaskStatus.COMPLETE) | Q(status=OnboardingTaskStatus.SKIPPED))
  44         ).values_list("task", flat=True)
  45     )
  46     if completed >= OrganizationOnboardingTask.REQUIRED_ONBOARDING_TASKS:
  47         try:
  48             with transaction.atomic():
  49                 OrganizationOption.objects.create(
  50                     organization_id=organization_id,
  51                     key="onboarding:complete",
  52                     value={"updated": timezone.now()},
  53                 )
  54         except IntegrityError:
  55             pass

  58 @project_created.connect(weak=False)
  59 def record_new_project(project, user, **kwargs):
  60     if user.is_authenticated():
  61         user_id = default_user_id = user.id
  62     else:
  63         user = user_id = None
  64         try:
  65             default_user_id = (
  66                 Organization.objects.get(id=project.organization_id).get_default_owner().id
  67             )
  68         except IndexError:
  69             logging.getLogger("sentry").warn(
  70                 "Cannot initiate onboarding for organization (%s) due to missing owners",
  71                 project.organization_id,
  72             )
  73             # XXX(dcramer): we cannot setup onboarding tasks without a user
  74             return
  75 
  76     analytics.record(
  77         "project.created",
  78         user_id=user_id,
  79         default_user_id=default_user_id,
  80         organization_id=project.organization_id,
  81         project_id=project.id,
  82         platform=project.platform,
  83     )
  84 
  85     success = OrganizationOnboardingTask.objects.record(
  86         organization_id=project.organization_id,
  87         task=OnboardingTask.FIRST_PROJECT,
  88         user=user,
  89         status=OnboardingTaskStatus.COMPLETE,
  90         project_id=project.id,
  91     )
  92     if not success:
  93         OrganizationOnboardingTask.objects.record(
  94             organization_id=project.organization_id,
  95             task=OnboardingTask.SECOND_PLATFORM,
  96             user=user,
  97             status=OnboardingTaskStatus.PENDING,
  98             project_id=project.id,
  99         )

 113 @first_event_received.connect(weak=False)
 114 def record_first_event(project, event, **kwargs):
 115     """
 116     Requires up to 2 database calls, but should only run with the first event in
 117     any project, so performance should not be a huge bottleneck.
 118     """
 119 
 120     # If complete, pass (creation fails due to organization, task unique constraint)
 121     # If pending, update.
 122     # If does not exist, create.
 123     rows_affected, created = OrganizationOnboardingTask.objects.create_or_update(
 124         organization_id=project.organization_id,
 125         task=OnboardingTask.FIRST_EVENT,
 126         status=OnboardingTaskStatus.PENDING,
 127         values={
 128             "status": OnboardingTaskStatus.COMPLETE,
 129             "project_id": project.id,
 130             "date_completed": project.first_event,
 131             "data": {"platform": event.platform},
 132         },
 133     )
 134 
 135     user = Organization.objects.get(id=project.organization_id).get_default_owner()
 136 
 137     if rows_affected or created:
 138         analytics.record(
 139             "first_event.sent",
 140             user_id=user.id,
 141             organization_id=project.organization_id,
 142             project_id=project.id,
 143             platform=event.platform,
 144         )
 145         return
 146 
 147     try:
 148         oot = OrganizationOnboardingTask.objects.filter(
 149             organization_id=project.organization_id, task=OnboardingTask.FIRST_EVENT
 150         )[0]
 151     except IndexError:
 152         return
 153 
 154     # Only counts if it's a new project and platform
 155     if oot.project_id != project.id and oot.data.get("platform", event.platform) != event.platform:
 156         rows_affected, created = OrganizationOnboardingTask.objects.create_or_update(
 157             organization_id=project.organization_id,
 158             task=OnboardingTask.SECOND_PLATFORM,
 159             status=OnboardingTaskStatus.PENDING,
 160             values={
 161                 "status": OnboardingTaskStatus.COMPLETE,
 162                 "project_id": project.id,
 163                 "date_completed": project.first_event,
 164                 "data": {"platform": event.platform},
 165             },
 166         )
 167         if rows_affected or created:
 168             analytics.record(
 169                 "second_platform.added",
 170                 user_id=user.id,
 171                 organization_id=project.organization_id,
 172                 project_id=project.id,
 173                 platform=event.platform,
 174             )

 233 @event_processed.connect(weak=False)
 234 def record_user_context_received(project, event, **kwargs):
 235     user_context = event.data.get("user")
 236     if not user_context:
 237         return
 238     # checking to see if only ip address is being sent (our js library does this automatically)
 239     # testing for this in test_no_user_tracking_for_ip_address_only
 240     # list(d.keys()) pattern is to make this python3 safe
 241     elif list(user_context.keys()) != ["ip_address"]:
 242         success = OrganizationOnboardingTask.objects.record(
 243             organization_id=project.organization_id,
 244             task=OnboardingTask.USER_CONTEXT,
 245             status=OnboardingTaskStatus.COMPLETE,
 246             project_id=project.id,
 247         )
 248         if success:
 249             user = Organization.objects.get(id=project.organization_id).get_default_owner()
 250             analytics.record(
 251                 "first_user_context.sent",
 252                 user_id=user.id,
 253                 organization_id=project.organization_id,
 254                 project_id=project.id,
 255             )
 256             try_mark_onboarding_complete(project.organization_id)

 281 @plugin_enabled.connect(weak=False)
 282 def record_plugin_enabled(plugin, project, user, **kwargs):
 283     if isinstance(plugin, IssueTrackingPlugin) or isinstance(plugin, IssueTrackingPlugin2):
 284         task = OnboardingTask.ISSUE_TRACKER
 285         status = OnboardingTaskStatus.PENDING
 286     elif isinstance(plugin, NotificationPlugin):
 287         task = OnboardingTask.ALERT_RULE
 288         status = OnboardingTaskStatus.COMPLETE
 289     else:
 290         return
 291 
 292     success = OrganizationOnboardingTask.objects.record(
 293         organization_id=project.organization_id,
 294         task=task,
 295         status=status,
 296         user=user,
 297         project_id=project.id,
 298         data={"plugin": plugin.slug},
 299     )
 300     if success:
 301         try_mark_onboarding_complete(project.organization_id)
 302 
 303     analytics.record(
 304         "plugin.enabled",
 305         user_id=user.id,
 306         organization_id=project.organization_id,
 307         project_id=project.id,
 308         plugin=plugin.slug,
 309     )

 329 @issue_tracker_used.connect(weak=False)
 330 def record_issue_tracker_used(plugin, project, user, **kwargs):
 331     rows_affected, created = OrganizationOnboardingTask.objects.create_or_update(
 332         organization_id=project.organization_id,
 333         task=OnboardingTask.ISSUE_TRACKER,
 334         status=OnboardingTaskStatus.PENDING,

 336             "status": OnboardingTaskStatus.COMPLETE,
