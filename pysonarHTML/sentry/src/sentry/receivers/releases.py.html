  33 def remove_resolved_link(link):
  34     # TODO(dcramer): ideally this would simply "undo" the link change,
  35     # but we don't know for a fact that the resolution was most recently from
  36     # the GroupLink
  37     with transaction.atomic():
  38         link.delete()
  39         affected = Group.objects.filter(status=GroupStatus.RESOLVED, id=link.group_id).update(
  40             status=GroupStatus.UNRESOLVED
  41         )
  42         if affected:
  43             Activity.objects.create(
  44                 project_id=link.project_id,
  45                 group_id=link.group_id,
  46                 type=Activity.SET_UNRESOLVED,
  47                 ident=link.group_id,
  48             )

  51 def resolved_in_commit(instance, created, **kwargs):
  52     current_datetime = timezone.now()
  53 
  54     groups = instance.find_referenced_groups()
  55 
  56     # Delete GroupLinks where message may have changed
  57     group_ids = {g.id for g in groups}
  58     group_links = GroupLink.objects.filter(
  59         linked_type=GroupLink.LinkedType.commit,
  60         relationship=GroupLink.Relationship.resolves,
  61         linked_id=instance.id,
  62     )
  63     for link in group_links:
  64         if link.group_id not in group_ids:
  65             remove_resolved_link(link)
  66 
  67     try:
  68         repo = Repository.objects.get(id=instance.repository_id)
  69     except Repository.DoesNotExist:
  70         repo = None
  71 
  72     for group in groups:
  73         try:
  74             # XXX(dcramer): This code is somewhat duplicated from the
  75             # project_group_index mutation api
  76             with transaction.atomic():
  77                 GroupLink.objects.create(
  78                     group_id=group.id,
  79                     project_id=group.project_id,
  80                     linked_type=GroupLink.LinkedType.commit,
  81                     relationship=GroupLink.Relationship.resolves,
  82                     linked_id=instance.id,
  83                 )
  84 
  85                 if instance.author:
  86                     user_list = list(instance.author.find_users())
  87                 else:
  88                     user_list = ()
  89 
  90                 if user_list:
  91                     acting_user = user_list[0]
  92                     Activity.objects.create(
  93                         project_id=group.project_id,
  94                         group=group,
  95                         type=Activity.SET_RESOLVED_IN_COMMIT,
  96                         ident=instance.id,
  97                         user=acting_user,
  98                         data={"commit": instance.id},
  99                     )
 100                     self_assign_issue = UserOption.objects.get_value(
 101                         user=acting_user, key="self_assign_issue", default="0"
 102                     )
 103                     if self_assign_issue == "1" and not group.assignee_set.exists():
 104                         GroupAssignee.objects.assign(
 105                             group=group, assigned_to=acting_user, acting_user=acting_user
 106                         )
 107 
 108                     # while we only create activity and assignment for one user we want to
 109                     # subscribe every user
 110                     for user in user_list:
 111                         GroupSubscription.objects.subscribe(
 112                             user=user, group=group, reason=GroupSubscriptionReason.status_change
 113                         )
 114 
 115                 else:
 116                     Activity.objects.create(
 117                         project_id=group.project_id,
 118                         group=group,
 119                         type=Activity.SET_RESOLVED_IN_COMMIT,
 120                         ident=instance.id,
 121                         data={"commit": instance.id},
 122                     )
 123                 Group.objects.filter(id=group.id).update(
 124                     status=GroupStatus.RESOLVED, resolved_at=current_datetime
 125                 )
 126         except IntegrityError:
 127             pass
 128         else:
 129             if repo is not None:
 130                 if repo.integration_id is not None:
 131                     analytics.record(
 132                         "integration.resolve.commit",
 133                         provider=repo.provider,
 134                         id=repo.integration_id,
 135                         organization_id=repo.organization_id,
 136                     )
 137                 user = user_list[0] if user_list else None
 138 
 139                 issue_resolved.send_robust(
 140                     organization_id=repo.organization_id,
 141                     user=user,
 142                     group=group,
 143                     project=group.project,
 144                     resolution_type="with_commit",
 145                     sender="resolved_with_commit",
 146                 )

 149 def resolved_in_pull_request(instance, created, **kwargs):
 150     groups = instance.find_referenced_groups()
 151 
 152     # Delete GroupLinks where message may have changed
 153     group_ids = {g.id for g in groups}
 154     group_links = GroupLink.objects.filter(
 155         linked_type=GroupLink.LinkedType.pull_request,
 156         relationship=GroupLink.Relationship.resolves,
 157         linked_id=instance.id,
 158     )
 159     for link in group_links:
 160         if link.group_id not in group_ids:
 161             remove_resolved_link(link)
 162 
 163     try:
 164         repo = Repository.objects.get(id=instance.repository_id)
 165     except Repository.DoesNotExist:
 166         repo = None
 167 
 168     for group in groups:
 169         try:
 170             with transaction.atomic():
 171                 GroupLink.objects.create(
 172                     group_id=group.id,
 173                     project_id=group.project_id,
 174                     linked_type=GroupLink.LinkedType.pull_request,
 175                     relationship=GroupLink.Relationship.resolves,
 176                     linked_id=instance.id,
 177                 )
 178 
 179                 if instance.author:
 180                     user_list = list(instance.author.find_users())
 181                 else:
 182                     user_list = ()
 183                 if user_list:
 184                     Activity.objects.create(
 185                         project_id=group.project_id,
 186                         group=group,
 187                         type=Activity.SET_RESOLVED_IN_PULL_REQUEST,
 188                         ident=instance.id,
 189                         user=user_list[0],
 190                         data={"pull_request": instance.id},
 191                     )
 192                     GroupAssignee.objects.assign(
 193                         group=group, assigned_to=user_list[0], acting_user=user_list[0]
 194                     )
 195                 else:
 196                     Activity.objects.create(
 197                         project_id=group.project_id,
 198                         group=group,
 199                         type=Activity.SET_RESOLVED_IN_PULL_REQUEST,
 200                         ident=instance.id,
 201                         data={"pull_request": instance.id},
 202                     )
 203         except IntegrityError:
 204             pass
 205         else:
 206             if repo is not None and repo.integration_id is not None:
 207                 analytics.record(
 208                     "integration.resolve.pr",
 209                     provider=repo.provider,
 210                     id=repo.integration_id,
 211                     organization_id=repo.organization_id,
 212                 )
