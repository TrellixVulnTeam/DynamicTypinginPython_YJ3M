  84 @event_processed.connect(weak=False)
  85 def record_event_processed(project, event, **kwargs):
  86     feature_slugs = []
  87 
  88     platform = event.group.platform if event.group else event.platform
  89 
  90     # Platform
  91     if platform in manager.location_slugs("language"):
  92         feature_slugs.append(platform)
  93 
  94     # Release Tracking
  95     if event.get_tag("sentry:release"):
  96         feature_slugs.append("release_tracking")
  97 
  98     # Environment Tracking
  99     if event.get_tag("environment"):
 100         feature_slugs.append("environment_tracking")
 101 
 102     # User Tracking
 103     user_context = event.data.get("user")
 104     # We'd like them to tag with id or email.
 105     # Certain SDKs automatically tag with ip address.
 106     # Check to make sure more the ip address is being sent.
 107     # testing for this in test_no_user_tracking_for_ip_address_only
 108     # list(d.keys()) pattern is to make this python3 safe
 109     if user_context and list(user_context.keys()) != ["ip_address"]:
 110         feature_slugs.append("user_tracking")
 111 
 112     # Custom Tags
 113     if set(tag[0] for tag in event.tags) - DEFAULT_TAGS:
 114         feature_slugs.append("custom_tags")
 115 
 116     # Sourcemaps
 117     if has_sourcemap(event):
 118         feature_slugs.append("source_maps")
 119 
 120     # Breadcrumbs
 121     if event.data.get("breadcrumbs"):
 122         feature_slugs.append("breadcrumbs")
 123 
 124     if not feature_slugs:
 125         return
 126 
 127     FeatureAdoption.objects.bulk_record(project.organization_id, feature_slugs)

 172 @issue_resolved.connect(weak=False)
 173 def record_issue_resolved(organization_id, project, group, user, resolution_type, **kwargs):
 174     """ There are three main types of ways to resolve issues
 175         1) via a release (current release, next release, or other)
 176         2) via commit (in the UI with the commit hash (marked as "in_commit")
 177             or tagging the issue in a commit (marked as "with_commit"))
 178         3) now
 179     """
 180     if resolution_type in ("in_next_release", "in_release"):
 181         FeatureAdoption.objects.record(
 182             organization_id=organization_id, feature_slug="resolved_in_release", complete=True
 183         )
 184     if resolution_type == "with_commit":
 185         FeatureAdoption.objects.record(
 186             organization_id=organization_id, feature_slug="resolved_with_commit", complete=True
 187         )
 188 
 189     if user and user.is_authenticated():
 190         user_id = default_user_id = user.id
 191     else:
 192         user_id = None
 193         default_user_id = project.organization.get_default_owner().id
 194 
 195     analytics.record(
 196         "issue.resolved",
 197         user_id=user_id,
 198         default_user_id=default_user_id,
 199         organization_id=organization_id,
 200         group_id=group.id,
 201         resolution_type=resolution_type,
 202     )
