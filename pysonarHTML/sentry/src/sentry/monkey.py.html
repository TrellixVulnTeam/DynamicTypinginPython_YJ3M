   4 def register_scheme(name):
   5     try:
   6         import urlparse  # NOQA

  19 def patch_httprequest_repr():
  20     try:
  21         from django.http import HttpRequest
  22     except ImportError:
  23         # This module is potentially imported before Django is installed
  24         # during a setup.py run
  25         return
  26 
  27     # Intentionally strip all GET/POST/COOKIE values out of repr() for HttpRequest
  28     # and subclass WSGIRequest. This prevents sensitive information from getting
  29     # logged. This was yanked out of Django master anyhow.
  30     # https://code.djangoproject.com/ticket/12098
  31     def safe_httprequest_repr(self):
  32         return "<%s: %s %r>" % (self.__class__.__name__, self.method, self.get_full_path())
  33 
  34     HttpRequest.__repr__ = safe_httprequest_repr

  37 def patch_parse_cookie():
  38     try:
  39         from django.utils import six
  40         from django.utils.encoding import force_str
  41         from django.utils.six.moves import http_cookies
  42         from django import http
  43     except ImportError:
  44         # This module is potentially imported before Django is installed
  45         # during a setup.py run
  46         return
  47 
  48     # Backported from 1.8.15: https://github.com/django/django/blob/1.8.15/django/http/cookie.py#L91
  49     # See https://www.djangoproject.com/weblog/2016/sep/26/security-releases/ for more context.
  50     def safe_parse_cookie(cookie):
  51         """
  52         Return a dictionary parsed from a `Cookie:` header string.
  53         """
  54         cookiedict = {}
  55         if six.PY2:
  56             cookie = force_str(cookie)
  57         for chunk in cookie.split(";"):
  58             if "=" in chunk:
  59                 key, val = chunk.split("=", 1)
  60             else:
  61                 # Assume an empty name per
  62                 # https://bugzilla.mozilla.org/show_bug.cgi?id=169091
  63                 key, val = "", chunk
  64             key, val = key.strip(), val.strip()
  65             if key or val:
  66                 # unquote using Python's algorithm.
  67                 cookiedict[key] = http_cookies._unquote(val)
  68         return cookiedict
  69 
  70     http.parse_cookie = safe_parse_cookie

  73 def patch_django_views_debug():
  74     # Prevent exposing any Django SETTINGS on our debug error page
  75     # This information is not useful for Sentry development
  76     # and poses a significant security risk if this is exposed by accident
  77     # in any production system if, by change, it were deployed
  78     # with DEBUG=True.
  79     try:
  80         from django.views import debug
  81     except ImportError:
  82         return
  83 
  84     debug.get_safe_settings = lambda: {}
