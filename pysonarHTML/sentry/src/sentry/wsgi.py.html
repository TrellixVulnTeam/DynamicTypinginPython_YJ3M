  27 class FileWrapperWSGIHandler(WSGIHandler):
  28     """A WSGIHandler implementation that handles a StreamingHttpResponse
  29     from django to leverage wsgi.file_wrapper for delivering large streaming
  30     responses.
  31 
  32     Note: this was added natively into Django 1.8, so if by some reason,
  33     we upgraded, this wouldn't be relevant anymore."""
  34 
  35     def __call__(self, environ, start_response):
  36         response = super(FileWrapperWSGIHandler, self).__call__(environ, start_response)
  37         if hasattr(response, "streaming") and response.streaming:
  38             try:
  39                 response = environ["wsgi.file_wrapper"](response.streaming_content)
  40             except KeyError:
  41                 # In our case, we're shipping with uwsgi, so it's safer to assume
  42                 # that wsgi.file_wrapper does exist. It'd be exceptional otherwise.
  43                 pass
  44         return response
  45
