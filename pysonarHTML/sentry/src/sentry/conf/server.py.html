  29 def env(key, default="", type=None):
  30     """
  31     Extract an environment variable for use in configuration
  32 
  33     :param key: The environment variable to be extracted.
  34     :param default: The value to be returned if `key` is not found.
  35     :param type: The type of the returned object (defaults to the type of `default`).
  36     :return: The environment variable if it exists, else `default`.
  37     """
  38 
  39     # First check an internal cache, so we can `pop` multiple times
  40     # without actually losing the value.
  41     try:
  42         rv = env._cache[key]
  43     except KeyError:
  44         if "SENTRY_RUNNING_UWSGI" in os.environ:
  45             # We do this so when the process forks off into uwsgi
  46             # we want to actually be popping off values. This is so that
  47             # at runtime, the variables aren't actually available.
  48             fn = os.environ.pop
  49         else:
  50             fn = os.environ.__getitem__
  51 
  52         try:
  53             rv = fn(key)
  54             env._cache[key] = rv
  55         except KeyError:
  56             rv = default
  57 
  58     if type is None:
  59         type = type_from_value(default)
  60 
  61     return type(rv)
  62
