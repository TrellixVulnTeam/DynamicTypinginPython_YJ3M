  10 @register.tag
  11 def feature(parser, token):
  12     bits = token.split_contents()
  13     if len(bits) < 2:
  14         raise template.TemplateSyntaxError(
  15             "%r tag requires an argument" % token.contents.split()[0]
  16         )
  17 
  18     name = bits[1]
  19     params = bits[2:]
  20 
  21     nodelist_true = parser.parse(("else", "endfeature"))
  22     token = parser.next_token()
  23 
  24     if token.contents == "else":
  25         nodelist_false = parser.parse(("endfeature",))
  26         parser.delete_first_token()
  27     else:
  28         nodelist_false = template.NodeList()
  29 
  30     return FeatureNode(nodelist_true, nodelist_false, name, params)
  31 
  32 
  33 class FeatureNode(template.Node):
  34     def __init__(self, nodelist_true, nodelist_false, name, params):
  35         self.nodelist_true = nodelist_true
  36         self.nodelist_false = nodelist_false
  37         self.name = name
  38         self.params = [template.Variable(i) for i in params]
  39 
  40     def render(self, context):
  41         params = [i.resolve(context) for i in self.params]
  42         if "request" in context:
  43             user = context["request"].user
  44         else:
  45             user = None
  46 
  47         if not features.has(self.name, actor=user, *params):
  48             return self.nodelist_false.render(context)
  49 
  50         return self.nodelist_true.render(context)
