  16 @register.simple_tag
  17 def absolute_asset_url(module, path):
  18     """
  19     Returns a versioned absolute asset URL (located within Sentry's static files).
  20 
  21     Example:
  22       {% absolute_asset_url 'sentry' 'dist/sentry.css' %}
  23       =>  "http://sentry.example.com/_static/74d127b78dc7daf2c51f/sentry/dist/sentry.css"
  24     """
  25     return absolute_uri(get_asset_url(module, path))

  28 @register.simple_tag
  29 def crossorigin():
  30     """
  31     Returns an additional crossorigin="anonymous" snippet for use in a <script> tag if
  32     our asset urls are from a different domain than the system.url-prefix.
  33     """
  34     if absolute_uri(settings.STATIC_URL).startswith(options.get("system.url-prefix")):
  35         # They share the same domain prefix, so we don't need CORS
  36         return ""
  37     return ' crossorigin="anonymous"'

  40 @register.simple_tag(takes_context=True)
  41 def locale_js_include(context):
  42     """
  43     If the user has a non-English locale set, returns a <script> tag pointing
  44     to the relevant locale JavaScript file
  45     """
  46     request = context["request"]
  47 
  48     try:
  49         lang_code = request.LANGUAGE_CODE
  50     except AttributeError:
  51         # it's possible that request at this point, LANGUAGE_CODE hasn't be bound
  52         # to the Request object yet. This specifically happens when rendering our own
  53         # 500 error page, resulting in yet another error trying to render our error.
  54         return ""
  55 
  56     if lang_code == "en" or lang_code not in settings.SUPPORTED_LANGUAGES:
  57         return ""
  58 
  59     href = get_asset_url("sentry", "dist/locale/" + lang_code + ".js")
  60     return mark_safe('<script src="{0}"{1}></script>'.format(href, crossorigin()))
