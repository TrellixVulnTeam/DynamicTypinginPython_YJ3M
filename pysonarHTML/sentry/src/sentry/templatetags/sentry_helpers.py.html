  54 class AbsoluteUriNode(template.Node):
  55     def __init__(self, args, target_var):
  56         self.args = args
  57         self.target_var = target_var
  58 
  59     def render(self, context):
  60         from sentry.utils.http import absolute_uri
  61 
  62         # Attempt to resolve all arguments into actual variables.
  63         # This converts a value such as `"foo"` into the string `foo`
  64         # and will look up a value such as `foo` from the context as
  65         # the variable `foo`. If the variable does not exist, silently
  66         # resolve as an empty string, which matches the behavior
  67         # `SimpleTagNode`
  68         args = []
  69         for arg in self.args:
  70             try:
  71                 arg = template.Variable(arg).resolve(context)
  72             except template.VariableDoesNotExist:
  73                 arg = ""
  74             args.append(arg)
  75 
  76         # No args is just fine
  77         if not args:
  78             rv = ""
  79         # If there's only 1 argument, there's nothing to format
  80         elif len(args) == 1:
  81             rv = args[0]
  82         else:
  83             rv = args[0].format(*args[1:])
  84 
  85         rv = absolute_uri(rv)
  86 
  87         # We're doing an `as foo` and we want to assign the result
  88         # to a variable instead of actually returning.
  89         if self.target_var is not None:
  90             context[self.target_var] = rv
  91             rv = ""
  92 
  93         return rv

  96 @register.tag
  97 def absolute_uri(parser, token):
  98     bits = token.split_contents()[1:]
  99     # Check if the last two bits are `as {var}`
 100     if len(bits) >= 2 and bits[-2] == "as":
 101         target_var = bits[-1]
 102         bits = bits[:-2]
 103     else:
 104         target_var = None
 105     return AbsoluteUriNode(bits, target_var)

 164 @register.filter
 165 def small_count(v, precision=1):
 166     if not v:
 167         return 0
 168     z = [(1000000000, _("b")), (1000000, _("m")), (1000, _("k"))]
 169     v = int(v)
 170     for x, y in z:
 171         o, p = divmod(v, x)
 172         if o:
 173             if len(six.text_type(o)) > 2 or not p:
 174                 return "%d%s" % (o, y)
 175             return (u"%.{}f%s".format(precision)) % (v / float(x), y)
 176     return v

 204 @register.filter
 205 def timesince(value, now=None):
 206     from django.template.defaultfilters import timesince
 207 
 208     if now is None:
 209         now = timezone.now()
 210     if not value:
 211         return _("never")
 212     if value < (now - timedelta(days=5)):
 213         return value.date()
 214     value = (" ".join(timesince(value, now).split(" ")[0:2])).strip(",")
 215     if value == _("0 minutes"):
 216         return _("just now")
 217     if value == _("1 day"):
 218         return _("yesterday")
 219     return _("%s ago") % value

 222 @register.filter
 223 def duration(value):
 224     if not value:
 225         return "0s"
 226     # value is assumed to be in ms
 227     value = value / 1000.0
 228     hours, minutes, seconds = 0, 0, 0
 229     if value > 3600:
 230         hours = value / 3600
 231         value = value % 3600
 232     if value > 60:
 233         minutes = value / 60
 234         value = value % 60
 235     seconds = value
 236     output = []
 237     if hours:
 238         output.append("%dh" % hours)
 239     if minutes:
 240         output.append("%dm" % minutes)
 241     if seconds > 1:
 242         output.append("%0.2fs" % seconds)
 243     elif seconds:
 244         output.append("%dms" % (seconds * 1000))
 245     return "".join(output)
