  27 class PendingBuffer(object):
  28     def __init__(self, size):
  29         assert size > 0
  30         self.buffer = [None] * size
  31         self.size = size
  32         self.pointer = 0
  33 
  34     def full(self):
  35         return self.pointer == self.size
  36 
  37     def empty(self):
  38         return self.pointer == 0
  39 
  40     def append(self, item):
  41         assert not self.full()
  42         self.buffer[self.pointer] = item
  43         self.pointer += 1
  44 
  45     def clear(self):
  46         self.pointer = 0
  47 
  48     def flush(self):
  49         rv = self.buffer[: self.pointer]
  50         self.clear()
  51         return rv

  54 class RedisBuffer(Buffer):
  55     key_expire = 60 * 60  # 1 hour
  56     pending_key = "b:p"
  57 
  58     def __init__(self, pending_partitions=1, incr_batch_size=2, **options):
  59         self.cluster, options = get_cluster_from_options("SENTRY_BUFFER_OPTIONS", options)
  60         self.pending_partitions = pending_partitions
  61         self.incr_batch_size = incr_batch_size
  62         assert self.pending_partitions > 0
  63         assert self.incr_batch_size > 0
  64 
  65     def validate(self):
  66         try:
  67             with self.cluster.all() as client:
  68                 client.ping()
  69         except Exception as e:
  70             raise InvalidConfiguration(six.text_type(e))

  72     def _coerce_val(self, value):
  73         if isinstance(value, models.Model):
  74             value = value.pk
  75         return force_bytes(value, errors="replace")
  76 
  77     def _make_key(self, model, filters):
  78         """
  79         Returns a Redis-compatible key for the model given filters.
  80         """
  81         return "b:k:%s:%s" % (
  82             model._meta,
  83             md5_text(
  84                 "&".join(
  85                     "%s=%s" % (k, self._coerce_val(v)) for k, v in sorted(six.iteritems(filters))
  86                 )
  87             ).hexdigest(),
  88         )
  89 
  90     def _make_pending_key(self, partition=None):
  91         """
  92         Returns the key to be used for the pending buffer.
  93         When partitioning is enabled, there is a key for each
  94         partition, without it, there's only the default pending_key
  95         """
  96         if partition is None:
  97             return self.pending_key
  98         assert partition >= 0
  99         return "%s:%d" % (self.pending_key, partition)
 100 
 101     def _make_pending_key_from_key(self, key):
 102         """
 103         Return the pending_key for a given key. This is used
 104         to route a key into the correct pending buffer. If partitioning
 105         is disabled, route into the no partition buffer.
 106         """
 107         if self.pending_partitions == 1:
 108             return self.pending_key
 109         return self._make_pending_key(crc32(key) % self.pending_partitions)
 110 
 111     def _make_lock_key(self, key):
 112         return "l:%s" % (key,)

 114     def _dump_values(self, values):
 115         result = {}
 116         for k, v in six.iteritems(values):
 117             result[k] = self._dump_value(v)
 118         return result
 119 
 120     def _dump_value(self, value):
 121         if isinstance(value, six.string_types):
 122             type_ = "s"
 123         elif isinstance(value, datetime):
 124             type_ = "d"
 125             value = value.strftime("%s.%f")
 126         elif isinstance(value, int):
 127             type_ = "i"
 128         elif isinstance(value, float):
 129             type_ = "f"
 130         else:
 131             raise TypeError(type(value))
 132         return (type_, six.text_type(value))
 133 
 134     def _load_values(self, payload):
 135         result = {}
 136         for k, (t, v) in six.iteritems(payload):
 137             result[k] = self._load_value((t, v))
 138         return result
 139 
 140     def _load_value(self, payload):
 141         (type_, value) = payload
 142         if type_ == "s":
 143             return value
 144         elif type_ == "d":
 145             return datetime.fromtimestamp(float(value)).replace(tzinfo=timezone.utc)
 146         elif type_ == "i":
 147             return int(value)
 148         elif type_ == "f":
 149             return float(value)
 150         else:
 151             raise TypeError("invalid type: {}".format(type_))

 153     def incr(self, model, columns, filters, extra=None, signal_only=None):
 154         """
 155         Increment the key by doing the following:
 156 
 157         - Insert/update a hashmap based on (model, columns)
 158             - Perform an incrby on counters
 159             - Perform a set (last write wins) on extra
 160             - Perform a set on signal_only (only if True)
 161         - Add hashmap key to pending flushes
 162         """
 163 
 164         # TODO(dcramer): longer term we'd rather not have to serialize values
 165         # here (unless it's to JSON)
 166         key = self._make_key(model, filters)
 167         pending_key = self._make_pending_key_from_key(key)
 168         # We can't use conn.map() due to wanting to support multiple pending
 169         # keys (one per Redis partition)
 170         conn = self.cluster.get_local_client_for_key(key)
 171 
 172         pipe = conn.pipeline()
 173         pipe.hsetnx(key, "m", "%s.%s" % (model.__module__, model.__name__))
 174         # TODO(dcramer): once this goes live in production, we can kill the pickle path
 175         # (this is to ensure a zero downtime deploy where we can transition event processing)
 176         pipe.hsetnx(key, "f", pickle.dumps(filters))
 177         # pipe.hsetnx(key, 'f', json.dumps(self._dump_values(filters)))
 178         for column, amount in six.iteritems(columns):
 179             pipe.hincrby(key, "i+" + column, amount)
 180 
 181         if extra:
 182             # Group tries to serialize 'score', so we'd need some kind of processing
 183             # hook here
 184             # e.g. "update score if last_seen or times_seen is changed"
 185             for column, value in six.iteritems(extra):
 186                 # TODO(dcramer): once this goes live in production, we can kill the pickle path
 187                 # (this is to ensure a zero downtime deploy where we can transition event processing)
 188                 pipe.hset(key, "e+" + column, pickle.dumps(value))
 189                 # pipe.hset(key, 'e+' + column, json.dumps(self._dump_value(value)))
 190 
 191         if signal_only is True:
 192             pipe.hset(key, "s", "1")
 193 
 194         pipe.expire(key, self.key_expire)
 195         pipe.zadd(pending_key, time(), key)
 196         pipe.execute()
 197 
 198         metrics.incr(
 199             "buffer.incr",
 200             skip_internal=True,
 201             tags={"module": model.__module__, "model": model.__name__},
 202         )

 204     def process_pending(self, partition=None):
 205         if partition is None and self.pending_partitions > 1:
 206             # If we're using partitions, this one task fans out into
 207             # N subtasks instead.
 208             for i in range(self.pending_partitions):
 209                 process_pending.apply_async(kwargs={"partition": i})
 210             # Explicitly also run over the unpartitioned buffer as well
 211             # to ease in transition. In practice, this should just be
 212             # super fast and is fine to do redundantly.
 213 
 214         pending_key = self._make_pending_key(partition)
 215         client = self.cluster.get_routing_client()
 216         lock_key = self._make_lock_key(pending_key)
 217         # prevent a stampede due to celerybeat + periodic task
 218         if not client.set(lock_key, "1", nx=True, ex=60):
 219             return
 220 
 221         pending_buffer = PendingBuffer(self.incr_batch_size)
 222 
 223         try:
 224             keycount = 0
 225             with self.cluster.all() as conn:
 226                 results = conn.zrange(pending_key, 0, -1)
 227 
 228             with self.cluster.all() as conn:
 229                 for host_id, keys in six.iteritems(results.value):
 230                     if not keys:
 231                         continue
 232                     keycount += len(keys)
 233                     for key in keys:
 234                         pending_buffer.append(key)
 235                         if pending_buffer.full():
 236                             process_incr.apply_async(kwargs={"batch_keys": pending_buffer.flush()})
 237                     conn.target([host_id]).zrem(pending_key, *keys)
 238 
 239             # queue up remainder of pending keys
 240             if not pending_buffer.empty():
 241                 process_incr.apply_async(kwargs={"batch_keys": pending_buffer.flush()})
 242 
 243             metrics.timing("buffer.pending-size", keycount)
 244         finally:
 245             client.delete(lock_key)

 247     def process(self, key=None, batch_keys=None):
 248         assert not (key is None and batch_keys is None)
 249         assert not (key is not None and batch_keys is not None)
 250 
 251         if key is not None:
 252             batch_keys = [key]
 253 
 254         for key in batch_keys:
 255             self._process_single_incr(key)
 256 
 257     def _process_single_incr(self, key):
 258         client = self.cluster.get_routing_client()
 259         lock_key = self._make_lock_key(key)
 260         # prevent a stampede due to the way we use celery etas + duplicate
 261         # tasks
 262         if not client.set(lock_key, "1", nx=True, ex=10):
 263             metrics.incr("buffer.revoked", tags={"reason": "locked"}, skip_internal=False)
 264             self.logger.debug("buffer.revoked.locked", extra={"redis_key": key})
 265             return
 266 
 267         pending_key = self._make_pending_key_from_key(key)
 268 
 269         try:
 270             conn = self.cluster.get_local_client_for_key(key)
 271             pipe = conn.pipeline()
 272             pipe.hgetall(key)
 273             pipe.zrem(pending_key, key)
 274             pipe.delete(key)
 275             values = pipe.execute()[0]
 276 
 277             if not values:
 278                 metrics.incr("buffer.revoked", tags={"reason": "empty"}, skip_internal=False)
 279                 self.logger.debug("buffer.revoked.empty", extra={"redis_key": key})
 280                 return
 281 
 282             model = import_string(values.pop("m"))
 283             if values["f"].startswith("{"):
 284                 filters = self._load_values(json.loads(values.pop("f")))
 285             else:
 286                 # TODO(dcramer): legacy pickle support - remove in Sentry 9.1
 287                 filters = pickle.loads(values.pop("f"))
 288 
 289             incr_values = {}
 290             extra_values = {}
 291             signal_only = None
 292             for k, v in six.iteritems(values):
 293                 if k.startswith("i+"):
 294                     incr_values[k[2:]] = int(v)
 295                 elif k.startswith("e+"):
 296                     if v.startswith("["):
 297                         extra_values[k[2:]] = self._load_value(json.loads(v))
 298                     else:
 299                         # TODO(dcramer): legacy pickle support - remove in Sentry 9.1
 300                         extra_values[k[2:]] = pickle.loads(v)
 301                 elif k == "s":
 302                     signal_only = bool(int(v))  # Should be 1 if set
 303 
 304             super(RedisBuffer, self).process(model, incr_values, filters, extra_values, signal_only)
 305         finally:
 306             client.delete(lock_key)
