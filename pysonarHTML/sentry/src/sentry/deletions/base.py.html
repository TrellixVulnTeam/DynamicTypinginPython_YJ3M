  60     def get_child_relations(self, instance):

  75         return child_relations

  77     def filter_relations(self, child_relations):
  78         if not self.skip_models or not child_relations:
  79             return child_relations
  80 
  81         return list(
  82             [rel for rel in child_relations if rel.params.get("model") not in self.skip_models]
  83         )

  85     def delete_bulk(self, instance_list):
  86         """
  87         Delete a batch of objects bound to this task.
  88 
  89         This **should** not be called with arbitrary types, but rather should
  90         be used for only the base type this task was instantiated against.
  91         """
  92         self.mark_deletion_in_progress(instance_list)
  93 
  94         child_relations = self.get_child_relations_bulk(instance_list)
  95         child_relations = self.extend_relations_bulk(child_relations, instance_list)
  96         child_relations = self.filter_relations(child_relations)
  97         if child_relations:
  98             has_more = self.delete_children(child_relations)
  99             if has_more:
 100                 return has_more
 101 
 102         for instance in instance_list:
 103             child_relations = self.get_child_relations(instance)
 104             child_relations = self.extend_relations(child_relations, instance)
 105             child_relations = self.filter_relations(child_relations)
 106             if child_relations:
 107                 has_more = self.delete_children(child_relations)
 108                 if has_more:
 109                     return has_more
 110 
 111         return self.delete_instance_bulk(instance_list)
 112

 114         raise NotImplementedError

 116     def delete_instance_bulk(self, instance_list):
 117         for instance in instance_list:

 120     def delete_children(self, relations):
 121         # Ideally this runs through the deletion manager
 122         for relation in relations:
 123             task = self.manager.get(
 124                 transaction_id=self.transaction_id,
 125                 actor_id=self.actor_id,
 126                 task=relation.task,
 127                 **relation.params
 128             )
 129             has_more = True
 130             while has_more:
 131                 has_more = task.chunk()
 132         return False

 138 class ModelDeletionTask(BaseDeletionTask):
 139     DEFAULT_QUERY_LIMIT = None
 140     manager_name = "objects"
 141 
 142     def __init__(self, manager, model, query, query_limit=None, order_by=None, **kwargs):
 143         super(ModelDeletionTask, self).__init__(manager, **kwargs)
 144         self.model = model
 145         self.query = query
 146         self.query_limit = query_limit or self.DEFAULT_QUERY_LIMIT or self.chunk_size
 147         self.order_by = order_by
 148 
 149     def __repr__(self):
 150         return "<%s: model=%s query=%s order_by=%s transaction_id=%s actor_id=%s>" % (
 151             type(self),
 152             self.model,
 153             self.query,
 154             self.order_by,
 155             self.transaction_id,
 156             self.actor_id,
 157         )
 158 
 159     def extend_relations(self, child_relations, obj):
 160         from sentry.deletions import default_manager
 161 
 162         return child_relations + [rel(obj) for rel in default_manager.dependencies[self.model]]
 163 
 164     def extend_relations_bulk(self, child_relations, obj_list):
 165         from sentry.deletions import default_manager
 166 
 167         return child_relations + [
 168             rel(obj_list) for rel in default_manager.bulk_dependencies[self.model]
 169         ]
 170 
 171     def chunk(self, num_shards=None, shard_id=None):
 172         """
 173         Deletes a chunk of this instance's data. Return ``True`` if there is
 174         more work, or ``False`` if the entity has been removed.
 175         """
 176         query_limit = self.query_limit
 177         remaining = self.chunk_size
 178         while remaining > 0:
 179             queryset = getattr(self.model, self.manager_name).filter(**self.query)
 180             if self.order_by:
 181                 queryset = queryset.order_by(self.order_by)
 182 
 183             if num_shards:
 184                 assert num_shards > 1
 185                 assert shard_id < num_shards
 186                 queryset = queryset.extra(
 187                     where=[
 188                         u"id %% {num_shards} = {shard_id}".format(
 189                             num_shards=num_shards, shard_id=shard_id
 190                         )
 191                     ]
 192                 )
 193 
 194             queryset = list(queryset[:query_limit])
 195             if not queryset:
 196                 return False
 197 
 198             self.delete_bulk(queryset)
 199             remaining -= query_limit
 200         return True
 201 
 202     def delete_instance_bulk(self, instance_list):
 203         # slow, but ensures Django cascades are handled
 204         for instance in instance_list:
 205             self.delete_instance(instance)
 206 
 207     def delete_instance(self, instance):
 208         instance_id = instance.id
 209         try:
 210             instance.delete()
 211         finally:
 212             # Don't log Group and Event child object deletions.
 213             model_name = type(instance).__name__
 214             if not _leaf_re.search(model_name):
 215                 self.logger.info(
 216                     "object.delete.executed",
 217                     extra={
 218                         "object_id": instance_id,
 219                         "transaction_id": self.transaction_id,
 220                         "app_label": instance._meta.app_label,
 221                         "model": model_name,
 222                     },
 223                 )
 224 
 225     def get_actor(self):
 226         from sentry.models import User
 227 
 228         if self.actor_id:
 229             try:
 230                 return User.objects.get_from_cache(id=self.actor_id)
 231             except User.DoesNotExist:
 232                 pass
 233         return None
 234 
 235     def mark_deletion_in_progress(self, instance_list):
 236         for instance in instance_list:
 237             status = getattr(instance, "status", None)
 238             if status not in (ObjectStatus.DELETION_IN_PROGRESS, None):
 239                 instance.update(status=ObjectStatus.DELETION_IN_PROGRESS)

 242 class BulkModelDeletionTask(ModelDeletionTask):
 243     """
 244     An efficient mechanism for deleting larger volumes of rows in one pass,
 245     but will hard fail if the relations have resident foreign relations.
 246 
 247     Note: Does NOT support child relations.
 248     """
 249 
 250     DEFAULT_CHUNK_SIZE = 10000
 251 
 252     def __init__(self, manager, model, query, partition_key=None, **kwargs):
 253         super(BulkModelDeletionTask, self).__init__(manager, model, query, **kwargs)
 254 
 255         self.partition_key = partition_key
 256 
 257     def chunk(self):
 258         return self.delete_instance_bulk()
 259 
 260     def delete_instance_bulk(self):
 261         try:
 262             return bulk_delete_objects(
 263                 model=self.model,
 264                 limit=self.chunk_size,
 265                 transaction_id=self.transaction_id,
 266                 partition_key=self.partition_key,
 267                 **self.query
 268             )
 269         finally:
 270             # Don't log Group and Event child object deletions.
 271             model_name = self.model.__name__
 272             if not _leaf_re.search(model_name):
 273                 self.logger.info(
 274                     "object.delete.bulk_executed",
 275                     extra=dict(
 276                         {
 277                             "transaction_id": self.transaction_id,
 278                             "app_label": self.model._meta.app_label,
 279                             "model": model_name,
 280                         },
 281                         **self.query
 282                     ),
 283                 )
