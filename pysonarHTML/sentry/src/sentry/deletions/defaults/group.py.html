  11 class EventDataDeletionTask(BaseDeletionTask):
  12     """
  13     Deletes nodestore data, EventAttachment and UserReports for group
  14     """
  15 
  16     DEFAULT_CHUNK_SIZE = 10000
  17 
  18     def __init__(self, manager, group_id, project_id, **kwargs):
  19         self.group_id = group_id
  20         self.project_id = project_id
  21         self.last_event = None
  22         super(EventDataDeletionTask, self).__init__(manager, **kwargs)
  23 
  24     def chunk(self):
  25         conditions = []
  26         if self.last_event is not None:
  27             conditions.extend(
  28                 [
  29                     ["timestamp", "<=", self.last_event.timestamp],
  30                     [
  31                         ["timestamp", "<", self.last_event.timestamp],
  32                         ["event_id", "<", self.last_event.event_id],
  33                     ],
  34                 ]
  35             )
  36 
  37         events = eventstore.get_unfetched_events(
  38             filter=eventstore.Filter(
  39                 conditions=conditions, project_ids=[self.project_id], group_ids=[self.group_id]
  40             ),
  41             limit=self.DEFAULT_CHUNK_SIZE,
  42             referrer="deletions.group",
  43             orderby=["-timestamp", "-event_id"],
  44         )
  45 
  46         if not events:
  47             return False
  48 
  49         self.last_event = events[-1]
  50 
  51         # Remove from nodestore
  52         node_ids = [Event.generate_node_id(self.project_id, event.event_id) for event in events]
  53         nodestore.delete_multi(node_ids)
  54 
  55         # Remove EventAttachment and UserReport
  56         event_ids = [event.event_id for event in events]
  57         EventAttachment.objects.filter(event_id__in=event_ids, project_id=self.project_id).delete()
  58         UserReport.objects.filter(event_id__in=event_ids, project_id=self.project_id).delete()
  59 
  60         return True
