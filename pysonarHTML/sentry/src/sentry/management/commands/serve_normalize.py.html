  23 def catch_errors(f):
  24     def wrapper(*args, **kwargs):
  25         error = None
  26         try:
  27             return f(*args, **kwargs)
  28         except Exception as e:
  29             error = force_str(six.text_type(e)) + " " + force_str(traceback.format_exc())
  30 
  31         try:
  32             return encode({"result": None, "error": error, "metrics": None})
  33         except (ValueError, TypeError) as e:
  34             try:
  35                 # Encoding error, try to send the exception instead
  36                 return encode(
  37                     {
  38                         "result": None,
  39                         "error": force_str(six.text_type(e))
  40                         + " "
  41                         + force_str(traceback.format_exc()),
  42                         "metrics": None,
  43                         "encoding_error": True,
  44                     }
  45                 )
  46             except Exception:
  47                 return b"{}"
  48 
  49     return wrapper

 107 class MetricCollector(object):
 108     def __init__(self):
 109         self.is_linux = sys.platform.startswith("linux")
 110         self.pid = os.getpid()
 111 
 112     def collect_metrics(self):
 113         metrics = {"time": time.time()}
 114 
 115         usage = resource.getrusage(resource.RUSAGE_SELF)
 116         usage_dict = {attr: getattr(usage, attr) for attr in dir(usage) if attr.startswith("ru_")}
 117         metrics.update(usage_dict)
 118 
 119         if self.is_linux:
 120             with open("/proc/{}/status".format(self.pid)) as procfh:
 121                 metrics["proc"] = procfh.read()
 122 
 123         return metrics

 126 class EventNormalizeHandler(SocketServer.BaseRequestHandler):
 127     """
 128     The request handler class for our server.
 129     It is instantiated once per connection to the server, and must
 130     override the handle() method to implement communication to the
 131     client.
 132     """
 133 
 134     BUFFER_SIZE = 4096
 135     SOCKET_TIMEOUT = 10.0
 136 
 137     def handle(self):
 138         self.server.socket.settimeout(self.SOCKET_TIMEOUT)
 139 
 140         chunks = []
 141 
 142         # Receive the data
 143         while True:
 144             rcvd = self.request.recv(self.BUFFER_SIZE)
 145             if rcvd is None:
 146                 raise ValueError("Received None")
 147 
 148             if not rcvd:
 149                 break
 150             chunks.append(rcvd)
 151 
 152         self.data = "".join(chunks)
 153 
 154         response = self.handle_data()
 155         self.request.sendall(response)
 156         self.request.close()
 157 
 158     def handle_data(self):
 159         return handle_data(self.data)

 162 class Command(BaseCommand):
 163     help = "Start a socket server for event normalization"
 164 
 165     option_list = BaseCommand.option_list + (
 166         make_option(
 167             "--unix",
 168             dest="socket_file",
 169             help='Unix socket to bind to. Example: "/tmp/normalize.sock"',
 170         ),
 171         make_option(
 172             "--net",
 173             dest="network_socket",
 174             help='Network socket to bind to. Example: "127.0.0.1:1234"',
 175         ),
 176         make_option(
 177             "--threading", action="store_true", dest="threading", help="Start a threading server"
 178         ),
 179         make_option(
 180             "--forking", action="store_true", dest="forking", help="Start a forking server"
 181         ),
 182     )
 183 
 184     def _check_socket_path(self, socket_file):
 185         if os.path.exists(socket_file):
 186             file_mode = os.stat(socket_file).st_mode
 187             if not stat.S_ISSOCK(file_mode):
 188                 raise CommandError("File already exists and is not a socket")
 189 
 190         # Make sure the socket does not already exist
 191         try:
 192             os.unlink(socket_file)
 193         except OSError:
 194             if os.path.exists(socket_file):
 195                 raise
 196 
 197     def handle(self, **options):
 198         socket_file = options.get("socket_file")
 199         network_socket = options.get("network_socket")
 200         threading = options.get("threading")
 201         forking = options.get("forking")
 202         if threading and forking:
 203             raise CommandError("Pick one: threading or forking.")
 204         if socket_file and network_socket:
 205             raise CommandError("Only one socket allowed at a time")
 206 
 207         if threading:
 208             server_type = "threading"
 209         elif forking:
 210             server_type = "forking"
 211         else:
 212             server_type = "single-threaded"
 213         self.stdout.write("Server type: %s\n" % (server_type,))
 214 
 215         if socket_file:
 216             self.socket_file = os.path.abspath(socket_file)
 217             self._check_socket_path(socket_file)
 218             self.stdout.write("Binding to unix socket: %s\n" % (socket_file,))
 219             if threading:
 220                 server = SocketServer.ThreadingUnixStreamServer(socket_file, EventNormalizeHandler)
 221                 server.daemon_threads = True
 222             elif forking:
 223                 server = ForkingUnixStreamServer(socket_file, EventNormalizeHandler)
 224             else:
 225                 server = SocketServer.UnixStreamServer(socket_file, EventNormalizeHandler)
 226         elif network_socket:
 227             host, port = network_socket.split(":")
 228             port = int(port)
 229             self.stdout.write("Binding to network socket: %s:%s\n" % (host, port))
 230             if threading:
 231                 server = SocketServer.ThreadingTCPServer((host, port), EventNormalizeHandler)
 232                 server.daemon_threads = True
 233             elif forking:
 234                 server = SocketServer.ForkingTCPServer((host, port), EventNormalizeHandler)
 235             else:
 236                 server = SocketServer.TCPServer((host, port), EventNormalizeHandler)
 237         else:
 238             raise CommandError("No connection option specified")
 239 
 240         server.serve_forever()
