   1 from __future__ import absolute_import, unicode_literals

   3 import io
   4 import os

   6 import six
   7 from django.conf import settings
   8 from django.db.migrations.loader import MigrationLoader
   9 from django.core.management.commands import makemigrations
  10 
  11 template = """Django migrations lock file. This helps us avoid migration conflicts on master.
  12 If you have a conflict in this file, it means that someone has committed a migration
  13 ahead of you.
  14 
  15 To resolve this, rebase against latest master and regenerate your migration. This file
  16 will then be regenerated, and you should be able to merge without conflicts.
  17 
  18 %s
  19 """
  20 
  21 
  22 class Command(makemigrations.Command):
  23     """
  24     Generates a lockfile so that Git will detect merge conflicts if there's a migration
  25     on master that doesn't exist in a branch.
  26     """
  27 
  28     def handle(self, *app_labels, **options):
  29         if not options["name"]:
  30             self.stderr.write(
  31                 "Please name your migrations using `-n <migration_name>`. "
  32                 "For example, `-n backfill_my_new_table`"
  33             )
  34             return
  35         super(Command, self).handle(*app_labels, **options)
  36         loader = MigrationLoader(None, ignore_no_migrations=True)
  37 
  38         latest_migration_by_app = {}
  39         for migration in six.itervalues(loader.disk_migrations):
  40             name = migration.name
  41             app_label = migration.app_label
  42             if (
  43                 settings.MIGRATIONS_LOCKFILE_APP_WHITELIST
  44                 and app_label not in settings.MIGRATIONS_LOCKFILE_APP_WHITELIST
  45             ):
  46                 continue
  47             latest_migration_by_app[app_label] = max(
  48                 latest_migration_by_app.get(app_label, ""), name
  49             )
  50 
  51         result = "\n".join(
  52             "{}: {}".format(app_label, name)
  53             for app_label, name in sorted(latest_migration_by_app.items())
  54         )
  55 
  56         with io.open(
  57             os.path.join(settings.MIGRATIONS_LOCKFILE_PATH, "migrations_lockfile.txt"), "w"
  58         ) as f:
  59             f.write(template % result)
