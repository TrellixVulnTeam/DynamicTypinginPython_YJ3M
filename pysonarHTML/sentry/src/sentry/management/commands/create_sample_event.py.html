   6 class Command(BaseCommand):
   7     help = "Creates a sample event in Sentry (if applicable)"
   8 
   9     def add_arguments(self, parser):
  10         parser.add_argument(
  11             "--project", dest="project", help="project ID or team-slug/project-slug"
  12         ),
  13         parser.add_argument("--platform", dest="platform"),
  14 
  15     def handle(self, **options):
  16         from django.conf import settings
  17         from sentry.models import Project
  18         from sentry.utils.samples import create_sample_event
  19 
  20         if not options["project"]:
  21             project = Project.objects.get(id=settings.SENTRY_PROJECT)
  22         else:
  23             if options["project"].isdigit():
  24                 project = Project.objects.get(id=options["project"])
  25             elif "/" in options["project"]:
  26                 t_slug, p_slug = options["project"].split("/", 1)
  27                 project = Project.objects.get(slug=p_slug, teams__slug=t_slug)
  28             else:
  29                 raise CommandError(
  30                     "Project must be specified as team-slug/project-slug or a project id"
  31                 )
  32 
  33         platform = options["platform"]
  34         event = create_sample_event(project, platform)
  35         if not event:
  36             raise CommandError("Unable to create an event for platform %r" % (platform,))
  37 
  38         self.stdout.write("Event created: %s" % (event.group.get_absolute_url(),))
