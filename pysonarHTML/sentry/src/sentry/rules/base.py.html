   1 """
   2 Rules apply either before an event gets stored, or immediately after.
   3 
   4 Basic actions:
   5 
   6 - I want to get notified when [X]
   7 - I want to group events when [X]
   8 - I want to scrub data when [X]
   9 
  10 Expanded:
  11 
  12 - I want to get notified when an event is first seen
  13 - I want to get notified when an event is marked as a regression
  14 - I want to get notified when the rate of an event increases by [100%]
  15 - I want to get notified when an event has been seen more than [100] times
  16 - I want to get notified when an event matches [conditions]
  17 - I want to group events when an event matches [conditions]
  18 
  19 Rules get broken down into two phases:
  20 
  21 - An action
  22 - A rule condition
  23 
  24 A condition itself may actually be any number of things, but that is determined
  25 by the rule's logic. Each rule condition may be associated with a form.
  26 
  27 - [ACTION:I want to get notified when] [RULE:an event is first seen]
  28 - [ACTION:I want to group events when] [RULE:an event matches [FORM]]
  29 """

  37 # Encapsulates a reference to the callback, including arguments. The `key`
  38 # attribute may be specifically used to key the callbacks when they are
  39 # collated during rule processing.
  40 CallbackFuture = namedtuple("CallbackFuture", ["callback", "kwargs", "key"])

  43 class RuleDescriptor(type):
  44     def __new__(cls, *args, **kwargs):
  45         new_cls = super(RuleDescriptor, cls).__new__(cls, *args, **kwargs)
  46         new_cls.id = "%s.%s" % (new_cls.__module__, new_cls.__name__)
  47         return new_cls

  50 @six.add_metaclass(RuleDescriptor)
  51 class RuleBase(object):
  52     label = None
  53     form_cls = None
  54 
  55     logger = logging.getLogger("sentry.rules")
  56 
  57     def __init__(self, project, data=None, rule=None):
  58         self.project = project
  59         self.data = data or {}
  60         self.had_data = data is not None
  61         self.rule = rule
  62 
  63     def is_enabled(self):
  64         return True
  65 
  66     def get_option(self, key, default=None):
  67         return self.data.get(key, default)
  68 
  69     def get_form_instance(self):
  70         if self.had_data:
  71             data = self.data
  72         else:
  73             data = None
  74         return self.form_cls(data)
  75 
  76     def render_label(self):
  77         return self.label.format(**self.data)
  78 
  79     def validate_form(self):
  80         if not self.form_cls:
  81             return True
  82 
  83         form = self.get_form_instance()
  84 
  85         return form.is_valid()
  86 
  87     def future(self, callback, key=None, **kwargs):
  88         return CallbackFuture(callback=callback, key=key, kwargs=kwargs)
