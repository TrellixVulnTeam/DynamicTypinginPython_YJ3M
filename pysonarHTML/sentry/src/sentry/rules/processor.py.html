  21 class RuleProcessor(object):
  22     logger = logging.getLogger("sentry.rules")
  23 
  24     def __init__(self, event, is_new, is_regression, is_new_group_environment, has_reappeared):
  25         self.event = event
  26         self.group = event.group
  27         self.project = event.project
  28 
  29         self.is_new = is_new
  30         self.is_regression = is_regression
  31         self.is_new_group_environment = is_new_group_environment
  32         self.has_reappeared = has_reappeared
  33 
  34         self.grouped_futures = {}
  35 
  36     def get_rules(self):
  37         return Rule.get_for_project(self.project.id)
  38 
  39     def get_rule_status(self, rule):
  40         key = "grouprulestatus:1:%s" % hash_values([self.group.id, rule.id])
  41         rule_status = cache.get(key)
  42         if rule_status is None:
  43             rule_status, _ = GroupRuleStatus.objects.get_or_create(
  44                 rule=rule, group=self.group, defaults={"project": self.project}
  45             )
  46             cache.set(key, rule_status, 300)
  47         return rule_status
  48 
  49     def condition_matches(self, condition, state, rule):
  50         condition_cls = rules.get(condition["id"])
  51         if condition_cls is None:
  52             self.logger.warn("Unregistered condition %r", condition["id"])
  53             return
  54 
  55         condition_inst = condition_cls(self.project, data=condition, rule=rule)
  56         return safe_execute(condition_inst.passes, self.event, state, _with_transaction=False)
  57 
  58     def get_state(self):
  59         return EventState(
  60             is_new=self.is_new,
  61             is_regression=self.is_regression,
  62             is_new_group_environment=self.is_new_group_environment,
  63             has_reappeared=self.has_reappeared,
  64         )

  66     def apply_rule(self, rule):
  67         match = rule.data.get("action_match") or Rule.DEFAULT_ACTION_MATCH
  68         condition_list = rule.data.get("conditions", ())
  69         frequency = rule.data.get("frequency") or Rule.DEFAULT_FREQUENCY
  70 
  71         # XXX(dcramer): if theres no condition should we really skip it,
  72         # or should we just apply it blindly?
  73         if not condition_list:
  74             return
  75 
  76         if (
  77             rule.environment_id is not None
  78             and self.event.get_environment().id != rule.environment_id
  79         ):
  80             return
  81 
  82         status = self.get_rule_status(rule)
  83 
  84         now = timezone.now()
  85         freq_offset = now - timedelta(minutes=frequency)
  86 
  87         if status.last_active and status.last_active > freq_offset:
  88             return
  89 
  90         state = self.get_state()
  91 
  92         condition_iter = (self.condition_matches(c, state, rule) for c in condition_list)
  93 
  94         if match == "all":
  95             passed = all(condition_iter)
  96         elif match == "any":
  97             passed = any(condition_iter)
  98         elif match == "none":
  99             passed = not any(condition_iter)
 100         else:
 101             self.logger.error("Unsupported action_match %r for rule %d", match, rule.id)
 102             return
 103 
 104         if passed:
 105             passed = (
 106                 GroupRuleStatus.objects.filter(id=status.id)
 107                 .exclude(last_active__gt=freq_offset)
 108                 .update(last_active=now)
 109             )
 110 
 111         if not passed:
 112             return
 113 
 114         if randrange(10) == 0:
 115             analytics.record(
 116                 "issue_alert.fired",
 117                 issue_id=self.group.id,
 118                 project_id=rule.project.id,
 119                 organization_id=rule.project.organization.id,
 120                 rule_id=rule.id,
 121             )
 122 
 123         for action in rule.data.get("actions", ()):
 124             action_cls = rules.get(action["id"])
 125             if action_cls is None:
 126                 self.logger.warn("Unregistered action %r", action["id"])
 127                 continue
 128 
 129             action_inst = action_cls(self.project, data=action, rule=rule)
 130             results = safe_execute(
 131                 action_inst.after, event=self.event, state=state, _with_transaction=False
 132             )
 133             if results is None:
 134                 self.logger.warn("Action %s did not return any futures", action["id"])
 135                 continue
 136 
 137             for future in results:
 138                 key = future.key if future.key is not None else future.callback
 139                 rule_future = RuleFuture(rule=rule, kwargs=future.kwargs)
 140 
 141                 if key not in self.grouped_futures:
 142                     self.grouped_futures[key] = (future.callback, [rule_future])
 143                 else:
 144                     self.grouped_futures[key][1].append(rule_future)
 145 
 146     def apply(self):
 147         # we should only apply rules on unresolved issues
 148         if not self.event.group.is_unresolved():
 149             return six.itervalues({})
 150 
 151         self.grouped_futures.clear()
 152         for rule in self.get_rules():
 153             self.apply_rule(rule)
 154         return six.itervalues(self.grouped_futures)
