  10 class MatchType(object):
  11     EQUAL = "eq"
  12     NOT_EQUAL = "ne"
  13     STARTS_WITH = "sw"
  14     ENDS_WITH = "ew"
  15     CONTAINS = "co"
  16     NOT_CONTAINS = "nc"
  17     IS_SET = "is"
  18     NOT_SET = "ns"
  19 
  20 
  21 MATCH_CHOICES = OrderedDict(
  22     [
  23         (MatchType.EQUAL, "equals"),
  24         (MatchType.NOT_EQUAL, "does not equal"),
  25         (MatchType.STARTS_WITH, "starts with"),
  26         (MatchType.ENDS_WITH, "ends with"),
  27         (MatchType.CONTAINS, "contains"),
  28         (MatchType.NOT_CONTAINS, "does not contain"),
  29         (MatchType.IS_SET, "is set"),
  30         (MatchType.NOT_SET, "is not set"),
  31     ]
  32 )
  33

  35 class TaggedEventForm(forms.Form):
  36     key = forms.CharField(widget=forms.TextInput())
  37     match = forms.ChoiceField(MATCH_CHOICES.items(), widget=forms.Select())
  38     value = forms.CharField(widget=forms.TextInput(), required=False)
  39 
  40     def clean(self):
  41         super(TaggedEventForm, self).clean()
  42 
  43         match = self.cleaned_data.get("match")
  44         value = self.cleaned_data.get("value")
  45 
  46         if match not in (MatchType.IS_SET, MatchType.NOT_SET) and not value:
  47             raise forms.ValidationError("This field is required.")

  50 class TaggedEventCondition(EventCondition):
  51     form_cls = TaggedEventForm
  52     label = u"An event's tags match {key} {match} {value}"
  53 
  54     form_fields = {
  55         "key": {"type": "string", "placeholder": "key"},
  56         "match": {"type": "choice", "choices": MATCH_CHOICES.items()},
  57         "value": {"type": "string", "placeholder": "value"},
  58     }
  59 
  60     def passes(self, event, state, **kwargs):
  61         key = self.get_option("key")
  62         match = self.get_option("match")
  63         value = self.get_option("value")
  64 
  65         if not (key and match):
  66             return False
  67 
  68         key = key.lower()
  69 
  70         tags = (
  71             k
  72             for gen in (
  73                 (k.lower() for k, v in event.tags),
  74                 (tagstore.get_standardized_key(k) for k, v in event.tags),
  75             )
  76             for k in gen
  77         )
  78 
  79         if match == MatchType.IS_SET:
  80             return key in tags
  81 
  82         elif match == MatchType.NOT_SET:
  83             return key not in tags
  84 
  85         if not value:
  86             return False
  87 
  88         value = value.lower()
  89 
  90         values = (
  91             v.lower()
  92             for k, v in event.tags
  93             if k.lower() == key or tagstore.get_standardized_key(k) == key
  94         )
  95 
  96         if match == MatchType.EQUAL:
  97             for t_value in values:
  98                 if t_value == value:
  99                     return True
 100             return False
 101 
 102         elif match == MatchType.NOT_EQUAL:
 103             for t_value in values:
 104                 if t_value == value:
 105                     return False
 106             return True
 107 
 108         elif match == MatchType.STARTS_WITH:
 109             for t_value in values:
 110                 if t_value.startswith(value):
 111                     return True
 112             return False
 113 
 114         elif match == MatchType.ENDS_WITH:
 115             for t_value in values:
 116                 if t_value.endswith(value):
 117                     return True
 118             return False
 119 
 120         elif match == MatchType.CONTAINS:
 121             for t_value in values:
 122                 if value in t_value:
 123                     return True
 124             return False
 125 
 126         elif match == MatchType.NOT_CONTAINS:
 127             for t_value in values:
 128                 if value in t_value:
 129                     return False
 130             return True

 132     def render_label(self):
 133         data = {
 134             "key": self.data["key"],
 135             "value": self.data["value"],
 136             "match": MATCH_CHOICES[self.data["match"]],
 137         }
 138         return self.label.format(**data)
