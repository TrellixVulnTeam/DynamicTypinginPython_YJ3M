  88     def get_rate(self, event, interval, environment_id):
  89         _, duration = intervals[interval]

  94     def is_guessed_to_be_created_on_project_creation(self):
  95         """
  96         Best effort approximation on whether a rule with this condition was created on project creation based on how
  97         closely the rule and project are created; and if the label matches the default name used on project creation.
  98         :return:
  99             bool: True if rule is approximated to be created on project creation, False otherwise.
 100         """
 101         delta = abs(self.rule.date_added - self.project.date_added)
 102         return delta.total_seconds() < 30 and self.rule.label == DEFAULT_RULE_LABEL

 108     def query_hook(self, event, start, end, environment_id):
 109         return self.tsdb.get_sums(

 111             keys=[event.group_id],

 113             end=end,
 114             environment_id=environment_id,
