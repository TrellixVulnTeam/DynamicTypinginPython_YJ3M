   1 from __future__ import absolute_import

   3 from collections import OrderedDict
   4 
   5 from django import forms
   6 from sentry.constants import LOG_LEVELS, LOG_LEVELS_MAP
   7 
   8 from sentry.rules.conditions.base import EventCondition
   9 
  10 LEVEL_CHOICES = OrderedDict(
  11     [(u"{0}".format(k), v) for k, v in sorted(LOG_LEVELS.items(), key=lambda x: x[0], reverse=True)]
  12 )
  13 
  14 
  15 class MatchType(object):
  16     EQUAL = "eq"
  17     LESS_OR_EQUAL = "lte"
  18     GREATER_OR_EQUAL = "gte"
  19 
  20 
  21 MATCH_CHOICES = OrderedDict(
  22     [
  23         (MatchType.EQUAL, "equal to"),
  24         (MatchType.LESS_OR_EQUAL, "less than or equal to"),
  25         (MatchType.GREATER_OR_EQUAL, "greater than or equal to"),
  26     ]
  27 )
  28 
  29 
  30 class LevelEventForm(forms.Form):
  31     level = forms.ChoiceField(choices=LEVEL_CHOICES.items())
  32     match = forms.ChoiceField(choices=MATCH_CHOICES.items())
  33 
  34 
  35 class LevelCondition(EventCondition):
  36     form_cls = LevelEventForm
  37     label = "An event's level is {match} {level}"
  38     form_fields = {
  39         "level": {"type": "choice", "choices": LEVEL_CHOICES.items()},
  40         "match": {"type": "choice", "choices": MATCH_CHOICES.items()},
  41     }
  42 
  43     def passes(self, event, state, **kwargs):
  44         desired_level = self.get_option("level")
  45         desired_match = self.get_option("match")
  46 
  47         if not (desired_level and desired_match):
  48             return False
  49 
  50         desired_level = int(desired_level)
  51         # Fetch the event level from the tags since event.level is
  52         # event.group.level which may have changed
  53         try:
  54             level = LOG_LEVELS_MAP[event.get_tag("level")]
  55         except KeyError:
  56             return False
  57 
  58         if desired_match == MatchType.EQUAL:
  59             return level == desired_level
  60         elif desired_match == MatchType.GREATER_OR_EQUAL:
  61             return level >= desired_level
  62         elif desired_match == MatchType.LESS_OR_EQUAL:
  63             return level <= desired_level
  64         return False
  65 
  66     def render_label(self):
  67         data = {
  68             "level": LEVEL_CHOICES[self.data["level"]],
  69             "match": MATCH_CHOICES[self.data["match"]],
  70         }
  71         return self.label.format(**data)
