  60 class EventAttributeCondition(EventCondition):
  61     """
  62     Attributes are a mapping of <logical-key>.<property>.
  63 
  64     For example:
  65 
  66     - message
  67     - platform
  68     - exception.{type,value}
  69     - user.{id,ip_address,email,FIELD}
  70     - http.{method,url}
  71     - stacktrace.{code,module,filename}
  72     - extra.{FIELD}
  73     """
  74 
  75     # TODO(dcramer): add support for stacktrace.vars.[name]
  76 
  77     form_cls = EventAttributeForm
  78     label = u"An event's {attribute} value {match} {value}"
  79 
  80     form_fields = {
  81         "attribute": {
  82             "type": "choice",
  83             "placeholder": "i.e. exception.type",
  84             "choices": [[a, a] for a in ATTR_CHOICES],
  85         },
  86         "match": {"type": "choice", "choices": MATCH_CHOICES.items()},
  87         "value": {"type": "string", "placeholder": "value"},
  88     }

  90     def _get_attribute_values(self, event, attr):
  91         # TODO(dcramer): we should validate attributes (when we can) before
  92         path = attr.split(".")
  93 
  94         if path[0] == "platform":
  95             if len(path) != 1:
  96                 return []
  97             return [event.platform]
  98 
  99         if path[0] == "message":
 100             if len(path) != 1:
 101                 return []
 102             return [event.message]
 103         elif path[0] == "environment":
 104             return [event.get_tag("environment")]
 105 
 106         elif path[0] == "type":
 107             return [event.data["type"]]
 108 
 109         elif len(path) == 1:
 110             return []
 111 
 112         elif path[0] == "extra":
 113             path.pop(0)
 114             value = event.data["extra"]
 115             while path:
 116                 bit = path.pop(0)
 117                 value = value.get(bit)
 118                 if not value:
 119                     return []
 120 
 121             if isinstance(value, (list, tuple)):
 122                 return value
 123             return [value]
 124 
 125         elif len(path) != 2:
 126             return []
 127 
 128         elif path[0] == "exception":
 129             if path[1] not in ("type", "value"):
 130                 return []
 131 
 132             return [getattr(e, path[1]) for e in event.interfaces["exception"].values]
 133 
 134         elif path[0] == "user":
 135             if path[1] in ("id", "ip_address", "email", "username"):
 136                 return [getattr(event.interfaces["user"], path[1])]
 137             return [getattr(event.interfaces["user"].data, path[1])]
 138 
 139         elif path[0] == "http":
 140             if path[1] not in ("url", "method"):
 141                 return []
 142 
 143             return [getattr(event.interfaces["request"], path[1])]
 144 
 145         elif path[0] == "stacktrace":
 146             stacks = event.interfaces.get("stacktrace")
 147             if stacks:
 148                 stacks = [stacks]
 149             else:
 150                 stacks = [
 151                     e.stacktrace for e in event.interfaces["exception"].values if e.stacktrace
 152                 ]
 153 
 154             result = []
 155             for st in stacks:
 156                 for frame in st.frames:
 157                     if path[1] in ("filename", "module"):
 158                         result.append(getattr(frame, path[1]))
 159                     elif path[1] == "code":
 160                         if frame.pre_context:
 161                             result.extend(frame.pre_context)
 162                         if frame.context_line:
 163                             result.append(frame.context_line)
 164                         if frame.post_context:
 165                             result.extend(frame.post_context)
 166             return result
 167         return []

 169     def render_label(self):
 170         data = {
 171             "attribute": self.data["attribute"],
 172             "value": self.data["value"],
 173             "match": MATCH_CHOICES[self.data["match"]],
 174         }
 175         return self.label.format(**data)

 177     def passes(self, event, state, **kwargs):
 178         attr = self.get_option("attribute")
 179         match = self.get_option("match")
 180         value = self.get_option("value")
 181 
 182         if not (attr and match and value):
 183             return False
 184 
 185         value = value.lower()
 186         attr = attr.lower()
 187 
 188         try:
 189             attribute_values = self._get_attribute_values(event, attr)
 190         except KeyError:
 191             attribute_values = []
 192 
 193         attribute_values = [six.text_type(v).lower() for v in attribute_values if v is not None]
 194 
 195         if match == MatchType.EQUAL:
 196             for a_value in attribute_values:
 197                 if a_value == value:
 198                     return True
 199             return False
 200 
 201         elif match == MatchType.NOT_EQUAL:
 202             for a_value in attribute_values:
 203                 if a_value == value:
 204                     return False
 205             return True
 206 
 207         elif match == MatchType.STARTS_WITH:
 208             for a_value in attribute_values:
 209                 if a_value.startswith(value):
 210                     return True
 211             return False
 212 
 213         elif match == MatchType.ENDS_WITH:
 214             for a_value in attribute_values:
 215                 if a_value.endswith(value):
 216                     return True
 217             return False
 218 
 219         elif match == MatchType.CONTAINS:
 220             for a_value in attribute_values:
 221                 if value in a_value:
 222                     return True
 223             return False
 224 
 225         elif match == MatchType.NOT_CONTAINS:
 226             for a_value in attribute_values:
 227                 if value in a_value:
 228                     return False
 229             return True
 230 
 231         elif match == MatchType.IS_SET:
 232             return bool(attribute_values)
 233 
 234         elif match == MatchType.NOT_SET:
 235             return not attribute_values
