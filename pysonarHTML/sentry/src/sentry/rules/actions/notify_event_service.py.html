  30 class NotifyEventServiceAction(EventAction):
  31     form_cls = NotifyEventServiceForm

  33     prompt = "Send a notification via an integration"

  35     def __init__(self, *args, **kwargs):
  36         super(NotifyEventServiceAction, self).__init__(*args, **kwargs)
  37         self.form_fields = {
  38             "service": {
  39                 "type": "choice",
  40                 "choices": [[i.slug, i.title] for i in self.get_services()],

  44     def after(self, event, state):
  45         service = self.get_option("service")
  46 
  47         extra = {"event_id": event.event_id}
  48         if not service:
  49             self.logger.info("rules.fail.is_configured", extra=extra)
  50             return
  51 
  52         plugin = None
  53         app = None
  54         try:
  55             app = SentryApp.objects.get(slug=service)
  56         except SentryApp.DoesNotExist:
  57             pass
  58 
  59         if app:
  60             kwargs = {"sentry_app": app}
  61             metrics.incr("notifications.sent", instance=app.slug, skip_internal=False)
  62             yield self.future(notify_sentry_app, **kwargs)
  63 
  64         try:
  65             plugin = plugins.get(service)
  66         except KeyError:
  67             if not app:
  68                 # If we can't find the sentry app OR plugin,
  69                 # we've removed the plugin no need to error, just skip.
  70                 extra["plugin"] = service
  71                 self.logger.info("rules.fail.plugin_does_not_exist", extra=extra)
  72                 return
  73 
  74         if plugin:
  75             if not plugin.is_enabled(self.project):
  76                 extra["project_id"] = self.project.id
  77                 self.logger.info("rules.fail.is_enabled", extra=extra)
  78                 return
  79 
  80             group = event.group
  81 
  82             if not plugin.should_notify(group=group, event=event):
  83                 extra["group_id"] = group.id
  84                 self.logger.info("rule.fail.should_notify", extra=extra)
  85                 return
  86 
  87             metrics.incr("notifications.sent", instance=plugin.slug, skip_internal=False)
  88             yield self.future(plugin.rule_notify)
  89 
  90     def get_sentry_app_services(self):
  91         apps = SentryApp.objects.filter(
  92             installations__organization_id=self.project.organization_id,
  93             is_alertable=True,
  94             installations__status=SentryAppInstallationStatus.INSTALLED,
  95         ).distinct()
  96         results = [SentryAppService(app) for app in apps]
  97         return results
  98 
  99     def get_plugins(self):
 100         from sentry.plugins.bases.notify import NotificationPlugin
 101 
 102         results = []
 103         for plugin in plugins.for_project(self.project, version=1):
 104             if not isinstance(plugin, NotificationPlugin):
 105                 continue
 106             results.append(PluginService(plugin))
 107 
 108         for plugin in plugins.for_project(self.project, version=2):
 109             for notifier in safe_execute(plugin.get_notifiers, _with_transaction=False) or ():
 110                 results.append(PluginService(notifier))
 111 
 112         return results
 113 
 114     def get_services(self):
 115         services = self.get_plugins()
 116         services += self.get_sentry_app_services()
 117         return services
 118 
 119     def get_form_instance(self):
 120         return self.form_cls(self.data, services=self.get_services())
