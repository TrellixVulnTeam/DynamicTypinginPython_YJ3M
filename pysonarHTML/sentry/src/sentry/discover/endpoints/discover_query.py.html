  27     def handle_results(self, snuba_results, requested_query, projects):
  28         if "project.name" in requested_query["selected_columns"]:

  30             snuba_results["meta"].insert(

  34                 snuba_results["meta"] = [
  35                     field for field in snuba_results["meta"] if field["name"] != "project.id"

  39                 if "project.id" in result:

  66     def do_query(self, projects, request, **kwargs):
  67         requested_query = deepcopy(kwargs)
  68 
  69         selected_columns = kwargs["selected_columns"]
  70         groupby_columns = kwargs["groupby"]
  71 
  72         if "project.name" in requested_query["selected_columns"]:
  73             selected_columns.remove("project.name")
  74             if "project.id" not in selected_columns:
  75                 selected_columns.append("project.id")
  76 
  77         if "project.name" in requested_query["groupby"]:
  78             groupby_columns.remove("project.name")
  79             if "project.id" not in groupby_columns:
  80                 groupby_columns.append("project.id")
  81 
  82         for aggregation in kwargs["aggregations"]:
  83             if aggregation[1] == "project.name":
  84                 aggregation[1] = "project.id"
  85 
  86         if not kwargs["aggregations"]:
  87 
  88             data_fn = partial(transform_aliases_and_query, referrer="discover", **kwargs)
  89             return self.paginate(
  90                 request=request,
  91                 on_results=lambda results: self.handle_results(results, requested_query, projects),
  92                 paginator=GenericOffsetPaginator(data_fn=data_fn),
  93                 max_per_page=1000,
  94             )
  95         else:
  96             snuba_results = transform_aliases_and_query(referrer="discover", **kwargs)
  97             return Response(
  98                 self.handle_results(snuba_results, requested_query, projects), status=200
  99             )

 101     def post(self, request, organization):
 102         if not features.has("organizations:discover", organization, actor=request.user):
 103             return Response(status=404)
 104 
 105         try:
 106             requested_projects = set(map(int, request.data.get("projects", [])))
 107         except (ValueError, TypeError):
 108             raise ResourceDoesNotExist()
 109         projects = self._get_projects_by_id(requested_projects, request, organization)
 110 
 111         serializer = DiscoverQuerySerializer(data=request.data)
 112 
 113         if not serializer.is_valid():
 114             return Response(serializer.errors, status=400)
 115 
 116         serialized = serializer.validated_data
 117 
 118         has_aggregations = len(serialized.get("aggregations")) > 0
 119 
 120         selected_columns = (
 121             serialized.get("conditionFields", []) + []
 122             if has_aggregations
 123             else serialized.get("fields", [])
 124         )
 125 
 126         projects_map = {}
 127         for project in projects:
 128             projects_map[project.id] = project.slug
 129 
 130         # Make sure that all selected fields are in the group by clause if there
 131         # are aggregations
 132         groupby = serialized.get("groupby") or []
 133         fields = serialized.get("fields") or []
 134         if has_aggregations:
 135             for field in fields:
 136                 if field not in groupby:
 137                     groupby.append(field)
 138 
 139         return self.do_query(
 140             projects=projects_map,
 141             start=serialized.get("start"),
 142             end=serialized.get("end"),
 143             groupby=groupby,
 144             selected_columns=selected_columns,
 145             conditions=serialized.get("conditions"),
 146             orderby=serialized.get("orderby"),
 147             limit=serialized.get("limit"),
 148             aggregations=serialized.get("aggregations"),
 149             rollup=serialized.get("rollup"),
 150             filter_keys={"project.id": projects_map.keys()},
 151             arrayjoin=serialized.get("arrayjoin"),
 152             request=request,
 153             turbo=serialized.get("turbo"),
 154         )
