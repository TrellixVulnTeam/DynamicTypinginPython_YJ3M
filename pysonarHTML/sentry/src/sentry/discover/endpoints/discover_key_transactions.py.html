  46 class KeyTransactionEndpoint(KeyTransactionBase):
  47     permission_classes = (KeyTransactionPermission,)
  48 
  49     def post(self, request, organization):
  50         """ Create a Key Transaction """
  51         if not self.has_feature(request, organization):
  52             return Response(status=404)
  53 
  54         project = self.get_project(request, organization)
  55 
  56         base_filter = {"organization": organization, "owner": request.user}
  57 
  58         with transaction.atomic():
  59             serializer = KeyTransactionSerializer(data=request.data, context=base_filter)
  60             if serializer.is_valid():
  61                 data = serializer.validated_data
  62                 base_filter["transaction"] = data["transaction"]
  63                 base_filter["project"] = project
  64 
  65                 if KeyTransaction.objects.filter(**base_filter).count() > 0:
  66                     return Response(status=204)
  67 
  68                 KeyTransaction.objects.create(**base_filter)
  69                 return Response(status=201)
  70             return Response(serializer.errors, status=400)
  71 
  72     def get(self, request, organization):
  73         """ Get the Key Transactions for a user """
  74         if not self.has_feature(request, organization):
  75             return Response(status=404)
  76 
  77         params = self.get_filter_params(request, organization)
  78         fields = request.GET.getlist("field")[:]
  79         orderby = self.get_orderby(request)
  80 
  81         queryset = KeyTransaction.objects.filter(organization=organization, owner=request.user)
  82 
  83         results = {}
  84         if queryset.exists():
  85             results = key_transaction_query(
  86                 fields,
  87                 request.GET.get("query"),
  88                 params,
  89                 orderby,
  90                 "discover.key_transactions",
  91                 queryset,
  92             )
  93 
  94         return Response(
  95             self.handle_results_with_meta(request, organization, params["project_id"], results),
  96             status=200,
  97         )
  98 
  99     def delete(self, request, organization):
 100         """ Remove a Key transaction for a user """
 101         if not self.has_feature(request, organization):
 102             return Response(status=404)
 103 
 104         project = self.get_project(request, organization)
 105         transaction = request.data["transaction"]
 106 
 107         try:
 108             model = KeyTransaction.objects.get(
 109                 transaction=transaction,
 110                 organization=organization,
 111                 project=project,
 112                 owner=request.user,
 113             )
 114         except KeyTransaction.DoesNotExist:
 115             return Response(status=204)
 116 
 117         model.delete()
 118 
 119         return Response(status=204)

 125     def get(self, request, organization):
 126         """ Get the Key Transactions for a user """
 127         if not self.has_feature(request, organization):
 128             return Response(status=404)
 129 
 130         queryset = KeyTransaction.objects.filter(organization=organization, owner=request.user)
 131 
 132         def get_event_stats(query_columns, query, params, rollup, reference_event=None):
 133             return key_transaction_timeseries_query(
 134                 selected_columns=query_columns,
 135                 query=query,
 136                 params=params,
 137                 rollup=rollup,
 138                 referrer="api.organization-event-stats.key-transactions",
 139                 queryset=queryset,
 140             )
 141 
 142         return Response(
 143             self.get_event_stats_data(request, organization, get_event_stats), status=200
 144         )
