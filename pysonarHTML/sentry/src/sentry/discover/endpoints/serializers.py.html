  89     def validate_conditions(self, value):
  90         # Handle error (exception_stacks), stack(exception_frames)
  91         return [self.get_condition(condition) for condition in value]

  93     def validate_aggregations(self, value):
  94         valid_functions = set(["count()", "uniq", "avg", "sum"])
  95         requested_functions = set(agg[0] for agg in value)
  96 
  97         if not requested_functions.issubset(valid_functions):
  98             invalid_functions = ", ".join((requested_functions - valid_functions))
  99 
 100             raise serializers.ValidationError(
 101                 u"Invalid aggregate function - {}".format(invalid_functions)
 102             )
 103 
 104         return value

 106     def get_array_field(self, field):
 107         pattern = r"^(error|stack)\..+"
 108         term = re.search(pattern, field)
 109         if term and SENTRY_SNUBA_MAP.get(field):
 110             return term
 111         return None

 113     def get_condition(self, condition):
 114         array_field = self.get_array_field(condition[0])
 115         has_equality_operator = condition[1] in ("=", "!=")
 116 
 117         # Cast boolean values to 1 / 0
 118         if isinstance(condition[2], bool):
 119             condition[2] = int(condition[2])
 120 
 121         # Strip double quotes on strings
 122         if isinstance(condition[2], six.string_types):
 123             match = re.search(r'^"(.*)"$', condition[2])
 124             if match:
 125                 condition[2] = match.group(1)
 126 
 127         # Apply has function to any array field if it's = / != and not part of arrayjoin
 128         if array_field and has_equality_operator and (array_field.group(1) != self.arrayjoin):
 129             value = condition[2]
 130 
 131             if isinstance(value, six.string_types):
 132                 value = u"'{}'".format(value)
 133 
 134             bool_value = 1 if condition[1] == "=" else 0
 135 
 136             return [["has", [array_field.group(0), value]], "=", bool_value]
 137 
 138         return condition

 173     def validate_projects(self, projects):
 174         organization = self.context["organization"]
 175 
 176         projects = set(projects)
 177         if projects == ALL_ACCESS_PROJECTS:
 178             return projects
 179 
 180         org_projects = set(
 181             Project.objects.filter(
 182                 organization=organization, id__in=projects, status=ProjectStatus.VISIBLE
 183             ).values_list("id", flat=True)
 184         )
 185         if projects != org_projects:
 186             raise PermissionDenied
 187 
 188         return projects

 190     def validate(self, data):
 191         query = {}
 192         query_keys = [
 193             "environment",
 194             "query",
 195             "fields",
 196             "conditions",
 197             "aggregations",
 198             "range",
 199             "start",
 200             "end",
 201             "orderby",
 202             "limit",
 203             "widths",
 204             "yAxis",
 205             "display",
 206         ]
 207 
 208         for key in query_keys:
 209             if data.get(key) is not None:
 210                 query[key] = data[key]
 211 
 212         version = data.get("version", 1)
 213         self.validate_version_fields(version, query)
 214         if version == 2:
 215             if len(query["fields"]) < 1:
 216                 raise serializers.ValidationError("You must include at least one field.")
 217 
 218         if data["projects"] == ALL_ACCESS_PROJECTS:
 219             data["projects"] = []
 220             query["all_projects"] = True
 221 
 222         if "query" in query:
 223             try:
 224                 get_filter(
 225                     query["query"],
 226                     {
 227                         "project_id": data["projects"],
 228                         "organization_id": self.context["organization"],
 229                     },
 230                 )
 231             except InvalidSearchQuery as err:
 232                 raise serializers.ValidationError("Cannot save invalid query: {}".format(err))
 233 
 234         return {
 235             "name": data["name"],
 236             "project_ids": data["projects"],
 237             "query": query,
 238             "version": version,
 239         }

 241     def validate_version_fields(self, version, query):
 242         try:
 243             not_allowed = self.disallowed_fields[version]
 244         except KeyError:
 245             raise serializers.ValidationError("Invalid version requested.")
 246         bad_fields = set(query.keys()) & not_allowed
 247         if bad_fields:
 248             raise serializers.ValidationError(
 249                 "You cannot use the %s attribute(s) with the selected version"
 250                 % ", ".join(bad_fields)
 251             )

 257     def validate(self, data):
 258         data = super(KeyTransactionSerializer, self).validate(data)
 259         base_filter = self.context.copy()
 260         # Limit the number of key transactions
 261         if KeyTransaction.objects.filter(**base_filter).count() >= MAX_KEY_TRANSACTIONS:
 262             raise serializers.ValidationError(
 263                 "At most {} Key Transactions can be added".format(MAX_KEY_TRANSACTIONS)
 264             )
 265         return data
