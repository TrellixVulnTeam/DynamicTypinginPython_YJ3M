  17 class DiscoverSavedQueriesEndpoint(OrganizationEndpoint):
  18     permission_classes = (DiscoverSavedQueryPermission,)
  19 
  20     def has_feature(self, organization, request):
  21         return features.has(
  22             "organizations:discover", organization, actor=request.user
  23         ) or features.has("organizations:discover-query", organization, actor=request.user)
  24 
  25     def get(self, request, organization):
  26         """
  27         List saved queries for organization
  28         """
  29         if not self.has_feature(organization, request):
  30             return self.respond(status=404)
  31 
  32         queryset = (
  33             DiscoverSavedQuery.objects.filter(organization=organization)
  34             .select_related("created_by")
  35             .prefetch_related("projects")
  36             .extra(select={"lower_name": "lower(name)"})
  37         )
  38         query = request.query_params.get("query")
  39         if query:
  40             tokens = tokenize_query(query)
  41             for key, value in six.iteritems(tokens):
  42                 if key == "name" or key == "query":
  43                     value = " ".join(value)
  44                     queryset = queryset.filter(name__icontains=value)
  45                 elif key == "version":
  46                     value = " ".join(value)
  47                     queryset = queryset.filter(version=value)
  48                 else:
  49                     queryset = queryset.none()
  50 
  51         sort_by = request.query_params.get("sortBy")
  52         if sort_by in ("name", "-name"):
  53             order_by = "-lower_name" if sort_by.startswith("-") else "lower_name"
  54         elif sort_by in ("dateCreated", "-dateCreated"):
  55             order_by = "-date_created" if sort_by.startswith("-") else "date_created"
  56         elif sort_by in ("dateUpdated", "-dateUpdated"):
  57             order_by = "-date_updated" if sort_by.startswith("-") else "date_updated"
  58         elif sort_by == "myqueries":
  59             order_by = [
  60                 Case(When(created_by_id=request.user.id, then=-1), default="created_by_id"),
  61                 "lower_name",
  62             ]
  63         else:
  64             order_by = "lower_name"
  65         if not isinstance(order_by, list):
  66             order_by = [order_by]
  67         queryset = queryset.order_by(*order_by)
  68 
  69         # Old discover expects all queries and uses this parameter.
  70         if request.query_params.get("all") == "1":
  71             saved_queries = list(queryset.all())
  72             return Response(serialize(saved_queries), status=200)
  73 
  74         def data_fn(offset, limit):
  75             return list(queryset[offset : offset + limit])
  76 
  77         return self.paginate(
  78             request=request,
  79             paginator=GenericOffsetPaginator(data_fn=data_fn),
  80             on_results=lambda x: serialize(x, request.user),
  81             default_per_page=25,
  82         )
  83 
  84     def post(self, request, organization):
  85         """
  86         Create a saved query
  87         """
  88         if not self.has_feature(organization, request):
  89             return self.respond(status=404)
  90 
  91         serializer = DiscoverSavedQuerySerializer(
  92             data=request.data, context={"organization": organization}
  93         )
  94 
  95         if not serializer.is_valid():
  96             return Response(serializer.errors, status=400)
  97 
  98         data = serializer.validated_data
  99 
 100         model = DiscoverSavedQuery.objects.create(
 101             organization=organization,
 102             name=data["name"],
 103             query=data["query"],
 104             version=data["version"],
 105             created_by=request.user if request.user.is_authenticated() else None,
 106         )
 107 
 108         model.set_projects(data["project_ids"])
 109 
 110         return Response(serialize(model), status=201)
