  23 class DiscoverSavedQuery(Model):
  24     """
  25     A saved Discover query
  26     """
  27 
  28     __core__ = False
  29 
  30     projects = models.ManyToManyField("sentry.Project", through=DiscoverSavedQueryProject)
  31     organization = FlexibleForeignKey("sentry.Organization")
  32     created_by = FlexibleForeignKey("sentry.User", null=True, on_delete=models.SET_NULL)
  33     name = models.CharField(max_length=255)
  34     query = JSONField()
  35     version = models.IntegerField(null=True)
  36     date_created = models.DateTimeField(auto_now_add=True)
  37     date_updated = models.DateTimeField(auto_now=True)
  38 
  39     class Meta:
  40         app_label = "sentry"
  41         db_table = "sentry_discoversavedquery"
  42 
  43     __repr__ = sane_repr("organization_id", "created_by", "name")
  44 
  45     def set_projects(self, project_ids):
  46         with transaction.atomic():
  47             DiscoverSavedQueryProject.objects.filter(discover_saved_query=self).exclude(
  48                 project__in=project_ids
  49             ).delete()
  50 
  51             existing_project_ids = DiscoverSavedQueryProject.objects.filter(
  52                 discover_saved_query=self
  53             ).values_list("project", flat=True)
  54 
  55             new_project_ids = list(set(project_ids) - set(existing_project_ids))
  56 
  57             DiscoverSavedQueryProject.objects.bulk_create(
  58                 [
  59                     DiscoverSavedQueryProject(project_id=project_id, discover_saved_query=self)
  60                     for project_id in new_project_ids
  61                 ]
  62             )
