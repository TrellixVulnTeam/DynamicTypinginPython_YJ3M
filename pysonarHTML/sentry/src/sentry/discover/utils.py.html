  13 def parse_columns_in_functions(col, context=None, index=None):
  14     """
  15     Checks expressions for arguments that should be considered a column while
  16     ignoring strings that represent clickhouse function names
  17 
  18     if col is a list, means the expression has functions and we need
  19     to parse for arguments that should be considered column names.
  20 
  21     Assumptions here:
  22      * strings that represent clickhouse function names are always followed by a list or tuple
  23      * strings that are quoted with single quotes are used as string literals for CH
  24      * otherwise we should attempt to get the snuba column name (or custom tag)
  25     """
  26 
  27     function_name_index = get_function_index(col)
  28 
  29     if function_name_index is not None:
  30         # if this is non zero, that means there are strings before this index
  31         # that should be converted to snuba column names
  32         # e.g. ['func1', ['column', 'func2', ['arg1']]]
  33         if function_name_index > 0:
  34             for i in six.moves.xrange(0, function_name_index):
  35                 if context is not None:
  36                     context[i] = get_snuba_column_name(col[i])
  37 
  38         args = col[function_name_index + 1]
  39 
  40         # check for nested functions in args
  41         if get_function_index(args):
  42             # look for columns
  43             return parse_columns_in_functions(args, args)
  44 
  45         # check each argument for column names
  46         else:
  47             for (i, arg) in enumerate(args):
  48                 parse_columns_in_functions(arg, args, i)
  49     else:
  50         # probably a column name
  51         if context is not None and index is not None:
  52             context[index] = get_snuba_column_name(col)

  55 def transform_aliases_and_query(**kwargs):
  56     """
  57     Convert aliases in selected_columns, groupby, aggregation, conditions,
  58     orderby and arrayjoin fields to their internal Snuba format and post the
  59     query to Snuba. Convert back translated aliases before returning snuba
  60     results.
  61 
  62     :deprecated: This method is deprecated. You should use sentry.snuba.discover instead.
  63     """
  64 
  65     arrayjoin_map = {"error": "exception_stacks", "stack": "exception_frames"}
  66 
  67     translated_columns = {}
  68     derived_columns = set()
  69 
  70     selected_columns = kwargs.get("selected_columns")
  71     groupby = kwargs.get("groupby")
  72     aggregations = kwargs.get("aggregations")
  73     conditions = kwargs.get("conditions")
  74     filter_keys = kwargs["filter_keys"]
  75     arrayjoin = kwargs.get("arrayjoin")
  76     orderby = kwargs.get("orderby")
  77     having = kwargs.get("having", [])
  78     dataset = Dataset.Events
  79 
  80     if selected_columns:
  81         for (idx, col) in enumerate(selected_columns):
  82             if isinstance(col, list):
  83                 # if list, means there are potentially nested functions and need to
  84                 # iterate and translate potential columns
  85                 parse_columns_in_functions(col)
  86                 selected_columns[idx] = col
  87                 translated_columns[col[2]] = col[2]
  88                 derived_columns.add(col[2])
  89             else:
  90                 name = get_snuba_column_name(col)
  91                 selected_columns[idx] = name
  92                 translated_columns[name] = col
  93 
  94     if groupby:
  95         for (idx, col) in enumerate(groupby):
  96             if col not in derived_columns:
  97                 name = get_snuba_column_name(col)
  98             else:
  99                 name = col
 100 
 101             groupby[idx] = name
 102             translated_columns[name] = col
 103 
 104     for aggregation in aggregations or []:
 105         derived_columns.add(aggregation[2])
 106         if isinstance(aggregation[1], six.string_types):
 107             aggregation[1] = get_snuba_column_name(aggregation[1])
 108         elif isinstance(aggregation[1], (set, tuple, list)):
 109             aggregation[1] = [get_snuba_column_name(col) for col in aggregation[1]]
 110 
 111     for col in filter_keys.keys():
 112         name = get_snuba_column_name(col)
 113         filter_keys[name] = filter_keys.pop(col)
 114 
 115     if conditions:
 116         aliased_conditions = []
 117         for condition in conditions:
 118             field = condition[0]
 119             if not isinstance(field, (list, tuple)) and field in derived_columns:
 120                 having.append(condition)
 121             else:
 122                 aliased_conditions.append(condition)
 123         kwargs["conditions"] = aliased_conditions
 124 
 125     if having:
 126         kwargs["having"] = having
 127 
 128     if orderby:
 129         orderby = orderby if isinstance(orderby, (list, tuple)) else [orderby]
 130         translated_orderby = []
 131 
 132         for field_with_order in orderby:
 133             field = field_with_order.lstrip("-")
 134             translated_orderby.append(
 135                 u"{}{}".format(
 136                     "-" if field_with_order.startswith("-") else "",
 137                     field if field in derived_columns else get_snuba_column_name(field),
 138                 )
 139             )
 140 
 141         kwargs["orderby"] = translated_orderby
 142 
 143     kwargs["arrayjoin"] = arrayjoin_map.get(arrayjoin, arrayjoin)
 144     kwargs["dataset"] = dataset
 145 
 146     result = aliased_query(**kwargs)
 147 
 148     snuba_filter = eventstore.Filter(
 149         rollup=kwargs.get("rollup"),
 150         start=kwargs.get("start"),
 151         end=kwargs.get("end"),
 152         orderby=kwargs.get("orderby"),
 153     )
 154     return transform_results(result, translated_columns, snuba_filter)
