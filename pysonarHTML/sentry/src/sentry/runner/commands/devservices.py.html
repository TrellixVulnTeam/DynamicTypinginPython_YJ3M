  33 def ensure_interface(ports):
  34     # If there is no interface specified, make sure the
  35     # default interface is 127.0.0.1
  36     rv = {}
  37     for k, v in ports.items():
  38         if not isinstance(v, tuple):
  39             v = ("127.0.0.1", v)
  40         rv[k] = v
  41     return rv

  57 def attach(project, fast, service):
  58     """
  59     Run a single devservice in foreground, as opposed to `up` which runs all of
  60     them in the background.
  61 
  62     Accepts a single argument, the name of the service to spawn. The service
  63     will run with output printed to your terminal, and the ability to kill it
  64     with ^C. This is used in devserver.
  65 
  66     Note: This does not update images, you will have to use `devservices up`
  67     for that.
  68     """
  69 
  70     os.environ["SENTRY_SKIP_BACKEND_VALIDATION"] = "1"
  71 
  72     from sentry.runner import configure
  73 
  74     configure()
  75 
  76     client = get_docker_client()
  77     containers = _prepare_containers(project, silent=True)
  78     if service not in containers:
  79         raise click.ClickException("Service `{}` is not known or not enabled.".format(service))
  80 
  81     container = _start_service(client, service, containers, project, fast=fast, always_start=True)
  82 
  83     def exit_handler(*_):
  84         try:
  85             click.echo("Stopping {}".format(service))
  86             container.stop()
  87             click.echo("Removing {}".format(service))
  88             container.remove()
  89         except KeyboardInterrupt:
  90             pass
  91 
  92     signal.signal(signal.SIGINT, exit_handler)
  93     signal.signal(signal.SIGTERM, exit_handler)
  94 
  95     for line in container.logs(stream=True, since=int(time.time() - 20)):
  96         click.echo(line, nl=False)

 104 def up(services, project, exclude, fast):
 105     """
 106     Run/update dependent services.
 107 
 108     The default is everything, however you may pass positional arguments to specify
 109     an explicit list of services to bring up.
 110 
 111     You may also exclude services, for example: --exclude redis --exclude postgres.
 112     """
 113     os.environ["SENTRY_SKIP_BACKEND_VALIDATION"] = "1"
 114 
 115     from sentry.runner import configure
 116 
 117     configure()
 118 
 119     containers = _prepare_containers(project, silent=True)
 120 
 121     for service in exclude:
 122         if service not in containers:
 123             click.secho(
 124                 "Service `{}` is not known or not enabled.\n".format(service), err=True, fg="red",
 125             )
 126             click.secho(
 127                 "Services that are available:\n" + "\n".join(containers.keys()) + "\n", err=True,
 128             )
 129             raise click.Abort()
 130 
 131     if services:
 132         selected_containers = {}
 133         for service in services:
 134             if service not in containers:
 135                 click.secho(
 136                     "Service `{}` is not known or not enabled.\n".format(service),
 137                     err=True,
 138                     fg="red",
 139                 )
 140                 click.secho(
 141                     "Services that are available:\n" + "\n".join(containers.keys()) + "\n",
 142                     err=True,
 143                 )
 144                 raise click.Abort()
 145             selected_containers[service] = containers[service]
 146         containers = selected_containers
 147 
 148     if fast:
 149         click.secho(
 150             "> Warning! Fast mode completely eschews any image updating, so services may be stale.",
 151             err=True,
 152             fg="red",
 153         )
 154 
 155     client = get_docker_client()
 156     get_or_create(client, "network", project)
 157 
 158     for name, container_options in containers.items():
 159         if name in exclude:
 160             continue
 161         _start_service(client, name, containers, project, fast=fast)

 164 def _prepare_containers(project, silent=False):
 165     from django.conf import settings
 166     from sentry import options as sentry_options
 167 
 168     containers = {}
 169 
 170     for name, options in settings.SENTRY_DEVSERVICES.items():
 171         options = options.copy()
 172         test_fn = options.pop("only_if", None)
 173         if test_fn and not test_fn(settings, sentry_options):
 174             if not silent:
 175                 click.secho(
 176                     "! Skipping {} due to only_if condition".format(name), err=True, fg="cyan"
 177                 )
 178             continue
 179 
 180         options["network"] = project
 181         options["detach"] = True
 182         options["name"] = project + "_" + name
 183         options.setdefault("ports", {})
 184         options.setdefault("environment", {})
 185         options.setdefault("restart_policy", {"Name": "on-failure"})
 186         options["ports"] = ensure_interface(options["ports"])
 187         containers[name] = options
 188 
 189     # keys are service names
 190     # a service has 1 container exactly, the container name being value["name"]
 191     return containers

 194 def _start_service(client, name, containers, project, fast=False, always_start=False):
 195     from django.conf import settings
 196     import docker
 197 
 198     options = containers[name]
 199 
 200     # HACK(mattrobenolt): special handle snuba backend because it needs to
 201     # handle different values based on the eventstream backend
 202     # For snuba, we can't run the full suite of devserver, but can only
 203     # run the api.
 204     if name == "snuba" and "snuba" in settings.SENTRY_EVENTSTREAM:
 205         options["environment"].pop("DEFAULT_BROKERS", None)
 206         options["command"] = ["devserver", "--no-workers"]
 207 
 208     for key, value in options["environment"].items():
 209         options["environment"][key] = value.format(containers=containers)
 210 
 211     pull = options.pop("pull", False)
 212     if not fast:
 213         if pull:
 214             click.secho("> Pulling image '%s'" % options["image"], err=True, fg="green")
 215             client.images.pull(options["image"])
 216         else:
 217             # We want make sure to pull everything on the first time,
 218             # (the image doesn't exist), regardless of pull=True.
 219             try:
 220                 client.images.get(options["image"])
 221             except docker.errors.NotFound:
 222                 click.secho("> Pulling image '%s'" % options["image"], err=True, fg="green")
 223                 client.images.pull(options["image"])
 224 
 225     for mount in options.get("volumes", {}).keys():
 226         if "/" not in mount:
 227             get_or_create(client, "volume", project + "_" + mount)
 228             options["volumes"][project + "_" + mount] = options["volumes"].pop(mount)
 229 
 230     listening = ""
 231     if options["ports"]:
 232         listening = "(listening: %s)" % ", ".join(map(text_type, options["ports"].values()))
 233 
 234     # If a service is associated with the devserver, then do not run the created container.
 235     # This was mainly added since it was not desirable for reverse_proxy to occupy port 8000 on the
 236     # first "devservices up".
 237     # See https://github.com/getsentry/sentry/pull/18362#issuecomment-616785458
 238     with_devserver = options.pop("with_devserver", False)
 239 
 240     # Two things call _start_service.
 241     # devservices up, and devservices attach.
 242     # Containers that should be started on-demand with devserver
 243     # should ONLY be started via the latter, which sets `always_start`.
 244     if with_devserver and not always_start:
 245         click.secho(
 246             "> Not starting container '%s' because it should be started on-demand with devserver."
 247             % options["name"],
 248             fg="yellow",
 249         )
 250         # XXX: if always_start=False, do not expect to have a container returned 100% of the time.
 251         return None
 252 
 253     container = None
 254     try:
 255         container = client.containers.get(options["name"])
 256     except docker.errors.NotFound:
 257         pass
 258 
 259     if container is not None:
 260         # devservices which are marked with pull True will need their containers
 261         # to be recreated with the freshly pulled image.
 262         should_reuse_container = not pull
 263 
 264         # Except if the container is started as part of devserver we should reuse it.
 265         # Or, if we're in fast mode (devservices up --fast)
 266         if with_devserver or fast:
 267             should_reuse_container = True
 268 
 269         if should_reuse_container:
 270             click.secho(
 271                 "> Starting EXISTING container '%s' %s" % (container.name, listening),
 272                 err=True,
 273                 fg="yellow",
 274             )
 275             # Note that if the container is already running, this will noop.
 276             # This makes repeated `devservices up` quite fast.
 277             container.start()
 278             return container
 279 
 280         click.secho("> Stopping container '%s'" % container.name, err=True, fg="yellow")
 281         container.stop()
 282         click.secho("> Removing container '%s'" % container.name, err=True, fg="yellow")
 283         container.remove()
 284 
 285     click.secho("> Creating container '%s'" % options["name"], err=True, fg="yellow")
 286     container = client.containers.create(**options)
 287     click.secho("> Starting container '%s' %s" % (container.name, listening), err=True, fg="yellow")
 288     container.start()
 289     return container

 332 
 333     configure()
 334 
 335     containers = _prepare_containers(project, silent=True)
 336 
 337     if services:
 338         selected_containers = {}
 339         for service in services:
 340             # XXX: This code is also fairly duplicated in here at this point, so dedupe in the future.
 341             if service not in containers:
 342                 click.secho(
 343                     "Service `{}` is not known or not enabled.\n".format(service),
 344                     err=True,
 345                     fg="red",
 346                 )
 347                 click.secho(
 348                     "Services that are available:\n" + "\n".join(containers.keys()) + "\n",
 349                     err=True,
 350                 )
 351                 raise click.Abort()
 352             selected_containers[service] = containers[service]
 353         containers = selected_containers

 355     click.confirm(

 357 This will delete these services and all of their data:
 358 
 359 %s
 360 
 361 Are you sure you want to continue?"""
 362         % "\n".join(containers.keys()),
 363         abort=True,
 364     )
 365 
 366     for service_name, container_options in containers.items():
 367         try:
 368             container = client.containers.get(container_options["name"])
 369         except docker.errors.NotFound:

 378         container.stop()
 379         click.secho("> Removing '%s' container" % container_options["name"], err=True, fg="red")
 380         container.remove()
 381 
 382     prefix = project + "_"
 383 
 384     for volume in client.volumes.list():
 385         if volume.name.startswith(prefix):
 386             if not services or volume.name[len(prefix) :] in services:
 387                 click.secho("> Removing '%s' volume" % volume.name, err=True, fg="red")
 388                 volume.remove()
 389 
 390     if not services:
 391         try:
 392             network = client.networks.get(project)
 393         except docker.errors.NotFound:
 394             pass
 395         else:
 396             click.secho("> Removing '%s' network" % network.name, err=True, fg="red")
 397             network.remove()
