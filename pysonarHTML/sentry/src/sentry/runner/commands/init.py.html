  15     from sentry.runner.settings import discover_configs, generate_settings
  16 
  17     if directory is not None:
  18         os.environ["SENTRY_CONF"] = directory

  20     directory, py, yaml = discover_configs()
  21 
  22     # In this case, the config is pointing directly to a file, so we
  23     # must maintain old behavior, and just abort
  24     if yaml is None and os.path.isfile(py):
  25         # TODO: Link to docs explaining about new behavior of SENTRY_CONF?
  26         raise click.ClickException(
  27             "Found legacy '%s' file, so aborting." % click.format_filename(py)
  28         )
  29 
  30     if yaml is None:
  31         raise click.ClickException("DIRECTORY must not be a file.")
  32 
  33     if directory and not os.path.exists(directory):
  34         os.makedirs(directory)
  35 
  36     py_contents, yaml_contents = generate_settings(dev)
  37 
  38     if os.path.isfile(yaml):
  39         click.confirm(
  40             "File already exists at '%s', overwrite?" % click.format_filename(yaml), abort=True
  41         )
  42 
  43     with click.open_file(yaml, "w") as fp:
  44         fp.write(yaml_contents)
  45 
  46     if os.path.isfile(py):
  47         click.confirm(
  48             "File already exists at '%s', overwrite?" % click.format_filename(py), abort=True
  49         )
