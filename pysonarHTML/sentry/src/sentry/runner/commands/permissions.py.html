   8 def user_param_to_user(value):
   9     from sentry.utils.auth import find_users
  10 
  11     users = find_users(value)
  12     if not users:
  13         raise click.ClickException(u"No user matching `{}`".format(value))
  14     if len(users) > 1:
  15         raise click.ClickException(u"Found more than one user matching `{}`".format(value))
  16     user = users[0]
  17     if not user.is_superuser:
  18         raise click.ClickException(
  19             u"User `{}` does not have superuser status".format(user.username)
  20         )
  21     return user
