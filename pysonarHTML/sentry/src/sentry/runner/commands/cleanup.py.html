  19 def get_project(value):
  20     from sentry.models import Project
  21 
  22     try:
  23         if value.isdigit():
  24             return int(value)
  25         if "/" not in value:
  26             return None
  27         org, proj = value.split("/", 1)
  28         return Project.objects.get(organization__slug=org, slug=proj).id
  29     except Project.DoesNotExist:
  30         return None

  41 def multiprocess_worker(task_queue):
  42     # Configure within each Process
  43     import logging
  44     from sentry.utils.imports import import_string
  45 
  46     logger = logging.getLogger("sentry.cleanup")
  47 
  48     configured = False
  49 
  50     while True:
  51         j = task_queue.get()
  52         if j == _STOP_WORKER:
  53             task_queue.task_done()
  54             return
  55 
  56         # On first task, configure Sentry environment
  57         if not configured:
  58             from sentry.runner import configure
  59 
  60             configure()
  61 
  62             from sentry import models
  63             from sentry import deletions
  64             from sentry import similarity
  65 
  66             skip_models = [
  67                 # Handled by other parts of cleanup
  68                 models.EventAttachment,
  69                 models.UserReport,
  70                 models.Group,
  71                 models.GroupEmailThread,
  72                 models.GroupRuleStatus,
  73                 # Handled by TTL
  74                 similarity.features,
  75             ] + [b[0] for b in EXTRA_BULK_QUERY_DELETES]
  76 
  77             configured = True
  78 
  79         model, chunk = j
  80         model = import_string(model)
  81 
  82         try:
  83             task = deletions.get(
  84                 model=model,
  85                 query={"id__in": chunk},
  86                 skip_models=skip_models,
  87                 transaction_id=uuid4().hex,
  88             )
  89 
  90             while True:
  91                 if not task.chunk():
  92                     break
  93         except Exception as e:
  94             logger.exception(e)
  95         finally:
  96             task_queue.task_done()

 122 def cleanup(days, project, concurrency, silent, model, router, timed):
 123     """Delete a portion of trailing data based on creation date.
 124 
 125     All data that is older than `--days` will be deleted.  The default for
 126     this is 30 days.  In the default setting all projects will be truncated
 127     but if you have a specific project you want to limit this to this can be
 128     done with the `--project` flag which accepts a project ID or a string
 129     with the form `org/project` where both are slugs.
 130     """
 131     if concurrency < 1:
 132         click.echo("Error: Minimum concurrency is 1", err=True)
 133         raise click.Abort()
 134 
 135     os.environ["_SENTRY_CLEANUP"] = "1"
 136 
 137     # Make sure we fork off multiprocessing pool
 138     # before we import or configure the app
 139     from multiprocessing import Process, JoinableQueue as Queue
 140 
 141     pool = []
 142     task_queue = Queue(1000)
 143     for _ in xrange(concurrency):
 144         p = Process(target=multiprocess_worker, args=(task_queue,))
 145         p.daemon = True
 146         p.start()
 147         pool.append(p)
 148 
 149     from sentry.runner import configure
 150 
 151     configure()
 152 
 153     from django.db import router as db_router
 154     from sentry.app import nodestore
 155     from sentry.db.deletion import BulkDeleteQuery
 156     from sentry import models
 157     from sentry.data_export.models import ExportedData
 158 
 159     if timed:
 160         import time
 161         from sentry.utils import metrics
 162 
 163         start_time = time.time()
 164 
 165     # list of models which this query is restricted to
 166     model_list = {m.lower() for m in model}
 167 
 168     def is_filtered(model):
 169         if router is not None and db_router.db_for_write(model) != router:
 170             return True
 171         if not model_list:
 172             return False
 173         return model.__name__.lower() not in model_list
 174 
 175     # Deletions that use `BulkDeleteQuery` (and don't need to worry about child relations)
 176     # (model, datetime_field, order_by)
 177     BULK_QUERY_DELETES = [
 178         (models.EventAttachment, "date_added", None),
 179         (models.UserReport, "date_added", None),
 180         (models.GroupEmailThread, "date", None),
 181         (models.GroupRuleStatus, "date_added", None),
 182     ] + EXTRA_BULK_QUERY_DELETES
 183 
 184     # Deletions that use the `deletions` code path (which handles their child relations)
 185     # (model, datetime_field, order_by)
 186     DELETES = ((models.Group, "last_seen", "last_seen"),)
 187 
 188     if not silent:
 189         click.echo("Removing expired values for LostPasswordHash")
 190 
 191     if is_filtered(models.LostPasswordHash):
 192         if not silent:
 193             click.echo(">> Skipping LostPasswordHash")
 194     else:
 195         models.LostPasswordHash.objects.filter(
 196             date_added__lte=timezone.now() - timedelta(hours=48)
 197         ).delete()
 198 
 199     if not silent:
 200         click.echo("Removing expired values for OrganizationMember")
 201 
 202     if is_filtered(models.OrganizationMember):
 203         if not silent:
 204             click.echo(">> Skipping OrganizationMember")
 205     else:
 206         expired_threshold = timezone.now() - timedelta(days=days)
 207         models.OrganizationMember.delete_expired(expired_threshold)
 208 
 209     for model in [models.ApiGrant, models.ApiToken]:
 210         if not silent:
 211             click.echo(u"Removing expired values for {}".format(model.__name__))
 212 
 213         if is_filtered(model):
 214             if not silent:
 215                 click.echo(u">> Skipping {}".format(model.__name__))
 216         else:
 217             queryset = model.objects.filter(
 218                 expires_at__lt=(timezone.now() - timedelta(days=API_TOKEN_TTL_IN_DAYS))
 219             )
 220 
 221             # SentryAppInstallations are associated to ApiTokens. We're okay
 222             # with these tokens sticking around so that the Integration can
 223             # refresh them, but all other non-associated tokens should be
 224             # deleted.
 225             if model is models.ApiToken:
 226                 queryset = queryset.filter(sentry_app_installation__isnull=True)
 227 
 228             queryset.delete()
 229 
 230     if not silent:
 231         click.echo("Removing expired files associated with ExportedData")
 232 
 233     if is_filtered(ExportedData):
 234         if not silent:
 235             click.echo(">> Skipping ExportedData files")
 236     else:
 237         queryset = ExportedData.objects.filter(date_expired__lt=(timezone.now()))
 238         for item in queryset:
 239             item.delete_file()
 240 
 241     project_id = None
 242     if project:
 243         click.echo("Bulk NodeStore deletion not available for project selection", err=True)
 244         project_id = get_project(project)
 245         if project_id is None:
 246             click.echo("Error: Project not found", err=True)
 247             raise click.Abort()
 248     else:
 249         if not silent:
 250             click.echo("Removing old NodeStore values")
 251 
 252         cutoff = timezone.now() - timedelta(days=days)
 253         try:
 254             nodestore.cleanup(cutoff)
 255         except NotImplementedError:
 256             click.echo("NodeStore backend does not support cleanup operation", err=True)
 257 
 258     for bqd in BULK_QUERY_DELETES:
 259         if len(bqd) == 4:
 260             model, dtfield, order_by, chunk_size = bqd
 261         else:
 262             chunk_size = 10000
 263             model, dtfield, order_by = bqd
 264 
 265         if not silent:
 266             click.echo(
 267                 u"Removing {model} for days={days} project={project}".format(
 268                     model=model.__name__, days=days, project=project or "*"
 269                 )
 270             )
 271         if is_filtered(model):
 272             if not silent:
 273                 click.echo(">> Skipping %s" % model.__name__)
 274         else:
 275             BulkDeleteQuery(
 276                 model=model, dtfield=dtfield, days=days, project_id=project_id, order_by=order_by
 277             ).execute(chunk_size=chunk_size)
 278 
 279     for model, dtfield, order_by in DELETES:
 280         if not silent:
 281             click.echo(
 282                 u"Removing {model} for days={days} project={project}".format(
 283                     model=model.__name__, days=days, project=project or "*"
 284                 )
 285             )
 286 
 287         if is_filtered(model):
 288             if not silent:
 289                 click.echo(">> Skipping %s" % model.__name__)
 290         else:
 291             imp = ".".join((model.__module__, model.__name__))
 292 
 293             q = BulkDeleteQuery(
 294                 model=model, dtfield=dtfield, days=days, project_id=project_id, order_by=order_by
 295             )
 296 
 297             for chunk in q.iterator(chunk_size=100):
 298                 task_queue.put((imp, chunk))
 299 
 300             task_queue.join()
 301 
 302     # Clean up FileBlob instances which are no longer used and aren't super
 303     # recent (as there could be a race between blob creation and reference)
 304     if not silent:
 305         click.echo("Cleaning up unused FileBlob references")
 306     if is_filtered(models.FileBlob):
 307         if not silent:
 308             click.echo(">> Skipping FileBlob")
 309     else:
 310         cleanup_unused_files(silent)
 311 
 312     # Shut down our pool
 313     for _ in pool:
 314         task_queue.put(_STOP_WORKER)
 315 
 316     # And wait for it to drain
 317     for p in pool:
 318         p.join()
 319 
 320     if timed:
 321         duration = int(time.time() - start_time)
 322         metrics.timing("cleanup.duration", duration, instance=router, sample_rate=1.0)
 323         click.echo("Clean up took %s second(s)." % duration)

 326 def cleanup_unused_files(quiet=False):
 327     """
 328     Remove FileBlob's (and thus the actual files) if they are no longer
 329     referenced by any File.
 330 
 331     We set a minimum-age on the query to ensure that we don't try to remove
 332     any blobs which are brand new and potentially in the process of being
 333     referenced.
 334     """
 335     from sentry.models import File, FileBlob, FileBlobIndex
 336 
 337     if quiet:
 338         from sentry.utils.query import RangeQuerySetWrapper
 339     else:
 340         from sentry.utils.query import RangeQuerySetWrapperWithProgressBar as RangeQuerySetWrapper
 341 
 342     cutoff = timezone.now() - timedelta(days=1)
 343     queryset = FileBlob.objects.filter(timestamp__lte=cutoff)
 344 
 345     for blob in RangeQuerySetWrapper(queryset):
 346         if FileBlobIndex.objects.filter(blob=blob).exists():
 347             continue
 348         if File.objects.filter(blob=blob).exists():
 349             continue
 350         blob.delete()
