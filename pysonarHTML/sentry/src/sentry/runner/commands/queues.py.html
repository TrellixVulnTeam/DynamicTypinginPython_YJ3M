  22     if backend is None:
  23         raise click.ClickException("unknown broker type")

  29     else:
  30         queues = sorted(queues, reverse=reverse)

  45     if get_queue_by_name(queue) is None:

  48     if backend is None:
  49         raise click.ClickException("unknown broker type")

  63     click.echo("Poof, %d messages deleted" % backend.purge_queue(queue), err=True)
