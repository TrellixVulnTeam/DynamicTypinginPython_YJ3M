  36 
  37     try:
  38         key = manager.lookup_key(option)

  40         raise click.ClickException("unknown option: %s" % option)
  41     value = manager.get(key.name)
  42     if silent:
  43         click.echo(value)

  45     # TODO(mattrobenolt): Add help to option keys
  46     # if key.help:
