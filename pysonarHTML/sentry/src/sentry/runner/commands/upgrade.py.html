  29 def _has_applied_django_migration(connection, app_name, migration):
  30     cursor = connection.cursor()
  31     try:
  32         cursor.execute(
  33             "SELECT 1 FROM django_migrations WHERE app = %s AND name = %s LIMIT 1",
  34             [app_name, migration],
  35         )
  36     except ProgrammingError as exc:
  37         if 'relation "django_migrations" does not exist' in six.text_type(exc):
  38             return False
  39         raise
  40     else:
  41         return bool(cursor.fetchall())
  42     finally:
  43         cursor.close()

  46 def _has_applied_south_migration(connection, app_name, migration):

  48         return True

  60 def _fake_django_migration(connection, app_name, migration, verbosity=0):
  61     if _has_applied_django_migration(connection, app_name, migration):
  62         return False
  63 
  64     if verbosity:
  65         click.echo("Faking migration for {}.{}".format(app_name, migration))
  66 
  67     cursor = connection.cursor()
  68     try:
  69         cursor.execute(
  70             "INSERT INTO django_migrations (app, name, applied) VALUES (%s, %s, NOW())",
  71             [app_name, migration],
  72         )
  73     finally:
  74         cursor.close()

  88 def _migrate_from_south(verbosity):
  89     from django.db.migrations.recorder import MigrationRecorder
  90 
  91     connection = connections["default"]
  92 
  93     if not _has_south_history(connection):
  94         return
  95 
  96     # ensure the django_migrations relation exists
  97     recorder = MigrationRecorder(connection)
  98     recorder.ensure_schema()
  99 
 100     for django_app_name, django_migration in DJANGO_MIGRATIONS:
 101         _fake_django_migration(connection, django_app_name, django_migration, verbosity)
 102 
 103     for (
 104         south_app_name,
 105         south_migration,
 106         django_app_name,
 107         django_migration,
 108         south_migration_required,
 109         south_migration_required_error,
 110     ) in settings.SOUTH_MIGRATION_CONVERSIONS:
 111         if _has_applied_south_migration(connection, south_app_name, south_migration):
 112             _fake_django_migration(connection, django_app_name, django_migration)
 113         elif south_migration_required:
 114             raise Exception(south_migration_required_error)

 117 def _upgrade(interactive, traceback, verbosity, repair, with_nodestore):
 118     from django.core.management import call_command as dj_call_command
 119 
 120     # migrate legacy south history into new django migrations automatically
 121     _migrate_from_south(verbosity)
 122 
 123     dj_call_command(
 124         "migrate",
 125         interactive=interactive,
 126         traceback=traceback,
 127         verbosity=verbosity,
 128         migrate=True,
 129         merge=True,
 130         ignore_ghost_migrations=True,
 131     )
 132 
 133     if with_nodestore:
 134         from sentry import nodestore
 135 
 136         nodestore.bootstrap()
 137 
 138     if repair:
 139         from sentry.runner import call_command
 140 
 141         call_command("sentry.runner.commands.repair.repair")

 160 def upgrade(ctx, verbosity, traceback, noinput, lock, no_repair, with_nodestore):
 161     "Perform any pending database migrations and upgrades."
 162 
 163     if lock:
 164         from sentry.app import locks
 165         from sentry.utils.locking import UnableToAcquireLock
 166 
 167         lock = locks.get("upgrade", duration=0)
 168         try:
 169             with lock.acquire():
 170                 _upgrade(not noinput, traceback, verbosity, not no_repair, with_nodestore)
 171         except UnableToAcquireLock:
 172             raise click.ClickException("Unable to acquire `upgrade` lock.")
 173     else:
 174         _upgrade(not noinput, traceback, verbosity, not no_repair, with_nodestore)
