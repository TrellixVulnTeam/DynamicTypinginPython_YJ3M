  19 def sort_dependencies():
  20     """
  21     Similar to Django's except that we discard the important of natural keys
  22     when sorting dependencies (i.e. it works without them).
  23     """
  24     from django.apps import apps
  25 
  26     # Process the list of models, and get the list of dependencies
  27     model_dependencies = []
  28     models = set()
  29     for app_config in apps.get_app_configs():
  30         model_list = app_config.get_models()
  31 
  32         for model in model_list:
  33             models.add(model)
  34             # Add any explicitly defined dependencies
  35             if hasattr(model, "natural_key"):
  36                 deps = getattr(model.natural_key, "dependencies", [])
  37                 if deps:
  38                     deps = [apps.get_model(*d.split(".")) for d in deps]
  39             else:
  40                 deps = []
  41 
  42             # Now add a dependency for any FK relation with a model that
  43             # defines a natural key
  44             for field in model._meta.fields:
  45                 if hasattr(field.rel, "to"):
  46                     rel_model = field.rel.to
  47                     if rel_model != model:
  48                         deps.append(rel_model)
  49 
  50             # Also add a dependency for any simple M2M relation with a model
  51             # that defines a natural key.  M2M relations with explicit through
  52             # models don't count as dependencies.
  53             for field in model._meta.many_to_many:
  54                 rel_model = field.rel.to
  55                 if rel_model != model:
  56                     deps.append(rel_model)
  57             model_dependencies.append((model, deps))
  58 
  59     model_dependencies.reverse()
  60     # Now sort the models to ensure that dependencies are met. This
  61     # is done by repeatedly iterating over the input list of models.
  62     # If all the dependencies of a given model are in the final list,
  63     # that model is promoted to the end of the final list. This process
  64     # continues until the input list is empty, or we do a full iteration
  65     # over the input models without promoting a model to the final list.
  66     # If we do a full iteration without a promotion, that means there are
  67     # circular dependencies in the list.
  68     model_list = []
  69     while model_dependencies:
  70         skipped = []
  71         changed = False
  72         while model_dependencies:
  73             model, deps = model_dependencies.pop()
  74 
  75             # If all of the models in the dependency list are either already
  76             # on the final model list, or not on the original serialization list,
  77             # then we've found another model with all it's dependencies satisfied.
  78             found = True
  79             for candidate in ((d not in models or d in model_list) for d in deps):
  80                 if not candidate:
  81                     found = False
  82             if found:
  83                 model_list.append(model)
  84                 changed = True
  85             else:
  86                 skipped.append((model, deps))
  87         if not changed:
  88             raise RuntimeError(
  89                 "Can't resolve dependencies for %s in serialized app list."
  90                 % ", ".join(
  91                     "%s.%s" % (model._meta.app_label, model._meta.object_name)
  92                     for model, deps in sorted(skipped, key=lambda obj: obj[0].__name__)
  93                 )
  94             )
  95         model_dependencies = skipped
  96 
  97     return model_list

 108 def export(dest, silent, indent, exclude):
 109     "Exports core metadata for the Sentry installation."
 110 
 111     if exclude is None:
 112         exclude = ()
 113     else:
 114         exclude = exclude.lower().split(",")
 115 
 116     from django.core import serializers
 117 
 118     def yield_objects():
 119         # Collate the objects to be serialized.
 120         for model in sort_dependencies():
 121             if (
 122                 not getattr(model, "__core__", True)
 123                 or model.__name__.lower() in exclude
 124                 or model._meta.proxy
 125             ):
 126                 if not silent:
 127                     click.echo(">> Skipping model <%s>" % (model.__name__,), err=True)
 128                 continue
 129 
 130             queryset = model._base_manager.order_by(model._meta.pk.name)
 131             for obj in queryset.iterator():
 132                 yield obj
 133 
 134     if not silent:
 135         click.echo(">> Beginning export", err=True)
 136     serializers.serialize(
 137         "json", yield_objects(), indent=indent, stream=dest, use_natural_foreign_keys=True
 138     )
