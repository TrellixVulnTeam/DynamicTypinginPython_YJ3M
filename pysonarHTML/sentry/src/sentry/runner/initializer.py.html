  25     #     ],

  28         try:
  29             plugin = ep.load()

  31             import traceback

  34                 "Failed to load plugin %r:\n%s" % (ep.name, traceback.format_exc()), err=True

  37                 raise

  42         init_plugin(plugin)

  47     for integration_path in settings.SENTRY_DEFAULT_INTEGRATIONS:
  48         try:

  50         except Exception:
  51             import traceback
  52

  55                 err=True,
  56             )
  57         else:
  58             integrations.register(integration_cls)

 138 def bootstrap_options(settings, config=None):
 139     """
 140     Quickly bootstrap options that come in from a config file
 141     and convert options into Django settings that are
 142     required to even initialize the rest of the app.
 143     """
 144     # Make sure our options have gotten registered
 145     from sentry.options import load_defaults
 146 
 147     load_defaults()
 148 
 149     options = {}
 150     if config is not None:
 151         # Attempt to load our config yaml file
 152         from sentry.utils.yaml import safe_load
 153         from yaml.parser import ParserError
 154         from yaml.scanner import ScannerError
 155 
 156         try:
 157             with open(config, "rb") as fp:
 158                 options = safe_load(fp)
 159         except IOError:
 160             # Gracefully fail if yaml file doesn't exist
 161             pass
 162         except (AttributeError, ParserError, ScannerError) as e:
 163             from .importer import ConfigurationError
 164 
 165             raise ConfigurationError("Malformed config.yml file: %s" % six.text_type(e))
 166 
 167         # Empty options file, so fail gracefully
 168         if options is None:
 169             options = {}
 170         # Options needs to be a dict
 171         elif not isinstance(options, dict):
 172             from .importer import ConfigurationError
 173 
 174             raise ConfigurationError("Malformed config.yml file")
 175 
 176     from sentry.conf.server import DEAD
 177 
 178     # First move options from settings into options
 179     for k, v in six.iteritems(options_mapper):
 180         if getattr(settings, v, DEAD) is not DEAD and k not in options:
 181             warnings.warn(DeprecatedSettingWarning(options_mapper[k], "SENTRY_OPTIONS['%s']" % k))
 182             options[k] = getattr(settings, v)
 183 
 184     # Stuff everything else into SENTRY_OPTIONS
 185     # these will be validated later after bootstrapping
 186     for k, v in six.iteritems(options):
 187         settings.SENTRY_OPTIONS[k] = v
 188 
 189     # Now go back through all of SENTRY_OPTIONS and promote
 190     # back into settings. This catches the case when values are defined
 191     # only in SENTRY_OPTIONS and no config.yml file
 192     for o in (settings.SENTRY_DEFAULT_OPTIONS, settings.SENTRY_OPTIONS):
 193         for k, v in six.iteritems(o):
 194             if k in options_mapper:
 195                 # Map the mail.backend aliases to something Django understands
 196                 if k == "mail.backend":
 197                     try:
 198                         v = settings.SENTRY_EMAIL_BACKEND_ALIASES[v]
 199                     except KeyError:
 200                         pass
 201                 # Escalate the few needed to actually get the app bootstrapped into settings
 202                 setattr(settings, options_mapper[k], v)

 205 def configure_structlog():
 206     """
 207     Make structlog comply with all of our options.
 208     """
 209     from django.conf import settings
 210     import logging.config
 211     import structlog
 212     from sentry import options
 213     from sentry.logging import LoggingFormat
 214 
 215     WrappedDictClass = structlog.threadlocal.wrap_dict(dict)
 216     kwargs = {
 217         "context_class": WrappedDictClass,
 218         "wrapper_class": structlog.stdlib.BoundLogger,
 219         "cache_logger_on_first_use": True,
 220         "processors": [
 221             structlog.stdlib.add_log_level,
 222             structlog.stdlib.PositionalArgumentsFormatter(),
 223             structlog.processors.format_exc_info,
 224             structlog.processors.StackInfoRenderer(),
 225             structlog.processors.UnicodeDecoder(),
 226         ],
 227     }
 228 
 229     fmt_from_env = os.environ.get("SENTRY_LOG_FORMAT")
 230     if fmt_from_env:
 231         settings.SENTRY_OPTIONS["system.logging-format"] = fmt_from_env.lower()
 232 
 233     fmt = options.get("system.logging-format")
 234 
 235     if fmt == LoggingFormat.HUMAN:
 236         from sentry.logging.handlers import HumanRenderer
 237 
 238         kwargs["processors"].extend(
 239             [structlog.processors.ExceptionPrettyPrinter(), HumanRenderer()]
 240         )
 241     elif fmt == LoggingFormat.MACHINE:
 242         from sentry.logging.handlers import JSONRenderer
 243 
 244         kwargs["processors"].append(JSONRenderer())
 245 
 246     structlog.configure(**kwargs)
 247 
 248     lvl = os.environ.get("SENTRY_LOG_LEVEL")
 249 
 250     if lvl and lvl not in logging._levelNames:
 251         raise AttributeError("%s is not a valid logging level." % lvl)
 252 
 253     settings.LOGGING["root"].update({"level": lvl or settings.LOGGING["default_level"]})
 254 
 255     if lvl:
 256         for logger in settings.LOGGING["overridable"]:
 257             try:
 258                 settings.LOGGING["loggers"][logger].update({"level": lvl})
 259             except KeyError:
 260                 raise KeyError("%s is not a defined logger." % logger)
 261 
 262     logging.config.dictConfig(settings.LOGGING)

 338 def setup_services(validate=True):
 339     from sentry import (
 340         analytics,
 341         buffer,
 342         digests,
 343         newsletter,
 344         nodestore,
 345         quotas,
 346         ratelimits,
 347         search,
 348         tagstore,
 349         tsdb,
 350     )
 351     from .importer import ConfigurationError
 352     from sentry.utils.settings import reraise_as

 354     service_list = (
 355         analytics,
 356         buffer,
 357         digests,
 358         newsletter,
 359         nodestore,
 360         quotas,
 361         ratelimits,
 362         search,
 363         tagstore,
 364         tsdb,
 365     )
 366 
 367     for service in service_list:
 368         if validate:
 369             try:

 384                     ConfigurationError(
 385                         u"{} service failed to call setup()\n{}".format(

 436 def bind_cache_to_option_store():
 437     # The default ``OptionsStore`` instance is initialized without the cache
 438     # backend attached. The store itself utilizes the cache during normal
 439     # operation, but can't use the cache before the options (which typically
 440     # includes the cache configuration) have been bootstrapped from the legacy
 441     # settings and/or configuration values. Those options should have been
 442     # loaded at this point, so we can plug in the cache backend before
 443     # continuing to initialize the remainder of the application.
 444     from django.core.cache import cache as default_cache
 445     from sentry.options import default_store
 446 
 447     default_store.cache = default_cache

 450 def show_big_error(message):
 451     if isinstance(message, six.string_types):
 452         lines = message.strip().splitlines()
 453     else:
 454         lines = message
 455     maxline = max(map(len, lines))
 456     click.echo("", err=True)
 457     click.secho("!!!%s!!!" % ("!" * min(maxline, 80),), err=True, fg="red")
 458     click.secho("!! %s !!" % "".center(maxline), err=True, fg="red")
 459     for line in lines:
 460         click.secho("!! %s !!" % line.center(maxline), err=True, fg="red")
 461     click.secho("!! %s !!" % "".center(maxline), err=True, fg="red")
 462     click.secho("!!!%s!!!" % ("!" * min(maxline, 80),), err=True, fg="red")
 463     click.echo("", err=True)

 466 def apply_legacy_settings(settings):
 467     from sentry import options
 468 
 469     # SENTRY_USE_QUEUE used to determine if Celery was eager or not
 470     if hasattr(settings, "SENTRY_USE_QUEUE"):
 471         warnings.warn(
 472             DeprecatedSettingWarning(
 473                 "SENTRY_USE_QUEUE",
 474                 "CELERY_ALWAYS_EAGER",
 475                 "https://docs.sentry.io/on-premise/server/queue/",
 476             )
 477         )
 478         settings.CELERY_ALWAYS_EAGER = not settings.SENTRY_USE_QUEUE
 479 
 480     for old, new in (
 481         ("SENTRY_ADMIN_EMAIL", "system.admin-email"),
 482         ("SENTRY_URL_PREFIX", "system.url-prefix"),
 483         ("SENTRY_SYSTEM_MAX_EVENTS_PER_MINUTE", "system.rate-limit"),
 484         ("SENTRY_ENABLE_EMAIL_REPLIES", "mail.enable-replies"),
 485         ("SENTRY_SMTP_HOSTNAME", "mail.reply-hostname"),
 486         ("MAILGUN_API_KEY", "mail.mailgun-api-key"),
 487         ("SENTRY_FILESTORE", "filestore.backend"),
 488         ("SENTRY_FILESTORE_OPTIONS", "filestore.options"),
 489         ("GOOGLE_CLIENT_ID", "auth-google.client-id"),
 490         ("GOOGLE_CLIENT_SECRET", "auth-google.client-secret"),
 491     ):
 492         if new not in settings.SENTRY_OPTIONS and hasattr(settings, old):
 493             warnings.warn(DeprecatedSettingWarning(old, "SENTRY_OPTIONS['%s']" % new))
 494             settings.SENTRY_OPTIONS[new] = getattr(settings, old)
 495 
 496     if hasattr(settings, "SENTRY_REDIS_OPTIONS"):
 497         if "redis.clusters" in settings.SENTRY_OPTIONS:
 498             raise Exception(
 499                 "Cannot specify both SENTRY_OPTIONS['redis.clusters'] option and SENTRY_REDIS_OPTIONS setting."
 500             )
 501         else:
 502             warnings.warn(
 503                 DeprecatedSettingWarning(
 504                     "SENTRY_REDIS_OPTIONS",
 505                     'SENTRY_OPTIONS["redis.clusters"]',
 506                     removed_in_version="8.5",
 507                 )
 508             )
 509             settings.SENTRY_OPTIONS["redis.clusters"] = {"default": settings.SENTRY_REDIS_OPTIONS}
 510     else:
 511         # Provide backwards compatibility to plugins expecting there to be a
 512         # ``SENTRY_REDIS_OPTIONS`` setting by using the ``default`` cluster.
 513         # This should be removed when ``SENTRY_REDIS_OPTIONS`` is officially
 514         # deprecated. (This also assumes ``FLAG_NOSTORE`` on the configuration
 515         # option.)
 516         settings.SENTRY_REDIS_OPTIONS = options.get("redis.clusters")["default"]
 517 
 518     if not hasattr(settings, "SENTRY_URL_PREFIX"):
 519         url_prefix = options.get("system.url-prefix", silent=True)
 520         if not url_prefix:
 521             # HACK: We need to have some value here for backwards compatibility
 522             url_prefix = "http://sentry.example.com"
 523         settings.SENTRY_URL_PREFIX = url_prefix
 524 
 525     if settings.TIME_ZONE != "UTC":
 526         # non-UTC timezones are not supported
 527         show_big_error("TIME_ZONE should be set to UTC")
 528 
 529     # Set ALLOWED_HOSTS if it's not already available
 530     if not settings.ALLOWED_HOSTS:
 531         settings.ALLOWED_HOSTS = ["*"]
 532 
 533     if hasattr(settings, "SENTRY_ALLOW_REGISTRATION"):
 534         warnings.warn(
 535             DeprecatedSettingWarning(
 536                 "SENTRY_ALLOW_REGISTRATION", 'SENTRY_FEATURES["auth:register"]'
 537             )
 538         )
 539         settings.SENTRY_FEATURES["auth:register"] = settings.SENTRY_ALLOW_REGISTRATION
 540 
 541     settings.DEFAULT_FROM_EMAIL = settings.SENTRY_OPTIONS.get(
 542         "mail.from", settings.SENTRY_DEFAULT_OPTIONS.get("mail.from")
 543     )
 544 
 545     # HACK(mattrobenolt): This is a one-off assertion for a system.secret-key value.
 546     # If this becomes a pattern, we could add another flag to the OptionsManager to cover this, but for now
 547     # this is the only value that should prevent the app from booting up. Currently FLAG_REQUIRED is used to
 548     # trigger the Installation Wizard, not abort startup.
 549     if not settings.SENTRY_OPTIONS.get("system.secret-key"):
 550         from .importer import ConfigurationError
 551 
 552         raise ConfigurationError(
 553             "`system.secret-key` MUST be set. Use 'sentry config generate-secret-key' to get one."
 554         )

 557 def validate_snuba():
 558     """
 559     Make sure everything related to Snuba is in sync.
 560 
 561     This covers a few cases:
 562 
 563     * When you have features related to Snuba, you must also
 564       have Snuba fully configured correctly to continue.
 565     * If you have Snuba specific search/tagstore/tsdb backends,
 566       you must also have a Snuba compatible eventstream backend
 567       otherwise no data will be written into Snuba.
 568     * If you only have Snuba related eventstream, yell that you
 569       probably want the other backends otherwise things are weird.
 570     """
 571     if not settings.DEBUG:
 572         return
 573 
 574     has_all_snuba_required_backends = (
 575         settings.SENTRY_SEARCH == "sentry.search.snuba.EventsDatasetSnubaSearchBackend"
 576         and settings.SENTRY_TAGSTORE == "sentry.tagstore.snuba.SnubaTagStorage"
 577         and
 578         # TODO(mattrobenolt): Remove ServiceDelegator check
 579         settings.SENTRY_TSDB
 580         in ("sentry.tsdb.redissnuba.RedisSnubaTSDB", "sentry.utils.services.ServiceDelegator")
 581     )
 582 
 583     eventstream_is_snuba = (
 584         settings.SENTRY_EVENTSTREAM == "sentry.eventstream.snuba.SnubaEventStream"
 585         or settings.SENTRY_EVENTSTREAM == "sentry.eventstream.kafka.KafkaEventStream"
 586     )
 587 
 588     # All good here, it doesn't matter what else is going on
 589     if has_all_snuba_required_backends and eventstream_is_snuba:
 590         return
 591 
 592     from sentry.features import requires_snuba as snuba_features
 593 
 594     snuba_enabled_features = set()
 595 
 596     for feature in snuba_features:
 597         if settings.SENTRY_FEATURES.get(feature, False):
 598             snuba_enabled_features.add(feature)
 599 
 600     if snuba_enabled_features and not eventstream_is_snuba:
 601         from .importer import ConfigurationError
 602 
 603         show_big_error(
 604             """
 605 You have features enabled which require Snuba,
 606 but you don't have any Snuba compatible configuration.
 607 
 608 Features you have enabled:
 609 %s
 610 
 611 See: https://github.com/getsentry/snuba#sentry--snuba
 612 """
 613             % "\n".join(snuba_enabled_features)
 614         )
 615         raise ConfigurationError("Cannot continue without Snuba configured.")
 616 
 617     if not eventstream_is_snuba:
 618         from .importer import ConfigurationError
 619 
 620         show_big_error(
 621             """
 622 It appears that you are requiring Snuba,
 623 but your SENTRY_EVENTSTREAM is not compatible.
 624 
 625 Current settings:
 626 
 627 SENTRY_SEARCH = %r
 628 SENTRY_TAGSTORE = %r
 629 SENTRY_TSDB = %r
 630 SENTRY_EVENTSTREAM = %r
 631 
 632 See: https://github.com/getsentry/snuba#sentry--snuba"""
 633             % (
 634                 settings.SENTRY_SEARCH,
 635                 settings.SENTRY_TAGSTORE,
 636                 settings.SENTRY_TSDB,
 637                 settings.SENTRY_EVENTSTREAM,
 638             )
 639         )
 640         raise ConfigurationError("Cannot continue without Snuba configured correctly.")
 641 
 642     if eventstream_is_snuba and not has_all_snuba_required_backends:
 643         show_big_error(
 644             """
 645 You are using a Snuba compatible eventstream
 646 without configuring search/tagstore/tsdb also to use Snuba.
 647 This is probably not what you want.
 648 
 649 Current settings:
 650 
 651 SENTRY_SEARCH = %r
 652 SENTRY_TAGSTORE = %r
 653 SENTRY_TSDB = %r
 654 SENTRY_EVENTSTREAM = %r
 655 
 656 See: https://github.com/getsentry/snuba#sentry--snuba"""
 657             % (
 658                 settings.SENTRY_SEARCH,
 659                 settings.SENTRY_TAGSTORE,
