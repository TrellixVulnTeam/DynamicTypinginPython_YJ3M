  10 class CaseInsensitiveChoice(Choice):
  11     def convert(self, value, param, ctx):
  12         self.choices = [choice.upper() for choice in self.choices]

  16 def configuration(f):
  17     "Load and configure Sentry."
  18     import click
  19     from functools import update_wrapper
  20 
  21     @click.pass_context
  22     def inner(ctx, *args, **kwargs):
  23         # HACK: We can't call `configure()` from within tests
  24         # since we don't load config files from disk, so we
  25         # need a way to bypass this initialization step
  26         if os.environ.get("_SENTRY_SKIP_CONFIGURATION") != "1":
  27             from sentry.runner import configure
  28 
  29             configure()
  30         return ctx.invoke(f, *args, **kwargs)
  31 
  32     return update_wrapper(inner, f)

  35 def log_options(default=None):
  36     def decorator(f):
  37         """
  38         Give ability to configure global logging level/format.
  39         Must be used before configuration.
  40         """
  41         import click
  42         from functools import update_wrapper
  43         from sentry.logging import LoggingFormat
  44 
  45         formats = [LoggingFormat.HUMAN, LoggingFormat.MACHINE]
  46 
  47         @click.pass_context
  48         @click.option(
  49             "--loglevel",
  50             "-l",
  51             default=default,
  52             help="Global logging level. Use wisely.",
  53             envvar="SENTRY_LOG_LEVEL",
  54             type=CaseInsensitiveChoice(LOG_LEVELS),
  55         )
  56         @click.option(
  57             "--logformat",
  58             default=default,
  59             help="Log line format.",
  60             envvar="SENTRY_LOG_FORMAT",
  61             type=CaseInsensitiveChoice(formats),
  62         )
  63         def inner(ctx, loglevel=None, logformat=None, *args, **kwargs):
  64             if loglevel:
  65                 os.environ["SENTRY_LOG_LEVEL"] = loglevel
  66             if logformat:
  67                 os.environ["SENTRY_LOG_FORMAT"] = logformat.lower()
  68             return ctx.invoke(f, *args, **kwargs)
  69 
  70         return update_wrapper(inner, f)
  71 
  72     return decorator
