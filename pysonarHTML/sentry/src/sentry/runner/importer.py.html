  23 class Importer(object):
  24     def __init__(self, name, config_path, default_settings=None, callback=None):

  26         self.config_path = config_path
  27         self.default_settings = default_settings
  28         self.callback = callback

  38     def load_module(self, fullname):
  39         # Check to make sure it's not already in sys.modules in case of a reload()
  40         if fullname in sys.modules:
  41             return sys.modules[fullname]  # pragma: no cover
  42 
  43         try:
  44             mod = self._load_module(fullname)
  45         except Exception as e:
  46             from sentry.utils.settings import reraise_as
  47 
  48             msg = six.text_type(e)
  49             if msg:
  50                 msg = "%s: %s" % (type(e).__name__, msg)
  51             else:
  52                 msg = type(e).__name__
  53             reraise_as(ConfigurationError(msg))
  54         else:
  55             # Install into sys.modules explicitly
  56             sys.modules[fullname] = mod
  57             if self.callback is not None:
  58                 self.callback(mod)
  59             return mod
  60 
  61     def _load_module(self, fullname):
  62         if self.default_settings:
  63             from importlib import import_module
  64 
  65             default_settings_mod = import_module(self.default_settings)
  66         else:
  67             default_settings_mod = None
  68 
  69         settings_mod = imp.new_module(self.name)
  70 
  71         # Django doesn't play too nice without the config file living as a real
  72         # file, so let's fake it.
  73         settings_mod.__file__ = self.config_path
  74 
  75         # install the default settings for this app
  76         load_settings(default_settings_mod, settings=settings_mod)
  77 
  78         # install the custom settings for this app
  79         load_settings(self.config_path, settings=settings_mod, silent=True)
  80 
  81         install_plugin_apps("sentry.apps", settings_mod)
  82 
  83         return settings_mod

  86 def load_settings(mod_or_filename, settings, silent=False):
  87     if isinstance(mod_or_filename, six.string_types):
  88         conf = imp.new_module("temp_config")
  89         conf.__file__ = mod_or_filename
  90 
  91         try:
  92             with open(mod_or_filename) as source_file:
  93                 six.exec_(source_file.read(), conf.__dict__)
  94         except IOError as e:
  95             import errno
  96 
  97             if silent and e.errno in (errno.ENOENT, errno.EISDIR):
  98                 return settings
  99             e.strerror = "Unable to load configuration file (%s)" % e.strerror
 100             raise
 101     else:
 102         conf = mod_or_filename
 103 
 104     add_settings(conf, settings=settings)

 107 def install_plugin_apps(entry_point, settings):
 108     # entry_points={
 109     #    'sentry.apps': [
 110     #         'phabricator = sentry_phabricator'
 111     #     ],
 112     # },
 113     from pkg_resources import iter_entry_points
 114 
 115     installed_apps = list(settings.INSTALLED_APPS)
 116     for ep in iter_entry_points(entry_point):
 117         if ep.module_name not in installed_apps:
 118             installed_apps.append(ep.module_name)
 119     settings.INSTALLED_APPS = tuple(installed_apps)

 122 def add_settings(mod, settings):
 123     """
 124     Adds all settings that are part of ``mod`` to the global settings object.
 125     Special cases ``EXTRA_APPS`` to append the specified applications to the
 126     list of ``INSTALLED_APPS``.
 127     """
 128 
 129     for setting in dir(mod):
 130         if not setting.isupper():
 131             continue
 132 
 133         setting_value = getattr(mod, setting)
 134         if setting in ("INSTALLED_APPS",) and isinstance(setting_value, six.string_types):
 135             setting_value = (setting_value,)  # In case the user forgot the comma.
 136 
 137         # Any setting that starts with EXTRA_ and matches a setting that is a list or tuple
 138         # will automatically append the values to the current setting.
 139         # It might make sense to make this less magical
 140         if setting[:6] == "EXTRA_":
 141             base_setting = setting[6:]
 142             if isinstance(getattr(settings, base_setting), (list, tuple)):
 143                 curval = getattr(settings, base_setting)
 144                 setattr(settings, base_setting, curval + type(curval)(setting_value))
 145                 continue
 146 
 147         setattr(settings, setting, setting_value)
