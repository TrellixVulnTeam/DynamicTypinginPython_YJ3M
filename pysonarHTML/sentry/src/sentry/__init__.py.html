  14 def _get_git_revision(path):
  15     if not os.path.exists(os.path.join(path, ".git")):
  16         return None
  17     try:
  18         revision = check_output(["git", "rev-parse", "HEAD"], cwd=path, env=os.environ)
  19     except Exception:
  20         # binary didn't exist, wasn't on path, etc
  21         return None
  22     return revision.strip()

  25 def get_revision():
  26     """
  27     :returns: Revision number of this branch/checkout, if available. None if
  28         no revision number can be determined.
  29     """
  30     if "SENTRY_BUILD" in os.environ:
  31         return os.environ["SENTRY_BUILD"]
  32     package_dir = os.path.dirname(__file__)
  33     checkout_dir = os.path.normpath(os.path.join(package_dir, os.pardir, os.pardir))
  34     path = os.path.join(checkout_dir)
  35     if os.path.exists(path):
  36         return _get_git_revision(path)
  37     return None

  40 def get_version():
  41     if __build__:
  42         return "%s.%s" % (__version__, __build__)
  43     return __version__

  46 def is_docker():
  47     # One of these environment variables are guaranteed to exist
  48     # from our official docker images.
  49     # SENTRY_VERSION is from a tagged release, and SENTRY_BUILD is from a
  50     # a git based image.
  51     return "SENTRY_VERSION" in os.environ or "SENTRY_BUILD" in os.environ
