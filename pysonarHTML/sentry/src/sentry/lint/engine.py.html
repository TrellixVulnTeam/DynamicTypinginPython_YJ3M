  62         files_to_check = get_files(".")

  91 def js_lint(file_list=None, parseable=False, format=False):
  92 
  93     # We require eslint in path but we actually call an eslint wrapper
  94     eslint_path = get_node_modules_bin("eslint")
  95 
  96     # Note, in CI, we run a relaxed version of our eslint rules (.eslint.relax.js)
  97     eslint_wrapper_path = get_sentry_bin("eslint-travis-wrapper")
  98 
  99     if not os.path.exists(eslint_path):
 100         sys.stdout.write("!! Skipping JavaScript linting because eslint is not installed.\n")
 101         return False
 102 
 103     js_file_list = get_js_files(file_list, snapshots=True)
 104 
 105     has_errors = False
 106     if js_file_list:
 107         if os.environ.get("CI"):
 108             cmd = [eslint_wrapper_path, "--ext", ".js,.jsx,.ts,.tsx"]
 109         else:
 110             cmd = [eslint_path, "--ext", ".js,.jsx,.ts,.tsx"]
 111 
 112         if format:
 113             cmd.append("--fix")
 114         if parseable:
 115             cmd.append("--format=checkstyle")
 116 
 117         status = Popen(cmd + js_file_list).wait()
 118         has_errors = status != 0
 119 
 120     return has_errors

 148 def yarn_check(file_list):
 149     """
 150     Checks if package.json was modified WITHOUT a corresponding change in the Yarn
 151     lockfile. This can happen if a user manually edited package.json without running Yarn.
 152 
 153     This is a user prompt right now because there ARE cases where you can touch package.json
 154     without a Yarn lockfile change, e.g. Jest config changes, license changes, etc.
 155     """
 156 
 157     if file_list is None or os.environ.get("SKIP_YARN_CHECK"):
 158         return False
 159 
 160     if "package.json" in file_list and "yarn.lock" not in file_list:
 161         sys.stdout.write(
 162             "\033[33m"
 163             + """Warning: package.json modified without accompanying yarn.lock modifications.
 164 
 165 If you updated a dependency/devDependency in package.json, you must run `yarn install` to update the lockfile.
 166 
 167 To skip this check, run `SKIP_YARN_CHECK=1 git commit [options]`"""
 168             + "\033[0m"
 169             + "\n"
 170         )
 171         return True
 172 
 173     return False

 176 def is_prettier_valid(project_root, prettier_path):
 177     if not os.path.exists(prettier_path):
 178         sys.stderr.write(
 179             "[sentry.lint] Skipping JavaScript formatting because prettier is not installed.\n"
 180         )
 181         return False
 182 
 183     # Get Prettier version from package.json
 184     package_version = None
 185     package_json_path = os.path.join(project_root, "package.json")
 186     with open(package_json_path) as package_json:
 187         try:
 188             package_version = json.load(package_json)["devDependencies"]["prettier"]
 189         except KeyError:
 190             sys.stderr.write("!! Prettier missing from package.json\n")
 191             return False
 192 
 193     prettier_version = subprocess.check_output([prettier_path, "--version"]).rstrip()
 194     if prettier_version != package_version:
 195         sys.stderr.write(
 196             u"[sentry.lint] Prettier is out of date: {} (expected {}). Please run `yarn install`.\n".format(
 197                 prettier_version, package_version
 198             )
 199         )
 200         return False
 201 
 202     return True

 205 def js_lint_format(file_list=None):
 206     """
 207     We only format JavaScript code as part of this pre-commit hook. It is not part
 208     of the lint engine. This uses eslint's `--fix` formatting feature.
 209     """
 210     eslint_path = get_node_modules_bin("eslint")
 211     project_root = get_project_root()
 212     prettier_path = get_prettier_path()
 213 
 214     if not os.path.exists(eslint_path):
 215         sys.stdout.write(
 216             "!! Skipping JavaScript linting and formatting because eslint is not installed.\n"
 217         )
 218         return False
 219 
 220     if not is_prettier_valid(project_root, prettier_path):
 221         return False
 222 
 223     js_file_list = get_js_files(file_list)
 224 
 225     # manually exclude some bad files
 226     js_file_list = [x for x in js_file_list if "/javascript/example-project/" not in x]
 227     cmd = [eslint_path, "--fix"]
 228 
 229     has_package_json_errors = (
 230         False
 231         if "package.json" not in file_list
 232         else run_formatter([prettier_path, "--write"], ["package.json"])
 233     )
 234 
 235     has_errors = run_formatter(cmd, js_file_list)
 236 
 237     return has_errors or has_package_json_errors

 277 def run_formatter(cmd, file_list, prompt_on_changes=True):
 278     if not file_list:
 279         return False
 280 
 281     has_errors = False
 282 
 283     status = subprocess.Popen(cmd + file_list).wait()
 284     has_errors = status != 0
 285     if has_errors:
 286         return True
 287 
 288     # this is not quite correct, but it at least represents what would be staged
 289     output = subprocess.check_output(["git", "diff", "--color"] + file_list)
 290     if output:
 291         sys.stdout.write("[sentry.lint] applied changes from autoformatting\n")
 292         sys.stdout.write(output)
 293         if prompt_on_changes:
 294             with open("/dev/tty") as fp:
 295                 sys.stdout.write("\033[1m" + "Stage this patch and continue? [Y/n] " + "\033[0m\n")
 296                 if fp.readline().strip() not in ("Y", "y", ""):
 297                     sys.stderr.write("[sentry.lint] Unstaged changes have not been staged.\n")
 298                     if not os.environ.get("SENTRY_SKIP_FORCE_PATCH"):
 299                         sys.stderr.write("[sentry.lint] Aborted!\n")
 300                         sys.exit(1)
 301                 else:
 302                     status = subprocess.Popen(["git", "update-index", "--add"] + file_list).wait()
 303         has_errors = status != 0
 304     return has_errors

 307 def run(
 308     file_list=None,
 309     format=True,
 310     lint=True,
 311     js=True,
 312     py=True,
 313     less=True,
 314     yarn=True,
 315     test=False,
 316     parseable=False,
 317 ):
 318     old_sysargv = sys.argv
 319 
 320     try:
 321         sys.argv = [os.path.join(os.path.dirname(__file__), os.pardir, os.pardir, os.pardir)]
 322         results = []
 323 
 324         # packages
 325         if yarn:
 326             results.append(yarn_check(file_list))
 327 
 328         # bail early if a deps failed
 329         if any(results):
 330             return 1
 331 
 332         if format:
 333             if py:
 334                 # python autoformatting is now done via pre-commit (black)
 335                 pass
 336             if js:
 337                 # run eslint with --fix and skip these linters down below
 338                 results.append(js_lint_format(file_list))
 339             if less:
 340                 results.append(less_format(file_list))
 341 
 342         # bail early if a formatter failed
 343         if any(results):
 344             return 1
 345 
 346         if lint:
 347             if py:
 348                 pass  # flake8 linting was moved to pre-commit
 349             if js:
 350                 # stylelint `--fix` doesn't work well
 351                 results.append(js_stylelint(file_list, parseable=parseable, format=format))
 352 
 353                 if not format:
 354                     # these tasks are called when we need to format, so skip it here
 355                     results.append(js_lint(file_list, parseable=parseable, format=format))
 356 
 357         if test:
 358             if js:
 359                 results.append(js_test(file_list))
 360 
 361         if any(results):
 362             return 1
 363         return 0
 364     finally:
 365         sys.argv = old_sysargv
