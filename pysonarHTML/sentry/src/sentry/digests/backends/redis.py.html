  23 class RedisBackend(Backend):
  24     """
  25     Implements the digest backend API, backed by Redis.
  26 
  27     Each timeline is modeled as a sorted set, and also maintains a separate key
  28     that contains the last time the digest was processed (used for scheduling.)
  29 
  30     .. code::
  31 
  32         redis:6379> ZREVRANGEBYSCORE "d:t:mail:p:1" inf -inf WITHSCORES
  33         1) "433be20b807c4cd49a132de69c0f6c55"
  34         2) "1444847625"
  35         3) "0f9d5fe4b5b3400fab85d9a841aa8467"
  36         4) "1444847625"
  37         ...
  38 
  39     The timeline contains references to several records, which are stored
  40     separately, encoded using the codec provided to the backend:
  41 
  42     .. code::
  43 
  44         redis:6379> GET "d:t:mail:p:1:r:433be20b807c4cd49a132de69c0f6c55"
  45         [ binary content ]
  46 
  47     When the timeline is ready to be digested, the timeline set is renamed,
  48     creating a digest set (in this case the key would be ``d:t:mail:p:1:d``),
  49     that represents a snapshot of the timeline contents at that point in time.
  50     (If the digest set already exists, the timeline contents are instead
  51     unioned into the digest set and then the timeline is cleared.) This allows
  52     new records to be added to the timeline that will be processed after the
  53     next scheduling interval without the risk of data loss due to race
  54     conditions between the record addition and digest generation and delivery.
  55 
  56     Schedules are modeled as two sorted sets -- one for ``waiting`` items, and
  57     one for ``ready`` items. Items in the ``waiting`` set are scored by the
  58     time at which they should be transitioned to the ``ready`` set.  Items in
  59     the ``ready`` set are scored by the time at which they were scheduled to be
  60     added to the ``ready`` set. Iterating each set from oldest to newest yields
  61     the highest priority items for action (moving from the ``waiting`` to
  62     ``ready`` set, or delivering a digest for the ``waiting`` and ``ready``
  63     set, respectively.)
  64 
  65     .. code::
  66 
  67         redis:6379> ZREVRANGEBYSCORE "d:s:w" inf -inf WITHSCORES
  68         1) "mail:p:1"
  69         2) "1444847638"
  70 
  71     """
  72 
  73     def __init__(self, **options):
  74         self.cluster, options = get_cluster_from_options("SENTRY_DIGESTS_OPTIONS", options)
  75         self.locks = LockManager(RedisLockBackend(self.cluster))
  76 
  77         self.namespace = options.pop("namespace", "d")
  78 
  79         # Sets the time-to-live (in seconds) for records, timelines, and
  80         # digests. This can (and should) be a relatively high value, since
  81         # timelines, digests, and records should all be deleted after they have
  82         # been processed -- this is mainly to ensure stale data doesn't hang
  83         # around too long in the case of a configuration error. This should be
  84         # larger than the maximum scheduling delay to ensure data is not evicted
  85         # too early.
  86         self.ttl = options.pop("ttl", 60 * 60)
  87 
  88         super(RedisBackend, self).__init__(**options)
  89 
  90     def validate(self):
  91         logger.debug("Validating Redis version...")
  92         check_cluster_versions(self.cluster, Version((2, 8, 9)), label="Digests")
  93 
  94     def _get_connection(self, key):
  95         return self.cluster.get_local_client_for_key(u"{}:t:{}".format(self.namespace, key))
  96 
  97     def _get_timeline_lock(self, key, duration):
  98         lock_key = u"{}:t:{}".format(self.namespace, key)
  99         return self.locks.get(lock_key, duration=duration, routing_key=lock_key)

 101     def add(self, key, record, increment_delay=None, maximum_delay=None, timestamp=None):
 102         if timestamp is None:
 103             timestamp = time.time()
 104 
 105         if increment_delay is None:
 106             increment_delay = self.increment_delay
 107 
 108         if maximum_delay is None:
 109             maximum_delay = self.maximum_delay
 110 
 111         # Redis returns "true" and "false" as "1" and "None", so we just cast
 112         # them back to the appropriate boolean here.
 113         return bool(
 114             script(
 115                 self._get_connection(key),
 116                 [key],
 117                 [
 118                     "ADD",
 119                     self.namespace,
 120                     self.ttl,
 121                     timestamp,
 122                     key,
 123                     record.key,
 124                     self.codec.encode(record.value),
 125                     record.timestamp,  # TODO: check type
 126                     increment_delay,
 127                     maximum_delay,
 128                     self.capacity if self.capacity else -1,
 129                     self.truncation_chance,
 130                 ],
 131             )
 132         )
 133 
 134     def __schedule_partition(self, host, deadline, timestamp):
 135         return script(
 136             self.cluster.get_local_client(host),
 137             ["-"],
 138             ["SCHEDULE", self.namespace, self.ttl, timestamp, deadline],
 139         )
 140 
 141     def schedule(self, deadline, timestamp=None):
 142         if timestamp is None:
 143             timestamp = time.time()
 144 
 145         for host in self.cluster.hosts:
 146             try:
 147                 for key, timestamp in self.__schedule_partition(host, deadline, timestamp):
 148                     yield ScheduleEntry(key, float(timestamp))
 149             except Exception as error:
 150                 logger.error(
 151                     "Failed to perform scheduling for partition %r due to error: %r",
 152                     host,
 153                     error,
 154                     exc_info=True,
 155                 )
 156 
 157     def __maintenance_partition(self, host, deadline, timestamp):
 158         return script(
 159             self.cluster.get_local_client(host),
 160             ["-"],
 161             ["MAINTENANCE", self.namespace, self.ttl, timestamp, deadline],
 162         )
 163 
 164     def maintenance(self, deadline, timestamp=None):
 165         if timestamp is None:
 166             timestamp = time.time()
 167 
 168         for host in self.cluster.hosts:
 169             try:
 170                 self.__maintenance_partition(host, deadline, timestamp)
 171             except Exception as error:
 172                 logger.error(
 173                     "Failed to perform maintenance on digest partition %r due to error: %r",
 174                     host,
 175                     error,
 176                     exc_info=True,
 177                 )

 179     @contextmanager
 180     def digest(self, key, minimum_delay=None, timestamp=None):
 181         if minimum_delay is None:
 182             minimum_delay = self.minimum_delay
 183 
 184         if timestamp is None:
 185             timestamp = time.time()
 186 
 187         connection = self._get_connection(key)
 188         with self._get_timeline_lock(key, duration=30).acquire():
 189             try:
 190                 response = script(
 191                     connection,
 192                     [key],
 193                     [
 194                         "DIGEST_OPEN",
 195                         self.namespace,
 196                         self.ttl,
 197                         timestamp,
 198                         key,
 199                         self.capacity if self.capacity else -1,
 200                     ],
 201                 )
 202             except ResponseError as e:
 203                 if "err(invalid_state):" in six.text_type(e):
 204                     six.raise_from(InvalidState("Timeline is not in the ready state."), e)
 205                 else:
 206                     raise
 207 
 208             records = map(
 209                 lambda key__value__timestamp: Record(
 210                     key__value__timestamp[0],
 211                     self.codec.decode(key__value__timestamp[1])
 212                     if key__value__timestamp[1] is not None
 213                     else None,
 214                     float(key__value__timestamp[2]),
 215                 ),
 216                 response,
 217             )
 218 
 219             # If the record value is `None`, this means the record data was
 220             # missing (it was presumably evicted by Redis) so we don't need to
 221             # return it here.
 222             yield [record for record in records if record.value is not None]
 223 
 224             script(
 225                 connection,
 226                 [key],
 227                 ["DIGEST_CLOSE", self.namespace, self.ttl, timestamp, key, minimum_delay]
 228                 + [record.key for record in records],
 229             )
 230 
 231     def delete(self, key, timestamp=None):
 232         if timestamp is None:
 233             timestamp = time.time()
 234 
 235         connection = self._get_connection(key)
 236         with self._get_timeline_lock(key, duration=30).acquire():
 237             script(connection, [key], ["DELETE", self.namespace, self.ttl, timestamp, key])
