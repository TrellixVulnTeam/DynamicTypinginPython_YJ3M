  25 class Backend(Service):
  26     """
  27     A digest backend coordinates the addition of records to timelines, as well
  28     as scheduling their digestion (processing.) This allows for summarizations
  29     of activity that was recorded as having occurred during a window of time.
  30 
  31     A timeline is the central abstraction for digests. A timeline is a
  32     reverse-chronological set of records. Timelines are identified by a unique
  33     key. Records within a timeline are also identified by a key that is unique
  34     with respect to the timeline they are a part of.
  35 
  36     A timeline can be in one of two states: "waiting" or "ready".
  37 
  38     When the first record is added to a timeline, the timeline transitions to
  39     the "ready" state, and the digest is immediately available to be digested
  40     and delivered. (This immediate state change to "ready" allows notifications
  41     to be delivered with lower latency.)
  42 
  43     After delivery, the digest transitions to the "waiting" state for the
  44     duration of the delay interval. If more items are added to the digest
  45     during this waiting period, the schedule is extended incrementally (up to
  46     the value defined by the maximum delay option) to allow grouping the more
  47     items into a single notification.
  48 
  49     When the "waiting" period is over, the timeline transitions back to the
  50     "ready" state, which causes the timeline to be again digested and
  51     delivered. After the timeline is digested, it transitions back to the
  52     "waiting" state if it contained records. If the timeline did not contain
  53     any records when it was digested, it can be deleted (although deletion may
  54     be preempted by a new record being added to the timeline, requiring it to
  55     be transitioned to "waiting" instead.)
  56     """
  57 
  58     __all__ = ("add", "delete", "digest", "enabled", "maintenance", "schedule", "validate")

  60     def __init__(self, **options):
  61         # The ``minimum_delay`` option defines the default minimum amount of
  62         # time (in seconds) to wait between scheduling digests for delivery
  63         # after the initial scheduling.
  64         self.minimum_delay = options.pop("minimum_delay", 60 * 5)
  65 
  66         # The ``maximum_delay`` option defines the default maximum amount of
  67         # time (in seconds) to wait between scheduling digests for delivery.
  68         self.maximum_delay = options.pop("maximum_delay", 60 * 30)
  69 
  70         # The ``increment_delay`` option defines how long each observation of
  71         # an event should delay scheduling (up until the ``maximum_delay``
  72         # after the last time a digest was processed.)
  73         self.increment_delay = options.pop("increment_delay", 30)
  74 
  75         # The ``codec`` option provides the strategy for encoding and decoding
  76         # records in the timeline.
  77         self.codec = load(options.pop("codec", DEFAULT_CODEC))
  78 
  79         # The ``capacity`` option defines the maximum number of items that
  80         # should be contained within a timeline. (Whether this is a hard or
  81         # soft limit is backend dependent -- see the ``truncation_chance`` option.)
  82         self.capacity = options.pop("capacity", None)
  83         if self.capacity is not None and self.capacity < 1:
  84             raise ValueError("Timeline capacity must be at least 1 if used.")
  85 
  86         # The ``truncation_chance`` option defines the probability that an
  87         # ``add`` operation will trigger a truncation of the timeline to keep
  88         # it's size close to the defined capacity. A value of 1 will cause the
  89         # timeline to be truncated on every ``add`` operation (effectively
  90         # making it a hard limit), while a lower probability will increase the
  91         # chance of the timeline growing past it's intended capacity, but
  92         # increases the performance of ``add`` operations (by avoiding
  93         # truncation, which is a potentially expensive operation, especially on
  94         # large data sets.)
  95         if self.capacity:
  96             self.truncation_chance = options.pop("truncation_chance", 1.0 / self.capacity)
  97         else:
  98             if options.get("truncation_chance") is not None:
  99                 raise TypeError(
 100                     'No timeline capacity has been set, "truncation_chance" must be None.'
 101                 )
 102             else:
 103                 self.truncation_chance = 0.0

 126     def digest(self, key, minimum_delay=None):
 127         """
 128         Extract records from a timeline for processing.
 129 
 130         This method acts as a context manager. The target of the ``as`` clause
 131         is an iterator contains all of the records contained within the digest.
 132 
 133         If the context manager successfully exits, all records that were part
 134         of the digest are removed from the timeline and the timeline is placed
 135         back in the "waiting" state. If an exception is raised during the
 136         execution of the context manager, all records are preserved and no
 137         state change occurs so that the next invocation will contain all
 138         records that the were included previously, as well as any records that
 139         were added in between invocations. (This means that the caller must
 140         either retry the digest operation until it succeeds, or wait for the
 141         operation to be rescheduled as part of the maintenance process for the
 142         items to be processed.)
 143 
 144         Typically, the block that is surrounded by context manager includes all
 145         of the processing logic necessary to summarize the timeline contents
 146         (since this process is generally has no side effects), while any
 147         irrevocable action -- such as sending an email -- should occur after
 148         the context manager has exited, to ensure that action is performed at
 149         most once.
 150 
 151         For example::
 152 
 153             with timelines.digest('project:1') as records:
 154                 message = build_digest_email(records)
 155 
 156             message.send_async()
 157 
 158         """
 159         raise NotImplementedError

 172     def maintenance(self, deadline):
 173         """
 174         Identify timelines that appear to be stuck in the ready state.
 175 
 176         This method moves all timelines that are in the ready state back to the
 177         waiting state if their schedule time is prior to the deadline. (This
 178         does not reschedule any tasks directly, and should generally be
 179         performed as part of the scheduler task, before the ``schedule``
 180         call.)
 181 
 182         This is designed to handle the situation where task execution is
 183         managed by a separate system such as RabbitMQ & Celery from scheduling.
 184         A digest task may not be able to be successfully retried after a failure
 185         (e.g. if the process executing the task can no longer communicate with
 186         the messaging broker) which can result in a task remaining in the ready
 187         state without an execution plan.
 188 
 189         This may cause issues when asynchronously processing timelines and
 190         there is a severe backlog of timelines to be digested. Timelines in the
 191         "ready" state that were scheduled for execution prior to the deadline
 192         may still have outstanding tasks associated with them -- remember that
 193         without the ability to interrogate the queue, we are unable to identify
 194         if these tasks have finished but were unable to be removed from the
 195         schedule, failed outright, or are still pending. As part of
 196         maintenance, those timelines are moved back to the "waiting" state for
 197         rescheduling, and if a pending task for a timeline that was previously
 198         in the "ready" state but is now back in the "waiting" state actually
 199         executes, it will fail to be executed. The timeline will later be moved
 200         back to the "ready" state and rescheduled -- so it's contents will
 201         eventually be processed -- but this may be significantly delayed from
 202         the originally scheduled time. Both the queue backlog as well as the
 203         frequency of the digest task raising an exception when a timeline is in
 204         an invalid state should be monitored. If these exceptions happen
 205         frequently -- especially during periods of abnormal queue growth -- the
 206         frequency of maintenance tasks should be decreased, or the deadline
 207         should be pushed further towards the past (execution grace period
 208         increased) or both.
 209         """
 210         raise NotImplementedError
