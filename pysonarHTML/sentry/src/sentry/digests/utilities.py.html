  11 def get_digest_metadata(digest):
  12     start = None
  13     end = None
  14 
  15     counts = Counter()
  16     for rule, groups in six.iteritems(digest):
  17         counts.update(groups.keys())
  18 
  19         for group, records in six.iteritems(groups):
  20             for record in records:
  21                 if start is None or record.datetime < start:
  22                     start = record.datetime
  23 
  24                 if end is None or record.datetime > end:
  25                     end = record.datetime
  26 
  27     return start, end, counts

  54 def get_event_from_groups_in_digest(digest):
  55     """
  56     get_event_from_groups_in_digest(digest: Digest)
  57 
  58     Gets the first event from each group in the digest
  59     """
  60     events = []
  61     for rule_groups in six.itervalues(digest):
  62         for group_records in six.itervalues(rule_groups):
  63             events.append(group_records[0].value.event)
  64     return set(events)

  67 def build_custom_digest(original_digest, events):
  68     """
  69     build_custom_digest(original_digest: Digest, events: Set[Events]) -> Digest
  70     """
  71     user_digest = OrderedDict()
  72     for rule, rule_groups in six.iteritems(original_digest):
  73         user_rule_groups = OrderedDict()
  74         for group, group_records in six.iteritems(rule_groups):
  75             user_group_records = [
  76                 record for record in group_records if record.value.event in events
  77             ]
  78             if user_group_records:
  79                 user_rule_groups[group] = user_group_records
  80         if user_rule_groups:
  81             user_digest[rule] = user_rule_groups
  82     return user_digest

  85 def build_events_by_actor(project_id, events, user_ids):
  86     """
  87     build_events_by_actor(project_id: Int, events: Set(Events), user_ids: Set[Int]) -> Map[Actor, Set(Events)]
  88     """
  89     events_by_actor = defaultdict(set)
  90     for event in events:
  91         # TODO(LB): I Know this is inefficient.
  92         # ProjectOwnership.get_owners is O(n) queries and I'm doing that O(len(events)) times
  93         # I will create a follow-up PR to address this method's efficiency problem
  94         # Just wanted to make as few changes as possible for now.
  95         actors, __ = ProjectOwnership.get_owners(project_id, event.data)
  96         if actors == ProjectOwnership.Everyone:
  97             actors = [Actor(user_id, User) for user_id in user_ids]
  98         for actor in actors:
  99             events_by_actor[actor].add(event)
 100     return events_by_actor

 103 def convert_actors_to_users(events_by_actor, user_ids):
 104     """
 105     convert_actors_to_user_set(events_by_actor: Map[Actor, Set(Events)], user_ids: List(Int)) -> Map[user_id: Int, Set(Events)]
 106     """
 107     events_by_user = defaultdict(set)
 108     team_actors = [actor for actor in six.iterkeys(events_by_actor) if actor.type == Team]
 109     teams_to_user_ids = team_actors_to_user_ids(team_actors, user_ids)
 110     for actor, events in six.iteritems(events_by_actor):
 111         if actor.type == Team:
 112             try:
 113                 team_user_ids = teams_to_user_ids[actor.id]
 114             except KeyError:
 115                 pass
 116             else:
 117                 for user_id in team_user_ids:
 118                     events_by_user[user_id].update(events)
 119         elif actor.type == User:
 120             events_by_user[actor.id].update(events)
 121         else:
 122             raise ValueError("Unknown Actor type: %s" % actor.type)
 123     return events_by_user

 126 def team_actors_to_user_ids(team_actors, user_ids):
 127     """
 128     team_actors_to_user_ids(team_actors: List(Actors), user_ids: List(Int)) -> Map[team_id:Int, user_ids:Set(Int)]
 129 
 130     Will not include a team in the result if there are no active members in a team.
 131     """
 132     team_ids = [actor.id for actor in team_actors]
 133     members = OrganizationMemberTeam.objects.filter(
 134         team_id__in=team_ids, is_active=True, organizationmember__user_id__in=user_ids
 135     ).select_related("organizationmember")
 136 
 137     team_members = defaultdict(set)
 138     for member in members:
 139         user_id = member.organizationmember.user_id
 140         team_members[member.team_id].add(user_id)
 141 
 142     return team_members
