  21 def split_key(key):
  22     from sentry.mail.adapter import ActionTargetType
  23 
  24     key_parts = key.split(":", 4)
  25     project_id = key_parts[2]
  26     # XXX: We transitioned to new style keys (len == 5) a while ago on sentry.io. But
  27     # on-prem users might transition at any time, so we need to keep this transition
  28     # code around for a while, maybe indefinitely.
  29     if len(key_parts) == 5:
  30         target_type = ActionTargetType(key_parts[3])
  31         target_identifier = key_parts[4] if key_parts[4] else None
  32     else:
  33         target_type = ActionTargetType.ISSUE_OWNERS
  34         target_identifier = None
  35     return Project.objects.get(pk=project_id), target_type, target_identifier

  55 def fetch_state(project, records):
  56     # This reads a little strange, but remember that records are returned in
  57     # reverse chronological order, and we query the database in chronological
  58     # order.
  59     # NOTE: This doesn't account for any issues that are filtered out later.
  60     start = records[-1].datetime
  61     end = records[0].datetime
  62 
  63     groups = Group.objects.in_bulk(record.value.event.group_id for record in records)
  64     return {
  65         "project": project,
  66         "groups": groups,
  67         "rules": Rule.objects.in_bulk(
  68             itertools.chain.from_iterable(record.value.rules for record in records)
  69         ),
  70         "event_counts": tsdb.get_sums(tsdb.models.group, groups.keys(), start, end),
  71         "user_counts": tsdb.get_distinct_counts_totals(
  72             tsdb.models.users_affected_by_group, groups.keys(), start, end
  73         ),
  74     }

  77 def attach_state(project, groups, rules, event_counts, user_counts):
  78     for id, group in six.iteritems(groups):
  79         assert group.project_id == project.id, "Group must belong to Project"
  80         group.project = project
  81         group.event_count = 0
  82         group.user_count = 0
  83 
  84     for id, rule in six.iteritems(rules):
  85         assert rule.project_id == project.id, "Rule must belong to Project"
  86         rule.project = project
  87 
  88     for id, event_count in six.iteritems(event_counts):
  89         groups[id].event_count = event_count
  90 
  91     for id, user_count in six.iteritems(user_counts):
  92         groups[id].user_count = user_count
  93 
  94     return {"project": project, "groups": groups, "rules": rules}

  97 class Pipeline(object):
  98     def __init__(self):
  99         self.operations = []
 100 
 101     def __call__(self, sequence):
 102         return reduce(lambda x, operation: operation(x), self.operations, sequence)
 103 
 104     def apply(self, function):
 105         def operation(sequence):
 106             result = function(sequence)
 107             logger.debug("%r applied to %s items.", function, len(sequence))
 108             return result
 109 
 110         self.operations.append(operation)
 111         return self
 112 
 113     def filter(self, function):
 114         def operation(sequence):
 115             result = [s for s in sequence if function(s)]
 116             logger.debug("%r filtered %s items to %s.", function, len(sequence), len(result))
 117             return result
 118 
 119         self.operations.append(operation)
 120         return self
 121 
 122     def map(self, function):
 123         def operation(sequence):
 124             result = [function(s) for s in sequence]
 125             logger.debug("%r applied to %s items.", function, len(sequence))
 126             return result
 127 
 128         self.operations.append(operation)
 129         return self
 130 
 131     def reduce(self, function, initializer):
 132         def operation(sequence):
 133             result = reduce(function, sequence, initializer(sequence))
 134             logger.debug("%r reduced %s items to %s.", function, len(sequence), len(result))
 135             return result
 136 
 137         self.operations.append(operation)
 138         return self

 141 def rewrite_record(record, project, groups, rules):
 142     event = record.value.event
 143 
 144     # Reattach the group to the event.
 145     group = groups.get(event.group_id)
 146     if group is not None:
 147         event.group = group
 148     else:
 149         logger.debug("%r could not be associated with a group.", record)
 150         return
 151 
 152     return Record(
 153         record.key,
 154         Notification(event, [_f for _f in [rules.get(id) for id in record.value.rules] if _f]),
 155         record.timestamp,
 156     )

 159 def group_records(groups, record):
 160     group = record.value.event.group
 161     rules = record.value.rules
 162     if not rules:
 163         logger.debug("%r has no associated rules, and will not be added to any groups.", record)
 164 
 165     for rule in rules:
 166         groups[rule][group].append(record)
 167 
 168     return groups

 171 def sort_group_contents(rules):
 172     for key, groups in six.iteritems(rules):
 173         rules[key] = OrderedDict(
 174             sorted(
 175                 groups.items(),
 176                 # x = (group, records)
 177                 key=lambda x: (x[0].event_count, x[0].user_count),
 178                 reverse=True,
 179             )
 180         )
 181     return rules
 182 
 183 
 184 def sort_rule_groups(rules):
 185     return OrderedDict(
 186         sorted(
 187             rules.items(),
 188             # x = (rule, groups)
 189             key=lambda x: len(x[1]),
 190             reverse=True,
 191         )
 192     )

 195 def build_digest(project, records, state=None):
 196     records = list(records)
 197     if not records:
 198         return
 199 
 200     # XXX: This is a hack to allow generating a mock digest without actually
 201     # doing any real IO!
 202     if state is None:
 203         state = fetch_state(project, records)
 204 
 205     state = attach_state(**state)
 206 
 207     def check_group_state(record):
 208         return record.value.event.group.get_status() == GroupStatus.UNRESOLVED
 209 
 210     pipeline = (
 211         Pipeline()
 212         .map(functools.partial(rewrite_record, **state))
 213         .filter(bool)
 214         .filter(check_group_state)
 215         .reduce(group_records, lambda sequence: defaultdict(lambda: defaultdict(list)))
 216         .apply(sort_group_contents)
 217         .apply(sort_rule_groups)
 218     )
 219 
 220     return pipeline(records)
