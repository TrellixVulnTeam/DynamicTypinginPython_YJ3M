  72 def build_query_params_from_request(request, organization, projects, environments):
  73     query_kwargs = {"projects": projects, "sort_by": request.GET.get("sort", DEFAULT_SORT_OPTION)}
  74 
  75     limit = request.GET.get("limit")
  76     if limit:
  77         try:
  78             query_kwargs["limit"] = int(limit)
  79         except ValueError:
  80             raise ValidationError("invalid limit")
  81 
  82     # TODO: proper pagination support
  83     if request.GET.get("cursor"):
  84         try:
  85             query_kwargs["cursor"] = Cursor.from_string(request.GET.get("cursor"))
  86         except ValueError:
  87             raise ParseError(detail="Invalid cursor parameter.")
  88     query = request.GET.get("query", "is:unresolved").strip()
  89     if query:
  90         try:
  91             search_filters = convert_query_values(
  92                 parse_search_query(query), projects, request.user, environments
  93             )
  94         except InvalidSearchQuery as e:
  95             raise ValidationError(
  96                 u"Your search query could not be parsed: {}".format(six.text_type(e))
  97             )
  98 
  99         validate_search_filter_permissions(organization, search_filters, request.user)
 100         query_kwargs["search_filters"] = search_filters
 101 
 102     return query_kwargs

 113 def validate_search_filter_permissions(organization, search_filters, user):
 114     """
 115     Verifies that an organization is allowed to perform the query that they
 116     submitted.
 117     If the org is using a feature they don't have access to, raises
 118     `ValidationError` with information which part of the query they don't have
 119     access to.
 120     :param search_filters:
 121     """
 122     # If the organization has advanced search, then no need to perform any
 123     # other checks since they're allowed to use all search features
 124     if features.has("organizations:advanced-search", organization):
 125         return
 126 
 127     for search_filter in search_filters:
 128         for feature_condition, feature_name in advanced_search_features:
 129             if feature_condition(search_filter):
 130                 advanced_search_feature_gated.send_robust(
 131                     user=user, organization=organization, sender=validate_search_filter_permissions
 132                 )
 133                 raise ValidationError(
 134                     u"You need access to the advanced search feature to use {}".format(feature_name)
 135                 )

 156 class InCommitValidator(serializers.Serializer):
 157     commit = serializers.CharField(required=True)
 158     repository = serializers.CharField(required=True)
 159 
 160     def validate_repository(self, value):
 161         project = self.context["project"]
 162         try:
 163             value = Repository.objects.get(organization_id=project.organization_id, name=value)
 164         except Repository.DoesNotExist:
 165             raise serializers.ValidationError("Unable to find the given repository.")
 166         return value
 167 
 168     def validate(self, attrs):
 169         attrs = super(InCommitValidator, self).validate(attrs)
 170         repository = attrs.get("repository")
 171         commit = attrs.get("commit")
 172         if not repository:
 173             raise serializers.ValidationError(
 174                 {"repository": ["Unable to find the given repository."]}
 175             )
 176         if not commit:
 177             raise serializers.ValidationError({"commit": ["Unable to find the given commit."]})
 178         try:
 179             commit = Commit.objects.get(repository_id=repository.id, key=commit)
 180         except Commit.DoesNotExist:
 181             raise serializers.ValidationError({"commit": ["Unable to find the given commit."]})
 182         return commit

 198         project = self.context["project"]
 199         if value == "latest":
 200             try:
 201                 value = (
 202                     Release.objects.filter(
 203                         projects=project, organization_id=project.organization_id
 204                     )

 260     def validate_assignedTo(self, value):
 261         if (
 262             value
 263             and value.type is User
 264             and not self.context["project"].member_set.filter(user_id=value.id).exists()
 265         ):
 266             raise serializers.ValidationError("Cannot assign to non-team member")

 269             value

 292 def handle_discard(request, group_list, projects, user):
 293     for project in projects:
 294         if not features.has("projects:discard-groups", project, actor=user):
 295             return Response({"detail": ["You do not have that feature enabled"]}, status=400)
 296 
 297     # grouped by project_id
 298     groups_to_delete = defaultdict(list)
 299 
 300     for group in group_list:
 301         with transaction.atomic():
 302             try:
 303                 tombstone = GroupTombstone.objects.create(
 304                     previous_group_id=group.id,
 305                     actor_id=user.id if user else None,
 306                     **{name: getattr(group, name) for name in TOMBSTONE_FIELDS_FROM_GROUP}
 307                 )
 308             except IntegrityError:
 309                 # in this case, a tombstone has already been created
 310                 # for a group, so no hash updates are necessary
 311                 pass
 312             else:
 313                 groups_to_delete[group.project_id].append(group)
 314 
 315                 GroupHash.objects.filter(group=group).update(
 316                     group=None, group_tombstone_id=tombstone.id
 317                 )
 318 
 319     for project in projects:
 320         _delete_groups(request, project, groups_to_delete.get(project.id), delete_type="discard")
 321 
 322     return Response(status=204)

 325 def _delete_groups(request, project, group_list, delete_type):
 326     if not group_list:
 327         return
 328 
 329     # deterministic sort for sanity, and for very large deletions we'll
 330     # delete the "smaller" groups first
 331     group_list.sort(key=lambda g: (g.times_seen, g.id))
 332     group_ids = [g.id for g in group_list]
 333 
 334     Group.objects.filter(id__in=group_ids).exclude(
 335         status__in=[GroupStatus.PENDING_DELETION, GroupStatus.DELETION_IN_PROGRESS]
 336     ).update(status=GroupStatus.PENDING_DELETION)
 337 
 338     eventstream_state = eventstream.start_delete_groups(project.id, group_ids)
 339     transaction_id = uuid4().hex
 340 
 341     GroupHash.objects.filter(project_id=project.id, group__id__in=group_ids).delete()
 342 
 343     delete_groups_task.apply_async(
 344         kwargs={
 345             "object_ids": group_ids,
 346             "transaction_id": transaction_id,
 347             "eventstream_state": eventstream_state,
 348         },
 349         countdown=3600,
 350     )
 351 
 352     for group in group_list:
 353         create_audit_entry(
 354             request=request,
 355             transaction_id=transaction_id,
 356             logger=audit_logger,
 357             organization_id=project.organization_id,
 358             target_object=group.id,
 359         )
 360 
 361         delete_logger.info(
 362             "object.delete.queued",
 363             extra={
 364                 "object_id": group.id,
 365                 "transaction_id": transaction_id,
 366                 "model": type(group).__name__,
 367             },
 368         )
 369 
 370         issue_deleted.send_robust(
 371             group=group, user=request.user, delete_type=delete_type, sender=_delete_groups
 372         )

 375 def delete_groups(request, projects, organization_id, search_fn):
 376     """
 377     `search_fn` refers to the `search.query` method with the appropriate
 378     project, org, environment, and search params already bound
 379     """
 380     group_ids = request.GET.getlist("id")
 381     if group_ids:
 382         group_list = list(
 383             Group.objects.filter(
 384                 project__in=projects,
 385                 project__organization_id=organization_id,
 386                 id__in=set(group_ids),
 387             ).exclude(status__in=[GroupStatus.PENDING_DELETION, GroupStatus.DELETION_IN_PROGRESS])
 388         )
 389     else:
 390         try:
 391             # bulk mutations are limited to 1000 items
 392             # TODO(dcramer): it'd be nice to support more than this, but its
 393             # a bit too complicated right now
 394             cursor_result, _ = search_fn({"limit": 1000, "paginator_options": {"max_limit": 1000}})
 395         except ValidationError as exc:
 396             return Response({"detail": six.text_type(exc)}, status=400)
 397 
 398         group_list = list(cursor_result)
 399 
 400     if not group_list:
 401         return Response(status=204)
 402 
 403     groups_by_project_id = defaultdict(list)
 404     for group in group_list:
 405         groups_by_project_id[group.project_id].append(group)
 406 
 407     for project in projects:
 408         _delete_groups(request, project, groups_by_project_id.get(project.id), delete_type="delete")
 409 
 410     return Response(status=204)

 413 def self_subscribe_and_assign_issue(acting_user, group):
 414     # Used during issue resolution to assign to acting user
 415     # returns None if the user didn't elect to self assign on resolution
 416     # or the group is assigned already, otherwise returns Actor
 417     # representation of current user
 418     if acting_user:
 419         GroupSubscription.objects.subscribe(
 420             user=acting_user, group=group, reason=GroupSubscriptionReason.status_change
 421         )
 422         self_assign_issue = UserOption.objects.get_value(
 423             user=acting_user, key="self_assign_issue", default="0"
 424         )
 425         if self_assign_issue == "1" and not group.assignee_set.exists():
 426             return Actor(type=User, id=acting_user.id)

 429 def track_update_groups(function):
 430     def wrapper(request, projects, *args, **kwargs):
 431         try:
 432             response = function(request, projects, *args, **kwargs)
 433         except Exception:
 434             metrics.incr("group.update.http_response", sample_rate=1.0, tags={"status": 500})
 435             # Continue raising the error now that we've incr the metric
 436             raise
 437 
 438         serializer = GroupValidator(
 439             data=request.data,
 440             partial=True,
 441             context={"project": projects[0], "access": getattr(request, "access", None)},
 442         )
 443         results = dict(serializer.validated_data) if serializer.is_valid() else {}
 444         tags = {key: True for key in results.keys()}
 445         tags["status"] = response.status_code
 446 
 447         metrics.incr("group.update.http_response", sample_rate=1.0, tags=tags)
 448         return response
 449 
 450     return wrapper

 454 def update_groups(request, projects, organization_id, search_fn):
 455     group_ids = request.GET.getlist("id")
 456     if group_ids:
 457         group_list = Group.objects.filter(
 458             project__organization_id=organization_id, project__in=projects, id__in=group_ids
 459         )
 460         # filter down group ids to only valid matches
 461         group_ids = [g.id for g in group_list]
 462         if not group_ids:
 463             return Response(status=204)
 464     else:
 465         group_list = None
 466 
 467     # TODO(jess): We may want to look into refactoring GroupValidator
 468     # to support multiple projects, but this is pretty complicated
 469     # because of the assignee validation. Punting on this for now.
 470     for project in projects:
 471         serializer = GroupValidator(
 472             data=request.data,
 473             partial=True,
 474             context={"project": project, "access": getattr(request, "access", None)},
 475         )
 476         if not serializer.is_valid():
 477             return Response(serializer.errors, status=400)
 478 
 479     result = dict(serializer.validated_data)
 480 
 481     # so we won't have to requery for each group
 482     project_lookup = {p.id: p for p in projects}
 483 
 484     acting_user = request.user if request.user.is_authenticated() else None
 485 
 486     if not group_ids:
 487         try:
 488             # bulk mutations are limited to 1000 items
 489             # TODO(dcramer): it'd be nice to support more than this, but its
 490             # a bit too complicated right now
 491             cursor_result, _ = search_fn({"limit": 1000, "paginator_options": {"max_limit": 1000}})
 492         except ValidationError as exc:
 493             return Response({"detail": six.text_type(exc)}, status=400)
 494 
 495         group_list = list(cursor_result)
 496         group_ids = [g.id for g in group_list]
 497 
 498     is_bulk = len(group_ids) > 1
 499 
 500     group_project_ids = {g.project_id for g in group_list}
 501     # filter projects down to only those that have groups in the search results
 502     projects = [p for p in projects if p.id in group_project_ids]
 503 
 504     queryset = Group.objects.filter(id__in=group_ids)
 505 
 506     discard = result.get("discard")
 507     if discard:
 508 
 509         return handle_discard(request, list(queryset), projects, acting_user)
 510 
 511     statusDetails = result.pop("statusDetails", result)
 512     status = result.get("status")
 513     release = None
 514     commit = None
 515 
 516     if status in ("resolved", "resolvedInNextRelease"):
 517         if status == "resolvedInNextRelease" or statusDetails.get("inNextRelease"):
 518             # TODO(jess): We may want to support this for multi project, but punting on it for now
 519             if len(projects) > 1:
 520                 return Response(
 521                     {"detail": "Cannot set resolved in next release for multiple projects."},
 522                     status=400,
 523                 )
 524             release = (
 525                 statusDetails.get("inNextRelease")
 526                 or Release.objects.filter(
 527                     projects=projects[0], organization_id=projects[0].organization_id
 528                 )
 529                 .extra(select={"sort": "COALESCE(date_released, date_added)"})
 530                 .order_by("-sort")[0]
 531             )
 532             activity_type = Activity.SET_RESOLVED_IN_RELEASE
 533             activity_data = {
 534                 # no version yet
 535                 "version": ""
 536             }
 537             status_details = {
 538                 "inNextRelease": True,
 539                 "actor": serialize(extract_lazy_object(request.user), request.user),
 540             }
 541             res_type = GroupResolution.Type.in_next_release
 542             res_type_str = "in_next_release"
 543             res_status = GroupResolution.Status.pending
 544         elif statusDetails.get("inRelease"):
 545             # TODO(jess): We could update validation to check if release
 546             # applies to multiple projects, but I think we agreed to punt
 547             # on this for now
 548             if len(projects) > 1:
 549                 return Response(
 550                     {"detail": "Cannot set resolved in release for multiple projects."}, status=400
 551                 )
 552             release = statusDetails["inRelease"]
 553             activity_type = Activity.SET_RESOLVED_IN_RELEASE
 554             activity_data = {
 555                 # no version yet
 556                 "version": release.version
 557             }
 558             status_details = {
 559                 "inRelease": release.version,
 560                 "actor": serialize(extract_lazy_object(request.user), request.user),
 561             }
 562             res_type = GroupResolution.Type.in_release
 563             res_type_str = "in_release"
 564             res_status = GroupResolution.Status.resolved
 565         elif statusDetails.get("inCommit"):
 566             # TODO(jess): Same here, this is probably something we could do, but
 567             # punting for now.
 568             if len(projects) > 1:
 569                 return Response(
 570                     {"detail": "Cannot set resolved in commit for multiple projects."}, status=400
 571                 )
 572             commit = statusDetails["inCommit"]
 573             activity_type = Activity.SET_RESOLVED_IN_COMMIT
 574             activity_data = {"commit": commit.id}
 575             status_details = {
 576                 "inCommit": serialize(commit, request.user),
 577                 "actor": serialize(extract_lazy_object(request.user), request.user),
 578             }
 579             res_type_str = "in_commit"
 580         else:
 581             res_type_str = "now"
 582             activity_type = Activity.SET_RESOLVED
 583             activity_data = {}
 584             status_details = {}
 585 
 586         now = timezone.now()
 587         metrics.incr("group.resolved", instance=res_type_str, skip_internal=True)
 588 
 589         # if we've specified a commit, let's see if its already been released
 590         # this will allow us to associate the resolution to a release as if we
 591         # were simply using 'inRelease' above
 592         # Note: this is different than the way commit resolution works on deploy
 593         # creation, as a given deploy is connected to an explicit release, and
 594         # in this case we're simply choosing the most recent release which contains
 595         # the commit.
 596         if commit and not release:
 597             # TODO(jess): If we support multiple projects for release / commit resolution,
 598             # we need to update this to find the release for each project (we shouldn't assume
 599             # it's the same)
 600             try:
 601                 release = (
 602                     Release.objects.filter(projects__in=projects, releasecommit__commit=commit)
 603                     .extra(select={"sort": "COALESCE(date_released, date_added)"})
 604                     .order_by("-sort")[0]
 605                 )
 606                 res_type = GroupResolution.Type.in_release
 607                 res_status = GroupResolution.Status.resolved
 608             except IndexError:
 609                 release = None
 610 
 611         for group in group_list:
 612             with transaction.atomic():
 613                 resolution = None
 614                 if release:
 615                     resolution_params = {
 616                         "release": release,
 617                         "type": res_type,
 618                         "status": res_status,
 619                         "actor_id": request.user.id if request.user.is_authenticated() else None,
 620                     }
 621                     resolution, created = GroupResolution.objects.get_or_create(
 622                         group=group, defaults=resolution_params
 623                     )
 624                     if not created:
 625                         resolution.update(datetime=timezone.now(), **resolution_params)
 626 
 627                 if commit:
 628                     GroupLink.objects.create(
 629                         group_id=group.id,
 630                         project_id=group.project_id,
 631                         linked_type=GroupLink.LinkedType.commit,
 632                         relationship=GroupLink.Relationship.resolves,
 633                         linked_id=commit.id,
 634                     )
 635 
 636                 affected = Group.objects.filter(id=group.id).update(
 637                     status=GroupStatus.RESOLVED, resolved_at=now
 638                 )
 639                 if not resolution:
 640                     created = affected
 641 
 642                 group.status = GroupStatus.RESOLVED
 643                 group.resolved_at = now
 644 
 645                 assigned_to = self_subscribe_and_assign_issue(acting_user, group)
 646                 if assigned_to is not None:
 647                     result["assignedTo"] = assigned_to
 648 
 649                 if created:
 650                     activity = Activity.objects.create(
 651                         project=project_lookup[group.project_id],
 652                         group=group,
 653                         type=activity_type,
 654                         user=acting_user,
 655                         ident=resolution.id if resolution else None,
 656                         data=activity_data,
 657                     )
 658                     # TODO(dcramer): we need a solution for activity rollups
 659                     # before sending notifications on bulk changes
 660                     if not is_bulk:
 661                         activity.send_notification()
 662 
 663             issue_resolved.send_robust(
 664                 organization_id=organization_id,
 665                 user=acting_user or request.user,
 666                 group=group,
 667                 project=project_lookup[group.project_id],
 668                 resolution_type=res_type_str,
 669                 sender=update_groups,
 670             )
 671 
 672             kick_off_status_syncs.apply_async(
 673                 kwargs={"project_id": group.project_id, "group_id": group.id}
 674             )
 675 
 676         result.update({"status": "resolved", "statusDetails": status_details})
 677 
 678     elif status:
 679         new_status = STATUS_CHOICES[result["status"]]
 680 
 681         with transaction.atomic():
 682             happened = queryset.exclude(status=new_status).update(status=new_status)
 683 
 684             GroupResolution.objects.filter(group__in=group_ids).delete()
 685 
 686             if new_status == GroupStatus.IGNORED:
 687                 metrics.incr("group.ignored", skip_internal=True)
 688 
 689                 ignore_duration = (
 690                     statusDetails.pop("ignoreDuration", None)
 691                     or statusDetails.pop("snoozeDuration", None)
 692                 ) or None
 693                 ignore_count = statusDetails.pop("ignoreCount", None) or None
 694                 ignore_window = statusDetails.pop("ignoreWindow", None) or None
 695                 ignore_user_count = statusDetails.pop("ignoreUserCount", None) or None
 696                 ignore_user_window = statusDetails.pop("ignoreUserWindow", None) or None
 697                 if ignore_duration or ignore_count or ignore_user_count:
 698                     if ignore_duration:
 699                         ignore_until = timezone.now() + timedelta(minutes=ignore_duration)
 700                     else:
 701                         ignore_until = None
 702                     for group in group_list:
 703                         state = {}
 704                         if ignore_count and not ignore_window:
 705                             state["times_seen"] = group.times_seen
 706                         if ignore_user_count and not ignore_user_window:
 707                             state["users_seen"] = group.count_users_seen()
 708                         GroupSnooze.objects.create_or_update(
 709                             group=group,
 710                             values={
 711                                 "until": ignore_until,
 712                                 "count": ignore_count,
 713                                 "window": ignore_window,
 714                                 "user_count": ignore_user_count,
 715                                 "user_window": ignore_user_window,
 716                                 "state": state,
 717                                 "actor_id": request.user.id
 718                                 if request.user.is_authenticated()
 719                                 else None,
 720                             },
 721                         )
 722                         result["statusDetails"] = {
 723                             "ignoreCount": ignore_count,
 724                             "ignoreUntil": ignore_until,
 725                             "ignoreUserCount": ignore_user_count,
 726                             "ignoreUserWindow": ignore_user_window,
 727                             "ignoreWindow": ignore_window,
 728                             "actor": serialize(extract_lazy_object(request.user), request.user),
 729                         }
 730                 else:
 731                     GroupSnooze.objects.filter(group__in=group_ids).delete()
 732                     ignore_until = None
 733                     result["statusDetails"] = {}
 734             else:
 735                 result["statusDetails"] = {}
 736 
 737         if group_list and happened:
 738             if new_status == GroupStatus.UNRESOLVED:
 739                 activity_type = Activity.SET_UNRESOLVED
 740                 activity_data = {}
 741             elif new_status == GroupStatus.IGNORED:
 742                 activity_type = Activity.SET_IGNORED
 743                 activity_data = {
 744                     "ignoreCount": ignore_count,
 745                     "ignoreDuration": ignore_duration,
 746                     "ignoreUntil": ignore_until,
 747                     "ignoreUserCount": ignore_user_count,
 748                     "ignoreUserWindow": ignore_user_window,
 749                     "ignoreWindow": ignore_window,
 750                 }
 751 
 752                 groups_by_project_id = defaultdict(list)
 753                 for group in group_list:
 754                     groups_by_project_id[group.project_id].append(group)
 755 
 756                 for project in projects:
 757                     project_groups = groups_by_project_id.get(project.id)
 758                     if project_groups:
 759                         issue_ignored.send_robust(
 760                             project=project,
 761                             user=acting_user,
 762                             group_list=project_groups,
 763                             activity_data=activity_data,
 764                             sender=update_groups,
 765                         )
 766 
 767             for group in group_list:
 768                 group.status = new_status
 769 
 770                 activity = Activity.objects.create(
 771                     project=project_lookup[group.project_id],
 772                     group=group,
 773                     type=activity_type,
 774                     user=acting_user,
 775                     data=activity_data,
 776                 )
 777                 # TODO(dcramer): we need a solution for activity rollups
 778                 # before sending notifications on bulk changes
 779                 if not is_bulk:
 780                     if acting_user:
 781                         GroupSubscription.objects.subscribe(
 782                             user=acting_user,
 783                             group=group,
 784                             reason=GroupSubscriptionReason.status_change,
 785                         )
 786                     activity.send_notification()
 787 
 788                 if new_status == GroupStatus.UNRESOLVED:
 789                     kick_off_status_syncs.apply_async(
 790                         kwargs={"project_id": group.project_id, "group_id": group.id}
 791                     )
 792 
 793     if "assignedTo" in result:
 794         assigned_actor = result["assignedTo"]
 795         if assigned_actor:
 796             for group in group_list:
 797                 resolved_actor = assigned_actor.resolve()
 798 
 799                 GroupAssignee.objects.assign(group, resolved_actor, acting_user)
 800             result["assignedTo"] = serialize(
 801                 assigned_actor.resolve(), acting_user, ActorSerializer()
 802             )
 803         else:
 804             for group in group_list:
 805                 GroupAssignee.objects.deassign(group, acting_user)
 806 
 807     is_member_map = {
 808         project.id: project.member_set.filter(user=acting_user).exists() for project in projects
 809     }
 810     if result.get("hasSeen"):
 811         for group in group_list:
 812             if is_member_map.get(group.project_id):
 813                 instance, created = create_or_update(
 814                     GroupSeen,
 815                     group=group,
 816                     user=acting_user,
 817                     project=project_lookup[group.project_id],
 818                     values={"last_seen": timezone.now()},
 819                 )
 820     elif result.get("hasSeen") is False:
 821         GroupSeen.objects.filter(group__in=group_ids, user=acting_user).delete()
 822 
 823     if result.get("isBookmarked"):
 824         for group in group_list:
 825             GroupBookmark.objects.get_or_create(
 826                 project=project_lookup[group.project_id], group=group, user=acting_user
 827             )
 828             GroupSubscription.objects.subscribe(
 829                 user=acting_user, group=group, reason=GroupSubscriptionReason.bookmark
 830             )
 831     elif result.get("isBookmarked") is False:
 832         GroupBookmark.objects.filter(group__in=group_ids, user=acting_user).delete()
 833 
 834     # TODO(dcramer): we could make these more efficient by first
 835     # querying for rich rows are present (if N > 2), flipping the flag
 836     # on those rows, and then creating the missing rows
 837     if result.get("isSubscribed") in (True, False):
 838         is_subscribed = result["isSubscribed"]
 839         for group in group_list:
 840             # NOTE: Subscribing without an initiating event (assignment,
 841             # commenting, etc.) clears out the previous subscription reason
 842             # to avoid showing confusing messaging as a result of this
 843             # action. It'd be jarring to go directly from "you are not
 844             # subscribed" to "you were subscribed due since you were
 845             # assigned" just by clicking the "subscribe" button (and you
 846             # may no longer be assigned to the issue anyway.)
 847             GroupSubscription.objects.create_or_update(
 848                 user=acting_user,
 849                 group=group,
 850                 project=project_lookup[group.project_id],
 851                 values={"is_active": is_subscribed, "reason": GroupSubscriptionReason.unknown},
 852             )
 853 
 854         result["subscriptionDetails"] = {
 855             "reason": SUBSCRIPTION_REASON_MAP.get(GroupSubscriptionReason.unknown, "unknown")
 856         }
 857 
 858     if "isPublic" in result:
 859         # We always want to delete an existing share, because triggering
 860         # an isPublic=True even when it's already public, should trigger
 861         # regenerating.
 862         for group in group_list:
 863             if GroupShare.objects.filter(group=group).delete():
 864                 result["shareId"] = None
 865                 Activity.objects.create(
 866                     project=project_lookup[group.project_id],
 867                     group=group,
 868                     type=Activity.SET_PRIVATE,
 869                     user=acting_user,
 870                 )
 871 
 872     if result.get("isPublic"):
 873         for group in group_list:
 874             share, created = GroupShare.objects.get_or_create(
 875                 project=project_lookup[group.project_id], group=group, user=acting_user
 876             )
 877             if created:
 878                 result["shareId"] = share.uuid
 879                 Activity.objects.create(
 880                     project=project_lookup[group.project_id],
 881                     group=group,
 882                     type=Activity.SET_PUBLIC,
 883                     user=acting_user,
 884                 )
 885 
 886     # XXX(dcramer): this feels a bit shady like it should be its own
 887     # endpoint
 888     if result.get("merge") and len(group_list) > 1:
 889         # don't allow merging cross project
 890         if len(projects) > 1:
 891             return Response({"detail": "Merging across multiple projects is not supported"})
 892         group_list_by_times_seen = sorted(
 893             group_list, key=lambda g: (g.times_seen, g.id), reverse=True
 894         )
 895         primary_group, groups_to_merge = group_list_by_times_seen[0], group_list_by_times_seen[1:]
 896 
 897         group_ids_to_merge = [g.id for g in groups_to_merge]
 898         eventstream_state = eventstream.start_merge(
 899             primary_group.project_id, group_ids_to_merge, primary_group.id
 900         )
 901 
 902         Group.objects.filter(id__in=group_ids_to_merge).update(status=GroupStatus.PENDING_MERGE)
 903 
 904         transaction_id = uuid4().hex
 905         merge_groups.delay(
 906             from_object_ids=group_ids_to_merge,
 907             to_object_id=primary_group.id,
 908             transaction_id=transaction_id,
 909             eventstream_state=eventstream_state,
 910         )
 911 
 912         Activity.objects.create(
 913             project=project_lookup[primary_group.project_id],
 914             group=primary_group,
 915             type=Activity.MERGE,
 916             user=acting_user,
 917             data={"issues": [{"id": c.id} for c in groups_to_merge]},
 918         )
 919 
 920         result["merge"] = {
 921             "parent": six.text_type(primary_group.id),
 922             "children": [six.text_type(g.id) for g in groups_to_merge],
 923         }
 924 
 925     return Response(result)
