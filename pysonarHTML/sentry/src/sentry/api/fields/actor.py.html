  12 class Actor(namedtuple("Actor", "id type")):
  13     def get_actor_id(self):
  14         return "%s:%d" % (self.type.__name__.lower(), self.id)
  15 
  16     @classmethod
  17     def from_actor_identifier(cls, actor_identifier):
  18         """
  19         Returns an Actor tuple corresponding to a User or Team associated with
  20         the given identifier.
  21 
  22         Forms `actor_identifier` can take:
  23             1231 -> look up User by id
  24             "1231" -> look up User by id
  25             "user:1231" -> look up User by id
  26             "team:1231" -> look up Team by id
  27             "maiseythedog" -> look up User by username
  28             "maisey@dogsrule.com" -> look up User by primary email
  29         """
  30         # If we have an integer, fall back to assuming it's a User
  31         if isinstance(actor_identifier, six.integer_types):
  32             return Actor(actor_identifier, User)
  33 
  34         # If the actor_identifier is a simple integer as a string,
  35         # we're also a User
  36         if actor_identifier.isdigit():
  37             return Actor(int(actor_identifier), User)
  38 
  39         if actor_identifier.startswith("user:"):
  40             return cls(int(actor_identifier[5:]), User)
  41 
  42         if actor_identifier.startswith("team:"):
  43             return cls(int(actor_identifier[5:]), Team)
  44 
  45         try:
  46             return Actor(find_users(actor_identifier)[0].id, User)
  47         except IndexError:
  48             raise serializers.ValidationError("Unable to resolve actor identifier")

  50     def resolve(self):
  51         return self.type.objects.get(id=self.id)
  52 
  53     @classmethod
  54     def resolve_many(cls, actors):
  55         if not actors:
  56             return []
  57 
  58         actors_by_type = defaultdict(list)
  59         for actor in actors:
  60             actors_by_type[actor.type].append(actor)
  61 
  62         results = []
  63         for type, actors in actors_by_type.items():
  64             results.extend(list(type.objects.filter(id__in=[a.id for a in actors])))
  65 
  66         return results
  67 
  68     @classmethod
  69     def resolve_dict(cls, actor_dict):
  70         actors_by_type = defaultdict(list)
  71         for actor in actor_dict.values():
  72             actors_by_type[actor.type].append(actor)
  73 
  74         resolved_actors = {}
  75         for type, actors in actors_by_type.items():
  76             resolved_actors[type] = {
  77                 actor.id: actor for actor in type.objects.filter(id__in=[a.id for a in actors])
  78             }
  79 
  80         return {key: resolved_actors[value.type][value.id] for key, value in actor_dict.items()}

  84     def to_representation(self, value):

  87     def to_internal_value(self, data):
  88         if not data:
  89             return None
  90 
  91         try:
  92             return Actor.from_actor_identifier(data)
  93         except Exception:
  94             raise serializers.ValidationError("Unknown actor input")
