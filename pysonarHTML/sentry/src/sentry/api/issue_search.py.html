  25 class IssueSearchVisitor(SearchVisitor):
  26     key_mappings = {
  27         "assigned_to": ["assigned"],
  28         "bookmarked_by": ["bookmarks"],
  29         "subscribed_by": ["subscribed"],
  30         "first_release": ["first-release", "firstRelease"],
  31         "first_seen": ["age", "firstSeen"],
  32         "last_seen": ["lastSeen"],
  33         "active_at": ["activeSince"],
  34         # TODO: Special case this in the backends, since they currently rely
  35         # on date_from and date_to explicitly
  36         "date": ["event.timestamp"],
  37         "times_seen": ["timesSeen"],
  38         "sentry:dist": ["dist"],
  39     }
  40     numeric_keys = SearchVisitor.numeric_keys.union(["times_seen"])
  41     date_keys = SearchVisitor.date_keys.union(["active_at", "date"])
  42 
  43     @cached_property
  44     def is_filter_translators(self):
  45         is_filter_translators = {
  46             "assigned": (SearchKey("unassigned"), SearchValue(False)),
  47             "unassigned": (SearchKey("unassigned"), SearchValue(True)),
  48         }
  49         for status_key, status_value in STATUS_CHOICES.items():
  50             is_filter_translators[status_key] = (SearchKey("status"), SearchValue(status_value))
  51         return is_filter_translators
  52 
  53     def visit_is_filter(self, node, children):
  54         # the key is "is" here, which we don't need
  55         negation, _, _, search_value = children
  56 
  57         if search_value.raw_value not in self.is_filter_translators:
  58             raise InvalidSearchQuery(
  59                 'Invalid value for "is" search, valid values are {}'.format(
  60                     sorted(self.is_filter_translators.keys())
  61                 )
  62             )
  63 
  64         search_key, search_value = self.is_filter_translators[search_value.raw_value]
  65 
  66         operator = "!=" if self.is_negated(negation) else "="
  67 
  68         return SearchFilter(search_key, operator, search_value)
  69 
  70     def visit_boolean_operator(self, node, children):
  71         raise InvalidSearchQuery(
  72             'Boolean statements containing "OR" or "AND" are not supported in this search'
  73         )

  76 def parse_search_query(query):
  77     try:
  78         tree = event_search_grammar.parse(query)
  79     except IncompleteParseError as e:
  80         raise InvalidSearchQuery(
  81             "%s %s"
  82             % (
  83                 u"Parse error: %r (column %d)." % (e.expr.name, e.column()),
  84                 "This is commonly caused by unmatched-parentheses. Enclose any text in double quotes.",
  85             )
  86         )
  87     return IssueSearchVisitor().visit(tree)

 119 def convert_query_values(search_filters, projects, user, environments):
 120     """
 121     Accepts a collection of SearchFilter objects and converts their values into
 122     a specific format, based on converters specified in `value_converters`.
 123     :param search_filters: Collection of `SearchFilter` objects.
 124     :param projects: List of projects being searched across
 125     :param user: The user making the search
 126     :return: New collection of `SearchFilters`, which may have converted values.
 127     """
 128 
 129     def convert_search_filter(search_filter):
 130         if search_filter.key.name in value_converters:
 131             converter = value_converters[search_filter.key.name]
 132             new_value = converter(search_filter.value.raw_value, projects, user, environments)
 133             search_filter = search_filter._replace(value=SearchValue(new_value))
 134         elif isinstance(search_filter, AggregateFilter):
 135             raise InvalidSearchQuery(
 136                 u"Aggregate filters ({}) are not supported in issue searches.".format(
 137                     search_filter.key.name
 138                 )
 139             )
 140         return search_filter
 141 
 142     return map(convert_search_filter, search_filters)
