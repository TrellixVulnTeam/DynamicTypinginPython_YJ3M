  11 class AvatarSerializer(serializers.Serializer):
  12     avatar_photo = AvatarField(required=False)
  13     avatar_type = serializers.ChoiceField(
  14         choices=(("upload", "upload"), ("gravatar", "gravatar"), ("letter_avatar", "letter_avatar"))
  15     )
  16 
  17     def validate(self, attrs):
  18         attrs = super(AvatarSerializer, self).validate(attrs)
  19         if attrs.get("avatar_type") == "upload":
  20             model_type = self.context["type"]
  21             has_existing_file = model_type.objects.filter(
  22                 file__isnull=False, **self.context["kwargs"]
  23             ).exists()
  24             if not has_existing_file and not attrs.get("avatar_photo"):
  25                 raise serializers.ValidationError(
  26                     {"avatar_type": "Cannot set avatar_type to upload without avatar_photo"}
  27                 )
  28         return attrs

  31 class AvatarMixin(object):
  32     object_type = None
  33     model = None
  34 
  35     def get(self, request, **kwargs):
  36         obj = kwargs[self.object_type]
  37         return Response(serialize(obj, request.user))
  38 
  39     def get_serializer_context(self, obj, **kwargs):
  40         return {"type": self.model, "kwargs": {self.object_type: obj}}
  41 
  42     def get_avatar_filename(self, obj):
  43         return u"{}.png".format(obj.id)
  44 
  45     def put(self, request, **kwargs):
  46         obj = kwargs[self.object_type]
  47         serializer = AvatarSerializer(data=request.data, context=self.get_serializer_context(obj))
  48         if not serializer.is_valid():
  49             return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
  50 
  51         result = serializer.validated_data
  52 
  53         self.model.save_avatar(
  54             relation={self.object_type: obj},
  55             type=result["avatar_type"],
  56             avatar=result.get("avatar_photo"),
  57             filename=self.get_avatar_filename(obj),
  58         )
  59 
  60         return Response(serialize(obj, request.user))
