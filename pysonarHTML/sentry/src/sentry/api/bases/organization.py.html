  41     def is_not_2fa_compliant(self, request, organization):
  42         return (
  43             organization.flags.require_2fa
  44             and not Authenticator.objects.user_has_2fa(request.user)
  45             and not is_active_superuser(request)
  46         )

  48     def needs_sso(self, request, organization):
  49         # XXX(dcramer): this is very similar to the server-rendered views
  50         # logic for checking valid SSO
  51         if not request.access.requires_sso:
  52             return False
  53         if not auth.has_completed_sso(request, organization.id):
  54             return True
  55         if not request.access.sso_is_valid:
  56             return True
  57         return False

  59     def has_object_permission(self, request, view, organization):
  60         self.determine_access(request, organization)
  61         allowed_scopes = set(self.scope_map.get(request.method, []))
  62         return any(request.access.has_scope(s) for s in allowed_scopes)

 147     def get_requested_project_ids(self, request):
 148         """
 149         Returns the project ids that were requested by the request.
 150 
 151         To determine the projects to filter this endpoint by with full
 152         permission checking, use ``get_projects``, instead.
 153         """
 154         try:
 155             return set(map(int, request.GET.getlist("project")))
 156         except ValueError:
 157             raise ParseError(detail="Invalid project parameter. Values must be numbers.")
 158 
 159     def get_projects(
 160         self, request, organization, force_global_perms=False, include_all_accessible=False
 161     ):
 162         """
 163         Determines which project ids to filter the endpoint by. If a list of
 164         project ids is passed in via the `project` querystring argument then
 165         validate that these projects can be accessed. If not passed, then
 166         return all project ids that the user can access within this
 167         organization.
 168 
 169         :param request:
 170         :param organization: Organization to fetch projects for
 171         :param force_global_perms: Permission override. Allows subclasses to
 172         perform their own validation and allow the user to access any project
 173         in the organization. This is a hack to support the old
 174         `request.auth.has_scope` way of checking permissions, don't use it
 175         for anything else, we plan to remove this once we remove uses of
 176         `auth.has_scope`.
 177         :param include_all_accessible: Whether to factor the organization
 178         allow_joinleave flag into permission checks. We should ideally
 179         standardize how this is used and remove this parameter.
 180         :return: A list of Project objects, or raises PermissionDenied.
 181         """
 182         project_ids = self.get_requested_project_ids(request)
 183         return self._get_projects_by_id(
 184             project_ids, request, organization, force_global_perms, include_all_accessible
 185         )
 186 
 187     def _get_projects_by_id(
 188         self,
 189         project_ids,
 190         request,
 191         organization,
 192         force_global_perms=False,
 193         include_all_accessible=False,
 194     ):
 195         qs = Project.objects.filter(organization=organization, status=ProjectStatus.VISIBLE)
 196         user = getattr(request, "user", None)
 197 
 198         # A project_id of -1 means 'all projects I have access to'
 199         # While no project_ids means 'all projects I am a member of'.
 200         if project_ids == ALL_ACCESS_PROJECTS:
 201             include_all_accessible = True
 202             project_ids = set()
 203 
 204         requested_projects = project_ids.copy()
 205         if project_ids:
 206             qs = qs.filter(id__in=project_ids)
 207 
 208         with sentry_sdk.start_span(op="fetch_organization_projects") as span:
 209             projects = list(qs)
 210             span.set_data("Project Count", len(projects))
 211         with sentry_sdk.start_span(op="apply_project_permissions") as span:
 212             span.set_data("Project Count", len(projects))
 213             if force_global_perms:
 214                 span.set_tag("mode", "force_global_perms")
 215             else:
 216                 if (
 217                     user
 218                     and is_active_superuser(request)
 219                     or requested_projects
 220                     or include_all_accessible
 221                 ):
 222                     span.set_tag("mode", "has_project_access")
 223                     func = request.access.has_project_access
 224                 else:
 225                     span.set_tag("mode", "has_project_membership")
 226                     func = request.access.has_project_membership
 227                 projects = [p for p in qs if func(p)]
 228 
 229         project_ids = set(p.id for p in projects)
 230 
 231         if requested_projects and project_ids != requested_projects:
 232             raise PermissionDenied
 233 
 234         return projects

 240     def get_filter_params(self, request, organization, date_filter_optional=False):
 241         """
 242         Extracts common filter parameters from the request and returns them
 243         in a standard format.
 244         :param request:
 245         :param organization: Organization to get params for
 246         :param date_filter_optional: Defines what happens if no date filter
 247         parameters are passed. If False, no date filtering occurs. If True, we
 248         provide default values.
 249         :return: A dict with keys:
 250          - start: start date of the filter
 251          - end: end date of the filter
 252          - project_id: A list of project ids to filter on
 253          - environment(optional): If environments were passed in, a list of
 254          environment names
 255         """
 256         # get the top level params -- projects, time range, and environment
 257         # from the request
 258         try:
 259             start, end = get_date_range_from_params(request.GET, optional=date_filter_optional)
 260         except InvalidParams as e:
 261             raise ParseError(detail=u"Invalid date range: {}".format(e))
 262 
 263         with sentry_sdk.start_span(op="PERF: org.get_filter_params - projects"):
 264             try:
 265                 projects = self.get_projects(request, organization)
 266             except ValueError:
 267                 raise ParseError(detail="Invalid project ids")
 268 
 269         if not projects:
 270             raise NoProjects
 271 
 272         environments = self.get_environments(request, organization)
 273         params = {
 274             "start": start,
 275             "end": end,
 276             "project_id": [p.id for p in projects],
 277             "organization_id": organization.id,
 278         }
 279         if environments:
 280             params["environment"] = [env.name for env in environments]
 281             params["environment_objects"] = environments
 282 
 283         return params

 285     def convert_args(self, request, organization_slug, *args, **kwargs):
 286         with sentry_sdk.start_span(op="PERF: org.convert_args - organization (cache)"):
 287             try:
 288                 organization = Organization.objects.get_from_cache(slug=organization_slug)
 289             except Organization.DoesNotExist:
 290                 raise ResourceDoesNotExist
 291 
 292         with sentry_sdk.start_span(
 293             op="check_object_permissions_on_organization", description=organization_slug
 294         ):
 295             self.check_object_permissions(request, organization)
 296 
 297         bind_organization_context(organization)
 298 
 299         request._request.organization = organization
 300 
 301         # Track the 'active' organization when the request came from
 302         # a cookie based agent (react app)
 303         # Never track any org (regardless of whether the user does or doesn't have
 304         # membership in that org) when the user is in active superuser mode
 305         if request.auth is None and request.user and not is_active_superuser(request):
 306             request.session["activeorg"] = organization.slug
 307 
 308         kwargs["organization"] = organization
 309         return (args, kwargs)

 315     def get_projects(self, request, organization):
 316         """
 317         Get all projects the current user or API token has access to. More
 318         detail in the parent class's method of the same name.
 319         """
 320         has_valid_api_key = False
 321         if isinstance(request.auth, ApiKey):
 322             if request.auth.organization_id != organization.id:
 323                 return []
 324             has_valid_api_key = request.auth.has_scope(
 325                 "project:releases"
 326             ) or request.auth.has_scope("project:write")
 327 
 328         if not (
 329             has_valid_api_key
 330             or (getattr(request, "user", None) and request.user.is_authenticated())
 331         ):
 332             return []
 333 
 334         return super(OrganizationReleasesBaseEndpoint, self).get_projects(
 335             request, organization, force_global_perms=has_valid_api_key, include_all_accessible=True
 336         )

 338     def has_release_permission(self, request, organization, release):
 339         """
 340         Does the given request have permission to access this release, based
 341         on the projects to which the release is attached?
 342 
 343         If the given request has an actor (user or ApiKey), cache the results
 344         for a minute on the unique combination of actor,org,release, and project
 345         ids.
 346         """
 347         actor_id = None
 348         has_perms = None
 349         if getattr(request, "user", None) and request.user.id:
 350             actor_id = "user:%s" % request.user.id
 351         if getattr(request, "auth", None) and request.auth.id:
 352             actor_id = "apikey:%s" % request.auth.id
 353         if actor_id is not None:
 354             project_ids = sorted(self.get_requested_project_ids(request))
 355             key = "release_perms:1:%s" % hash_values(
 356                 [actor_id, organization.id, release.id] + project_ids
 357             )
 358             has_perms = cache.get(key)
 359         if has_perms is None:
 360             has_perms = ReleaseProject.objects.filter(
 361                 release=release, project__in=self.get_projects(request, organization)
 362             ).exists()
 363             if actor_id is not None:
 364                 cache.set(key, has_perms, 60)
 365 
 366         return has_perms
