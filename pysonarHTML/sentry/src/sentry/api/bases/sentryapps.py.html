  17 def catch_raised_errors(func):
  18     @wraps(func)
  19     def wrapped(self, *args, **kwargs):
  20         try:
  21             return func(self, *args, **kwargs)
  22         except APIError as e:
  23             return Response({"detail": e.msg}, status=400)
  24 
  25     return wrapped

  28 def ensure_scoped_permission(request, allowed_scopes):
  29     """
  30     Verifies the User making the request has at least one required scope for
  31     the endpoint being requested.
  32 
  33     If no scopes were specified in a ``scope_map``, it means the endpoint should
  34     not be accessible. That is, this function expects every accessible endpoint
  35     to have a list of scopes.
  36 
  37     That list of scopes may be empty, implying that the User does not need any
  38     specific scope and the endpoint is public.
  39     """
  40     # If no scopes were found at all, the endpoint should not be accessible.
  41     if allowed_scopes is None:
  42         return False
  43 
  44     # If there are no scopes listed, it implies a public endpoint.
  45     if len(allowed_scopes) == 0:
  46         return True
  47 
  48     return any(request.access.has_scope(s) for s in set(allowed_scopes))

  52     @wraps(func)
  53     def wrapped(self, *args, **kwargs):

  57     return wrapped

  60 class SentryAppsPermission(SentryPermission):
  61     scope_map = {
  62         # GET is ideally a public endpoint but for now we are allowing for
  63         # anyone who has member permissions or above.
  64         "GET": (
  65             "event:read",
  66             "event:write",
  67             "event:admin",
  68             "project:releases",
  69             "project:read",
  70             "org:read",
  71             "member:read",
  72             "team:read",
  73         ),
  74         "POST": ("org:read", "org:integrations", "org:write", "org:admin"),
  75     }
  76 
  77     def has_object_permission(self, request, view, organization):
  78         if not hasattr(request, "user") or not request.user:
  79             return False
  80 
  81         self.determine_access(request, organization)
  82 
  83         if is_active_superuser(request):
  84             return True
  85 
  86         # User must be a part of the Org they're trying to create the app in.
  87         if organization not in request.user.get_orgs():
  88             raise Http404
  89 
  90         return ensure_scoped_permission(request, self.scope_map.get(request.method))

  99 class SentryAppsBaseEndpoint(IntegrationPlatformEndpoint):
 100     permission_classes = (SentryAppsPermission,)
 101 
 102     def convert_args(self, request, *args, **kwargs):
 103         # This baseclass is the the SentryApp collection endpoints:
 104         #
 105         #       [GET, POST] /sentry-apps
 106         #
 107         # The GET endpoint is public and doesn't require (or handle) any query
 108         # params or request body.
 109         #
 110         # The POST endpoint is for creating a Sentry App. Part of that creation
 111         # is associating it with the Organization that it's created within.
 112         #
 113         # So in the case of POST requests, we want to pull the Organization out
 114         # of the request body so that we can ensure the User making the request
 115         # has access to it.
 116         #
 117         # Since ``convert_args`` is conventionally where you materialize model
 118         # objects from URI params, we're applying the same logic for a param in
 119         # the request body.
 120         #
 121         if not request.json_body or "organization" not in request.json_body:
 122             return (args, kwargs)
 123 
 124         organization = request.user.get_orgs().get(slug=request.json_body["organization"])
 125 
 126         self.check_object_permissions(request, organization)
 127 
 128         kwargs["organization"] = organization
 129         return (args, kwargs)

 132 class SentryAppPermission(SentryPermission):
 133     unpublished_scope_map = {
 134         "GET": ("org:read", "org:integrations", "org:write", "org:admin"),
 135         "PUT": ("org:read", "org:integrations", "org:write", "org:admin"),
 136         "POST": ("org:write", "org:admin"),  # used for publishing an app
 137         "DELETE": ("org:write", "org:admin"),
 138     }
 139 
 140     published_scope_map = {
 141         # GET is ideally a public endpoint but for now we are allowing for
 142         # anyone who has member permissions or above.
 143         "GET": (
 144             "event:read",
 145             "event:write",
 146             "event:admin",
 147             "project:releases",
 148             "project:read",
 149             "org:read",
 150             "member:read",
 151             "team:read",
 152         ),
 153         "PUT": ("org:write", "org:admin"),
 154         "POST": ("org:write", "org:admin"),
 155         "DELETE": ("org:admin"),
 156     }
 157 
 158     @property
 159     def scope_map(self):
 160         return self.published_scope_map
 161 
 162     def has_object_permission(self, request, view, sentry_app):
 163         if not hasattr(request, "user") or not request.user:
 164             return False
 165 
 166         self.determine_access(request, sentry_app.owner)
 167 
 168         if is_active_superuser(request):
 169             return True
 170 
 171         # if app is unpublished, user must be in the Org who owns the app.
 172         if not sentry_app.is_published:
 173             if sentry_app.owner not in request.user.get_orgs():
 174                 raise Http404
 175 
 176         # TODO(meredith): make a better way to allow for public
 177         # endpoints. we can't use ensure_scoped_permission now
 178         # that the public endpoint isn't denoted by '()'
 179         if sentry_app.is_published and request.method == "GET":
 180             return True
 181 
 182         return ensure_scoped_permission(
 183             request, self._scopes_for_sentry_app(sentry_app).get(request.method)
 184         )
 185 
 186     def _scopes_for_sentry_app(self, sentry_app):
 187         if sentry_app.is_published:
 188             return self.published_scope_map
 189         else:
 190             return self.unpublished_scope_map

 251 class SentryAppInstallationPermission(SentryPermission):
 252     scope_map = {
 253         "GET": ("org:read", "org:integrations", "org:write", "org:admin"),
 254         "DELETE": ("org:integrations", "org:write", "org:admin"),
 255         # NOTE(mn): The only POST endpoint right now is to create External
 256         # Issues, which uses this baseclass since it's nested under an
 257         # installation.
 258         #
 259         # The scopes below really only make sense for that endpoint. Any other
 260         # nested endpoints will probably need different scopes - figure out how
 261         # to deal with that when it happens.
 262         "POST": ("org:integrations", "event:write", "event:admin"),
 263     }
 264 
 265     def has_permission(self, request, *args, **kwargs):
 266         # To let the app mark the installation as installed, we don't care about permissions
 267         if (
 268             hasattr(request, "user")
 269             and hasattr(request.user, "is_sentry_app")
 270             and request.user.is_sentry_app
 271             and request.method == "PUT"
 272         ):
 273             return True
 274         return super(SentryAppInstallationPermission, self).has_permission(request, *args, **kwargs)
 275 
 276     def has_object_permission(self, request, view, installation):
 277         if not hasattr(request, "user") or not request.user:
 278             return False
 279 
 280         self.determine_access(request, installation.organization)
 281 
 282         if is_active_superuser(request):
 283             return True
 284 
 285         # if user is an app, make sure it's for that same app
 286         if request.user.is_sentry_app:
 287             return request.user == installation.sentry_app.proxy_user
 288 
 289         if installation.organization not in request.user.get_orgs():
 290             raise Http404
 291 
 292         return ensure_scoped_permission(request, self.scope_map.get(request.method))
