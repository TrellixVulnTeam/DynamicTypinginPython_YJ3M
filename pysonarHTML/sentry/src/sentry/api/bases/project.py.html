  25 class ProjectPermission(OrganizationPermission):
  26     scope_map = {
  27         "GET": ["project:read", "project:write", "project:admin"],
  28         "POST": ["project:write", "project:admin"],
  29         "PUT": ["project:write", "project:admin"],
  30         "DELETE": ["project:admin"],

  33     def has_object_permission(self, request, view, project):
  34         result = super(ProjectPermission, self).has_object_permission(
  35             request, view, project.organization
  36         )
  37 
  38         if not result:
  39             return result
  40         if project.teams.exists():
  41             return any(
  42                 has_team_permission(request, team, self.scope_map) for team in project.teams.all()
  43             )
  44         elif is_system_auth(request.auth):
  45             return True
  46         elif request.user.is_authenticated():
  47             # this is only for team-less projects
  48             if is_active_superuser(request):
  49                 return True
  50             elif request.user.is_sentry_app:
  51                 return SentryApp.check_project_permission_for_sentry_app_user(request.user, project)
  52             try:
  53                 role = (
  54                     OrganizationMember.objects.filter(
  55                         organization=project.organization, user=request.user
  56                     )
  57                     .values_list("role", flat=True)
  58                     .get()
  59                 )
  60             except OrganizationMember.DoesNotExist:
  61                 # this should probably never happen?
  62                 return False
  63 
  64             return roles.get(role).is_global
  65         elif hasattr(request.auth, "project_id") and project.id == request.auth.project_id:
  66             return True
  67 
  68         return False

 121     def convert_args(self, request, organization_slug, project_slug, *args, **kwargs):
 122         try:
 123             project = (
 124                 Project.objects.filter(organization__slug=organization_slug, slug=project_slug)
 125                 .select_related("organization")
 126                 .prefetch_related("teams")
 127                 .get()
 128             )
 129         except Project.DoesNotExist:
 130             try:
 131                 # Project may have been renamed
 132                 redirect = ProjectRedirect.objects.select_related("project")
 133                 redirect = redirect.get(
 134                     organization__slug=organization_slug, redirect_slug=project_slug
 135                 )
 136                 # Without object permissions don't reveal the rename
 137                 self.check_object_permissions(request, redirect.project)
 138 
 139                 # get full path so that we keep query strings
 140                 requested_url = request.get_full_path()
 141                 new_url = requested_url.replace(
 142                     "projects/%s/%s/" % (organization_slug, project_slug),
 143                     "projects/%s/%s/" % (organization_slug, redirect.project.slug),
 144                 )
 145 
 146                 # Resource was moved/renamed if the requested url is different than the new url
 147                 if requested_url != new_url:
 148                     raise ProjectMoved(new_url, redirect.project.slug)
 149 
 150                 # otherwise project doesn't exist
 151                 raise ResourceDoesNotExist
 152             except ProjectRedirect.DoesNotExist:
 153                 raise ResourceDoesNotExist
 154 
 155         if project.status != ProjectStatus.VISIBLE:
 156             raise ResourceDoesNotExist
 157 
 158         self.check_object_permissions(request, project)
 159 
 160         with configure_scope() as scope:
 161             scope.set_tag("project", project.id)
 162 
 163         bind_organization_context(project.organization)
 164 
 165         request._request.organization = project.organization
 166 
 167         kwargs["project"] = project
 168         return (args, kwargs)

 170     def get_filter_params(self, request, project, date_filter_optional=False):
 171         """Similar to the version on the organization just for a single project."""
 172         # get the top level params -- projects, time range, and environment
 173         # from the request
 174         try:
 175             start, end = get_date_range_from_params(request.GET, optional=date_filter_optional)
 176         except InvalidParams as e:
 177             raise ProjectEventsError(six.text_type(e))
 178 
 179         environments = [env.name for env in get_environments(request, project.organization)]
 180         params = {"start": start, "end": end, "project_id": [project.id]}
 181         if environments:
 182             params["environment"] = environments
 183 
 184         return params

 186     def handle_exception(self, request, exc):
 187         if isinstance(exc, ProjectMoved):
 188             response = Response(
 189                 {"slug": exc.detail["detail"]["extra"]["slug"], "detail": exc.detail["detail"]},
 190                 status=exc.status_code,
 191             )
 192             response["Location"] = exc.detail["detail"]["extra"]["url"]
 193             return response
 194         return super(ProjectEndpoint, self).handle_exception(request, exc)
