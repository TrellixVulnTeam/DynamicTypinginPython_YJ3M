  58     def get_snuba_query_args_legacy(self, request, organization):
  59         params = self.get_filter_params(request, organization)
  60 
  61         group_ids = request.GET.getlist("group")
  62         if group_ids:
  63             # TODO(mark) This parameter should be removed in the long term.
  64             # Instead of using this parameter clients should use `issue.id`
  65             # in their query string.
  66             try:
  67                 group_ids = set(map(int, [_f for _f in group_ids if _f]))
  68             except ValueError:
  69                 raise ParseError(detail="Invalid group parameter. Values must be numbers")
  70 
  71             projects = Project.objects.filter(
  72                 organization=organization, group__id__in=group_ids
  73             ).distinct()
  74             if any(p for p in projects if not request.access.has_project_access(p)):
  75                 raise PermissionDenied
  76             params["group_ids"] = list(group_ids)
  77             params["project_id"] = list(set([p.id for p in projects] + params["project_id"]))
  78 
  79         query = request.GET.get("query")
  80         try:
  81             _filter = get_filter(query, params)
  82         except InvalidSearchQuery as e:
  83             raise ParseError(detail=six.text_type(e))
  84 
  85         snuba_args = {
  86             "start": _filter.start,
  87             "end": _filter.end,
  88             "conditions": _filter.conditions,
  89             "filter_keys": _filter.filter_keys,
  90         }
  91 
  92         return snuba_args

  94     def quantize_date_params(self, request, params):
  95         # We only need to perform this rounding on relative date periods
  96         if "statsPeriod" not in request.GET:
  97             return params
  98         results = params.copy()
  99         duration = (params["end"] - params["start"]).total_seconds()
 100         # Only perform rounding on durations longer than an hour
 101         if duration > 3600:
 102             # Round to 15 minutes if over 30 days, otherwise round to the minute
 103             round_to = 15 * 60 if duration >= 30 * 24 * 3600 else 60
 104             for key in ["start", "end"]:
 105                 results[key] = snuba.quantize_time(
 106                     params[key], params.get("organization_id", 0), duration=round_to
 107                 )
 108         return results

 111 class OrganizationEventsV2EndpointBase(OrganizationEventsEndpointBase):
 112     def handle_results_with_meta(self, request, organization, project_ids, results):
 113         with sentry_sdk.start_span(op="discover.endpoint", description="base.handle_results"):
 114             data = self.handle_data(request, organization, project_ids, results.get("data"))
 115             if not data:
 116                 return {"data": [], "meta": {}}
 117 
 118             meta = {
 119                 value["name"]: get_json_meta_type(value["name"], value["type"])
 120                 for value in results["meta"]
 121             }
 122             # Ensure all columns in the result have types.
 123             for key in data[0]:
 124                 if key not in meta:
 125                     meta[key] = "string"
 126             return {"meta": meta, "data": data}
 127 
 128     def handle_data(self, request, organization, project_ids, results):
 129         if not results:
 130             return results
 131 
 132         first_row = results[0]
 133 
 134         # TODO(mark) move all of this result formatting into discover.query()
 135         # once those APIs are used across the application.
 136         if "transaction.status" in first_row:
 137             for row in results:
 138                 row["transaction.status"] = SPAN_STATUS_CODE_TO_NAME.get(row["transaction.status"])
 139 
 140         fields = request.GET.getlist("field")
 141         if "issue" in fields:  # Look up the short ID and return that in the results
 142             issue_ids = set(row.get("issue.id") for row in results)
 143             issues = Group.issues_mapping(issue_ids, project_ids, organization)
 144             for result in results:
 145                 if "issue.id" in result:
 146                     result["issue"] = issues.get(result["issue.id"], "unknown")
 147 
 148         if not ("project.id" in first_row or "projectid" in first_row):
 149             return results
 150 
 151         for result in results:
 152             for key in ("projectid", "project.id"):
 153                 if key in result:
 154                     if key not in fields:
 155                         del result[key]
 156 
 157         return results

 159     def get_event_stats_data(self, request, organization, get_event_stats, top_events=False):
 160         try:
 161             with sentry_sdk.start_span(
 162                 op="discover.endpoint", description="base.stats_query_creation"
 163             ):
 164                 columns = request.GET.getlist("yAxis", ["count()"])
 165                 query = request.GET.get("query")
 166                 try:
 167                     params = self.get_filter_params(request, organization)
 168                 except NoProjects:
 169                     return {"data": []}
 170                 params = self.quantize_date_params(request, params)
 171                 rollup = get_rollup_from_request(
 172                     request,
 173                     params,
 174                     "1h",
 175                     InvalidSearchQuery(
 176                         "Your interval and date range would create too many results. "
 177                         "Use a larger interval, or a smaller date range."
 178                     ),
 179                 )
 180                 # Backwards compatibility for incidents which uses the old
 181                 # column aliases as it straddles both versions of events/discover.
 182                 # We will need these aliases until discover2 flags are enabled for all
 183                 # users.
 184                 column_map = {
 185                     "user_count": "count_unique(user)",
 186                     "event_count": "count()",
 187                     "epm()": "epm(%d)" % rollup,
 188                     "eps()": "eps(%d)" % rollup,
 189                 }
 190                 query_columns = [column_map.get(column, column) for column in columns]
 191                 reference_event = self.reference_event(
 192                     request, organization, params.get("start"), params.get("end")
 193                 )
 194 
 195             with sentry_sdk.start_span(op="discover.endpoint", description="base.stats_query"):
 196                 result = get_event_stats(query_columns, query, params, rollup, reference_event)
 197         except (discover.InvalidSearchQuery, snuba.QueryOutsideRetentionError) as error:
 198             raise ParseError(detail=six.text_type(error))
 199         serializer = SnubaTSResultSerializer(organization, None, request.user)
 200 
 201         with sentry_sdk.start_span(op="discover.endpoint", description="base.stats_serialization"):
 202             if top_events:
 203                 results = {}
 204                 for key, event_result in six.iteritems(result):
 205                     if len(query_columns) > 1:
 206                         results[key] = self.serialize_multiple_axis(
 207                             serializer, event_result, columns, query_columns
 208                         )
 209                     else:
 210                         # Need to get function alias if count is a field, but not the axis
 211                         results[key] = serializer.serialize(
 212                             event_result, get_function_alias(query_columns[0])
 213                         )
 214                 return results
 215             elif len(query_columns) > 1:
 216                 return self.serialize_multiple_axis(serializer, result, columns, query_columns)
 217             else:
 218                 return serializer.serialize(result)
 219 
 220     def serialize_multiple_axis(self, serializer, event_result, columns, query_columns):
 221         # Return with requested yAxis as the key
 222         result = {
 223             columns[index]: serializer.serialize(
 224                 event_result, get_function_alias(query_column), order=index
 225             )
 226             for index, query_column in enumerate(query_columns)
 227         }
 228         # Set order if multi-axis + top events
 229         if "order" in event_result.data:
 230             result["order"] = event_result.data["order"]
 231         return result
