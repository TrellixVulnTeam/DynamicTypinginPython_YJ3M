  66     def build_cursor_link(self, request, name, cursor):
  67         querystring = u"&".join(
  68             u"{0}={1}".format(urlquote(k), urlquote(v))
  69             for k, v in six.iteritems(request.GET)
  70             if k != "cursor"
  71         )
  72         base_url = absolute_uri(urlquote(request.path))
  73         if querystring:
  74             base_url = u"{0}?{1}".format(base_url, querystring)
  75         else:
  76             base_url = base_url + "?"
  77 
  78         return LINK_HEADER.format(
  79             uri=base_url,
  80             cursor=six.text_type(cursor),
  81             name=name,
  82             has_results="true" if bool(cursor) else "false",
  83         )

  88     def handle_exception(self, request, exc):
  89         try:
  90             response = super(Endpoint, self).handle_exception(exc)
  91         except Exception:
  92             import sys
  93             import traceback
  94 
  95             sys.stderr.write(traceback.format_exc())
  96             event_id = capture_exception()
  97             context = {"detail": "Internal Error", "errorId": event_id}
  98             response = Response(context, status=500)
  99             response.exception = True
 100         return response

 105     def load_json_body(self, request):
 106         """
 107         Attempts to load the request body when it's JSON.
 108 
 109         The end result is ``request.json_body`` having a value. When it can't
 110         load the body as JSON, for any reason, ``request.json_body`` is None.
 111 
 112         The request flow is unaffected and no exceptions are ever raised.
 113         """
 114 
 115         request.json_body = None
 116 
 117         if not request.META.get("CONTENT_TYPE", "").startswith("application/json"):
 118             return
 119 
 120         if not len(request.body):
 121             return
 122 
 123         try:
 124             request.json_body = json.loads(request.body)
 125         except JSONDecodeError:
 126             return

 128     def initialize_request(self, request, *args, **kwargs):
 129         # XXX: Since DRF 3.x, when the request is passed into
 130         # `initialize_request` it's set as an internal variable on the returned
 131         # request. Then when we call `rv.auth` it attempts to authenticate,
 132         # fails and sets `user` and `auth` to None on the internal request. We
 133         # keep track of these here and reassign them as needed.
 134         orig_auth = getattr(request, "auth", None)
 135         orig_user = getattr(request, "user", None)
 136         rv = super(Endpoint, self).initialize_request(request, *args, **kwargs)
 137         # If our request is being made via our internal API client, we need to
 138         # stitch back on auth and user information
 139         if getattr(request, "__from_api_client__", False):
 140             if rv.auth is None:
 141                 rv.auth = orig_auth
 142             if rv.user is None:
 143                 rv.user = orig_user
 144         return rv

 146     @csrf_exempt
 147     @allow_cors_options
 148     def dispatch(self, request, *args, **kwargs):
 149         """
 150         Identical to rest framework's dispatch except we add the ability
 151         to convert arguments (for common URL params).
 152         """
 153         with sentry_sdk.start_span(op="base.dispatch.setup", description=type(self).__name__):
 154             self.args = args
 155             self.kwargs = kwargs
 156             request = self.initialize_request(request, *args, **kwargs)
 157             self.load_json_body(request)
 158             self.request = request
 159             self.headers = self.default_response_headers  # deprecate?
 160 
 161         # Tags that will ultimately flow into the metrics backend at the end of
 162         # the request (happens via middleware/stats.py).
 163         request._metric_tags = {}
 164 
 165         if settings.SENTRY_API_RESPONSE_DELAY:
 166             with sentry_sdk.start_span(
 167                 op="base.dispatch.sleep", description=type(self).__name__,
 168             ) as span:
 169                 span.set_data("SENTRY_API_RESPONSE_DELAY", settings.SENTRY_API_RESPONSE_DELAY)
 170                 time.sleep(settings.SENTRY_API_RESPONSE_DELAY / 1000.0)
 171 
 172         origin = request.META.get("HTTP_ORIGIN", "null")
 173         # A "null" value should be treated as no Origin for us.
 174         # See RFC6454 for more information on this behavior.
 175         if origin == "null":
 176             origin = None
 177 
 178         try:
 179             with sentry_sdk.start_span(op="base.dispatch.request", description=type(self).__name__):
 180                 if origin and request.auth:
 181                     allowed_origins = request.auth.get_allowed_origins()
 182                     if not is_valid_origin(origin, allowed=allowed_origins):
 183                         response = Response("Invalid origin: %s" % (origin,), status=400)
 184                         self.response = self.finalize_response(request, response, *args, **kwargs)
 185                         return self.response
 186 
 187                 self.initial(request, *args, **kwargs)
 188 
 189                 # Get the appropriate handler method
 190                 if request.method.lower() in self.http_method_names:
 191                     handler = getattr(self, request.method.lower(), self.http_method_not_allowed)
 192 
 193                     (args, kwargs) = self.convert_args(request, *args, **kwargs)
 194                     self.args = args
 195                     self.kwargs = kwargs
 196                 else:
 197                     handler = self.http_method_not_allowed
 198 
 199                 if getattr(request, "access", None) is None:
 200                     # setup default access
 201                     request.access = access.from_request(request)
 202 
 203             with sentry_sdk.start_span(
 204                 op="base.dispatch.execute",
 205                 description="{}.{}".format(type(self).__name__, handler.__name__),
 206             ):
 207                 response = handler(request, *args, **kwargs)
 208 
 209         except Exception as exc:
 210             response = self.handle_exception(request, exc)
 211 
 212         if origin:
 213             self.add_cors_headers(request, response)
 214 
 215         self.response = self.finalize_response(request, response, *args, **kwargs)
 216 
 217         return self.response

 219     def add_cors_headers(self, request, response):
 220         response["Access-Control-Allow-Origin"] = request.META["HTTP_ORIGIN"]
 221         response["Access-Control-Allow-Methods"] = ", ".join(self.http_method_names)

 223     def add_cursor_headers(self, request, response, cursor_result):
 224         if cursor_result.hits is not None:
 225             response["X-Hits"] = cursor_result.hits
 226         if cursor_result.max_hits is not None:
 227             response["X-Max-Hits"] = cursor_result.max_hits
 228         response["Link"] = ", ".join(
 229             [
 230                 self.build_cursor_link(request, "previous", cursor_result.prev),
 231                 self.build_cursor_link(request, "next", cursor_result.next),
 232             ]
 233         )

 238     def paginate(
 239         self,
 240         request,
 241         on_results=None,
 242         paginator=None,
 243         paginator_cls=Paginator,
 244         default_per_page=100,
 245         max_per_page=100,
 246         **paginator_kwargs
 247     ):
 248         assert (paginator and not paginator_kwargs) or (paginator_cls and paginator_kwargs)
 249 
 250         try:
 251             per_page = int(request.GET.get("per_page", default_per_page))
 252         except ValueError:
 253             raise ParseError(detail="Invalid per_page parameter.")
 254 
 255         input_cursor = None
 256         if request.GET.get("cursor"):
 257             try:
 258                 input_cursor = Cursor.from_string(request.GET.get("cursor"))
 259             except ValueError:
 260                 raise ParseError(detail="Invalid cursor parameter.")
 261 
 262         max_per_page = max(max_per_page, default_per_page)
 263         if per_page > max_per_page:
 264             raise ParseError(
 265                 detail="Invalid per_page value. Cannot exceed {}.".format(max_per_page)
 266             )
 267 
 268         if not paginator:
 269             paginator = paginator_cls(**paginator_kwargs)
 270 
 271         try:
 272             with sentry_sdk.start_span(
 273                 op="base.paginate.get_result", description=type(self).__name__,
 274             ) as span:
 275                 span.set_data("Limit", per_page)
 276                 cursor_result = paginator.get_result(limit=per_page, cursor=input_cursor)
 277         except BadPaginationError as e:
 278             raise ParseError(detail=six.text_type(e))
 279 
 280         # map results based on callback
 281         if on_results:
 282             with sentry_sdk.start_span(
 283                 op="base.paginate.on_results", description=type(self).__name__,
 284             ):
 285                 results = on_results(cursor_result.results)
 286         else:
 287             results = cursor_result.results
 288 
 289         response = Response(results)
 290         self.add_cursor_headers(request, response, cursor_result)
 291         return response

 294 class EnvironmentMixin(object):
 295     def _get_environment_func(self, request, organization_id):
 296         """\
 297         Creates a function that when called returns the ``Environment``
 298         associated with a request object, or ``None`` if no environment was
 299         provided. If the environment doesn't exist, an ``Environment.DoesNotExist``
 300         exception will be raised.
 301 
 302         This returns as a callable since some objects outside of the API
 303         endpoint need to handle the "environment was provided but does not
 304         exist" state in addition to the two non-exceptional states (the
 305         environment was provided and exists, or the environment was not
 306         provided.)
 307         """
 308         return functools.partial(self._get_environment_from_request, request, organization_id)
 309 
 310     def _get_environment_id_from_request(self, request, organization_id):
 311         environment = self._get_environment_from_request(request, organization_id)
 312         return environment and environment.id
 313 
 314     def _get_environment_from_request(self, request, organization_id):
 315         if not hasattr(request, "_cached_environment"):
 316             environment_param = request.GET.get("environment")
 317             if environment_param is None:
 318                 environment = None
 319             else:
 320                 environment = Environment.get_for_organization_id(
 321                     name=environment_param, organization_id=organization_id
 322                 )
 323 
 324             request._cached_environment = environment
 325 
 326         return request._cached_environment

 329 class StatsMixin(object):
 330     def _parse_args(self, request, environment_id=None):
 331         resolution = request.GET.get("resolution")
 332         if resolution:
 333             resolution = self._parse_resolution(resolution)
 334             assert resolution in tsdb.get_rollups()
 335 
 336         end = request.GET.get("until")
 337         if end:
 338             end = to_datetime(float(end))
 339         else:
 340             end = datetime.utcnow().replace(tzinfo=utc)
 341 
 342         start = request.GET.get("since")
 343         if start:
 344             start = to_datetime(float(start))
 345             assert start <= end, "start must be before or equal to end"
 346         else:
 347             start = end - timedelta(days=1, seconds=-1)
 348 
 349         if not resolution:
 350             resolution = tsdb.get_optimal_rollup(start, end)
 351 
 352         return {
 353             "start": start,
 354             "end": end,
 355             "rollup": resolution,
 356             "environment_ids": environment_id and [environment_id],
 357         }
 358 
 359     def _parse_resolution(self, value):
 360         if value.endswith("h"):
 361             return int(value[:-1]) * ONE_HOUR
 362         elif value.endswith("d"):
 363             return int(value[:-1]) * ONE_DAY
 364         elif value.endswith("m"):
 365             return int(value[:-1]) * ONE_MINUTE
 366         elif value.endswith("s"):
 367             return int(value[:-1])
 368         else:
 369             raise ValueError(value)
