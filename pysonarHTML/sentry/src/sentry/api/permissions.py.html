  28 class ScopedPermission(permissions.BasePermission):
  29     """
  30     Permissions work depending on the type of authentication:
  31 
  32     - A user inherits permissions based on their membership role. These are
  33       still dictated as common scopes, but they can't be checked until the
  34       has_object_permission hook is called.
  35     - ProjectKeys (legacy) are granted only project based scopes. This
  36     - APIKeys specify their scope, and work as expected.
  37     """
  38 
  39     scope_map = {"HEAD": (), "GET": (), "POST": (), "PUT": (), "PATCH": (), "DELETE": ()}
  40 
  41     def has_permission(self, request, view):
  42         # session-based auth has all scopes for a logged in user
  43         if not getattr(request, "auth", None):
  44             return request.user.is_authenticated()
  45 
  46         allowed_scopes = set(self.scope_map.get(request.method, []))
  47         current_scopes = request.auth.get_scopes()
  48         return any(s in allowed_scopes for s in current_scopes)
  49 
  50     def has_object_permission(self, request, view, obj):
  51         return False

  54 class SuperuserPermission(permissions.BasePermission):
  55     def has_permission(self, request, view):
  56         if is_active_superuser(request):
  57             return True
  58         if request.user.is_authenticated() and request.user.is_superuser:
  59             raise SuperuserRequired
  60         return False

  63 class SentryPermission(ScopedPermission):
  64     def is_not_2fa_compliant(self, request, organization):
  65         return False

  67     def needs_sso(self, request, organization):
  68         return False

  70     def determine_access(self, request, organization):
  71         from sentry.api.base import logger
  72 
  73         if request.user and request.user.is_authenticated() and request.auth:
  74             request.access = access.from_request(
  75                 request, organization, scopes=request.auth.get_scopes()
  76             )
  77 
  78         elif request.auth:
  79             request.access = access.from_auth(request.auth, organization)
  80 
  81         else:
  82             request.access = access.from_request(request, organization)
  83 
  84             if auth.is_user_signed_request(request):
  85                 # if the user comes from a signed request
  86                 # we let them pass if sso is enabled
  87                 logger.info(
  88                     "access.signed-sso-passthrough",
  89                     extra={"organization_id": organization.id, "user_id": request.user.id},
  90                 )
  91             elif request.user.is_authenticated():
  92                 # session auth needs to confirm various permissions
  93                 if self.needs_sso(request, organization):
  94 
  95                     logger.info(
  96                         "access.must-sso",
  97                         extra={"organization_id": organization.id, "user_id": request.user.id},
  98                     )
  99 
 100                     raise SsoRequired(organization)
 101 
 102                 if self.is_not_2fa_compliant(request, organization):
 103                     logger.info(
 104                         "access.not-2fa-compliant",
 105                         extra={"organization_id": organization.id, "user_id": request.user.id},
 106                     )
 107                     raise TwoFactorRequired()
