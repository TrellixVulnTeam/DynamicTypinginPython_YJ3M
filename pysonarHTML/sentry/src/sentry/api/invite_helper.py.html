  12 INVITE_COOKIE = "pending-invite"
  13 COOKIE_MAX_AGE = 60 * 60 * 24 * 7  # 7 days

  16 def add_invite_cookie(request, response, member_id, token):
  17     url = reverse("sentry-accept-invite", args=[member_id, token])
  18     response.set_cookie(
  19         INVITE_COOKIE,
  20         urlencode({"memberId": member_id, "token": token, "url": url}),
  21         max_age=COOKIE_MAX_AGE,
  22     )
  23 
  24 
  25 def remove_invite_cookie(request, response):
  26     if INVITE_COOKIE in request.COOKIES:
  27         response.delete_cookie(INVITE_COOKIE)
  28 
  29 
  30 def get_invite_cookie(request):
  31     if INVITE_COOKIE not in request.COOKIES:
  32         return None
  33 
  34     # memberId should be coerced back to an integer
  35     invite_data = dict(parse_qsl(request.COOKIES.get(INVITE_COOKIE)))
  36     invite_data["memberId"] = int(invite_data["memberId"])
  37 
  38     return invite_data

  41 class ApiInviteHelper(object):
  42     @classmethod
  43     def from_cookie_or_email(cls, request, organization, email, instance=None, logger=None):
  44         """
  45         Initializes the ApiInviteHelper by locating the pending organization
  46         member via the currently set pending invite cookie, or via the passed
  47         email if no cookie is currently set.
  48         """
  49         pending_invite = get_invite_cookie(request)
  50 
  51         try:
  52             if pending_invite is not None:
  53                 om = OrganizationMember.objects.get(
  54                     id=pending_invite["memberId"], token=pending_invite["token"]
  55                 )
  56             else:
  57                 om = OrganizationMember.objects.get(
  58                     email=email, organization=organization, user=None
  59                 )
  60         except OrganizationMember.DoesNotExist:
  61             # Unable to locate the pending organization member. Cannot setup
  62             # the invite helper.
  63             return None
  64 
  65         return cls(
  66             request=request, member_id=om.id, token=om.token, instance=instance, logger=logger
  67         )
  68 
  69     @classmethod
  70     def from_cookie(cls, request, instance=None, logger=None):
  71         org_invite = get_invite_cookie(request)
  72 
  73         if not org_invite:
  74             return None
  75 
  76         try:
  77             return ApiInviteHelper(
  78                 request=request,
  79                 member_id=org_invite["memberId"],
  80                 token=org_invite["token"],
  81                 instance=instance,
  82                 logger=logger,
  83             )
  84         except OrganizationMember.DoesNotExist:
  85             if logger:
  86                 logger.error("Invalid pending invite cookie", exc_info=True)
  87             return None
  88 
  89     def __init__(self, request, member_id, token, instance=None, logger=None):
  90         self.request = request
  91         self.member_id = member_id
  92         self.token = token
  93         self.instance = instance
  94         self.logger = logger
  95         self.om = self.organization_member
  96 
  97     def handle_success(self):
  98         member_joined.send_robust(
  99             member=self.om,
 100             organization=self.om.organization,
 101             sender=self.instance if self.instance else self,
 102         )
 103 
 104     def handle_member_already_exists(self):
 105         if self.logger:
 106             self.logger.info(
 107                 "Pending org invite not accepted - User already org member",
 108                 extra={"organization_id": self.om.organization.id, "user_id": self.request.user.id},
 109             )
 110 
 111     def handle_invite_not_approved(self):
 112         if not self.invite_approved:
 113             self.om.delete()
 114 
 115     @property
 116     def organization_member(self):
 117         return OrganizationMember.objects.select_related("organization").get(pk=self.member_id)
 118 
 119     @property
 120     def member_pending(self):
 121         return self.om.is_pending
 122 
 123     @property
 124     def invite_approved(self):
 125         return self.om.invite_approved
 126 
 127     @property
 128     def valid_token(self):
 129         if self.token is None:
 130             return False
 131         if self.om.token_expired:
 132             return False
 133         return constant_time_compare(self.om.token or self.om.legacy_token, self.token)
 134 
 135     @property
 136     def user_authenticated(self):
 137         return self.request.user.is_authenticated()
 138 
 139     @property
 140     def needs_2fa(self):
 141         org_requires_2fa = self.om.organization.flags.require_2fa.is_set
 142         user_has_2fa = Authenticator.objects.user_has_2fa(self.request.user.id)
 143         return org_requires_2fa and not user_has_2fa
 144 
 145     @property
 146     def member_already_exists(self):
 147         if not self.user_authenticated:
 148             return False
 149 
 150         return OrganizationMember.objects.filter(
 151             organization=self.om.organization, user=self.request.user
 152         ).exists()
 153 
 154     @property
 155     def valid_request(self):
 156         return (
 157             self.member_pending
 158             and self.invite_approved
 159             and self.valid_token
 160             and self.user_authenticated
 161             and not self.needs_2fa
 162         )
 163 
 164     def accept_invite(self, user=None):
 165         om = self.om
 166 
 167         if user is None:
 168             user = self.request.user
 169 
 170         if self.member_already_exists:
 171             self.handle_member_already_exists()
 172             om.delete()
 173         else:
 174             om.set_user(user)
 175             om.save()
 176 
 177             create_audit_entry(
 178                 self.request,
 179                 actor=user,
 180                 organization=om.organization,
 181                 target_object=om.id,
 182                 target_user=user,
 183                 event=AuditLogEntryEvent.MEMBER_ACCEPT,
 184                 data=om.get_audit_log_data(),
 185             )
 186 
 187             self.handle_success()
 188             metrics.incr("organization.invite-accepted", sample_rate=1.0)
