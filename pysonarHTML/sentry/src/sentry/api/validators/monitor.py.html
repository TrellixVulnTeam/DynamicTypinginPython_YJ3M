  44 class CronJobValidator(serializers.Serializer):
  45     schedule_type = serializers.ChoiceField(
  46         choices=zip(SCHEDULE_TYPES.keys(), SCHEDULE_TYPES.keys())
  47     )
  48     schedule = ObjectField()
  49     checkin_margin = EmptyIntegerField(required=False, default=None)
  50     max_runtime = EmptyIntegerField(required=False, default=None)
  51 
  52     def validate_schedule_type(self, value):
  53         if value:
  54             value = SCHEDULE_TYPES[value]
  55         return value
  56 
  57     def validate(self, attrs):
  58         if "schedule_type" in attrs:
  59             schedule_type = attrs["schedule_type"]
  60         else:
  61             schedule_type = self.instance["schedule_type"]
  62 
  63         value = attrs.get("schedule")
  64         if not value:
  65             return attrs
  66 
  67         if schedule_type == ScheduleType.INTERVAL:
  68             if not isinstance(value, list):
  69                 raise ValidationError("Invalid value for schedule_type")
  70             if not isinstance(value[0], int):
  71                 raise ValidationError("Invalid value for schedule unit count (index 0)")
  72             if value[1] not in INTERVAL_NAMES:
  73                 raise ValidationError("Invalid value for schedule unit name (index 1)")
  74         elif schedule_type == ScheduleType.CRONTAB:
  75             if not isinstance(value, six.string_types):
  76                 raise ValidationError("Invalid value for schedule_type")
  77             value = value.strip()
  78             if value.startswith("@"):
  79                 try:
  80                     value = NONSTANDARD_CRONTAB_SCHEDULES[value]
  81                 except KeyError:
  82                     raise ValidationError("Schedule was not parseable")
  83             if not croniter.is_valid(value):
  84                 raise ValidationError("Schedule was not parseable")
  85             attrs["schedule"] = value
  86         return attrs

  89 class MonitorValidator(serializers.Serializer):
  90     project = ProjectField()
  91     name = serializers.CharField()
  92     status = serializers.ChoiceField(
  93         choices=zip(MONITOR_STATUSES.keys(), MONITOR_STATUSES.keys()), default="active"
  94     )
  95     type = serializers.ChoiceField(choices=zip(MONITOR_TYPES.keys(), MONITOR_TYPES.keys()))
  96     config = ObjectField()
  97 
  98     def validate(self, attrs):
  99         attrs = super(MonitorValidator, self).validate(attrs)
 100         type = self.instance["type"] if self.instance else self.initial_data.get("type")
 101         if type in MONITOR_TYPES:
 102             type = MONITOR_TYPES[type]
 103         if type == MonitorType.CRON_JOB:
 104             validator = CronJobValidator(
 105                 instance=self.instance.get("config", {}) if self.instance else {},
 106                 data=attrs.get("config", {}),
 107                 partial=self.partial,
 108             )
 109             validator.is_valid(raise_exception=True)
 110             attrs["config"] = validator.validated_data
 111         elif not type:
 112             return attrs
 113         else:
 114             raise NotImplementedError
 115         return attrs

 127     def update(self, instance, validated_data):
 128         has_config = "config" in instance or "config" in validated_data
 129         if has_config:
 130             config = instance.get("config", {})
 131             config.update(validated_data.get("config", {}))
 132         instance.update(validated_data)
 133         if has_config:
 134             instance["config"] = config
 135         return instance
