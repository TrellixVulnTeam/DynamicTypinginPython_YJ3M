  13 class SentryAPIException(APIException):
  14     code = ""
  15     message = ""
  16 
  17     def __init__(self, code=None, message=None, detail=None, **kwargs):
  18         # Note that we no longer call the base `__init__` here. This is because
  19         # DRF now forces all detail messages that subclass `APIException` to a
  20         # string, which breaks our format.
  21         # https://www.django-rest-framework.org/community/3.0-announcement/#miscellaneous-notes
  22         if detail is None:
  23             detail = {
  24                 "code": code or self.code,
  25                 "message": message or self.message,
  26                 "extra": kwargs,
  27             }
  28 
  29         self.detail = {"detail": detail}
