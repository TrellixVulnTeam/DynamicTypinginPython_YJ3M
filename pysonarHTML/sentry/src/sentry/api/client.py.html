  14 class ApiError(Exception):
  15     def __init__(self, status_code, body):
  16         self.status_code = status_code
  17         self.body = body
  18 
  19     def __str__(self):
  20         return u"status={} body={}".format(self.status_code, self.body)
  21 
  22     def __repr__(self):
  23         return u"<ApiError: {}>".format(self)

  26 class ApiClient(object):
  27     prefix = "/api/0"
  28 
  29     ApiError = ApiError
  30 
  31     def request(
  32         self,
  33         method,
  34         path,
  35         user=None,
  36         auth=None,
  37         params=None,
  38         data=None,
  39         is_sudo=None,
  40         is_superuser=None,
  41         request=None,
  42     ):
  43         if self.prefix not in path:
  44             full_path = self.prefix + path
  45         else:
  46             full_path = path
  47 
  48         # we explicitly do not allow you to override the request *and* the user
  49         # as then other checks like is_superuser would need overwritten
  50         assert not (request and (user or auth)), "use either request or auth"
  51 
  52         resolver_match = resolve(full_path)
  53         callback, callback_args, callback_kwargs = resolver_match
  54 
  55         if data:
  56             # we encode to ensure compatibility
  57             data = json.loads(json.dumps(data))
  58 
  59         rf = APIRequestFactory()
  60         mock_request = getattr(rf, method.lower())(full_path, data or {})
  61         # Flag to our API class that we should trust this auth passed through
  62         mock_request.__from_api_client__ = True
  63 
  64         if request:
  65             mock_request.auth = getattr(request, "auth", None)
  66             mock_request.user = request.user
  67 
  68             if is_sudo is None:
  69                 mock_request.is_sudo = lambda: request.is_sudo()
  70             else:
  71                 mock_request.is_sudo = lambda: is_sudo
  72             mock_request.session = request.session
  73 
  74             if is_superuser is None:
  75                 mock_request.superuser = request.superuser
  76             else:
  77                 mock_request.superuser = Superuser(mock_request)
  78         else:
  79             mock_request.auth = auth
  80             mock_request.user = user
  81             mock_request.is_sudo = lambda: is_sudo
  82             mock_request.session = {}
  83             mock_request.superuser = Superuser(mock_request)
  84 
  85         mock_request.is_superuser = lambda: mock_request.superuser.is_active
  86 
  87         if request:
  88             # superuser checks require access to IP
  89             mock_request.META["REMOTE_ADDR"] = request.META["REMOTE_ADDR"]
  90 
  91         force_authenticate(mock_request, user, auth)
  92 
  93         if params:
  94             mock_request.GET._mutable = True
  95             mock_request.GET.update(params)
  96             mock_request.GET._mutable = False
  97 
  98         if data:
  99             mock_request.POST._mutable = True
 100             mock_request.POST.update(data)
 101             mock_request.POST._mutable = False
 102 
 103         response = callback(mock_request, *callback_args, **callback_kwargs)
 104 
 105         if 200 <= response.status_code < 400:
 106             return response
 107         raise self.ApiError(response.status_code, response.data)
