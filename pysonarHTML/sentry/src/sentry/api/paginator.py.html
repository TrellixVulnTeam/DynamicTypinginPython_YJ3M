  31                 self.key, self.desc = order_by[1:], True
  32             else:
  33                 self.key, self.desc = order_by, False

  36             self.desc = False

  41     def _is_asc(self, is_prev):
  42         return (self.desc and is_prev) or not (self.desc or is_prev)

  44     def _build_queryset(self, value, is_prev):
  45         queryset = self.queryset
  46 
  47         # "asc" controls whether or not we need to change the ORDER BY to
  48         # ascending.  If we're sorting by DESC but we're using a previous
  49         # page cursor, we'll change the ordering to ASC and reverse the
  50         # list below (this is so we know how to get the before/after row).
  51         # If we're sorting ASC _AND_ we're not using a previous page cursor,
  52         # then we'll need to resume using ASC.
  53         asc = self._is_asc(is_prev)
  54 
  55         # We need to reverse the ORDER BY if we're using a cursor for a
  56         # previous page so we know exactly where we ended last page.  The
  57         # results will get reversed back to the requested order below.
  58         if self.key:
  59             if self.key in queryset.query.order_by:
  60                 if not asc:
  61                     index = queryset.query.order_by.index(self.key)
  62                     queryset.query.order_by[index] = "-%s" % (queryset.query.order_by[index])
  63             elif ("-%s" % self.key) in queryset.query.order_by:
  64                 if asc:
  65                     index = queryset.query.order_by.index("-%s" % (self.key))
  66                     queryset.query.order_by[index] = queryset.query.order_by[index][1:]
  67             else:
  68                 if asc:
  69                     queryset = queryset.order_by(self.key)
  70                 else:
  71                     queryset = queryset.order_by("-%s" % self.key)
  72 
  73         if value:
  74             assert self.key
  75             if self.key in queryset.query.extra:
  76                 col_query, col_params = queryset.query.extra[self.key]
  77                 col_params = col_params[:]
  78             else:
  79                 col_query, col_params = quote_name(self.key), []
  80             col_params.append(value)
  81 
  82             if asc:
  83                 queryset = queryset.extra(
  84                     where=["%s.%s >= %%s" % (queryset.model._meta.db_table, col_query)],
  85                     params=col_params,
  86                 )
  87             else:
  88                 queryset = queryset.extra(
  89                     where=["%s.%s <= %%s" % (queryset.model._meta.db_table, col_query)],
  90                     params=col_params,
  91                 )
  92 
  93         return queryset

 101     def get_result(self, limit=100, cursor=None, count_hits=False, known_hits=None):
 102         # cursors are:
 103         #   (identifier(integer), row offset, is_prev)
 104         if cursor is None:
 105             cursor = Cursor(0, 0, 0)
 106 
 107         limit = min(limit, self.max_limit)
 108 
 109         if cursor.value:
 110             cursor_value = self.value_from_cursor(cursor)
 111         else:
 112             cursor_value = 0
 113 
 114         queryset = self._build_queryset(cursor_value, cursor.is_prev)
 115 
 116         # TODO(dcramer): this does not yet work correctly for ``is_prev`` when
 117         # the key is not unique
 118         if count_hits:
 119             hits = self.count_hits(MAX_HITS_LIMIT)
 120         elif known_hits is not None:
 121             hits = known_hits
 122         else:
 123             hits = None
 124 
 125         offset = cursor.offset
 126         # The extra amount is needed so we can decide in the ResultCursor if there is
 127         # more on the next page.
 128         extra = 1
 129         # this effectively gets us the before row, and the current (after) row
 130         # every time. Do not offset if the provided cursor value was empty since
 131         # there is nothing to traverse past.
 132         # We need to actually fetch the before row so that we can compare it to the
 133         # cursor value. This allows us to handle an edge case where the first row
 134         # for a given cursor is the same row that generated the cursor on the
 135         # previous page, but we want to display since it has had its its sort value
 136         # updated.
 137         if cursor.is_prev and cursor.value:
 138             extra += 1
 139 
 140         stop = offset + limit + extra
 141         results = list(queryset[offset:stop])
 142 
 143         if cursor.is_prev and cursor.value:
 144             # If the first result is equal to the cursor_value then it's safe to filter
 145             # it out, since the value hasn't been updated
 146             if results and self.get_item_key(results[0], for_prev=True) == cursor.value:
 147                 results = results[1:]
 148             # Otherwise we may have fetched an extra row, just drop it off the end if so.
 149             elif len(results) == offset + limit + extra:
 150                 results = results[:-1]
 151 
 152         if cursor.is_prev:
 153             results.reverse()
 154 
 155         return build_cursor(
 156             results=results,
 157             limit=limit,
 158             hits=hits,
 159             max_hits=MAX_HITS_LIMIT if count_hits else None,
 160             cursor=cursor,
 161             is_desc=self.desc,
 162             key=self.get_item_key,
 163             on_results=self.on_results,
 164         )
 165 
 166     def count_hits(self, max_hits):
 167         if not max_hits:
 168             return 0
 169         hits_query = self.queryset.values()[:max_hits].query
 170         # clear out any select fields (include select_related) and pull just the id
 171         hits_query.clear_select_clause()
 172         hits_query.add_fields(["id"])
 173         hits_query.clear_ordering(force_empty=True)
 174         try:
 175             h_sql, h_params = hits_query.sql_with_params()
 176         except EmptyResultSet:
 177             return 0
 178         cursor = connections[self.queryset.db].cursor()
 179         cursor.execute(u"SELECT COUNT(*) FROM ({}) as t".format(h_sql), h_params)
 180         return cursor.fetchone()[0]

 196         value = getattr(item, self.key)

 209 class OffsetPaginator(object):
 210     def __init__(
 211         self, queryset, order_by=None, max_limit=MAX_LIMIT, max_offset=None, on_results=None
 212     ):
 213         self.key = (
 214             order_by
 215             if order_by is None or isinstance(order_by, (list, tuple, set))
 216             else (order_by,)
 217         )
 218         self.queryset = queryset
 219         self.max_limit = max_limit
 220         self.max_offset = max_offset
 221         self.on_results = on_results
 222 
 223     def get_result(self, limit=100, cursor=None):
 224         # offset is page #
 225         # value is page limit
 226         if cursor is None:
 227             cursor = Cursor(0, 0, 0)
 228 
 229         limit = min(limit, self.max_limit)
 230 
 231         queryset = self.queryset
 232         if self.key:
 233             queryset = queryset.order_by(*self.key)
 234 
 235         page = cursor.offset
 236         offset = cursor.offset * cursor.value
 237         stop = offset + (cursor.value or limit) + 1
 238 
 239         if self.max_offset is not None and offset >= self.max_offset:
 240             raise BadPaginationError("Pagination offset too large")
 241         if offset < 0:
 242             raise BadPaginationError("Pagination offset cannot be negative")
 243 
 244         results = list(queryset[offset:stop])
 245         if cursor.value != limit:
 246             results = results[-(limit + 1) :]
 247 
 248         next_cursor = Cursor(limit, page + 1, False, len(results) > limit)
 249         prev_cursor = Cursor(limit, page - 1, True, page > 0)
 250 
 251         results = list(results[:limit])
 252         if self.on_results:
 253             results = self.on_results(results)
 254 
 255         return CursorResult(results=results, next=next_cursor, prev=prev_cursor)

 258 class MergingOffsetPaginator(OffsetPaginator):
 259     """This paginator uses a function to first look up items from an
 260     independently paginated resource to only then fall back to a query set.
 261     This is for instance useful if you want to query snuba for the primary
 262     sort order and then look up data in postgres.
 263     """
 264 
 265     def __init__(
 266         self,
 267         queryset,
 268         data_load_func,
 269         apply_to_queryset,
 270         key_from_model=None,
 271         key_from_data=None,
 272         max_limit=MAX_LIMIT,
 273         on_results=None,
 274     ):
 275         super(MergingOffsetPaginator, self).__init__(
 276             queryset, max_limit=max_limit, on_results=on_results
 277         )
 278         self.data_load_func = data_load_func
 279         self.apply_to_queryset = apply_to_queryset
 280         self.key_from_model = key_from_model or (lambda x: x.id)
 281         self.key_from_data = key_from_data or (lambda x: x)
 282 
 283     def get_result(self, limit=100, cursor=None):
 284         if cursor is None:
 285             cursor = Cursor(0, 0, 0)
 286 
 287         limit = min(limit, self.max_limit)
 288 
 289         page = cursor.offset
 290         offset = cursor.offset * cursor.value
 291         limit = (cursor.value or limit) + 1
 292 
 293         if self.max_offset is not None and offset >= self.max_offset:
 294             raise BadPaginationError("Pagination offset too large")
 295         if offset < 0:
 296             raise BadPaginationError("Pagination offset cannot be negative")
 297 
 298         primary_results = self.data_load_func(offset=offset, limit=limit)
 299 
 300         queryset = self.apply_to_queryset(self.queryset, primary_results)
 301 
 302         mapping = {}
 303         for model in queryset:
 304             mapping[self.key_from_model(model)] = model
 305 
 306         results = []
 307         for row in primary_results:
 308             model = mapping.get(self.key_from_data(row))
 309             if model is not None:
 310                 results.append(model)
 311 
 312         next_cursor = Cursor(limit, page + 1, False, len(primary_results) > limit)
 313         prev_cursor = Cursor(limit, page - 1, True, page > 0)
 314         results = list(results[:limit])
 315 
 316         if self.on_results:
 317             results = self.on_results(results)
 318 
 319         return CursorResult(results=results, next=next_cursor, prev=prev_cursor)

 322 def reverse_bisect_left(a, x, lo=0, hi=None):
 323     """\
 324     Similar to ``bisect.bisect_left``, but expects the data in the array ``a``
 325     to be provided in descending order, rather than the ascending order assumed
 326     by ``bisect_left``.
 327 
 328     The returned index ``i`` partitions the array ``a`` into two halves so that:
 329 
 330     - left side: ``all(val > x for val in a[lo:i])``
 331     - right side: ``all(val <= x for val in a[i:hi])``
 332     """
 333     if lo < 0:
 334         raise ValueError("lo must be non-negative")
 335 
 336     if hi is None or hi > len(a):
 337         hi = len(a)
 338 
 339     while lo < hi:
 340         mid = (lo + hi) // 2
 341         if a[mid] > x:
 342             lo = mid + 1
 343         else:
 344             hi = mid
 345 
 346     return lo

 349 class SequencePaginator(object):
 350     def __init__(self, data, reverse=False, max_limit=MAX_LIMIT, on_results=None):
 351         self.scores, self.values = (
 352             map(list, zip(*sorted(data, reverse=reverse))) if data else ([], [])
 353         )
 354         self.reverse = reverse
 355         self.search = functools.partial(
 356             reverse_bisect_left if reverse else bisect.bisect_left, self.scores
 357         )
 358         self.max_limit = max_limit
 359         self.on_results = on_results
 360 
 361     def get_result(self, limit, cursor=None, count_hits=False, known_hits=None):
 362         limit = min(limit, self.max_limit)
 363 
 364         if cursor is None:
 365             cursor = Cursor(0, 0, False)
 366 
 367         assert cursor.offset > -1
 368 
 369         if cursor.value == 0:
 370             position = len(self.scores) if cursor.is_prev else 0
 371         else:
 372             position = self.search(cursor.value)
 373 
 374         position = position + cursor.offset
 375 
 376         if cursor.is_prev:
 377             # TODO: It might make sense to ensure that this hi value is at
 378             # least the length of the page + 1 if we want to ensure we return a
 379             # full page of results when paginating backwards while data is
 380             # being mutated.
 381             hi = min(position, len(self.scores))
 382             lo = max(hi - limit, 0)
 383         else:
 384             lo = max(position, 0)
 385             hi = min(lo + limit, len(self.scores))
 386 
 387         if self.scores:
 388             prev_score = self.scores[min(lo, len(self.scores) - 1)]
 389             prev_cursor = Cursor(
 390                 prev_score, lo - self.search(prev_score, hi=lo), True, True if lo > 0 else False
 391             )
 392 
 393             next_score = self.scores[min(hi, len(self.scores) - 1)]
 394             next_cursor = Cursor(
 395                 next_score,
 396                 hi - self.search(next_score, hi=hi),
 397                 False,
 398                 True if hi < len(self.scores) else False,
 399             )
 400         else:
 401             prev_cursor = Cursor(cursor.value, cursor.offset, True, False)
 402             next_cursor = Cursor(cursor.value, cursor.offset, False, False)
 403 
 404         results = self.values[lo:hi]
 405         if self.on_results:
 406             results = self.on_results(results)
 407 
 408         if known_hits is not None:
 409             hits = min(known_hits, MAX_HITS_LIMIT)
 410         elif count_hits:
 411             hits = min(len(self.scores), MAX_HITS_LIMIT)
 412         else:
 413             hits = None
 414 
 415         return CursorResult(
 416             results,
 417             prev=prev_cursor,
 418             next=next_cursor,
 419             hits=hits,
 420             max_hits=MAX_HITS_LIMIT if hits is not None else None,
 421         )

 424 class GenericOffsetPaginator(object):
 425     """
 426     A paginator for getting pages of results for a query using the OFFSET/LIMIT
 427     mechanism.
 428 
 429     This class makes the assumption that the query provides a static,
 430     totally-ordered view on the data, so that the next page of data can be
 431     retrieved by incrementing OFFSET to the next multiple of LIMIT with no
 432     overlaps or gaps from the previous page.
 433 
 434     It is potentially less performant than a ranged query solution that might
 435     not to have to look at as many rows.
 436 
 437     Can either take data as a list or dictionary with data as value in order to
 438     return full object if necessary. (if isinstance statement)
 439     """
 440 
 441     def __init__(self, data_fn):
 442         self.data_fn = data_fn
 443 
 444     def get_result(self, limit, cursor=None):
 445         assert limit > 0
 446         offset = cursor.offset if cursor is not None else 0
 447         # Request 1 more than limit so we can tell if there is another page
 448         data = self.data_fn(offset=offset, limit=limit + 1)
 449 
 450         if isinstance(data, list):
 451             has_more = len(data) == limit + 1
 452             if has_more:
 453                 data.pop()
 454         elif isinstance(data.get("data"), list):
 455             has_more = len(data["data"]) == limit + 1
 456             if has_more:
 457                 data["data"].pop()
 458         else:
 459             raise NotImplementedError
 460 
 461         # Since we are not issuing ranged queries, our cursors always have
 462         # `value=0` (ie. all rows have the same value), and so offset naturally
 463         # becomes the absolute row offset from the beginning of the entire
 464         # dataset, which is the same meaning as SQLs `OFFSET`.
 465         return CursorResult(
 466             data,
 467             prev=Cursor(0, max(0, offset - limit), True, offset > 0),
 468             next=Cursor(0, max(0, offset + limit), False, has_more),
 469         )
 470         # TODO use Cursor.value as the `end` argument to data_fn() so that
 471         # subsequent pages returned using these cursors are using the same end
 472         # date for queries, this should stop drift from new incoming events.
 473

 475 class CombinedQuerysetPaginator(object):
 476     multiplier = 1000000  # Use microseconds for date keys.
 477 
 478     def __init__(self, order_by, querysets, on_results=None):
 479         if order_by:
 480             if order_by.startswith("-"):
 481                 self.key, self.desc = order_by[1:], True
 482             else:
 483                 self.key, self.desc = order_by, False
 484         else:
 485             self.key = None
 486             self.desc = False
 487 
 488         self.querysets = querysets
 489         self.on_results = on_results
 490         # IMPROVEMENT: Ensure each model in the querysets has attribute of self.key
 491         # ... or accept a function that can map to the proper key based on model instance
 492         # use case could be sorting by name/label for AlertRule / Rule
 493         # For now...just be diligent in your usage.
 494 
 495     def get_item_key(self, item, for_prev=False):
 496         if self.key == "date_added":  # Could generalize this more
 497             return self.multiplier * float(getattr(item, self.key).strftime("%s.%f"))
 498         else:
 499             value = getattr(item, self.key)
 500             return math.floor(value) if self._is_asc(for_prev) else math.ceil(value)
 501 
 502     def value_from_cursor(self, cursor):
 503         if self.key == "date_added":  # Could generalize this more
 504             return datetime.fromtimestamp(float(cursor.value) / self.multiplier).replace(
 505                 tzinfo=timezone.utc
 506             )
 507         else:
 508             return cursor.value
 509 
 510     def _is_asc(self, is_prev):
 511         return (self.desc and is_prev) or not (self.desc or is_prev)
 512 
 513     def _build_combined_querysets(self, value, is_prev, limit, extra):
 514         asc = self._is_asc(is_prev)
 515 
 516         if asc:
 517             order_by = self.key
 518             filter_condition = "%s__gte" % self.key
 519         else:
 520             order_by = "-%s" % self.key
 521             filter_condition = "%s__lte" % self.key
 522 
 523         filters = {}
 524         if value is not None:
 525             assert self.key
 526             filters[filter_condition] = value
 527 
 528         combined_querysets = list()
 529         for queryset in self.querysets:
 530             queryset = queryset.filter(**filters).order_by(order_by)[: (limit + extra)]
 531             combined_querysets += list(queryset)
 532 
 533         combined_querysets.sort(
 534             key=lambda item: (getattr(item, self.key), type(item).__name__), reverse=not asc
 535         )
 536         return combined_querysets
 537 
 538     def get_result(self, cursor=None, limit=100):
 539         if cursor is None:
 540             cursor = Cursor(0, 0, 0)
 541 
 542         if cursor.value:
 543             cursor_value = self.value_from_cursor(cursor)
 544         else:
 545             cursor_value = None
 546 
 547         limit = min(limit, MAX_LIMIT)
 548 
 549         offset = cursor.offset
 550         extra = 1
 551         if cursor.is_prev and cursor.value:
 552             extra += 1
 553         combined_querysets = self._build_combined_querysets(
 554             cursor_value, cursor.is_prev, limit, extra
 555         )
 556 
 557         stop = offset + limit + extra
 558         results = list(combined_querysets[offset:stop])
 559 
 560         if cursor.is_prev and cursor.value:
 561             # If the first result is equal to the cursor_value then it's safe to filter
 562             # it out, since the value hasn't been updated
 563             if results and self.get_item_key(results[0], for_prev=True) == cursor.value:
 564                 results = results[1:]
 565             # Otherwise we may have fetched an extra row, just drop it off the end if so.
 566             elif len(results) == offset + limit + extra:
 567                 results = results[:-1]
 568 
 569         # We reversed the results when generating the querysets, so we need to reverse back now.
 570         if cursor.is_prev:
 571             results.reverse()
 572 
 573         return build_cursor(
 574             results=results,
 575             cursor=cursor,
 576             key=self.get_item_key,
 577             limit=limit,
 578             is_desc=self.desc,
 579             on_results=self.on_results,
 580         )
