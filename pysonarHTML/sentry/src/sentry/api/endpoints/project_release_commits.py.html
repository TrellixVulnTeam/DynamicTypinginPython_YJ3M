  10 class ProjectReleaseCommitsEndpoint(ProjectEndpoint):
  11     doc_section = DocSection.RELEASES
  12     permission_classes = (ProjectReleasePermission,)
  13 
  14     def get(self, request, project, version):
  15         """
  16         List a Project Release's Commits
  17         ````````````````````````````````
  18 
  19         Retrieve a list of commits for a given release.
  20 
  21         :pparam string organization_slug: the slug of the organization the
  22                                           release belongs to.
  23         :pparam string project_slug: the slug of the project to list the
  24                                      release files of.
  25         :pparam string version: the version identifier of the release.
  26         :auth: required
  27         """
  28         try:
  29             release = Release.objects.get(
  30                 organization_id=project.organization_id, projects=project, version=version
  31             )
  32         except Release.DoesNotExist:
  33             raise ResourceDoesNotExist
  34 
  35         queryset = ReleaseCommit.objects.filter(release=release).select_related(
  36             "commit", "commit__author"
  37         )
  38 
  39         return self.paginate(
  40             request=request,
  41             queryset=queryset,
  42             order_by="order",
  43             on_results=lambda x: serialize([rc.commit for rc in x], request.user),
  44         )
