  35 class TeamSerializer(serializers.ModelSerializer):
  36     slug = serializers.RegexField(r"^[a-z0-9_\-]+$", max_length=50)
  37 
  38     class Meta:
  39         model = Team
  40         fields = ("name", "slug")
  41 
  42     def validate_slug(self, value):
  43         qs = Team.objects.filter(slug=value, organization=self.instance.organization).exclude(
  44             id=self.instance.id
  45         )
  46         if qs.exists():
  47             raise serializers.ValidationError('The slug "%s" is already in use.' % (value,))
  48         return value

  55     def get(self, request, team):
  56         """
  57         Retrieve a Team
  58         ```````````````
  59 
  60         Return details on an individual team.
  61 
  62         :pparam string organization_slug: the slug of the organization the
  63                                           team belongs to.
  64         :pparam string team_slug: the slug of the team to get.
  65         :auth: required
  66         """
  67         context = serialize(team, request.user)
  68         context["organization"] = serialize(team.organization, request.user)
  69 
  70         return Response(context)

  72     @attach_scenarios([update_team_scenario])
  73     def put(self, request, team):
  74         """
  75         Update a Team
  76         `````````````
  77 
  78         Update various attributes and configurable settings for the given
  79         team.
  80 
  81         :pparam string organization_slug: the slug of the organization the
  82                                           team belongs to.
  83         :pparam string team_slug: the slug of the team to get.
  84         :param string name: the new name for the team.
  85         :param string slug: a new slug for the team.  It has to be unique
  86                             and available.
  87         :auth: required
  88         """
  89         serializer = TeamSerializer(team, data=request.data, partial=True)
  90         if serializer.is_valid():
  91             team = serializer.save()
  92 
  93             self.create_audit_entry(
  94                 request=request,
  95                 organization=team.organization,
  96                 target_object=team.id,
  97                 event=AuditLogEntryEvent.TEAM_EDIT,
  98                 data=team.get_audit_log_data(),
  99             )
 100 
 101             return Response(serialize(team, request.user))
 102 
 103         return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)

 105     @sudo_required
 106     def delete(self, request, team):
 107         """
 108         Delete a Team
 109         `````````````
 110 
 111         Schedules a team for deletion.
 112 
 113         **Note:** Deletion happens asynchronously and therefor is not
 114         immediate.  However once deletion has begun the state of a project
 115         changes and will be hidden from most public views.
 116         """
 117         updated = Team.objects.filter(id=team.id, status=TeamStatus.VISIBLE).update(
 118             status=TeamStatus.PENDING_DELETION
 119         )
 120         if updated:
 121             transaction_id = uuid4().hex
 122 
 123             self.create_audit_entry(
 124                 request=request,
 125                 organization=team.organization,
 126                 target_object=team.id,
 127                 event=AuditLogEntryEvent.TEAM_REMOVE,
 128                 data=team.get_audit_log_data(),
 129                 transaction_id=transaction_id,
 130             )
 131 
 132             delete_team.apply_async(kwargs={"object_id": team.id, "transaction_id": transaction_id})
 133 
 134             delete_logger.info(
 135                 "object.delete.queued",
 136                 extra={
 137                     "object_id": team.id,
 138                     "transaction_id": transaction_id,
 139                     "model": type(team).__name__,
 140                 },
 141             )
 142 
 143         return Response(status=204)
