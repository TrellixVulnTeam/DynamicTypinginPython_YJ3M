  32 class AcceptProjectTransferEndpoint(Endpoint):
  33     authentication_classes = (SessionAuthentication,)
  34     permission_classes = (IsAuthenticated,)

  36     def get_validated_data(self, data, user):
  37         try:
  38             data = unsign(force_str(data))
  39         except SignatureExpired:
  40             raise InvalidPayload("Project transfer link has expired.")
  41         except BadSignature:
  42             raise InvalidPayload("Could not approve transfer, please make sure link is valid.")
  43 
  44         if data["user_id"] != user.id:
  45             raise InvalidPayload("Invalid permissions")
  46 
  47         try:
  48             project = Project.objects.get(
  49                 id=data["project_id"], organization_id=data["from_organization_id"]
  50             )
  51         except Project.DoesNotExist:
  52             raise InvalidPayload("Project no longer exists")
  53 
  54         return data, project
  55 
  56     @sudo_required
  57     def get(self, request):
  58         try:
  59             data = request.GET["data"]
  60         except KeyError:
  61             raise Http404
  62 
  63         try:
  64             data, project = self.get_validated_data(data, request.user)
  65         except InvalidPayload as e:
  66             return Response({"detail": six.text_type(e)}, status=400)
  67 
  68         organizations = Organization.objects.filter(
  69             status=OrganizationStatus.ACTIVE,
  70             id__in=OrganizationMember.objects.filter(
  71                 user=request.user, role=roles.get_top_dog().id
  72             ).values_list("organization_id", flat=True),
  73         )
  74 
  75         return Response(
  76             {
  77                 "organizations": serialize(
  78                     list(organizations),
  79                     request.user,
  80                     DetailedOrganizationSerializerWithProjectsAndTeams(),
  81                     access=request.access,
  82                 ),
  83                 "project": {"slug": project.slug, "id": project.id},
  84             }
  85         )

  87     @sudo_required
  88     def post(self, request):
  89         try:
  90             data = request.data["data"]
  91         except KeyError:
  92             raise Http404
  93 
  94         try:
  95             data, project = self.get_validated_data(data, request.user)
  96         except InvalidPayload as e:
  97             return Response({"detail": six.text_type(e)}, status=400)
  98 
  99         transaction_id = data["transaction_id"]
 100 
 101         org_slug = request.data.get("organization")
 102         team_id = request.data.get("team")
 103 
 104         if org_slug is not None and team_id is not None:
 105             return Response(
 106                 {"detail": "Choose either a team or an organization, not both"}, status=400
 107             )
 108 
 109         if org_slug is None and team_id is None:
 110             return Response(
 111                 {"detail": "Choose either a team or an organization to transfer the project to"},
 112                 status=400,
 113             )
 114 
 115         if team_id:
 116             try:
 117                 team = Team.objects.get(id=team_id)
 118             except Team.DoesNotExist:
 119                 return Response({"detail": "Invalid team"}, status=400)
 120 
 121             # check if user is an owner of the team's org
 122             is_team_org_owner = OrganizationMember.objects.filter(
 123                 user__is_active=True,
 124                 user=request.user,
 125                 role=roles.get_top_dog().id,
 126                 organization_id=team.organization_id,
 127             ).exists()
 128 
 129             if not is_team_org_owner:
 130                 return Response({"detail": "Invalid team"}, status=400)
 131 
 132             project.transfer_to(team=team)
 133 
 134         if org_slug:
 135             try:
 136                 organization = Organization.objects.get(slug=org_slug)
 137             except Organization.DoesNotExist:
 138                 return Response({"detail": "Invalid organization"}, status=400)
 139 
 140             # check if user is an owner of the organization
 141             is_org_owner = OrganizationMember.objects.filter(
 142                 user__is_active=True,
 143                 user=request.user,
 144                 role=roles.get_top_dog().id,
 145                 organization_id=organization.id,
 146             ).exists()
 147 
 148             if not is_org_owner:
 149                 return Response({"detail": "Invalid organization"}, status=400)
 150 
 151             project.transfer_to(organization=organization)
 152 
 153         self.create_audit_entry(
 154             request=request,
 155             organization=project.organization,
 156             target_object=project.id,
 157             event=AuditLogEntryEvent.PROJECT_ACCEPT_TRANSFER,
 158             data=project.get_audit_log_data(),
 159             transaction_id=transaction_id,
 160         )
 161 
 162         return Response(status=204)
