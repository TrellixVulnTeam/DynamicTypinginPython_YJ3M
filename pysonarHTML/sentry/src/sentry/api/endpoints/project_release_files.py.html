  54 class ProjectReleaseFilesEndpoint(ProjectEndpoint):
  55     doc_section = DocSection.RELEASES
  56     permission_classes = (ProjectReleasePermission,)
  57 
  58     @attach_scenarios([list_files_scenario])
  59     def get(self, request, project, version):
  60         """
  61         List a Project Release's Files
  62         ``````````````````````````````
  63 
  64         Retrieve a list of files for a given release.
  65 
  66         :pparam string organization_slug: the slug of the organization the
  67                                           release belongs to.
  68         :pparam string project_slug: the slug of the project to list the
  69                                      release files of.
  70         :pparam string version: the version identifier of the release.
  71         :qparam string query: If set, this parameter is used to search files.
  72         :auth: required
  73         """
  74         query = request.GET.get("query")
  75 
  76         try:
  77             release = Release.objects.get(
  78                 organization_id=project.organization_id, projects=project, version=version
  79             )
  80         except Release.DoesNotExist:
  81             raise ResourceDoesNotExist
  82 
  83         file_list = (
  84             ReleaseFile.objects.filter(release=release).select_related("file").order_by("name")
  85         )
  86 
  87         if query:
  88             file_list = file_list.filter(Q(name__icontains=query))
  89 
  90         return self.paginate(
  91             request=request,
  92             queryset=file_list,
  93             order_by="name",
  94             paginator_cls=OffsetPaginator,
  95             max_offset=MAX_RELEASE_FILES_OFFSET,
  96             on_results=lambda r: serialize(load_dist(r), request.user),
  97         )

  99     @attach_scenarios([upload_file_scenario])
 100     def post(self, request, project, version):
 101         """
 102         Upload a New Project Release File
 103         `````````````````````````````````
 104 
 105         Upload a new file for the given release.
 106 
 107         Unlike other API requests, files must be uploaded using the
 108         traditional multipart/form-data content-type.
 109 
 110         The optional 'name' attribute should reflect the absolute path
 111         that this file will be referenced as. For example, in the case of
 112         JavaScript you might specify the full web URI.
 113 
 114         :pparam string organization_slug: the slug of the organization the
 115                                           release belongs to.
 116         :pparam string project_slug: the slug of the project to change the
 117                                      release of.
 118         :pparam string version: the version identifier of the release.
 119         :param string name: the name (full path) of the file.
 120         :param string dist: the name of the dist.
 121         :param file file: the multipart encoded file.
 122         :param string header: this parameter can be supplied multiple times
 123                               to attach headers to the file.  Each header
 124                               is a string in the format ``key:value``.  For
 125                               instance it can be used to define a content
 126                               type.
 127         :auth: required
 128         """
 129         try:
 130             release = Release.objects.get(
 131                 organization_id=project.organization_id, projects=project, version=version
 132             )
 133         except Release.DoesNotExist:
 134             raise ResourceDoesNotExist
 135 
 136         logger = logging.getLogger("sentry.files")
 137         logger.info("projectreleasefile.start")
 138 
 139         if "file" not in request.data:
 140             return Response({"detail": "Missing uploaded file"}, status=400)
 141 
 142         fileobj = request.data["file"]
 143 
 144         full_name = request.data.get("name", fileobj.name)
 145         if not full_name or full_name == "file":
 146             return Response({"detail": "File name must be specified"}, status=400)
 147 
 148         name = full_name.rsplit("/", 1)[-1]
 149 
 150         if _filename_re.search(name):
 151             return Response(
 152                 {"detail": "File name must not contain special whitespace characters"}, status=400
 153             )
 154 
 155         dist_name = request.data.get("dist")
 156         dist = None
 157         if dist_name:
 158             dist = release.add_dist(dist_name)
 159 
 160         headers = {"Content-Type": fileobj.content_type}
 161         for headerval in request.data.getlist("header") or ():
 162             try:
 163                 k, v = headerval.split(":", 1)
 164             except ValueError:
 165                 return Response({"detail": "header value was not formatted correctly"}, status=400)
 166             else:
 167                 if _filename_re.search(v):
 168                     return Response(
 169                         {"detail": "header value must not contain special whitespace characters"},
 170                         status=400,
 171                     )
 172                 headers[k] = v.strip()
 173 
 174         file = File.objects.create(name=name, type="release.file", headers=headers)
 175         file.putfile(fileobj, logger=logger)
 176 
 177         try:
 178             with transaction.atomic():
 179                 releasefile = ReleaseFile.objects.create(
 180                     organization_id=release.organization_id,
 181                     release=release,
 182                     file=file,
 183                     name=full_name,
 184                     dist=dist,
 185                 )
 186         except IntegrityError:
 187             file.delete()
 188             return Response({"detail": ERR_FILE_EXISTS}, status=409)
 189 
 190         return Response(serialize(releasefile, request.user), status=201)
