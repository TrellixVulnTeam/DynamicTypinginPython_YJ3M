  57     def download(self, debug_file_id, project):
  58         rate_limited = ratelimits.is_limited(
  59             project=project,
  60             key="rl:DSymFilesEndpoint:download:%s:%s" % (debug_file_id, project.id),
  61             limit=10,
  62         )
  63         if rate_limited:
  64             logger.info(
  65                 "notification.rate_limited",
  66                 extra={"project_id": project.id, "project_debug_file_id": debug_file_id},
  67             )
  68             return HttpResponse({"Too many download requests"}, status=403)
  69 
  70         debug_file = ProjectDebugFile.objects.filter(id=debug_file_id).first()
  71 
  72         if debug_file is None:
  73             raise Http404
  74 
  75         try:
  76             fp = debug_file.file.getfile()
  77             response = StreamingHttpResponse(
  78                 iter(lambda: fp.read(4096), b""), content_type="application/octet-stream"
  79             )
  80             response["Content-Length"] = debug_file.file.size
  81             response["Content-Disposition"] = 'attachment; filename="%s%s"' % (
  82                 posixpath.basename(debug_file.debug_id),
  83                 debug_file.file_extension,
  84             )
  85             return response
  86         except IOError:
  87             raise Http404

  89     def get(self, request, project):
  90         """
  91         List a Project's Debug Information Files
  92         ````````````````````````````````````````
  93 
  94         Retrieve a list of debug information files for a given project.
  95 
  96         :pparam string organization_slug: the slug of the organization the
  97                                           file belongs to.
  98         :pparam string project_slug: the slug of the project to list the
  99                                      DIFs of.
 100         :qparam string query: If set, this parameter is used to locate DIFs with.
 101         :qparam string id: If set, the specified DIF will be sent in the response.
 102         :qparam string file_formats: If set, only DIFs with these formats will be returned.
 103         :auth: required
 104         """
 105         download_requested = request.GET.get("id") is not None
 106         if download_requested and (request.access.has_scope("project:write")):
 107             return self.download(request.GET.get("id"), project)
 108 
 109         code_id = request.GET.get("code_id")
 110         debug_id = request.GET.get("debug_id")
 111         query = request.GET.get("query")
 112         file_formats = request.GET.getlist("file_formats")
 113 
 114         # If this query contains a debug identifier, normalize it to allow for
 115         # more lenient queries (e.g. supporting Breakpad ids). Use the index to
 116         # speed up such queries.
 117         if query and len(query) <= 45 and not debug_id:
 118             try:
 119                 debug_id = normalize_debug_id(query.strip())
 120             except SymbolicError:
 121                 pass
 122 
 123         if debug_id:
 124             # If a debug ID is specified, do not consider the stored code
 125             # identifier and strictly filter by debug identifier. Often there
 126             # are mismatches in the code identifier in PEs.
 127             q = Q(debug_id__exact=debug_id)
 128         elif code_id:
 129             q = Q(code_id__exact=code_id)
 130         elif query:
 131             q = (
 132                 Q(object_name__icontains=query)
 133                 | Q(debug_id__icontains=query)
 134                 | Q(code_id__icontains=query)
 135                 | Q(cpu_name__icontains=query)
 136                 | Q(file__headers__icontains=query)
 137             )
 138 
 139             known_file_format = DIF_MIMETYPES.get(query)
 140             if known_file_format:
 141                 q |= Q(file__headers__icontains=known_file_format)
 142         else:
 143             q = Q()
 144 
 145         file_format_q = Q()
 146         for file_format in file_formats:
 147             known_file_format = DIF_MIMETYPES.get(file_format)
 148             if known_file_format:
 149                 file_format_q |= Q(file__headers__icontains=known_file_format)
 150 
 151         q &= file_format_q
 152 
 153         queryset = ProjectDebugFile.objects.filter(q, project=project).select_related("file")
 154 
 155         return self.paginate(
 156             request=request,
 157             queryset=queryset,
 158             order_by="-id",
 159             paginator_cls=OffsetPaginator,
 160             default_per_page=20,
 161             on_results=lambda x: serialize(x, request.user),
 162         )

 164     def delete(self, request, project):
 165         """
 166         Delete a specific Project's Debug Information File
 167         ```````````````````````````````````````````````````
 168 
 169         Delete a debug information file for a given project.
 170 
 171         :pparam string organization_slug: the slug of the organization the
 172                                           file belongs to.
 173         :pparam string project_slug: the slug of the project to delete the
 174                                      DIF.
 175         :qparam string id: The id of the DIF to delete.
 176         :auth: required
 177         """
 178 
 179         if request.GET.get("id") and (request.access.has_scope("project:write")):
 180             with transaction.atomic():
 181                 debug_file = (
 182                     ProjectDebugFile.objects.filter(id=request.GET.get("id"), project=project)
 183                     .select_related("file")
 184                     .first()
 185                 )
 186                 if debug_file is not None:
 187                     debug_file.delete()
 188                     return Response(status=204)
 189 
 190         return Response(status=404)

 235 def find_missing_chunks(organization, chunks):
 236     """Returns a list of chunks which are missing for an org."""
 237     owned = set(
 238         FileBlobOwner.objects.filter(
 239             blob__checksum__in=chunks, organization=organization
 240         ).values_list("blob__checksum", flat=True)
 241     )
 242     return list(set(chunks) - owned)

 246     permission_classes = (ProjectReleasePermission,)

 248     def post(self, request, project):
 249         """
 250         Assemble one or multiple chunks (FileBlob) into debug files
 251         ````````````````````````````````````````````````````````````
 252 
 253         :auth: required
 254         """
 255         schema = {
 256             "type": "object",
 257             "patternProperties": {
 258                 "^[0-9a-f]{40}$": {
 259                     "type": "object",
 260                     "required": ["name", "chunks"],
 261                     "properties": {
 262                         "name": {"type": "string"},
 263                         "debug_id": {"type": "string"},
 264                         "chunks": {
 265                             "type": "array",
 266                             "items": {"type": "string", "pattern": "^[0-9a-f]{40}$"},
 267                         },
 268                     },
 269                     "additionalProperties": True,
 270                 }
 271             },
 272             "additionalProperties": False,
 273         }
 274 
 275         try:
 276             files = json.loads(request.body)
 277             jsonschema.validate(files, schema)
 278         except jsonschema.ValidationError as e:
 279             return Response({"error": str(e).splitlines()[0]}, status=400)
 280         except BaseException:
 281             return Response({"error": "Invalid json body"}, status=400)
 282 
 283         file_response = {}
 284 
 285         for checksum, file_to_assemble in six.iteritems(files):
 286             name = file_to_assemble.get("name", None)
 287             debug_id = file_to_assemble.get("debug_id", None)
 288             chunks = file_to_assemble.get("chunks", [])
 289 
 290             # First, check the cached assemble status. During assembling, a
 291             # ProjectDebugFile will be created and we need to prevent a race
 292             # condition.
 293             state, detail = get_assemble_status(AssembleTask.DIF, project.id, checksum)
 294             if state == ChunkFileState.OK:
 295                 file_response[checksum] = {
 296                     "state": state,
 297                     "detail": None,
 298                     "missingChunks": [],
 299                     "dif": detail,
 300                 }
 301                 continue
 302             elif state is not None:
 303                 file_response[checksum] = {"state": state, "detail": detail, "missingChunks": []}
 304                 continue
 305 
 306             # Next, check if this project already owns the ProjectDebugFile.
 307             # This can under rare circumstances yield more than one file
 308             # which is why we use first() here instead of get().
 309             dif = (
 310                 ProjectDebugFile.objects.filter(project=project, file__checksum=checksum)
 311                 .select_related("file")
 312                 .order_by("-id")
 313                 .first()
 314             )
 315 
 316             if dif is not None:
 317                 file_response[checksum] = {
 318                     "state": ChunkFileState.OK,
 319                     "detail": None,
 320                     "missingChunks": [],
 321                     "dif": serialize(dif),
 322                 }
 323                 continue
 324 
 325             # There is neither a known file nor a cached state, so we will
 326             # have to create a new file.  Assure that there are checksums.
 327             # If not, we assume this is a poll and report NOT_FOUND
 328             if not chunks:
 329                 file_response[checksum] = {"state": ChunkFileState.NOT_FOUND, "missingChunks": []}
 330                 continue
 331 
 332             # Check if all requested chunks have been uploaded.
 333             missing_chunks = find_missing_chunks(project.organization, chunks)
 334             if missing_chunks:
 335                 file_response[checksum] = {
 336                     "state": ChunkFileState.NOT_FOUND,
 337                     "missingChunks": missing_chunks,
 338                 }
 339                 continue
 340 
 341             # We don't have a state yet, this means we can now start
 342             # an assemble job in the background.
 343             set_assemble_status(AssembleTask.DIF, project.id, checksum, ChunkFileState.CREATED)
 344 
 345             from sentry.tasks.assemble import assemble_dif
 346 
 347             assemble_dif.apply_async(
 348                 kwargs={
 349                     "project_id": project.id,
 350                     "name": name,
 351                     "debug_id": debug_id,
 352                     "checksum": checksum,
 353                     "chunks": chunks,
 354                 }
 355             )
 356 
 357             file_response[checksum] = {"state": ChunkFileState.CREATED, "missingChunks": []}
 358 
 359         return Response(file_response, status=200)

 362 class SourceMapsEndpoint(ProjectEndpoint):
 363     # doc_section = DocSection.PROJECTS
 364     permission_classes = (ProjectReleasePermission,)
 365 
 366     def get(self, request, project):
 367         """
 368         List a Project's Source Map Archives
 369         ````````````````````````````````````
 370 
 371         Retrieve a list of source map archives (releases, later bundles) for a given project.
 372 
 373         :pparam string organization_slug: the slug of the organization the
 374                                           source map archive belongs to.
 375         :pparam string project_slug: the slug of the project to list the
 376                                      source map archives of.
 377         :qparam string query: If set, this parameter is used to locate source map archives with.
 378         :auth: required
 379         """
 380         query = request.GET.get("query")
 381 
 382         try:
 383             queryset = (
 384                 Release.objects.filter(projects=project, organization_id=project.organization_id)
 385                 .annotate(
 386                     has_file=Exists(
 387                         ReleaseFile.objects.filter(release=OuterRef("id")).values_list("id")
 388                     )
 389                 )
 390                 .values("id", "version", "date_added")
 391                 .filter(has_file=True)
 392             )
 393         except Release.DoesNotExist:
 394             raise ResourceDoesNotExist
 395 
 396         if query:
 397             query_q = Q(version__icontains=query)
 398 
 399             suffix_match = _release_suffix.match(query)
 400             if suffix_match is not None:
 401                 query_q |= Q(version__icontains="%s+%s" % suffix_match.groups())
 402 
 403             queryset = queryset.filter(query_q)
 404 
 405         def expose_release(release, count):
 406             return {
 407                 "type": "release",
 408                 "id": release["id"],
 409                 "name": release["version"],
 410                 "date": release["date_added"],
 411                 "fileCount": count,
 412             }
 413 
 414         def serialize_results(results):
 415             file_counts = (
 416                 Release.objects.filter(id__in=[r["id"] for r in results])
 417                 .annotate(count=Count("releasefile"))
 418                 .values("count", "id")
 419             )
 420             file_count_map = {r["id"]: r["count"] for r in file_counts}
 421             return serialize(
 422                 [expose_release(r, file_count_map[r["id"]]) for r in results], request.user
 423             )
 424 
 425         return self.paginate(
 426             request=request,
 427             queryset=queryset,
 428             order_by="-date_added",
 429             paginator_cls=OffsetPaginator,
 430             default_per_page=10,
 431             on_results=serialize_results,
 432         )

 434     def delete(self, request, project):
 435         """
 436         Delete an Archive
 437         ```````````````````````````````````````````````````
 438 
 439         Delete all artifacts inside given archive.
 440 
 441         :pparam string organization_slug: the slug of the organization the
 442                                             archive belongs to.
 443         :pparam string project_slug: the slug of the project to delete the
 444                                         archive of.
 445         :qparam string id: The id of the archive to delete.
 446         :auth: required
 447         """
 448 
 449         archive_id = request.GET.get("id")
 450 
 451         if archive_id:
 452             with transaction.atomic():
 453                 release = Release.objects.get(
 454                     organization_id=project.organization_id, projects=project, id=archive_id
 455                 )
 456                 if release is not None:
 457                     release_files = ReleaseFile.objects.filter(release=release)
 458                     release_files.delete()
 459                     return Response(status=204)
 460 
 461         return Response(status=404)
