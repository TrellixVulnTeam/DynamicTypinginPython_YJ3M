  55 class TeamSerializer(serializers.Serializer):
  56     name = serializers.CharField(max_length=64, required=False, allow_null=True, allow_blank=True)
  57     slug = serializers.RegexField(
  58         r"^[a-z0-9_\-]+$",
  59         max_length=50,
  60         required=False,
  61         allow_null=True,
  62         error_messages={
  63             "invalid": _(
  64                 "Enter a valid slug consisting of lowercase letters, "
  65                 "numbers, underscores or hyphens."
  66             )
  67         },
  68     )
  69 
  70     def validate(self, attrs):
  71         if not (attrs.get("name") or attrs.get("slug")):
  72             raise serializers.ValidationError("Name or slug is required")
  73         return attrs

  80     @attach_scenarios([list_organization_teams_scenario])
  81     def get(self, request, organization):
  82         """
  83         List an Organization's Teams
  84         ````````````````````````````
  85 
  86         Return a list of teams bound to a organization.
  87 
  88         :pparam string organization_slug: the slug of the organization for
  89                                           which the teams should be listed.
  90         :param string detailed:      Specify "0" to return team details that do not include projects
  91         :param string is_not_member: Specify "1" to *only* return team details of teams that user is not a member of
  92         :auth: required
  93         """
  94         # TODO(dcramer): this should be system-wide default for organization
  95         # based endpoints
  96         if request.auth and hasattr(request.auth, "project"):
  97             return Response(status=403)
  98 
  99         with sentry_sdk.start_span(op="PERF: OrgTeam.get - filter"):
 100             queryset = Team.objects.filter(
 101                 organization=organization, status=TeamStatus.VISIBLE
 102             ).order_by("slug")
 103 
 104         if request.GET.get("is_not_member", "0") == "1":
 105             user_teams = Team.objects.get_for_user(organization=organization, user=request.user)
 106             queryset = queryset.exclude(id__in=[ut.id for ut in user_teams])
 107 
 108         query = request.GET.get("query")
 109 
 110         with sentry_sdk.start_span(op="PERF: OrgTeam.get - tokenize"):
 111             if query:
 112                 tokens = tokenize_query(query)
 113                 for key, value in six.iteritems(tokens):
 114                     if key == "query":
 115                         value = " ".join(value)
 116                         queryset = queryset.filter(
 117                             Q(name__icontains=value) | Q(slug__icontains=value)
 118                         )
 119                     else:
 120                         queryset = queryset.none()
 121 
 122         is_detailed = request.GET.get("detailed", "1") != "0"
 123 
 124         with sentry_sdk.start_span(op="PERF: OrgTeam.get - serialize"):
 125             serializer = (
 126                 team_serializers.TeamWithProjectsSerializer
 127                 if is_detailed
 128                 else team_serializers.TeamSerializer
 129             )
 130 
 131         return self.paginate(
 132             request=request,
 133             queryset=queryset,
 134             order_by="slug",
 135             on_results=lambda x: serialize(x, request.user, serializer()),
 136             paginator_cls=OffsetPaginator,
 137         )

 139     @attach_scenarios([create_new_team_scenario])
 140     def post(self, request, organization):
 141         """
 142         Create a new Team
 143         ``````````````````
 144 
 145         Create a new team bound to an organization.  Only the name of the
 146         team is needed to create it, the slug can be auto generated.
 147 
 148         :pparam string organization_slug: the slug of the organization the
 149                                           team should be created for.
 150         :param string name: the optional name of the team.
 151         :param string slug: the optional slug for this team.  If
 152                             not provided it will be auto generated from the
 153                             name.
 154         :auth: required
 155         """
 156         serializer = TeamSerializer(data=request.data)
 157 
 158         if serializer.is_valid():
 159             result = serializer.validated_data
 160 
 161             try:
 162                 with transaction.atomic():
 163                     team = Team.objects.create(
 164                         name=result.get("name") or result["slug"],
 165                         slug=result.get("slug"),
 166                         organization=organization,
 167                     )
 168             except IntegrityError:
 169                 return Response(
 170                     {
 171                         "non_field_errors": [CONFLICTING_SLUG_ERROR],
 172                         "detail": CONFLICTING_SLUG_ERROR,
 173                     },
 174                     status=409,
 175                 )
 176             else:
 177                 team_created.send_robust(
 178                     organization=organization, user=request.user, team=team, sender=self.__class__
 179                 )
 180 
 181             if request.user.is_authenticated():
 182                 try:
 183                     member = OrganizationMember.objects.get(
 184                         user=request.user, organization=organization
 185                     )
 186                 except OrganizationMember.DoesNotExist:
 187                     pass
 188                 else:
 189                     OrganizationMemberTeam.objects.create(team=team, organizationmember=member)
 190 
 191             self.create_audit_entry(
 192                 request=request,
 193                 organization=organization,
 194                 target_object=team.id,
 195                 event=AuditLogEntryEvent.TEAM_ADD,
 196                 data=team.get_audit_log_data(),
 197             )
 198 
 199             return Response(serialize(team, request.user), status=201)
 200         return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
