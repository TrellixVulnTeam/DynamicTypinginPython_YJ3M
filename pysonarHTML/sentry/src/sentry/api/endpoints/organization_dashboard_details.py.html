  18 def remove_widgets(dashboard_widgets, widget_data):
  19     """
  20     Removes current widgets belonging to dashboard not in widget_data.
  21     Returns remaining widgets.
  22     """
  23     widget_ids = [wd["id"] for wd in widget_data]
  24     dashboard_widgets.exclude(id__in=widget_ids).delete()
  25     return dashboard_widgets.filter(id__in=widget_ids)

  28 def reorder_widgets(dashboard_id, widget_data):
  29     """
  30     Reorders Widgets given the relative order desired,
  31     reorders widgets in the next possible set of numbers
  32     i.e if order of widgets is 1, 2, 3
  33         the reordered widgets will have order 4, 5, 6
  34     """
  35     dashboard_widgets = Widget.objects.filter(dashboard_id=dashboard_id)
  36     dashboard_widgets = list(remove_widgets(dashboard_widgets, widget_data))
  37 
  38     # dashboard_widgets and widget_data should now have the same widgets
  39     widget_data.sort(key=lambda x: x["order"])
  40 
  41     next_order = get_next_dashboard_order(dashboard_id)
  42     for index, data in enumerate(widget_data):
  43         for widget in dashboard_widgets:
  44             if widget.id == data["id"]:
  45                 widget.order = next_order + index
  46                 widget.save()
  47                 break

  55 class DashboardWithWidgetsSerializer(serializers.Serializer):
  56     title = serializers.CharField(required=False, allow_blank=True, allow_null=True)
  57     widgets = ListField(child=WidgetSerializer(), required=False, allow_null=True)
  58 
  59     def validate_widgets(self, widgets):
  60         if len(widgets) != len(set([w["order"] for w in widgets])):
  61             raise ValidationError("Widgets must not have duplicate order values.")
  62 
  63         widgets_count = Widget.objects.filter(
  64             id__in=[w["id"] for w in widgets],
  65             dashboard_id=self.context["dashboard_id"],
  66             status=ObjectStatus.VISIBLE,
  67         ).count()
  68 
  69         if len(widgets) != widgets_count:
  70             raise ValidationError(
  71                 "All widgets must exist within this dashboard prior to reordering."
  72             )
  73 
  74         return widgets

  98         Delete an Organization's Dashboard

 114     def put(self, request, organization, dashboard):
 115         """
 116         Edit an Organization's Dashboard
 117         ```````````````````````````````````
 118 
 119         Edit an individual organization's dashboard as well as
 120         bulk edits on widgets (i.e. rearranging widget order).
 121 
 122         :pparam string organization_slug: the slug of the organization the
 123                                           dashboard belongs to.
 124         :pparam int dashboard_id: the id of the dashboard.
 125         :param array widgets: the array of widgets (consisting of a widget id and the order)
 126                             to be updated.
 127         :auth: required
 128         """
 129         serializer = DashboardWithWidgetsSerializer(
 130             data=request.data, context={"dashboard_id": dashboard.id}
 131         )
 132 
 133         if not serializer.is_valid():
 134             return Response(serializer.errors, status=400)
 135         data = serializer.validated_data
 136         try:
 137             with transaction.atomic():
 138                 title = data.get("title")
 139                 if title:
 140                     dashboard.update(title=data["title"])
 141 
 142                 widgets = data.get("widgets")
 143                 if widgets:
 144                     reorder_widgets(dashboard.id, widgets)
 145         except IntegrityError:
 146             return self.respond({"Dashboard with that title already exists"}, status=409)
 147 
 148         return self.respond(serialize(dashboard, request.user), status=200)
