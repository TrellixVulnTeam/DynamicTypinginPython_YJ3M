  41             import sys
  42

  45             del exc_tb
  46 
  47         # XXX(dcramer): we experienced an issue where the identity still existed,
  48         # and given that this is a cheap query, lets error hard in that case
  49         assert not UserSocialAuth.objects.filter(user=user, id=identity_id).exists()
  50 
  51         logger.info(
  52             "user.identity.disconnect",

  54                 "user_id": user.id,
