  11 class OrganizationAuthProviderDetailsEndpoint(OrganizationEndpoint):
  12     permission_classes = (OrganizationAuthProviderPermission,)

  14     def get(self, request, organization):
  15         """
  16         Retrieve details about Organization's SSO settings and
  17         currently installed auth_provider
  18         ``````````````````````````````````````````````````````
  19 
  20         :pparam string organization_slug: the organization short name
  21         :auth: required
  22         """
  23         try:
  24             auth_provider = AuthProvider.objects.get(organization=organization)
  25         except AuthProvider.DoesNotExist:
  26             # This is a valid state where org does not have an auth provider
  27             # configured, make sure we respond with a 20x
  28             return Response(status=status.HTTP_204_NO_CONTENT)
  29 
  30         # cache organization so that we don't need to query for org when serializing
  31         auth_provider._organization_cache = organization
  32 
  33         return Response(serialize(auth_provider, request.user))
