  78 def clean_newline_inputs(value, case_insensitive=True):
  79     result = []
  80     for v in value.split("\n"):
  81         if case_insensitive:
  82             v = v.lower()
  83         v = v.strip()
  84         if v:
  85             result.append(v)
  86     return result

 132     def validate(self, data):
 133         max_delay = (
 134             data["digestsMaxDelay"]
 135             if "digestsMaxDelay" in data
 136             else self.context["project"].get_option("digests:mail:maximum_delay")
 137         )
 138         min_delay = (
 139             data["digestsMinDelay"]
 140             if "digestsMinDelay" in data
 141             else self.context["project"].get_option("digests:mail:minimum_delay")
 142         )
 143 
 144         if min_delay is not None and max_delay and max_delay is not None and min_delay > max_delay:
 145             raise serializers.ValidationError(
 146                 {"digestsMinDelay": "The minimum delay on digests must be lower than the maximum."}
 147             )
 148 
 149         return data

 159     def validate_slug(self, slug):
 160         if slug in RESERVED_PROJECT_SLUGS:
 161             raise serializers.ValidationError(
 162                 'The slug "%s" is reserved and not allowed.' % (slug,)
 163             )
 164         project = self.context["project"]
 165         other = (
 166             Project.objects.filter(slug=slug, organization=project.organization)
 167             .exclude(id=project.id)
 168             .first()
 169         )
 170         if other is not None:
 171             raise serializers.ValidationError(
 172                 "Another project (%s) is already using that slug" % other.name
 173             )
 174         return slug

 195     def validate_symbolSources(self, sources_json):
 196         if not sources_json:
 197             return sources_json
 198 
 199         from sentry import features
 200 
 201         organization = self.context["project"].organization
 202         request = self.context["request"]
 203         has_sources = features.has(
 204             "organizations:custom-symbol-sources", organization, actor=request.user
 205         )
 206 
 207         if not has_sources:
 208             raise serializers.ValidationError(
 209                 "Organization is not allowed to set custom symbol sources"

 242     def validate_copy_from_project(self, other_project_id):
 243         try:
 244             other_project = Project.objects.filter(
 245                 id=other_project_id, organization_id=self.context["project"].organization_id
 246             ).prefetch_related("teams")[0]
 247         except IndexError:
 248             raise serializers.ValidationError("Project to copy settings from not found.")
 249 
 250         request = self.context["request"]
 251         if not request.access.has_project_access(other_project):
 252             raise serializers.ValidationError(
 253                 "Project settings cannot be copied from a project you do not have access to."
 254             )
 255 
 256         for project_team in other_project.projectteam_set.all():
 257             if not request.access.has_team_scope(project_team.team, "team:write"):
 258                 raise serializers.ValidationError(
 259                     "Project settings cannot be copied from a project with a team you do not have write access to."
 260                 )
 261 
 262         return other_project_id

 284     def _get_unresolved_count(self, project):
 285         queryset = Group.objects.filter(status=GroupStatus.UNRESOLVED, project=project)
 286 
 287         resolve_age = project.get_option("sentry:resolve_age", None)
 288         if resolve_age:
 289             queryset = queryset.filter(
 290                 last_seen__gte=timezone.now() - timedelta(hours=int(resolve_age))
 291             )
 292 
 293         return queryset.count()

 316     @attach_scenarios([update_project_scenario])
 317     def put(self, request, project):
 318         """
 319         Update a Project
 320         ````````````````
 321 
 322         Update various attributes and configurable settings for the given
 323         project.  Only supplied values are updated.
 324 
 325         :pparam string organization_slug: the slug of the organization the
 326                                           project belongs to.
 327         :pparam string project_slug: the slug of the project to update.
 328         :param string name: the new name for the project.
 329         :param string slug: the new slug for the project.
 330         :param string team: the slug of new team for the project. Note, will be deprecated
 331                             soon when multiple teams can have access to a project.
 332         :param string platform: the new platform for the project.
 333         :param boolean isBookmarked: in case this API call is invoked with a
 334                                      user context this allows changing of
 335                                      the bookmark flag.
 336         :param int digestsMinDelay:
 337         :param int digestsMaxDelay:
 338         :auth: required
 339         """
 340         has_project_write = (request.auth and request.auth.has_scope("project:write")) or (
 341             request.access and request.access.has_scope("project:write")
 342         )
 343 
 344         changed_proj_settings = {}
 345 
 346         if has_project_write:
 347             serializer_cls = ProjectAdminSerializer
 348         else:
 349             serializer_cls = ProjectMemberSerializer
 350 
 351         serializer = serializer_cls(
 352             data=request.data, partial=True, context={"project": project, "request": request}
 353         )
 354         if not serializer.is_valid():
 355             return Response(serializer.errors, status=400)
 356 
 357         result = serializer.validated_data
 358 
 359         if not has_project_write:
 360             # options isn't part of the serializer, but should not be editable by members
 361             for key in chain(six.iterkeys(ProjectAdminSerializer().fields), ["options"]):
 362                 if request.data.get(key) and not result.get(key):
 363                     return Response(
 364                         {"detail": ["You do not have permission to perform this action."]},
 365                         status=403,
 366                     )
 367 
 368         changed = False
 369 
 370         old_slug = None
 371         if result.get("slug"):
 372             old_slug = project.slug
 373             project.slug = result["slug"]
 374             changed = True
 375             changed_proj_settings["new_slug"] = project.slug
 376 
 377         if result.get("name"):
 378             project.name = result["name"]
 379             changed = True
 380             changed_proj_settings["new_project"] = project.name
 381 
 382         old_team_id = None
 383         new_team = None
 384         if result.get("team"):
 385             return Response(
 386                 {"detail": ["Editing a team via this endpoint has been deprecated."]}, status=400
 387             )
 388 
 389         if result.get("platform"):
 390             project.platform = result["platform"]
 391             changed = True
 392 
 393         if changed:
 394             project.save()
 395             if old_team_id is not None:
 396                 ProjectTeam.objects.filter(project=project, team_id=old_team_id).update(
 397                     team=new_team
 398                 )
 399 
 400             if old_slug:
 401                 ProjectRedirect.record(project, old_slug)
 402 
 403         if result.get("isBookmarked"):
 404             try:
 405                 with transaction.atomic():
 406                     ProjectBookmark.objects.create(project_id=project.id, user=request.user)
 407             except IntegrityError:
 408                 pass
 409         elif result.get("isBookmarked") is False:
 410             ProjectBookmark.objects.filter(project_id=project.id, user=request.user).delete()
 411 
 412         if result.get("digestsMinDelay"):
 413             project.update_option("digests:mail:minimum_delay", result["digestsMinDelay"])
 414         if result.get("digestsMaxDelay"):
 415             project.update_option("digests:mail:maximum_delay", result["digestsMaxDelay"])
 416         if result.get("subjectPrefix") is not None:
 417             if project.update_option("mail:subject_prefix", result["subjectPrefix"]):
 418                 changed_proj_settings["mail:subject_prefix"] = result["subjectPrefix"]
 419         if result.get("subjectTemplate"):
 420             project.update_option("mail:subject_template", result["subjectTemplate"])
 421         if result.get("scrubIPAddresses") is not None:
 422             if project.update_option("sentry:scrub_ip_address", result["scrubIPAddresses"]):
 423                 changed_proj_settings["sentry:scrub_ip_address"] = result["scrubIPAddresses"]
 424         if result.get("groupingConfig") is not None:
 425             if project.update_option("sentry:grouping_config", result["groupingConfig"]):
 426                 changed_proj_settings["sentry:grouping_config"] = result["groupingConfig"]
 427         if result.get("groupingEnhancements") is not None:
 428             if project.update_option(
 429                 "sentry:grouping_enhancements", result["groupingEnhancements"]
 430             ):
 431                 changed_proj_settings["sentry:grouping_enhancements"] = result[
 432                     "groupingEnhancements"
 433                 ]
 434         if result.get("groupingEnhancementsBase") is not None:
 435             if project.update_option(
 436                 "sentry:grouping_enhancements_base", result["groupingEnhancementsBase"]
 437             ):
 438                 changed_proj_settings["sentry:grouping_enhancements_base"] = result[
 439                     "groupingEnhancementsBase"
 440                 ]
 441         if result.get("fingerprintingRules") is not None:
 442             if project.update_option("sentry:fingerprinting_rules", result["fingerprintingRules"]):
 443                 changed_proj_settings["sentry:fingerprinting_rules"] = result["fingerprintingRules"]
 444         if result.get("securityToken") is not None:
 445             if project.update_option("sentry:token", result["securityToken"]):
 446                 changed_proj_settings["sentry:token"] = result["securityToken"]
 447         if result.get("securityTokenHeader") is not None:
 448             if project.update_option("sentry:token_header", result["securityTokenHeader"]):
 449                 changed_proj_settings["sentry:token_header"] = result["securityTokenHeader"]
 450         if result.get("verifySSL") is not None:
 451             if project.update_option("sentry:verify_ssl", result["verifySSL"]):
 452                 changed_proj_settings["sentry:verify_ssl"] = result["verifySSL"]
 453         if result.get("dataScrubber") is not None:
 454             if project.update_option("sentry:scrub_data", result["dataScrubber"]):
 455                 changed_proj_settings["sentry:scrub_data"] = result["dataScrubber"]

 508                 user=request.user, key="mail:alert", value=0, project=project

 512         if has_project_write:

 514             if "sentry:origins" in options:
 515                 project.update_option(
 516                     "sentry:origins", clean_newline_inputs(options["sentry:origins"])
 517                 )
 518             if "sentry:resolve_age" in options:
 519                 project.update_option("sentry:resolve_age", int(options["sentry:resolve_age"]))
 520             if "sentry:scrub_data" in options:
 521                 project.update_option("sentry:scrub_data", bool(options["sentry:scrub_data"]))

 626     @attach_scenarios([delete_project_scenario])
 627     @sudo_required
 628     def delete(self, request, project):
 629         """
 630         Delete a Project
 631         ````````````````
 632 
 633         Schedules a project for deletion.
 634 
 635         Deletion happens asynchronously and therefor is not immediate.
 636         However once deletion has begun the state of a project changes and
 637         will be hidden from most public views.
 638 
 639         :pparam string organization_slug: the slug of the organization the
 640                                           project belongs to.
 641         :pparam string project_slug: the slug of the project to delete.
 642         :auth: required
 643         """
 644         if project.is_internal_project():
 645             return Response(
 646                 '{"error": "Cannot remove projects internally used by Sentry."}',
 647                 status=status.HTTP_403_FORBIDDEN,
 648             )
 649 
 650         updated = Project.objects.filter(id=project.id, status=ProjectStatus.VISIBLE).update(
 651             status=ProjectStatus.PENDING_DELETION
 652         )
 653         if updated:
 654             transaction_id = uuid4().hex
 655 
 656             delete_project.apply_async(
 657                 kwargs={"object_id": project.id, "transaction_id": transaction_id}, countdown=3600
 658             )
 659 
 660             self.create_audit_entry(
 661                 request=request,
 662                 organization=project.organization,
 663                 target_object=project.id,
 664                 event=AuditLogEntryEvent.PROJECT_REMOVE,
 665                 data=project.get_audit_log_data(),
 666                 transaction_id=transaction_id,
 667             )
 668 
 669             delete_logger.info(
 670                 "object.delete.queued",
 671                 extra={
 672                     "object_id": project.id,
 673                     "transaction_id": transaction_id,
 674                     "model": type(project).__name__,
 675                 },
 676             )
 677 
 678             project.rename_on_pending_deletion()
 679 
 680         return Response(status=204)
