  17 class ProjectOwnershipSerializer(serializers.Serializer):
  18     raw = serializers.CharField(allow_blank=True)
  19     fallthrough = serializers.BooleanField()
  20     autoAssignment = serializers.BooleanField()
  21 
  22     def validate(self, attrs):
  23         if not attrs.get("raw", "").strip():
  24             return attrs
  25         try:
  26             rules = parse_rules(attrs["raw"])
  27         except ParseError as e:
  28             raise serializers.ValidationError(
  29                 {
  30                     "raw": u"Parse error: %r (line %d, column %d)"
  31                     % (e.expr.name, e.line(), e.column())
  32                 }
  33             )
  34 
  35         schema = dump_schema(rules)
  36 
  37         owners = {o for rule in rules for o in rule.owners}
  38         actors = resolve_actors(owners, self.context["ownership"].project_id)
  39 
  40         bad_actors = []
  41         for owner, actor in six.iteritems(actors):
  42             if actor is None:
  43                 if owner.type == "user":
  44                     bad_actors.append(owner.identifier)
  45                 elif owner.type == "team":
  46                     bad_actors.append(u"#{}".format(owner.identifier))
  47 
  48         if bad_actors:
  49             raise serializers.ValidationError(
  50                 {"raw": u"Invalid rule owners: {}".format(", ".join(bad_actors))}
  51             )
  52 
  53         attrs["schema"] = schema
  54         return attrs
  55 
  56     def save(self):
  57         ownership = self.context["ownership"]
  58 
  59         changed = False
  60         if "raw" in self.validated_data:
  61             raw = self.validated_data["raw"]
  62             if not raw.strip():
  63                 raw = None
  64 
  65             if ownership.raw != raw:
  66                 ownership.raw = raw
  67                 ownership.schema = self.validated_data.get("schema")
  68                 changed = True
  69 
  70         if "fallthrough" in self.validated_data:
  71             fallthrough = self.validated_data["fallthrough"]
  72             if ownership.fallthrough != fallthrough:
  73                 ownership.fallthrough = fallthrough
  74                 changed = True
  75 
  76         changed = self.__modify_auto_assignment(ownership) or changed
  77 
  78         if changed:
  79             now = timezone.now()
  80             if ownership.date_created is None:
  81                 ownership.date_created = now
  82             ownership.last_updated = now
  83             ownership.save()
  84 
  85         return ownership
  86 
  87     def __modify_auto_assignment(self, ownership):
  88         auto_assignment = self.validated_data.get("autoAssignment")
  89 
  90         if auto_assignment is None:
  91             return False
  92 
  93         changed = ownership.auto_assignment != auto_assignment
  94 
  95         if changed:
  96             ownership.auto_assignment = auto_assignment
  97 
  98         return changed

 101 class ProjectOwnershipEndpoint(ProjectEndpoint):
 102     def get_ownership(self, project):
 103         try:
 104             return ProjectOwnership.objects.get(project=project)
 105         except ProjectOwnership.DoesNotExist:
 106             return ProjectOwnership(project=project, date_created=None, last_updated=None)
 107 
 108     def get(self, request, project):
 109         """
 110         Retrieve a Project's Ownership configuration
 111         ````````````````````````````````````````````
 112 
 113         Return details on a project's ownership configuration.
 114 
 115         :auth: required
 116         """
 117         return Response(serialize(self.get_ownership(project), request.user))
 118 
 119     def put(self, request, project):
 120         """
 121         Update a Project's Ownership configuration
 122         ``````````````````````````````````````````
 123 
 124         Updates a project's ownership configuration settings. Only the
 125         attributes submitted are modified.
 126 
 127         :param string raw: Raw input for ownership configuration.
 128         :param boolean fallthrough: Indicate if there is no match on explicit rules,
 129                                     to fall through and make everyone an implicit owner.
 130         :auth: required
 131         """
 132         serializer = ProjectOwnershipSerializer(
 133             data=request.data, partial=True, context={"ownership": self.get_ownership(project)}
 134         )
 135         if serializer.is_valid():
 136             ownership = serializer.save()
 137             ownership_rule_created.send_robust(project=project, sender=self.__class__)
 138             return Response(serialize(ownership, request.user))
 139         return Response(serializer.errors, status=400)
