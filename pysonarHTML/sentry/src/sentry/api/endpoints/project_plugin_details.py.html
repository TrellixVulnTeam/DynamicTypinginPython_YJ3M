  29 class ProjectPluginDetailsEndpoint(ProjectEndpoint):
  30     def _get_plugin(self, plugin_id):
  31         try:
  32             return plugins.get(plugin_id)
  33         except KeyError:
  34             raise ResourceDoesNotExist
  35 
  36     def get(self, request, project, plugin_id):
  37         plugin = self._get_plugin(plugin_id)
  38 
  39         try:
  40             context = serialize(plugin, request.user, PluginWithConfigSerializer(project))
  41         except PluginIdentityRequired as e:
  42             context = serialize(plugin, request.user, PluginSerializer(project))
  43             context["config_error"] = six.text_type(e)
  44             context["auth_url"] = reverse("socialauth_associate", args=[plugin.slug])
  45 
  46         return Response(context)
  47 
  48     def post(self, request, project, plugin_id):
  49         """
  50         Enable plugin, Test plugin or Reset plugin values
  51         """
  52         plugin = self._get_plugin(plugin_id)
  53 
  54         if request.data.get("test") and plugin.is_testable():
  55             test_results = plugin.test_configuration_and_get_test_results(project)
  56             return Response({"detail": test_results}, status=200)
  57 
  58         if request.data.get("reset"):
  59             plugin = self._get_plugin(plugin_id)
  60             plugin.reset_options(project=project)
  61             context = serialize(plugin, request.user, PluginWithConfigSerializer(project))
  62 
  63             self.create_audit_entry(
  64                 request=request,
  65                 organization=project.organization,
  66                 target_object=project.id,
  67                 event=AuditLogEntryEvent.INTEGRATION_EDIT,
  68                 data={"integration": plugin_id, "project": project.slug},
  69             )
  70 
  71             return Response(context, status=200)
  72 
  73         if not plugin.can_disable:
  74             return Response({"detail": ERR_ALWAYS_ENABLED}, status=400)
  75 
  76         plugin.enable(project)
  77 
  78         self.create_audit_entry(
  79             request=request,
  80             organization=project.organization,
  81             target_object=project.id,
  82             event=AuditLogEntryEvent.INTEGRATION_ADD,
  83             data={"integration": plugin_id, "project": project.slug},
  84         )
  85 
  86         return Response(status=201)
  87 
  88     def delete(self, request, project, plugin_id):
  89         """
  90         Disable plugin
  91         """
  92         plugin = self._get_plugin(plugin_id)
  93 
  94         if not plugin.can_disable:
  95             return Response({"detail": ERR_ALWAYS_ENABLED}, status=400)
  96 
  97         plugin.disable(project)
  98 
  99         self.create_audit_entry(
 100             request=request,
 101             organization=project.organization,
 102             target_object=project.id,
 103             event=AuditLogEntryEvent.INTEGRATION_REMOVE,
 104             data={"integration": plugin_id, "project": project.slug},
 105         )
 106 
 107         return Response(status=204)

 109     def put(self, request, project, plugin_id):
 110         plugin = self._get_plugin(plugin_id)
 111 
 112         config = [
 113             serialize_field(project, plugin, c)
 114             for c in plugin.get_config(project=project, user=request.user, initial=request.data)
 115         ]
 116 
 117         cleaned = {}
 118         errors = {}
 119         for field in config:
 120             key = field["name"]
 121             value = request.data.get(key)
 122 
 123             if field.get("required") and not value:
 124                 errors[key] = ERR_FIELD_REQUIRED
 125 
 126             try:
 127                 value = plugin.validate_config_field(
 128                     project=project, name=key, value=value, actor=request.user
 129                 )
 130             except (
 131                 forms.ValidationError,
 132                 serializers.ValidationError,
 133                 InvalidIdentity,
 134                 PluginError,
 135             ) as e:
 136                 errors[key] = six.text_type(e)
 137 
 138             if not errors.get(key):
 139                 cleaned[key] = value
 140 
 141         if not errors:
 142             try:
 143                 cleaned = plugin.validate_config(
 144                     project=project, config=cleaned, actor=request.user
 145                 )
 146             except (InvalidIdentity, PluginError) as e:
 147                 errors["__all__"] = six.text_type(e)
 148 
 149         if errors:
 150             return Response({"errors": errors}, status=400)
 151 
 152         for key, value in six.iteritems(cleaned):
 153             if value is None:
 154                 plugin.unset_option(project=project, key=key)
 155             else:
 156                 plugin.set_option(project=project, key=key, value=value)
 157 
 158         context = serialize(plugin, request.user, PluginWithConfigSerializer(project))
 159 
 160         plugin_enabled.send(plugin=plugin, project=project, user=request.user, sender=self)
 161 
 162         self.create_audit_entry(
 163             request=request,
 164             organization=project.organization,
 165             target_object=project.id,
 166             event=AuditLogEntryEvent.INTEGRATION_EDIT,
 167             data={"integration": plugin_id, "project": project.slug},
 168         )
 169 
 170         return Response(context)
