  20 class GroupTagKeyValuesEndpoint(GroupEndpoint, EnvironmentMixin):
  21     doc_section = DocSection.EVENTS

  23     # XXX: this scenario does not work for some inexplicable reasons
  24     # @attach_scenarios([list_tag_values_scenario])
  25     def get(self, request, group, key):
  26         """
  27         List a Tag's Values
  28         ```````````````````
  29 
  30         Return a list of values associated with this key for an issue.
  31 
  32         :pparam string issue_id: the ID of the issue to retrieve.
  33         :pparam string key: the tag key to look the values up for.
  34         :auth: required
  35         """
  36         lookup_key = tagstore.prefix_reserved_key(key)
  37 
  38         environment_ids = [e.id for e in get_environments(request, group.project.organization)]
  39 
  40         try:
  41             tagstore.get_tag_key(group.project_id, None, lookup_key)
  42         except tagstore.TagKeyNotFound:
  43             raise ResourceDoesNotExist
  44 
  45         sort = request.GET.get("sort")
  46         if sort == "date":
  47             order_by = "-last_seen"
  48         elif sort == "age":
  49             order_by = "-first_seen"
  50         else:
  51             order_by = "-id"
  52 
  53         if key == "user":
  54             serializer_cls = UserTagValueSerializer(group.project_id)
  55         else:
  56             serializer_cls = None
  57 
  58         paginator = tagstore.get_group_tag_value_paginator(
  59             group.project_id, group.id, environment_ids, lookup_key, order_by=order_by
  60         )
  61 
  62         return self.paginate(
  63             request=request,
  64             paginator=paginator,
  65             on_results=lambda results: serialize(results, request.user, serializer_cls),
  66         )
