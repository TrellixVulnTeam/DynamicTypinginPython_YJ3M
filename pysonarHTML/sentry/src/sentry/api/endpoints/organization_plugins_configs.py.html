  13 class OrganizationPluginsConfigsEndpoint(OrganizationEndpoint):
  14     def get(self, request, organization):
  15 
  16         """
  17         List one or more plugin configurations, including a `projectList` for each plugin which contains
  18         all the projects that have that specific plugin both configured and enabled.
  19 
  20         - similar to the `OrganizationPluginsEndpoint`, and can eventually replace it
  21 
  22         :qparam plugins array[string]: an optional list of plugin ids (slugs) if you want specific plugins.
  23                                     If not set, will return configurations for all plugins.
  24         """
  25 
  26         desired_plugins = []
  27         for slug in request.GET.getlist("plugins") or ():
  28             # if the user request a plugin that doesn't exist, throw 404
  29             try:
  30                 desired_plugins.append(plugins.get(slug))
  31             except KeyError:
  32                 return Response({"detail": "Plugin %s not found" % slug}, status=404)
  33 
  34         # if no plugins were specified, grab all plugins but limit by those that have the ability to be configured
  35         if not desired_plugins:
  36             desired_plugins = list(plugins.plugin_that_can_be_configured())
  37 
  38         # `keys_to_check` are the ProjectOption keys that tell us if a plugin is enabled (e.g. `plugin:enabled`) or are
  39         # configured properly, meaning they have the required information - plugin.required_field - needed for the
  40         # plugin to work (ex:`opsgenie:api_key`)
  41         keys_to_check = []
  42         for plugin in desired_plugins:
  43             keys_to_check.append("%s:enabled" % plugin.slug)
  44             if plugin.required_field:
  45                 keys_to_check.append("%s:%s" % (plugin.slug, plugin.required_field))
  46 
  47         # Get all the project options for org that have truthy values
  48         project_options = ProjectOption.objects.filter(
  49             key__in=keys_to_check, project__organization=organization
  50         ).exclude(value__in=[False, ""])
  51 
  52         """
  53         This map stores info about whether a plugin is configured and/or enabled
  54         {
  55             "plugin_slug": {
  56                 "project_id": { "enabled": True, "configured": False },
  57             },
  58         }
  59         """
  60         info_by_plugin_project = {}
  61         for project_option in project_options:
  62             [slug, field] = project_option.key.split(":")
  63             project_id = project_option.project_id
  64 
  65             # first add to the set of all projects by plugin
  66             info_by_plugin_project.setdefault(slug, {}).setdefault(
  67                 project_id, {"enabled": False, "configured": False}
  68             )
  69 
  70             # next check if enabled
  71             if field == "enabled":
  72                 info_by_plugin_project[slug][project_id]["enabled"] = True
  73             # if the projectoption is not the enable field, it's configuration field
  74             else:
  75                 info_by_plugin_project[slug][project_id]["configured"] = True
  76 
  77         # get the IDs of all projects for found project options and grab them from the DB
  78         project_id_set = set([project_option.project_id for project_option in project_options])
  79         projects = Project.objects.filter(id__in=project_id_set)
  80 
  81         # create a key/value map of our projects
  82         project_map = {project.id: project for project in projects}
  83 
  84         # iterate through the desired plugins and serialize them
  85         serialized_plugins = []
  86         for plugin in desired_plugins:
  87             serialized_plugin = serialize(plugin, request.user, PluginSerializer())
  88 
  89             serialized_plugin["projectList"] = []
  90 
  91             info_by_project = info_by_plugin_project.get(plugin.slug, {})
  92 
  93             # iterate through the projects
  94             for project_id, plugin_info in six.iteritems(info_by_project):
  95                 project = project_map[project_id]
  96 
  97                 # only include plugins which are configured
  98                 if not plugin_info["configured"]:
  99                     continue
 100 
 101                 serialized_plugin["projectList"].append(
 102                     {
 103                         "projectId": project.id,
 104                         "projectSlug": project.slug,
 105                         "projectName": project.name,  # TODO(steve): do we need?
 106                         "enabled": plugin_info["enabled"],
 107                         "configured": plugin_info["configured"],  # TODO(steve): do we need?
 108                         "projectPlatform": project.platform,
 109                     }
 110                 )
 111             # sort by the projectSlug
 112             serialized_plugin["projectList"].sort(key=lambda x: x["projectSlug"])
 113             serialized_plugins.append(serialized_plugin)
 114 
 115         return Response(serialized_plugins)
