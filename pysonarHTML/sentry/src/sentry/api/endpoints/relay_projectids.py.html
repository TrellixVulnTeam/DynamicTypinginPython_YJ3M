  11 class RelayProjectIdsEndpoint(Endpoint):
  12     authentication_classes = (RelayAuthentication,)
  13     permission_classes = (RelayPermission,)
  14 
  15     def post(self, request):
  16         relay = request.relay
  17         assert relay is not None  # should be provided during Authentication
  18 
  19         project_ids = {}
  20         for public_key in request.relay_request_data.get("publicKeys") or ():
  21             if not ProjectKey.looks_like_api_key(public_key):
  22                 continue
  23 
  24             try:
  25                 pk = ProjectKey.objects.get_from_cache(public_key=public_key)
  26             except ProjectKey.DoesNotExist:
  27                 continue
  28 
  29             # NB: Do not validate pk here (is_active or store). Relay should
  30             # also receive a mapping for disabled public keys and then perform
  31             # the full project config fetch.
  32 
  33             project_ids[public_key] = pk.project_id
  34 
  35         return Response({"projectIds": project_ids}, status=200)
