  28 def _get_timezone_choices():
  29     results = []
  30     for tz in pytz.common_timezones:
  31         now = datetime.now(pytz.timezone(tz))
  32         offset = now.strftime("%z")
  33         results.append((int(offset), tz, "(UTC%s) %s" % (offset, tz)))
  34     results.sort()
  35 
  36     for i in range(len(results)):
  37         results[i] = results[i][1:]
  38     return results

  58 class BaseUserSerializer(serializers.ModelSerializer):
  59     def validate_username(self, value):
  60         if User.objects.filter(username__iexact=value).exclude(id=self.instance.id).exists():
  61             raise serializers.ValidationError("That username is already in use.")
  62         return value
  63 
  64     def validate(self, attrs):
  65         attrs = super(BaseUserSerializer, self).validate(attrs)
  66 
  67         if self.instance.email == self.instance.username:
  68             if attrs.get("username", self.instance.email) != self.instance.email:
  69                 # ... this probably needs to handle newsletters and such?
  70                 attrs.setdefault("email", attrs["username"])
  71 
  72         return attrs
  73 
  74     def update(self, instance, validated_data):
  75         if "isActive" not in validated_data:
  76             validated_data["isActive"] = instance.is_active
  77         return super(BaseUserSerializer, self).update(instance, validated_data)

 109 class UserDetailsEndpoint(UserEndpoint):
 110     def get(self, request, user):
 111         """
 112         Retrieve User Details
 113         `````````````````````
 114 
 115         Return details for an account's details and options such as: full name, timezone, 24hr times, language,
 116         stacktrace_order.
 117 
 118         :auth: required
 119         """
 120         return Response(serialize(user, request.user, DetailedUserSerializer()))
 121 
 122     def put(self, request, user):
 123         """
 124         Update Account Appearance options
 125         `````````````````````````````````
 126 
 127         Update account appearance options. Only supplied values are updated.
 128 
 129         :pparam string user_id: user id
 130         :param string language: language preference
 131         :param string stacktrace_order: One of -1 (default), 1 (most recent call last), 2 (most recent call first).
 132         :param string timezone: timezone option
 133         :param clock_24_hours boolean: use 24 hour clock
 134         :auth: required
 135         """
 136 
 137         if is_active_superuser(request):
 138             serializer_cls = SuperuserUserSerializer
 139         else:
 140             serializer_cls = UserSerializer
 141         serializer = serializer_cls(user, data=request.data, partial=True)
 142 
 143         serializer_options = UserOptionsSerializer(
 144             data=request.data.get("options", {}), partial=True
 145         )
 146 
 147         # This serializer should NOT include privileged fields e.g. password
 148         if not serializer.is_valid() or not serializer_options.is_valid():
 149             return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
 150 
 151         # map API keys to keys in model
 152         key_map = {
 153             "language": "language",
 154             "timezone": "timezone",
 155             "stacktraceOrder": "stacktrace_order",
 156             "clock24Hours": "clock_24_hours",
 157         }
 158 
 159         options_result = serializer_options.validated_data
 160 
 161         for key in key_map:
 162             if key in options_result:
 163                 UserOption.objects.set_value(
 164                     user=user, key=key_map.get(key, key), value=options_result.get(key)
 165                 )
 166 
 167         user = serializer.save()
 168 
 169         return Response(serialize(user, request.user, DetailedUserSerializer()))

 171     @sudo_required
 172     def delete(self, request, user):
 173         """
 174         Delete User Account
 175 
 176         Also removes organizations if they are an owner
 177         :pparam string user_id: user id
 178         :param boolean hard_delete: Completely remove the user from the database (requires super user)
 179         :param list organizations: List of organization ids to remove
 180         :auth required:
 181         """
 182 
 183         serializer = DeleteUserSerializer(data=request.data)
 184 
 185         if not serializer.is_valid():
 186             return Response(status=status.HTTP_400_BAD_REQUEST)
 187 
 188         # from `frontend/remove_account.py`
 189         org_list = Organization.objects.filter(
 190             member_set__role__in=[x.id for x in roles.with_scope("org:admin")],
 191             member_set__user=user,
 192             status=OrganizationStatus.VISIBLE,
 193         )
 194 
 195         org_results = []
 196         for org in org_list:
 197             org_results.append({"organization": org, "single_owner": org.has_single_owner()})
 198 
 199         avail_org_slugs = set([o["organization"].slug for o in org_results])
 200         orgs_to_remove = set(serializer.validated_data.get("organizations")).intersection(
 201             avail_org_slugs
 202         )
 203 
 204         for result in org_results:
 205             if result["single_owner"]:
 206                 orgs_to_remove.add(result["organization"].slug)
 207 
 208         for org_slug in orgs_to_remove:
 209             client.delete(
 210                 path=u"/organizations/{}/".format(org_slug), request=request, is_sudo=True
 211             )
 212 
 213         remaining_org_ids = [
 214             o.id for o in org_list if o.slug in avail_org_slugs.difference(orgs_to_remove)
 215         ]
 216 
 217         if remaining_org_ids:
 218             OrganizationMember.objects.filter(
 219                 organization__in=remaining_org_ids, user=user
 220             ).delete()
 221 
 222         logging_data = {"actor_id": request.user.id, "ip_address": request.META["REMOTE_ADDR"]}
 223 
 224         hard_delete = serializer.validated_data.get("hardDelete", False)
 225 
 226         # Only active superusers can hard delete accounts
 227         if hard_delete and not is_active_superuser(request):
 228             return Response(
 229                 {"detail": "Only superusers may hard delete a user account"},
 230                 status=status.HTTP_403_FORBIDDEN,
 231             )
 232 
 233         is_current_user = request.user.id == user.id
 234 
 235         if hard_delete:
 236             user.delete()
 237             delete_logger.info("user.removed", extra=logging_data)
 238         else:
 239             User.objects.filter(id=user.id).update(is_active=False)
 240             delete_logger.info("user.deactivate", extra=logging_data)
 241 
 242         # if the user deleted their own account log them out
 243         if is_current_user:
 244             logout(request)
 245 
 246         return Response(status=status.HTTP_204_NO_CONTENT)
