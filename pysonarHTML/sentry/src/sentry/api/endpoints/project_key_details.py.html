  39 class ProjectKeyDetailsEndpoint(ProjectEndpoint):
  40     doc_section = DocSection.PROJECTS

  42     def get(self, request, project, key_id):
  43         try:
  44             key = ProjectKey.objects.get(
  45                 project=project, public_key=key_id, roles=F("roles").bitor(ProjectKey.roles.store)
  46             )
  47         except ProjectKey.DoesNotExist:
  48             raise ResourceDoesNotExist
  49 
  50         return Response(serialize(key, request.user), status=200)

  52     def put(self, request, project, key_id):
  53         """
  54         Update a Client Key
  55         ```````````````````
  56 
  57         Update a client key.  This can be used to rename a key.
  58 
  59         :pparam string organization_slug: the slug of the organization the
  60                                           client keys belong to.
  61         :pparam string project_slug: the slug of the project the client keys
  62                                      belong to.
  63         :pparam string key_id: the ID of the key to update.
  64         :param string name: the new name for the client key.
  65         :auth: required
  66         """
  67         try:
  68             key = ProjectKey.objects.get(
  69                 project=project, public_key=key_id, roles=F("roles").bitor(ProjectKey.roles.store)
  70             )
  71         except ProjectKey.DoesNotExist:
  72             raise ResourceDoesNotExist
  73 
  74         serializer = ProjectKeySerializer(data=request.data, partial=True)
  75         default_version = get_default_sdk_version_for_project(project)
  76 
  77         if serializer.is_valid():
  78             result = serializer.validated_data
  79 
  80             if result.get("name"):
  81                 key.label = result["name"]
  82 
  83             if not result.get("browserSdkVersion"):
  84                 key.data = {"browserSdkVersion": default_version}
  85             else:
  86                 key.data = {"browserSdkVersion": result["browserSdkVersion"]}
  87 
  88             if result.get("isActive") is True:
  89                 key.status = ProjectKeyStatus.ACTIVE
  90             elif result.get("isActive") is False:
  91                 key.status = ProjectKeyStatus.INACTIVE
  92 
  93             if features.has("projects:rate-limits", project):
  94                 ratelimit = result.get("rateLimit", -1)
  95                 if (
  96                     ratelimit is None
  97                     or ratelimit != -1
  98                     and ratelimit
  99                     and (ratelimit["count"] is None or ratelimit["window"] is None)
 100                 ):
 101                     key.rate_limit_count = None
 102                     key.rate_limit_window = None
 103                 elif result.get("rateLimit"):
 104                     key.rate_limit_count = result["rateLimit"]["count"]
 105                     key.rate_limit_window = result["rateLimit"]["window"]
 106 
 107             key.save()
 108 
 109             self.create_audit_entry(
 110                 request=request,
 111                 organization=project.organization,
 112                 target_object=key.id,
 113                 event=AuditLogEntryEvent.PROJECTKEY_EDIT,
 114                 data=key.get_audit_log_data(),
 115             )
 116 
 117             return Response(serialize(key, request.user), status=200)
 118         return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)

 120     @attach_scenarios([delete_key_scenario])
 121     def delete(self, request, project, key_id):
 122         """
 123         Delete a Client Key
 124         ```````````````````
 125 
 126         Delete a client key.
 127 
 128         :pparam string organization_slug: the slug of the organization the
 129                                           client keys belong to.
 130         :pparam string project_slug: the slug of the project the client keys
 131                                      belong to.
 132         :pparam string key_id: the ID of the key to delete.
 133         :auth: required
 134         """
 135         try:
 136             key = ProjectKey.objects.get(
 137                 project=project, public_key=key_id, roles=F("roles").bitor(ProjectKey.roles.store)
 138             )
 139         except ProjectKey.DoesNotExist:
 140             raise ResourceDoesNotExist
 141 
 142         self.create_audit_entry(
 143             request=request,
 144             organization=project.organization,
 145             target_object=key.id,
 146             event=AuditLogEntryEvent.PROJECTKEY_REMOVE,
 147             data=key.get_audit_log_data(),
 148         )
 149 
 150         key.delete()
 151 
 152         return Response(status=204)
