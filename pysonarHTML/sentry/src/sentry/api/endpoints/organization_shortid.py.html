  15 @scenario("ResolveShortId")
  16 def resolve_short_id_scenario(runner):
  17     group = Group.objects.filter(project=runner.default_project).first()
  18     runner.request(
  19         method="GET",
  20         path="/organizations/%s/shortids/%s/" % (runner.org.slug, group.qualified_short_id),

  24 class ShortIdLookupEndpoint(OrganizationEndpoint):
  25     doc_section = DocSection.ORGANIZATIONS
  26 
  27     @attach_scenarios([resolve_short_id_scenario])
  28     def get(self, request, organization, short_id):
  29         """
  30         Resolve a Short ID
  31         ``````````````````
  32 
  33         This resolves a short ID to the project slug and internal issue ID.
  34 
  35         :pparam string organization_slug: the slug of the organization the
  36                                           short ID should be looked up in.
  37         :pparam string short_id: the short ID to look up.
  38         :auth: required
  39         """
  40         try:
  41             group = Group.objects.by_qualified_short_id(organization.id, short_id)
  42         except Group.DoesNotExist:
  43             raise ResourceDoesNotExist()
  44 
  45         return Response(
  46             {
  47                 "organizationSlug": organization.slug,
  48                 "projectSlug": group.project.slug,
  49                 "groupId": six.text_type(group.id),
  50                 "group": serialize(group, request.user),
  51                 "shortId": group.qualified_short_id,
  52             }
  53         )
