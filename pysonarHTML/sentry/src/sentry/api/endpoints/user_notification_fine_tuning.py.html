  22 KEY_MAP = {
  23     "alerts": {"key": "mail:alert", "type": int},
  24     "workflow": {"key": "workflow:notifications", "type": ""},
  25     "deploy": {"key": "deploy-emails", "type": ""},
  26     "reports": {"key": "reports:disabled-organizations", "type": ""},
  27     "email": {"key": "mail:email", "type": ""},

  31 class UserNotificationFineTuningEndpoint(UserEndpoint):
  32     def get(self, request, user, notification_type):
  33         if notification_type not in KEY_MAP:
  34             return Response(
  35                 {"detail": "Unknown notification type: %s." % notification_type},
  36                 status=status.HTTP_404_NOT_FOUND,
  37             )
  38 
  39         notifications = UserNotificationsSerializer()
  40 
  41         serialized = serialize(
  42             user,
  43             request.user,
  44             notifications,
  45             notification_option_key=KEY_MAP[notification_type]["key"],
  46         )
  47         return Response(serialized)
  48 
  49     def put(self, request, user, notification_type):
  50         """
  51         Update user notification options
  52         ````````````````````````````````
  53 
  54         Updates user's notification options on a per project or organization basis.
  55         Expected payload is a map/dict whose key is a project or org id and value varies depending on `notification_type`.
  56 
  57         For `alerts`, `workflow`, `email` it expects a key of projectId
  58         For `deploy` and `reports` it expects a key of organizationId
  59 
  60         For `alerts`, `workflow`, `deploy`, it expects a value of:
  61             - "-1" = for "default" value (i.e. delete the option)
  62             - "0"  = disabled
  63             - "1"  = enabled
  64         For `reports` it is only a boolean.
  65         For `email` it is a verified email (string).
  66 
  67         :auth required:
  68         :pparam string notification_type:  One of:  alerts, workflow, reports, deploy, email
  69         :param map: Expects a map of id -> value (enabled or email)
  70         """
  71 
  72         if notification_type not in KEY_MAP:
  73             return Response(
  74                 {"detail": "Unknown notification type: %s." % notification_type},
  75                 status=status.HTTP_404_NOT_FOUND,
  76             )
  77 
  78         key = KEY_MAP[notification_type]
  79         filter_args = {"user": user, "key": key["key"]}
  80 
  81         if notification_type == "reports":
  82             (user_option, created) = UserOption.objects.get_or_create(**filter_args)
  83 
  84             value = set(user_option.value or [])
  85 
  86             # set of org ids that user is a member of
  87             org_ids = self.get_org_ids(user)
  88             for org_id, enabled in request.data.items():
  89                 org_id = int(org_id)
  90                 # We want "0" to be falsey
  91                 enabled = int(enabled)
  92 
  93                 # make sure user is in org
  94                 if org_id not in org_ids:
  95                     return Response(
  96                         {
  97                             "detail": "User does not belong to at least one of the \
  98                             requested orgs (org_id: %s)."
  99                             % org_id
 100                         },
 101                         status=status.HTTP_403_FORBIDDEN,
 102                     )
 103 
 104                 # list contains org ids that should have reports DISABLED
 105                 # so if enabled need to check if org_id exists in list (because by default
 106                 # they will have reports enabled)
 107                 if enabled and org_id in value:
 108                     value.remove(org_id)
 109                 elif not enabled:
 110                     value.add(org_id)
 111 
 112             user_option.update(value=list(value))
 113             return Response(status=status.HTTP_204_NO_CONTENT)
 114 
 115         if notification_type in ["alerts", "workflow", "email"]:
 116             update_key = "project"
 117             parent_ids = set(self.get_project_ids(user))
 118         else:
 119             update_key = "organization"
 120             parent_ids = set(self.get_org_ids(user))
 121 
 122         try:
 123             ids_to_update = set([int(i) for i in request.data.keys()])
 124         except ValueError:
 125             return Response(
 126                 {"detail": "Invalid id value provided. Id values should be integers."},
 127                 status=status.HTTP_400_BAD_REQUEST,
 128             )
 129 
 130         # make sure that the ids we are going to update are a subset of the user's
 131         # list of orgs or projects
 132         if not ids_to_update.issubset(parent_ids):
 133             bad_ids = ids_to_update - parent_ids
 134             return Response(
 135                 {
 136                     "detail": "At least one of the requested projects is not \
 137                     available to this user, because the user does not belong \
 138                     to the necessary teams. (ids of unavailable projects: %s)"
 139                     % bad_ids
 140                 },
 141                 status=status.HTTP_403_FORBIDDEN,
 142             )
 143 
 144         if notification_type == "email":
 145             # make sure target emails exist and are verified
 146             emails_to_check = set(request.data.values())
 147             emails = UserEmail.objects.filter(
 148                 user=user, email__in=emails_to_check, is_verified=True
 149             )
 150 
 151             # Is there a better way to check this?
 152             if len(emails) != len(emails_to_check):
 153                 return Response(
 154                     {
 155                         "detail": "Invalid email value(s) provided. Email values \
 156                         must be verified emails for the given user."
 157                     },
 158                     status=status.HTTP_400_BAD_REQUEST,
 159                 )
 160 
 161         with transaction.atomic():
 162             for id in request.data:
 163                 val = request.data[id]
 164                 int_val = int(val) if notification_type != "email" else None
 165 
 166                 filter_args["%s_id" % update_key] = id
 167 
 168                 # 'email' doesn't have a default to delete, and it's a string
 169                 # -1 is a magic value to use "default" value, so just delete option
 170                 if int_val == -1:
 171                     UserOption.objects.filter(**filter_args).delete()
 172                 else:
 173                     user_option, _ = UserOption.objects.get_or_create(**filter_args)
 174 
 175                     # Values have been saved as strings for `mail:alerts` *shrug*
 176                     # `reports:disabled-organizations` requires an array of ids
 177                     user_option.update(value=int_val if key["type"] is int else six.text_type(val))
 178 
 179             return Response(status=status.HTTP_204_NO_CONTENT)
 180 
 181     def get_org_ids(self, user):
 182         """ Get org ids for user """
 183         return set(
 184             OrganizationMember.objects.filter(
 185                 user=user, organization__status=OrganizationStatus.ACTIVE
 186             ).values_list("organization_id", flat=True)
 187         )
 188 
 189     def get_project_ids(self, user):
 190         """ Get project ids that user has access to """
 191         return set(
 192             ProjectTeam.objects.filter(
 193                 team_id__in=OrganizationMemberTeam.objects.filter(
 194                     organizationmember__user=user
 195                 ).values_list("team_id", flat=True)
 196             ).values_list("project_id", flat=True)
 197         )
