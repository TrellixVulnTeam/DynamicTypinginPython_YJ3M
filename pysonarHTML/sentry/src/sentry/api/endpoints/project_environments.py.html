  14 def list_project_environments_scenario(runner):
  15     runner.request(
  16         method="GET",
  17         path="/projects/%s/%s/environments/" % (runner.org.slug, runner.default_project.slug),
  18     )

  21 class ProjectEnvironmentsEndpoint(ProjectEndpoint):
  22     doc_section = DocSection.PROJECTS
  23 
  24     @attach_scenarios([list_project_environments_scenario])
  25     def get(self, request, project):
  26         """
  27         List a Project's Environments
  28         ```````````````````````````````
  29 
  30         Return environments for a given project.
  31 
  32         :qparam string visibility: when omitted only visible environments are
  33                                    returned. Set to ``"hidden"`` for only hidden
  34                                    environments, or ``"all"`` for both hidden
  35                                    and visible environments.
  36 
  37         :pparam string organization_slug: the slug of the organization the project
  38                                           belongs to.
  39 
  40         :pparam string project_slug: the slug of the project.
  41 
  42         :auth: required
  43         """
  44 
  45         queryset = (
  46             EnvironmentProject.objects.filter(
  47                 project=project,
  48                 # Including the organization_id is necessary for postgres to use indexes
  49                 # efficiently.
  50                 environment__organization_id=project.organization_id,
  51             )
  52             .exclude(
  53                 # HACK(mattrobenolt): We don't want to surface the
  54                 # "No Environment" environment to the UI since it
  55                 # doesn't really exist. This might very likely change
  56                 # with new tagstore backend in the future, but until
  57                 # then, we're hiding it since it causes more problems
  58                 # than it's worth.
  59                 environment__name=""
  60             )
  61             .select_related("environment")
  62             .order_by("environment__name")
  63         )
  64 
  65         visibility = request.GET.get("visibility", "visible")
  66         if visibility not in environment_visibility_filter_options:
  67             return Response(
  68                 {
  69                     "detail": u"Invalid value for 'visibility', valid values are: {!r}".format(
  70                         environment_visibility_filter_options.keys()
  71                     )
  72                 },
  73                 status=400,
  74             )
  75 
  76         add_visibility_filters = environment_visibility_filter_options[visibility]
  77         queryset = add_visibility_filters(queryset)
  78 
  79         return Response(serialize(list(queryset), request.user))
