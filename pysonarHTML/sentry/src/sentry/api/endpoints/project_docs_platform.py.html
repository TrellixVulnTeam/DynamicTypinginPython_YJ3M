  15 def replace_keys(html, project_key):
  16     if project_key is None:
  17         return html
  18     html = html.replace("___DSN___", project_key.dsn_private)
  19     html = html.replace("___PUBLIC_DSN___", project_key.dsn_public)
  20     html = html.replace("___PUBLIC_KEY___", project_key.public_key)
  21     html = html.replace("___SECRET_KEY___", project_key.secret_key)
  22     html = html.replace("___PROJECT_ID___", six.text_type(project_key.project_id))
  23     html = html.replace("___MINIDUMP_URL___", project_key.minidump_endpoint)
  24     html = html.replace("___UNREAL_URL___", project_key.unreal_endpoint)
  25     html = html.replace(
  26         "___RELAY_CDN_URL___",
  27         absolute_uri(reverse("sentry-js-sdk-loader", args=[project_key.public_key])),
  28     )
  29 
  30     # If we actually render this in the main UI we can also provide
  31     # extra information about the project (org slug and project slug)
  32     if "___PROJECT_NAME___" in html or "___ORG_NAME___" in html:
  33         project = project_key.project
  34         org = project.organization
  35         html = html.replace("___ORG_NAME___", six.text_type(org.slug))
  36         html = html.replace("___PROJECT_NAME___", six.text_type(project.slug))
  37 
  38     return html
