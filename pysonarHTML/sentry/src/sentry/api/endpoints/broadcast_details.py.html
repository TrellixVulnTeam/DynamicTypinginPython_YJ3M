  20 class BroadcastDetailsEndpoint(Endpoint):
  21     permission_classes = (IsAuthenticated,)
  22 
  23     def _get_broadcast(self, request, broadcast_id):
  24         if is_active_superuser(request) and request.access.has_permission("broadcasts.admin"):
  25             queryset = Broadcast.objects.all()
  26         else:
  27             queryset = Broadcast.objects.filter(
  28                 Q(date_expires__isnull=True) | Q(date_expires__gt=timezone.now()), is_active=True
  29             )
  30 
  31         try:
  32             return queryset.get(id=broadcast_id)
  33         except Broadcast.DoesNotExist:
  34             raise ResourceDoesNotExist
  35 
  36     def _get_validator(self, request):
  37         if is_active_superuser(request):
  38             return AdminBroadcastValidator
  39         return BroadcastValidator
  40 
  41     def _get_serializer(self, request):
  42         if is_active_superuser(request):
  43             return AdminBroadcastSerializer
  44         return BroadcastSerializer
  45 
  46     def _serialize_response(self, request, broadcast):
  47         serializer_cls = self._get_serializer(request)
  48         return self.respond(serialize(broadcast, request.user, serializer=serializer_cls()))
  49 
  50     def get(self, request, broadcast_id):
  51         broadcast = self._get_broadcast(request, broadcast_id)
  52         return self._serialize_response(request, broadcast)
  53 
  54     def put(self, request, broadcast_id):
  55         broadcast = self._get_broadcast(request, broadcast_id)
  56         validator = self._get_validator(request)(data=request.data, partial=True)
  57         if not validator.is_valid():
  58             return self.respond(validator.errors, status=400)
  59 
  60         result = validator.validated_data
  61 
  62         update_kwargs = {}
  63         if result.get("title"):
  64             update_kwargs["title"] = result["title"]
  65         if result.get("message"):
  66             update_kwargs["message"] = result["message"]
  67         if result.get("link"):
  68             update_kwargs["link"] = result["link"]
  69         if result.get("isActive") is not None:
  70             update_kwargs["is_active"] = result["isActive"]
  71         if result.get("dateExpires", -1) != -1:
  72             update_kwargs["date_expires"] = result["dateExpires"]
  73         if result.get("cta"):
  74             update_kwargs["cta"] = result["cta"]
  75         if update_kwargs:
  76             with transaction.atomic():
  77                 broadcast.update(**update_kwargs)
  78                 logger.info(
  79                     "broadcasts.update",
  80                     extra={
  81                         "ip_address": request.META["REMOTE_ADDR"],
  82                         "user_id": request.user.id,
  83                         "broadcast_id": broadcast.id,
  84                         "data": update_kwargs,
  85                     },
  86                 )
  87 
  88         if result.get("hasSeen"):
  89             try:
  90                 with transaction.atomic():
  91                     BroadcastSeen.objects.create(broadcast=broadcast, user=request.user)
  92             except IntegrityError:
  93                 pass
  94 
  95         return self._serialize_response(request, broadcast)
