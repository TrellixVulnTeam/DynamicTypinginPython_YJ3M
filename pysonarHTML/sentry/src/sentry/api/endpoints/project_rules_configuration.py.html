   8 class ProjectRulesConfigurationEndpoint(ProjectEndpoint):
   9     permission_classes = (StrictProjectPermission,)
  10 
  11     def get(self, request, project):
  12         """
  13         Retrieve the list of configuration options for a given project.
  14         """
  15 
  16         action_list = []
  17         condition_list = []
  18 
  19         has_issue_alerts_targeting = (
  20             project.flags.has_issue_alerts_targeting
  21             or request.query_params.get("issue_alerts_targeting") == "1"
  22         )
  23         # TODO: conditions need to be based on actions
  24         for rule_type, rule_cls in rules:
  25             node = rule_cls(project)
  26             context = {"id": node.id, "label": node.label, "enabled": node.is_enabled()}
  27             if hasattr(node, "prompt"):
  28                 context["prompt"] = node.prompt
  29 
  30             if (
  31                 node.id == "sentry.mail.actions.NotifyEmailAction"
  32                 and not has_issue_alerts_targeting
  33             ):
  34                 continue
  35 
  36             if hasattr(node, "form_fields"):
  37                 context["formFields"] = node.form_fields
  38 
  39             # It is possible for a project to have no services. In that scenario we do
  40             # not want the front end to render the action as the action does not have
  41             # options.
  42             if (
  43                 node.id == "sentry.rules.actions.notify_event_service.NotifyEventServiceAction"
  44                 and len(node.get_services()) == 0
  45             ):
  46                 continue
  47 
  48             if rule_type.startswith("condition/"):
  49                 condition_list.append(context)
  50             elif rule_type.startswith("action/"):
  51                 action_list.append(context)
  52 
  53         context = {"actions": action_list, "conditions": condition_list}
  54 
  55         return Response(context)
