  49     def _can_create_team_member(self, request, organization, team_slug):
  50         """
  51         User can join or add a member to a team:
  52 
  53         * If they are an active superuser
  54         * If they are a team admin or have global write access
  55         * If the open membership organization setting is enabled
  56         """
  57 
  58         return (
  59             is_active_superuser(request)
  60             or self._can_admin_team(request, organization, team_slug)
  61             or organization.flags.allow_joinleave
  62         )
  63 
  64     def _can_delete(self, request, member, organization, team_slug):
  65         """
  66         User can remove a member from a team:
  67 
  68         * If they are an active superuser
  69         * If they are removing their own membership
  70         * If they are a team admin or have global write access
  71         """
  72 
  73         if is_active_superuser(request):
  74             return True
  75 
  76         if not request.user.is_authenticated():
  77             return False
  78 
  79         if request.user.id == member.user_id:
  80             return True
  81 
  82         if self._can_admin_team(request, organization, team_slug):
  83             return True
  84 
  85         return False
  86 
  87     def _can_admin_team(self, request, organization, team_slug):
  88         global_roles = [r.id for r in roles.with_scope("org:write") if r.is_global]
  89         team_roles = [r.id for r in roles.with_scope("team:write")]
  90 
  91         # must be a team admin or have global write access
  92         return OrganizationMember.objects.filter(
  93             Q(role__in=global_roles)
  94             | Q(organizationmemberteam__team__slug=team_slug, role__in=team_roles),
  95             organization=organization,
  96             user__id=request.user.id,
  97             user__is_active=True,
  98         ).exists()
  99 
 100     def _get_member(self, request, organization, member_id):
 101         if member_id == "me":
 102             queryset = OrganizationMember.objects.filter(
 103                 organization=organization, user__id=request.user.id, user__is_active=True
 104             )
 105         else:
 106             queryset = OrganizationMember.objects.filter(
 107                 Q(user__is_active=True) | Q(user__isnull=True),
 108                 organization=organization,
 109                 id=member_id,
 110             )
 111         return queryset.select_related("user").get()

 113     def _create_access_request(self, request, team, member):
 114         omt, created = OrganizationAccessRequest.objects.get_or_create(team=team, member=member)
 115 
 116         if not created:
 117             return
 118 
 119         requester = request.user if request.user != member.user else None
 120         if requester:
 121             omt.update(requester=requester)
 122 
 123         omt.send_request_email()

 125     def post(self, request, organization, member_id, team_slug):
 126         """
 127         Join, request access to or add a member to a team.
 128 
 129         If the user needs permission to join the team, an access request will
 130         be generated and the returned status code will be 202.
 131 
 132         If the user is already a member of the team, this will simply return
 133         a 204.
 134         """
 135         try:
 136             member = self._get_member(request, organization, member_id)
 137         except OrganizationMember.DoesNotExist:
 138             raise ResourceDoesNotExist
 139 
 140         if not request.user.is_authenticated():
 141             return Response(status=status.HTTP_401_UNAUTHORIZED)
 142 
 143         try:
 144             team = Team.objects.get(organization=organization, slug=team_slug)
 145         except Team.DoesNotExist:
 146             raise ResourceDoesNotExist
 147 
 148         try:
 149             omt = OrganizationMemberTeam.objects.get(team=team, organizationmember=member)
 150         except OrganizationMemberTeam.DoesNotExist:
 151             if self._can_create_team_member(request, organization, team_slug):
 152                 omt = OrganizationMemberTeam.objects.create(team=team, organizationmember=member)
 153             else:
 154                 self._create_access_request(request, team, member)
 155                 return Response(status=202)
 156 
 157         else:
 158             return Response(status=204)
 159 
 160         self.create_audit_entry(
 161             request=request,
 162             organization=organization,
 163             target_object=omt.id,
 164             target_user=member.user,
 165             event=AuditLogEntryEvent.MEMBER_JOIN_TEAM,
 166             data=omt.get_audit_log_data(),
 167         )
 168 
 169         return Response(serialize(team, request.user, TeamWithProjectsSerializer()), status=201)
 170 
 171     def delete(self, request, organization, member_id, team_slug):
 172         """
 173         Leave or remove a member from a team
 174         """
 175         try:
 176             member = self._get_member(request, organization, member_id)
 177         except OrganizationMember.DoesNotExist:
 178             raise ResourceDoesNotExist
 179 
 180         if not self._can_delete(request, member, organization, team_slug):
 181             return Response({"detail": ERR_INSUFFICIENT_ROLE}, status=400)
 182 
 183         try:
 184             team = Team.objects.get(organization=organization, slug=team_slug)
 185         except Team.DoesNotExist:
 186             raise ResourceDoesNotExist
 187 
 188         try:
 189             omt = OrganizationMemberTeam.objects.get(team=team, organizationmember=member)
 190         except OrganizationMemberTeam.DoesNotExist:
 191             pass
 192         else:
 193             self.create_audit_entry(
 194                 request=request,
 195                 organization=organization,
 196                 target_object=omt.id,
 197                 target_user=member.user,
 198                 event=AuditLogEntryEvent.MEMBER_LEAVE_TEAM,
 199                 data=omt.get_audit_log_data(),
 200             )
 201             omt.delete()
 202 
 203         return Response(serialize(team, request.user, TeamWithProjectsSerializer()), status=200)
