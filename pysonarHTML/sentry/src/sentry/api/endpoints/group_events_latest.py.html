  13 @scenario("GetLatestGroupSample")
  14 def get_latest_group_sample_scenario(runner):
  15     project = runner.default_project

  20 class GroupEventsLatestEndpoint(GroupEndpoint):
  21     doc_section = DocSection.EVENTS
  22 
  23     @attach_scenarios([get_latest_group_sample_scenario])
  24     def get(self, request, group):
  25         """
  26         Retrieve the Latest Event for an Issue
  27         ``````````````````````````````````````
  28 
  29         Retrieves the details of the latest event for an issue.
  30 
  31         :pparam string group_id: the ID of the issue
  32         """
  33         environments = [e.name for e in get_environments(request, group.project.organization)]
  34 
  35         event = group.get_latest_event_for_environments(environments)
  36 
  37         if not event:
  38             return Response({"detail": "No events found for group"}, status=404)
  39 
  40         try:
  41             return client.get(
  42                 u"/projects/{}/{}/events/{}/".format(
  43                     event.organization.slug, event.project.slug, event.event_id
  44                 ),
  45                 request=request,
  46             )
  47         except client.ApiError as e:
  48             return Response(e.body, status=e.status_code)
