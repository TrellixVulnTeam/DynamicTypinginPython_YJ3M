  34 class ProjectSerializer(serializers.Serializer):
  35     name = serializers.CharField(max_length=64, required=True)
  36     slug = serializers.RegexField(r"^[a-z0-9_\-]+$", max_length=50, required=False, allow_null=True)
  37     platform = serializers.CharField(required=False, allow_blank=True, allow_null=True)
  38     default_rules = serializers.BooleanField(required=False, initial=True)
  39 
  40     def validate_platform(self, value):
  41         if Project.is_valid_platform(value):
  42             return value
  43         raise serializers.ValidationError("Invalid platform")

  46 # While currently the UI suggests teams are a parent of a project, in reality
  47 # the project is the core component, and which team it is on is simply an
  48 # attribute. Because you can already change the team of a project via mutating
  49 # it, and because Sentry intends to remove teams as a hierarchy item, we
  50 # allow you to view a teams projects, as well as create a new project as long
  51 # as you are a member of that team and have project scoped permissions.

  67     @attach_scenarios([list_team_projects_scenario])
  68     def get(self, request, team):
  69         """
  70         List a Team's Projects
  71         ``````````````````````
  72 
  73         Return a list of projects bound to a team.
  74 
  75         :pparam string organization_slug: the slug of the organization the
  76                                           team belongs to.
  77         :pparam string team_slug: the slug of the team to list the projects of.
  78         :auth: required
  79         """
  80         if request.auth and hasattr(request.auth, "project"):
  81             queryset = Project.objects.filter(id=request.auth.project.id)
  82         else:
  83             queryset = Project.objects.filter(teams=team, status=ProjectStatus.VISIBLE)
  84 
  85         stats_period = request.GET.get("statsPeriod")
  86         if stats_period not in (None, "", "24h", "14d", "30d"):
  87             return Response(
  88                 {"error": {"params": {"stats_period": {"message": ERR_INVALID_STATS_PERIOD}}}},
  89                 status=400,
  90             )
  91         elif not stats_period:
  92             # disable stats
  93             stats_period = None
  94 
  95         return self.paginate(
  96             request=request,
  97             queryset=queryset,
  98             order_by="slug",
  99             on_results=lambda x: serialize(
 100                 x,
 101                 request.user,
 102                 ProjectSummarySerializer(
 103                     environment_id=self._get_environment_id_from_request(
 104                         request, team.organization.id
 105                     ),
 106                     stats_period=stats_period,
 107                 ),
 108             ),
 109             paginator_cls=OffsetPaginator,
 110         )

 112     @attach_scenarios([create_project_scenario])
 113     def post(self, request, team):
 114         """
 115         Create a New Project
 116         ````````````````````
 117 
 118         Create a new project bound to a team.
 119 
 120         :pparam string organization_slug: the slug of the organization the
 121                                           team belongs to.
 122         :pparam string team_slug: the slug of the team to create a new project
 123                                   for.
 124         :param string name: the name for the new project.
 125         :param string slug: optionally a slug for the new project.  If it's
 126                             not provided a slug is generated from the name.
 127         :param bool default_rules: create default rules (defaults to True)
 128         :auth: required
 129         """
 130         serializer = ProjectSerializer(data=request.data)
 131 
 132         if serializer.is_valid():
 133             result = serializer.validated_data
 134 
 135             with transaction.atomic():
 136                 try:
 137                     with transaction.atomic():
 138                         project = Project.objects.create(
 139                             name=result["name"],
 140                             slug=result.get("slug"),
 141                             organization=team.organization,
 142                             platform=result.get("platform"),
 143                         )
 144                 except IntegrityError:
 145                     return Response(
 146                         {"detail": "A project with this slug already exists."}, status=409
 147                     )
 148                 else:
 149                     project.add_team(team)
 150 
 151                 # XXX: create sample event?
 152 
 153                 self.create_audit_entry(
 154                     request=request,
 155                     organization=team.organization,
 156                     target_object=project.id,
 157                     event=AuditLogEntryEvent.PROJECT_ADD,
 158                     data=project.get_audit_log_data(),
 159                 )
 160 
 161                 project_created.send(
 162                     project=project,
 163                     user=request.user,
 164                     default_rules=result.get("default_rules", True),
 165                     sender=self,
 166                 )
 167 
 168             return Response(serialize(project, request.user), status=201)
 169         return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
