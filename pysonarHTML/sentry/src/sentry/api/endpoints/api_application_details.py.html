  53     def put(self, request, app_id):
  54         try:
  55             instance = ApiApplication.objects.get(
  56                 owner=request.user, client_id=app_id, status=ApiApplicationStatus.active
  57             )
  58         except ApiApplication.DoesNotExist:
  59             raise ResourceDoesNotExist
  60 
  61         serializer = ApiApplicationSerializer(data=request.data, partial=True)
  62 
  63         if serializer.is_valid():
  64             result = serializer.validated_data
  65             kwargs = {}
  66             if "name" in result:
  67                 kwargs["name"] = result["name"]
  68             if "allowedOrigins" in result:
  69                 kwargs["allowed_origins"] = "\n".join(result["allowedOrigins"])
  70             if "redirectUris" in result:
  71                 kwargs["redirect_uris"] = "\n".join(result["redirectUris"])
  72             if "homepageUrl" in result:
  73                 kwargs["homepage_url"] = result["homepageUrl"]
  74             if "privacyUrl" in result:
  75                 kwargs["privacy_url"] = result["privacyUrl"]
  76             if "termsUrl" in result:
  77                 kwargs["terms_url"] = result["termsUrl"]
  78             if kwargs:
  79                 instance.update(**kwargs)
  80             return Response(serialize(instance, request.user), status=200)
  81         return Response(serializer.errors, status=400)

  83     def delete(self, request, app_id):
  84         try:
  85             instance = ApiApplication.objects.get(
  86                 owner=request.user, client_id=app_id, status=ApiApplicationStatus.active
  87             )
  88         except ApiApplication.DoesNotExist:

  90 
  91         updated = ApiApplication.objects.filter(id=instance.id).update(
  92             status=ApiApplicationStatus.pending_deletion

  96 
  97             delete_api_application.apply_async(
  98                 kwargs={"object_id": instance.id, "transaction_id": transaction_id}, countdown=3600
