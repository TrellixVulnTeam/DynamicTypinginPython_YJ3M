  14 class UserAuthenticatorDetailsEndpoint(UserEndpoint):
  15     permission_classes = (OrganizationUserPermission,)
  16 
  17     @sudo_required
  18     def get(self, request, user, auth_id):
  19         """
  20         Get Authenticator Interface
  21         ```````````````````````````
  22 
  23         Retrieves authenticator interface details for user depending on user enrollment status
  24 
  25         :pparam string user_id: user id or "me" for current user
  26         :pparam string auth_id: authenticator model id
  27 
  28         :auth: required
  29         """
  30 
  31         try:
  32             authenticator = Authenticator.objects.get(user=user, id=auth_id)
  33         except (ValueError, Authenticator.DoesNotExist):
  34             return Response(status=status.HTTP_404_NOT_FOUND)
  35 
  36         interface = authenticator.interface
  37 
  38         # User is enrolled to auth interface:
  39         # - display interface details
  40         # - show enrolled specific details like:
  41         #    - created at, last used dates
  42         #    - phone number for SMS
  43         #    - recovery codes
  44         response = serialize(interface)
  45 
  46         if interface.interface_id == "recovery":
  47             response["codes"] = interface.get_unused_codes()
  48         if interface.interface_id == "sms":
  49             response["phone"] = interface.phone_number
  50         if interface.interface_id == "u2f":
  51             response["devices"] = interface.get_registered_devices()
  52 
  53         return Response(response)

  55     @sudo_required
  56     def put(self, request, user, auth_id):
  57         """
  58         Modify authenticator interface
  59         ``````````````````````````````
  60 
  61         Currently, only supports regenerating recovery codes
  62 
  63         :pparam string user_id: user id or 'me' for current user
  64         :pparam int auth_id: authenticator model id
  65 
  66         :auth required:
  67         """
  68 
  69         try:
  70             authenticator = Authenticator.objects.get(user=user, id=auth_id)
  71         except (ValueError, Authenticator.DoesNotExist):
  72             return Response(status=status.HTTP_404_NOT_FOUND)
  73 
  74         interface = authenticator.interface
  75 
  76         if interface.interface_id == "recovery":
  77             interface.regenerate_codes()
  78 
  79             capture_security_activity(
  80                 account=user,
  81                 type="recovery-codes-regenerated",
  82                 actor=request.user,
  83                 ip_address=request.META["REMOTE_ADDR"],
  84                 context={"authenticator": authenticator},
  85                 send_email=True,
  86             )
  87         return Response(serialize(interface))

  89     @sudo_required
  90     def delete(self, request, user, auth_id, interface_device_id=None):
  91         """
  92         Remove authenticator
  93         ````````````````````
  94 
  95         :pparam string user_id: user id or 'me' for current user
  96         :pparam string auth_id: authenticator model id
  97         :pparam string interface_device_id: some interfaces (u2f) allow multiple devices
  98 
  99         :auth required:
 100         """
 101 
 102         try:
 103             authenticator = Authenticator.objects.get(user=user, id=auth_id)
 104         except (ValueError, Authenticator.DoesNotExist):
 105             return Response(status=status.HTTP_404_NOT_FOUND)
 106 
 107         interface = authenticator.interface
 108 
 109         # Remove a single device and not entire authentication method
 110         if interface.interface_id == "u2f" and interface_device_id is not None:
 111             device_name = interface.get_device_name(interface_device_id)
 112             # Can't remove if this is the last device, will return False if so
 113             if not interface.remove_u2f_device(interface_device_id):
 114                 return Response(status=status.HTTP_500_INTERNAL_SERVER_ERROR)
 115             interface.authenticator.save()
 116 
 117             capture_security_activity(
 118                 account=user,
 119                 type="mfa-removed",
 120                 actor=request.user,
 121                 ip_address=request.META["REMOTE_ADDR"],
 122                 context={"authenticator": authenticator, "device_name": device_name},
 123                 send_email=True,
 124             )
 125             return Response(status=status.HTTP_204_NO_CONTENT)
 126 
 127         # if the user's organization requires 2fa,
 128         # don't delete the last auth method
 129         enrolled_methods = Authenticator.objects.all_interfaces_for_user(user, ignore_backup=True)
 130         last_2fa_method = len(enrolled_methods) == 1
 131         require_2fa = user.get_orgs_require_2fa().exists()
 132 
 133         if require_2fa and last_2fa_method:
 134             return Response(
 135                 {"detail": "Cannot delete authenticator because organization requires 2FA"},
 136                 status=status.HTTP_403_FORBIDDEN,
 137             )
 138 
 139         with transaction.atomic():
 140             authenticator.delete()
 141 
 142             # if we delete an actual authenticator and all that
 143             # remains are backup interfaces, then we kill them in the
 144             # process.
 145             if not interface.is_backup_interface:
 146                 interfaces = Authenticator.objects.all_interfaces_for_user(user)
 147                 backup_interfaces = [x for x in interfaces if x.is_backup_interface]
 148                 if len(backup_interfaces) == len(interfaces):
 149                     for iface in backup_interfaces:
 150                         iface.authenticator.delete()
 151 
 152                     # wait to generate entries until all pending writes
 153                     # have been sent to db
 154                     for iface in backup_interfaces:
 155                         capture_security_activity(
 156                             account=request.user,
 157                             type="mfa-removed",
 158                             actor=request.user,
 159                             ip_address=request.META["REMOTE_ADDR"],
 160                             context={"authenticator": iface.authenticator},
 161                             send_email=False,
 162                         )
 163 
 164             capture_security_activity(
 165                 account=user,
 166                 type="mfa-removed",
 167                 actor=request.user,
 168                 ip_address=request.META["REMOTE_ADDR"],
 169                 context={"authenticator": authenticator},
 170                 send_email=not interface.is_backup_interface,
 171             )
 172 
 173         return Response(status=status.HTTP_204_NO_CONTENT)
