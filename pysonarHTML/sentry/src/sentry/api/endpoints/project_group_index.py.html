  67     def _search(self, request, project, extra_query_kwargs=None):
  68         try:
  69             environment = self._get_environment_from_request(request, project.organization_id)
  70         except Environment.DoesNotExist:
  71             # XXX: The 1000 magic number for `max_hits` is an abstraction leak
  72             # from `sentry.api.paginator.BasePaginator.get_result`.
  73             result = CursorResult([], None, None, hits=0, max_hits=1000)
  74             query_kwargs = {}
  75         else:
  76             environments = [environment] if environment is not None else environment
  77             query_kwargs = build_query_params_from_request(
  78                 request, project.organization, [project], environments
  79             )
  80             if extra_query_kwargs is not None:
  81                 assert "environment" not in extra_query_kwargs
  82                 query_kwargs.update(extra_query_kwargs)
  83 
  84             query_kwargs["environments"] = environments
  85             result = search.query(**query_kwargs)
  86         return result, query_kwargs

  89     @attach_scenarios([list_project_issues_scenario])
  90     def get(self, request, project):
  91         """
  92         List a Project's Issues
  93         ```````````````````````
  94 
  95         Return a list of issues (groups) bound to a project.  All parameters are
  96         supplied as query string parameters.
  97 
  98         A default query of ``is:unresolved`` is applied. To return results
  99         with other statuses send an new query value (i.e. ``?query=`` for all
 100         results).
 101 
 102         The ``statsPeriod`` parameter can be used to select the timeline
 103         stats which should be present. Possible values are: ``""`` (disable),
 104         ``"24h"``, ``"14d"``
 105 
 106         :qparam string statsPeriod: an optional stat period (can be one of
 107                                     ``"24h"``, ``"14d"``, and ``""``).
 108         :qparam bool shortIdLookup: if this is set to true then short IDs are
 109                                     looked up by this function as well.  This
 110                                     can cause the return value of the function
 111                                     to return an event issue of a different
 112                                     project which is why this is an opt-in.
 113                                     Set to `1` to enable.
 114         :qparam querystring query: an optional Sentry structured search
 115                                    query.  If not provided an implied
 116                                    ``"is:unresolved"`` is assumed.)
 117         :qparam string environment: this restricts the issues to ones containing
 118                                     events from this environment
 119         :pparam string organization_slug: the slug of the organization the
 120                                           issues belong to.
 121         :pparam string project_slug: the slug of the project the issues
 122                                      belong to.
 123         :auth: required
 124         """
 125         stats_period = request.GET.get("statsPeriod")
 126         if stats_period not in (None, "", "24h", "14d"):
 127             return Response({"detail": ERR_INVALID_STATS_PERIOD}, status=400)
 128         elif stats_period is None:
 129             # default
 130             stats_period = "24h"
 131         elif stats_period == "":
 132             # disable stats
 133             stats_period = None
 134 
 135         serializer = functools.partial(
 136             StreamGroupSerializer,
 137             environment_func=self._get_environment_func(request, project.organization_id),
 138             stats_period=stats_period,
 139         )
 140 
 141         query = request.GET.get("query", "").strip()
 142         if query:
 143             matching_group = None
 144             matching_event = None
 145             event_id = normalize_event_id(query)
 146             if event_id:
 147                 # check to see if we've got an event ID
 148                 try:
 149                     matching_group = Group.objects.from_event_id(project, event_id)
 150                 except Group.DoesNotExist:
 151                     pass
 152                 else:
 153                     matching_event = eventstore.get_event_by_id(project.id, event_id)
 154             elif matching_group is None:
 155                 matching_group = get_by_short_id(
 156                     project.organization_id, request.GET.get("shortIdLookup"), query
 157                 )
 158                 if matching_group is not None and matching_group.project_id != project.id:
 159                     matching_group = None
 160 
 161             if matching_group is not None:
 162                 matching_event_environment = None
 163 
 164                 try:
 165                     matching_event_environment = (
 166                         matching_event.get_environment().name if matching_event else None
 167                     )
 168                 except Environment.DoesNotExist:
 169                     pass
 170 
 171                 response = Response(
 172                     serialize(
 173                         [matching_group],
 174                         request.user,
 175                         serializer(
 176                             matching_event_id=getattr(matching_event, "event_id", None),
 177                             matching_event_environment=matching_event_environment,
 178                         ),
 179                     )
 180                 )
 181                 response["X-Sentry-Direct-Hit"] = "1"
 182                 return response
 183 
 184         try:
 185             cursor_result, query_kwargs = self._search(request, project, {"count_hits": True})
 186         except ValidationError as exc:
 187             return Response({"detail": six.text_type(exc)}, status=400)
 188 
 189         results = list(cursor_result)
 190 
 191         context = serialize(results, request.user, serializer())
 192 
 193         # HACK: remove auto resolved entries
 194         # TODO: We should try to integrate this into the search backend, since
 195         # this can cause us to arbitrarily return fewer results than requested.
 196         status = [
 197             search_filter
 198             for search_filter in query_kwargs.get("search_filters", [])
 199             if search_filter.key.name == "status"
 200         ]
 201         if status and status[0].value.raw_value == GroupStatus.UNRESOLVED:
 202             context = [r for r in context if r["status"] == "unresolved"]
 203 
 204         response = Response(context)
 205 
 206         self.add_cursor_headers(request, response, cursor_result)
 207 
 208         if results and query not in DEFAULT_SAVED_SEARCH_QUERIES:
 209             advanced_search.send(project=project, sender=request.user)
 210             analytics.record(
 211                 "project_issue.searched",
 212                 user_id=request.user.id,
 213                 organization_id=project.organization_id,
 214                 project_id=project.id,
 215                 query=query,
 216             )
 217 
 218         return response

 220     @attach_scenarios([bulk_update_issues_scenario])
 221     def put(self, request, project):
 222         """
 223         Bulk Mutate a List of Issues
 224         ````````````````````````````
 225 
 226         Bulk mutate various attributes on issues.  The list of issues
 227         to modify is given through the `id` query parameter.  It is repeated
 228         for each issue that should be modified.
 229 
 230         - For non-status updates, the `id` query parameter is required.
 231         - For status updates, the `id` query parameter may be omitted
 232           for a batch "update all" query.
 233         - An optional `status` query parameter may be used to restrict
 234           mutations to only events with the given status.
 235 
 236         The following attributes can be modified and are supplied as
 237         JSON object in the body:
 238 
 239         If any ids are out of scope this operation will succeed without
 240         any data mutation.
 241 
 242         :qparam int id: a list of IDs of the issues to be mutated.  This
 243                         parameter shall be repeated for each issue.  It
 244                         is optional only if a status is mutated in which
 245                         case an implicit `update all` is assumed.
 246         :qparam string status: optionally limits the query to issues of the
 247                                specified status.  Valid values are
 248                                ``"resolved"``, ``"unresolved"`` and
 249                                ``"ignored"``.
 250         :pparam string organization_slug: the slug of the organization the
 251                                           issues belong to.
 252         :pparam string project_slug: the slug of the project the issues
 253                                      belong to.
 254         :param string status: the new status for the issues.  Valid values
 255                               are ``"resolved"``, ``"resolvedInNextRelease"``,
 256                               ``"unresolved"``, and ``"ignored"``.
 257         :param map statusDetails: additional details about the resolution.
 258                                   Valid values are ``"inRelease"``, ``"inNextRelease"``,
 259                                   ``"inCommit"``,  ``"ignoreDuration"``, ``"ignoreCount"``,
 260                                   ``"ignoreWindow"``, ``"ignoreUserCount"``, and
 261                                   ``"ignoreUserWindow"``.
 262         :param int ignoreDuration: the number of minutes to ignore this issue.
 263         :param boolean isPublic: sets the issue to public or private.
 264         :param boolean merge: allows to merge or unmerge different issues.
 265         :param string assignedTo: the user or team that should be assigned to
 266                                   this issue. Can be of the form ``"<user_id>"``,
 267                                   ``"user:<user_id>"``, ``"<username>"``,
 268                                   ``"<user_primary_email>"``, or ``"team:<team_id>"``.
 269         :param boolean hasSeen: in case this API call is invoked with a user
 270                                 context this allows changing of the flag
 271                                 that indicates if the user has seen the
 272                                 event.
 273         :param boolean isBookmarked: in case this API call is invoked with a
 274                                      user context this allows changing of
 275                                      the bookmark flag.
 276         :auth: required
 277         """
 278 
 279         search_fn = functools.partial(self._search, request, project)
 280         return update_groups(request, [project], project.organization_id, search_fn)

 283     def delete(self, request, project):
 284         """
 285         Bulk Remove a List of Issues
 286         ````````````````````````````
 287 
 288         Permanently remove the given issues. The list of issues to
 289         modify is given through the `id` query parameter.  It is repeated
 290         for each issue that should be removed.
 291 
 292         Only queries by 'id' are accepted.
 293 
 294         If any ids are out of scope this operation will succeed without
 295         any data mutation.
 296 
 297         :qparam int id: a list of IDs of the issues to be removed.  This
 298                         parameter shall be repeated for each issue.
 299         :pparam string organization_slug: the slug of the organization the
 300                                           issues belong to.
 301         :pparam string project_slug: the slug of the project the issues
 302                                      belong to.
 303         :auth: required
 304         """
 305         search_fn = functools.partial(self._search, request, project)
 306         return delete_groups(request, [project], project.organization_id, search_fn)
