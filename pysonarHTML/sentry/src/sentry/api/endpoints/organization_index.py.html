  54 class OrganizationIndexEndpoint(Endpoint):
  55     doc_section = DocSection.ORGANIZATIONS
  56     permission_classes = (OrganizationPermission,)
  57 
  58     @attach_scenarios([list_your_organizations_scenario])
  59     def get(self, request):
  60         """
  61         List your Organizations
  62         ```````````````````````
  63 
  64         Return a list of organizations available to the authenticated
  65         session.  This is particularly useful for requests with an
  66         user bound context.  For API key based requests this will
  67         only return the organization that belongs to the key.
  68 
  69         :qparam bool owner: restrict results to organizations in which you are
  70                             an organization owner
  71 
  72         :auth: required
  73         """
  74         owner_only = request.GET.get("owner") in ("1", "true")
  75 
  76         queryset = Organization.objects.distinct()
  77 
  78         if request.auth and not request.user.is_authenticated():
  79             if hasattr(request.auth, "project"):
  80                 queryset = queryset.filter(id=request.auth.project.organization_id)
  81             elif request.auth.organization is not None:
  82                 queryset = queryset.filter(id=request.auth.organization.id)
  83 
  84         elif owner_only:
  85             # This is used when closing an account
  86             queryset = queryset.filter(
  87                 member_set__role=roles.get_top_dog().id,
  88                 member_set__user=request.user,
  89                 status=OrganizationStatus.VISIBLE,
  90             )
  91             org_results = []
  92             for org in sorted(queryset, key=lambda x: x.name):
  93                 # O(N) query
  94                 org_results.append(
  95                     {"organization": serialize(org), "singleOwner": org.has_single_owner()}
  96                 )
  97 
  98             return Response(org_results)
  99 
 100         elif not (is_active_superuser(request) and request.GET.get("show") == "all"):
 101             queryset = queryset.filter(
 102                 id__in=OrganizationMember.objects.filter(user=request.user).values("organization")
 103             )
 104 
 105         query = request.GET.get("query")
 106         if query:
 107             tokens = tokenize_query(query)
 108             for key, value in six.iteritems(tokens):
 109                 if key == "query":
 110                     value = " ".join(value)
 111                     queryset = queryset.filter(
 112                         Q(name__icontains=value)
 113                         | Q(slug__icontains=value)
 114                         | Q(members__email__iexact=value)
 115                     )
 116                 elif key == "slug":
 117                     queryset = queryset.filter(in_iexact("slug", value))
 118                 elif key == "email":
 119                     queryset = queryset.filter(in_iexact("members__email", value))
 120                 elif key == "platform":
 121                     queryset = queryset.filter(
 122                         project__in=ProjectPlatform.objects.filter(platform__in=value).values(
 123                             "project_id"
 124                         )
 125                     )
 126                 elif key == "id":
 127                     queryset = queryset.filter(id__in=value)
 128                 elif key == "status":
 129                     try:
 130                         queryset = queryset.filter(
 131                             status__in=[OrganizationStatus[v.upper()] for v in value]
 132                         )
 133                     except KeyError:
 134                         queryset = queryset.none()
 135                 elif key == "member_id":
 136                     queryset = queryset.filter(
 137                         id__in=OrganizationMember.objects.filter(id__in=value).values(
 138                             "organization"
 139                         )
 140                     )
 141                 else:
 142                     queryset = queryset.none()
 143 
 144         sort_by = request.GET.get("sortBy")
 145         if sort_by == "members":
 146             queryset = queryset.annotate(member_count=Count("member_set"))
 147             order_by = "-member_count"
 148             paginator_cls = OffsetPaginator
 149         elif sort_by == "projects":
 150             queryset = queryset.annotate(project_count=Count("project"))
 151             order_by = "-project_count"
 152             paginator_cls = OffsetPaginator
 153         elif sort_by == "events":
 154             queryset = queryset.annotate(event_count=Sum("stats__events_24h")).filter(
 155                 stats__events_24h__isnull=False
 156             )
 157             order_by = "-event_count"
 158             paginator_cls = OffsetPaginator
 159         else:
 160             order_by = "-date_added"
 161             paginator_cls = DateTimePaginator
 162 
 163         return self.paginate(
 164             request=request,
 165             queryset=queryset,
 166             order_by=order_by,
 167             on_results=lambda x: serialize(x, request.user),
 168             paginator_cls=paginator_cls,
 169         )

 171     # XXX: endpoint useless for end-users as it needs user context.
 172     def post(self, request):
 173         """
 174         Create a New Organization
 175         `````````````````````````
 176 
 177         Create a new organization owned by the request's user.  To create
 178         an organization only the name is required.
 179 
 180         :param string name: the human readable name for the new organization.
 181         :param string slug: the unique URL slug for this organization.  If
 182                             this is not provided a slug is automatically
 183                             generated based on the name.
 184         :param bool agreeTerms: a boolean signaling you agree to the applicable
 185                                 terms of service and privacy policy.
 186         :auth: required, user-context-needed
 187         """
 188         if not request.user.is_authenticated():
 189             return Response({"detail": "This endpoint requires user info"}, status=401)
 190 
 191         if not features.has("organizations:create", actor=request.user):
 192             return Response(
 193                 {"detail": "Organizations are not allowed to be created by this user."}, status=401
 194             )
 195 
 196         limit = options.get("api.rate-limit.org-create")
 197         if limit and ratelimiter.is_limited(
 198             u"org-create:{}".format(request.user.id), limit=limit, window=3600
 199         ):
 200             return Response(
 201                 {"detail": "You are attempting to create too many organizations too quickly."},
 202                 status=429,
 203             )
 204 
 205         serializer = OrganizationSerializer(data=request.data)
 206 
 207         if serializer.is_valid():
 208             result = serializer.validated_data
 209 
 210             try:
 211                 with transaction.atomic():
 212                     org = Organization.objects.create(name=result["name"], slug=result.get("slug"))
 213 
 214                     om = OrganizationMember.objects.create(
 215                         organization=org, user=request.user, role=roles.get_top_dog().id
 216                     )
 217 
 218                     if result.get("defaultTeam"):
 219                         team = org.team_set.create(name=org.name)
 220 
 221                         OrganizationMemberTeam.objects.create(
 222                             team=team, organizationmember=om, is_active=True
 223                         )
 224 
 225                     self.create_audit_entry(
 226                         request=request,
 227                         organization=org,
 228                         target_object=org.id,
 229                         event=AuditLogEntryEvent.ORG_ADD,
 230                         data=org.get_audit_log_data(),
 231                     )
 232 
 233                     analytics.record(
 234                         "organization.created",
 235                         org,
 236                         actor_id=request.user.id if request.user.is_authenticated() else None,
 237                     )
 238 
 239             except IntegrityError:
 240                 return Response(
 241                     {"detail": "An organization with this slug already exists."}, status=409
 242                 )
 243 
 244             # failure on sending this signal is acceptable
 245             if result.get("agreeTerms"):
 246                 terms_accepted.send_robust(
 247                     user=request.user,
 248                     organization=org,
 249                     ip_address=request.META["REMOTE_ADDR"],
 250                     sender=type(self),
 251                 )
 252 
 253             return Response(serialize(org, request.user), status=201)
 254         return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
