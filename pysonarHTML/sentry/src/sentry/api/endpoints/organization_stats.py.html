  19 class OrganizationStatsEndpoint(OrganizationEndpoint, EnvironmentMixin, StatsMixin):
  20     doc_section = DocSection.ORGANIZATIONS
  21 
  22     @attach_scenarios([retrieve_event_counts_organization])
  23     def get(self, request, organization):
  24         """
  25         Retrieve Event Counts for an Organization
  26         `````````````````````````````````````````
  27 
  28         .. caution::
  29            This endpoint may change in the future without notice.
  30 
  31         Return a set of points representing a normalized timestamp and the
  32         number of events seen in the period.
  33 
  34         :pparam string organization_slug: the slug of the organization for
  35                                           which the stats should be
  36                                           retrieved.
  37         :qparam string stat: the name of the stat to query (``"received"``,
  38                              ``"rejected"``, ``"blacklisted"``)
  39         :qparam timestamp since: a timestamp to set the start of the query
  40                                  in seconds since UNIX epoch.
  41         :qparam timestamp until: a timestamp to set the end of the query
  42                                  in seconds since UNIX epoch.
  43         :qparam string resolution: an explicit resolution to search
  44                                    for (one of ``10s``, ``1h``, and ``1d``)
  45         :auth: required
  46         """
  47         group = request.GET.get("group", "organization")
  48         if group == "organization":
  49             keys = [organization.id]
  50         elif group == "project":
  51             team_list = Team.objects.get_for_user(organization=organization, user=request.user)
  52 
  53             project_ids = request.GET.getlist("projectID")
  54             if not project_ids:
  55                 project_list = []
  56                 for team in team_list:
  57                     project_list.extend(Project.objects.get_for_user(team=team, user=request.user))
  58             else:
  59                 project_list = Project.objects.filter(teams__in=team_list, id__in=project_ids)
  60             keys = list({p.id for p in project_list})
  61         else:
  62             raise ValueError("Invalid group: %s" % group)
  63 
  64         if "id" in request.GET:
  65             id_filter_set = frozenset(map(int, request.GET.getlist("id")))
  66             keys = [k for k in keys if k in id_filter_set]
  67 
  68         if not keys:
  69             return Response([])
  70 
  71         stat_model = None
  72         stat = request.GET.get("stat", "received")
  73         query_kwargs = {}
  74         if stat == "received":
  75             if group == "project":
  76                 stat_model = tsdb.models.project_total_received
  77             else:
  78                 stat_model = tsdb.models.organization_total_received
  79         elif stat == "rejected":
  80             if group == "project":
  81                 stat_model = tsdb.models.project_total_rejected
  82             else:
  83                 stat_model = tsdb.models.organization_total_rejected
  84         elif stat == "blacklisted":
  85             if group == "project":
  86                 stat_model = tsdb.models.project_total_blacklisted
  87             else:
  88                 stat_model = tsdb.models.organization_total_blacklisted
  89         elif stat == "generated":
  90             if group == "project":
  91                 stat_model = tsdb.models.project
  92                 try:
  93                     query_kwargs["environment_id"] = self._get_environment_id_from_request(
  94                         request, organization.id
  95                     )
  96                 except Environment.DoesNotExist:
  97                     raise ResourceDoesNotExist
  98 
  99         if stat_model is None:
 100             raise ValueError("Invalid group: %s, stat: %s" % (group, stat))
 101 
 102         data = tsdb.get_range(
 103             model=stat_model, keys=keys, **self._parse_args(request, **query_kwargs)
 104         )
 105 
 106         if group == "organization":
 107             data = data[organization.id]
 108 
 109         return Response(data)
