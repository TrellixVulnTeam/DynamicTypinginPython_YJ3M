  19 def map_value_to_constant(constant, value):
  20     value = value.upper()
  21     if value == "OK":
  22         value = "ACTIVE"
  23     if not hasattr(constant, value):
  24         raise ValueError(value)
  25     return getattr(constant, value)

  28 class OrganizationMonitorsEndpoint(OrganizationEndpoint):
  29     def get(self, request, organization):
  30         """
  31         Retrieve monitors for an organization
  32         `````````````````````````````````````
  33 
  34         :pparam string organization_slug: the slug of the organization
  35         :auth: required
  36         """
  37         if not features.has("organizations:monitors", organization, actor=request.user):
  38             raise ResourceDoesNotExist
  39 
  40         try:
  41             filter_params = self.get_filter_params(request, organization, date_filter_optional=True)
  42         except NoProjects:
  43             return self.respond([])
  44 
  45         queryset = Monitor.objects.filter(
  46             organization_id=organization.id, project_id__in=filter_params["project_id"]
  47         ).exclude(status__in=[MonitorStatus.PENDING_DELETION, MonitorStatus.DELETION_IN_PROGRESS])
  48         query = request.GET.get("query")
  49         if query:
  50             tokens = tokenize_query(query)
  51             for key, value in six.iteritems(tokens):
  52                 if key == "query":
  53                     value = " ".join(value)
  54                     queryset = queryset.filter(Q(name__icontains=value) | Q(id__iexact=value))
  55                 elif key == "id":
  56                     queryset = queryset.filter(in_iexact("id", value))
  57                 elif key == "name":
  58                     queryset = queryset.filter(in_iexact("name", value))
  59                 elif key == "status":
  60                     try:
  61                         queryset = queryset.filter(
  62                             status__in=map_value_to_constant(MonitorStatus, value)
  63                         )
  64                     except ValueError:
  65                         queryset = queryset.none()
  66                 elif key == "type":
  67                     try:
  68                         queryset = queryset.filter(
  69                             status__in=map_value_to_constant(MonitorType, value)
  70                         )
  71                     except ValueError:
  72                         queryset = queryset.none()
  73 
  74                 elif key == "id":
  75                     queryset = queryset.filter(id__in=value)
  76                 else:
  77                     queryset = queryset.none()
  78 
  79         queryset = queryset.extra(
  80             select={"is_error": "sentry_monitor.status = %s" % (MonitorStatus.ERROR,)}
  81         )
  82 
  83         return self.paginate(
  84             request=request,
  85             queryset=queryset,
  86             order_by=("-is_error", "-name"),
  87             on_results=lambda x: serialize(x, request.user),
  88             paginator_cls=OffsetPaginator,
  89         )

  91     def post(self, request, organization):
  92         """
  93         Create a monitor
  94         ````````````````
  95 
  96         :pparam string organization_slug: the slug of the organization
  97         :auth: required
  98         """
  99         validator = MonitorValidator(
 100             data=request.data, context={"organization": organization, "access": request.access}
 101         )
 102         if not validator.is_valid():
 103             return self.respond(validator.errors, status=400)
 104 
 105         result = validator.validated_data
 106 
 107         monitor = Monitor.objects.create(
 108             project_id=result["project"].id,
 109             organization_id=organization.id,
 110             name=result["name"],
 111             status=result["status"],
 112             type=result["type"],
 113             config=result["config"],
 114         )
 115         self.create_audit_entry(
 116             request=request,
 117             organization=organization,
 118             target_object=monitor.id,
 119             event=AuditLogEntryEvent.MONITOR_ADD,
 120             data=monitor.get_audit_log_data(),
 121         )
 122 
 123         return self.respond(serialize(monitor, request.user), status=201)
