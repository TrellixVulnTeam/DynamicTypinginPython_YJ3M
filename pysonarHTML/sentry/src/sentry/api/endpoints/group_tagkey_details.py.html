  20 class GroupTagKeyDetailsEndpoint(GroupEndpoint, EnvironmentMixin):
  21     doc_section = DocSection.EVENTS
  22 
  23     # XXX: this scenario does not work for some inexplicable reasons
  24     # @attach_scenarios([list_tag_details_scenario])
  25     def get(self, request, group, key):
  26         """
  27         Retrieve Tag Details
  28         ````````````````````
  29 
  30         Returns details for given tag key related to an issue.
  31 
  32         :pparam string issue_id: the ID of the issue to retrieve.
  33         :pparam string key: the tag key to look the values up for.
  34         :auth: required
  35         """
  36         lookup_key = tagstore.prefix_reserved_key(key)
  37 
  38         try:
  39             environment_id = self._get_environment_id_from_request(
  40                 request, group.project.organization_id
  41             )
  42         except Environment.DoesNotExist:
  43             # if the environment doesn't exist then the tag can't possibly exist
  44             raise ResourceDoesNotExist
  45 
  46         try:
  47             group_tag_key = tagstore.get_group_tag_key(
  48                 group.project_id, group.id, environment_id, lookup_key
  49             )
  50         except tagstore.GroupTagKeyNotFound:
  51             raise ResourceDoesNotExist
  52 
  53         if group_tag_key.count is None:
  54             group_tag_key.count = tagstore.get_group_tag_value_count(
  55                 group.project_id, group.id, environment_id, lookup_key
  56             )
  57 
  58         if group_tag_key.top_values is None:
  59             group_tag_key.top_values = tagstore.get_top_group_tag_values(
  60                 group.project_id, group.id, environment_id, lookup_key
  61             )
  62 
  63         return Response(serialize(group_tag_key, request.user))
