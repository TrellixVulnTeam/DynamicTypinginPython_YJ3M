 119 class OrganizationSerializer(serializers.Serializer):
 120     name = serializers.CharField(max_length=64)
 121     slug = serializers.RegexField(r"^[a-z0-9_\-]+$", max_length=50)
 122     accountRateLimit = EmptyIntegerField(
 123         min_value=0, max_value=1000000, required=False, allow_null=True
 124     )
 125     projectRateLimit = EmptyIntegerField(
 126         min_value=50, max_value=100, required=False, allow_null=True
 127     )
 128     avatar = AvatarField(required=False, allow_null=True)
 129     avatarType = serializers.ChoiceField(
 130         choices=(("upload", "upload"), ("letter_avatar", "letter_avatar")),
 131         required=False,
 132         allow_null=True,
 133     )
 134 
 135     openMembership = serializers.BooleanField(required=False)
 136     allowSharedIssues = serializers.BooleanField(required=False)
 137     enhancedPrivacy = serializers.BooleanField(required=False)
 138     dataScrubber = serializers.BooleanField(required=False)
 139     dataScrubberDefaults = serializers.BooleanField(required=False)
 140     sensitiveFields = ListField(child=serializers.CharField(), required=False)
 141     safeFields = ListField(child=serializers.CharField(), required=False)
 142     storeCrashReports = serializers.IntegerField(min_value=-1, max_value=20, required=False)
 143     attachmentsRole = serializers.CharField(required=True)
 144     eventsMemberAdmin = serializers.BooleanField(required=False)
 145     scrubIPAddresses = serializers.BooleanField(required=False)
 146     scrapeJavaScript = serializers.BooleanField(required=False)
 147     isEarlyAdopter = serializers.BooleanField(required=False)
 148     require2FA = serializers.BooleanField(required=False)
 149     trustedRelays = ListField(child=serializers.CharField(), required=False)
 150     allowJoinRequests = serializers.BooleanField(required=False)

 153     @memoize
 154     def _has_legacy_rate_limits(self):
 155         org = self.context["organization"]
 156         return OrganizationOption.objects.filter(
 157             organization=org, key__in=LEGACY_RATE_LIMIT_OPTIONS
 158         ).exists()
 159 
 160     def _has_sso_enabled(self):
 161         org = self.context["organization"]
 162         return AuthProvider.objects.filter(organization=org).exists()
 163 
 164     def validate_slug(self, value):
 165         # Historically, the only check just made sure there was more than 1
 166         # character for the slug, but since then, there are many slugs that
 167         # fit within this new imposed limit. We're not fixing existing, but
 168         # just preventing new bad values.
 169         if len(value) < 3:
 170             raise serializers.ValidationError(
 171                 'This slug "%s" is too short. Minimum of 3 characters.' % (value,)
 172             )
 173         if value in RESERVED_ORGANIZATION_SLUGS:
 174             raise serializers.ValidationError(
 175                 'This slug "%s" is reserved and not allowed.' % (value,)
 176             )
 177         qs = Organization.objects.filter(slug=value).exclude(id=self.context["organization"].id)
 178         if qs.exists():
 179             raise serializers.ValidationError('The slug "%s" is already in use.' % (value,))
 180         return value
 181 
 182     def validate_relayPiiConfig(self, value):
 183         organization = self.context["organization"]
 184         return validate_pii_config_update(organization, value)
 185 
 186     def validate_sensitiveFields(self, value):
 187         if value and not all(value):
 188             raise serializers.ValidationError("Empty values are not allowed.")
 189         return value
 190 
 191     def validate_safeFields(self, value):
 192         if value and not all(value):
 193             raise serializers.ValidationError("Empty values are not allowed.")
 194         return value
 195 
 196     def validate_attachmentsRole(self, value):
 197         try:
 198             roles.get(value)
 199         except KeyError:
 200             raise serializers.ValidationError("Invalid role")
 201         return value
 202 
 203     def validate_require2FA(self, value):
 204         user = self.context["user"]
 205         has_2fa = Authenticator.objects.user_has_2fa(user)
 206         if value and not has_2fa:
 207             raise serializers.ValidationError(ERR_NO_2FA)
 208 
 209         if value and self._has_sso_enabled():
 210             raise serializers.ValidationError(ERR_SSO_ENABLED)
 211         return value
 212 
 213     def validate_trustedRelays(self, value):
 214         from sentry import features
 215 
 216         organization = self.context["organization"]
 217         request = self.context["request"]
 218         has_relays = features.has("organizations:relay", organization, actor=request.user)
 219         if not has_relays:
 220             raise serializers.ValidationError(
 221                 "Organization does not have the relay feature enabled"
 222             )
 223         return value
 224 
 225     def validate_accountRateLimit(self, value):
 226         if not self._has_legacy_rate_limits:
 227             raise serializers.ValidationError(
 228                 "The accountRateLimit option cannot be configured for this organization"
 229             )
 230         return value
 231 
 232     def validate_projectRateLimit(self, value):
 233         if not self._has_legacy_rate_limits:
 234             raise serializers.ValidationError(
 235                 "The accountRateLimit option cannot be configured for this organization"
 236             )
 237         return value

 239     def validate(self, attrs):
 240         attrs = super(OrganizationSerializer, self).validate(attrs)
 241         if attrs.get("avatarType") == "upload":
 242             has_existing_file = OrganizationAvatar.objects.filter(
 243                 organization=self.context["organization"], file__isnull=False
 244             ).exists()
 245             if not has_existing_file and not attrs.get("avatar"):
 246                 raise serializers.ValidationError(
 247                     {"avatarType": "Cannot set avatarType to upload without avatar"}
 248                 )
 249         return attrs

 251     def save(self):
 252         org = self.context["organization"]
 253         changed_data = {}
 254 
 255         for key, option, type_, default_value in ORG_OPTIONS:
 256             if key not in self.initial_data:
 257                 continue
 258             try:
 259                 option_inst = OrganizationOption.objects.get(organization=org, key=option)
 260             except OrganizationOption.DoesNotExist:
 261                 OrganizationOption.objects.set_value(
 262                     organization=org, key=option, value=type_(self.initial_data[key])
 263                 )
 264 
 265                 if self.initial_data[key] != default_value:
 266                     changed_data[key] = u"to {}".format(self.initial_data[key])
 267             else:
 268                 option_inst.value = self.initial_data[key]
 269                 # check if ORG_OPTIONS changed
 270                 if option_inst.has_changed("value"):
 271                     old_val = option_inst.old_value("value")
 272                     changed_data[key] = u"from {} to {}".format(old_val, option_inst.value)
 273                 option_inst.save()
 274 
 275         if "openMembership" in self.initial_data:
 276             org.flags.allow_joinleave = self.initial_data["openMembership"]
 277         if "allowSharedIssues" in self.initial_data:
 278             org.flags.disable_shared_issues = not self.initial_data["allowSharedIssues"]
 279         if "enhancedPrivacy" in self.initial_data:
 280             org.flags.enhanced_privacy = self.initial_data["enhancedPrivacy"]
 281         if "isEarlyAdopter" in self.initial_data:
 282             org.flags.early_adopter = self.initial_data["isEarlyAdopter"]
 283         if "require2FA" in self.initial_data:
 284             org.flags.require_2fa = self.initial_data["require2FA"]
 285         if "name" in self.initial_data:
 286             org.name = self.initial_data["name"]
 287         if "slug" in self.initial_data:
 288             org.slug = self.initial_data["slug"]
 289 
 290         org_tracked_field = {
 291             "name": org.name,
 292             "slug": org.slug,
 293             "default_role": org.default_role,
 294             "flag_field": {
 295                 "allow_joinleave": org.flags.allow_joinleave.is_set,
 296                 "enhanced_privacy": org.flags.enhanced_privacy.is_set,
 297                 "disable_shared_issues": org.flags.disable_shared_issues.is_set,
 298                 "early_adopter": org.flags.early_adopter.is_set,
 299                 "require_2fa": org.flags.require_2fa.is_set,
 300             },
 301         }
 302 
 303         # check if fields changed
 304         for f, v in six.iteritems(org_tracked_field):
 305             if f != "flag_field":
 306                 if org.has_changed(f):
 307                     old_val = org.old_value(f)
 308                     changed_data[f] = u"from {} to {}".format(old_val, v)
 309             else:
 310                 # check if flag fields changed
 311                 for f, v in six.iteritems(org_tracked_field["flag_field"]):
 312                     if org.flag_has_changed(f):
 313                         changed_data[f] = u"to {}".format(v)
 314 
 315         org.save()
 316 
 317         if "avatar" in self.initial_data or "avatarType" in self.initial_data:
 318             OrganizationAvatar.save_avatar(
 319                 relation={"organization": org},
 320                 type=self.initial_data.get("avatarType", "upload"),
 321                 avatar=self.initial_data.get("avatar"),
 322                 filename=u"{}.png".format(org.slug),
 323             )
 324         if "require2FA" in self.initial_data and self.initial_data["require2FA"] is True:
 325             org.handle_2fa_required(self.context["request"])
 326         return org, changed_data

 329 class OwnerOrganizationSerializer(OrganizationSerializer):
 330     defaultRole = serializers.ChoiceField(choices=roles.get_choices())
 331     cancelDeletion = serializers.BooleanField(required=False)
 332 
 333     def save(self, *args, **kwargs):
 334         org = self.context["organization"]
 335         cancel_deletion = "cancelDeletion" in self.initial_data and org.status in DELETION_STATUSES
 336         if "defaultRole" in self.initial_data:
 337             org.default_role = self.initial_data["defaultRole"]
 338         if cancel_deletion:
 339             org.status = OrganizationStatus.VISIBLE
 340         return super(OwnerOrganizationSerializer, self).save(*args, **kwargs)

 343 class OrganizationDetailsEndpoint(OrganizationEndpoint):
 344     doc_section = DocSection.ORGANIZATIONS
 345 
 346     @attach_scenarios([retrieve_organization_scenario])
 347     def get(self, request, organization):
 348         """
 349         Retrieve an Organization
 350         ````````````````````````
 351 
 352         Return details on an individual organization including various details
 353         such as membership access, features, and teams.
 354 
 355         :pparam string organization_slug: the slug of the organization the
 356                                           team should be created for.
 357         :param string detailed: Specify '0' to retrieve details without projects and teams.
 358         :auth: required
 359         """
 360         is_detailed = request.GET.get("detailed", "1") != "0"
 361         serializer = (
 362             org_serializers.DetailedOrganizationSerializerWithProjectsAndTeams
 363             if is_detailed
 364             else org_serializers.DetailedOrganizationSerializer
 365         )
 366         context = serialize(organization, request.user, serializer(), access=request.access)
 367         return self.respond(context)
 368 
 369     @attach_scenarios([update_organization_scenario])
 370     def put(self, request, organization):
 371         """
 372         Update an Organization
 373         ``````````````````````
 374 
 375         Update various attributes and configurable settings for the given
 376         organization.
 377 
 378         :pparam string organization_slug: the slug of the organization the
 379                                           team should be created for.
 380         :param string name: an optional new name for the organization.
 381         :param string slug: an optional new slug for the organization.  Needs
 382                             to be available and unique.
 383         :auth: required
 384         """
 385         if request.access.has_scope("org:admin"):
 386             serializer_cls = OwnerOrganizationSerializer
 387         else:
 388             serializer_cls = OrganizationSerializer
 389 
 390         was_pending_deletion = organization.status in DELETION_STATUSES
 391 
 392         serializer = serializer_cls(
 393             data=request.data,
 394             partial=True,
 395             context={"organization": organization, "user": request.user, "request": request},
 396         )
 397         if serializer.is_valid():
 398             organization, changed_data = serializer.save()
 399 
 400             if was_pending_deletion:
 401                 self.create_audit_entry(
 402                     request=request,
 403                     organization=organization,
 404                     target_object=organization.id,
 405                     event=AuditLogEntryEvent.ORG_RESTORE,
 406                     data=organization.get_audit_log_data(),
 407                 )
 408                 delete_logger.info(
 409                     "object.delete.canceled",
 410                     extra={"object_id": organization.id, "model": Organization.__name__},
 411                 )
 412             elif changed_data:
 413                 self.create_audit_entry(
 414                     request=request,
 415                     organization=organization,
 416                     target_object=organization.id,
 417                     event=AuditLogEntryEvent.ORG_EDIT,
 418                     data=changed_data,
 419                 )
 420 
 421             return self.respond(
 422                 serialize(
 423                     organization,
 424                     request.user,
 425                     org_serializers.DetailedOrganizationSerializerWithProjectsAndTeams(),
 426                     access=request.access,
 427                 )
 428             )
 429         return self.respond(serializer.errors, status=status.HTTP_400_BAD_REQUEST)

 431     def handle_delete(self, request, organization):
 432         """
 433         This method exists as a way for getsentry to override this endpoint with less duplication.
 434         """
 435         if not request.user.is_authenticated():
 436             return self.respond({"detail": ERR_NO_USER}, status=401)
 437         if organization.is_default:
 438             return self.respond({"detail": ERR_DEFAULT_ORG}, status=400)
 439         updated = Organization.objects.filter(
 440             id=organization.id, status=OrganizationStatus.VISIBLE
 441         ).update(status=OrganizationStatus.PENDING_DELETION)
 442         if updated:
 443             transaction_id = uuid4().hex
 444             countdown = 86400
 445             entry = self.create_audit_entry(
 446                 request=request,
 447                 organization=organization,
 448                 target_object=organization.id,
 449                 event=AuditLogEntryEvent.ORG_REMOVE,
 450                 data=organization.get_audit_log_data(),
 451                 transaction_id=transaction_id,
 452             )
 453             organization.send_delete_confirmation(entry, countdown)
 454             delete_organization.apply_async(
 455                 kwargs={
 456                     "object_id": organization.id,
 457                     "transaction_id": transaction_id,
 458                     "actor_id": request.user.id,
 459                 },
 460                 countdown=countdown,
 461             )
 462             delete_logger.info(
 463                 "object.delete.queued",
 464                 extra={
 465                     "object_id": organization.id,
 466                     "transaction_id": transaction_id,
 467                     "model": Organization.__name__,
 468                 },
 469             )
 470         context = serialize(
 471             organization,
 472             request.user,
 473             org_serializers.DetailedOrganizationSerializerWithProjectsAndTeams(),
 474             access=request.access,
 475         )
 476         return self.respond(context, status=202)
 477 
 478     @sudo_required
 479     def delete(self, request, organization):
 480         """
 481         Delete an Organization
 482         ``````````````````````
 483         Schedules an organization for deletion.  This API endpoint cannot
 484         be invoked without a user context for security reasons.  This means
 485         that at present an organization can only be deleted from the
 486         Sentry UI.
 487         Deletion happens asynchronously and therefore is not immediate.
 488         However once deletion has begun the state of an organization changes and
 489         will be hidden from most public views.
 490         :pparam string organization_slug: the slug of the organization the
 491                                           team should be created for.
 492         :auth: required, user-context-needed
 493         """
 494         return self.handle_delete(request, organization)
