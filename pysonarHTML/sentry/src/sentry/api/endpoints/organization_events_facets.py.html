  16 class OrganizationEventsFacetsEndpoint(OrganizationEventsEndpointBase):
  17     def get(self, request, organization):
  18         with sentry_sdk.start_span(op="discover.endpoint", description="filter_params") as span:
  19             span.set_data("organization", organization)
  20             if not self.has_feature(organization, request):
  21                 return Response(status=404)
  22             try:
  23                 params = self.get_filter_params(request, organization)
  24             except NoProjects:
  25                 return Response([])
  26             params = self.quantize_date_params(request, params)
  27             self._validate_project_ids(request, organization, params)
  28 
  29         with sentry_sdk.start_span(op="discover.endpoint", description="discover_query"):
  30             try:
  31                 facets = discover.get_facets(
  32                     query=request.GET.get("query"),
  33                     params=params,
  34                     referrer="api.organization-events-facets.top-tags",
  35                 )
  36             except (discover.InvalidSearchQuery, snuba.QueryOutsideRetentionError) as error:
  37                 raise ParseError(detail=six.text_type(error))
  38 
  39         with sentry_sdk.start_span(op="discover.endpoint", description="populate_results") as span:
  40             span.set_data("facet_count", len(facets or []))
  41             resp = defaultdict(lambda: {"key": "", "topValues": []})
  42             for row in facets:
  43                 values = resp[row.key]
  44                 values["key"] = tagstore.get_standardized_key(row.key)
  45                 values["topValues"].append(
  46                     {
  47                         "name": tagstore.get_tag_value_label(row.key, row.value),
  48                         "value": row.value,
  49                         "count": row.count,
  50                     }
  51                 )
  52             if "project" in resp:
  53                 # Replace project ids with slugs as that is what we generally expose to users
  54                 # and filter out projects that the user doesn't have access too.
  55                 projects = {p.id: p.slug for p in self.get_projects(request, organization)}
  56                 filtered_values = []
  57                 for v in resp["project"]["topValues"]:
  58                     if v["value"] in projects:
  59                         name = projects[v["value"]]
  60                         v.update({"name": name})
  61                         filtered_values.append(v)
  62 
  63                 resp["project"]["topValues"] = filtered_values
  64 
  65         return Response(resp.values())

  67     def _validate_project_ids(self, request, organization, params):
  68         project_ids = params["project_id"]
  69 
  70         has_global_views = features.has(
  71             "organizations:global-views", organization, actor=request.user
  72         )
  73 
  74         if not has_global_views and len(project_ids) > 1:
  75             raise ParseError(detail="You cannot view events from multiple projects.")
  76 
  77         return project_ids
