  13 class AcceptOrganizationInvite(Endpoint):
  14     # Disable authentication and permission requirements.
  15     permission_classes = []
  16 
  17     def respond_invalid(self, request):
  18         return Response(status=status.HTTP_400_BAD_REQUEST, data={"details": "Invalid invite code"})
  19 
  20     def get_helper(self, request, member_id, token):
  21         return ApiInviteHelper(request=request, member_id=member_id, instance=self, token=token)
  22 
  23     def get(self, request, member_id, token):
  24         try:
  25             helper = self.get_helper(request, member_id, token)
  26         except OrganizationMember.DoesNotExist:
  27             return self.respond_invalid(request)
  28 
  29         om = helper.om
  30         organization = om.organization
  31 
  32         if not helper.member_pending or not helper.valid_token or not om.invite_approved:
  33             return self.respond_invalid(request)
  34 
  35         # Keep track of the invite email for when we land back on the login page
  36         request.session["invite_email"] = om.email
  37 
  38         try:
  39             auth_provider = AuthProvider.objects.get(organization=organization)
  40         except AuthProvider.DoesNotExist:
  41             auth_provider = None
  42 
  43         data = {
  44             "orgSlug": organization.slug,
  45             "needsAuthentication": not helper.user_authenticated,
  46             "needs2fa": helper.needs_2fa,
  47             "needsSso": auth_provider is not None,
  48             # If they're already a member of the organization its likely
  49             # they're using a shared account and either previewing this invite
  50             # or are incorrectly expecting this to create a new account.
  51             "existingMember": helper.member_already_exists,
  52         }
  53 
  54         response = Response(None)
  55 
  56         # Allow users to register an account when accepting an invite
  57         if not helper.user_authenticated:
  58             request.session["can_register"] = True
  59             add_invite_cookie(request, response, member_id, token)
  60 
  61             # When SSO is required do *not* set a next_url to return to accept
  62             # invite. The invite will be accepted after SSO is completed.
  63             url = (
  64                 reverse("sentry-accept-invite", args=[member_id, token])
  65                 if not auth_provider
  66                 else "/"
  67             )
  68             auth.initiate_login(self.request, next_url=url)
  69 
  70         # If the org has SSO setup, we'll store the invite cookie to later
  71         # associate the org member after authentication. We can avoid needing
  72         # to come back to the accept invite page since 2FA will *not* be
  73         # required if SSO is required.
  74         if auth_provider is not None:
  75             add_invite_cookie(request, response, member_id, token)
  76             provider = auth_provider.get_provider()
  77             data["ssoProvider"] = provider.name
  78 
  79         if helper.needs_2fa:
  80             add_invite_cookie(request, response, member_id, token)
  81 
  82         response.data = data
  83 
  84         return response
  85 
  86     def post(self, request, member_id, token):
  87         try:
  88             helper = self.get_helper(request, member_id, token)
  89         except OrganizationMember.DoesNotExist:
  90             return self.respond_invalid(request)
  91 
  92         if not helper.valid_request:
  93             return Response(
  94                 status=status.HTTP_400_BAD_REQUEST,
  95                 data={"details": "unable to accept organization invite"},
  96             )
  97 
  98         if helper.member_already_exists:
  99             response = Response(
 100                 status=status.HTTP_400_BAD_REQUEST, data={"details": "member already exists"}
 101             )
 102         else:
 103             response = Response(status=status.HTTP_204_NO_CONTENT)
 104 
 105         helper.accept_invite()
 106         remove_invite_cookie(request, response)
 107 
 108         return response
