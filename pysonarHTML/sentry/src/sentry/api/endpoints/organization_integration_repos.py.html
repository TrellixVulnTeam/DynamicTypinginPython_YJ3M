  15     def get(self, request, organization, integration_id):
  16         """
  17         Get the list of repositories available in an integration
  18         ````````````````````````````````````````````````````````
  19 
  20         Gets all repositories that an integration makes available,
  21         and indicates whether or not you can search repositories
  22         by name.
  23 
  24         :qparam string search: Name fragment to search repositories by.
  25         """
  26         integration = self.get_integration(organization, integration_id)
  27 
  28         if integration.status == ObjectStatus.DISABLED:
  29             context = {"repos": []}
  30             return self.respond(context)
  31 
  32         install = integration.get_installation(organization.id)
  33 
  34         if isinstance(install, RepositoryMixin):
  35             try:
  36                 repositories = install.get_repositories(request.GET.get("search"))
  37             except IntegrationError as e:
  38                 return self.respond({"detail": six.text_type(e)}, status=400)
  39 
  40             context = {"repos": repositories, "searchable": install.repo_search}
  41             return self.respond(context)
  42 
  43         return self.respond({"detail": "Repositories not supported"}, status=400)
