  19 class OrganizationUserReportsEndpoint(OrganizationEndpoint):
  20     permission_classes = (OrganizationUserReportsPermission,)
  21 
  22     @attach_scenarios([list_org_user_reports_scenario])
  23     def get(self, request, organization):
  24         """
  25         List an Organization's User Feedback
  26         ``````````````````````````````
  27 
  28         Return a list of user feedback items within this organization. Can be
  29         filtered by projects/environments/creation date.
  30 
  31         :pparam string organization_slug: the slug of the organization.
  32         :pparam string project_slug: the slug of the project.
  33         :auth: required
  34         """
  35         try:
  36             filter_params = self.get_filter_params(request, organization, date_filter_optional=True)
  37         except NoProjects:
  38             return Response([])
  39 
  40         queryset = UserReport.objects.filter(
  41             project_id__in=filter_params["project_id"], group__isnull=False
  42         ).select_related("group")
  43         if "environment" in filter_params:
  44             queryset = queryset.filter(environment__name__in=filter_params["environment"])
  45         if filter_params["start"] and filter_params["end"]:
  46             queryset = queryset.filter(
  47                 date_added__range=(filter_params["start"], filter_params["end"])
  48             )
  49 
  50         status = request.GET.get("status", "unresolved")
  51         if status == "unresolved":
  52             queryset = queryset.filter(group__status=GroupStatus.UNRESOLVED)
  53         elif status:
  54             return self.respond({"status": "Invalid status choice"}, status=400)
  55 
  56         return self.paginate(
  57             request=request,
  58             queryset=queryset,
  59             order_by="-date_added",
  60             on_results=lambda x: serialize(x, request.user, UserReportWithGroupSerializer()),
  61             paginator_cls=DateTimePaginator,
  62         )
