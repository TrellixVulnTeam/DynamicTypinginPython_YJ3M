  53 class OrganizationMemberSerializer(serializers.Serializer):
  54     email = AllowedEmailField(max_length=75, required=True)
  55     role = serializers.ChoiceField(choices=roles.get_choices(), required=True)
  56     teams = ListField(required=False, allow_null=False, default=[])
  57     sendInvite = serializers.BooleanField(required=False, default=True, write_only=True)

  59     def validate_email(self, email):
  60         queryset = OrganizationMember.objects.filter(
  61             Q(email=email) | Q(user__email__iexact=email, user__is_active=True),
  62             organization=self.context["organization"],
  63         )
  64 
  65         if queryset.filter(invite_status=InviteStatus.APPROVED.value).exists():
  66             raise serializers.ValidationError("The user %s is already a member" % email)
  67 
  68         if not self.context.get("allow_existing_invite_request"):
  69             if queryset.filter(
  70                 Q(invite_status=InviteStatus.REQUESTED_TO_BE_INVITED.value)
  71                 | Q(invite_status=InviteStatus.REQUESTED_TO_JOIN.value)
  72             ).exists():
  73                 raise serializers.ValidationError(
  74                     "There is an existing invite request for %s" % email
  75                 )
  76         return email
  77 
  78     def validate_teams(self, teams):
  79         valid_teams = list(
  80             Team.objects.filter(
  81                 organization=self.context["organization"], status=TeamStatus.VISIBLE, slug__in=teams
  82             )
  83         )
  84 
  85         if len(valid_teams) != len(teams):
  86             raise serializers.ValidationError("Invalid teams")
  87 
  88         return valid_teams
  89 
  90     def validate_role(self, role):
  91         if role not in {r.id for r in self.context["allowed_roles"]}:
  92             raise serializers.ValidationError("You do not have permission to invite that role.")
  93 
  94         return role

 100     def get(self, request, organization):
 101         queryset = (
 102             OrganizationMember.objects.filter(
 103                 Q(user__is_active=True) | Q(user__isnull=True),
 104                 organization=organization,
 105                 invite_status=InviteStatus.APPROVED.value,
 106             )
 107             .select_related("user")
 108             .order_by("email", "user__email")
 109         )
 110 
 111         query = request.GET.get("query")
 112 
 113         if query:
 114             tokens = tokenize_query(query)
 115             for key, value in six.iteritems(tokens):
 116                 if key == "email":
 117                     queryset = queryset.filter(
 118                         Q(email__in=value)
 119                         | Q(user__email__in=value)
 120                         | Q(user__emails__email__in=value)
 121                     )
 122 
 123                 elif key == "scope":
 124                     queryset = queryset.filter(role__in=[r.id for r in roles.with_any_scope(value)])
 125 
 126                 elif key == "role":
 127                     queryset = queryset.filter(role__in=value)
 128 
 129                 elif key == "isInvited":
 130                     isInvited = "true" in value
 131                     queryset = queryset.filter(user__isnull=isInvited)
 132 
 133                 elif key == "ssoLinked":
 134                     ssoFlag = OrganizationMember.flags["sso:linked"]
 135                     ssoLinked = "true" in value
 136                     if ssoLinked:
 137                         queryset = queryset.filter(flags=F("flags").bitor(ssoFlag))
 138                     else:
 139                         queryset = queryset.filter(flags=F("flags").bitand(~ssoFlag))
 140 
 141                 elif key == "has2fa":
 142                     has2fa = "true" in value
 143                     if has2fa:
 144                         types = [a.type for a in available_authenticators(ignore_backup=True)]
 145                         queryset = queryset.filter(
 146                             user__authenticator__isnull=False, user__authenticator__type__in=types
 147                         ).distinct()
 148                     else:
 149                         queryset = queryset.filter(user__authenticator__isnull=True)
 150 
 151                 elif key == "query":
 152                     value = " ".join(value)
 153                     queryset = queryset.filter(
 154                         Q(email__icontains=value)
 155                         | Q(user__email__icontains=value)
 156                         | Q(user__name__icontains=value)
 157                     )
 158                 else:
 159                     queryset = queryset.none()
 160 
 161         return self.paginate(
 162             request=request,
 163             queryset=queryset,
 164             on_results=lambda x: serialize(x, request.user),
 165             paginator_cls=OffsetPaginator,
 166         )

 168     def post(self, request, organization):
 169         """
 170         Add a Member to Organization
 171         ````````````````````````````
 172 
 173         Invite a member to the organization.
 174 
 175         :pparam string organization_slug: the slug of the organization the member will belong to
 176         :param string email: the email address to invite
 177         :param string role: the role of the new member
 178         :param array teams: the slugs of the teams the member should belong to.
 179 
 180         :auth: required
 181         """
 182         if not features.has("organizations:invite-members", organization, actor=request.user):
 183             return Response(
 184                 {"organization": "Your organization is not allowed to invite members"}, status=403
 185             )
 186 
 187         _, allowed_roles = get_allowed_roles(request, organization)
 188 
 189         serializer = OrganizationMemberSerializer(
 190             data=request.data,
 191             context={
 192                 "organization": organization,
 193                 "allowed_roles": allowed_roles,
 194                 "allow_existing_invite_request": True,
 195             },
 196         )
 197 
 198         if not serializer.is_valid():
 199             return Response(serializer.errors, status=400)
 200 
 201         result = serializer.validated_data
 202 
 203         with transaction.atomic():
 204             # remove any invitation requests for this email before inviting
 205             OrganizationMember.objects.filter(
 206                 Q(invite_status=InviteStatus.REQUESTED_TO_BE_INVITED.value)
 207                 | Q(invite_status=InviteStatus.REQUESTED_TO_JOIN.value),
 208                 email=result["email"],
 209                 organization=organization,
 210             ).delete()
 211 
 212             om = OrganizationMember(
 213                 organization=organization,
 214                 email=result["email"],
 215                 role=result["role"],
 216                 inviter=request.user,
 217             )
 218 
 219             if settings.SENTRY_ENABLE_INVITES:
 220                 om.token = om.generate_token()
 221             om.save()
 222 
 223         if result["teams"]:
 224             lock = locks.get(u"org:member:{}".format(om.id), duration=5)
 225             with TimedRetryPolicy(10)(lock.acquire):
 226                 save_team_assignments(om, result["teams"])
 227 
 228         if settings.SENTRY_ENABLE_INVITES and result.get("sendInvite"):
 229             om.send_invite_email()
 230             member_invited.send_robust(
 231                 member=om, user=request.user, sender=self, referrer=request.data.get("referrer")
 232             )
 233 
 234         self.create_audit_entry(
 235             request=request,
 236             organization_id=organization.id,
 237             target_object=om.id,
 238             data=om.get_audit_log_data(),
 239             event=AuditLogEntryEvent.MEMBER_INVITE
 240             if settings.SENTRY_ENABLE_INVITES
 241             else AuditLogEntryEvent.MEMBER_ADD,
 242         )
 243 
 244         return Response(serialize(om), status=201)
