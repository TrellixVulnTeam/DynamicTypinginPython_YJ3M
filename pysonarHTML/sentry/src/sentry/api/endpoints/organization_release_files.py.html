  20 def load_dist(results):
  21     # Dists are pretty uncommon.  In case they do appear load them now
  22     # as trying to join this on the DB does terrible things with large
  23     # offsets (it would otherwise generate a left outer join).
  24     dists = dict.fromkeys(x.dist_id for x in results)
  25     if not dists:
  26         return results
  27 
  28     for dist in Distribution.objects.filter(pk__in=dists.keys()):
  29         dists[dist.id] = dist
  30 
  31     for result in results:
  32         if result.dist_id is not None:
  33             dist = dists.get(result.dist_id)
  34             if dist is not None:
  35                 result.dist = dist
  36 
  37     return results

  40 class OrganizationReleaseFilesEndpoint(OrganizationReleasesBaseEndpoint):
  41     doc_section = DocSection.RELEASES

  43     def get(self, request, organization, version):
  44         """
  45         List an Organization Release's Files
  46         ````````````````````````````````````
  47 
  48         Retrieve a list of files for a given release.
  49 
  50         :pparam string organization_slug: the slug of the organization the
  51                                           release belongs to.
  52         :pparam string version: the version identifier of the release.
  53         :auth: required
  54         """
  55         try:
  56             release = Release.objects.get(organization_id=organization.id, version=version)
  57         except Release.DoesNotExist:
  58             raise ResourceDoesNotExist
  59 
  60         if not self.has_release_permission(request, organization, release):
  61             raise ResourceDoesNotExist
  62 
  63         file_list = (
  64             ReleaseFile.objects.filter(release=release).select_related("file").order_by("name")
  65         )
  66 
  67         return self.paginate(
  68             request=request,
  69             queryset=file_list,
  70             order_by="name",
  71             paginator_cls=OffsetPaginator,
  72             max_offset=MAX_RELEASE_FILES_OFFSET,
  73             on_results=lambda r: serialize(load_dist(r), request.user),
  74         )

  76     def post(self, request, organization, version):
  77         """
  78         Upload a New Organization Release File
  79         ``````````````````````````````````````
  80 
  81         Upload a new file for the given release.
  82 
  83         Unlike other API requests, files must be uploaded using the
  84         traditional multipart/form-data content-type.
  85 
  86         The optional 'name' attribute should reflect the absolute path
  87         that this file will be referenced as. For example, in the case of
  88         JavaScript you might specify the full web URI.
  89 
  90         :pparam string organization_slug: the slug of the organization the
  91                                           release belongs to.
  92         :pparam string version: the version identifier of the release.
  93         :param string name: the name (full path) of the file.
  94         :param file file: the multipart encoded file.
  95         :param string dist: the name of the dist.
  96         :param string header: this parameter can be supplied multiple times
  97                               to attach headers to the file.  Each header
  98                               is a string in the format ``key:value``.  For
  99                               instance it can be used to define a content
 100                               type.
 101         :auth: required
 102         """
 103         try:
 104             release = Release.objects.get(organization_id=organization.id, version=version)
 105         except Release.DoesNotExist:
 106             raise ResourceDoesNotExist
 107 
 108         logger = logging.getLogger("sentry.files")
 109         logger.info("organizationreleasefile.start")
 110 
 111         if not self.has_release_permission(request, organization, release):
 112             raise ResourceDoesNotExist
 113 
 114         if "file" not in request.data:
 115             return Response({"detail": "Missing uploaded file"}, status=400)
 116 
 117         fileobj = request.data["file"]
 118 
 119         full_name = request.data.get("name", fileobj.name)
 120         if not full_name or full_name == "file":
 121             return Response({"detail": "File name must be specified"}, status=400)
 122 
 123         name = full_name.rsplit("/", 1)[-1]
 124 
 125         if _filename_re.search(name):
 126             return Response(
 127                 {"detail": "File name must not contain special whitespace characters"}, status=400
 128             )
 129 
 130         dist_name = request.data.get("dist")
 131         dist = None
 132         if dist_name:
 133             dist = release.add_dist(dist_name)
 134 
 135         headers = {"Content-Type": fileobj.content_type}
 136         for headerval in request.data.getlist("header") or ():
 137             try:
 138                 k, v = headerval.split(":", 1)
 139             except ValueError:
 140                 return Response({"detail": "header value was not formatted correctly"}, status=400)
 141             else:
 142                 if _filename_re.search(v):
 143                     return Response(
 144                         {"detail": "header value must not contain special whitespace characters"},
 145                         status=400,
 146                     )
 147                 headers[k] = v.strip()
 148 
 149         file = File.objects.create(name=name, type="release.file", headers=headers)
 150         file.putfile(fileobj, logger=logger)
 151 
 152         try:
 153             with transaction.atomic():
 154                 releasefile = ReleaseFile.objects.create(
 155                     organization_id=release.organization_id,
 156                     release=release,
 157                     file=file,
 158                     name=full_name,
 159                     dist=dist,
 160                 )
 161         except IntegrityError:
 162             file.delete()
 163             return Response({"detail": ERR_FILE_EXISTS}, status=409)
 164 
 165         return Response(serialize(releasefile, request.user), status=201)
