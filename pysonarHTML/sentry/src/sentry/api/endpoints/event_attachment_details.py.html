  16 class EventAttachmentDetailsPermission(ProjectPermission):
  17     def has_object_permission(self, request, view, project):
  18         result = super(EventAttachmentDetailsPermission, self).has_object_permission(
  19             request, view, project
  20         )
  21 
  22         if not result:
  23             return result
  24 
  25         if is_system_auth(request.auth) or is_active_superuser(request):
  26             return True
  27 
  28         if not request.user.is_authenticated():
  29             return False
  30 
  31         organization = project.organization
  32         required_role = (
  33             organization.get_option("sentry:attachments_role") or ATTACHMENTS_ROLE_DEFAULT
  34         )
  35 
  36         try:
  37             current_role = (
  38                 OrganizationMember.objects.filter(organization=organization, user=request.user)
  39                 .values_list("role", flat=True)
  40                 .get()
  41             )
  42         except OrganizationMember.DoesNotExist:
  43             return False
  44 
  45         required_role = roles.get(required_role)
  46         current_role = roles.get(current_role)
  47         return current_role.priority >= required_role.priority

  53     def download(self, attachment):
  54         file = attachment.file
  55         fp = file.getfile()
  56         response = StreamingHttpResponse(
  57             iter(lambda: fp.read(4096), b""),
  58             content_type=file.headers.get("content-type", "application/octet-stream"),
  59         )
  60         response["Content-Length"] = file.size
  61         response["Content-Disposition"] = 'attachment; filename="%s"' % posixpath.basename(
  62             " ".join(attachment.name.split())
  63         )
  64         return response

  66     def get(self, request, project, event_id, attachment_id):
  67         """
  68         Retrieve an Attachment
  69         ``````````````````````
  70 
  71         :pparam string organization_slug: the slug of the organization the
  72                                           issues belong to.
  73         :pparam string project_slug: the slug of the project the event
  74                                      belongs to.
  75         :pparam string event_id: the id of the event.
  76         :pparam string attachment_id: the id of the attachment.
  77         :auth: required
  78         """
  79         if not features.has(
  80             "organizations:event-attachments", project.organization, actor=request.user
  81         ):
  82             return self.respond(status=404)
  83 
  84         event = eventstore.get_event_by_id(project.id, event_id)
  85         if event is None:
  86             return self.respond({"detail": "Event not found"}, status=404)
  87 
  88         try:
  89             attachment = (
  90                 EventAttachment.objects.filter(
  91                     project_id=project.id, event_id=event.event_id, id=attachment_id
  92                 )
  93                 .select_related("file")
  94                 .get()
  95             )
  96         except EventAttachment.DoesNotExist:
  97             return self.respond({"detail": "Attachment not found"}, status=404)
  98 
  99         if request.GET.get("download") is not None:
 100             return self.download(attachment)
 101 
 102         return self.respond(
 103             {
 104                 "id": six.text_type(attachment.id),
 105                 "name": attachment.name,
 106                 "headers": attachment.file.headers,
 107                 "size": attachment.file.size,
 108                 "sha1": attachment.file.checksum,
 109                 "dateCreated": attachment.file.timestamp,
 110             }
 111         )

 113     def delete(self, request, project, event_id, attachment_id):
 114         """
 115         Delete an Event Attachment by ID
 116         ````````````````````````````````
 117 
 118         Delete an attachment on the given event.
 119 
 120         :pparam string event_id: the identifier of the event.
 121         :pparam string attachment_id: the identifier of the attachment.
 122         :auth: required
 123         """
 124         if not features.has(
 125             "organizations:event-attachments", project.organization, actor=request.user
 126         ):
 127             return self.respond(status=404)
 128 
 129         try:
 130             attachment = (
 131                 EventAttachment.objects.filter(
 132                     project_id=project.id, event_id=event_id, id=attachment_id
 133                 )
 134                 .select_related("file")
 135                 .get()
 136             )
 137         except EventAttachment.DoesNotExist:
 138             return self.respond({"detail": "Attachment not found"}, status=404)
 139 
 140         attachment.delete()
 141         return self.respond(status=204)
