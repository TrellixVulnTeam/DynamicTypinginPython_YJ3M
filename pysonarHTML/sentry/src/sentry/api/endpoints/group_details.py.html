  77     def _get_activity(self, request, group, num):
  78         activity_items = set()
  79         activity = []
  80         activity_qs = (
  81             Activity.objects.filter(group=group).order_by("-datetime").select_related("user")
  82         )
  83         # we select excess so we can filter dupes
  84         for item in activity_qs[: num * 2]:
  85             sig = (item.type, item.ident, item.user_id)
  86             # TODO: we could just generate a signature (hash(text)) for notes
  87             # so there's no special casing
  88             if item.type == Activity.NOTE:
  89                 activity.append(item)
  90             elif sig not in activity_items:
  91                 activity_items.add(sig)
  92                 activity.append(item)
  93 
  94         activity.append(
  95             Activity(
  96                 id=0,
  97                 project=group.project,
  98                 group=group,
  99                 type=Activity.FIRST_SEEN,
 100                 datetime=group.first_seen,
 101             )
 102         )
 103 
 104         return activity[:num]

 106     def _get_seen_by(self, request, group):
 107         seen_by = list(
 108             GroupSeen.objects.filter(group=group).select_related("user").order_by("-last_seen")
 109         )
 110         return serialize(seen_by, request.user)

 112     def _get_actions(self, request, group):
 113         project = group.project
 114 
 115         action_list = []
 116         for plugin in plugins.for_project(project, version=1):
 117             results = safe_execute(
 118                 plugin.actions, request, group, action_list, _with_transaction=False
 119             )
 120 
 121             if not results:
 122                 continue
 123 
 124             action_list = results
 125 
 126         for plugin in plugins.for_project(project, version=2):
 127             for action in (
 128                 safe_execute(plugin.get_actions, request, group, _with_transaction=False) or ()
 129             ):
 130                 action_list.append(action)
 131 
 132         return action_list
 133 
 134     def _get_available_issue_plugins(self, request, group):
 135         project = group.project
 136 
 137         plugin_issues = []
 138         for plugin in plugins.for_project(project, version=1):
 139             if isinstance(plugin, IssueTrackingPlugin2):
 140                 plugin_issues = safe_execute(
 141                     plugin.plugin_issues, request, group, plugin_issues, _with_transaction=False
 142                 )
 143         return plugin_issues

 145     def _get_context_plugins(self, request, group):
 146         project = group.project
 147         return serialize(
 148             [
 149                 plugin
 150                 for plugin in plugins.for_project(project, version=None)
 151                 if plugin.has_project_conf()
 152                 and hasattr(plugin, "get_custom_contexts")
 153                 and plugin.get_custom_contexts()
 154             ],
 155             request.user,
 156             PluginSerializer(project),

 159     def _get_release_info(self, request, group, version):
 160         try:
 161             release = Release.objects.get(
 162                 projects=group.project,
 163                 organization_id=group.project.organization_id,
 164                 version=version,
 165             )
 166         except Release.DoesNotExist:
 167             release = {"version": version}
 168         return serialize(release, request.user)
 169 
 170     def _get_first_last_release_info(self, request, group, versions):
 171         releases = {
 172             release.version: release
 173             for release in Release.objects.filter(
 174                 projects=group.project,
 175                 organization_id=group.project.organization_id,
 176                 version__in=versions,
 177             )
 178         }
 179         serialized_releases = serialize(
 180             [releases.get(version) for version in versions], request.user,
 181         )
 182         # Default to a dictionary if the release object wasn't found and not serialized
 183         return [
 184             item if item is not None else {"version": version}
 185             for item, version in zip(serialized_releases, versions)
 186         ]

 188     @attach_scenarios([retrieve_aggregate_scenario])
 189     def get(self, request, group):
 190         """
 191         Retrieve an Issue
 192         `````````````````
 193 
 194         Return details on an individual issue. This returns the basic stats for
 195         the issue (title, last seen, first seen), some overall numbers (number
 196         of comments, user reports) as well as the summarized event data.
 197 
 198         :pparam string issue_id: the ID of the issue to retrieve.
 199         :auth: required
 200         """
 201         try:
 202             # TODO(dcramer): handle unauthenticated/public response
 203 
 204             organization = group.project.organization
 205             environments = get_environments(request, organization)
 206             environment_ids = [e.id for e in environments]
 207 
 208             # WARNING: the rest of this endpoint relies on this serializer
 209             # populating the cache SO don't move this :)
 210             data = serialize(
 211                 group, request.user, GroupSerializerSnuba(environment_ids=environment_ids)
 212             )
 213 
 214             # TODO: these probably should be another endpoint
 215             activity = self._get_activity(request, group, num=100)
 216             seen_by = self._get_seen_by(request, group)
 217 
 218             first_release = group.get_first_release()
 219 
 220             if first_release is not None:
 221                 last_release = group.get_last_release()
 222             else:
 223                 last_release = None
 224 
 225             action_list = self._get_actions(request, group)
 226 
 227             if first_release is not None and last_release is not None:
 228                 first_release, last_release = self._get_first_last_release_info(
 229                     request, group, [first_release, last_release]
 230                 )
 231             elif first_release is not None:
 232                 first_release = self._get_release_info(request, group, first_release)
 233             elif last_release is not None:
 234                 last_release = self._get_release_info(request, group, last_release)
 235 
 236             get_range = functools.partial(tsdb.get_range, environment_ids=environment_ids)
 237 
 238             tags = tagstore.get_group_tag_keys(
 239                 group.project_id, group.id, environment_ids, limit=100
 240             )
 241             if not environment_ids:
 242                 user_reports = UserReport.objects.filter(group=group)
 243             else:
 244                 user_reports = UserReport.objects.filter(
 245                     group=group, environment_id__in=environment_ids
 246                 )
 247 
 248             now = timezone.now()
 249             hourly_stats = tsdb.rollup(
 250                 get_range(
 251                     model=tsdb.models.group, keys=[group.id], end=now, start=now - timedelta(days=1)
 252                 ),
 253                 3600,
 254             )[group.id]
 255             daily_stats = tsdb.rollup(
 256                 get_range(
 257                     model=tsdb.models.group,
 258                     keys=[group.id],
 259                     end=now,
 260                     start=now - timedelta(days=30),
 261                 ),
 262                 3600 * 24,
 263             )[group.id]
 264 
 265             participants = list(
 266                 User.objects.filter(
 267                     groupsubscription__is_active=True, groupsubscription__group=group
 268                 )
 269             )
 270 
 271             data.update(
 272                 {
 273                     "firstRelease": first_release,
 274                     "lastRelease": last_release,
 275                     "activity": serialize(activity, request.user),
 276                     "seenBy": seen_by,
 277                     "participants": serialize(participants, request.user),
 278                     "pluginActions": action_list,
 279                     "pluginIssues": self._get_available_issue_plugins(request, group),
 280                     "pluginContexts": self._get_context_plugins(request, group),
 281                     "userReportCount": user_reports.count(),
 282                     "tags": sorted(serialize(tags, request.user), key=lambda x: x["name"]),
 283                     "stats": {"24h": hourly_stats, "30d": daily_stats},
 284                 }
 285             )
 286 
 287             # the current release is the 'latest seen' release within the
 288             # environment even if it hasnt affected this issue
 289             if environments:
 290                 with sentry_sdk.start_span(op="GroupDetailsEndpoint.get.current_release") as span:
 291                     span.set_data("Environment Count", len(environments))
 292                     span.set_data(
 293                         "Raw Parameters",
 294                         {
 295                             "group.id": group.id,
 296                             "group.project_id": group.project_id,
 297                             "group.project.organization_id": group.project.organization_id,
 298                             "environments": [{"id": e.id, "name": e.name} for e in environments],
 299                         },
 300                     )
 301 
 302                     try:
 303                         current_release = GroupRelease.objects.filter(
 304                             group_id=group.id,
 305                             environment__in=[env.name for env in environments],
 306                             release_id=(
 307                                 ReleaseEnvironment.objects.filter(
 308                                     release_id__in=ReleaseProject.objects.filter(
 309                                         project_id=group.project_id
 310                                     ).values_list("release_id", flat=True),
 311                                     organization_id=group.project.organization_id,
 312                                     environment_id__in=environment_ids,
 313                                 )
 314                                 .order_by("-first_seen")
 315                                 .values_list("release_id", flat=True)
 316                             )[:1],
 317                         )[0]
 318                     except IndexError:
 319                         current_release = None
 320 
 321                 data["currentRelease"] = serialize(
 322                     current_release, request.user, GroupReleaseWithStatsSerializer()
 323                 )
 324 
 325             metrics.incr("group.update.http_response", sample_rate=1.0, tags={"status": 200})
 326             return Response(data)
 327         except Exception:
 328             metrics.incr("group.update.http_response", sample_rate=1.0, tags={"status": 500})
 329             raise

 331     @attach_scenarios([update_aggregate_scenario])
 332     def put(self, request, group):
 333         """
 334         Update an Issue
 335         ```````````````
 336 
 337         Updates an individual issues's attributes.  Only the attributes
 338         submitted are modified.
 339 
 340         :pparam string issue_id: the ID of the group to retrieve.
 341         :param string status: the new status for the issue.  Valid values
 342                               are ``"resolved"``, ``resolvedInNextRelease``,
 343                               ``"unresolved"``, and ``"ignored"``.
 344         :param string assignedTo: the user or team that should be assigned to
 345                                   this issue. Can be of the form ``"<user_id>"``,
 346                                   ``"user:<user_id>"``, ``"<username>"``,
 347                                   ``"<user_primary_email>"``, or ``"team:<team_id>"``.
 348         :param boolean hasSeen: in case this API call is invoked with a user
 349                                 context this allows changing of the flag
 350                                 that indicates if the user has seen the
 351                                 event.
 352         :param boolean isBookmarked: in case this API call is invoked with a
 353                                      user context this allows changing of
 354                                      the bookmark flag.
 355         :param boolean isSubscribed:
 356         :param boolean isPublic: sets the issue to public or private.
 357         :auth: required
 358         """
 359         try:
 360             discard = request.data.get("discard")
 361 
 362             # TODO(dcramer): we need to implement assignedTo in the bulk mutation
 363             # endpoint
 364             response = client.put(
 365                 path=u"/projects/{}/{}/issues/".format(
 366                     group.project.organization.slug, group.project.slug
 367                 ),
 368                 params={"id": group.id},
 369                 data=request.data,
 370                 request=request,
 371             )
 372 
 373             # if action was discard, there isn't a group to serialize anymore
 374             if discard:
 375                 return response
 376 
 377             # we need to fetch the object against as the bulk mutation endpoint
 378             # only returns a delta, and object mutation returns a complete updated
 379             # entity.
 380             # TODO(dcramer): we should update the API and have this be an explicit
 381             # flag (or remove it entirely) so that delta's are the primary response
 382             # for mutation.
 383             group = Group.objects.get(id=group.id)
 384 
 385             serialized = serialize(
 386                 group,
 387                 request.user,
 388                 GroupSerializer(
 389                     environment_func=self._get_environment_func(
 390                         request, group.project.organization_id
 391                     )
 392                 ),
 393             )
 394             metrics.incr(
 395                 "group.update.http_response", sample_rate=1.0, tags={"status": response.status_code}
 396             )
 397             return Response(serialized, status=response.status_code)
 398         except client.ApiError as e:
 399             metrics.incr(
 400                 "group.update.http_response", sample_rate=1.0, tags={"status": e.status_code}
 401             )
 402             return Response(e.body, status=e.status_code)
 403         except Exception:
 404             metrics.incr("group.update.http_response", sample_rate=1.0, tags={"status": 500})
 405             raise

 407     @attach_scenarios([delete_aggregate_scenario])
 408     def delete(self, request, group):
 409         """
 410         Remove an Issue
 411         ```````````````
 412 
 413         Removes an individual issue.
 414 
 415         :pparam string issue_id: the ID of the issue to delete.
 416         :auth: required
 417         """
 418         try:
 419             from sentry.tasks.deletion import delete_groups
 420 
 421             updated = (
 422                 Group.objects.filter(id=group.id)
 423                 .exclude(
 424                     status__in=[GroupStatus.PENDING_DELETION, GroupStatus.DELETION_IN_PROGRESS]
 425                 )
 426                 .update(status=GroupStatus.PENDING_DELETION)
 427             )
 428             if updated:
 429                 project = group.project
 430 
 431                 eventstream_state = eventstream.start_delete_groups(group.project_id, [group.id])
 432                 transaction_id = uuid4().hex
 433 
 434                 GroupHash.objects.filter(project_id=group.project_id, group__id=group.id).delete()
 435 
 436                 delete_groups.apply_async(
 437                     kwargs={
 438                         "object_ids": [group.id],
 439                         "transaction_id": transaction_id,
 440                         "eventstream_state": eventstream_state,
 441                     },
 442                     countdown=3600,
 443                 )
 444 
 445                 self.create_audit_entry(
 446                     request=request,
 447                     organization_id=project.organization_id if project else None,
 448                     target_object=group.id,
 449                     transaction_id=transaction_id,
 450                 )
 451 
 452                 delete_logger.info(
 453                     "object.delete.queued",
 454                     extra={
 455                         "object_id": group.id,
 456                         "transaction_id": transaction_id,
 457                         "model": type(group).__name__,
 458                     },
 459                 )
 460 
 461                 issue_deleted.send_robust(
 462                     group=group, user=request.user, delete_type="delete", sender=self.__class__
 463                 )
 464             metrics.incr("group.update.http_response", sample_rate=1.0, tags={"status": 200})
 465             return Response(status=202)
 466         except Exception:
 467             metrics.incr("group.update.http_response", sample_rate=1.0, tags={"status": 500})
 468             raise
