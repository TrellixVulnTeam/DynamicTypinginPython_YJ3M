  23 class ApproveInviteRequestSerializer(serializers.Serializer):
  24     approve = serializers.BooleanField(required=True, write_only=True)
  25 
  26     def validate_approve(self, approve):
  27         request = self.context["request"]
  28         organization = self.context["organization"]
  29         member = self.context["member"]
  30         allowed_roles = self.context["allowed_roles"]
  31 
  32         if not features.has("organizations:invite-members", organization, actor=request.user):
  33             raise serializers.ValidationError(ERR_CANNOT_INVITE)
  34 
  35         if (
  36             organization.get_option("sentry:join_requests") is False
  37             and member.invite_status == InviteStatus.REQUESTED_TO_JOIN.value
  38         ):
  39             raise serializers.ValidationError(ERR_JOIN_REQUESTS_DISABLED)
  40 
  41         # members cannot invite roles higher than their own
  42         if member.role not in {r.id for r in allowed_roles}:
  43             raise serializers.ValidationError(ERR_INSUFFICIENT_ROLE)
  44 
  45         return approve

  59     def _get_member(self, organization, member_id):
  60         try:
  61             return OrganizationMember.objects.get(
  62                 Q(invite_status=InviteStatus.REQUESTED_TO_BE_INVITED.value)
  63                 | Q(invite_status=InviteStatus.REQUESTED_TO_JOIN.value),
  64                 organization=organization,
  65                 user__isnull=True,
  66                 id=member_id,
  67             )
  68         except ValueError:
  69             raise OrganizationMember.DoesNotExist()
  70 
  71     def get(self, request, organization, member_id):
  72         try:
  73             member = self._get_member(organization, member_id)
  74         except OrganizationMember.DoesNotExist:
  75             raise ResourceDoesNotExist
  76 
  77         return Response(
  78             serialize(member, serializer=OrganizationMemberWithTeamsSerializer()),
  79             status=status.HTTP_200_OK,
  80         )

  82     def put(self, request, organization, member_id):
  83         """
  84         Update an invite request to Organization
  85         ````````````````````````````````````````
  86 
  87         Update and/or approve an invite request to an organization.
  88 
  89         :pparam string organization_slug: the slug of the organization the member will belong to
  90         :param string member_id: the member ID
  91         :param boolean approve: allows the member to be invited
  92         :param string role: the suggested role of the new member
  93         :param array teams: the suggested slugs of the teams the member should belong to.
  94 
  95         :auth: required
  96         """
  97 
  98         try:
  99             member = self._get_member(organization, member_id)
 100         except OrganizationMember.DoesNotExist:
 101             raise ResourceDoesNotExist
 102 
 103         serializer = OrganizationMemberSerializer(
 104             data=request.data,
 105             context={"organization": organization, "allowed_roles": roles.get_all()},
 106             partial=True,
 107         )
 108 
 109         if not serializer.is_valid():
 110             return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
 111 
 112         result = serializer.validated_data
 113 
 114         if result.get("role"):
 115             member.update(role=result["role"])
 116 
 117         if "teams" in result:
 118             save_team_assignments(member, result["teams"])
 119 
 120         if "approve" in request.data:
 121             _, allowed_roles = get_allowed_roles(request, organization)
 122 
 123             serializer = ApproveInviteRequestSerializer(
 124                 data=request.data,
 125                 context={
 126                     "request": request,
 127                     "organization": organization,
 128                     "member": member,
 129                     "allowed_roles": allowed_roles,
 130                 },
 131             )
 132 
 133             if not serializer.is_valid():
 134                 return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
 135 
 136             result = serializer.validated_data
 137 
 138             if result.get("approve") and not member.invite_approved:
 139                 member.approve_invite()
 140                 member.save()
 141 
 142                 if settings.SENTRY_ENABLE_INVITES:
 143                     member.send_invite_email()
 144                     member_invited.send_robust(
 145                         member=member,
 146                         user=request.user,
 147                         sender=self,
 148                         referrer=request.data.get("referrer"),
 149                     )
 150 
 151                 self.create_audit_entry(
 152                     request=request,
 153                     organization_id=organization.id,
 154                     target_object=member.id,
 155                     data=member.get_audit_log_data(),
 156                     event=AuditLogEntryEvent.MEMBER_INVITE
 157                     if settings.SENTRY_ENABLE_INVITES
 158                     else AuditLogEntryEvent.MEMBER_ADD,
 159                 )
 160 
 161         return Response(
 162             serialize(member, serializer=OrganizationMemberWithTeamsSerializer()),
 163             status=status.HTTP_200_OK,
 164         )

 166     def delete(self, request, organization, member_id):
 167         """
 168         Delete an invite request to Organization
 169         ````````````````````````````````````````
 170 
 171         Delete an invite request to an organization.
 172 
 173         :pparam string organization_slug: the slug of the organization the member would belong to
 174         :param string member_id: the member ID
 175 
 176         :auth: required
 177         """
 178 
 179         try:
 180             member = self._get_member(organization, member_id)
 181         except OrganizationMember.DoesNotExist:
 182             raise ResourceDoesNotExist
 183 
 184         member.delete()
 185 
 186         self.create_audit_entry(
 187             request=request,
 188             organization_id=organization.id,
 189             target_object=member.id,
 190             data=member.get_audit_log_data(),
 191             event=AuditLogEntryEvent.INVITE_REQUEST_REMOVE,
 192         )
 193 
 194         return Response(status=status.HTTP_204_NO_CONTENT)
