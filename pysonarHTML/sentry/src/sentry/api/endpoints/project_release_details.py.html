  24     def get(self, request, project, version):
  25         """
  26         Retrieve a Project's Release
  27         ````````````````````````````
  28 
  29         Return details on an individual release.
  30 
  31         :pparam string organization_slug: the slug of the organization the
  32                                           release belongs to.
  33         :pparam string project_slug: the slug of the project to retrieve the
  34                                      release of.
  35         :pparam string version: the version identifier of the release.
  36         :auth: required
  37         """
  38         with_health = request.GET.get("health") == "1"
  39         summary_stats_period = request.GET.get("summaryStatsPeriod") or "14d"
  40         health_stats_period = request.GET.get("healthStatsPeriod") or ("24h" if with_health else "")
  41         if summary_stats_period not in STATS_PERIODS:
  42             raise ParseError(detail=get_stats_period_detail("summaryStatsPeriod", STATS_PERIODS))
  43         if health_stats_period and health_stats_period not in STATS_PERIODS:
  44             raise ParseError(detail=get_stats_period_detail("healthStatsPeriod", STATS_PERIODS))
  45 
  46         try:
  47             release = Release.objects.get(
  48                 organization_id=project.organization_id, projects=project, version=version
  49             )
  50         except Release.DoesNotExist:
  51             raise ResourceDoesNotExist
  52 
  53         if with_health:
  54             release._for_project_id = project.id
  55 
  56         return Response(
  57             serialize(
  58                 release,
  59                 request.user,
  60                 project=project,
  61                 with_health_data=with_health,
  62                 summary_stats_period=summary_stats_period,
  63                 health_stats_period=health_stats_period,
  64             )
  65         )

  67     def put(self, request, project, version):
  68         """
  69         Update a Project's Release
  70         ``````````````````````````
  71 
  72         Update a release.  This can change some metadata associated with
  73         the release (the ref, url, and dates).
  74 
  75         :pparam string organization_slug: the slug of the organization the
  76                                           release belongs to.
  77         :pparam string project_slug: the slug of the project to change the
  78                                      release of.
  79         :pparam string version: the version identifier of the release.
  80         :param string ref: an optional commit reference.  This is useful if
  81                            a tagged version has been provided.
  82         :param url url: a URL that points to the release.  This can be the
  83                         path to an online interface to the sourcecode
  84                         for instance.
  85         :param datetime dateReleased: an optional date that indicates when
  86                                       the release went live.  If not provided
  87                                       the current time is assumed.
  88         :auth: required
  89         """
  90         try:
  91             release = Release.objects.get(
  92                 organization_id=project.organization_id, projects=project, version=version
  93             )
  94         except Release.DoesNotExist:
  95             raise ResourceDoesNotExist
  96 
  97         serializer = ReleaseSerializer(data=request.data, partial=True)
  98 
  99         if not serializer.is_valid():
 100             return Response(serializer.errors, status=400)
 101 
 102         result = serializer.validated_data
 103 
 104         was_released = bool(release.date_released)
 105 
 106         kwargs = {}
 107         if result.get("dateReleased"):
 108             kwargs["date_released"] = result["dateReleased"]
 109         if result.get("ref"):
 110             kwargs["ref"] = result["ref"]
 111         if result.get("url"):
 112             kwargs["url"] = result["url"]
 113 
 114         if kwargs:
 115             release.update(**kwargs)
 116 
 117         commit_list = result.get("commits")
 118         if commit_list:
 119             hook = ReleaseHook(project)
 120             # TODO(dcramer): handle errors with release payloads
 121             hook.set_commits(release.version, commit_list)
 122 
 123         if not was_released and release.date_released:
 124             Activity.objects.create(
 125                 type=Activity.RELEASE,
 126                 project=project,
 127                 ident=Activity.get_version_ident(release.version),
 128                 data={"version": release.version},
 129                 datetime=release.date_released,
 130             )
 131 
 132         return Response(serialize(release, request.user))

 134     def delete(self, request, project, version):
 135         """
 136         Delete a Project's Release
 137         ``````````````````````````
 138 
 139         Permanently remove a release and all of its files.
 140 
 141         :pparam string organization_slug: the slug of the organization the
 142                                           release belongs to.
 143         :pparam string project_slug: the slug of the project to delete the
 144                                      release of.
 145         :pparam string version: the version identifier of the release.
 146         :auth: required
 147         """
 148         try:
 149             release = Release.objects.get(
 150                 organization_id=project.organization_id, projects=project, version=version
 151             )
 152         except Release.DoesNotExist:
 153             raise ResourceDoesNotExist
 154 
 155         try:
 156             release.safe_delete()
 157         except UnsafeReleaseDeletion as e:
 158             return Response({"detail": six.text_type(e)}, status=400)
 159 
 160         return Response(status=204)
