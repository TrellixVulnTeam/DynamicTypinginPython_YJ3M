  41 class GroupEventsEndpoint(GroupEndpoint, EnvironmentMixin):
  42     doc_section = DocSection.EVENTS

  44     @attach_scenarios([list_available_samples_scenario])
  45     def get(self, request, group):
  46         """
  47         List an Issue's Events
  48         ``````````````````````
  49 
  50         This endpoint lists an issue's events.
  51         :qparam bool full: if this is set to true then the event payload will
  52                            include the full event body, including the stacktrace.
  53                            Set to 1 to enable.
  54 
  55         :pparam string issue_id: the ID of the issue to retrieve.
  56 
  57         :auth: required
  58         """
  59 
  60         try:
  61             environments = get_environments(request, group.project.organization)
  62             query, tags = self._get_search_query_and_tags(request, group, environments)
  63         except InvalidQuery as exc:
  64             return Response({"detail": six.text_type(exc)}, status=400)
  65         except (NoResults, ResourceDoesNotExist):
  66             return Response([])
  67 
  68         try:
  69             start, end = get_date_range_from_params(request.GET, optional=True)
  70         except InvalidParams as e:
  71             raise ParseError(detail=six.text_type(e))
  72 
  73         try:
  74             return self._get_events_snuba(request, group, environments, query, tags, start, end)
  75         except GroupEventsError as exc:
  76             raise ParseError(detail=six.text_type(exc))

  78     def _get_events_snuba(self, request, group, environments, query, tags, start, end):
  79         default_end = timezone.now()
  80         default_start = default_end - timedelta(days=90)
  81         params = {
  82             "group_ids": [group.id],
  83             "project_id": [group.project_id],
  84             "start": start if start else default_start,
  85             "end": end if end else default_end,
  86         }
  87         direct_hit_resp = get_direct_hit_response(request, query, params, "api.group-events")
  88         if direct_hit_resp:
  89             return direct_hit_resp
  90 
  91         if environments:
  92             params["environment"] = [env.name for env in environments]
  93 
  94         full = request.GET.get("full", False)
  95         try:
  96             snuba_filter = get_filter(request.GET.get("query", None), params)
  97         except InvalidSearchQuery as e:
  98             raise ParseError(detail=six.text_type(e))
  99 
 100         snuba_filter.conditions.append(["event.type", "!=", "transaction"])
 101 
 102         data_fn = partial(eventstore.get_events, referrer="api.group-events", filter=snuba_filter)
 103         serializer = EventSerializer() if full else SimpleEventSerializer()
 104         return self.paginate(
 105             request=request,
 106             on_results=lambda results: serialize(results, request.user, serializer),
 107             paginator=GenericOffsetPaginator(data_fn=data_fn),
 108         )

 110     def _get_search_query_and_tags(self, request, group, environments=None):
 111         raw_query = request.GET.get("query")
 112 
 113         if raw_query:
 114             query_kwargs = parse_query([group.project], raw_query, request.user, environments)
 115             query = query_kwargs.pop("query", None)
 116             tags = query_kwargs.pop("tags", {})
 117         else:
 118             query = None
 119             tags = {}
 120 
 121         if environments:
 122             env_names = set(env.name for env in environments)
 123             if "environment" in tags:
 124                 # If a single environment was passed as part of the query, then
 125                 # we'll just search for that individual environment in this
 126                 # query, even if more are selected.
 127                 if tags["environment"] not in env_names:
 128                     # An event can only be associated with a single
 129                     # environment, so if the environments associated with
 130                     # the request don't contain the environment provided as a
 131                     # tag lookup, the query cannot contain any valid results.
 132                     raise NoResults
 133             else:
 134                 # XXX: Handle legacy backends here. Just store environment as a
 135                 # single tag if we only have one so that we don't break existing
 136                 # usage.
 137                 tags["environment"] = list(env_names) if len(env_names) > 1 else env_names.pop()
 138 
 139         return query, tags
