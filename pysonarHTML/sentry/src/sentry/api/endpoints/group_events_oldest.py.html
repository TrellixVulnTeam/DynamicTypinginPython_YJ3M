  20 class GroupEventsOldestEndpoint(GroupEndpoint):
  21     doc_section = DocSection.EVENTS
  22 
  23     @attach_scenarios([get_oldest_group_sample_scenario])
  24     def get(self, request, group):
  25         """
  26         Retrieve the Oldest Event for an Issue
  27         ``````````````````````````````````````
  28 
  29         Retrieves the details of the oldest event for an issue.
  30 
  31         :pparam string group_id: the ID of the issue
  32         """
  33 
  34         environments = [e.name for e in get_environments(request, group.project.organization)]
  35 
  36         event = group.get_oldest_event_for_environments(environments)
  37 
  38         if not event:
  39             return Response({"detail": "No events found for group"}, status=404)
  40 
  41         try:
  42             return client.get(
  43                 u"/projects/{}/{}/events/{}/".format(
  44                     event.organization.slug, event.project.slug, event.event_id
  45                 ),
  46                 request=request,
  47             )
  48         except client.ApiError as e:
  49             return Response(e.body, status=e.status_code)
