  23 class BroadcastIndexEndpoint(OrganizationEndpoint):
  24     permission_classes = (OrganizationPermission,)
  25 
  26     def _get_serializer(self, request):
  27         if is_active_superuser(request):
  28             return AdminBroadcastSerializer
  29         return BroadcastSerializer
  30 
  31     def _serialize_objects(self, items, request):
  32         serializer_cls = self._get_serializer(request)
  33         return serialize(items, request.user, serializer=serializer_cls())
  34 
  35     def _secondary_filtering(self, request, organization_slug, queryset):
  36         # used in the SASS product
  37         return list(queryset)

  47     def get(self, request, organization=None):
  48         if (
  49             request.GET.get("show") == "all"
  50             and is_active_superuser(request)
  51             and request.access.has_permission("broadcasts.admin")
  52         ):
  53             # superusers can slice and dice
  54             queryset = Broadcast.objects.all().order_by("-date_added")
  55         else:
  56             # only allow active broadcasts if they're not a superuser
  57             queryset = Broadcast.objects.filter(
  58                 Q(date_expires__isnull=True) | Q(date_expires__gt=timezone.now()), is_active=True
  59             ).order_by("-date_added")
  60 
  61         query = request.GET.get("query")
  62         if query:
  63             tokens = tokenize_query(query)
  64             for key, value in six.iteritems(tokens):
  65                 if key == "query":
  66                     value = " ".join(value)
  67                     queryset = queryset.filter(
  68                         Q(title__icontains=value)
  69                         | Q(message__icontains=value)
  70                         | Q(link__icontains=value)
  71                     )
  72                 elif key == "id":
  73                     queryset = queryset.filter(id__in=value)
  74                 elif key == "link":
  75                     queryset = queryset.filter(in_icontains("link", value))
  76                 elif key == "status":
  77                     filters = []
  78                     for v in value:
  79                         v = v.lower()
  80                         if v == "active":
  81                             filters.append(
  82                                 Q(date_expires__isnull=True, is_active=True)
  83                                 | Q(date_expires__gt=timezone.now(), is_active=True)
  84                             )
  85                         elif v == "inactive":
  86                             filters.append(Q(date_expires__lt=timezone.now()) | Q(is_active=False))
  87                         else:
  88                             queryset = queryset.none()
  89                     if filters:
  90                         queryset = queryset.filter(six.moves.reduce(or_, filters))
  91                 else:
  92                     queryset = queryset.none()
  93 
  94         if organization:
  95             data = self._secondary_filtering(request, organization, queryset)
  96             return self.respond(self._serialize_objects(data, request))
  97 
  98         sort_by = request.GET.get("sortBy")
  99         if sort_by == "expires":
 100             order_by = "-date_expires"
 101             paginator_cls = DateTimePaginator
 102         else:
 103             order_by = "-date_added"
 104             paginator_cls = DateTimePaginator
 105 
 106         return self.paginate(
 107             request=request,
 108             queryset=queryset,
 109             order_by=order_by,
 110             on_results=lambda x: self._serialize_objects(x, request),
 111             paginator_cls=paginator_cls,
 112         )

 114     def put(self, request):
 115         validator = BroadcastValidator(data=request.data, partial=True)
 116         if not validator.is_valid():
 117             return self.respond(validator.errors, status=400)
 118 
 119         result = validator.validated_data
 120 
 121         queryset = Broadcast.objects.filter(is_active=True)
 122 
 123         ids = request.GET.getlist("id")
 124         if ids:
 125             queryset = queryset.filter(id__in=ids)
 126 
 127         if result.get("hasSeen"):
 128             if not request.user.is_authenticated():
 129                 return self.respond(status=401)
 130 
 131             if ids:
 132                 unseen_queryset = queryset
 133             else:
 134                 unseen_queryset = queryset.exclude(
 135                     id__in=queryset.filter(broadcastseen__user=request.user).values("id")
 136                 )
 137 
 138             for broadcast in unseen_queryset:
 139                 try:
 140                     with transaction.atomic():
 141                         BroadcastSeen.objects.create(broadcast=broadcast, user=request.user)
 142                 except IntegrityError:
 143                     pass
 144 
 145         return self.respond(result)
 146 
 147     def post(self, request):
 148         if not (is_active_superuser(request) and request.access.has_permission("broadcasts.admin")):
 149             return self.respond(status=401)
 150 
 151         validator = AdminBroadcastValidator(data=request.data)
 152         if not validator.is_valid():
 153             return self.respond(validator.errors, status=400)
 154 
 155         result = validator.validated_data
 156 
 157         with transaction.atomic():
 158             broadcast = Broadcast.objects.create(
 159                 title=result["title"],
 160                 message=result["message"],
 161                 link=result["link"],
 162                 cta=result["cta"],
 163                 is_active=result.get("isActive") or False,
 164                 date_expires=result.get("dateExpires"),
 165             )
 166             logger.info(
 167                 "broadcasts.create",
 168                 extra={
 169                     "ip_address": request.META["REMOTE_ADDR"],
 170                     "user_id": request.user.id,
 171                     "broadcast_id": broadcast.id,
 172                 },
 173             )
 174 
 175         if result.get("hasSeen"):
 176             try:
 177                 with transaction.atomic():
 178                     BroadcastSeen.objects.create(broadcast=broadcast, user=request.user)
 179             except IntegrityError:
 180                 pass
 181 
 182         return self.respond(self._serialize_objects(broadcast, request))
