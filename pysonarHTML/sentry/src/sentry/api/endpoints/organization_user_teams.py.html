  13 class OrganizationUserTeamsEndpoint(OrganizationEndpoint):
  14     doc_section = DocSection.TEAMS
  15 
  16     def get(self, request, organization):
  17         """
  18         List your Teams In the Current Organization
  19         ```````````````````````````````````````````
  20 
  21         Return a list of the teams available to the authenticated session and
  22         with the supplied organization. If the user is a super user, then all
  23         teams within the organization are returned.
  24         """
  25         if is_active_superuser(request):
  26             # retrieve all teams within the organization
  27             queryset = Team.objects.filter(
  28                 organization=organization, status=TeamStatus.VISIBLE
  29             ).order_by("slug")
  30             return Response(serialize(list(queryset), request.user, TeamWithProjectsSerializer()))
  31         else:
  32             return Response(
  33                 serialize(list(request.access.teams), request.user, TeamWithProjectsSerializer())
  34             )
