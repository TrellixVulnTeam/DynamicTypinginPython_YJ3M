  41 def is_internal_relay(request, public_key):
  42     """
  43     Checks if the relay is allowed to register, otherwise raises an exception
  44     """
  45     if (
  46         settings.DEBUG
  47         or request.META.get("REMOTE_ADDR", None) in settings.INTERNAL_IPS
  48         or public_key in settings.SENTRY_RELAY_WHITELIST_PK
  49     ):
  50         return True
  51     return False

  54 class RelayRegisterChallengeEndpoint(Endpoint):
  55     authentication_classes = ()
  56     permission_classes = ()
  57 
  58     def post(self, request):
  59         """
  60         Requests to Register a Relay
  61         ````````````````````````````
  62 
  63         Registers the relay with the sentry installation.  If a relay boots
  64         it will always attempt to invoke this endpoint.
  65         """
  66         try:
  67             json_data = json.loads(request.body)
  68         except ValueError:
  69             return Response({"detail": "No valid json body"}, status=status.HTTP_400_BAD_REQUEST)
  70 
  71         serializer = RelayRegisterChallengeSerializer(data=json_data)
  72 
  73         if not serializer.is_valid():
  74             return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
  75 
  76         if not settings.SENTRY_RELAY_OPEN_REGISTRATION and not is_internal_relay(
  77             request, json_data.get("public_key")
  78         ):
  79             return Response(
  80                 {"detail": "Relay is not allowed to register"}, status=status.HTTP_401_UNAUTHORIZED
  81             )
  82 
  83         sig = get_header_relay_signature(request)
  84         if not sig:
  85             return Response(
  86                 {"detail": "Missing relay signature"}, status=status.HTTP_400_BAD_REQUEST
  87             )
  88 
  89         try:
  90             challenge = create_register_challenge(request.body, sig)
  91         except Exception as exc:
  92             return Response(
  93                 {"detail": str(exc).splitlines()[0]}, status=status.HTTP_400_BAD_REQUEST
  94             )
  95 
  96         relay_id = six.text_type(challenge["relay_id"])
  97         if relay_id != get_header_relay_id(request):
  98             return Response(
  99                 {"detail": "relay_id in payload did not match header"},
 100                 status=status.HTTP_400_BAD_REQUEST,
 101             )
 102 
 103         try:
 104             relay = Relay.objects.get(relay_id=relay_id)
 105         except Relay.DoesNotExist:
 106             pass
 107         else:
 108             if relay.public_key != six.text_type(challenge["public_key"]):
 109                 # This happens if we have an ID collision or someone copies an existing id
 110                 return Response(
 111                     {"detail": "Attempted to register agent with a different public key"},
 112                     status=status.HTTP_400_BAD_REQUEST,
 113                 )
 114 
 115         default_cache.set(
 116             "relay-auth:%s" % relay_id,
 117             {"token": challenge["token"], "public_key": six.text_type(challenge["public_key"])},
 118             60,
 119         )
 120         return Response(
 121             serialize(
 122                 {"relay_id": six.text_type(challenge["relay_id"]), "token": challenge["token"]}
 123             )
 124         )

 127 class RelayRegisterResponseEndpoint(Endpoint):
 128     authentication_classes = ()
 129     permission_classes = ()
 130 
 131     def post(self, request):
 132         """
 133         Registers a Relay
 134         `````````````````
 135 
 136         Registers the relay with the sentry installation.  If a relay boots
 137         it will always attempt to invoke this endpoint.
 138         """
 139 
 140         try:
 141             json_data = json.loads(request.body)
 142         except ValueError:
 143             return Response({"detail": "No valid json body"}, status=status.HTTP_400_BAD_REQUEST)
 144 
 145         serializer = RelayRegisterResponseSerializer(data=json_data)
 146 
 147         if not serializer.is_valid():
 148             return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
 149 
 150         sig = get_header_relay_signature(request)
 151         if not sig:
 152             return Response(
 153                 {"detail": "Missing relay signature"}, status=status.HTTP_400_BAD_REQUEST
 154             )
 155 
 156         relay_id = six.text_type(get_register_response_relay_id(request.body))
 157         if relay_id != get_header_relay_id(request):
 158             return Response(
 159                 {"detail": "relay_id in payload did not match header"},
 160                 status=status.HTTP_400_BAD_REQUEST,
 161             )
 162 
 163         params = default_cache.get("relay-auth:%s" % relay_id)
 164         if params is None:
 165             return Response({"detail": "Challenge expired"}, status=status.HTTP_401_UNAUTHORIZED)
 166 
 167         key = PublicKey.parse(params["public_key"])
 168 
 169         try:
 170             validate_register_response(key, request.body, sig)
 171         except Exception as exc:
 172             return Response(
 173                 {"detail": str(exc).splitlines()[0]}, status=status.HTTP_400_BAD_REQUEST
 174             )
 175 
 176         is_internal = is_internal_relay(request, params["public_key"])
 177         try:
 178             relay = Relay.objects.get(relay_id=relay_id)
 179         except Relay.DoesNotExist:
 180             relay = Relay.objects.create(
 181                 relay_id=relay_id, public_key=params["public_key"], is_internal=is_internal
 182             )
 183         else:
 184             relay.last_seen = timezone.now()
 185             relay.is_internal = is_internal
 186             relay.save()
 187         default_cache.delete("relay-auth:%s" % relay_id)
 188 
 189         return Response(serialize({"relay_id": relay.relay_id}))
