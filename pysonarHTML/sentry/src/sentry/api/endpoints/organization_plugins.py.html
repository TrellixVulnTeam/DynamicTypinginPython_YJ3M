  13 class OrganizationPluginsEndpoint(OrganizationEndpoint):
  14     def get(self, request, organization):
  15         all_plugins = dict([(p.slug, p) for p in plugins.all()])
  16 
  17         if "plugins" in request.GET:
  18             if request.GET.get("plugins") == "_all":
  19                 return Response(
  20                     serialize([p for p in plugins.all()], request.user, PluginSerializer())
  21                 )
  22 
  23             desired_plugins = set(request.GET.getlist("plugins"))
  24         else:
  25             desired_plugins = set(all_plugins.keys())
  26 
  27         # Ignore plugins that are not available to this Sentry install.
  28         desired_plugins = desired_plugins & set(all_plugins.keys())
  29 
  30         # Each tuple represents an enabled Plugin (of only the ones we care
  31         # about) and its corresponding Project.
  32         enabled_plugins = ProjectOption.objects.filter(
  33             key__in=["%s:enabled" % slug for slug in desired_plugins],
  34             project__organization=organization,
  35         ).select_related("project")
  36 
  37         resources = []
  38 
  39         for project_option in enabled_plugins:
  40             resources.append(
  41                 serialize(
  42                     all_plugins[project_option.key.split(":")[0]],
  43                     request.user,
  44                     OrganizationPluginSerializer(project_option.project),
  45                 )
  46             )
  47 
  48         return Response(resources)
