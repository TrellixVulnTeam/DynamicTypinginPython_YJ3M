  22     def download(self, releasefile):
  23         file = releasefile.file
  24         fp = file.getfile()
  25         response = StreamingHttpResponse(
  26             iter(lambda: fp.read(4096), b""),
  27             content_type=file.headers.get("content-type", "application/octet-stream"),
  28         )
  29         response["Content-Length"] = file.size
  30         response["Content-Disposition"] = 'attachment; filename="%s"' % posixpath.basename(
  31             " ".join(releasefile.name.split())
  32         )
  33         return response

  35     def get(self, request, organization, version, file_id):
  36         """
  37         Retrieve an Organization Release's File
  38         ```````````````````````````````````````
  39 
  40         Return details on an individual file within a release.  This does
  41         not actually return the contents of the file, just the associated
  42         metadata.
  43 
  44         :pparam string organization_slug: the slug of the organization the
  45                                           release belongs to.
  46         :pparam string version: the version identifier of the release.
  47         :pparam string file_id: the ID of the file to retrieve.
  48         :auth: required
  49         """
  50         try:
  51             release = Release.objects.get(organization_id=organization.id, version=version)
  52         except Release.DoesNotExist:
  53             raise ResourceDoesNotExist
  54 
  55         if not self.has_release_permission(request, organization, release):
  56             raise ResourceDoesNotExist
  57 
  58         try:
  59             releasefile = ReleaseFile.objects.get(release=release, id=file_id)
  60         except ReleaseFile.DoesNotExist:
  61             raise ResourceDoesNotExist
  62 
  63         download_requested = request.GET.get("download") is not None
  64         if download_requested and (request.access.has_scope("project:write")):
  65             return self.download(releasefile)
  66         elif download_requested:
  67             return Response(status=403)
  68         return Response(serialize(releasefile, request.user))

  70     def put(self, request, organization, version, file_id):
  71         """
  72         Update an Organization Release's File
  73         `````````````````````````````````````
  74 
  75         Update metadata of an existing file.  Currently only the name of
  76         the file can be changed.
  77 
  78         :pparam string organization_slug: the slug of the organization the
  79                                           release belongs to.
  80         :pparam string version: the version identifier of the release.
  81         :pparam string file_id: the ID of the file to update.
  82         :param string name: the new name of the file.
  83         :param string dist: the name of the dist.
  84         :auth: required
  85         """
  86         try:
  87             release = Release.objects.get(organization_id=organization.id, version=version)
  88         except Release.DoesNotExist:
  89             raise ResourceDoesNotExist
  90 
  91         if not self.has_release_permission(request, organization, release):
  92             raise ResourceDoesNotExist
  93 
  94         try:
  95             releasefile = ReleaseFile.objects.get(release=release, id=file_id)
  96         except ReleaseFile.DoesNotExist:
  97             raise ResourceDoesNotExist
  98 
  99         serializer = ReleaseFileSerializer(data=request.data)
 100 
 101         if not serializer.is_valid():
 102             return Response(serializer.errors, status=400)
 103 
 104         result = serializer.validated_data
 105 
 106         releasefile.update(name=result["name"])
 107 
 108         return Response(serialize(releasefile, request.user))

 111         """

 124         """

 126             release = Release.objects.get(organization_id=organization.id, version=version)
 127         except Release.DoesNotExist:
 128             raise ResourceDoesNotExist
 129 
 130         if not self.has_release_permission(request, organization, release):
 131             raise ResourceDoesNotExist
 132 
 133         try:
 134             releasefile = ReleaseFile.objects.get(release=release, id=file_id)
 135         except ReleaseFile.DoesNotExist:
 136             raise ResourceDoesNotExist

 138         file = releasefile.file
 139 
 140         # TODO(dcramer): this doesnt handle a failure from file.deletefile() to
