  11 class OrganizationEventDetailsEndpoint(OrganizationEventsEndpointBase):
  12     def get(self, request, organization, project_slug, event_id):
  13         if not self.has_feature(organization, request):
  14             return Response(status=404)
  15 
  16         try:
  17             project = Project.objects.get(
  18                 slug=project_slug, organization_id=organization.id, status=ProjectStatus.VISIBLE
  19             )
  20         except Project.DoesNotExist:
  21             return Response(status=404)
  22 
  23         # Check access to the project as this endpoint doesn't use membership checks done
  24         # get_filter_params().

  26             return Response(status=404)
  27 
  28         # We return the requested event if we find a match regardless of whether
  29         # it occurred within the range specified
  30         event = eventstore.get_event_by_id(project.id, event_id)
  31 
  32         if event is None:
  33             return Response({"detail": "Event not found"}, status=404)
  34 
  35         data = serialize(event)
  36         data["projectSlug"] = project_slug
  37 
  38         return Response(data)
