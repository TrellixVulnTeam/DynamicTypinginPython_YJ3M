  39     def _search(self, request, organization, projects, environments, extra_query_kwargs=None):
  40         query_kwargs = build_query_params_from_request(
  41             request, organization, projects, environments
  42         )
  43         if extra_query_kwargs is not None:
  44             assert "environment" not in extra_query_kwargs
  45             query_kwargs.update(extra_query_kwargs)
  46 
  47         query_kwargs["environments"] = environments if environments else None
  48         result = search.query(**query_kwargs)
  49         return result, query_kwargs

  51     def get(self, request, organization):
  52         """
  53         List an Organization's Issues
  54         `````````````````````````````
  55 
  56         Return a list of issues (groups) bound to an organization.  All parameters are
  57         supplied as query string parameters.
  58 
  59         A default query of ``is:unresolved`` is applied. To return results
  60         with other statuses send an new query value (i.e. ``?query=`` for all
  61         results).
  62 
  63         The ``groupStatsPeriod`` parameter can be used to select the timeline
  64         stats which should be present. Possible values are: '' (disable),
  65         '24h', '14d'
  66 
  67         The ``statsPeriod`` parameter can be used to select a date window starting
  68         from now. Ex. ``14d``.
  69 
  70         The ``start`` and ``end`` parameters can be used to select an absolute
  71         date period to fetch issues from.
  72 
  73         :qparam string statsPeriod: an optional stat period (can be one of
  74                                     ``"24h"``, ``"14d"``, and ``""``).
  75         :qparam string groupStatsPeriod: an optional stat period (can be one of
  76                                     ``"24h"``, ``"14d"``, and ``""``).
  77         :qparam string start:       Beginning date. You must also provide ``end``.
  78         :qparam string end:         End date. You must also provide ``start``.
  79         :qparam bool shortIdLookup: if this is set to true then short IDs are
  80                                     looked up by this function as well.  This
  81                                     can cause the return value of the function
  82                                     to return an event issue of a different
  83                                     project which is why this is an opt-in.
  84                                     Set to `1` to enable.
  85         :qparam querystring query: an optional Sentry structured search
  86                                    query.  If not provided an implied
  87                                    ``"is:unresolved"`` is assumed.)
  88         :pparam string organization_slug: the slug of the organization the
  89                                           issues belong to.
  90         :auth: required
  91         """
  92         stats_period = request.GET.get("groupStatsPeriod")
  93         if stats_period not in (None, "", "24h", "14d"):
  94             return Response({"detail": ERR_INVALID_STATS_PERIOD}, status=400)
  95         elif stats_period is None:
  96             # default
  97             stats_period = "24h"
  98         elif stats_period == "":
  99             # disable stats
 100             stats_period = None
 101 
 102         environments = self.get_environments(request, organization)
 103 
 104         serializer = functools.partial(
 105             StreamGroupSerializerSnuba,
 106             environment_ids=[env.id for env in environments],
 107             stats_period=stats_period,
 108         )
 109 
 110         projects = self.get_projects(request, organization)
 111         project_ids = [p.id for p in projects]
 112 
 113         if not projects:
 114             return Response([])
 115 
 116         if len(projects) > 1 and not features.has(
 117             "organizations:global-views", organization, actor=request.user
 118         ):
 119             return Response(
 120                 {"detail": "You do not have the multi project stream feature enabled"}, status=400
 121             )
 122 
 123         # we ignore date range for both short id and event ids
 124         query = request.GET.get("query", "").strip()
 125         if query:
 126             # check to see if we've got an event ID
 127             event_id = normalize_event_id(query)
 128             if event_id:
 129                 # For a direct hit lookup we want to use any passed project ids
 130                 # (we've already checked permissions on these) plus any other
 131                 # projects that the user is a member of. This gives us a better
 132                 # chance of returning the correct result, even if the wrong
 133                 # project is selected.
 134                 direct_hit_projects = set(project_ids) | set(
 135                     [project.id for project in request.access.projects]
 136                 )
 137                 groups = list(Group.objects.filter_by_event_id(direct_hit_projects, event_id))
 138                 if len(groups) == 1:
 139                     response = Response(
 140                         serialize(groups, request.user, serializer(matching_event_id=event_id))
 141                     )
 142                     response["X-Sentry-Direct-Hit"] = "1"
 143                     return response
 144 
 145                 if groups:
 146                     return Response(serialize(groups, request.user, serializer()))
 147 
 148             group = get_by_short_id(organization.id, request.GET.get("shortIdLookup"), query)
 149             if group is not None:
 150                 # check all projects user has access to
 151                 if request.access.has_project_access(group.project):
 152                     response = Response(serialize([group], request.user, serializer()))
 153                     response["X-Sentry-Direct-Hit"] = "1"
 154                     return response
 155 
 156         # If group ids specified, just ignore any query components
 157         try:
 158             group_ids = set(map(int, request.GET.getlist("group")))
 159         except ValueError:
 160             return Response({"detail": "Group ids must be integers"}, status=400)
 161 
 162         if group_ids:
 163             groups = list(Group.objects.filter(id__in=group_ids, project_id__in=project_ids))
 164             if any(g for g in groups if not request.access.has_project_access(g.project)):
 165                 raise PermissionDenied
 166             return Response(serialize(groups, request.user, serializer()))
 167 
 168         try:
 169             start, end = get_date_range_from_params(request.GET)
 170         except InvalidParams as e:
 171             return Response({"detail": six.text_type(e)}, status=400)
 172 
 173         try:
 174             cursor_result, query_kwargs = self._search(
 175                 request,
 176                 organization,
 177                 projects,
 178                 environments,
 179                 {"count_hits": True, "date_to": end, "date_from": start},
 180             )
 181         except (ValidationError, discover.InvalidSearchQuery) as exc:
 182             return Response({"detail": six.text_type(exc)}, status=400)
 183 
 184         results = list(cursor_result)
 185 
 186         context = serialize(results, request.user, serializer())
 187 
 188         # HACK: remove auto resolved entries
 189         # TODO: We should try to integrate this into the search backend, since
 190         # this can cause us to arbitrarily return fewer results than requested.
 191         status = [
 192             search_filter
 193             for search_filter in query_kwargs.get("search_filters", [])
 194             if search_filter.key.name == "status"
 195         ]
 196         if status and status[0].value.raw_value == GroupStatus.UNRESOLVED:
 197             context = [r for r in context if r["status"] == "unresolved"]
 198 
 199         response = Response(context)
 200 
 201         self.add_cursor_headers(request, response, cursor_result)
 202 
 203         # TODO(jess): add metrics that are similar to project endpoint here
 204 
 205         return response

 207     def put(self, request, organization):
 208         """
 209         Bulk Mutate a List of Issues
 210         ````````````````````````````
 211 
 212         Bulk mutate various attributes on issues.  The list of issues
 213         to modify is given through the `id` query parameter.  It is repeated
 214         for each issue that should be modified.
 215 
 216         - For non-status updates, the `id` query parameter is required.
 217         - For status updates, the `id` query parameter may be omitted
 218           for a batch "update all" query.
 219         - An optional `status` query parameter may be used to restrict
 220           mutations to only events with the given status.
 221 
 222         The following attributes can be modified and are supplied as
 223         JSON object in the body:
 224 
 225         If any ids are out of scope this operation will succeed without
 226         any data mutation.
 227 
 228         :qparam int id: a list of IDs of the issues to be mutated.  This
 229                         parameter shall be repeated for each issue.  It
 230                         is optional only if a status is mutated in which
 231                         case an implicit `update all` is assumed.
 232         :qparam string status: optionally limits the query to issues of the
 233                                specified status.  Valid values are
 234                                ``"resolved"``, ``"unresolved"`` and
 235                                ``"ignored"``.
 236         :pparam string organization_slug: the slug of the organization the
 237                                           issues belong to.
 238         :param string status: the new status for the issues.  Valid values
 239                               are ``"resolved"``, ``"resolvedInNextRelease"``,
 240                               ``"unresolved"``, and ``"ignored"``. Status
 241                               updates that include release data are only allowed
 242                               for groups within a single project.
 243         :param map statusDetails: additional details about the resolution.
 244                                   Valid values are ``"inRelease"``, ``"inNextRelease"``,
 245                                   ``"inCommit"``,  ``"ignoreDuration"``, ``"ignoreCount"``,
 246                                   ``"ignoreWindow"``, ``"ignoreUserCount"``, and
 247                                   ``"ignoreUserWindow"``. Status detail
 248                                   updates that include release data are only allowed
 249                                   for groups within a single project.
 250         :param int ignoreDuration: the number of minutes to ignore this issue.
 251         :param boolean isPublic: sets the issue to public or private.
 252         :param boolean merge: allows to merge or unmerge different issues.
 253         :param string assignedTo: the user or team that should be assigned to
 254                                   these issues. Can be of the form ``"<user_id>"``,
 255                                   ``"user:<user_id>"``, ``"<username>"``,
 256                                   ``"<user_primary_email>"``, or ``"team:<team_id>"``.
 257                                   Bulk assigning issues is limited to groups
 258                                   within a single project.
 259         :param boolean hasSeen: in case this API call is invoked with a user
 260                                 context this allows changing of the flag
 261                                 that indicates if the user has seen the
 262                                 event.
 263         :param boolean isBookmarked: in case this API call is invoked with a
 264                                      user context this allows changing of
 265                                      the bookmark flag.
 266         :auth: required
 267         """
 268 
 269         projects = self.get_projects(request, organization)
 270         if len(projects) > 1 and not features.has(
 271             "organizations:global-views", organization, actor=request.user
 272         ):
 273             return Response(
 274                 {"detail": "You do not have the multi project stream feature enabled"}, status=400
 275             )
 276 
 277         search_fn = functools.partial(
 278             self._search,
 279             request,
 280             organization,
 281             projects,
 282             self.get_environments(request, organization),
 283         )
 284         return update_groups(request, projects, organization.id, search_fn)

 294 
 295         Only queries by 'id' are accepted.
 296 
 297         If any ids are out of scope this operation will succeed without
 298         any data mutation.
 299 
 300         :qparam int id: a list of IDs of the issues to be removed.  This
 301                         parameter shall be repeated for each issue.
 302         :pparam string organization_slug: the slug of the organization the
 303                                           issues belong to.
 304         :auth: required
 305         """
 306         projects = self.get_projects(request, organization)
 307         if len(projects) > 1 and not features.has(
 308             "organizations:global-views", organization, actor=request.user
 309         ):
 310             return Response(
 311                 {"detail": "You do not have the multi project stream feature enabled"}, status=400
 312             )
 313 
 314         search_fn = functools.partial(
 315             self._search,
 316             request,
 317             organization,
 318             projects,
 319             self.get_environments(request, organization),
