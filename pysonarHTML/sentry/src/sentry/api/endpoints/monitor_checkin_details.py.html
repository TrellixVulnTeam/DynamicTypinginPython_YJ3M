  43     def convert_args(self, request, monitor_id, checkin_id, *args, **kwargs):
  44         try:
  45             monitor = Monitor.objects.get(guid=monitor_id)
  46         except Monitor.DoesNotExist:
  47             raise ResourceDoesNotExist
  48 
  49         project = Project.objects.get_from_cache(id=monitor.project_id)
  50         if project.status != ProjectStatus.VISIBLE:
  51             raise ResourceDoesNotExist
  52 
  53         if hasattr(request.auth, "project_id") and project.id != request.auth.project_id:
  54             return self.respond(status=400)
  55 
  56         if not features.has("organizations:monitors", project.organization, actor=request.user):
  57             raise ResourceDoesNotExist
  58 
  59         self.check_object_permissions(request, project)
  60 
  61         with configure_scope() as scope:
  62             scope.set_tag("project", project.id)
  63 
  64         bind_organization_context(project.organization)
  65 
  66         try:
  67             checkin = MonitorCheckIn.objects.get(monitor=monitor, guid=checkin_id)
  68         except MonitorCheckIn.DoesNotExist:
  69             raise ResourceDoesNotExist
  70 
  71         request._request.organization = project.organization
  72 
  73         kwargs.update({"checkin": checkin, "monitor": monitor, "project": project})
  74         return (args, kwargs)

  76     def get(self, request, project, monitor, checkin):
  77         """
  78         Retrieve a check-in
  79         ```````````````````
  80 
  81         :pparam string monitor_id: the id of the monitor.
  82         :pparam string checkin_id: the id of the check-in.
  83         :auth: required
  84         """
  85         # we dont allow read permission with DSNs
  86         if isinstance(request.auth, ProjectKey):
  87             return self.respond(status=401)
  88 
  89         return self.respond(serialize(checkin, request.user))

  91     def put(self, request, project, monitor, checkin):
  92         """
  93         Update a check-in
  94         `````````````````
  95 
  96         :pparam string monitor_id: the id of the monitor.
  97         :pparam string checkin_id: the id of the check-in.
  98         :auth: required
  99         """
 100         if checkin.status in CheckInStatus.FINISHED_VALUES:
 101             return self.respond(status=400)
 102 
 103         serializer = CheckInSerializer(
 104             data=request.data, partial=True, context={"project": project, "request": request}
 105         )
 106         if not serializer.is_valid():
 107             return self.respond(serializer.errors, status=400)
 108 
 109         result = serializer.validated_data
 110 
 111         current_datetime = timezone.now()
 112         params = {"date_updated": current_datetime}
 113         if "duration" in result:
 114             params["duration"] = result["duration"]
 115         if "status" in result:
 116             params["status"] = getattr(CheckInStatus, result["status"].upper())
 117 
 118         with transaction.atomic():
 119             checkin.update(**params)
 120             if checkin.status == CheckInStatus.ERROR:
 121                 if not monitor.mark_failed(current_datetime):
 122                     return self.respond(serialize(checkin, request.user), status=208)
 123             else:
 124                 monitor_params = {
 125                     "last_checkin": current_datetime,
 126                     "next_checkin": monitor.get_next_scheduled_checkin(current_datetime),
 127                 }
 128                 if checkin.status == CheckInStatus.OK:
 129                     monitor_params["status"] = MonitorStatus.OK
 130                 Monitor.objects.filter(id=monitor.id).exclude(
 131                     last_checkin__gt=current_datetime
 132                 ).update(**monitor_params)
 133 
 134         return self.respond(serialize(checkin, request.user))
