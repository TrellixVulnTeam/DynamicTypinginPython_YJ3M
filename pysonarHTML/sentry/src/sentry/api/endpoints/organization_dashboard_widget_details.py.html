  34     def put(self, request, organization, dashboard, widget):
  35         """
  36         Edit a Widget on an Organization's Dashboard
  37         ````````````````````````````````````````````
  38 
  39         Edit a widget on an organization's dashboard.
  40 
  41         :pparam string organization_slug: the slug of the organization the
  42                                           dashboard belongs to.
  43         :pparam int dashboard_id: the id of the dashboard.
  44         :pparam int widget_id: the id of the widget.
  45         :param string title: the title of the widget.
  46         :param string displayType: the widget display type (i.e. line or table).
  47         :param array displayOptions: the widget display options are special
  48                                     variables necessary to displaying the widget correctly.
  49         :param array dataSources: the sources of data for the widget to display.
  50                                 If supplied the entire set of data sources will be deleted
  51                                 and replaced with the input provided.
  52         :auth: required
  53         """
  54         # TODO(lb): better document displayType, displayOptions, and dataSources.
  55         serializer = WidgetSerializer(data=request.data, context={"organization": organization})
  56         if not serializer.is_valid():
  57             return Response(serializer.errors, status=400)
  58 
  59         data = serializer.validated_data
  60 
  61         with transaction.atomic():
  62             widget.update(
  63                 title=data.get("title", widget.title),
  64                 display_type=data.get("displayType", widget.display_type),
  65                 display_options=data.get("displayOptions", widget.display_options),
  66             )
  67 
  68             if "dataSources" in data:
  69                 WidgetDataSource.objects.filter(widget_id=widget.id).delete()
  70             for widget_data in data.get("dataSources", []):
  71                 WidgetDataSource.objects.create(
  72                     name=widget_data["name"],
  73                     data=widget_data["data"],
  74                     type=widget_data["type"],
  75                     order=widget_data["order"],
  76                     widget_id=widget.id,
  77                 )
  78 
  79         return Response(serialize(widget, request.user))
