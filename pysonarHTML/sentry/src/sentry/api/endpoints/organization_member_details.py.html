  38 def get_allowed_roles(request, organization, member=None):
  39     can_admin = request.access.has_scope("member:admin")
  40 
  41     allowed_roles = []
  42     if can_admin and not is_active_superuser(request):
  43         acting_member = member or OrganizationMember.objects.get(
  44             user=request.user, organization=organization
  45         )
  46         if member and roles.get(acting_member.role).priority < roles.get(member.role).priority:
  47             can_admin = False
  48         else:
  49             allowed_roles = [
  50                 r for r in roles.get_all() if r.priority <= roles.get(acting_member.role).priority
  51             ]
  52             can_admin = bool(allowed_roles)
  53     elif is_active_superuser(request):
  54         allowed_roles = roles.get_all()
  55     return (can_admin, allowed_roles)

  65 class RelaxedMemberPermission(OrganizationPermission):

  79     def _get_member(self, request, organization, member_id):
  80         if member_id == "me":
  81             queryset = OrganizationMember.objects.filter(
  82                 organization=organization, user__id=request.user.id, user__is_active=True
  83             )
  84         else:
  85             try:
  86                 queryset = OrganizationMember.objects.filter(
  87                     Q(user__is_active=True) | Q(user__isnull=True),
  88                     organization=organization,
  89                     id=member_id,
  90                     invite_status=InviteStatus.APPROVED.value,
  91                 )
  92             except ValueError:
  93                 raise OrganizationMember.DoesNotExist()
  94         return queryset.select_related("user").get()

  96     def _is_only_owner(self, member):
  97         if member.role != roles.get_top_dog().id:
  98             return False
  99 
 100         queryset = OrganizationMember.objects.filter(
 101             organization=member.organization_id,
 102             role=roles.get_top_dog().id,
 103             user__isnull=False,
 104             user__is_active=True,
 105         ).exclude(id=member.id)
 106         if queryset.exists():
 107             return False
 108 
 109         return True

 111     def _serialize_member(self, member, request, allowed_roles=None):
 112         context = serialize(member, serializer=OrganizationMemberWithTeamsSerializer())
 113 
 114         if request.access.has_scope("member:admin"):
 115             context["invite_link"] = member.get_invite_link()
 116             context["user"] = serialize(member.user, request.user, DetailedUserSerializer())
 117 
 118         context["isOnlyOwner"] = self._is_only_owner(member)
 119         context["roles"] = serialize(
 120             roles.get_all(), serializer=RoleSerializer(), allowed_roles=allowed_roles
 121         )
 122 
 123         return context

 125     def get(self, request, organization, member_id):
 126         """Currently only returns allowed invite roles for member invite"""
 127 
 128         try:
 129             member = self._get_member(request, organization, member_id)
 130         except OrganizationMember.DoesNotExist:
 131             raise ResourceDoesNotExist
 132 
 133         _, allowed_roles = get_allowed_roles(request, organization, member)
 134 
 135         context = self._serialize_member(member, request, allowed_roles)
 136 
 137         return Response(context)

 139     def put(self, request, organization, member_id):
 140         try:
 141             om = self._get_member(request, organization, member_id)
 142         except OrganizationMember.DoesNotExist:
 143             raise ResourceDoesNotExist
 144 
 145         serializer = OrganizationMemberSerializer(data=request.data, partial=True)
 146 
 147         if not serializer.is_valid():
 148             return Response(status=400)
 149 
 150         try:
 151             auth_provider = AuthProvider.objects.get(organization=organization)
 152             auth_provider = auth_provider.get_provider()
 153         except AuthProvider.DoesNotExist:
 154             auth_provider = None
 155 
 156         allowed_roles = None
 157         result = serializer.validated_data
 158 
 159         # XXX(dcramer): if/when this expands beyond reinvite we need to check
 160         # access level
 161         if result.get("reinvite"):
 162             if om.is_pending:
 163                 if result.get("regenerate"):
 164                     if request.access.has_scope("member:admin"):
 165                         om.regenerate_token()
 166                         om.save()
 167                     else:
 168                         return Response({"detail": ERR_INSUFFICIENT_SCOPE}, status=400)
 169                 if om.token_expired:
 170                     return Response({"detail": ERR_EXPIRED}, status=400)
 171                 om.send_invite_email()
 172             elif auth_provider and not getattr(om.flags, "sso:linked"):
 173                 om.send_sso_link_email(request.user, auth_provider)
 174             else:
 175                 # TODO(dcramer): proper error message
 176                 return Response({"detail": ERR_UNINVITABLE}, status=400)
 177 
 178         if "teams" in result:
 179             # dupe code from member_index
 180             # ensure listed teams are real teams
 181             teams = list(
 182                 Team.objects.filter(
 183                     organization=organization, status=TeamStatus.VISIBLE, slug__in=result["teams"]
 184                 )
 185             )
 186 
 187             if len(set(result["teams"])) != len(teams):
 188                 return Response({"teams": "Invalid team"}, status=400)
 189 
 190             with transaction.atomic():
 191                 # teams may be empty
 192                 OrganizationMemberTeam.objects.filter(organizationmember=om).delete()
 193                 OrganizationMemberTeam.objects.bulk_create(
 194                     [OrganizationMemberTeam(team=team, organizationmember=om) for team in teams]
 195                 )
 196 
 197         if result.get("role"):
 198             _, allowed_roles = get_allowed_roles(request, organization)
 199             allowed_role_ids = {r.id for r in allowed_roles}
 200 
 201             # A user cannot promote others above themselves
 202             if result["role"] not in allowed_role_ids:
 203                 return Response(
 204                     {"role": "You do not have permission to assign the given role."}, status=403
 205                 )
 206 
 207             # A user cannot demote a superior
 208             if om.role not in allowed_role_ids:
 209                 return Response(
 210                     {"role": "You do not have permission to assign a role to the given user."},
 211                     status=403,
 212                 )
 213 
 214             if om.user == request.user and (result["role"] != om.role):
 215                 return Response({"detail": "You cannot make changes to your own role."}, status=400)
 216 
 217             om.update(role=result["role"])
 218 
 219         self.create_audit_entry(
 220             request=request,
 221             organization=organization,
 222             target_object=om.id,
 223             target_user=om.user,
 224             event=AuditLogEntryEvent.MEMBER_EDIT,
 225             data=om.get_audit_log_data(),
 226         )
 227 
 228         context = self._serialize_member(om, request, allowed_roles)
 229 
 230         return Response(context)

 232     def delete(self, request, organization, member_id):
 233         try:
 234             om = self._get_member(request, organization, member_id)
 235         except OrganizationMember.DoesNotExist:
 236             raise ResourceDoesNotExist
 237 
 238         if request.user.is_authenticated() and not is_active_superuser(request):
 239             try:
 240                 acting_member = OrganizationMember.objects.get(
 241                     organization=organization, user=request.user
 242                 )
 243             except OrganizationMember.DoesNotExist:
 244                 return Response({"detail": ERR_INSUFFICIENT_ROLE}, status=400)
 245             else:
 246                 if acting_member != om:
 247                     if not request.access.has_scope("member:admin"):
 248                         return Response({"detail": ERR_INSUFFICIENT_SCOPE}, status=400)
 249                     elif not roles.can_manage(acting_member.role, om.role):
 250                         return Response({"detail": ERR_INSUFFICIENT_ROLE}, status=400)
 251 
 252         # TODO(dcramer): do we even need this check?
 253         elif not request.access.has_scope("member:admin"):
 254             return Response({"detail": ERR_INSUFFICIENT_SCOPE}, status=400)
 255 
 256         if self._is_only_owner(om):
 257             return Response({"detail": ERR_ONLY_OWNER}, status=403)
 258 
 259         audit_data = om.get_audit_log_data()
 260 
 261         with transaction.atomic():
 262             AuthIdentity.objects.filter(
 263                 user=om.user, auth_provider__organization=organization
 264             ).delete()
 265 
 266             om.delete()
 267 
 268         self.create_audit_entry(
 269             request=request,
 270             organization=organization,
 271             target_object=om.id,
 272             target_user=om.user,
 273             event=AuditLogEntryEvent.MEMBER_REMOVE,
 274             data=audit_data,
 275         )
 276 
 277         return Response(status=204)
