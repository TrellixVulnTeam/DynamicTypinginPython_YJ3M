  98 def debounce_update_release_health_data(organization, project_ids):
  99     """This causes a flush of snuba health data to the postgres tables once
 100     per minute for the given projects.
 101     """
 102     # Figure out which projects need to get updates from the snuba.
 103     should_update = {}
 104     cache_keys = ["debounce-health:%d" % id for id in project_ids]
 105     cache_data = cache.get_many(cache_keys)
 106     for project_id, cache_key in izip(project_ids, cache_keys):
 107         if cache_data.get(cache_key) is None:
 108             should_update[project_id] = cache_key
 109 
 110     if not should_update:
 111         return
 112 
 113     projects = {p.id: p for p in Project.objects.get_many_from_cache(should_update.keys())}
 114 
 115     # This gives us updates for all release-projects which have seen new
 116     # health data over the last days. It will miss releases where the last
 117     # date is longer than what `get_changed_project_release_model_adoptions`
 118     # considers recent.
 119     project_releases = get_changed_project_release_model_adoptions(should_update.keys())
 120 
 121     # Check which we already have rows for.
 122     existing = set(
 123         ReleaseProject.objects.filter(
 124             project_id__in=[x[0] for x in project_releases],
 125             release__version__in=[x[1] for x in project_releases],
 126         ).values_list("project_id", "release__version")
 127     )
 128     to_upsert = []
 129     for key in project_releases:
 130         if key not in existing:
 131             to_upsert.append(key)
 132 
 133     if to_upsert:
 134         dates = get_oldest_health_data_for_releases(to_upsert)
 135 
 136         for project_id, version in to_upsert:
 137             project = projects.get(project_id)
 138             if project is None:
 139                 # should not happen
 140                 continue
 141 
 142         # We might have never observed the release.  This for instance can
 143         # happen if the release only had health data so far.  For these cases
 144         # we want to create the release the first time we observed it on the
 145         # health side.
 146         release = Release.get_or_create(
 147             project=project, version=version, date_added=dates.get((project_id, version))
 148         )
 149 
 150         # Make sure that the release knows about this project.  Like we had before
 151         # the project might not have been associated with this release yet.
 152         release.add_project(project)
 153 
 154     # Debounce updates for a minute
 155     cache.set_many(dict(izip(should_update.values(), [True] * len(should_update))), 60)

 158 class OrganizationReleasesEndpoint(OrganizationReleasesBaseEndpoint, EnvironmentMixin):
 159     doc_section = DocSection.RELEASES

 161     @attach_scenarios([list_org_releases_scenario])
 162     def get(self, request, organization):
 163         """
 164         List an Organization's Releases
 165         ```````````````````````````````
 166         Return a list of releases for a given organization.
 167 
 168         :pparam string organization_slug: the organization short name
 169         :qparam string query: this parameter can be used to create a
 170                               "starts with" filter for the version.
 171         """
 172         query = request.GET.get("query")
 173         with_health = request.GET.get("health") == "1"
 174         flatten = request.GET.get("flatten") == "1"
 175         sort = request.GET.get("sort") or "date"
 176         health_stat = request.GET.get("healthStat") or "sessions"
 177         summary_stats_period = request.GET.get("summaryStatsPeriod") or "14d"
 178         health_stats_period = request.GET.get("healthStatsPeriod") or ("24h" if with_health else "")
 179         if summary_stats_period not in STATS_PERIODS:
 180             raise ParseError(detail=get_stats_period_detail("summaryStatsPeriod", STATS_PERIODS))
 181         if health_stats_period and health_stats_period not in STATS_PERIODS:
 182             raise ParseError(detail=get_stats_period_detail("healthStatsPeriod", STATS_PERIODS))
 183         if health_stat not in ("sessions", "users"):
 184             raise ParseError(detail="invalid healthStat")
 185 
 186         paginator_cls = OffsetPaginator
 187         paginator_kwargs = {}
 188 
 189         try:
 190             filter_params = self.get_filter_params(request, organization, date_filter_optional=True)
 191         except NoProjects:
 192             return Response([])
 193 
 194         # This should get us all the projects into postgres that have received
 195         # health data in the last 24 hours.  If health data is not requested
 196         # we don't upsert releases.
 197         if with_health:
 198             debounce_update_release_health_data(organization, filter_params["project_id"])
 199 
 200         queryset = Release.objects.filter(organization=organization).select_related("owner")
 201 
 202         if "environment" in filter_params:
 203             queryset = queryset.filter(
 204                 releaseprojectenvironment__environment__name__in=filter_params["environment"],
 205                 releaseprojectenvironment__project_id__in=filter_params["project_id"],
 206             )
 207 
 208         if query:
 209             query_q = Q(version__icontains=query)
 210 
 211             suffix_match = _release_suffix.match(query)
 212             if suffix_match is not None:
 213                 query_q |= Q(version__icontains="%s+%s" % suffix_match.groups())
 214 
 215             queryset = queryset.filter(query_q)
 216 
 217         select_extra = {}
 218         sort_query = None
 219 
 220         queryset = queryset.distinct()
 221         if flatten:
 222             select_extra["_for_project_id"] = "sentry_release_project.project_id"
 223 
 224         if sort == "date":
 225             sort_query = "COALESCE(sentry_release.date_released, sentry_release.date_added)"
 226         elif sort in (
 227             "crash_free_sessions",
 228             "crash_free_users",
 229             "sessions",
 230             "users",
 231             "sessions_24h",
 232             "users_24h",
 233         ):
 234             if not flatten:
 235                 return Response(
 236                     {"detail": "sorting by crash statistics requires flattening (flatten=1)"},
 237                     status=400,
 238                 )
 239             paginator_cls = MergingOffsetPaginator
 240             paginator_kwargs.update(
 241                 data_load_func=lambda offset, limit: get_project_releases_by_stability(
 242                     project_ids=filter_params["project_id"],
 243                     environments=filter_params.get("environment"),
 244                     scope=sort,
 245                     offset=offset,
 246                     stats_period=summary_stats_period,
 247                     limit=limit,
 248                 ),
 249                 apply_to_queryset=lambda queryset, rows: queryset.filter(
 250                     projects__id__in=list(x[0] for x in rows), version__in=list(x[1] for x in rows)
 251                 ),
 252                 key_from_model=lambda x: (x._for_project_id, x.version),
 253             )
 254         else:
 255             return Response({"detail": "invalid sort"}, status=400)
 256 
 257         if sort_query is not None:
 258             queryset = queryset.filter(projects__id__in=filter_params["project_id"])
 259             select_extra["sort"] = sort_query
 260             paginator_kwargs["order_by"] = "-sort"
 261 
 262         queryset = queryset.extra(select=select_extra)
 263         if filter_params["start"] and filter_params["end"]:
 264             queryset = queryset.extra(
 265                 where=[
 266                     "COALESCE(sentry_release.date_released, sentry_release.date_added) BETWEEN %s and %s"
 267                 ],
 268                 params=[filter_params["start"], filter_params["end"]],
 269             )
 270 
 271         return self.paginate(
 272             request=request,
 273             queryset=queryset,
 274             paginator_cls=paginator_cls,
 275             on_results=lambda x: serialize(
 276                 x,
 277                 request.user,
 278                 with_health_data=with_health,
 279                 health_stat=health_stat,
 280                 health_stats_period=health_stats_period,
 281                 summary_stats_period=summary_stats_period,
 282             ),
 283             **paginator_kwargs
 284         )

 286     @attach_scenarios([create_new_org_release_ref_scenario, create_new_org_release_commit_scenario])
 287     def post(self, request, organization):
 288         """
 289         Create a New Release for an Organization
 290         ````````````````````````````````````````
 291         Create a new release for the given Organization.  Releases are used by
 292         Sentry to improve its error reporting abilities by correlating
 293         first seen events with the release that might have introduced the
 294         problem.
 295         Releases are also necessary for sourcemaps and other debug features
 296         that require manual upload for functioning well.
 297 
 298         :pparam string organization_slug: the slug of the organization the
 299                                           release belongs to.
 300         :param string version: a version identifier for this release.  Can
 301                                be a version number, a commit hash etc.
 302         :param string ref: an optional commit reference.  This is useful if
 303                            a tagged version has been provided.
 304         :param url url: a URL that points to the release.  This can be the
 305                         path to an online interface to the sourcecode
 306                         for instance.
 307         :param array projects: a list of project slugs that are involved in
 308                                this release
 309         :param datetime dateReleased: an optional date that indicates when
 310                                       the release went live.  If not provided
 311                                       the current time is assumed.
 312         :param array commits: an optional list of commit data to be associated
 313                               with the release. Commits must include parameters
 314                               ``id`` (the sha of the commit), and can optionally
 315                               include ``repository``, ``message``, ``patch_set``,
 316                               ``author_name``, ``author_email``, and ``timestamp``.
 317                               See [release without integration example](/workflow/releases/).
 318         :param array refs: an optional way to indicate the start and end commits
 319                            for each repository included in a release. Head commits
 320                            must include parameters ``repository`` and ``commit``
 321                            (the HEAD sha). They can optionally include ``previousCommit``
 322                            (the sha of the HEAD of the previous release), which should
 323                            be specified if this is the first time you've sent commit data.
 324                            ``commit`` may contain a range in the form of ``previousCommit..commit``
 325         :auth: required
 326         """
 327         serializer = ReleaseSerializerWithProjects(data=request.data)
 328 
 329         if serializer.is_valid():
 330             result = serializer.validated_data
 331 
 332             allowed_projects = {p.slug: p for p in self.get_projects(request, organization)}
 333 
 334             projects = []
 335             for slug in result["projects"]:
 336                 if slug not in allowed_projects:
 337                     return Response({"projects": ["Invalid project slugs"]}, status=400)
 338                 projects.append(allowed_projects[slug])
 339 
 340             # release creation is idempotent to simplify user
 341             # experiences
 342             try:
 343                 with transaction.atomic():
 344                     release, created = (
 345                         Release.objects.create(
 346                             organization_id=organization.id,
 347                             version=result["version"],
 348                             ref=result.get("ref"),
 349                             url=result.get("url"),
 350                             owner=result.get("owner"),
 351                             date_released=result.get("dateReleased"),
 352                         ),
 353                         True,
 354                     )
 355             except IntegrityError:
 356                 release, created = (
 357                     Release.objects.get(organization_id=organization.id, version=result["version"]),
 358                     False,
 359                 )
 360             else:
 361                 release_created.send_robust(release=release, sender=self.__class__)
 362 
 363             new_projects = []
 364             for project in projects:
 365                 created = release.add_project(project)
 366                 if created:
 367                     new_projects.append(project)
 368 
 369             if release.date_released:
 370                 for project in new_projects:
 371                     Activity.objects.create(
 372                         type=Activity.RELEASE,
 373                         project=project,
 374                         ident=Activity.get_version_ident(result["version"]),
 375                         data={"version": result["version"]},
 376                         datetime=release.date_released,
 377                     )
 378 
 379             commit_list = result.get("commits")
 380             if commit_list:
 381                 release.set_commits(commit_list)
 382 
 383             refs = result.get("refs")
 384             if not refs:
 385                 refs = [
 386                     {
 387                         "repository": r["repository"],
 388                         "previousCommit": r.get("previousId"),
 389                         "commit": r["currentId"],
 390                     }
 391                     for r in result.get("headCommits", [])
 392                 ]
 393             if refs:
 394                 if not request.user.is_authenticated():
 395                     return Response(
 396                         {"refs": ["You must use an authenticated API token to fetch refs"]},
 397                         status=400,
 398                     )
 399                 fetch_commits = not commit_list
 400                 try:
 401                     release.set_refs(refs, request.user, fetch=fetch_commits)
 402                 except InvalidRepository as e:
 403                     return Response({"refs": [six.text_type(e)]}, status=400)
 404 
 405             if not created and not new_projects:
 406                 # This is the closest status code that makes sense, and we want
 407                 # a unique 2xx response code so people can understand when
 408                 # behavior differs.
 409                 #   208 Already Reported (WebDAV; RFC 5842)
 410                 status = 208
 411             else:
 412                 status = 201
 413 
 414             return Response(serialize(release, request.user), status=status)
 415         return Response(serializer.errors, status=400)
