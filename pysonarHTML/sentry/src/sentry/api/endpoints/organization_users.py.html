  20 class OrganizationUsersEndpoint(OrganizationEndpoint, EnvironmentMixin):
  21     doc_section = DocSection.ORGANIZATIONS
  22 
  23     @attach_scenarios([list_organization_users_scenario])
  24     def get(self, request, organization):
  25         """
  26         List an Organization's Users
  27         ````````````````````````````
  28 
  29         Return a list of users that belong to a given organization.
  30 
  31         :qparam string project: restrict results to users who have access to a given project ID
  32         :pparam string organization_slug: the slug of the organization for which the users
  33                                           should be listed.
  34         :auth: required
  35         """
  36         projects = self.get_projects(request, organization)
  37 
  38         with sentry_sdk.start_span(op="OrganizationUsersEndpoint.get_members") as span:
  39             qs = (
  40                 OrganizationMember.objects.filter(
  41                     user__is_active=True,
  42                     organization=organization,
  43                     teams__projectteam__project__in=projects,
  44                 )
  45                 .select_related("user")
  46                 .prefetch_related(
  47                     "teams", "teams__projectteam_set", "teams__projectteam_set__project"
  48                 )
  49                 .order_by("user__email")
  50                 .distinct()
  51             )
  52             organization_members = list(qs)
  53 
  54             span.set_data("Project Count", len(projects))
  55             span.set_data("Member Count", len(organization_members))
  56 
  57         return Response(
  58             serialize(
  59                 organization_members,
  60                 request.user,
  61                 serializer=OrganizationMemberWithProjectsSerializer(
  62                     project_ids=[p.id for p in projects]
  63                 ),
  64             )
  65         )
