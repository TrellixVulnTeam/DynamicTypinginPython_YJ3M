  23 class SentryAppsEndpoint(SentryAppsBaseEndpoint):
  24     def get(self, request):
  25         status = request.GET.get("status")
  26 
  27         if status == "published":
  28             queryset = SentryApp.objects.filter(status=SentryAppStatus.PUBLISHED)
  29 
  30         elif status == "unpublished":
  31             queryset = SentryApp.objects.filter(status=SentryAppStatus.UNPUBLISHED)
  32             if not is_active_superuser(request):
  33                 queryset = queryset.filter(owner__in=request.user.get_orgs())
  34         elif status == "internal":
  35             queryset = SentryApp.objects.filter(status=SentryAppStatus.INTERNAL)
  36             if not is_active_superuser(request):
  37                 queryset = queryset.filter(owner__in=request.user.get_orgs())
  38         else:
  39             if is_active_superuser(request):
  40                 queryset = SentryApp.objects.all()
  41             else:
  42                 queryset = SentryApp.objects.filter(status=SentryAppStatus.PUBLISHED)
  43 
  44         return self.paginate(
  45             request=request,
  46             queryset=queryset,
  47             order_by="-date_added",
  48             paginator_cls=OffsetPaginator,
  49             on_results=lambda x: serialize(x, request.user, access=request.access),
  50         )

  52     def post(self, request, organization):
  53         data = {
  54             "name": request.json_body.get("name"),
  55             "user": request.user,
  56             "author": request.json_body.get("author"),
  57             "organization": self._get_user_org(request),
  58             "webhookUrl": request.json_body.get("webhookUrl"),
  59             "redirectUrl": request.json_body.get("redirectUrl"),
  60             "isAlertable": request.json_body.get("isAlertable"),
  61             "isInternal": request.json_body.get("isInternal"),
  62             "verifyInstall": request.json_body.get("verifyInstall"),
  63             "scopes": request.json_body.get("scopes", []),
  64             "events": request.json_body.get("events", []),
  65             "schema": request.json_body.get("schema", {}),
  66             "overview": request.json_body.get("overview"),
  67             "allowedOrigins": request.json_body.get("allowedOrigins", []),
  68         }
  69 
  70         if self._has_hook_events(request) and not features.has(
  71             "organizations:integrations-event-hooks", organization, actor=request.user
  72         ):
  73 
  74             return Response(
  75                 {
  76                     "non_field_errors": [
  77                         "Your organization does not have access to the 'error' resource subscription."
  78                     ]
  79                 },
  80                 status=403,
  81             )
  82 
  83         serializer = SentryAppSerializer(data=data, access=request.access)
  84 
  85         if serializer.is_valid():
  86             data["redirect_url"] = data["redirectUrl"]
  87             data["webhook_url"] = data["webhookUrl"]
  88             data["is_alertable"] = data["isAlertable"]
  89             data["verify_install"] = data["verifyInstall"]
  90             data["allowed_origins"] = data["allowedOrigins"]
  91             data["is_internal"] = data.get("isInternal")
  92 
  93             creator = InternalCreator if data.get("isInternal") else Creator
  94             try:
  95                 sentry_app = creator.run(request=request, **data)
  96             except ValidationError as e:
  97                 # we generate and validate the slug here instead of the serializer since the slug never changes
  98                 return Response(e.detail, status=400)
  99 
 100             return Response(serialize(sentry_app, access=request.access), status=201)
 101 
 102         # log any errors with schema
 103         if "schema" in serializer.errors:
 104             for error_message in serializer.errors["schema"]:
 105                 name = "sentry_app.schema_validation_error"
 106                 log_info = {
 107                     "schema": json.dumps(data["schema"]),
 108                     "user_id": request.user.id,
 109                     "sentry_app_name": data["name"],
 110                     "organization_id": organization.id,
 111                     "error_message": error_message,
 112                 }
 113                 logger.info(name, extra=log_info)
 114                 analytics.record(name, **log_info)
 115         return Response(serializer.errors, status=400)
 116 
 117     def _get_user_org(self, request):
 118         return next(
 119             (
 120                 org
 121                 for org in request.user.get_orgs()
 122                 if org.slug == request.json_body["organization"]
 123             ),
 124             None,
 125         )
 126 
 127     def _has_hook_events(self, request):
 128         if not request.json_body.get("events"):
 129             return False
 130 
 131         return "error" in request.json_body["events"]
