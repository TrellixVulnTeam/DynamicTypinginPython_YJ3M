  17 ERR_INVALID_STATS_PERIOD = "Invalid stats_period. Valid choices are '', '24h', '14d', and '30d'"

  20 @scenario("ListOrganizationProjects")
  21 def list_organization_projects_scenario(runner):
  22     runner.request(method="GET", path="/organizations/%s/projects/" % runner.org.slug)

  25 class OrganizationProjectsEndpoint(OrganizationEndpoint, EnvironmentMixin):
  26     doc_section = DocSection.ORGANIZATIONS
  27 
  28     @attach_scenarios([list_organization_projects_scenario])
  29     def get(self, request, organization):
  30         """
  31         List an Organization's Projects
  32         ```````````````````````````````
  33 
  34         Return a list of projects bound to a organization.
  35 
  36         :pparam string organization_slug: the slug of the organization for
  37                                           which the projects should be listed.
  38         :auth: required
  39         """
  40         stats_period = request.GET.get("statsPeriod")
  41         if stats_period not in (None, "", "24h", "14d", "30d"):
  42             return Response(
  43                 {"error": {"params": {"stats_period": {"message": ERR_INVALID_STATS_PERIOD}}}},
  44                 status=400,
  45             )
  46         elif not stats_period:
  47             # disable stats
  48             stats_period = None
  49 
  50         if request.auth and not request.user.is_authenticated():
  51             # TODO: remove this, no longer supported probably
  52             if hasattr(request.auth, "project"):
  53                 team_list = list(request.auth.project.teams.all())
  54                 queryset = Project.objects.filter(id=request.auth.project.id)
  55             elif request.auth.organization is not None:
  56                 org = request.auth.organization
  57                 team_list = list(Team.objects.filter(organization=org))
  58                 queryset = Project.objects.filter(teams__in=team_list)
  59             else:
  60                 return Response(
  61                     {"detail": "Current access does not point to " "organization."}, status=400
  62                 )
  63         else:
  64             queryset = Project.objects.filter(organization=organization)
  65 
  66         order_by = ["slug"]
  67 
  68         if request.user.is_authenticated():
  69             queryset = queryset.extra(
  70                 select={
  71                     "is_bookmarked": """exists (
  72                         select *
  73                         from sentry_projectbookmark spb
  74                         where spb.project_id = sentry_project.id and spb.user_id = %s
  75                     )"""
  76                 },
  77                 select_params=(request.user.id,),
  78             )
  79             order_by.insert(0, "-is_bookmarked")
  80 
  81         query = request.GET.get("query")
  82         if query:
  83             tokens = tokenize_query(query)
  84             for key, value in six.iteritems(tokens):
  85                 if key == "query":
  86                     value = " ".join(value)
  87                     queryset = queryset.filter(Q(name__icontains=value) | Q(slug__icontains=value))
  88                 elif key == "id":
  89                     queryset = queryset.filter(id__in=value)
  90                 elif key == "slug":
  91                     queryset = queryset.filter(slug__in=value)
  92                 elif key == "team":
  93                     team_list = list(Team.objects.filter(slug__in=value))
  94                     queryset = queryset.filter(teams__in=team_list)
  95                 elif key == "!team":
  96                     team_list = list(Team.objects.filter(slug__in=value))
  97                     queryset = queryset.exclude(teams__in=team_list)
  98                 elif key == "is_member":
  99                     queryset = queryset.filter(teams__organizationmember__user=request.user)
 100                 else:
 101                     queryset = queryset.none()
 102 
 103         queryset = queryset.filter(status=ProjectStatus.VISIBLE).distinct()
 104 
 105         # TODO(davidenwang): remove this after frontend requires only paginated projects
 106         get_all_projects = request.GET.get("all_projects") == "1"
 107 
 108         if get_all_projects:
 109             queryset = queryset.order_by("slug").select_related("organization")
 110             return Response(serialize(list(queryset), request.user, ProjectSummarySerializer()))
 111         else:
 112 
 113             def serialize_on_result(result):
 114                 environment_id = self._get_environment_id_from_request(request, organization.id)
 115                 serializer = ProjectSummarySerializer(
 116                     environment_id=environment_id, stats_period=stats_period,
 117                 )
 118                 return serialize(result, request.user, serializer)
 119 
 120             return self.paginate(
 121                 request=request,
 122                 queryset=queryset,
 123                 order_by=order_by,
 124                 on_results=serialize_on_result,
 125                 paginator_cls=OffsetPaginator,
 126             )
