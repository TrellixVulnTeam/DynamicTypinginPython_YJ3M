  10 class OrganizationReleaseCommitsEndpoint(OrganizationReleasesBaseEndpoint):
  11     doc_section = DocSection.RELEASES
  12 
  13     def get(self, request, organization, version):
  14         """
  15         List an Organization Release's Commits
  16         ``````````````````````````````````````
  17 
  18         Retrieve a list of commits for a given release.
  19 
  20         :pparam string organization_slug: the slug of the organization the
  21                                           release belongs to.
  22         :pparam string version: the version identifier of the release.
  23         :auth: required
  24         """
  25         try:
  26             release = Release.objects.distinct().get(
  27                 organization_id=organization.id,
  28                 projects__in=self.get_projects(request, organization),
  29                 version=version,
  30             )
  31         except Release.DoesNotExist:
  32             raise ResourceDoesNotExist
  33 
  34         queryset = ReleaseCommit.objects.filter(release=release).select_related(
  35             "commit", "commit__author"
  36         )
  37 
  38         return self.paginate(
  39             request=request,
  40             queryset=queryset,
  41             order_by="order",
  42             on_results=lambda x: serialize([rc.commit for rc in x], request.user),
  43         )
