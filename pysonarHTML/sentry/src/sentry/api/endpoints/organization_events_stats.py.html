 110     def get_field(self, request, snuba_args):
 111         y_axis = request.GET.get("yAxis", None)
 112         # These aliases are used by v1 of events.
 113         if not y_axis or y_axis == "event_count":
 114             y_axis = "count()"
 115         elif y_axis == "user_count":
 116             y_axis = "count_unique(user)"
 117 
 118         snuba_filter = eventstore.Filter(
 119             {
 120                 "start": snuba_args.get("start"),
 121                 "end": snuba_args.get("end"),
 122                 "rollup": snuba_args.get("rollup"),
 123             }
 124         )
 125         try:
 126             resolved = resolve_field_list([y_axis], snuba_filter)
 127         except InvalidSearchQuery as err:
 128             raise ParseError(detail=six.text_type(err))
 129         try:
 130             aggregate = resolved["aggregations"][0]
 131         except IndexError:
 132             raise ParseError(detail="Invalid yAxis value requested.")
 133         aggregate[2] = "count"
 134         snuba_args["aggregations"] = [aggregate]
 135 
 136         return snuba_args
