  31 class OrganizationEventsEndpoint(OrganizationEventsEndpointBase):
  32     def get(self, request, organization):
  33         # Check for a direct hit on event ID
  34         query = request.GET.get("query", "").strip()
  35 
  36         try:
  37             direct_hit_resp = get_direct_hit_response(
  38                 request,
  39                 query,
  40                 self.get_filter_params(request, organization),
  41                 "api.organization-events-direct-hit",
  42             )
  43         except NoProjects:
  44             pass
  45         else:
  46             if direct_hit_resp:
  47                 return direct_hit_resp
  48 
  49         full = request.GET.get("full", False)
  50         try:
  51             snuba_args = self.get_snuba_query_args_legacy(request, organization)
  52         except NoProjects:
  53             # return empty result if org doesn't have projects
  54             # or user doesn't have access to projects in org
  55             data_fn = lambda *args, **kwargs: []
  56         else:
  57             data_fn = partial(
  58                 eventstore.get_events,
  59                 referrer="api.organization-events",
  60                 filter=eventstore.Filter(
  61                     start=snuba_args["start"],
  62                     end=snuba_args["end"],
  63                     conditions=snuba_args["conditions"],
  64                     project_ids=snuba_args["filter_keys"].get("project_id", None),
  65                     group_ids=snuba_args["filter_keys"].get("group_id", None),
  66                 ),
  67             )
  68 
  69         serializer = EventSerializer() if full else SimpleEventSerializer()
  70         return self.paginate(
  71             request=request,
  72             on_results=lambda results: serialize(results, request.user, serializer),
  73             paginator=GenericOffsetPaginator(data_fn=data_fn),
  74         )

  76     def handle_results(self, request, organization, project_ids, results):
  77         projects = {
  78             p["id"]: p["slug"]
  79             for p in Project.objects.filter(organization=organization, id__in=project_ids).values(
  80                 "id", "slug"
  81             )
  82         }
  83 
  84         fields = request.GET.getlist("field")
  85 
  86         if "project.name" in fields:
  87             for result in results:
  88                 result["project.name"] = projects[result["project.id"]]
  89                 if "project.id" not in fields:
  90                     del result["project.id"]
  91 
  92         return results

  95 class OrganizationEventsV2Endpoint(OrganizationEventsV2EndpointBase):
  96     def build_cursor_link(self, request, name, cursor):
  97         # The base API function only uses the last query parameter, but this endpoint
  98         # needs all the parameters, particularly for the "field" query param.
  99         querystring = u"&".join(
 100             u"{0}={1}".format(urlquote(query[0]), urlquote(value))
 101             for query in request.GET.lists()
 102             if query[0] != "cursor"
 103             for value in query[1]
 104         )
 105 
 106         base_url = absolute_uri(urlquote(request.path))
 107         if querystring:
 108             base_url = u"{0}?{1}".format(base_url, querystring)
 109         else:
 110             base_url = base_url + "?"
 111 
 112         return LINK_HEADER.format(
 113             uri=base_url,
 114             cursor=six.text_type(cursor),
 115             name=name,
 116             has_results="true" if bool(cursor) else "false",
 117         )

 119     def get(self, request, organization):
 120         if not self.has_feature(organization, request):
 121             return Response(status=404)
 122 
 123         with sentry_sdk.start_span(op="discover.endpoint", description="filter_params") as span:
 124             span.set_tag("organization", organization)
 125             try:
 126                 params = self.get_filter_params(request, organization)
 127             except NoProjects:
 128                 return Response([])
 129             params = self.quantize_date_params(request, params)
 130 
 131             has_global_views = features.has(
 132                 "organizations:global-views", organization, actor=request.user
 133             )
 134             if not has_global_views and len(params.get("project_id", [])) > 1:
 135                 raise ParseError(detail="You cannot view events from multiple projects.")
 136 
 137             if len(request.GET.getlist("field")) > MAX_FIELDS:
 138                 raise ParseError(
 139                     detail="You can view up to {0} fields at a time. Please delete some and try again.".format(
 140                         MAX_FIELDS
 141                     )
 142                 )
 143 
 144         def data_fn(offset, limit):
 145             return discover.query(
 146                 selected_columns=request.GET.getlist("field")[:],
 147                 query=request.GET.get("query"),
 148                 params=params,
 149                 reference_event=self.reference_event(
 150                     request, organization, params.get("start"), params.get("end")
 151                 ),
 152                 orderby=self.get_orderby(request),
 153                 offset=offset,
 154                 limit=limit,
 155                 referrer=request.GET.get("referrer", "api.organization-events-v2"),
 156                 auto_fields=True,
 157                 use_aggregate_conditions=True,
 158             )
 159 
 160         try:
 161             return self.paginate(
 162                 request=request,
 163                 paginator=GenericOffsetPaginator(data_fn=data_fn),
 164                 on_results=lambda results: self.handle_results_with_meta(
 165                     request, organization, params["project_id"], results
 166                 ),
 167             )
 168         except (discover.InvalidSearchQuery, snuba.QueryOutsideRetentionError) as error:
 169             raise ParseError(detail=six.text_type(error))
 170         except snuba.QueryIllegalTypeOfArgument:
 171             raise ParseError(detail="Invalid query. Argument to function is wrong type.")
 172         except snuba.SnubaError as error:
 173             message = "Internal error. Please try again."
 174             if isinstance(
 175                 error,
 176                 (
 177                     snuba.RateLimitExceeded,
 178                     snuba.QueryMemoryLimitExceeded,
 179                     snuba.QueryTooManySimultaneous,
 180                 ),
 181             ):
 182                 message = "Query timeout. Please try again. If the problem persists try a smaller date range or fewer projects."
 183             elif isinstance(
 184                 error,
 185                 (
 186                     snuba.UnqualifiedQueryError,
 187                     snuba.QueryExecutionError,
 188                     snuba.SchemaValidationError,
 189                 ),
 190             ):
 191                 sentry_sdk.capture_exception(error)
 192                 message = "Internal error. Your query failed to run."
 193 
 194             raise ParseError(detail=message)
