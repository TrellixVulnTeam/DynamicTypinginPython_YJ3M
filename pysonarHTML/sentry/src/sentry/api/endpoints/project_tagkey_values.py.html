  12 class ProjectTagKeyValuesEndpoint(ProjectEndpoint, EnvironmentMixin):
  13     doc_section = DocSection.PROJECTS
  14 
  15     def get(self, request, project, key):
  16         """
  17         List a Tag's Values
  18         ```````````````````
  19 
  20         Return a list of values associated with this key.  The `query`
  21         parameter can be used to to perform a "contains" match on
  22         values.
  23 
  24         :pparam string organization_slug: the slug of the organization.
  25         :pparam string project_slug: the slug of the project.
  26         :pparam string key: the tag key to look up.
  27         :auth: required
  28         """
  29         lookup_key = tagstore.prefix_reserved_key(key)
  30 
  31         try:
  32             environment_id = self._get_environment_id_from_request(request, project.organization_id)
  33         except Environment.DoesNotExist:
  34             # if the environment doesn't exist then the tag can't possibly exist
  35             raise ResourceDoesNotExist
  36 
  37         try:
  38             tagkey = tagstore.get_tag_key(project.id, environment_id, lookup_key)
  39         except tagstore.TagKeyNotFound:
  40             raise ResourceDoesNotExist
  41 
  42         start, end = get_date_range_from_params(request.GET)
  43 
  44         paginator = tagstore.get_tag_value_paginator(
  45             project.id,
  46             environment_id,
  47             tagkey.key,
  48             start=start,
  49             end=end,
  50             query=request.GET.get("query"),
  51             order_by="-last_seen",
  52         )
  53 
  54         return self.paginate(
  55             request=request,
  56             paginator=paginator,
  57             on_results=lambda results: serialize(results, request.user),
  58         )
