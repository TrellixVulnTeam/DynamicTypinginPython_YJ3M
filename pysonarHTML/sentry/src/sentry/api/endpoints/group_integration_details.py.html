  21 class GroupIntegrationDetailsEndpoint(GroupEndpoint):
  22     def _has_issue_feature(self, organization, user):
  23         has_issue_basic = features.has(
  24             "organizations:integrations-issue-basic", organization, actor=user
  25         )
  26 
  27         has_issue_sync = features.has(
  28             "organizations:integrations-issue-sync", organization, actor=user
  29         )
  30 
  31         return has_issue_sync or has_issue_basic

  33     def create_issue_activity(self, request, group, installation, external_issue):
  34         issue_information = {
  35             "title": external_issue.title,
  36             "provider": installation.model.get_provider().name,
  37             "location": installation.get_issue_url(external_issue.key),
  38             "label": installation.get_issue_display_name(external_issue) or external_issue.key,
  39         }
  40         Activity.objects.create(
  41             project=group.project,
  42             group=group,
  43             type=Activity.CREATE_ISSUE,
  44             user=request.user,
  45             data=issue_information,
  46         )

  48     @transaction_start("GroupIntegrationDetailsEndpoint")
  49     def get(self, request, group, integration_id):
  50         if not self._has_issue_feature(group.organization, request.user):
  51             return Response({"detail": MISSING_FEATURE_MESSAGE}, status=400)
  52 
  53         # Keep link/create separate since create will likely require
  54         # many external API calls that aren't necessary if the user is
  55         # just linking
  56         action = request.GET.get("action")
  57         if action not in {"link", "create"}:
  58             return Response({"detail": "Action is required and should be either link or create"})
  59 
  60         organization_id = group.project.organization_id
  61         try:
  62             integration = Integration.objects.get(id=integration_id, organizations=organization_id)
  63         except Integration.DoesNotExist:
  64             return Response(status=404)
  65 
  66         if not (
  67             integration.has_feature(IntegrationFeatures.ISSUE_BASIC)
  68             or integration.has_feature(IntegrationFeatures.ISSUE_SYNC)
  69         ):
  70             return Response(
  71                 {"detail": "This feature is not supported for this integration."}, status=400
  72             )
  73 
  74         try:
  75             return Response(
  76                 serialize(
  77                     integration,
  78                     request.user,
  79                     IntegrationIssueConfigSerializer(group, action, params=request.GET),
  80                     organization_id=organization_id,
  81                 )
  82             )
  83         except IntegrationError as e:
  84             return Response({"detail": six.text_type(e)}, status=400)

  87     @transaction_start("GroupIntegrationDetailsEndpoint")
  88     def put(self, request, group, integration_id):
  89         if not self._has_issue_feature(group.organization, request.user):
  90             return Response({"detail": MISSING_FEATURE_MESSAGE}, status=400)
  91 
  92         external_issue_id = request.data.get("externalIssue")
  93         if not external_issue_id:
  94             return Response({"externalIssue": ["Issue ID is required"]}, status=400)
  95 
  96         organization_id = group.project.organization_id
  97         try:
  98             integration = Integration.objects.get(id=integration_id, organizations=organization_id)
  99         except Integration.DoesNotExist:
 100             return Response(status=404)
 101 
 102         if not (
 103             integration.has_feature(IntegrationFeatures.ISSUE_BASIC)
 104             or integration.has_feature(IntegrationFeatures.ISSUE_SYNC)
 105         ):
 106             return Response(
 107                 {"detail": "This feature is not supported for this integration."}, status=400
 108             )
 109 
 110         installation = integration.get_installation(organization_id)
 111         try:
 112             data = installation.get_issue(external_issue_id, data=request.data)
 113         except IntegrationFormError as exc:
 114             return Response(exc.field_errors, status=400)
 115         except IntegrationError as e:
 116             return Response({"non_field_errors": [six.text_type(e)]}, status=400)
 117 
 118         defaults = {
 119             "title": data.get("title"),
 120             "description": data.get("description"),
 121             "metadata": data.get("metadata"),
 122         }
 123 
 124         external_issue_key = installation.make_external_key(data)
 125         external_issue, created = ExternalIssue.objects.get_or_create(
 126             organization_id=organization_id,
 127             integration_id=integration.id,
 128             key=external_issue_key,
 129             defaults=defaults,
 130         )
 131 
 132         if created:
 133             integration_issue_linked.send_robust(
 134                 integration=integration,
 135                 organization=group.project.organization,
 136                 user=request.user,
 137                 sender=self.__class__,
 138             )
 139         else:
 140             external_issue.update(**defaults)
 141 
 142         installation.store_issue_last_defaults(group.project_id, request.data)
 143         try:
 144             installation.after_link_issue(external_issue, data=request.data)
 145         except IntegrationFormError as exc:
 146             return Response(exc.field_errors, status=400)
 147         except IntegrationError as e:
 148             return Response({"non_field_errors": [six.text_type(e)]}, status=400)
 149 
 150         try:
 151             with transaction.atomic():
 152                 GroupLink.objects.create(
 153                     group_id=group.id,
 154                     project_id=group.project_id,
 155                     linked_type=GroupLink.LinkedType.issue,
 156                     linked_id=external_issue.id,
 157                     relationship=GroupLink.Relationship.references,
 158                 )
 159         except IntegrityError:
 160             return Response({"non_field_errors": ["That issue is already linked"]}, status=400)
 161 
 162         self.create_issue_activity(request, group, installation, external_issue)
 163 
 164         # TODO(jess): would be helpful to return serialized external issue
 165         # once we have description, title, etc
 166         url = data.get("url") or installation.get_issue_url(external_issue.key)
 167         context = {
 168             "id": external_issue.id,
 169             "key": external_issue.key,
 170             "url": url,
 171             "integrationId": external_issue.integration_id,
 172             "displayName": installation.get_issue_display_name(external_issue),
 173         }
 174         return Response(context, status=201)

 179             return Response({"detail": MISSING_FEATURE_MESSAGE}, status=400)

 249     @transaction_start("GroupIntegrationDetailsEndpoint")
 250     def delete(self, request, group, integration_id):
 251         if not self._has_issue_feature(group.organization, request.user):
 252             return Response({"detail": MISSING_FEATURE_MESSAGE}, status=400)
 253 
 254         # note here externalIssue refers to `ExternalIssue.id` whereas above
 255         # it refers to the id from the provider
 256         external_issue_id = request.GET.get("externalIssue")
 257         if not external_issue_id:
 258             return Response({"detail": "External ID required"}, status=400)
 259 
 260         organization_id = group.project.organization_id
 261         try:
 262             integration = Integration.objects.get(id=integration_id, organizations=organization_id)
 263         except Integration.DoesNotExist:
 264             return Response(status=404)
 265 
 266         if not (
 267             integration.has_feature(IntegrationFeatures.ISSUE_BASIC)
 268             or integration.has_feature(IntegrationFeatures.ISSUE_SYNC)
 269         ):
 270             return Response(
 271                 {"detail": "This feature is not supported for this integration."}, status=400
 272             )
 273 
 274         try:
 275             external_issue = ExternalIssue.objects.get(
 276                 organization_id=organization_id, integration_id=integration.id, id=external_issue_id
 277             )
 278         except ExternalIssue.DoesNotExist:
 279             return Response(status=404)
 280 
 281         with transaction.atomic():
 282             GroupLink.objects.filter(
 283                 group_id=group.id,
 284                 project_id=group.project_id,
 285                 linked_type=GroupLink.LinkedType.issue,
 286                 linked_id=external_issue_id,
 287                 relationship=GroupLink.Relationship.references,
 288             ).delete()
 289 
 290             # check if other groups reference this external issue
 291             # and delete if not
 292             if not GroupLink.objects.filter(
 293                 linked_type=GroupLink.LinkedType.issue, linked_id=external_issue_id
 294             ).exists():
 295                 external_issue.delete()
 296 
 297         return Response(status=204)
