  70 def get_serializer_field_metadata(serializer, fields=None):
  71     """Returns field metadata for serializer"""
  72     meta = []
  73     for field_name, field in serializer.fields.items():
  74         if (fields is None or field_name in fields) and field_name:
  75             try:
  76                 default = field.get_default()
  77             except SkipField:
  78                 default = None
  79             serialized_field = {
  80                 "name": field_name,
  81                 "defaultValue": default,
  82                 "read_only": field.read_only,
  83                 "required": field.required,
  84                 "type": "string",
  85             }
  86             if hasattr(field, "max_length") and field.max_length:
  87                 serialized_field["max_length"] = field.max_length
  88             if field.label:
  89                 serialized_field["label"] = field.label
  90 
  91             meta.append(serialized_field)
  92 
  93     return meta

  96 class UserAuthenticatorEnrollEndpoint(UserEndpoint):
  97     @sudo_required
  98     def get(self, request, user, interface_id):
  99         """
 100         Get Authenticator Interface
 101         ```````````````````````````
 102 
 103         Retrieves authenticator interface details for user depending on user enrollment status
 104 
 105         :pparam string user_id: user id or "me" for current user
 106         :pparam string interface_id: interface id
 107 
 108         :auth: required
 109         """
 110 
 111         interface = Authenticator.objects.get_interface(user, interface_id)
 112 
 113         # Not all interfaces allow multi enrollment
 114         if interface.is_enrolled and not interface.allow_multi_enrollment:
 115             return Response(ALREADY_ENROLLED_ERR, status=status.HTTP_400_BAD_REQUEST)
 116 
 117         # User is not enrolled in auth interface:
 118         # - display configuration form
 119         response = serialize(interface)
 120         response["form"] = get_serializer_field_metadata(serializer_map[interface_id]())
 121 
 122         # U2fInterface has no 'secret' attribute
 123         try:
 124             response["secret"] = interface.secret
 125         except AttributeError:
 126             pass
 127 
 128         if interface_id == "totp":
 129             response["qrcode"] = interface.get_provision_qrcode(user.email)
 130 
 131         if interface_id == "u2f":
 132             response["challenge"] = interface.start_enrollment()
 133             # XXX: Upgrading python-u2flib-server to 5.0.0 changes the response
 134             # format. Our current js u2f library expects the old format, so
 135             # massaging the data to include appId here
 136             app_id = response["challenge"]["appId"]
 137             for register_request in response["challenge"]["registerRequests"]:
 138                 register_request["appId"] = app_id
 139 
 140         return Response(response)
 141 
 142     @sudo_required
 143     def post(self, request, user, interface_id):
 144         """
 145         Enroll in authenticator interface
 146         `````````````````````````````````
 147 
 148         :pparam string user_id: user id or "me" for current user
 149         :pparam string interface_id: interface id
 150 
 151         :auth: required
 152         """
 153 
 154         # Using `request.user` here because superuser should not be able to set a user's 2fa
 155 
 156         # start activation
 157         serializer_cls = serializer_map.get(interface_id, None)
 158 
 159         if serializer_cls is None:
 160             return Response(status=status.HTTP_404_NOT_FOUND)
 161 
 162         serializer = serializer_cls(data=request.data)
 163 
 164         if not serializer.is_valid():
 165             return Response(serializer.errors, status=status.HTTP_400_BAD_REQUEST)
 166 
 167         interface = Authenticator.objects.get_interface(request.user, interface_id)
 168 
 169         # Not all interfaces allow multi enrollment
 170         #
 171         # This is probably un-needed because we catch
 172         # `Authenticator.AlreadyEnrolled` when attempting to enroll
 173         if interface.is_enrolled and not interface.allow_multi_enrollment:
 174             return Response(ALREADY_ENROLLED_ERR, status=status.HTTP_400_BAD_REQUEST)
 175 
 176         try:
 177             interface.secret = request.data["secret"]
 178         except KeyError:
 179             pass
 180 
 181         context = {}
 182         # Need to update interface with phone number before validating OTP
 183         if "phone" in request.data:
 184             interface.phone_number = serializer.data["phone"]
 185 
 186             # Disregarding value of 'otp', if no OTP was provided,
 187             # send text message to phone number with OTP
 188             if "otp" not in request.data:
 189                 if interface.send_text(for_enrollment=True, request=request._request):
 190                     return Response(status=status.HTTP_204_NO_CONTENT)
 191                 else:
 192                     # Error sending text message
 193                     return Response(SEND_SMS_ERR, status=status.HTTP_500_INTERNAL_SERVER_ERROR)
 194 
 195         # Attempt to validate OTP
 196         if "otp" in request.data and not interface.validate_otp(serializer.data["otp"]):
 197             return Response(INVALID_OTP_ERR, status=status.HTTP_400_BAD_REQUEST)
 198 
 199         # Try u2f enrollment
 200         if interface_id == "u2f":
 201             # What happens when this fails?
 202             interface.try_enroll(
 203                 serializer.data["challenge"],
 204                 serializer.data["response"],
 205                 serializer.data["deviceName"],
 206             )
 207             context.update({"device_name": serializer.data["deviceName"]})
 208 
 209         try:
 210             interface.enroll(request.user)
 211         except Authenticator.AlreadyEnrolled:
 212             return Response(ALREADY_ENROLLED_ERR, status=status.HTTP_400_BAD_REQUEST)
 213 
 214         context.update({"authenticator": interface.authenticator})
 215         capture_security_activity(
 216             account=request.user,
 217             type="mfa-added",
 218             actor=request.user,
 219             ip_address=request.META["REMOTE_ADDR"],
 220             context=context,
 221             send_email=True,
 222         )
 223         request.user.clear_lost_passwords()
 224         request.user.refresh_session_nonce(self.request)
 225         request.user.save()
 226         Authenticator.objects.auto_add_recovery_codes(request.user)
 227 
 228         response = Response(status=status.HTTP_204_NO_CONTENT)
 229 
 230         # If there is a pending organization invite accept after the
 231         # authenticator has been configured.
 232         invite_helper = ApiInviteHelper.from_cookie(request=request, instance=self, logger=logger)
 233 
 234         if invite_helper and invite_helper.valid_request:
 235             invite_helper.accept_invite()
 236             remove_invite_cookie(request, response)
 237 
 238         return response
