  21 class TeamStatsEndpoint(TeamEndpoint, EnvironmentMixin, StatsMixin):
  22     doc_section = DocSection.TEAMS
  23 
  24     @attach_scenarios([retrieve_event_counts_team])
  25     def get(self, request, team):
  26         """
  27         Retrieve Event Counts for a Team
  28         ````````````````````````````````
  29 
  30         .. caution::
  31            This endpoint may change in the future without notice.
  32 
  33         Return a set of points representing a normalized timestamp and the
  34         number of events seen in the period.
  35 
  36         Query ranges are limited to Sentry's configured time-series
  37         resolutions.
  38 
  39         :pparam string organization_slug: the slug of the organization.
  40         :pparam string team_slug: the slug of the team.
  41         :qparam string stat: the name of the stat to query (``"received"``,
  42                              ``"rejected"``)
  43         :qparam timestamp since: a timestamp to set the start of the query
  44                                  in seconds since UNIX epoch.
  45         :qparam timestamp until: a timestamp to set the end of the query
  46                                  in seconds since UNIX epoch.
  47         :qparam string resolution: an explicit resolution to search
  48                                    for (one of ``10s``, ``1h``, and ``1d``)
  49         :auth: required
  50         """
  51         try:
  52             environment_id = self._get_environment_id_from_request(request, team.organization_id)
  53         except Environment.DoesNotExist:
  54             raise ResourceDoesNotExist
  55 
  56         projects = Project.objects.get_for_user(team=team, user=request.user)
  57 
  58         if not projects:
  59             return Response([])
  60 
  61         data = list(
  62             tsdb.get_range(
  63                 model=tsdb.models.project,
  64                 keys=[p.id for p in projects],
  65                 **self._parse_args(request, environment_id)
  66             ).values()
  67         )
  68 
  69         summarized = []
  70         for n in range(len(data[0])):
  71             total = sum(d[n][1] for d in data)
  72             summarized.append((data[0][n][0], total))
  73 
  74         return Response(summarized)
