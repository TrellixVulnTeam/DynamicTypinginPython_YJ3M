  55     @attach_scenarios([retrieve_organization_release_scenario])
  56     def get(self, request, organization, version):
  57         """
  58         Retrieve an Organization's Release
  59         ``````````````````````````````````
  60 
  61         Return details on an individual release.
  62 
  63         :pparam string organization_slug: the slug of the organization the
  64                                           release belongs to.
  65         :pparam string version: the version identifier of the release.
  66         :auth: required
  67         """
  68         project_id = request.GET.get("project")
  69         with_health = request.GET.get("health") == "1"
  70         summary_stats_period = request.GET.get("summaryStatsPeriod") or "14d"
  71         health_stats_period = request.GET.get("healthStatsPeriod") or ("24h" if with_health else "")
  72         if summary_stats_period not in STATS_PERIODS:
  73             raise ParseError(detail=get_stats_period_detail("summaryStatsPeriod", STATS_PERIODS))
  74         if health_stats_period and health_stats_period not in STATS_PERIODS:
  75             raise ParseError(detail=get_stats_period_detail("healthStatsPeriod", STATS_PERIODS))
  76 
  77         try:
  78             release = Release.objects.get(organization_id=organization.id, version=version)
  79         except Release.DoesNotExist:
  80             raise ResourceDoesNotExist
  81 
  82         if not self.has_release_permission(request, organization, release):
  83             raise ResourceDoesNotExist
  84 
  85         if with_health and project_id:
  86             try:
  87                 project = Project.objects.get_from_cache(id=int(project_id))
  88             except (ValueError, Project.DoesNotExist):
  89                 raise ParseError(detail="Invalid project")
  90             release._for_project_id = project.id
  91 
  92         return Response(
  93             serialize(
  94                 release,
  95                 request.user,
  96                 with_health_data=with_health,
  97                 summary_stats_period=summary_stats_period,
  98                 health_stats_period=health_stats_period,
  99             )
 100         )

 102     @attach_scenarios([update_organization_release_scenario])
 103     def put(self, request, organization, version):
 104         """
 105         Update an Organization's Release
 106         ````````````````````````````````
 107 
 108         Update a release. This can change some metadata associated with
 109         the release (the ref, url, and dates).
 110 
 111         :pparam string organization_slug: the slug of the organization the
 112                                           release belongs to.
 113         :pparam string version: the version identifier of the release.
 114         :param string ref: an optional commit reference.  This is useful if
 115                            a tagged version has been provided.
 116         :param url url: a URL that points to the release.  This can be the
 117                         path to an online interface to the sourcecode
 118                         for instance.
 119         :param datetime dateReleased: an optional date that indicates when
 120                                       the release went live.  If not provided
 121                                       the current time is assumed.
 122         :param array commits: an optional list of commit data to be associated
 123                               with the release. Commits must include parameters
 124                               ``id`` (the sha of the commit), and can optionally
 125                               include ``repository``, ``message``, ``author_name``,
 126                               ``author_email``, and ``timestamp``.
 127         :param array refs: an optional way to indicate the start and end commits
 128                            for each repository included in a release. Head commits
 129                            must include parameters ``repository`` and ``commit``
 130                            (the HEAD sha). They can optionally include ``previousCommit``
 131                            (the sha of the HEAD of the previous release), which should
 132                            be specified if this is the first time you've sent commit data.
 133         :auth: required
 134         """
 135         try:
 136             release = Release.objects.get(organization_id=organization, version=version)
 137         except Release.DoesNotExist:
 138             raise ResourceDoesNotExist
 139 
 140         if not self.has_release_permission(request, organization, release):
 141             raise ResourceDoesNotExist
 142 
 143         serializer = OrganizationReleaseSerializer(data=request.data)
 144 
 145         if not serializer.is_valid():
 146             return Response(serializer.errors, status=400)
 147 
 148         result = serializer.validated_data
 149 
 150         was_released = bool(release.date_released)
 151 
 152         kwargs = {}
 153         if result.get("dateReleased"):
 154             kwargs["date_released"] = result["dateReleased"]
 155         if result.get("ref"):
 156             kwargs["ref"] = result["ref"]
 157         if result.get("url"):
 158             kwargs["url"] = result["url"]
 159 
 160         if kwargs:
 161             release.update(**kwargs)
 162 
 163         commit_list = result.get("commits")
 164         if commit_list:
 165             # TODO(dcramer): handle errors with release payloads
 166             release.set_commits(commit_list)
 167 
 168         refs = result.get("refs")
 169         if not refs:
 170             refs = [
 171                 {
 172                     "repository": r["repository"],
 173                     "previousCommit": r.get("previousId"),
 174                     "commit": r["currentId"],
 175                 }
 176                 for r in result.get("headCommits", [])
 177             ]
 178         if refs:
 179             if not request.user.is_authenticated():
 180                 return Response(
 181                     {"refs": ["You must use an authenticated API token to fetch refs"]}, status=400
 182                 )
 183             fetch_commits = not commit_list
 184             try:
 185                 release.set_refs(refs, request.user, fetch=fetch_commits)
 186             except InvalidRepository as e:
 187                 return Response({"refs": [six.text_type(e)]}, status=400)
 188 
 189         if not was_released and release.date_released:
 190             for project in release.projects.all():
 191                 Activity.objects.create(
 192                     type=Activity.RELEASE,
 193                     project=project,
 194                     ident=Activity.get_version_ident(release.version),
 195                     data={"version": release.version},
 196                     datetime=release.date_released,
 197                 )
 198 
 199         return Response(serialize(release, request.user))

 201     def delete(self, request, organization, version):
 202         """
 203         Delete an Organization's Release
 204         ````````````````````````````````
 205 
 206         Permanently remove a release and all of its files.
 207 
 208         :pparam string organization_slug: the slug of the organization the
 209                                           release belongs to.
 210         :pparam string version: the version identifier of the release.
 211         :auth: required
 212         """
 213         try:
 214             release = Release.objects.get(organization_id=organization.id, version=version)
 215         except Release.DoesNotExist:
 216             raise ResourceDoesNotExist
 217 
 218         if not self.has_release_permission(request, organization, release):
 219             raise ResourceDoesNotExist
 220 
 221         try:
 222             release.safe_delete()
 223         except UnsafeReleaseDeletion as e:
 224             return Response({"detail": six.text_type(e)}, status=400)
 225 
 226         return Response(status=204)
