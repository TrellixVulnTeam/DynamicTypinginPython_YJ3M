  70     def download(self, releasefile):
  71         file = releasefile.file
  72         fp = file.getfile()
  73         response = StreamingHttpResponse(
  74             iter(lambda: fp.read(4096), b""),
  75             content_type=file.headers.get("content-type", "application/octet-stream"),
  76         )
  77         response["Content-Length"] = file.size
  78         response["Content-Disposition"] = 'attachment; filename="%s"' % posixpath.basename(
  79             " ".join(releasefile.name.split())
  80         )
  81         return response

  83     @attach_scenarios([retrieve_file_scenario])
  84     def get(self, request, project, version, file_id):
  85         """
  86         Retrieve a Project Release's File
  87         `````````````````````````````````
  88 
  89         Return details on an individual file within a release.  This does
  90         not actually return the contents of the file, just the associated
  91         metadata.
  92 
  93         :pparam string organization_slug: the slug of the organization the
  94                                           release belongs to.
  95         :pparam string project_slug: the slug of the project to retrieve the
  96                                      file of.
  97         :pparam string version: the version identifier of the release.
  98         :pparam string file_id: the ID of the file to retrieve.
  99         :auth: required
 100         """
 101         try:
 102             release = Release.objects.get(
 103                 organization_id=project.organization_id, projects=project, version=version
 104             )
 105         except Release.DoesNotExist:
 106             raise ResourceDoesNotExist
 107 
 108         try:
 109             releasefile = ReleaseFile.objects.get(release=release, id=file_id)
 110         except ReleaseFile.DoesNotExist:
 111             raise ResourceDoesNotExist
 112 
 113         download_requested = request.GET.get("download") is not None
 114         if download_requested and (request.access.has_scope("project:write")):
 115             return self.download(releasefile)
 116         elif download_requested:
 117             return Response(status=403)
 118         return Response(serialize(releasefile, request.user))

 120     @attach_scenarios([update_file_scenario])
 121     def put(self, request, project, version, file_id):
 122         """
 123         Update a File
 124         `````````````
 125 
 126         Update metadata of an existing file.  Currently only the name of
 127         the file can be changed.
 128 
 129         :pparam string organization_slug: the slug of the organization the
 130                                           release belongs to.
 131         :pparam string project_slug: the slug of the project to update the
 132                                      file of.
 133         :pparam string version: the version identifier of the release.
 134         :pparam string file_id: the ID of the file to update.
 135         :param string name: the new name of the file.
 136         :auth: required
 137         """
 138         try:
 139             release = Release.objects.get(
 140                 organization_id=project.organization_id, projects=project, version=version
 141             )
 142         except Release.DoesNotExist:
 143             raise ResourceDoesNotExist
 144 
 145         try:
 146             releasefile = ReleaseFile.objects.get(release=release, id=file_id)
 147         except ReleaseFile.DoesNotExist:
 148             raise ResourceDoesNotExist
 149 
 150         serializer = ReleaseFileSerializer(data=request.data)
 151 
 152         if not serializer.is_valid():
 153             return Response(serializer.errors, status=400)
 154 
 155         result = serializer.validated_data
 156 
 157         releasefile.update(name=result["name"])
 158 
 159         return Response(serialize(releasefile, request.user))

 161     @attach_scenarios([delete_file_scenario])
 162     def delete(self, request, project, version, file_id):
 163         """
 164         Delete a File
 165         `````````````
 166 
 167         Permanently remove a file from a release.
 168 
 169         This will also remove the physical file from storage.
 170 
 171         :pparam string organization_slug: the slug of the organization the
 172                                           release belongs to.
 173         :pparam string project_slug: the slug of the project to delete the
 174                                      file of.
 175         :pparam string version: the version identifier of the release.
 176         :pparam string file_id: the ID of the file to delete.
 177         :auth: required
 178         """
 179         try:
 180             release = Release.objects.get(
 181                 organization_id=project.organization_id, projects=project, version=version
 182             )
 183         except Release.DoesNotExist:
 184             raise ResourceDoesNotExist
 185 
 186         try:
 187             releasefile = ReleaseFile.objects.get(release=release, id=file_id)
 188         except ReleaseFile.DoesNotExist:
 189             raise ResourceDoesNotExist
 190 
 191         file = releasefile.file
 192 
 193         # TODO(dcramer): this doesnt handle a failure from file.deletefile() to
 194         # the actual deletion of the db row
 195         releasefile.delete()
 196         file.delete()
 197 
 198         return Response(status=204)
