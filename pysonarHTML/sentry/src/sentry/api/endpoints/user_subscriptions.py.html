  32         # This returns a dict with `subscriber` and `subscriptions`

  53     def put(self, request, user):
  54         """
  55         Update Account Subscriptions
  56         ````````````````````````````
  57 
  58         Update account subscriptions to newsletter
  59 
  60         :param int listId: id of newsletter list
  61         :param boolean subscribed: should be subscribed to newsletter
  62         :auth: required
  63         """
  64         validator = NewsletterValidator(data=request.data)
  65         if not validator.is_valid():
  66             return self.respond(validator.errors, status=400)
  67 
  68         result = validator.validated_data
  69         email = UserEmail.get_primary_email(user)
  70 
  71         kwargs = {
  72             "list_id": result["listId"],
  73             "subscribed": result["subscribed"],
  74             "verified": email.is_verified,
  75         }
  76         if not result["subscribed"]:
  77             kwargs["unsubscribed_date"] = timezone.now()
  78         else:
  79             kwargs["subscribed_date"] = timezone.now()
  80 
  81         newsletter.create_or_update_subscription(user, **kwargs)
  82         return self.respond(status=204)

  84     def post(self, request, user):
  85         """
  86         Configure Newsletter Subscription
  87         `````````````````````````````````
  88 
  89         Update the default newsletter subscription.
  90 
  91         :param boolean subscribed: should be subscribed to newsletter
  92         :auth: required
  93         """
  94         validator = DefaultNewsletterValidator(data=request.data)
  95         if not validator.is_valid():
  96             return self.respond(validator.errors, status=400)
  97 
  98         result = validator.validated_data
  99         email = UserEmail.get_primary_email(user)
 100 
 101         kwargs = {
 102             "subscribed": result["subscribed"],
 103             "verified": email.is_verified,
 104             "list_ids": newsletter.get_default_list_ids(),
 105         }
 106         if not result["subscribed"]:
 107             kwargs["unsubscribed_date"] = timezone.now()
 108         else:
 109             kwargs["subscribed_date"] = timezone.now()
 110 
 111         newsletter.create_or_update_subscriptions(user, **kwargs)
 112 
 113         user.update(flags=F("flags").bitand(~User.flags.newsletter_consent_prompt))
 114 
 115         return self.respond(status=204)
