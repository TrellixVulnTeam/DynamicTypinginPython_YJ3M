  21 class ProjectStatsEndpoint(ProjectEndpoint, EnvironmentMixin, StatsMixin):
  22     doc_section = DocSection.PROJECTS
  23 
  24     @attach_scenarios([retrieve_event_counts_project])
  25     def get(self, request, project):
  26         """
  27         Retrieve Event Counts for a Project
  28         ```````````````````````````````````
  29 
  30         .. caution::
  31            This endpoint may change in the future without notice.
  32 
  33         Return a set of points representing a normalized timestamp and the
  34         number of events seen in the period.
  35 
  36         Query ranges are limited to Sentry's configured time-series
  37         resolutions.
  38 
  39         :pparam string organization_slug: the slug of the organization.
  40         :pparam string project_slug: the slug of the project.
  41         :qparam string stat: the name of the stat to query (``"received"``,
  42                              ``"rejected"``, ``"blacklisted"``, ``generated``)
  43         :qparam timestamp since: a timestamp to set the start of the query
  44                                  in seconds since UNIX epoch.
  45         :qparam timestamp until: a timestamp to set the end of the query
  46                                  in seconds since UNIX epoch.
  47         :qparam string resolution: an explicit resolution to search
  48                                    for (one of ``10s``, ``1h``, and ``1d``)
  49         :auth: required
  50         """
  51         stat = request.GET.get("stat", "received")
  52         query_kwargs = {}
  53         if stat == "received":
  54             stat_model = tsdb.models.project_total_received
  55         elif stat == "rejected":
  56             stat_model = tsdb.models.project_total_rejected
  57         elif stat == "blacklisted":
  58             stat_model = tsdb.models.project_total_blacklisted
  59         elif stat == "generated":
  60             stat_model = tsdb.models.project
  61             try:
  62                 query_kwargs["environment_id"] = self._get_environment_id_from_request(
  63                     request, project.organization_id
  64                 )
  65             except Environment.DoesNotExist:
  66                 raise ResourceDoesNotExist
  67         elif stat == "forwarded":
  68             stat_model = tsdb.models.project_total_forwarded
  69         else:
  70             try:
  71                 stat_model = FILTER_STAT_KEYS_TO_VALUES[stat]
  72             except KeyError:
  73                 raise ValueError("Invalid stat: %s" % stat)
  74 
  75         data = tsdb.get_range(
  76             model=stat_model, keys=[project.id], **self._parse_args(request, **query_kwargs)
  77         )[project.id]
  78 
  79         return Response(data)
