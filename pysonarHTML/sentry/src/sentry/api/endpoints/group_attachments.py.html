  23 class GroupAttachmentsEndpoint(GroupEndpoint, EnvironmentMixin):
  24     def get(self, request, group):
  25         """
  26         List Event Attachments
  27         ``````````````````````
  28 
  29         Returns a list of event attachments for an issue.
  30 
  31         :pparam string issue_id: the ID of the issue to retrieve.
  32         :pparam list   types:    a list of attachment types to filter for.
  33         :auth: required
  34         """
  35 
  36         if not features.has(
  37             "organizations:event-attachments", group.project.organization, actor=request.user
  38         ):
  39             return self.respond(status=404)
  40 
  41         attachments = EventAttachment.objects.filter(group_id=group.id).select_related("file")
  42 
  43         types = request.GET.getlist("types") or ()
  44         if types:
  45             attachments = attachments.filter(file__type__in=types)
  46 
  47         return self.paginate(
  48             default_per_page=20,
  49             request=request,
  50             queryset=attachments,
  51             order_by="-date_added",
  52             on_results=lambda x: serialize(x, request.user, GroupEventAttachmentSerializer()),
