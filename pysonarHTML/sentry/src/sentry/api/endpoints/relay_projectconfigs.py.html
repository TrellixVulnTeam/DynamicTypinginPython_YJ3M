  26 def _sample_apm():
  27     return random.random() < getattr(settings, "SENTRY_RELAY_ENDPOINT_APM_SAMPLING", 0)

  30 class RelayProjectConfigsEndpoint(Endpoint):
  31     authentication_classes = (RelayAuthentication,)
  32     permission_classes = (RelayPermission,)
  33 
  34     def post(self, request):
  35         with Hub.current.start_span(
  36             Span(op="http.server", transaction="RelayProjectConfigsEndpoint", sampled=_sample_apm())
  37         ):
  38             return self._post(request)
  39 
  40     def _post(self, request):
  41         relay = request.relay
  42         assert relay is not None  # should be provided during Authentication
  43 
  44         full_config_requested = request.relay_request_data.get("fullConfig")
  45 
  46         if full_config_requested and not relay.is_internal:
  47             return Response("Relay unauthorized for full config information", 403)
  48 
  49         with Hub.current.start_span(op="relay_fetch_projects"):
  50             project_ids = set(request.relay_request_data.get("projects") or ())
  51             if project_ids:
  52                 with metrics.timer("relay_project_configs.fetching_projects.duration"):
  53                     projects = {p.id: p for p in Project.objects.get_many_from_cache(project_ids)}
  54             else:
  55                 projects = {}
  56 
  57         with Hub.current.start_span(op="relay_fetch_orgs"):
  58             # Preload all organizations and their options to prevent repeated
  59             # database access when computing the project configuration.
  60             org_ids = set(project.organization_id for project in six.itervalues(projects))
  61             if org_ids:
  62                 with metrics.timer("relay_project_configs.fetching_orgs.duration"):
  63                     orgs = Organization.objects.get_many_from_cache(org_ids)
  64                     orgs = {o.id: o for o in orgs if request.relay.has_org_access(o)}
  65             else:
  66                 orgs = {}
  67             org_options = {
  68                 i: OrganizationOption.objects.get_all_values(i) for i in six.iterkeys(orgs)
  69             }
  70 
  71         with Hub.current.start_span(op="relay_fetch_keys"):
  72             project_keys = {}
  73             for key in ProjectKey.objects.filter(project_id__in=project_ids):
  74                 project_keys.setdefault(key.project_id, []).append(key)
  75 
  76         metrics.timing("relay_project_configs.projects_requested", len(project_ids))
  77         metrics.timing("relay_project_configs.projects_fetched", len(projects))
  78         metrics.timing("relay_project_configs.orgs_fetched", len(orgs))
  79 
  80         configs = {}
  81         for project_id in project_ids:
  82             configs[six.text_type(project_id)] = {"disabled": True}
  83 
  84             project = projects.get(int(project_id))
  85             if project is None:
  86                 continue
  87 
  88             organization = orgs.get(project.organization_id)
  89             if organization is None:
  90                 continue
  91 
  92             # Try to prevent organization from being fetched again in quotas.
  93             project.organization = organization
  94             project._organization_cache = organization
  95 
  96             with Hub.current.start_span(op="get_config"):
  97                 with metrics.timer("relay_project_configs.get_config.duration"):
  98                     project_config = config.get_project_config(
  99                         project,
 100                         org_options=org_options.get(organization.id) or {},
 101                         full_config=full_config_requested,
 102                         project_keys=project_keys.get(project.id) or [],
 103                     )
 104 
 105             configs[six.text_type(project_id)] = project_config.to_dict()
 106 
 107         if full_config_requested:
 108             projectconfig_cache.set_many(configs)
 109 
 110         return Response({"configs": configs}, status=200)
