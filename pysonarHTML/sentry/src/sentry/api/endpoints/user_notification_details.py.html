  46 class UserNotificationsSerializer(Serializer):
  47     def get_attrs(self, item_list, user, *args, **kwargs):
  48         data = list(
  49             UserOption.objects.filter(
  50                 user__in=item_list, organization=None, project=None
  51             ).select_related("user")
  52         )
  53 
  54         results = defaultdict(list)
  55 
  56         for uo in data:
  57             results[uo.user].append(uo)
  58 
  59         return results
  60 
  61     def serialize(self, obj, attrs, user, *args, **kwargs):
  62 
  63         raw_data = {option.key: option.value for option in attrs}
  64 
  65         data = {}
  66         for key in USER_OPTION_SETTINGS:
  67             uo = USER_OPTION_SETTINGS[key]
  68             val = raw_data.get(uo["key"], uo["default"])
  69             if uo["type"] == bool:
  70                 data[key] = bool(int(val))  # '1' is true, '0' is false
  71             elif uo["type"] == int:
  72                 data[key] = int(val)
  73 
  74         data["weeklyReports"] = True  # This cannot be overridden
  75 
  76         return data

  99         if serializer.is_valid():

 101                 db_key = USER_OPTION_SETTINGS[key]["key"]

 103                 (uo, created) = UserOption.objects.get_or_create(
