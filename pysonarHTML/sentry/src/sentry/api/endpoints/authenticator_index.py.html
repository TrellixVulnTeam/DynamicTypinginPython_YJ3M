  10 class AuthenticatorIndexEndpoint(Endpoint):

  13     def get(self, request):
  14         """Returns u2f interface for a user, otherwise an empty array
  15         """
  16 
  17         # Currently just expose u2f challenge, not sure if it's necessary to list all
  18         # authenticator interfaces that are enabled
  19         try:
  20             interface = Authenticator.objects.get_interface(request.user, "u2f")
  21             if not interface.is_enrolled:
  22                 raise LookupError()
  23         except LookupError:
  24             return Response([])
  25 
  26         challenge = interface.activate(request._request).challenge
  27 
  28         # I don't think we currently support multiple interfaces of the same type
  29         # but just future proofing I guess
  30         return Response([{"id": "u2f", "challenge": challenge}])
