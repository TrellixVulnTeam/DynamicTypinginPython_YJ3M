  24 class EventIdLookupEndpoint(OrganizationEndpoint):
  25     doc_section = DocSection.ORGANIZATIONS
  26 
  27     @attach_scenarios([resolve_event_id_scenario])
  28     def get(self, request, organization, event_id):
  29         """
  30         Resolve a Event ID
  31         ``````````````````
  32 
  33         This resolves a event ID to the project slug and internal issue ID and internal event ID.
  34 
  35         :pparam string organization_slug: the slug of the organization the
  36                                           event ID should be looked up in.
  37         :param string event_id: the event ID to look up.
  38         :auth: required
  39         """
  40         # Largely copied from ProjectGroupIndexEndpoint
  41         if len(event_id) != 32:
  42             return Response({"detail": "Event ID must be 32 characters."}, status=400)
  43 
  44         project_slugs_by_id = dict(
  45             Project.objects.filter(organization=organization).values_list("id", "slug")
  46         )
  47 
  48         try:
  49             snuba_filter = eventstore.Filter(
  50                 conditions=[["event.type", "!=", "transaction"]],
  51                 project_ids=project_slugs_by_id.keys(),
  52                 event_ids=[event_id],
  53             )
  54             event = eventstore.get_events(filter=snuba_filter, limit=1)[0]
  55         except IndexError:
  56             raise ResourceDoesNotExist()
  57         else:
  58             return Response(
  59                 {
  60                     "organizationSlug": organization.slug,
  61                     "projectSlug": project_slugs_by_id[event.project_id],
  62                     "groupId": six.text_type(event.group_id),
  63                     "eventId": six.text_type(event.event_id),
  64                     "event": serialize(event, request.user),
  65                 }
  66             )
