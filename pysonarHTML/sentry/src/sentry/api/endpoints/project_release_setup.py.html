  11 class ProjectReleaseSetupCompletionEndpoint(ProjectEndpoint):
  12     permission_classes = (ProjectReleasePermission,)
  13 
  14     def get(self, request, project):
  15         """
  16         Get list with release setup progress for a project
  17         1. tag an error
  18         2. link a repo
  19         3. associate commits
  20         4. tell sentry about a deploy
  21         """
  22 
  23         tag_key = "onboard_tag:1:%s" % (project.id)
  24         repo_key = "onboard_repo:1:%s" % (project.organization_id)
  25         commit_key = "onboard_commit:1:%s" % hash_values([project.organization_id, project.id])
  26         deploy_key = "onboard_deploy:1:%s" % hash_values([project.organization_id, project.id])
  27         onboard_cache = cache.get_many([tag_key, repo_key, commit_key, deploy_key])
  28 
  29         tag = onboard_cache.get(tag_key)
  30         if tag is None:
  31             tag = Group.objects.filter(project=project.id, first_release__isnull=False).exists()
  32             cache.set(tag_key, tag, 3600 if tag else 60)
  33 
  34         repo = onboard_cache.get(repo_key)
  35         if repo is None:
  36             repo = Repository.objects.filter(organization_id=project.organization_id).exists()
  37             cache.set(repo_key, repo, 3600 if repo else 60)
  38 
  39         commit = onboard_cache.get(commit_key)
  40         if commit is None:
  41             commit = ReleaseCommit.objects.filter(
  42                 organization_id=project.organization_id, release__projects=project.id
  43             ).exists()
  44             cache.set(commit_key, commit, 3600 if commit else 60)
  45 
  46         deploy = onboard_cache.get(deploy_key)
  47         if deploy is None:
  48             deploy = Deploy.objects.filter(
  49                 organization_id=project.organization_id, release__projects=project.id
  50             ).exists()
  51             cache.set(deploy_key, deploy, 3600 if deploy else 60)
  52 
  53         return Response(
  54             [
  55                 {"step": "tag", "complete": bool(tag)},
  56                 {"step": "repo", "complete": bool(repo)},
  57                 {"step": "commit", "complete": bool(commit)},
  58                 {"step": "deploy", "complete": bool(deploy)},
  59             ]
  60         )
