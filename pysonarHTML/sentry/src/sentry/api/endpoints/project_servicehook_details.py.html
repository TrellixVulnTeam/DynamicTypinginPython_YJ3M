  15 class ProjectServiceHookDetailsEndpoint(ProjectEndpoint):
  16     doc_section = DocSection.PROJECTS
  17 
  18     def get(self, request, project, hook_id):
  19         """
  20         Retrieve a Service Hook
  21         ```````````````````````
  22 
  23         Return a service hook bound to a project.
  24 
  25         :pparam string organization_slug: the slug of the organization the
  26                                           client keys belong to.
  27         :pparam string project_slug: the slug of the project the client keys
  28                                      belong to.
  29         :pparam string hook_id: the guid of the service hook.
  30         :auth: required
  31         """
  32         try:
  33             hook = ServiceHook.objects.get(project_id=project.id, guid=hook_id)
  34         except ServiceHook.DoesNotExist:
  35             raise ResourceDoesNotExist
  36         return self.respond(serialize(hook, request.user))

  38     def put(self, request, project, hook_id):
  39         """
  40         Update a Service Hook
  41         `````````````````````
  42 
  43         :pparam string organization_slug: the slug of the organization the
  44                                           client keys belong to.
  45         :pparam string project_slug: the slug of the project the client keys
  46                                      belong to.
  47         :pparam string hook_id: the guid of the service hook.
  48         :param string url: the url for the webhook
  49         :param array[string] events: the events to subscribe to
  50         :auth: required
  51         """
  52         if not request.user.is_authenticated():
  53             return self.respond(status=401)
  54 
  55         try:
  56             hook = ServiceHook.objects.get(project_id=project.id, guid=hook_id)
  57         except ServiceHook.DoesNotExist:
  58             raise ResourceDoesNotExist
  59 
  60         validator = ServiceHookValidator(data=request.data, partial=True)
  61         if not validator.is_valid():
  62             return self.respond(validator.errors, status=status.HTTP_400_BAD_REQUEST)
  63 
  64         result = validator.validated_data
  65 
  66         updates = {}
  67         if result.get("events") is not None:
  68             updates["events"] = result["events"]
  69         if result.get("url"):
  70             updates["url"] = result["url"]
  71         if result.get("version") is not None:
  72             updates["version"] = result["version"]
  73         if result.get("isActive") is True:
  74             updates["status"] = ObjectStatus.ACTIVE
  75         elif result.get("isActive") is False:
  76             updates["status"] = ObjectStatus.DISABLED
  77 
  78         with transaction.atomic():
  79             hook.update(**updates)
  80 
  81             self.create_audit_entry(
  82                 request=request,
  83                 organization=project.organization,
  84                 target_object=hook.id,
  85                 event=AuditLogEntryEvent.SERVICEHOOK_EDIT,
  86                 data=hook.get_audit_log_data(),
  87             )
  88 
  89         return self.respond(serialize(hook, request.user))

  96         :pparam string organization_slug: the slug of the organization the

  98         :pparam string project_slug: the slug of the project the client keys
  99                                      belong to.
 100         :pparam string hook_id: the guid of the service hook.
 101         :auth: required
 102         """
 103         if not request.user.is_authenticated():
 104             return self.respond(status=401)
 105 
 106         try:
 107             hook = ServiceHook.objects.get(project_id=project.id, guid=hook_id)
