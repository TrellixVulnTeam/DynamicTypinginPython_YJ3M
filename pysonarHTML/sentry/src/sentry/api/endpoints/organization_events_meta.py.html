  21 class OrganizationEventsMetaEndpoint(OrganizationEventsEndpointBase):
  22     def get(self, request, organization):
  23         with sentry_sdk.start_span(op="discover.endpoint", description="filter_params") as span:
  24             span.set_data("organization", organization)
  25             try:
  26                 params = self.get_filter_params(request, organization)
  27             except NoProjects:
  28                 return Response({"count": 0})
  29             params = self.quantize_date_params(request, params)
  30 
  31         try:
  32             result = discover.query(
  33                 selected_columns=["count()"],
  34                 params=params,
  35                 query=request.query_params.get("query"),
  36                 referrer="api.organization-events-meta",
  37             )
  38         except (discover.InvalidSearchQuery, snuba.QueryOutsideRetentionError) as error:
  39             raise ParseError(detail=six.text_type(error))
  40 
  41         return Response({"count": result["data"][0]["count"]})

  47 class OrganizationEventsRelatedIssuesEndpoint(OrganizationEventsEndpointBase, EnvironmentMixin):
  48     def get(self, request, organization):
  49         with sentry_sdk.start_span(op="discover.endpoint", description="filter_params") as span:
  50             span.set_data("organization", organization)
  51             try:
  52                 params = self.get_filter_params(request, organization)
  53             except NoProjects:
  54                 return Response([])
  55 
  56             possible_keys = ["transaction"]
  57             lookup_keys = {key: request.query_params.get(key) for key in possible_keys}
  58 
  59             if not any(lookup_keys.values()):
  60                 return Response(
  61                     {
  62                         "detail": "Must provide one of {} in order to find related events".format(
  63                             possible_keys
  64                         )
  65                     },
  66                     status=400,
  67                 )
  68 
  69         try:
  70             with sentry_sdk.start_span(op="discover.endpoint", description="filter_creation"):
  71                 projects = self.get_projects(request, organization)
  72                 query_kwargs = build_query_params_from_request(
  73                     request, organization, projects, params.get("environment")
  74                 )
  75                 query_kwargs["limit"] = 5
  76                 try:
  77                     # Need to escape quotes in case some "joker" has a transaction with quotes
  78                     transaction_name = UNESCAPED_QUOTE_RE.sub('\\"', lookup_keys["transaction"])
  79                     parsed_terms = parse_search_query('transaction:"{}"'.format(transaction_name))
  80                 except ParseError:
  81                     return Response({"detail": "Invalid transaction search"}, status=400)
  82 
  83                 if query_kwargs.get("search_filters"):
  84                     query_kwargs["search_filters"].extend(parsed_terms)
  85                 else:
  86                     query_kwargs["search_filters"] = parsed_terms
  87 
  88             with sentry_sdk.start_span(op="discover.endpoint", description="issue_search"):
  89                 results = search.query(**query_kwargs)
  90         except discover.InvalidSearchQuery as err:
  91             raise ParseError(detail=six.text_type(err))
  92 
  93         with sentry_sdk.start_span(op="discover.endpoint", description="serialize_results") as span:
  94             results = list(results)
  95             span.set_data("result_length", len(results))
  96             context = serialize(
  97                 results,
  98                 request.user,
  99                 GroupSerializer(
 100                     environment_func=self._get_environment_func(request, organization.id)
 101                 ),
 102             )
 103 
 104         return Response(context)
