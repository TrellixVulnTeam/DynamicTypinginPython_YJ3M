  20 class ProjectEventsEndpoint(ProjectEndpoint):
  21     doc_section = DocSection.EVENTS
  22 
  23     @attach_scenarios([list_project_available_samples_scenario])
  24     def get(self, request, project):
  25         """
  26         List a Project's Events
  27         ```````````````````````
  28 
  29         Return a list of events bound to a project.
  30 
  31         Note: This endpoint is experimental and may be removed without notice.
  32 
  33         :qparam bool full: if this is set to true then the event payload will
  34                            include the full event body, including the stacktrace.
  35                            Set to 1 to enable.
  36 
  37         :pparam string organization_slug: the slug of the organization the
  38                                           groups belong to.
  39         :pparam string project_slug: the slug of the project the groups
  40                                      belong to.
  41         """
  42         from sentry.api.paginator import GenericOffsetPaginator
  43 
  44         query = request.GET.get("query")
  45         conditions = []
  46         if query:
  47             conditions.append(
  48                 [["positionCaseInsensitive", ["message", "'%s'" % (query,)]], "!=", 0]
  49             )
  50 
  51         full = request.GET.get("full", False)
  52 
  53         data_fn = partial(
  54             eventstore.get_events,
  55             filter=eventstore.Filter(conditions=conditions, project_ids=[project.id]),
  56             referrer="api.project-events",
  57         )
  58 
  59         serializer = EventSerializer() if full else SimpleEventSerializer()
  60         return self.paginate(
  61             request=request,
  62             on_results=lambda results: serialize(results, request.user, serializer),
  63             paginator=GenericOffsetPaginator(data_fn=data_fn),
  64         )
