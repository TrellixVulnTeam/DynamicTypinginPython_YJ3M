  30 def add_email(email, user):
  31     """
  32     Adds an email to user account
  33 
  34     Can be either primary or secondary
  35     """
  36 
  37     # Bad email
  38     if email is None:
  39         raise InvalidEmailError
  40 
  41     try:
  42         with transaction.atomic():
  43             new_email = UserEmail.objects.create(user=user, email=email)
  44     except IntegrityError:
  45         raise DuplicateEmailError
  46 
  47     new_email.set_hash()
  48     new_email.save()
  49     user.send_confirm_email_singular(new_email)
  50     return new_email

  68     @sudo_required
  69     def post(self, request, user):
  70         """
  71         Adds a secondary email address
  72         ``````````````````````````````
  73 
  74         Adds a secondary email address to account.
  75 
  76         :param string email: email to add
  77         :auth required:
  78         """
  79 
  80         validator = EmailValidator(data=request.data)
  81         if not validator.is_valid():
  82             return self.respond(validator.errors, status=400)
  83 
  84         result = validator.validated_data
  85         email = result["email"].lower().strip()
  86 
  87         try:
  88             new_useremail = add_email(email, user)
  89         except DuplicateEmailError:
  90             new_useremail = user.emails.get(email__iexact=email)
  91             return self.respond(serialize(new_useremail, user=request.user), status=200)
  92         else:
  93             logger.info(
  94                 "user.email.add",
  95                 extra={
  96                     "user_id": user.id,
  97                     "ip_address": request.META["REMOTE_ADDR"],
  98                     "email": new_useremail.email,
  99                 },
 100             )
 101             return self.respond(serialize(new_useremail, user=request.user), status=201)

 103     @sudo_required
 104     def put(self, request, user):
 105         """
 106         Updates primary email
 107         `````````````````````
 108 
 109         Changes primary email
 110 
 111         :param string email: the email to set as primary email
 112         :auth required:
 113         """
 114 
 115         validator = EmailValidator(data=request.data)
 116         if not validator.is_valid():
 117             return self.respond(validator.errors, status=400)
 118 
 119         result = validator.validated_data
 120         old_email = user.email.lower()
 121         new_email = result["email"].lower()
 122 
 123         new_useremail = user.emails.filter(email__iexact=new_email).first()
 124 
 125         # If email doesn't exist for user, attempt to add new email
 126         if not new_useremail:
 127             try:
 128                 new_useremail = add_email(new_email, user)
 129             except DuplicateEmailError:
 130                 new_useremail = user.emails.get(email__iexact=new_email)
 131             else:
 132                 logger.info(
 133                     "user.email.add",
 134                     extra={
 135                         "user_id": user.id,
 136                         "ip_address": request.META["REMOTE_ADDR"],
 137                         "email": new_useremail.email,
 138                     },
 139                 )
 140                 new_email = new_useremail.email
 141 
 142         # Check if email is in use
 143         # TODO(dcramer): this needs rate limiting to avoid abuse
 144         # TODO(dcramer): this needs a lock/constraint
 145         if (
 146             User.objects.filter(Q(email__iexact=new_email) | Q(username__iexact=new_email))
 147             .exclude(id=user.id)
 148             .exists()
 149         ):
 150             return self.respond(
 151                 {"email": "That email address is already associated with another account."},
 152                 status=400,
 153             )
 154 
 155         if not new_useremail.is_verified:
 156             return self.respond(
 157                 {"email": "You must verify your email address before marking it as primary."},
 158                 status=400,
 159             )
 160 
 161         options = UserOption.objects.filter(user=user, key="mail:email")
 162         for option in options:
 163             if option.value != old_email:
 164                 continue
 165             option.update(value=new_email)
 166 
 167         has_new_username = old_email == user.username
 168 
 169         update_kwargs = {"email": new_email}
 170 
 171         if has_new_username and not User.objects.filter(username__iexact=new_email).exists():
 172             update_kwargs["username"] = new_email
 173 
 174         # NOTE(mattrobenolt): When changing your primary email address,
 175         # we explicitly want to invalidate existing lost password hashes,
 176         # so that in the event of a compromised inbox, an outstanding
 177         # password hash can't be used to gain access. We also feel this
 178         # is a large enough of a security concern to force logging
 179         # out other current sessions.
 180         user.clear_lost_passwords()
 181         user.refresh_session_nonce(request._request)
 182         update_kwargs["session_nonce"] = user.session_nonce
 183 
 184         user.update(**update_kwargs)
 185 
 186         logger.info(
 187             "user.email.edit",
 188             extra={
 189                 "user_id": user.id,
 190                 "ip_address": request.META["REMOTE_ADDR"],
 191                 "email": new_email,
 192             },
 193         )
 194 
 195         return self.respond(serialize(new_useremail, user=request.user))

 197     @sudo_required
 198     def delete(self, request, user):
 199         """
 200         Removes an email from account
 201         `````````````````````````````
 202 
 203         Removes an email from account, can not remove primary email
 204 
 205         :param string email: email to remove
 206         :auth required:
 207         """
 208         validator = EmailValidator(data=request.data)
 209         if not validator.is_valid():
 210             return self.respond(validator.errors, status=400)
 211 
 212         email = validator.validated_data["email"]
 213         primary_email = UserEmail.get_primary_email(user)
 214         del_email = UserEmail.objects.filter(user=user, email__iexact=email).first()
 215 
 216         # Don't allow deleting primary email?
 217         if primary_email == del_email:
 218             return self.respond({"detail": "Cannot remove primary email"}, status=400)
 219 
 220         del_email.delete()
 221 
 222         logger.info(
 223             "user.email.remove",
 224             extra={"user_id": user.id, "ip_address": request.META["REMOTE_ADDR"], "email": email},
 225         )
 226 
 227         return self.respond(status=204)
