  22 class DataScrubbingSelectorSuggestionsEndpoint(OrganizationEndpoint):
  23     doc_section = DocSection.ORGANIZATIONS
  24 
  25     @attach_scenarios([get_selector_suggestions_for_organization_scenario])
  26     def get(self, request, organization):
  27         """
  28         Generate a list of data scrubbing selectors from existing event data.
  29 
  30         This list is used to auto-complete settings in "Data Scrubbing" /
  31         "Security and Privacy" settings.
  32         """
  33 
  34         event_id = request.GET.get("eventId", None)
  35 
  36         # Filtering by the projects that self.get_projects returns deals with
  37         # permission concerns
  38         projects = self.get_projects(request, organization)
  39         project_ids = [project.id for project in projects]
  40 
  41         suggestions = {}
  42 
  43         if event_id:
  44             for event in eventstore.get_events(
  45                 filter=eventstore.Filter(
  46                     conditions=[["id", "=", event_id]], project_ids=project_ids
  47                 )
  48             ):
  49                 for selector in pii_selector_suggestions_from_event(dict(event.data)):
  50                     examples_ = suggestions.setdefault(selector["path"], [])
  51                     if selector["value"]:
  52                         examples_.append(selector["value"])
  53 
  54         return Response(
  55             {
  56                 "suggestions": [
  57                     {"type": "value", "value": value, "examples": examples}
  58                     for value, examples in six.iteritems(suggestions)
  59                 ]
  60             }
  61         )
