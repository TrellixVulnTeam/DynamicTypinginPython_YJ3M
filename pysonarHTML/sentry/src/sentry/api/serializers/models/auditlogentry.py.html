  10 def fix(data):
  11     # XXX(dcramer): At some point Organization.flags was serialized as a BitHandler
  12     if "flags" in data and isinstance(data["flags"], BitHandler):
  13         data["flags"] = int(data["flags"])
  14 
  15     # There was a point in time where full Team objects
  16     # got serialized into our AuditLogEntry.data, so these
  17     # values need to be stripped and reduced down to integers
  18     if data.get("teams"):
  19         if hasattr(data["teams"][0], "id"):
  20             data["teams"] = [t.id for t in data["teams"]]
  21 
  22     return data
