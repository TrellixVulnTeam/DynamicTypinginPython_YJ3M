  13 @register(Incident)
  14 class IncidentSerializer(Serializer):
  15     def get_attrs(self, item_list, user, **kwargs):
  16         attach_foreignkey(item_list, Incident.alert_rule, related=("snuba_query",))
  17         incident_projects = defaultdict(list)
  18         for incident_project in IncidentProject.objects.filter(
  19             incident__in=item_list
  20         ).select_related("project"):
  21             incident_projects[incident_project.incident_id].append(incident_project.project.slug)
  22 
  23         alert_rules = {
  24             d["id"]: d
  25             for d in serialize(set(i.alert_rule for i in item_list if i.alert_rule.id), user)
  26         }
  27 
  28         results = {}
  29         for incident in item_list:
  30             results[incident] = {"projects": incident_projects.get(incident.id, [])}
  31             results[incident]["alert_rule"] = alert_rules.get(six.text_type(incident.alert_rule.id))
  32 
  33         return results
  34 
  35     def serialize(self, obj, attrs, user):
  36         return {
  37             "id": six.text_type(obj.id),
  38             "identifier": six.text_type(obj.identifier),
  39             "organizationId": six.text_type(obj.organization_id),
  40             "projects": attrs["projects"],
  41             "alertRule": attrs["alert_rule"],
  42             "status": obj.status,
  43             "statusMethod": obj.status_method,
  44             "type": obj.type,
  45             "title": obj.title,
  46             "dateStarted": obj.date_started,
  47             "dateDetected": obj.date_detected,
  48             "dateCreated": obj.date_added,
  49             "dateClosed": obj.date_closed,
  50         }

  53 class DetailedIncidentSerializer(IncidentSerializer):
  54     def get_attrs(self, item_list, user, **kwargs):
  55         results = super(DetailedIncidentSerializer, self).get_attrs(item_list, user=user, **kwargs)
  56         subscribed_incidents = set()
  57         if user.is_authenticated():
  58             subscribed_incidents = set(
  59                 IncidentSubscription.objects.filter(incident__in=item_list, user=user).values_list(
  60                     "incident_id", flat=True
  61                 )
  62             )
  63 
  64         for item in item_list:
  65             results[item]["is_subscribed"] = item.id in subscribed_incidents
  66         return results
  67 
  68     def _get_incident_seen_list(self, incident, user):
  69         incident_seen = list(
  70             IncidentSeen.objects.filter(incident=incident)
  71             .select_related("user")
  72             .order_by("-last_seen")
  73         )
  74 
  75         seen_by_list = []
  76         has_seen = False
  77 
  78         for seen_by in incident_seen:
  79             if seen_by.user == user:
  80                 has_seen = True
  81             seen_by_list.append(serialize(seen_by))
  82 
  83         return {"seen_by": seen_by_list, "has_seen": has_seen}
  84 
  85     def serialize(self, obj, attrs, user):
  86         context = super(DetailedIncidentSerializer, self).serialize(obj, attrs, user)
  87         seen_list = self._get_incident_seen_list(obj, user)
  88 
  89         context["isSubscribed"] = attrs["is_subscribed"]
  90         context["seenBy"] = seen_list["seen_by"]
  91         context["hasSeen"] = seen_list["has_seen"]
  92         # The query we should use to get accurate results in Discover.
  93         context["discoverQuery"] = self._build_discover_query(obj)
  94 
  95         return context
  96 
  97     def _build_discover_query(self, incident):
  98         query = incident.alert_rule.snuba_query.query
  99         dataset = QueryDatasets(incident.alert_rule.snuba_query.dataset)
 100         condition = None
 101 
 102         if dataset == QueryDatasets.EVENTS:
 103             condition = "event.type:error"
 104         elif dataset == QueryDatasets.TRANSACTIONS:
 105             condition = "event.type:transaction"
 106 
 107         if condition:
 108             query = "{} {}".format(condition, query) if query else condition
 109 
 110         return query
