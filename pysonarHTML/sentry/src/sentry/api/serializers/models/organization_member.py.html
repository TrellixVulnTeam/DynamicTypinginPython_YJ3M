  43 class OrganizationMemberWithTeamsSerializer(OrganizationMemberSerializer):
  44     def get_attrs(self, item_list, user):
  45         attrs = super(OrganizationMemberWithTeamsSerializer, self).get_attrs(item_list, user)
  46 
  47         member_team_map = list(
  48             OrganizationMemberTeam.objects.filter(
  49                 team__status=TeamStatus.VISIBLE, organizationmember__in=item_list
  50             ).values_list("organizationmember_id", "team_id")
  51         )
  52 
  53         teams = {
  54             team.id: team
  55             for team in Team.objects.filter(id__in=[team_id for _, team_id in member_team_map])
  56         }
  57         results = defaultdict(list)
  58 
  59         # results is a map of member id -> team_slug[]
  60         for member_id, team_id in member_team_map:
  61             results[member_id].append(teams[team_id].slug)
  62 
  63         for item in item_list:
  64             teams = results.get(item.id, [])
  65             try:
  66                 attrs[item]["teams"] = teams
  67             except KeyError:
  68                 attrs[item] = {"teams": teams}
  69 
  70         return attrs
  71 
  72     def serialize(self, obj, attrs, user):
  73         d = super(OrganizationMemberWithTeamsSerializer, self).serialize(obj, attrs, user)
  74 
  75         d["teams"] = attrs.get("teams", [])
  76 
  77         return d

  80 class OrganizationMemberWithProjectsSerializer(OrganizationMemberSerializer):
  81     def __init__(self, *args, **kwargs):
  82         self.project_ids = set(kwargs.pop("project_ids", []))
  83         super(OrganizationMemberWithProjectsSerializer, self).__init__(*args, **kwargs)
  84 
  85     def get_attrs(self, item_list, user):
  86         attrs = super(OrganizationMemberWithProjectsSerializer, self).get_attrs(item_list, user)
  87         # Note: For this to be efficient, call
  88         # `.prefetch_related(
  89         #       'teams',
  90         #       'teams__projectteam_set',
  91         #       'teams__projectteam_set__project',
  92         # )` on your queryset before using this serializer
  93         for org_member in item_list:
  94             projects = set()
  95             for team in org_member.teams.all():
  96                 # Filter in python here so that we don't break the prefetch
  97                 if team.status != TeamStatus.VISIBLE:
  98                     continue
  99 
 100                 for project_team in team.projectteam_set.all():
 101                     if (
 102                         project_team.project_id in self.project_ids
 103                         and project_team.project.status == TeamStatus.VISIBLE
 104                     ):
 105                         projects.add(project_team.project.slug)
 106 
 107             projects = list(projects)
 108             projects.sort()
 109             attrs[org_member]["projects"] = projects
 110 
 111         return attrs
 112 
 113     def serialize(self, obj, attrs, user):
 114         d = super(OrganizationMemberWithProjectsSerializer, self).serialize(obj, attrs, user)
 115         d["projects"] = attrs.get("projects", [])
 116         return d
