  27 def expose_version_info(info):
  28     if info is None:
  29         return None
  30     version = {"raw": info["version_raw"]}
  31     if info["version_parsed"]:
  32         version.update(
  33             {
  34                 "major": info["version_parsed"]["major"],
  35                 "minor": info["version_parsed"]["minor"],
  36                 "patch": info["version_parsed"]["patch"],
  37                 "pre": info["version_parsed"]["pre"],
  38                 "buildCode": info["version_parsed"]["build_code"],
  39                 "components": info["version_parsed"]["components"],
  40             }
  41         )
  42     return {
  43         "package": info["package"],
  44         "version": version,
  45         "description": info["description"],
  46         "buildHash": info["build_hash"],
  47     }

  50 def get_users_for_authors(organization_id, authors, user=None):
  51     """
  52     Returns a dictionary of author_id => user, if a Sentry
  53     user object exists for that email. If there is no matching
  54     Sentry user, a {user, email} dict representation of that
  55     author is returned.
  56     e.g.
  57     {
  58         1: serialized(<User id=1>),
  59         2: {email: 'not-a-user@example.com', name: 'dunno'},
  60         ...
  61     }
  62     """
  63     # Filter users based on the emails provided in the commits
  64     user_emails = list(
  65         UserEmail.objects.filter(in_iexact("email", [a.email for a in authors])).order_by("id")
  66     )
  67 
  68     # Filter users belonging to the organization associated with
  69     # the release
  70     users = User.objects.filter(
  71         id__in={ue.user_id for ue in user_emails},
  72         is_active=True,
  73         sentry_orgmember_set__organization_id=organization_id,
  74     )
  75     users = serialize(list(users), user)
  76     users_by_id = {user["id"]: user for user in users}
  77 
  78     # Figure out which email address matches to a user
  79     users_by_email = {}
  80     for email in user_emails:
  81         if email.email not in users_by_email:
  82             user = users_by_id.get(six.text_type(email.user_id), None)
  83             # user can be None if there's a user associated
  84             # with user_email in separate organization
  85             if user:
  86                 users_by_email[email.email] = user
  87 
  88     results = {}
  89     for author in authors:
  90         results[six.text_type(author.id)] = users_by_email.get(
  91             author.email, {"name": author.name, "email": author.email}
  92         )
  93 
  94     return results

  99     def _get_commit_metadata(self, item_list, user):
 100         """
 101         Returns a dictionary of release_id => commit metadata,
 102         where each commit metadata dict contains commit_count
 103         and an array of authors.
 104         e.g.
 105         {
 106             1: {
 107                 'latest_commit': <Commit id=1>,
 108                 'authors': [<User id=1>, <User id=2>]
 109             },
 110             ...
 111         }
 112         """
 113         author_ids = set()
 114         for obj in item_list:
 115             author_ids.update(obj.authors)
 116 
 117         if author_ids:
 118             authors = list(CommitAuthor.objects.filter(id__in=author_ids))
 119         else:
 120             authors = []
 121 
 122         if authors:
 123             org_ids = set(item.organization_id for item in item_list)
 124             if len(org_ids) != 1:
 125                 users_by_author = {}
 126             else:
 127                 users_by_author = get_users_for_authors(
 128                     organization_id=org_ids.pop(), authors=authors, user=user
 129                 )
 130         else:
 131             users_by_author = {}
 132 
 133         commit_ids = set((o.last_commit_id for o in item_list if o.last_commit_id))
 134         if commit_ids:
 135             commit_list = list(Commit.objects.filter(id__in=commit_ids).select_related("author"))
 136             commits = {c.id: d for c, d in zip(commit_list, serialize(commit_list, user))}
 137         else:
 138             commits = {}
 139 
 140         result = {}
 141         for item in item_list:
 142             item_authors = []
 143             seen_authors = set()
 144             for user in (users_by_author.get(a) for a in item.authors):
 145                 if user and user["email"] not in seen_authors:
 146                     seen_authors.add(user["email"])
 147                     item_authors.append(user)
 148 
 149             result[item] = {
 150                 "authors": item_authors,
 151                 "last_commit": commits.get(item.last_commit_id),
 152             }
 153         return result

 155     def _get_deploy_metadata(self, item_list, user):
 156         """
 157         Returns a dictionary of release_id => commit metadata,
 158         where each commit metadata dict contains commit_count
 159         and an array of authors.
 160         e.g.
 161         {
 162             1: {
 163                 'latest_commit': <Commit id=1>,
 164                 'authors': [<User id=1>, <User id=2>]
 165             },
 166             ...
 167         }
 168         """
 169         deploy_ids = set((o.last_deploy_id for o in item_list if o.last_deploy_id))
 170         if deploy_ids:
 171             deploy_list = list(Deploy.objects.filter(id__in=deploy_ids))
 172             deploys = {d.id: c for d, c in zip(deploy_list, serialize(deploy_list, user))}
 173         else:
 174             deploys = {}
 175 
 176         result = {}
 177         for item in item_list:
 178             result[item] = {"last_deploy": deploys.get(item.last_deploy_id)}
 179         return result

 181     def __get_project_id_list(self, item_list):
 182         project_ids = set()
 183         need_fallback = False
 184 
 185         for release in item_list:
 186             if release._for_project_id is not None:
 187                 project_ids.add(release._for_project_id)
 188             else:
 189                 need_fallback = True
 190 
 191         if not need_fallback:
 192             return sorted(project_ids), True
 193 
 194         return (
 195             list(
 196                 ReleaseProject.objects.filter(release__in=item_list)
 197                 .values_list("project_id", flat=True)
 198                 .distinct()
 199             ),
 200             False,
 201         )

 203     def __get_release_data_no_environment(self, project, item_list):
 204         if project is not None:
 205             project_ids = [project.id]
 206             specialized = True
 207         else:
 208             project_ids, specialized = self.__get_project_id_list(item_list)
 209 
 210         first_seen = {}
 211         last_seen = {}
 212         tag_values = tagstore.get_release_tags(
 213             project_ids, environment_id=None, versions=[o.version for o in item_list]
 214         )
 215         for tv in tag_values:
 216             first_val = first_seen.get(tv.value)
 217             last_val = last_seen.get(tv.value)
 218             first_seen[tv.value] = min(tv.first_seen, first_val) if first_val else tv.first_seen
 219             last_seen[tv.value] = max(tv.last_seen, last_val) if last_val else tv.last_seen
 220 
 221         group_counts_by_release = {}
 222         if project is not None:
 223             for release_id, new_groups in ReleaseProject.objects.filter(
 224                 project=project, release__in=item_list
 225             ).values_list("release_id", "new_groups"):
 226                 group_counts_by_release[release_id] = {project.id: new_groups}
 227         else:
 228             for project_id, release_id, new_groups in ReleaseProject.objects.filter(
 229                 release__in=item_list, new_groups__isnull=False
 230             ).values_list("project_id", "release_id", "new_groups"):
 231                 group_counts_by_release.setdefault(release_id, {})[project_id] = new_groups
 232         return first_seen, last_seen, group_counts_by_release

 234     def __get_release_data_with_environment(self, project, item_list, environment):
 235         release_project_envs = ReleaseProjectEnvironment.objects.filter(
 236             release__in=item_list, environment=environment
 237         ).select_related("release")
 238         if project is not None:
 239             release_project_envs = release_project_envs.filter(project=project)
 240         first_seen = {}
 241         last_seen = {}
 242         for release_project_env in release_project_envs:
 243             first_seen[release_project_env.release.version] = release_project_env.first_seen
 244             last_seen[release_project_env.release.version] = release_project_env.last_seen
 245 
 246         group_counts_by_release = {}
 247         for project_id, release_id, new_groups in release_project_envs.annotate(
 248             aggregated_new_issues_count=Sum("new_issues_count")
 249         ).values_list("project_id", "release_id", "aggregated_new_issues_count"):
 250             group_counts_by_release.setdefault(release_id, {})[project_id] = new_groups
 251 
 252         return first_seen, last_seen, group_counts_by_release

 254     def get_attrs(self, item_list, user, **kwargs):
 255         project = kwargs.get("project")
 256         environment = kwargs.get("environment")
 257         with_health_data = kwargs.get("with_health_data", False)
 258         health_stat = kwargs.get("health_stat", None)
 259         health_stats_period = kwargs.get("health_stats_period")
 260         summary_stats_period = kwargs.get("summary_stats_period")
 261 
 262         if environment is None:
 263             first_seen, last_seen, issue_counts_by_release = self.__get_release_data_no_environment(
 264                 project, item_list
 265             )
 266         else:
 267             (
 268                 first_seen,
 269                 last_seen,
 270                 issue_counts_by_release,
 271             ) = self.__get_release_data_with_environment(project, item_list, environment)
 272 
 273         owners = {
 274             d["id"]: d for d in serialize(set(i.owner for i in item_list if i.owner_id), user)
 275         }
 276 
 277         release_metadata_attrs = self._get_commit_metadata(item_list, user)
 278         deploy_metadata_attrs = self._get_deploy_metadata(item_list, user)
 279 
 280         release_projects = defaultdict(list)
 281         project_releases = ReleaseProject.objects.filter(release__in=item_list).values(
 282             "new_groups",
 283             "release_id",
 284             "release__version",
 285             "project__slug",
 286             "project__name",
 287             "project__id",
 288             "project__platform",
 289         )
 290 
 291         platforms = ProjectPlatform.objects.filter(
 292             project_id__in=set(x["project__id"] for x in project_releases)
 293         ).values_list("project_id", "platform")
 294         platforms_by_project = defaultdict(list)
 295         for project_id, platform in platforms:
 296             platforms_by_project[project_id].append(platform)
 297 
 298         if with_health_data:
 299             health_data = get_release_health_data_overview(
 300                 [(pr["project__id"], pr["release__version"]) for pr in project_releases],
 301                 health_stats_period=health_stats_period,
 302                 summary_stats_period=summary_stats_period,
 303                 stat=health_stat,
 304             )
 305         else:
 306             health_data = None
 307 
 308         for pr in project_releases:
 309             pr_rv = {
 310                 "id": pr["project__id"],
 311                 "slug": pr["project__slug"],
 312                 "name": pr["project__name"],
 313                 "new_groups": pr["new_groups"],
 314                 "platform": pr["project__platform"],
 315                 "platforms": platforms_by_project.get(pr["project__id"]) or [],
 316             }
 317             if health_data is not None:
 318                 pr_rv["health_data"] = health_data.get((pr["project__id"], pr["release__version"]))
 319             release_projects[pr["release_id"]].append(pr_rv)
 320 
 321         result = {}
 322         for item in item_list:
 323             single_release_projects = release_projects.get(item.id, [])
 324 
 325             if item._for_project_id is not None:
 326                 single_release_projects = [
 327                     x for x in single_release_projects if x["id"] == item._for_project_id
 328                 ]
 329                 release_new_groups = (issue_counts_by_release.get(item.id) or {}).get(
 330                     item._for_project_id
 331                 ) or 0
 332             else:
 333                 release_new_groups = sum((issue_counts_by_release.get(item.id) or {}).values())
 334 
 335             p = {
 336                 "owner": owners[six.text_type(item.owner_id)] if item.owner_id else None,
 337                 "new_groups": release_new_groups,
 338                 "projects": single_release_projects,
 339                 "first_seen": first_seen.get(item.version),
 340                 "last_seen": last_seen.get(item.version),
 341             }
 342 
 343             p.update(release_metadata_attrs[item])
 344             p.update(deploy_metadata_attrs[item])
 345 
 346             result[item] = p
 347         return result

 357                 "crashFreeSessions": data["crash_free_sessions"],
 358                 "sessionsCrashed": data["sessions_crashed"],
 359                 "sessionsErrored": data["sessions_errored"],
 360                 "totalUsers": data["total_users"],
 361                 "totalUsers24h": data["total_users_24h"],
 362                 "totalSessions": data["total_sessions"],
 363                 "totalSessions24h": data["total_sessions_24h"],
 364                 "adoption": data["adoption"],
 365                 "stats": data.get("stats"),
 366                 "hasHealthData": data["has_health_data"],
 367             }
 368 
 369         def expose_project(project):
 370             rv = {
 371                 "id": project["id"],
 372                 "slug": project["slug"],
 373                 "name": project["name"],
 374                 "newGroups": project["new_groups"],
 375                 "platform": project["platform"],
 376                 "platforms": project["platforms"],
 377             }
 378             if "health_data" in project:
 379                 rv["healthData"] = expose_health_data(project["health_data"])
 380             return rv
 381 
 382         d = {
 383             "version": obj.version,
 384             "shortVersion": obj.version,
 385             "versionInfo": expose_version_info(obj.version_info),
 386             "ref": obj.ref,
 387             "url": obj.url,
 388             "dateReleased": obj.date_released,
 389             "dateCreated": obj.date_added,
 390             "data": obj.data,
 391             "newGroups": attrs["new_groups"],
 392             "owner": attrs["owner"],
 393             "commitCount": obj.commit_count,
 394             "lastCommit": attrs.get("last_commit"),
 395             "deployCount": obj.total_deploys,
 396             "lastDeploy": attrs.get("last_deploy"),
 397             "authors": attrs.get("authors", []),
 398             "projects": [expose_project(p) for p in attrs.get("projects", [])],
 399             "firstEvent": attrs.get("first_seen"),
 400             "lastEvent": attrs.get("last_seen"),
 401         }
 402         return d
