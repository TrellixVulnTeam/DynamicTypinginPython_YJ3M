  14             if action.target:

  16                     return "Send an email to " + action.target.email
  17                 elif action.target_type == action.TargetType.TEAM.value:
  18                     return "Send an email to members of #" + action.target.slug

  20             return "Send a PagerDuty notification to " + action.target_display
