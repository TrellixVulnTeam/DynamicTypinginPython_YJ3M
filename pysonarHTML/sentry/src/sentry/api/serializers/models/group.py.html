  86     def _get_subscriptions(self, item_list, user):
  87         """
  88         Returns a mapping of group IDs to a two-tuple of (subscribed: bool,
  89         subscription: GroupSubscription or None) for the provided user and
  90         groups.
  91         """
  92         if not item_list:
  93             return {}
  94 
  95         # Collect all of the projects to look up, and keep a set of groups that
  96         # are part of that project. (Note that the common -- but not only --
  97         # case here is that all groups are part of the same project.)
  98         projects = defaultdict(set)
  99         for group in item_list:
 100             projects[group.project].add(group)
 101 
 102         # Fetch the options for each project -- we'll need this to identify if
 103         # a user has totally disabled workflow notifications for a project.
 104         # NOTE: This doesn't use `values_list` because that bypasses field
 105         # value decoding, so the `value` field would not be unpickled.
 106         options = {
 107             option.project_id: option.value
 108             for option in UserOption.objects.filter(
 109                 Q(project__in=projects.keys()) | Q(project__isnull=True),
 110                 user=user,
 111                 key="workflow:notifications",
 112             )
 113         }
 114 
 115         # If there is a subscription record associated with the group, we can
 116         # just use that to know if a user is subscribed or not, as long as
 117         # notifications aren't disabled for the project.
 118         subscriptions = {
 119             subscription.group_id: subscription
 120             for subscription in GroupSubscription.objects.filter(
 121                 group__in=list(
 122                     itertools.chain.from_iterable(
 123                         map(
 124                             lambda project__groups: project__groups[1]
 125                             if not options.get(project__groups[0].id, options.get(None))
 126                             == UserOptionValue.no_conversations
 127                             else [],
 128                             projects.items(),
 129                         )
 130                     )
 131                 ),
 132                 user=user,
 133             )
 134         }
 135 
 136         # This is the user's default value for any projects that don't have
 137         # the option value specifically recorded. (The default
 138         # "participating_only" value is convention.)
 139         global_default_workflow_option = options.get(None, UserOptionValue.participating_only)
 140 
 141         results = {}
 142         for project, groups in projects.items():
 143             project_default_workflow_option = options.get(
 144                 project.id, global_default_workflow_option
 145             )
 146             for group in groups:
 147                 subscription = subscriptions.get(group.id)
 148                 if subscription is not None:
 149                     results[group.id] = (subscription.is_active, subscription)
 150                 else:
 151                     results[group.id] = (
 152                         (project_default_workflow_option == UserOptionValue.all_conversations, None)
 153                         if project_default_workflow_option != UserOptionValue.no_conversations
 154                         else disabled
 155                     )
 156 
 157         return results

 159     def get_attrs(self, item_list, user):
 160         from sentry.plugins.base import plugins
 161         from sentry.integrations import IntegrationFeatures
 162         from sentry.models import PlatformExternalIssue
 163 
 164         GroupMeta.objects.populate_cache(item_list)
 165 
 166         attach_foreignkey(item_list, Group.project)
 167 
 168         if user.is_authenticated() and item_list:
 169             bookmarks = set(
 170                 GroupBookmark.objects.filter(user=user, group__in=item_list).values_list(
 171                     "group_id", flat=True
 172                 )
 173             )
 174             seen_groups = dict(
 175                 GroupSeen.objects.filter(user=user, group__in=item_list).values_list(
 176                     "group_id", "last_seen"
 177                 )
 178             )
 179             subscriptions = self._get_subscriptions(item_list, user)
 180         else:
 181             bookmarks = set()
 182             seen_groups = {}
 183             subscriptions = defaultdict(lambda: (False, None))
 184 
 185         assignees = {
 186             a.group_id: a.assigned_actor()
 187             for a in GroupAssignee.objects.filter(group__in=item_list)
 188         }
 189         resolved_assignees = Actor.resolve_dict(assignees)
 190 
 191         ignore_items = {g.group_id: g for g in GroupSnooze.objects.filter(group__in=item_list)}
 192 
 193         resolved_item_list = [i for i in item_list if i.status == GroupStatus.RESOLVED]
 194         if resolved_item_list:
 195             release_resolutions = {
 196                 i[0]: i[1:]
 197                 for i in GroupResolution.objects.filter(group__in=resolved_item_list).values_list(
 198                     "group", "type", "release__version", "actor_id"
 199                 )
 200             }
 201 
 202             # due to our laziness, and django's inability to do a reasonable join here
 203             # we end up with two queries
 204             commit_results = list(
 205                 Commit.objects.extra(
 206                     select={"group_id": "sentry_grouplink.group_id"},
 207                     tables=["sentry_grouplink"],
 208                     where=[
 209                         "sentry_grouplink.linked_id = sentry_commit.id",
 210                         "sentry_grouplink.group_id IN ({})".format(
 211                             ", ".join(six.text_type(i.id) for i in resolved_item_list)
 212                         ),
 213                         "sentry_grouplink.linked_type = %s",
 214                         "sentry_grouplink.relationship = %s",
 215                     ],
 216                     params=[int(GroupLink.LinkedType.commit), int(GroupLink.Relationship.resolves)],
 217                 )
 218             )
 219             commit_resolutions = {
 220                 i.group_id: d for i, d in zip(commit_results, serialize(commit_results, user))
 221             }
 222         else:
 223             release_resolutions = {}
 224             commit_resolutions = {}
 225 
 226         actor_ids = set(r[-1] for r in six.itervalues(release_resolutions))
 227         actor_ids.update(r.actor_id for r in six.itervalues(ignore_items))
 228         if actor_ids:
 229             users = list(User.objects.filter(id__in=actor_ids, is_active=True))
 230             actors = {u.id: d for u, d in zip(users, serialize(users, user))}
 231         else:
 232             actors = {}
 233 
 234         share_ids = dict(
 235             GroupShare.objects.filter(group__in=item_list).values_list("group_id", "uuid")
 236         )
 237 
 238         result = {}
 239 
 240         seen_stats = self._get_seen_stats(item_list, user)
 241 
 242         annotations_by_group_id = defaultdict(list)
 243 
 244         organization_id_list = list(set(item.project.organization_id for item in item_list))
 245         # if no groups, then we can't proceed but this seems to be a valid use case
 246         if not item_list:
 247             return {}
 248         if len(organization_id_list) > 1:
 249             # this should never happen but if it does we should know about it
 250             logger.warn(
 251                 u"Found multiple organizations for groups: %s, with orgs: %s"
 252                 % ([item.id for item in item_list], organization_id_list)
 253             )
 254 
 255         # should only have 1 org at this point
 256         organization_id = organization_id_list[0]
 257 
 258         # find all the integration installs that have issue tracking
 259         for integration in Integration.objects.filter(organizations=organization_id):
 260             if not (
 261                 integration.has_feature(IntegrationFeatures.ISSUE_BASIC)
 262                 or integration.has_feature(IntegrationFeatures.ISSUE_SYNC)
 263             ):
 264                 continue
 265 
 266             install = integration.get_installation(organization_id)
 267             local_annotations_by_group_id = (
 268                 safe_execute(
 269                     install.get_annotations_for_group_list,
 270                     group_list=item_list,
 271                     _with_transaction=False,
 272                 )
 273                 or {}
 274             )
 275             merge_list_dictionaries(annotations_by_group_id, local_annotations_by_group_id)
 276 
 277         # find the external issues for sentry apps and add them in
 278         local_annotations_by_group_id = (
 279             safe_execute(
 280                 PlatformExternalIssue.get_annotations_for_group_list,
 281                 group_list=item_list,
 282                 _with_transaction=False,
 283             )
 284             or {}
 285         )
 286         merge_list_dictionaries(annotations_by_group_id, local_annotations_by_group_id)
 287 
 288         for item in item_list:
 289             active_date = item.active_at or item.first_seen
 290 
 291             annotations = []
 292             annotations.extend(annotations_by_group_id[item.id])
 293 
 294             # add the annotations for plugins
 295             # note that the model GroupMeta where all the information is stored is already cached at the top of this function
 296             # so these for loops doesn't make a bunch of queries
 297             for plugin in plugins.for_project(project=item.project, version=1):
 298                 safe_execute(plugin.tags, None, item, annotations, _with_transaction=False)
 299             for plugin in plugins.for_project(project=item.project, version=2):
 300                 annotations.extend(
 301                     safe_execute(plugin.get_annotations, group=item, _with_transaction=False) or ()
 302                 )
 303 
 304             resolution_actor = None
 305             resolution_type = None
 306             resolution = release_resolutions.get(item.id)
 307             if resolution:
 308                 resolution_type = "release"
 309                 resolution_actor = actors.get(resolution[-1])
 310             if not resolution:
 311                 resolution = commit_resolutions.get(item.id)
 312                 if resolution:
 313                     resolution_type = "commit"
 314 
 315             ignore_item = ignore_items.get(item.id)
 316             if ignore_item:
 317                 ignore_actor = actors.get(ignore_item.actor_id)
 318             else:
 319                 ignore_actor = None
 320 
 321             result[item] = {
 322                 "assigned_to": resolved_assignees.get(item.id),
 323                 "is_bookmarked": item.id in bookmarks,
 324                 "subscription": subscriptions[item.id],
 325                 "has_seen": seen_groups.get(item.id, active_date) > active_date,
 326                 "annotations": annotations,
 327                 "ignore_until": ignore_item,
 328                 "ignore_actor": ignore_actor,
 329                 "resolution": resolution,
 330                 "resolution_type": resolution_type,
 331                 "resolution_actor": resolution_actor,
 332                 "share_id": share_ids.get(item.id),
 333             }
 334 
 335             result[item].update(seen_stats.get(item, {}))
 336         return result

 338     def _get_status(self, attrs, obj):
 339         status = obj.status
 340         status_details = {}
 341         if attrs["ignore_until"]:
 342             snooze = attrs["ignore_until"]
 343             if snooze.is_valid(group=obj):
 344                 # counts return the delta remaining when window is not set
 345                 status_details.update(
 346                     {
 347                         "ignoreCount": (
 348                             snooze.count - (obj.times_seen - snooze.state["times_seen"])
 349                             if snooze.count and not snooze.window
 350                             else snooze.count
 351                         ),
 352                         "ignoreUntil": snooze.until,
 353                         "ignoreUserCount": (
 354                             snooze.user_count - (attrs["user_count"] - snooze.state["users_seen"])
 355                             if snooze.user_count and not snooze.user_window
 356                             else snooze.user_count
 357                         ),
 358                         "ignoreUserWindow": snooze.user_window,
 359                         "ignoreWindow": snooze.window,
 360                         "actor": attrs["ignore_actor"],
 361                     }
 362                 )
 363             else:
 364                 status = GroupStatus.UNRESOLVED
 365         if status == GroupStatus.UNRESOLVED and obj.is_over_resolve_age():
 366             status = GroupStatus.RESOLVED
 367             status_details["autoResolved"] = True
 368         if status == GroupStatus.RESOLVED:
 369             status_label = "resolved"
 370             if attrs["resolution_type"] == "release":
 371                 res_type, res_version, _ = attrs["resolution"]
 372                 if res_type in (GroupResolution.Type.in_next_release, None):
 373                     status_details["inNextRelease"] = True
 374                 elif res_type == GroupResolution.Type.in_release:
 375                     status_details["inRelease"] = res_version
 376                 status_details["actor"] = attrs["resolution_actor"]
 377             elif attrs["resolution_type"] == "commit":
 378                 status_details["inCommit"] = attrs["resolution"]
 379         elif status == GroupStatus.IGNORED:
 380             status_label = "ignored"
 381         elif status in [GroupStatus.PENDING_DELETION, GroupStatus.DELETION_IN_PROGRESS]:
 382             status_label = "pending_deletion"
 383         elif status == GroupStatus.PENDING_MERGE:
 384             status_label = "pending_merge"
 385         else:
 386             status_label = "unresolved"
 387         return status_details, status_label

 389     def _get_permalink(self, obj, user):
 390         # If user is not logged in and member of the organization,
 391         # do not return the permalink which contains private information i.e. org name.
 392         request = env.request
 393         is_superuser = request and is_active_superuser(request) and request.user == user
 394 
 395         # If user is a sentry_app then it's a proxy user meaning we can't do a org lookup via `get_orgs()`
 396         # because the user isn't an org member. Instead we can use the auth token and the installation
 397         # it's associated with to find out what organization the token has access to.
 398         is_valid_sentryapp = False
 399         if (
 400             request
 401             and getattr(request.user, "is_sentry_app", False)
 402             and isinstance(request.auth, ApiToken)
 403         ):
 404             is_valid_sentryapp = SentryAppInstallationToken.has_organization_access(
 405                 request.auth, obj.organization
 406             )
 407 
 408         if (
 409             is_superuser
 410             or is_valid_sentryapp
 411             or (user.is_authenticated() and user.get_orgs().filter(id=obj.organization.id).exists())
 412         ):
 413             with sentry_sdk.start_span(op="GroupSerializerBase.serialize.permalink.build"):
 414                 return obj.get_absolute_url()
 415         else:
 416             return None

 418     def _get_subscription(self, attrs):
 419         subscription_details = None
 420         if attrs["subscription"] is not disabled:
 421             is_subscribed, subscription = attrs["subscription"]
 422             if subscription is not None and subscription.is_active:
 423                 subscription_details = {
 424                     "reason": SUBSCRIPTION_REASON_MAP.get(subscription.reason, "unknown")
 425                 }
 426         else:
 427             is_subscribed = False
 428             subscription_details = {"disabled": True}
 429         return is_subscribed, subscription_details

 477     def _get_seen_stats(self, item_list, user):
 478         try:
 479             environment = self.environment_func()
 480         except Environment.DoesNotExist:
 481             user_counts = {}
 482             first_seen = {}
 483             last_seen = {}
 484             times_seen = {}
 485         else:
 486             project_id = item_list[0].project_id
 487             item_ids = [g.id for g in item_list]
 488             user_counts = tagstore.get_groups_user_counts(
 489                 [project_id], item_ids, environment_ids=environment and [environment.id]
 490             )
 491             first_seen = {}
 492             last_seen = {}
 493             times_seen = {}
 494             if environment is not None:
 495                 environment_tagvalues = tagstore.get_group_list_tag_value(
 496                     [project_id], item_ids, [environment.id], "environment", environment.name
 497                 )
 498                 for item_id, value in environment_tagvalues.items():
 499                     first_seen[item_id] = value.first_seen
 500                     last_seen[item_id] = value.last_seen
 501                     times_seen[item_id] = value.times_seen
 502             else:
 503                 for item in item_list:
 504                     first_seen[item.id] = item.first_seen
 505                     last_seen[item.id] = item.last_seen
 506                     times_seen[item.id] = item.times_seen
 507 
 508         attrs = {}
 509         for item in item_list:
 510             attrs[item] = {
 511                 "times_seen": times_seen.get(item.id, 0),
 512                 "first_seen": first_seen.get(item.id),  # TODO: missing?
 513                 "last_seen": last_seen.get(item.id),
 514                 "user_count": user_counts.get(item.id, 0),
 515             }
 516 
 517         return attrs

 520 class GroupStatsMixin(object):
 521     STATS_PERIOD_CHOICES = {
 522         "14d": StatsPeriod(14, timedelta(hours=24)),
 523         "24h": StatsPeriod(24, timedelta(hours=1)),
 524     }

 527         raise NotImplementedError

 529     def get_stats(self, item_list, user):
 530         if self.stats_period:
 531             # we need to compute stats at 1d (1h resolution), and 14d
 532             group_ids = [g.id for g in item_list]
 533 
 534             segments, interval = self.STATS_PERIOD_CHOICES[self.stats_period]
 535             now = timezone.now()
 536             query_params = {
 537                 "start": now - ((segments - 1) * interval),
 538                 "end": now,
 539                 "rollup": int(interval.total_seconds()),
 540             }
 541 
 542             return self.query_tsdb(group_ids, query_params)

 545 class StreamGroupSerializer(GroupSerializer, GroupStatsMixin):
 546     def __init__(
 547         self,
 548         environment_func=None,
 549         stats_period=None,
 550         matching_event_id=None,
 551         matching_event_environment=None,
 552     ):
 553         super(StreamGroupSerializer, self).__init__(environment_func)
 554 
 555         if stats_period is not None:
 556             assert stats_period in self.STATS_PERIOD_CHOICES
 557 
 558         self.stats_period = stats_period
 559         self.matching_event_id = matching_event_id
 560         self.matching_event_environment = matching_event_environment
 561 
 562     def query_tsdb(self, group_ids, query_params):
 563         try:
 564             environment = self.environment_func()
 565         except Environment.DoesNotExist:
 566             stats = {key: tsdb.make_series(0, **query_params) for key in group_ids}
 567         else:
 568             stats = tsdb.get_range(
 569                 model=tsdb.models.group,
 570                 keys=group_ids,
 571                 environment_ids=environment and [environment.id],
 572                 **query_params
 573             )
 574 
 575         return stats
 576 
 577     def get_attrs(self, item_list, user):
 578         attrs = super(StreamGroupSerializer, self).get_attrs(item_list, user)
 579 
 580         if self.stats_period:
 581             stats = self.get_stats(item_list, user)
 582             for item in item_list:
 583                 attrs[item].update({"stats": stats[item.id]})
 584 
 585         return attrs
 586 
 587     def serialize(self, obj, attrs, user):
 588         result = super(StreamGroupSerializer, self).serialize(obj, attrs, user)
 589 
 590         if self.stats_period:
 591             result["stats"] = {self.stats_period: attrs["stats"]}
 592 
 593         if self.matching_event_id:
 594             result["matchingEventId"] = self.matching_event_id
 595 
 596         if self.matching_event_environment:
 597             result["matchingEventEnvironment"] = self.matching_event_environment
 598 
 599         return result
