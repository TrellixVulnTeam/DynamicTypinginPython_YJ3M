  10 def get_users_for_pull_requests(item_list, user=None):
  11     authors = list(
  12         CommitAuthor.objects.filter(id__in=[i.author_id for i in item_list if i.author_id])
  13     )
  14 
  15     if authors:
  16         org_ids = set(item.organization_id for item in item_list)
  17         if len(org_ids) == 1:
  18             return get_users_for_authors(organization_id=org_ids.pop(), authors=authors, user=user)
  19     return {}

  22 @register(PullRequest)
  23 class PullRequestSerializer(Serializer):
  24     def get_attrs(self, item_list, user):
  25         users_by_author = get_users_for_pull_requests(item_list, user)
  26         repositories = list(Repository.objects.filter(id__in=[c.repository_id for c in item_list]))
  27         repository_map = {repository.id: repository for repository in repositories}
  28         serialized_repos = {r["id"]: r for r in serialize(repositories, user)}
  29 
  30         result = {}
  31         for item in item_list:
  32             repository_id = six.text_type(item.repository_id)
  33             external_url = ""
  34             if item.repository_id in repository_map:
  35                 external_url = self._external_url(repository_map[item.repository_id], item)
  36             result[item] = {
  37                 "repository": serialized_repos.get(repository_id, {}),
  38                 "external_url": external_url,
  39                 "user": users_by_author.get(six.text_type(item.author_id), {})
  40                 if item.author_id
  41                 else {},
  42             }
  43 
  44         return result

  46     def _external_url(self, repository, pull):
  47         from sentry.plugins.base import bindings
  48 
  49         provider_id = repository.provider
  50         if not provider_id or not provider_id.startswith("integrations:"):
  51             return None
  52         provider_cls = bindings.get("integration-repository.provider").get(provider_id)
  53         provider = provider_cls(provider_id)
  54         return provider.pull_request_url(repository, pull)
