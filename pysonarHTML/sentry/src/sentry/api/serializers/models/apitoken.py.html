   9 @register(ApiToken)
  10 class ApiTokenSerializer(Serializer):
  11     def get_attrs(self, item_list, user):
  12         apps = {
  13             d["id"]: d
  14             for d in serialize(set(i.application for i in item_list if i.application_id), user)
  15         }
  16 
  17         attrs = {}
  18         for item in item_list:
  19             attrs[item] = {
  20                 "application": (apps.get(item.application.client_id) if item.application else None)
  21             }
  22         return attrs
  23 
  24     def serialize(self, obj, attrs, user):
  25         data = {
  26             "id": six.text_type(obj.id),
  27             "scopes": obj.get_scopes(),
  28             "application": attrs["application"],
  29             "expiresAt": obj.expires_at,
  30             "dateCreated": obj.date_added,
  31             "state": attrs.get("state"),
  32         }
  33         if not attrs["application"]:
  34             data["token"] = obj.token
  35             data["refreshToken"] = obj.refresh_token
  36         return data
