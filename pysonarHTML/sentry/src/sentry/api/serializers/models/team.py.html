  26 def get_team_memberships(team_list, user):
  27     """Get memberships the user has in the provided team list"""
  28     if user.is_authenticated():
  29         return OrganizationMemberTeam.objects.filter(
  30             organizationmember__user=user, team__in=team_list
  31         ).values_list("team", flat=True)
  32     return []

  35 def get_member_totals(team_list, user):
  36     """Get the total number of members in each team"""
  37     if user.is_authenticated():
  38         query = (
  39             Team.objects.filter(
  40                 id__in=[t.pk for t in team_list],
  41                 organizationmember__invite_status=InviteStatus.APPROVED.value,
  42             )
  43             .annotate(member_count=Count("organizationmemberteam"))
  44             .values("id", "member_count")
  45         )
  46         return {item["id"]: item["member_count"] for item in query}
  47     return {}

  50 def get_org_roles(org_ids, user):
  51     """Get the role the user has in each org"""
  52     if user.is_authenticated():
  53         # map of org id to role
  54         return {
  55             om["organization_id"]: om["role"]
  56             for om in OrganizationMember.objects.filter(
  57                 user=user, organization__in=set(org_ids)
  58             ).values("role", "organization_id")
  59         }
  60     return {}

  63 def get_access_requests(item_list, user):
  64     if user.is_authenticated():
  65         return frozenset(
  66             OrganizationAccessRequest.objects.filter(
  67                 team__in=item_list, member__user=user
  68             ).values_list("team", flat=True)
  69         )
  70     return frozenset()

  74 class TeamSerializer(Serializer):
  75     def get_attrs(self, item_list, user):
  76         request = env.request
  77         org_ids = set([t.organization_id for t in item_list])
  78 
  79         org_roles = get_org_roles(org_ids, user)
  80 
  81         member_totals = get_member_totals(item_list, user)
  82         memberships = get_team_memberships(item_list, user)
  83         access_requests = get_access_requests(item_list, user)
  84 
  85         avatars = {a.team_id: a for a in TeamAvatar.objects.filter(team__in=item_list)}
  86 
  87         is_superuser = request and is_active_superuser(request) and request.user == user
  88         result = {}
  89 
  90         for team in item_list:
  91             is_member = team.id in memberships
  92             org_role = org_roles.get(team.organization_id)
  93             if is_member:
  94                 has_access = True
  95             elif is_superuser:
  96                 has_access = True
  97             elif team.organization.flags.allow_joinleave:
  98                 has_access = True
  99             elif org_role and roles.get(org_role).is_global:
 100                 has_access = True
 101             else:
 102                 has_access = False
 103             result[team] = {
 104                 "pending_request": team.id in access_requests,
 105                 "is_member": is_member,
 106                 "has_access": has_access,
 107                 "avatar": avatars.get(team.id),
 108                 "member_count": member_totals.get(team.id, 0),
 109             }
 110         return result
 111 
 112     def serialize(self, obj, attrs, user):
 113         if attrs.get("avatar"):
 114             avatar = {
 115                 "avatarType": attrs["avatar"].get_avatar_type_display(),
 116                 "avatarUuid": attrs["avatar"].ident if attrs["avatar"].file_id else None,
 117             }
 118         else:
 119             avatar = {"avatarType": "letter_avatar", "avatarUuid": None}
 120         return {
 121             "id": six.text_type(obj.id),
 122             "slug": obj.slug,
 123             "name": obj.name,
 124             "dateCreated": obj.date_added,
 125             "isMember": attrs["is_member"],
 126             "hasAccess": attrs["has_access"],
 127             "isPending": attrs["pending_request"],
 128             "memberCount": attrs["member_count"],
 129             "avatar": avatar,
 130         }

 133 class TeamWithProjectsSerializer(TeamSerializer):
 134     def get_attrs(self, item_list, user):
 135         project_teams = list(
 136             ProjectTeam.objects.filter(team__in=item_list, project__status=ProjectStatus.VISIBLE)
 137             .order_by("project__name", "project__slug")
 138             .select_related("project")
 139         )
 140 
 141         # TODO(dcramer): we should query in bulk for ones we're missing here
 142         orgs = {i.organization_id: i.organization for i in item_list}
 143 
 144         for project_team in project_teams:
 145             project_team.project._organization_cache = orgs[project_team.project.organization_id]
 146 
 147         projects = [pt.project for pt in project_teams]
 148         projects_by_id = {
 149             project.id: data for project, data in zip(projects, serialize(projects, user))
 150         }
 151 
 152         project_map = defaultdict(list)
 153         for project_team in project_teams:
 154             project_map[project_team.team_id].append(projects_by_id[project_team.project_id])
 155 
 156         result = super(TeamWithProjectsSerializer, self).get_attrs(item_list, user)
 157         for team in item_list:
 158             result[team]["projects"] = project_map[team.id]
 159         return result
 160 
 161     def serialize(self, obj, attrs, user):
 162         d = super(TeamWithProjectsSerializer, self).serialize(obj, attrs, user)
 163         d["projects"] = attrs["projects"]
 164         return d
