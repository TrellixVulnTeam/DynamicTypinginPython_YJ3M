  20 def get_crash_files(events):
  21     event_ids = [x.event_id for x in events if x.platform == "native"]
  22     rv = {}
  23     if event_ids:
  24         attachments = EventAttachment.objects.filter(event_id__in=event_ids).select_related("file")
  25         for attachment in attachments:
  26             if attachment.file.type in CRASH_FILE_TYPES:
  27                 rv[attachment.event_id] = attachment
  28     return rv

  35     def _get_entries(self, event, user, is_public=False):
  36         # XXX(dcramer): These are called entries for future-proofing
  37 
  38         platform = event.platform
  39         meta = event.data.get("_meta") or {}
  40         interface_list = []
  41 
  42         for key, interface in six.iteritems(event.interfaces):
  43             # we treat user as a special contextual item
  44             if key in self._reserved_keys:
  45                 continue
  46 
  47             data = interface.get_api_context(is_public=is_public, platform=platform)
  48             # data might not be returned for e.g. a public HTTP repr
  49             # However, spans can be an empty list and should still be included.
  50             if not data and interface.path != "spans":
  51                 continue
  52 
  53             entry = {"data": data, "type": interface.external_type}
  54 
  55             api_meta = None
  56             if meta.get(key):
  57                 api_meta = interface.get_api_meta(meta[key], is_public=is_public, platform=platform)
  58                 api_meta = meta_with_chunks(data, api_meta)
  59 
  60             interface_list.append((interface, entry, api_meta))
  61 
  62         interface_list.sort(key=lambda x: x[0].get_display_score(), reverse=True)
  63 
  64         return (
  65             [i[1] for i in interface_list],
  66             {k: {"data": i[2]} for k, i in enumerate(interface_list) if i[2]},
  67         )

  69     def _get_interface_with_meta(self, event, name, is_public=False):
  70         interface = event.get_interface(name)
  71         if not interface:
  72             return (None, None)
  73 
  74         platform = event.platform
  75         data = interface.get_api_context(is_public=is_public, platform=platform)
  76         event_meta = event.data.get("_meta") or {}
  77         if not data or not event_meta.get(name):
  78             return (data, None)
  79 
  80         api_meta = interface.get_api_meta(event_meta[name], is_public=is_public, platform=platform)
  81         # data might not be returned for e.g. a public HTTP repr
  82         if not api_meta:
  83             return (data, None)
  84 
  85         return (data, meta_with_chunks(data, api_meta))

  87     def _get_tags_with_meta(self, event):
  88         meta = get_path(event.data, "_meta", "tags") or {}
  89 
  90         # If we have meta, we need to get the tags in their original order
  91         # from the raw event body as the indexes need to line up with the
  92         # metadata indexes. In other cases we can use event.tags
  93         if meta:
  94             raw_tags = event.data.get("tags") or []
  95         else:
  96             raw_tags = event.tags
  97 
  98         tags = sorted(
  99             [
 100                 {
 101                     "key": kv[0].split("sentry:", 1)[-1],
 102                     "value": kv[1],
 103                     "_meta": meta.get(kv[0]) or get_path(meta, six.text_type(i), "1") or None,
 104                 }
 105                 for i, kv in enumerate(raw_tags)
 106                 if kv is not None and kv[0] is not None and kv[1] is not None
 107             ],
 108             key=lambda x: x["key"],
 109         )
 110 
 111         # Add 'query' for each tag to tell the UI what to use as query
 112         # params for this tag.
 113         for tag in tags:
 114             query = convert_user_tag_to_query(tag["key"], tag["value"])
 115             if query:
 116                 tag["query"] = query
 117 
 118         tags_meta = {
 119             six.text_type(i): {"value": e.pop("_meta")}
 120             for i, e in enumerate(tags)
 121             if e.get("_meta")
 122         }
 123 
 124         return (tags, meta_with_chunks(tags, tags_meta))

 126     def _get_attr_with_meta(self, event, attr, default=None):
 127         value = event.data.get(attr, default)
 128         meta = get_path(event.data, "_meta", attr)
 129         return (value, meta_with_chunks(value, meta))

 131     def _get_legacy_message_with_meta(self, event):
 132         meta = event.data.get("_meta")
 133 
 134         message = get_path(event.data, "logentry", "formatted")
 135         msg_meta = get_path(meta, "logentry", "formatted")
 136 
 137         if not message:
 138             message = get_path(event.data, "logentry", "message")
 139             msg_meta = get_path(meta, "logentry", "message")
 140 
 141         if not message:
 142             message = event.message
 143             msg_meta = None
 144 
 145         return (message, meta_with_chunks(message, msg_meta))

 147     def _get_release_info(self, user, event):
 148         version = event.get_tag("sentry:release")
 149         if not version:
 150             return None
 151         try:
 152             release = Release.objects.get(
 153                 projects=event.project,
 154                 organization_id=event.project.organization_id,
 155                 version=version,
 156             )
 157         except Release.DoesNotExist:
 158             return {"version": version}
 159         return serialize(release, user)
 160 
 161     def _get_user_report(self, user, event):
 162         try:
 163             user_report = UserReport.objects.get(event_id=event.event_id, project=event.project)
 164         except UserReport.DoesNotExist:
 165             user_report = None
 166         return serialize(user_report, user)

 168     def get_attrs(self, item_list, user, is_public=False):
 169         crash_files = get_crash_files(item_list)
 170         results = {}
 171         for item in item_list:
 172             # TODO(dcramer): convert to get_api_context
 173             (user_data, user_meta) = self._get_interface_with_meta(item, "user", is_public)
 174             (contexts_data, contexts_meta) = self._get_interface_with_meta(
 175                 item, "contexts", is_public
 176             )
 177             (sdk_data, sdk_meta) = self._get_interface_with_meta(item, "sdk", is_public)
 178 
 179             (entries, entries_meta) = self._get_entries(item, user, is_public=is_public)
 180 
 181             crash_file = crash_files.get(item.event_id)
 182 
 183             results[item] = {
 184                 "entries": entries,
 185                 "user": user_data,
 186                 "contexts": contexts_data or {},
 187                 "sdk": sdk_data,
 188                 "crash_file": serialize(crash_file, user=user),
 189                 "_meta": {
 190                     "entries": entries_meta,
 191                     "user": user_meta,
 192                     "contexts": contexts_meta,
 193                     "sdk": sdk_meta,
 194                 },
 195             }
 196         return results

 198     def should_display_error(self, error):
 199         name = error.get("name")
 200         if not isinstance(name, six.string_types):
 201             return True
 202 
 203         return (
 204             not name.startswith("breadcrumbs.")
 205             and not name.startswith("extra.")
 206             and ".frames." not in name
 207         )

 209     def serialize(self, obj, attrs, user):
 210         errors = [
 211             EventError(error).get_api_context()
 212             for error in get_path(obj.data, "errors", filter=True, default=())
 213             # TODO(ja): Temporary workaround to hide certain normalization errors.
 214             # Remove this and the test in tests/sentry/api/serializers/test_event.py
 215             if self.should_display_error(error)
 216         ]
 217 
 218         (message, message_meta) = self._get_legacy_message_with_meta(obj)
 219         (tags, tags_meta) = self._get_tags_with_meta(obj)
 220         (context, context_meta) = self._get_attr_with_meta(obj, "extra", {})
 221         (packages, packages_meta) = self._get_attr_with_meta(obj, "modules", {})
 222 
 223         received = obj.data.get("received")
 224         if received:
 225             # Sentry at one point attempted to record invalid types here.
 226             # Remove after June 2 2016
 227             try:
 228                 received = datetime.utcfromtimestamp(received).replace(tzinfo=timezone.utc)
 229             except TypeError:
 230                 received = None
 231 
 232         d = {
 233             "id": obj.event_id,
 234             "groupID": six.text_type(obj.group_id) if obj.group_id else None,
 235             "eventID": obj.event_id,
 236             "projectID": six.text_type(obj.project_id),
 237             "size": obj.size,
 238             "entries": attrs["entries"],
 239             "dist": obj.dist,
 240             # See GH-3248
 241             "message": message,
 242             "title": obj.title,
 243             "location": obj.location,
 244             "user": attrs["user"],
 245             "contexts": attrs["contexts"],
 246             "sdk": attrs["sdk"],
 247             # TODO(dcramer): move into contexts['extra']
 248             "context": context,
 249             "packages": packages,
 250             "type": obj.get_event_type(),
 251             "metadata": obj.get_event_metadata(),
 252             "tags": tags,
 253             "platform": obj.platform,
 254             "dateReceived": received,
 255             "errors": errors,
 256             "_meta": {
 257                 "entries": attrs["_meta"]["entries"],
 258                 "message": message_meta,
 259                 "user": attrs["_meta"]["user"],
 260                 "contexts": attrs["_meta"]["contexts"],
 261                 "sdk": attrs["_meta"]["sdk"],
 262                 "context": context_meta,
 263                 "packages": packages_meta,
 264                 "tags": tags_meta,
 265             },
 266         }
 267         # Serialize attributes that are specific to different types of events.
 268         if obj.get_event_type() == "transaction":
 269             d.update(self.__serialize_transaction_attrs(attrs, obj))
 270         else:
 271             d.update(self.__serialize_error_attrs(attrs, obj))
 272         return d

 274     def __serialize_transaction_attrs(self, attrs, obj):
 275         """
 276         Add attributes that are only present on transaction events.
 277         """
 278         return {
 279             "startTimestamp": obj.data.get("start_timestamp"),
 280             "endTimestamp": obj.data.get("timestamp"),
 281         }
 282 
 283     def __serialize_error_attrs(self, attrs, obj):
 284         """
 285         Add attributes that are present on error and default event types
 286         """
 287         return {
 288             "crashFile": attrs["crash_file"],
 289             "culprit": obj.culprit,
 290             "dateCreated": obj.datetime,
 291             "fingerprints": obj.get_hashes(),
 292             "groupingConfig": obj.get_grouping_config(),
 293         }

 302         return list(get_suggested_updates(SdkSetupState.from_event_json(obj.data)))

 304     def serialize(self, obj, attrs, user):
 305         result = super(DetailedEventSerializer, self).serialize(obj, attrs, user)

 328 class SimpleEventSerializer(EventSerializer):
 329     """
 330     Simple event serializer that renders a basic outline of an event without
 331     most interfaces/breadcrumbs. This can be used for basic event list queries
 332     where we don't need the full detail. The side effect is that, if the
 333     serialized events are actually SnubaEvents, we can render them without
 334     needing to fetch the event bodies from nodestore.
 335 
 336     NB it would be super easy to inadvertently add a property accessor here
 337     that would require a nodestore lookup for a SnubaEvent serialized using
 338     this serializer. You will only really notice you've done this when the
 339     organization event search API gets real slow.
 340     """
 341 
 342     def get_attrs(self, item_list, user):
 343         crash_files = get_crash_files(item_list)
 344         return {
 345             event: {"crash_file": serialize(crash_files.get(event.event_id), user=user)}
 346             for event in item_list
 347         }
 348 
 349     def serialize(self, obj, attrs, user):
 350         tags = [{"key": key.split("sentry:", 1)[-1], "value": value} for key, value in obj.tags]
 351         for tag in tags:
 352             query = convert_user_tag_to_query(tag["key"], tag["value"])
 353             if query:
 354                 tag["query"] = query
 355 
 356         user = obj.get_minimal_user()
 357 
 358         return {
 359             "id": six.text_type(obj.event_id),
 360             "event.type": six.text_type(obj.get_event_type()),
 361             "groupID": six.text_type(obj.group_id) if obj.group_id else None,
 362             "eventID": six.text_type(obj.event_id),
 363             "projectID": six.text_type(obj.project_id),
 364             # XXX for 'message' this doesn't do the proper resolution of logentry
 365             # etc. that _get_legacy_message_with_meta does.
 366             "message": obj.message,
 367             "title": obj.title,
 368             "location": obj.location,
 369             "culprit": obj.culprit,
 370             "user": user and user.get_api_context(),
 371             "tags": tags,
 372             "platform": obj.platform,
 373             "dateCreated": obj.datetime,
 374             # Needed to generate minidump links in UI
 375             "crashFile": attrs["crash_file"],
 376         }

 379 class ExternalEventSerializer(EventSerializer):

 381     Event serializer for the minimum event data needed to send to an external service. This

 387         for tag in tags:
 388             query = convert_user_tag_to_query(tag["key"], tag["value"])
 389             if query:
 390                 tag["query"] = query
 391 
 392         user = obj.get_minimal_user()
 393 
 394         return {
 395             "groupID": six.text_type(obj.group_id) if obj.group_id else None,
 396             "eventID": six.text_type(obj.event_id),
 397             "project": six.text_type(obj.project_id),
 398             # XXX for 'message' this doesn't do the proper resolution of logentry
 399             # etc. that _get_legacy_message_with_meta does.
 400             "message": obj.message,
 401             "title": obj.title,
 402             "location": obj.location,
 403             "culprit": obj.culprit,
 404             "user": user and user.get_api_context(),
 405             "tags": tags,
 406             "platform": obj.platform,
 407             "datetime": obj.datetime.strftime("%Y-%m-%dT%H:%M:%S.%fZ"),
 408         }
