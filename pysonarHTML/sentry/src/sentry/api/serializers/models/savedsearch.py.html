   9 @register(SavedSearch)
  10 class SavedSearchSerializer(Serializer):
  11     def get_attrs(self, item_list, user):
  12         if user.is_authenticated():
  13             user_defaults = tuple(
  14                 SavedSearchUserDefault.objects.filter(
  15                     savedsearch__in=item_list, user=user
  16                 ).values_list("savedsearch", flat=True)
  17             )
  18         else:
  19             user_defaults = ()
  20 
  21         attrs = {}
  22         for item in item_list:
  23             attrs[item] = {"isUserDefault": item.id in user_defaults}
  24         return attrs
  25 
  26     def serialize(self, obj, attrs, user):
  27         return {
  28             "id": six.text_type(obj.id),
  29             # TODO: Remove once we've completely deprecated Sentry 9
  30             "projectId": six.text_type(obj.project_id) if obj.project_id else None,
  31             "type": obj.type,
  32             "name": obj.name,
  33             "query": obj.query,
  34             # TODO: Remove once we've completely deprecated Sentry 9
  35             "isDefault": obj.is_default,
  36             # TODO: Remove once we've completely deprecated Sentry 9
  37             "isUserDefault": attrs["isUserDefault"],
  38             "dateCreated": obj.date_added,
  39             # TODO: Remove once we've completely deprecated Sentry 9
  40             "isPrivate": bool(obj.owner),
  41             "isGlobal": obj.is_global,
  42             "isPinned": obj.is_pinned,
  43             "isOrgCustom": obj.is_org_custom_search,
  44         }
