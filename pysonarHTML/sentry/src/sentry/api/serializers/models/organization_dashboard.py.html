   9 @register(Widget)
  10 class WidgetSerializer(Serializer):
  11     def get_attrs(self, item_list, user):
  12         result = {}
  13         data_sources = serialize(
  14             list(WidgetDataSource.objects.filter(widget_id__in=[i.id for i in item_list]))
  15         )
  16 
  17         for widget in item_list:
  18             widget_data_sources = [
  19                 d for d in data_sources if d["widgetId"] == six.text_type(widget.id)
  20             ]
  21             result[widget] = {"dataSources": widget_data_sources}
  22 
  23         return result
