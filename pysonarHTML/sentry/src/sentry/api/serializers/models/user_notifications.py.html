   9 # notification_option_key is one of:
  10 # - mail:alert
  11 # - workflow:notifications
  12 # - deploy-emails
  13 # - reports:disabled-organizations
  14 # - mail:email
  15 class UserNotificationsSerializer(Serializer):
  16     def get_attrs(self, item_list, user, **kwargs):
  17         notification_option_key = kwargs["notification_option_key"]
  18         filter_args = {}
  19 
  20         if notification_option_key in ["alerts", "workflow", "email"]:
  21             filter_args["project__isnull"] = False
  22         elif notification_option_key == "deploy":
  23             filter_args["organization__isnull"] = False
  24 
  25         data = list(
  26             UserOption.objects.filter(
  27                 key=notification_option_key, user__in=item_list, **filter_args
  28             ).select_related("user", "project", "organization")
  29         )
  30 
  31         results = defaultdict(list)
  32 
  33         for uo in data:
  34             results[uo.user].append(uo)
  35 
  36         return results
  37 
  38     def serialize(self, obj, attrs, user, **kwargs):
  39         notification_option_key = kwargs["notification_option_key"]
  40         data = {}
  41 
  42         for uo in attrs:
  43             if notification_option_key == "reports:disabled-organizations":
  44                 # UserOption for key=reports:disabled-organizations saves a list of orgIds
  45                 # that should not receive reports
  46                 # This UserOption should have both project + organization = None
  47                 for org_id in uo.value:
  48                     data[org_id] = 0
  49             elif uo.project is not None:
  50                 data[uo.project.id] = uo.value
  51             elif uo.organization is not None:
  52                 data[uo.organization.id] = uo.value
  53         return data
