  17 @register(GroupRelease)
  18 class GroupReleaseSerializer(Serializer):
  19     def get_attrs(self, item_list, user):
  20         release_list = list(Release.objects.filter(id__in=[i.release_id for i in item_list]))
  21         releases = {r.id: d for r, d in zip(release_list, serialize(release_list, user))}
  22 
  23         result = {}
  24         for item in item_list:
  25             result[item] = {"release": releases.get(item.release_id)}
  26         return result
  27 
  28     def serialize(self, obj, attrs, user):
  29         return {
  30             "release": attrs["release"],
  31             "environment": obj.environment,
  32             "firstSeen": obj.first_seen,
  33             "lastSeen": obj.last_seen,
  34         }

  37 class GroupReleaseWithStatsSerializer(GroupReleaseSerializer):
  38     STATS_PERIODS = {
  39         "24h": StatsPeriod(24, timedelta(hours=1)),
  40         "30d": StatsPeriod(30, timedelta(hours=24)),
  41     }
  42 
  43     def __init__(self, since=None, until=None):
  44         self.since = since
  45         self.until = until
  46 
  47     def get_attrs(self, item_list, user):
  48         attrs = super(GroupReleaseWithStatsSerializer, self).get_attrs(item_list, user)
  49 
  50         items = {}
  51         for item in item_list:
  52             items.setdefault(item.group_id, []).append(item.id)
  53             attrs[item]["stats"] = {}
  54 
  55         for key, (segments, interval) in six.iteritems(self.STATS_PERIODS):
  56             until = self.until or timezone.now()
  57             since = self.since or until - (segments * interval)
  58 
  59             try:
  60                 stats = tsdb.get_frequency_series(
  61                     model=tsdb.models.frequent_releases_by_group,
  62                     items=items,
  63                     start=since,
  64                     end=until,
  65                     rollup=int(interval.total_seconds()),
  66                 )
  67             except NotImplementedError:
  68                 # TODO(dcramer): probably should log this, but not worth
  69                 # erring out
  70                 stats = {}
  71 
  72             for item in item_list:
  73                 attrs[item]["stats"][key] = [
  74                     (k, v[item.id]) for k, v in stats.get(item.group_id, {})
  75                 ]
  76         return attrs
  77 
  78     def serialize(self, obj, attrs, user):
  79         result = super(GroupReleaseWithStatsSerializer, self).serialize(obj, attrs, user)
  80         result["stats"] = attrs["stats"]
  81         return result
