  16 @register(AlertRule)
  17 class AlertRuleSerializer(Serializer):
  18     def get_attrs(self, item_list, user, **kwargs):
  19         alert_rules = {item.id: item for item in item_list}
  20         attach_foreignkey(item_list, AlertRule.snuba_query, related=("environment",))
  21 
  22         result = defaultdict(dict)
  23         triggers = AlertRuleTrigger.objects.filter(alert_rule__in=item_list).order_by("label")
  24         serialized_triggers = serialize(list(triggers))
  25         for trigger, serialized in zip(triggers, serialized_triggers):
  26             alert_rule_triggers = result[alert_rules[trigger.alert_rule_id]].setdefault(
  27                 "triggers", []
  28             )
  29             alert_rule_triggers.append(serialized)
  30 
  31         return result
  32 
  33     def serialize(self, obj, attrs, user):
  34         env = obj.snuba_query.environment
  35         # Temporary: Translate aggregate back here from `tags[sentry:user]` to `user` for the frontend.
  36 
  37         aggregate = translate_aggregate_field(obj.snuba_query.aggregate, reverse=True)
  38 
  39         return {
  40             "id": six.text_type(obj.id),
  41             "name": obj.name,
  42             "organizationId": six.text_type(obj.organization_id),
  43             "status": obj.status,
  44             "dataset": obj.snuba_query.dataset,
  45             "query": obj.snuba_query.query,
  46             "aggregate": aggregate,
  47             # TODO: Start having the frontend expect seconds
  48             "timeWindow": obj.snuba_query.time_window / 60,
  49             "environment": env.name if env else None,
  50             # TODO: Start having the frontend expect seconds
  51             "resolution": obj.snuba_query.resolution / 60,
  52             "thresholdPeriod": obj.threshold_period,
  53             "triggers": attrs.get("triggers", []),
  54             "includeAllProjects": obj.include_all_projects,
  55             "dateModified": obj.date_modified,
  56             "dateCreated": obj.date_added,
  57         }

  60 class DetailedAlertRuleSerializer(AlertRuleSerializer):
  61     def get_attrs(self, item_list, user, **kwargs):
  62         result = super(DetailedAlertRuleSerializer, self).get_attrs(item_list, user, **kwargs)
  63         alert_rule_projects = AlertRule.objects.filter(
  64             id__in=[item.id for item in item_list]
  65         ).values_list("id", "snuba_query__subscriptions__project__slug")
  66         alert_rules = {item.id: item for item in item_list}
  67         for alert_rule_id, project_slug in alert_rule_projects:
  68             rule_result = result[alert_rules[alert_rule_id]].setdefault("projects", [])
  69             rule_result.append(project_slug)
  70 
  71         for alert_rule_id, project_slug in AlertRuleExcludedProjects.objects.filter(
  72             alert_rule__in=item_list
  73         ).values_list("alert_rule_id", "project__slug"):
  74             exclusions = result[alert_rules[alert_rule_id]].setdefault("excludedProjects", [])
  75             exclusions.append(project_slug)
  76         return result
  77 
  78     def serialize(self, obj, attrs, user):
  79         data = super(DetailedAlertRuleSerializer, self).serialize(obj, attrs, user)
  80         data["projects"] = sorted(attrs["projects"])
  81         data["excludedProjects"] = sorted(attrs.get("excludedProjects", []))
  82         return data

  85 class CombinedRuleSerializer(Serializer):
  86     def get_attrs(self, item_list, user, **kwargs):
  87         results = super(CombinedRuleSerializer, self).get_attrs(item_list, user)
  88 
  89         alert_rules = serialize([x for x in item_list if isinstance(x, AlertRule)], user=user)
  90         rules = serialize([x for x in item_list if isinstance(x, Rule)], user=user)
  91 
  92         for item in item_list:
  93             if isinstance(item, AlertRule):
  94                 results[item] = alert_rules.pop(0)
  95             elif isinstance(item, Rule):
  96                 results[item] = rules.pop(0)
  97 
  98         return results
  99 
 100     def serialize(self, obj, attrs, user, **kwargs):
 101         if isinstance(obj, AlertRule):
 102             attrs["type"] = "alert_rule"
 103             return attrs
 104         elif isinstance(obj, Rule):
 105             attrs["type"] = "rule"
 106             return attrs
 107         else:
 108             raise AssertionError("Invalid rule to serialize: %r" % type(obj))
