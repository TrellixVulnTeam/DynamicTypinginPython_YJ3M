  10 @register(CommitFileChange)

  12     def get_attrs(self, item_list, user):
  13         commits = list(
  14             Commit.objects.filter(id__in=[f.commit_id for f in item_list]).select_related("author")
  15         )
  16         users_by_author = get_users_for_commits(commits)
  17         commits_by_id = {commit.id: commit for commit in commits}
  18 
  19         repo_names_by_id = dict(
  20             Repository.objects.filter(
  21                 id__in=[commit.repository_id for commit in commits]
  22             ).values_list("id", "name")
  23         )
  24 
  25         result = {}
  26         for item in item_list:
  27             commit = commits_by_id[item.commit_id]
  28             result[item] = {
  29                 "user": users_by_author.get(six.text_type(commit.author_id), {})
  30                 if commit.author_id
  31                 else {},
  32                 "message": commit.message,
  33                 "repository_name": repo_names_by_id.get(commit.repository_id),
  34             }
  35 
  36         return result
  37 
  38     def serialize(self, obj, attrs, user):
  39         return {
  40             "id": six.text_type(obj.id),
  41             "orgId": obj.organization_id,
  42             "author": attrs.get("user", {}),
  43             "commitMessage": attrs.get("message", ""),
  44             "filename": obj.filename,
  45             "type": obj.type,
  46             "repoName": attrs.get("repository_name", ""),
  47         }
