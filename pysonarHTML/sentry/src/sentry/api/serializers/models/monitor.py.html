   5 from sentry.api.serializers import Serializer, register, serialize
   6 from sentry.models import Monitor, Project

   9 @register(Monitor)
  10 class MonitorSerializer(Serializer):
  11     def get_attrs(self, item_list, user):
  12         # TODO(dcramer): assert on relations
  13         projects = {
  14             d["id"]: d
  15             for d in serialize(
  16                 list(Project.objects.filter(id__in=[i.project_id for i in item_list])), user
  17             )
  18         }
  19 
  20         return {
  21             item: {"project": projects[six.text_type(item.project_id)] if item.project_id else None}
  22             for item in item_list
  23         }
  24 
  25     def serialize(self, obj, attrs, user):
  26         config = obj.config.copy()
  27         if "schedule_type" in config:
  28             config["schedule_type"] = obj.get_schedule_type_display()
  29         return {
  30             "id": six.text_type(obj.guid),
  31             "status": obj.get_status_display(),
  32             "type": obj.get_type_display(),
  33             "name": obj.name,
  34             "config": config,
  35             "lastCheckIn": obj.last_checkin,
  36             "nextCheckIn": obj.next_checkin,
  37             "dateCreated": obj.date_added,
  38             "project": attrs["project"],
  39         }
