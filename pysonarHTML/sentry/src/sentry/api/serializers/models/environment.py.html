  33 class GroupEnvironmentWithStatsSerializer(EnvironmentSerializer):
  34     STATS_PERIODS = {
  35         "24h": StatsPeriod(24, timedelta(hours=1)),
  36         "30d": StatsPeriod(30, timedelta(hours=24)),
  37     }
  38 
  39     def __init__(self, group, since=None, until=None):
  40         self.group = group
  41         self.since = since
  42         self.until = until
  43 
  44     def get_attrs(self, item_list, user):
  45         attrs = {item: {"stats": {}} for item in item_list}
  46         items = {self.group.id: []}
  47         for item in item_list:
  48             items[self.group.id].append(item.id)
  49 
  50         for key, (segments, interval) in six.iteritems(self.STATS_PERIODS):
  51             until = self.until or timezone.now()
  52             since = self.since or until - (segments * interval)
  53 
  54             try:
  55                 stats = tsdb.get_frequency_series(
  56                     model=tsdb.models.frequent_environments_by_group,
  57                     items=items,
  58                     start=since,
  59                     end=until,
  60                     rollup=int(interval.total_seconds()),
  61                 )
  62             except NotImplementedError:
  63                 # TODO(dcramer): probably should log this, but not worth
  64                 # erring out
  65                 stats = {}
  66 
  67             for item in item_list:
  68                 attrs[item]["stats"][key] = [
  69                     (k, v[item.id]) for k, v in stats.get(self.group.id, {})
  70                 ]
  71         return attrs
  72 
  73     def serialize(self, obj, attrs, user):
  74         result = super(GroupEnvironmentWithStatsSerializer, self).serialize(obj, attrs, user)
  75         result["stats"] = attrs["stats"]
  76         return result
