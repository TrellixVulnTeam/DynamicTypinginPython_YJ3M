  55 @register(OrganizationIntegration)
  56 class OrganizationIntegrationSerializer(Serializer):
  57     def serialize(self, obj, attrs, user):
  58         # XXX(epurkhiser): This is O(n) for integrations, especially since
  59         # we're using the IntegrationConfigSerializer which pulls in the
  60         # integration installation config object which very well may be making
  61         # API request for config options.
  62         integration = serialize(
  63             objects=obj.integration,
  64             user=user,
  65             serializer=IntegrationConfigSerializer(obj.organization.id),
  66         )
  67         try:
  68             installation = obj.integration.get_installation(obj.organization_id)
  69         except NotImplementedError:
  70             # slack doesn't have an installation implementation
  71             config_data = obj.config
  72         else:
  73             # just doing this to avoid querying for an object we already have
  74             installation._org_integration = obj
  75             config_data = installation.get_config_data()
  76 
  77         integration.update({"configData": config_data})
  78 
  79         return integration

 102 class IntegrationIssueConfigSerializer(IntegrationSerializer):
 103     def __init__(self, group, action, params=None):
 104         self.group = group
 105         self.action = action
 106         self.params = params
 107 
 108     def serialize(self, obj, attrs, user, organization_id=None):
 109         data = super(IntegrationIssueConfigSerializer, self).serialize(obj, attrs, user)
 110         installation = obj.get_installation(organization_id)
 111 
 112         if self.action == "link":
 113             config = installation.get_link_issue_config(self.group, params=self.params)
 114             data["linkIssueConfig"] = config
 115 
 116         if self.action == "create":
 117             config = installation.get_create_issue_config(self.group, params=self.params)
 118             data["createIssueConfig"] = config
 119 
 120         return data

 123 class IntegrationIssueSerializer(IntegrationSerializer):
 124     def __init__(self, group):
 125         self.group = group
 126 
 127     def get_attrs(self, item_list, user, **kwargs):
 128         external_issues = ExternalIssue.objects.filter(
 129             id__in=GroupLink.objects.filter(
 130                 group_id=self.group.id,
 131                 project_id=self.group.project_id,
 132                 linked_type=GroupLink.LinkedType.issue,
 133                 relationship=GroupLink.Relationship.references,
 134             ).values_list("linked_id", flat=True),
 135             integration_id__in=[i.id for i in item_list],
 136         )
 137 
 138         issues_by_integration = defaultdict(list)
 139         ints_by_id = {i.id: i for i in item_list}
 140         for ei in external_issues:
 141             # TODO(jess): move into an external issue serializer?
 142             installation = ints_by_id[ei.integration_id].get_installation(
 143                 self.group.organization.id
 144             )
 145             issues_by_integration[ei.integration_id].append(
 146                 {
 147                     "id": six.text_type(ei.id),
 148                     "key": ei.key,
 149                     "url": installation.get_issue_url(ei.key),
 150                     "title": ei.title,
 151                     "description": ei.description,
 152                     "displayName": installation.get_issue_display_name(ei),
 153                 }
 154             )
 155 
 156         return {
 157             item: {"external_issues": issues_by_integration.get(item.id, [])} for item in item_list
 158         }
 159 
 160     def serialize(self, obj, attrs, user):
 161         data = super(IntegrationIssueSerializer, self).serialize(obj, attrs, user)
 162         data["externalIssues"] = attrs.get("external_issues", [])
 163         return data
