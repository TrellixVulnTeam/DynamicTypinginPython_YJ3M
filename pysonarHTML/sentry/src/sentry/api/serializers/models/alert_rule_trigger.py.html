  17 class AlertRuleTriggerSerializer(Serializer):
  18     def get_attrs(self, item_list, user, **kwargs):
  19 
  20         triggers = {item.id: item for item in item_list}
  21         result = defaultdict(dict)
  22 
  23         actions = AlertRuleTriggerAction.objects.filter(alert_rule_trigger__in=item_list).order_by(
  24             "id"
  25         )
  26         serialized_actions = serialize(list(actions))
  27         for trigger, serialized in zip(actions, serialized_actions):
  28             triggers_actions = result[triggers[trigger.alert_rule_trigger_id]].setdefault(
  29                 "actions", []
  30             )
  31             triggers_actions.append(serialized)
  32 
  33         return result
