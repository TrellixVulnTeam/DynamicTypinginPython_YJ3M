  11 @register(Broadcast)
  12 class BroadcastSerializer(Serializer):
  13     def get_attrs(self, item_list, user):
  14         if not user.is_authenticated():
  15             seen = set()
  16         else:
  17             seen = set(
  18                 BroadcastSeen.objects.filter(broadcast__in=item_list, user=user).values_list(
  19                     "broadcast", flat=True
  20                 )
  21             )
  22 
  23         return {item: {"seen": item.id in seen} for item in item_list}
  24 
  25     def serialize(self, obj, attrs, user):
  26         return {
  27             "id": six.text_type(obj.id),
  28             "message": obj.message,
  29             "title": obj.title,
  30             "link": obj.link,
  31             "cta": obj.cta,
  32             "isActive": obj.is_active,
  33             "dateCreated": obj.date_added,
  34             "dateExpires": obj.date_expires,
  35             "hasSeen": attrs["seen"],
  36         }

  39 class AdminBroadcastSerializer(BroadcastSerializer):
  40     def get_attrs(self, item_list, user):
  41         attrs = super(AdminBroadcastSerializer, self).get_attrs(item_list, user)
  42         counts = dict(
  43             BroadcastSeen.objects.filter(broadcast__in=item_list)
  44             .values("broadcast")
  45             .distinct()
  46             .annotate(user_count=Count("broadcast"))
  47             .values_list("broadcast", "user_count")
  48         )
  49 
  50         for item in attrs:
  51             attrs[item]["user_count"] = counts.get(item.id, 0)
  52         return attrs
  53 
  54     def serialize(self, obj, attrs, user):
  55         context = super(AdminBroadcastSerializer, self).serialize(obj, attrs, user)
  56         context["userCount"] = attrs["user_count"]
  57         return context
