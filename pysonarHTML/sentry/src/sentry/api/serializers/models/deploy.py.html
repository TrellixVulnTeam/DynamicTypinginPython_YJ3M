   9 @register(Deploy)
  10 class DeploySerializer(Serializer):
  11     def get_attrs(self, item_list, user, **kwargs):
  12         environments = {
  13             id: name
  14             for id, name in Environment.objects.filter(
  15                 id__in=[d.environment_id for d in item_list]
  16             ).values_list("id", "name")
  17         }
  18 
  19         result = {}
  20         for item in item_list:
  21             result[item] = {"environment": environments.get(item.environment_id)}
  22 
  23         return result
  24 
  25     def serialize(self, obj, attrs, user, **kwargs):
  26         return {
  27             "id": six.text_type(obj.id),
  28             "environment": attrs.get("environment"),
  29             "dateStarted": obj.date_started,
  30             "dateFinished": obj.date_finished,
  31             "name": obj.name,
  32             "url": obj.url,
  33         }
