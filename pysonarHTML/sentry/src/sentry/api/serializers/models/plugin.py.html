  98 def serialize_field(project, plugin, field):
  99     data = {
 100         "name": six.text_type(field["name"]),
 101         "label": six.text_type(field.get("label") or field["name"].title().replace("_", " ")),
 102         "type": field.get("type", "text"),
 103         "required": field.get("required", True),
 104         "help": six.text_type(field["help"]) if field.get("help") else None,
 105         "placeholder": six.text_type(field["placeholder"]) if field.get("placeholder") else None,
 106         "choices": field.get("choices"),
 107         "readonly": field.get("readonly", False),
 108         "defaultValue": field.get("default"),
 109         "value": None,
 110     }
 111     if field.get("type") != "secret":
 112         data["value"] = plugin.get_option(field["name"], project)
 113     else:
 114         data["hasSavedValue"] = bool(field.get("has_saved_value", False))
 115         data["prefix"] = field.get("prefix", "")
 116 
 117     return data
