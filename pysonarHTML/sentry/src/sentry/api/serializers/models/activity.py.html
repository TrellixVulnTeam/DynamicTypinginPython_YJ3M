  12 @register(Activity)
  13 class ActivitySerializer(Serializer):
  14     def __init__(self, environment_func=None):
  15         self.environment_func = environment_func
  16 
  17     def get_attrs(self, item_list, user):
  18         # TODO(dcramer); assert on relations
  19         users = {d["id"]: d for d in serialize(set(i.user for i in item_list if i.user_id), user)}
  20 
  21         commit_ids = {
  22             i.data["commit"] for i in item_list if i.type == Activity.SET_RESOLVED_IN_COMMIT
  23         }
  24         if commit_ids:
  25             commit_list = list(Commit.objects.filter(id__in=commit_ids))
  26             commits_by_id = {c.id: d for c, d in zip(commit_list, serialize(commit_list, user))}
  27             commits = {
  28                 i: commits_by_id.get(i.data["commit"])
  29                 for i in item_list
  30                 if i.type == Activity.SET_RESOLVED_IN_COMMIT
  31             }
  32         else:
  33             commits = {}
  34 
  35         pull_request_ids = {
  36             i.data["pull_request"]
  37             for i in item_list
  38             if i.type == Activity.SET_RESOLVED_IN_PULL_REQUEST
  39         }
  40         if pull_request_ids:
  41             pull_request_list = list(PullRequest.objects.filter(id__in=pull_request_ids))
  42             pull_requests_by_id = {
  43                 c.id: d for c, d in zip(pull_request_list, serialize(pull_request_list, user))
  44             }
  45             pull_requests = {
  46                 i: pull_requests_by_id.get(i.data["pull_request"])
  47                 for i in item_list
  48                 if i.type == Activity.SET_RESOLVED_IN_PULL_REQUEST
  49             }
  50         else:
  51             pull_requests = {}
  52 
  53         groups = apply_values(
  54             functools.partial(serialize, user=user),
  55             Group.objects.in_bulk(
  56                 set(
  57                     i.data["source_id"] for i in item_list if i.type == Activity.UNMERGE_DESTINATION
  58                 )
  59                 | set(
  60                     i.data["destination_id"] for i in item_list if i.type == Activity.UNMERGE_SOURCE
  61                 )
  62             ),
  63         )
  64 
  65         return {
  66             item: {
  67                 "user": users[six.text_type(item.user_id)] if item.user_id else None,
  68                 "source": groups.get(item.data["source_id"])
  69                 if item.type == Activity.UNMERGE_DESTINATION
  70                 else None,
  71                 "destination": groups.get(item.data["destination_id"])
  72                 if item.type == Activity.UNMERGE_SOURCE
  73                 else None,
  74                 "commit": commits.get(item),
  75                 "pull_request": pull_requests.get(item),
  76             }
  77             for item in item_list
  78         }
  79 
  80     def serialize(self, obj, attrs, user):
  81         if obj.type == Activity.SET_RESOLVED_IN_COMMIT:
  82             data = {"commit": attrs["commit"]}
  83         elif obj.type == Activity.SET_RESOLVED_IN_PULL_REQUEST:
  84             data = {"pullRequest": attrs["pull_request"]}
  85         elif obj.type == Activity.UNMERGE_DESTINATION:
  86             data = {"fingerprints": obj.data["fingerprints"], "source": attrs["source"]}
  87         elif obj.type == Activity.UNMERGE_SOURCE:
  88             data = {"fingerprints": obj.data["fingerprints"], "destination": attrs["destination"]}
  89         else:
  90             data = obj.data
  91             # XXX: We had a problem where Users were embedded into the mentions
  92             # attribute of group notes which needs to be removed
  93             # While group_note update has been fixed there are still many skunky comments
  94             # in the database.
  95             data.pop("mentions", None)
  96 
  97         return {
  98             "id": six.text_type(obj.id),
  99             "user": attrs["user"],
 100             "type": obj.get_type_display(),
 101             "data": data,
 102             "dateCreated": obj.datetime,
 103         }

 106 class OrganizationActivitySerializer(ActivitySerializer):
 107     def get_attrs(self, item_list, user):
 108         from sentry.api.serializers import GroupSerializer
 109 
 110         # TODO(dcramer); assert on relations
 111         attrs = super(OrganizationActivitySerializer, self).get_attrs(item_list, user)
 112 
 113         groups = {
 114             d["id"]: d
 115             for d in serialize(
 116                 set([i.group for i in item_list if i.group_id]),
 117                 user,
 118                 GroupSerializer(environment_func=self.environment_func),
 119             )
 120         }
 121 
 122         projects = {d["id"]: d for d in serialize(set(i.project for i in item_list), user)}
 123 
 124         for item in item_list:
 125             attrs[item]["issue"] = groups[six.text_type(item.group_id)] if item.group_id else None
 126             attrs[item]["project"] = projects[six.text_type(item.project_id)]
 127         return attrs
 128 
 129     def serialize(self, obj, attrs, user):
 130         context = super(OrganizationActivitySerializer, self).serialize(obj, attrs, user)
 131         context["issue"] = attrs["issue"]
 132         context["project"] = attrs["project"]
 133         return context
