  26 def manytoone_to_dict(queryset, key, filter_func=None):
  27     result = defaultdict(list)
  28     for row in queryset:
  29         if filter_func and not filter_func(row):
  30             continue
  31         result[getattr(row, key)].append(row)
  32     return result

  36 class UserSerializer(Serializer):
  37     def _get_identities(self, item_list, user):
  38         if not (env.request and is_active_superuser(env.request)):
  39             item_list = [x for x in item_list if x == user]
  40 
  41         queryset = AuthIdentity.objects.filter(user__in=item_list).select_related(
  42             "auth_provider", "auth_provider__organization"
  43         )
  44 
  45         results = {i.id: [] for i in item_list}
  46         for item in queryset:
  47             results[item.user_id].append(item)
  48         return results

  50     def get_attrs(self, item_list, user):
  51         avatars = {a.user_id: a for a in UserAvatar.objects.filter(user__in=item_list)}
  52         identities = self._get_identities(item_list, user)
  53 
  54         emails = manytoone_to_dict(UserEmail.objects.filter(user__in=item_list), "user_id")
  55         authenticators = Authenticator.objects.bulk_users_have_2fa([i.id for i in item_list])
  56 
  57         data = {}
  58         for item in item_list:
  59             data[item] = {
  60                 "avatar": avatars.get(item.id),
  61                 "identities": identities.get(item.id),
  62                 "has2fa": authenticators[item.id],
  63                 "emails": emails[item.id],
  64             }
  65         return data

  67     def serialize(self, obj, attrs, user):

  69 
  70         d = {
  71             "id": six.text_type(obj.id),
  72             "name": obj.get_display_name(),
  73             "username": obj.username,
  74             "email": obj.email,
  75             "avatarUrl": get_gravatar_url(obj.email, size=32),
  76             "isActive": obj.is_active,
  77             "hasPasswordAuth": obj.password not in ("!", ""),
  78             "isManaged": obj.is_managed,
  79             "dateJoined": obj.date_joined,
  80             "lastLogin": obj.last_login,
  81             "has2fa": attrs["has2fa"],
  82             "lastActive": obj.last_active,
  83             "isSuperuser": obj.is_superuser,
  84             "isStaff": obj.is_staff,
  85             "experiments": experiment_assignments,
  86         }
  87 
  88         if obj == user:
  89             options = {
  90                 o.key: o.value for o in UserOption.objects.filter(user=user, project__isnull=True)
  91             }
  92             stacktrace_order = int(options.get("stacktrace_order", -1) or -1)
  93 
  94             d["options"] = {
  95                 "language": options.get("language") or "en",
  96                 "stacktraceOrder": stacktrace_order,
  97                 "timezone": options.get("timezone") or settings.SENTRY_DEFAULT_TIME_ZONE,
  98                 "clock24Hours": options.get("clock_24_hours") or False,
  99             }
 100 
 101             d["flags"] = {"newsletter_consent_prompt": bool(obj.flags.newsletter_consent_prompt)}
 102 
 103         if attrs.get("avatar"):
 104             avatar = {
 105                 "avatarType": attrs["avatar"].get_avatar_type_display(),
 106                 "avatarUuid": attrs["avatar"].ident if attrs["avatar"].file_id else None,
 107             }
 108         else:
 109             avatar = {"avatarType": "letter_avatar", "avatarUuid": None}
 110         d["avatar"] = avatar
 111 
 112         # TODO(dcramer): move this to DetailedUserSerializer
 113         if attrs["identities"] is not None:
 114             d["identities"] = [
 115                 {
 116                     "id": six.text_type(i.id),
 117                     "name": i.ident,
 118                     "organization": {
 119                         "slug": i.auth_provider.organization.slug,
 120                         "name": i.auth_provider.organization.name,
 121                     },
 122                     "provider": {

 125                     },
 126                     "dateSynced": i.last_synced,
 127                     "dateVerified": i.last_verified,
 128                 }
 129                 for i in attrs["identities"]
 130             ]
 131 
 132         d["emails"] = [
 133             {"id": six.text_type(e.id), "email": e.email, "is_verified": e.is_verified}
 134             for e in attrs["emails"]
 135         ]
 136 
 137         return d

 140 class DetailedUserSerializer(UserSerializer):
 141     def get_attrs(self, item_list, user):
 142         attrs = super(DetailedUserSerializer, self).get_attrs(item_list, user)
 143 
 144         # ignore things that aren't user controlled (like recovery codes)
 145         authenticators = manytoone_to_dict(
 146             Authenticator.objects.filter(user__in=item_list),
 147             "user_id",
 148             lambda x: not x.interface.is_backup_interface,
 149         )
 150 
 151         permissions = manytoone_to_dict(
 152             UserPermission.objects.filter(user__in=item_list), "user_id"
 153         )
 154 
 155         memberships = manytoone_to_dict(
 156             OrganizationMember.objects.filter(
 157                 user__in=item_list, organization__status=OrganizationStatus.VISIBLE
 158             ),
 159             "user_id",
 160         )
 161 
 162         for item in item_list:
 163             attrs[item]["authenticators"] = authenticators[item.id]
 164             attrs[item]["permissions"] = permissions[item.id]
 165 
 166             # org can reset 2FA if the user is only in one org
 167             attrs[item]["canReset2fa"] = len(memberships[item.id]) == 1
 168 
 169         return attrs
 170 
 171     def serialize(self, obj, attrs, user):
 172         d = super(DetailedUserSerializer, self).serialize(obj, attrs, user)
 173         # XXX(dcramer): we dont use is_active_superuser here as we simply
 174         # want to tell the UI that we're an authenticated superuser, and
 175         # for requests that require an *active* session, they should prompt
 176         # on-demand. This ensures things like links to the Sentry admin can
 177         # still easily be rendered.
 178         d["isSuperuser"] = obj.is_superuser
 179         d["permissions"] = [up.permission for up in attrs["permissions"]]
 180         d["authenticators"] = [
 181             {
 182                 "id": six.text_type(a.id),
 183                 "type": a.interface.interface_id,
 184                 "name": six.text_type(a.interface.name),
 185                 "dateCreated": a.created_at,
 186                 "dateUsed": a.last_used_at,
 187             }
 188             for a in attrs["authenticators"]
 189         ]
 190         d["canReset2fa"] = attrs["canReset2fa"]
 191         return d
