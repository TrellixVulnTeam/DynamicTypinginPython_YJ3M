  42 class OrganizationSerializer(Serializer):
  43     def get_attrs(self, item_list, user):
  44         avatars = {
  45             a.organization_id: a
  46             for a in OrganizationAvatar.objects.filter(organization__in=item_list)
  47         }
  48         data = {}
  49         for item in item_list:
  50             data[item] = {"avatar": avatars.get(item.id)}
  51         return data
  52 
  53     def serialize(self, obj, attrs, user):
  54         from sentry import features
  55         from sentry.features.base import OrganizationFeature
  56 
  57         if attrs.get("avatar"):
  58             avatar = {
  59                 "avatarType": attrs["avatar"].get_avatar_type_display(),
  60                 "avatarUuid": attrs["avatar"].ident if attrs["avatar"].file_id else None,
  61             }
  62         else:
  63             avatar = {"avatarType": "letter_avatar", "avatarUuid": None}
  64 
  65         status = OrganizationStatus(obj.status)
  66 
  67         # Retrieve all registered organization features
  68         org_features = features.all(feature_type=OrganizationFeature).keys()
  69         feature_list = set()
  70 
  71         for feature_name in org_features:
  72             if not feature_name.startswith("organizations:"):
  73                 continue
  74             if features.has(feature_name, obj, actor=user):
  75                 # Remove the organization scope prefix
  76                 feature_list.add(feature_name[len("organizations:") :])
  77 
  78         # Do not include the onboarding feature if OrganizationOptions exist
  79         if (
  80             "onboarding" in feature_list
  81             and OrganizationOption.objects.filter(organization=obj).exists()
  82         ):
  83             feature_list.remove("onboarding")
  84 
  85         # Include api-keys feature if they previously had any api-keys
  86         if "api-keys" not in feature_list and ApiKey.objects.filter(organization=obj).exists():
  87             feature_list.add("api-keys")
  88 
  89         # Organization flag features (not provided through the features module)
  90         if OrganizationOption.objects.filter(
  91             organization=obj, key__in=LEGACY_RATE_LIMIT_OPTIONS
  92         ).exists():
  93             feature_list.add("legacy-rate-limits")
  94         if getattr(obj.flags, "allow_joinleave"):
  95             feature_list.add("open-membership")
  96         if not getattr(obj.flags, "disable_shared_issues"):
  97             feature_list.add("shared-issues")
  98 
  99         return {
 100             "id": six.text_type(obj.id),
 101             "slug": obj.slug,
 102             "status": {"id": status.name.lower(), "name": status.label},
 103             "name": obj.name or obj.slug,
 104             "dateCreated": obj.date_added,
 105             "isEarlyAdopter": bool(obj.flags.early_adopter),
 106             "require2FA": bool(obj.flags.require_2fa),
 107             "avatar": avatar,
 108             "features": feature_list,
 109         }

 112 class OnboardingTasksSerializer(Serializer):
 113     def get_attrs(self, item_list, user, **kwargs):
 114         # Unique user list
 115         users = {item.user for item in item_list if item.user}
 116         serialized_users = serialize(users, user, UserSerializer())
 117         user_map = {user["id"]: user for user in serialized_users}
 118 
 119         data = {}
 120         for item in item_list:
 121             data[item] = {"user": user_map.get(six.text_type(item.user_id))}
 122         return data
 123 
 124     def serialize(self, obj, attrs, user):
 125         return {
 126             "task": OrganizationOnboardingTask.TASK_KEY_MAP.get(obj.task),
 127             "status": OrganizationOnboardingTask.STATUS_KEY_MAP.get(obj.status),
 128             "user": attrs.get("user"),
 129             "completionSeen": obj.completion_seen,
 130             "dateCompleted": obj.date_completed,
 131             "data": obj.data,
 132         }

 135 class DetailedOrganizationSerializer(OrganizationSerializer):
 136     def get_attrs(self, item_list, user, **kwargs):
 137         return super(DetailedOrganizationSerializer, self).get_attrs(item_list, user)
 138 
 139     def serialize(self, obj, attrs, user, access):
 140         from sentry import experiments
 141 
 142         onboarding_tasks = list(
 143             OrganizationOnboardingTask.objects.filter(organization=obj).select_related("user")
 144         )
 145 
 146         experiment_assignments = experiments.all(org=obj, actor=user)
 147 
 148         context = super(DetailedOrganizationSerializer, self).serialize(obj, attrs, user)
 149         max_rate = quotas.get_maximum_quota(obj)
 150         context["experiments"] = experiment_assignments
 151         context["quota"] = {
 152             "maxRate": max_rate[0],
 153             "maxRateInterval": max_rate[1],
 154             "accountLimit": int(
 155                 OrganizationOption.objects.get_value(
 156                     organization=obj,
 157                     key="sentry:account-rate-limit",
 158                     default=ACCOUNT_RATE_LIMIT_DEFAULT,
 159                 )
 160             ),
 161             "projectLimit": int(
 162                 OrganizationOption.objects.get_value(
 163                     organization=obj,
 164                     key="sentry:project-rate-limit",
 165                     default=PROJECT_RATE_LIMIT_DEFAULT,
 166                 )
 167             ),
 168         }
 169 
 170         context.update(
 171             {
 172                 "isDefault": obj.is_default,
 173                 "defaultRole": obj.default_role,
 174                 "availableRoles": [{"id": r.id, "name": r.name} for r in roles.get_all()],
 175                 "openMembership": bool(obj.flags.allow_joinleave),
 176                 "require2FA": bool(obj.flags.require_2fa),
 177                 "allowSharedIssues": not obj.flags.disable_shared_issues,
 178                 "enhancedPrivacy": bool(obj.flags.enhanced_privacy),
 179                 "dataScrubber": bool(
 180                     obj.get_option("sentry:require_scrub_data", REQUIRE_SCRUB_DATA_DEFAULT)
 181                 ),
 182                 "dataScrubberDefaults": bool(
 183                     obj.get_option("sentry:require_scrub_defaults", REQUIRE_SCRUB_DEFAULTS_DEFAULT)
 184                 ),
 185                 "sensitiveFields": obj.get_option(
 186                     "sentry:sensitive_fields", SENSITIVE_FIELDS_DEFAULT
 187                 )
 188                 or [],
 189                 "safeFields": obj.get_option("sentry:safe_fields", SAFE_FIELDS_DEFAULT) or [],
 190                 "storeCrashReports": convert_crashreport_count(
 191                     obj.get_option("sentry:store_crash_reports")
 192                 ),
 193                 "attachmentsRole": six.text_type(
 194                     obj.get_option("sentry:attachments_role", ATTACHMENTS_ROLE_DEFAULT)
 195                 ),
 196                 "eventsMemberAdmin": bool(
 197                     obj.get_option("sentry:events_member_admin", EVENTS_MEMBER_ADMIN_DEFAULT)
 198                 ),
 199                 "scrubIPAddresses": bool(
 200                     obj.get_option(
 201                         "sentry:require_scrub_ip_address", REQUIRE_SCRUB_IP_ADDRESS_DEFAULT
 202                     )
 203                 ),
 204                 "scrapeJavaScript": bool(
 205                     obj.get_option("sentry:scrape_javascript", SCRAPE_JAVASCRIPT_DEFAULT)
 206                 ),
 207                 "trustedRelays": obj.get_option("sentry:trusted-relays", TRUSTED_RELAYS_DEFAULT)
 208                 or [],
 209                 "allowJoinRequests": bool(
 210                     obj.get_option("sentry:join_requests", JOIN_REQUESTS_DEFAULT)
 211                 ),
 212                 "relayPiiConfig": six.text_type(obj.get_option("sentry:relay_pii_config") or u"")
 213                 or None,
 214             }
 215         )
 216         context["access"] = access.scopes
 217         if access.role is not None:
 218             context["role"] = access.role
 219         context["pendingAccessRequests"] = OrganizationAccessRequest.objects.filter(
 220             team__organization=obj
 221         ).count()
 222         context["onboardingTasks"] = serialize(onboarding_tasks, user, OnboardingTasksSerializer())
 223         return context

 226 class DetailedOrganizationSerializerWithProjectsAndTeams(DetailedOrganizationSerializer):
 227     def get_attrs(self, item_list, user, **kwargs):
 228         return super(DetailedOrganizationSerializerWithProjectsAndTeams, self).get_attrs(
 229             item_list, user
 230         )
 231 
 232     def _project_list(self, organization, access):
 233         member_projects = list(access.projects)
 234         member_project_ids = [p.id for p in member_projects]
 235         other_projects = list(
 236             Project.objects.filter(organization=organization, status=ProjectStatus.VISIBLE).exclude(
 237                 id__in=member_project_ids
 238             )
 239         )
 240         project_list = sorted(other_projects + member_projects, key=lambda x: x.slug)
 241 
 242         for project in project_list:
 243             project._organization_cache = organization
 244         return project_list
 245 
 246     def _team_list(self, organization, access):
 247         member_teams = list(access.teams)
 248         member_team_ids = [p.id for p in member_teams]
 249         other_teams = list(
 250             Team.objects.filter(organization=organization, status=TeamStatus.VISIBLE).exclude(
 251                 id__in=member_team_ids
 252             )
 253         )
 254         team_list = sorted(other_teams + member_teams, key=lambda x: x.slug)
 255 
 256         for team in team_list:
 257             team._organization_cache = organization
 258         return team_list
 259 
 260     def serialize(self, obj, attrs, user, access):
 261         from sentry.api.serializers.models.project import ProjectSummarySerializer
 262         from sentry.api.serializers.models.team import TeamSerializer
 263 
 264         context = super(DetailedOrganizationSerializerWithProjectsAndTeams, self).serialize(
 265             obj, attrs, user, access
 266         )
 267 
 268         team_list = self._team_list(obj, access)
 269         project_list = self._project_list(obj, access)
 270 
 271         context["teams"] = serialize(team_list, user, TeamSerializer())
 272         context["projects"] = serialize(project_list, user, ProjectSummarySerializer())
 273 
 274         return context
