  71     def get_access_by_project(self, item_list, user):
  72         request = env.request
  73 
  74         project_teams = list(
  75             ProjectTeam.objects.filter(project__in=item_list).select_related("team")
  76         )
  77 
  78         project_team_map = defaultdict(list)
  79 
  80         for pt in project_teams:
  81             project_team_map[pt.project_id].append(pt.team)
  82 
  83         team_memberships = get_team_memberships([pt.team for pt in project_teams], user)
  84         org_roles = get_org_roles([i.organization_id for i in item_list], user)
  85 
  86         is_superuser = request and is_active_superuser(request) and request.user == user
  87         result = {}
  88         for project in item_list:
  89             is_member = any(t.id in team_memberships for t in project_team_map.get(project.id, []))
  90             org_role = org_roles.get(project.organization_id)
  91             if is_member:
  92                 has_access = True
  93             elif is_superuser:
  94                 has_access = True
  95             elif project.organization.flags.allow_joinleave:
  96                 has_access = True
  97             elif org_role and roles.get(org_role).is_global:
  98                 has_access = True
  99             else:
 100                 has_access = False
 101             result[project] = {"is_member": is_member, "has_access": has_access}
 102         return result

 104     def get_attrs(self, item_list, user):
 105         def measure_span(op_tag):
 106             span = sentry_sdk.start_span(op="serialize.get_attrs.project.{}".format(op_tag))
 107             span.set_data("Object Count", len(item_list))
 108             return span
 109 
 110         with measure_span("preamble"):
 111             project_ids = [i.id for i in item_list]
 112             if user.is_authenticated() and item_list:
 113                 bookmarks = set(
 114                     ProjectBookmark.objects.filter(
 115                         user=user, project_id__in=project_ids
 116                     ).values_list("project_id", flat=True)
 117                 )
 118                 user_options = {
 119                     (u.project_id, u.key): u.value
 120                     for u in UserOption.objects.filter(
 121                         Q(user=user, project__in=item_list, key="mail:alert")
 122                         | Q(user=user, key="subscribe_by_default", project__isnull=True)
 123                     )
 124                 }
 125                 default_subscribe = user_options.get("subscribe_by_default", "1") == "1"
 126             else:
 127                 bookmarks = set()
 128                 user_options = {}
 129                 default_subscribe = False
 130 
 131             if self.stats_period:
 132                 # we need to compute stats at 1d (1h resolution), and 14d
 133                 project_ids = [o.id for o in item_list]
 134 
 135                 segments, interval = STATS_PERIOD_CHOICES[self.stats_period]
 136                 now = timezone.now()
 137                 stats = tsdb.get_range(
 138                     model=tsdb.models.project,
 139                     keys=project_ids,
 140                     end=now,
 141                     start=now - ((segments - 1) * interval),
 142                     rollup=int(interval.total_seconds()),
 143                     environment_ids=self.environment_id and [self.environment_id],
 144                 )
 145             else:
 146                 stats = None
 147 
 148             avatars = {a.project_id: a for a in ProjectAvatar.objects.filter(project__in=item_list)}
 149             project_ids = [i.id for i in item_list]
 150             platforms = ProjectPlatform.objects.filter(project_id__in=project_ids).values_list(
 151                 "project_id", "platform"
 152             )
 153             platforms_by_project = defaultdict(list)
 154             for project_id, platform in platforms:
 155                 platforms_by_project[project_id].append(platform)
 156 
 157         with measure_span("access"):
 158             result = self.get_access_by_project(item_list, user)
 159 
 160         with measure_span("features"):
 161             features_by_project = self._get_features_for_projects(item_list, user)
 162             for project, serialized in result.items():
 163                 serialized["features"] = features_by_project[project]
 164 
 165         with measure_span("other"):
 166             for project, serialized in result.items():
 167                 serialized.update(
 168                     {
 169                         "is_bookmarked": project.id in bookmarks,
 170                         "is_subscribed": bool(
 171                             user_options.get((project.id, "mail:alert"), default_subscribe)
 172                         ),
 173                         "avatar": avatars.get(project.id),
 174                         "platforms": platforms_by_project[project.id],
 175                     }
 176                 )
 177                 if stats:
 178                     serialized["stats"] = stats[project.id]
 179         return result

 181     @staticmethod
 182     def _get_features_for_projects(all_projects, user):
 183         # Arrange to call features.has_for_batch rather than features.has
 184         # for performance's sake
 185         projects_by_org = defaultdict(list)
 186         for project in all_projects:
 187             projects_by_org[project.organization].append(project)
 188 
 189         features_by_project = defaultdict(list)
 190         for feature_name in features.all(feature_type=ProjectFeature).keys():
 191             if not feature_name.startswith(_PROJECT_SCOPE_PREFIX):
 192                 continue
 193             abbreviated_feature = feature_name[len(_PROJECT_SCOPE_PREFIX) :]
 194             for (organization, projects) in projects_by_org.items():
 195                 result = features.has_for_batch(feature_name, organization, projects, user)
 196                 for (project, flag) in result.items():
 197                     if flag:
 198                         features_by_project[project].append(abbreviated_feature)
 199 
 200         for project in all_projects:
 201             if project.flags.has_releases:
 202                 features_by_project[project].append("releases")
 203 
 204         return features_by_project

 291 class ProjectSummarySerializer(ProjectWithTeamSerializer):
 292     def get_attrs(self, item_list, user):
 293         attrs = super(ProjectSummarySerializer, self).get_attrs(item_list, user)
 294 
 295         projects_with_user_reports = set(
 296             UserReport.objects.filter(project_id__in=item_list).values_list("project", flat=True)
 297         )
 298 
 299         project_envs = (
 300             EnvironmentProject.objects.filter(
 301                 project_id__in=[i.id for i in item_list],
 302                 # Including the organization_id is necessary for postgres to use indexes
 303                 # efficiently.
 304                 environment__organization_id=item_list[0].organization_id,
 305             )
 306             .exclude(
 307                 is_hidden=True,
 308                 # HACK(lb): avoiding the no environment value
 309             )
 310             .exclude(environment__name="")
 311             .values("project_id", "environment__name")
 312         )
 313 
 314         environments_by_project = defaultdict(list)
 315         for project_env in project_envs:
 316             environments_by_project[project_env["project_id"]].append(
 317                 project_env["environment__name"]
 318             )
 319 
 320         cursor = connection.cursor()
 321         cursor.execute(
 322             """
 323             select srpe.project_id, se.name, sr.version, date_finished
 324             from (
 325                 select *
 326                 -- Finally, filter to the top row for each project/environment.
 327                 from (
 328                     -- Next we join to deploys and rank based recency of latest deploy for each project/environment.
 329                     select srpe.project_id, srpe.release_id, srpe.environment_id, sd.date_finished,
 330                     row_number() OVER (partition by (srpe.project_id, srpe.environment_id) order by sd.date_finished desc) row_num
 331                     from
 332                     (
 333                         -- First we fetch all related ReleaseProjectEnvironments, then filter to the x most recent for
 334                         -- each project/environment that actually have a deploy. This cuts out a lot of data volume
 335                         select *
 336                         from (
 337                             select *, row_number() OVER (partition by (srpe.project_id, srpe.environment_id) order by srpe.id desc) row_num
 338                             from sentry_releaseprojectenvironment srpe
 339                             where srpe.last_deploy_id is not null
 340                             and project_id = ANY(%s)
 341                         ) srpe
 342                         where row_num <= %s
 343                     ) srpe
 344                     inner join sentry_deploy sd on sd.id = srpe.last_deploy_id
 345                     where sd.date_finished is not null
 346                 ) srpe
 347                 where row_num = 1
 348             ) srpe
 349             inner join sentry_release sr on sr.id = srpe.release_id
 350             inner join sentry_environment se on se.id = srpe.environment_id;
 351             """,
 352             ([p.id for p in item_list], 10),
 353         )
 354         deploys_by_project = defaultdict(dict)
 355 
 356         for project_id, env_name, release_version, date_finished in cursor.fetchall():
 357             deploys_by_project[project_id][env_name] = {
 358                 "version": release_version,
 359                 "dateFinished": date_finished,
 360             }
 361 
 362         # We just return the version key here so that we cut down on response size
 363         latest_release_versions = {
 364             release.actual_project_id: {"version": release.version}
 365             for release in bulk_fetch_project_latest_releases(item_list)
 366         }
 367 
 368         for item in item_list:
 369             attrs[item]["latest_release"] = latest_release_versions.get(item.id)
 370             attrs[item]["deploys"] = deploys_by_project.get(item.id)
 371             attrs[item]["environments"] = environments_by_project.get(item.id, [])
 372             attrs[item]["has_user_reports"] = item.id in projects_with_user_reports
 373 
 374         return attrs

 402 def bulk_fetch_project_latest_releases(projects):
 403     """
 404     Fetches the latest release for each of the passed projects
 405     :param projects:
 406     :return: List of Releases, each with an additional `actual_project_id`
 407     attribute representing the project that they're the latest release for. If
 408     no release found, no entry will be returned for the given project.
 409     """
 410     # XXX: This query could be very inefficient for projects with a large
 411     # number of releases. To work around this, we only check 20 releases
 412     # ordered by highest release id, which is generally correlated with
 413     # most recent releases for a project. This could potentially result in
 414     # not having the correct most recent release, but in practice will
 415     # likely work fine.
 416     release_project_join_sql = """
 417         JOIN (
 418             SELECT *
 419             FROM sentry_release_project lrp
 420             WHERE lrp.project_id = p.id
 421             ORDER BY lrp.release_id DESC
 422             LIMIT 20
 423         ) lrp ON lrp.release_id = lrr.id
 424     """
 425 
 426     return list(
 427         Release.objects.raw(
 428             u"""
 429         SELECT lr.project_id as actual_project_id, r.*
 430         FROM (
 431             SELECT (
 432                 SELECT lrr.id
 433                 FROM sentry_release lrr
 434                 {}
 435                 WHERE lrp.project_id = p.id
 436                 ORDER BY COALESCE(lrr.date_released, lrr.date_added) DESC
 437                 LIMIT 1
 438             ) as release_id,
 439             p.id as project_id
 440             FROM sentry_project p
 441             WHERE p.id IN ({})
 442         ) as lr
 443         JOIN sentry_release r
 444         ON r.id = lr.release_id
 445         """.format(
 446                 release_project_join_sql, ", ".join(six.text_type(i.id) for i in projects)
 447             )
 448         )
 449     )
