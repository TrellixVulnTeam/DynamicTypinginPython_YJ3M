  17 def make_health_id(lookup, value):
  18     # Convert a lookup and value into
  19     # a string that can be used back in a request query.
  20     return "%s:%s" % (lookup.name, lookup.encoder(value))

  35 def geo_by_addr(ip):
  36     try:
  37         geo = _geo_by_addr(ip)
  38     except Exception:
  39         geo = None
  40 
  41     if not geo:
  42         return
  43 
  44     rv = {}
  45     for k in "country_code", "city", "region":
  46         d = geo.get(k)
  47         if isinstance(d, six.binary_type):
  48             d = d.decode("ISO-8859-1")
  49         rv[k] = d

  54 def serialize_eventusers(organization, item_list, user, lookup):
  55     if not item_list:
  56         return {}
  57 
  58     # We have no reliable way to map the tag value format
  59     # back into real EventUser rows. EventUser is only unique
  60     # per-project, and this is an organization aggregate.
  61     # This means a single value maps to multiple rows.
  62     filters = reduce(
  63         or_,
  64         [Q(hash=EventUser.hash_from_tag(tag), project_id=project) for tag, project in item_list],
  65     )
  66 
  67     eu_by_key = {(eu.tag_value, eu.project_id): eu for eu in EventUser.objects.filter(filters)}
  68 
  69     projects = serialize_projects(organization, {i[1] for i in item_list}, user)
  70 
  71     rv = {}
  72     for tag, project in item_list:
  73         eu = eu_by_key.get((tag, project))
  74         if eu is None:
  75             attr, value = tag.split(":", 1)
  76             eu = EventUser(project_id=project, **{EventUser.attr_from_keyword(attr): value})
  77         rv[(tag, project)] = {
  78             HEALTH_ID_KEY: make_health_id(lookup, [eu.tag_value, eu.project_id]),
  79             "value": {
  80                 "id": six.text_type(eu.id) if eu.id else None,
  81                 "project": projects.get(eu.project_id),
  82                 "hash": eu.hash,
  83                 "tagValue": eu.tag_value,
  84                 "identifier": eu.ident,
  85                 "username": eu.username,
  86                 "email": eu.email,
  87                 "ipAddress": eu.ip_address,
  88                 "dateCreated": eu.date_added,
  89                 "label": eu.get_label(),
  90                 "name": eu.get_display_name(),
  91                 "geo": geo_by_addr(eu.ip_address),
  92             },
  93         }
  94     return rv

 127 def zerofill(data, start, end, rollup):
 128     rv = []
 129     start = (int(to_timestamp(start)) // rollup) * rollup
 130     end = ((int(to_timestamp(end)) // rollup) * rollup) + rollup
 131     i = 0
 132     for key in six.moves.xrange(start, end, rollup):
 133         try:
 134             while data[i][0] < key:
 135                 rv.append(data[i])
 136                 i += 1
 137             if data[i][0] == key:
 138                 rv.append(data[i])
 139                 i += 1
 140                 continue
 141         except IndexError:
 142             pass
 143 
 144         rv.append((key, []))
 145     return rv

 148 class SnubaLookup(object):
 149     """
 150     A SnubaLookup consists of all of the attributes needed to facilitate making
 151     a query for a column in Snuba. This covers which columns are selected, the extra conditions
 152     that need to be applied, how values are serialized in/out of Snuba, etc.
 153     """
 154 
 155     __slots__ = (
 156         "name",
 157         "tagkey",
 158         "columns",
 159         "selected_columns",
 160         "conditions",
 161         "serializer",
 162         "encoder",
 163         "filter_key",
 164     )
 165     __registry = {}
 166 
 167     def __init__(
 168         self,
 169         name,
 170         tagkey=None,
 171         extra=None,
 172         selected_columns=None,
 173         conditions=None,
 174         serializer=serialize_noop,
 175         encoder=encoder_noop,
 176         filter_key=None,
 177     ):
 178         cls = type(self)
 179         assert name not in cls.__registry
 180         self.name = name
 181         self.tagkey = tagkey or name
 182         self.columns = [self.tagkey] + list(extra or [])
 183         self.serializer = partial(serializer, lookup=self)
 184         self.encoder = encoder
 185         self.conditions = conditions or [[self.tagkey, "IS NOT NULL", None]]
 186         self.selected_columns = selected_columns or []
 187         self.filter_key = filter_key or self.tagkey
 188         cls.__registry[name] = self
 189 
 190     @classmethod
 191     def get(cls, name):
 192         return cls.__registry[name]

 236 class BaseSnubaSerializer(object):
 237     def __init__(self, organization, lookup, user):
 238         self.organization = organization
 239         self.lookup = lookup
 240         self.user = user
 241 
 242     def get_attrs(self, item_list):
 243         if self.lookup is None:
 244             return item_list
 245 
 246         return self.lookup.serializer(self.organization, item_list, self.user)

 249 class SnubaResultSerializer(BaseSnubaSerializer):
 250     """
 251     Serializer for the top values Snuba results.
 252     """
 253 
 254     def serialize(self, result):
 255         counts_by_value = {
 256             value_from_row(r, self.lookup.columns): r["count"] for r in result.previous["data"]
 257         }
 258         projects = serialize_projects(
 259             self.organization,
 260             {p for r in result.current["data"] for p in r.get("top_projects", [])},
 261             self.user,
 262         )
 263         attrs = self.get_attrs(
 264             [value_from_row(r, self.lookup.columns) for r in result.current["data"]]
 265         )
 266 
 267         data = []
 268         for r in result.current["data"]:
 269             value = value_from_row(r, self.lookup.columns)
 270             row = {
 271                 "count": r["count"],
 272                 "lastCount": counts_by_value.get(value, 0),
 273                 self.lookup.name: attrs.get(value),
 274             }
 275             if "top_projects" in r:
 276                 row["topProjects"] = [projects[p] for p in r["top_projects"]]
 277             if "total_projects" in r:
 278                 row["totalProjects"] = r["total_projects"]
 279 
 280             data.append(row)
 281 
 282         return {
 283             "data": data,
 284             "totals": {
 285                 "count": result.current["totals"]["count"],
 286                 "lastCount": result.previous["totals"]["count"],
 287             },
 288         }

 291 class SnubaTSResultSerializer(BaseSnubaSerializer):
 292     """
 293     Serializer for time-series Snuba data.
 294     """
 295 
 296     def serialize(self, result, column="count", order=None):
 297         data = [
 298             (key, list(group))
 299             for key, group in itertools.groupby(result.data["data"], key=lambda r: r["time"])
 300         ]
 301         if self.lookup:
 302             attrs = self.get_attrs(
 303                 [value_from_row(r, self.lookup.columns) for _, v in data for r in v]
 304             )
 305         rv = []
 306         for k, v in data:
 307             row = []
 308             for r in v:
 309                 item = {"count": r.get(column, 0)}
 310                 if self.lookup:
 311                     value = value_from_row(r, self.lookup.columns)
 312                     item[self.lookup.name] = (attrs.get(value),)
 313                 row.append(item)
 314             rv.append((k, row))
 315 
 316         res = {"data": zerofill(rv, result.start, result.end, result.rollup)}
 317 
 318         if result.data.get("totals"):
 319             res["totals"] = {"count": result.data["totals"][column]}
 320         # If order is passed let that overwrite whats in data since its order for multi-axis
 321         if order is not None:
 322             res["order"] = order
 323         elif "order" in result.data:
 324             res["order"] = result.data["order"]
 325 
 326         return res
