  29         """
  30 
  31         if COMMIT_RANGE_DELIMITER in value:
  32             startCommit, endCommit = value.split(COMMIT_RANGE_DELIMITER)
  33

  35                 raise serializers.ValidationError(
  36                     "Commit cannot begin or end with %s" % COMMIT_RANGE_DELIMITER
  37                 )
  38 
  39             if len(startCommit) > MAX_COMMIT_LENGTH or len(endCommit) > MAX_COMMIT_LENGTH:
  40                 raise serializers.ValidationError(
  41                     "Start or end commit too long - max is %s chars each" % MAX_COMMIT_LENGTH
  42                 )

  44             return value
