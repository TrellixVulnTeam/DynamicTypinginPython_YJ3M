   9 def camel_to_snake_case(value):
  10     """
  11     Splits CamelCase and converts to lower case with underscores.
  12     """
  13     return re_camel_case.sub(r"_\1", value).strip("_").lower()

  16 def snake_to_camel_case(value):
  17     """
  18     Converts a string from snake_case to camelCase
  19     """
  20     words = value.strip("_").split("_")
  21     return words[0].lower() + "".join([word.capitalize() for word in words[1:]])

  24 def convert_dict_key_case(obj, converter):
  25     """
  26     Recursively converts the keys of a dictionary using the provided converter
  27     param.
  28     """
  29     if isinstance(obj, list):
  30         return [convert_dict_key_case(x, converter) for x in obj]
  31     elif not isinstance(obj, dict):
  32         return obj
  33 
  34     obj = obj.copy()
  35     for key in list(six.iterkeys(obj)):
  36         converted_key = converter(key)
  37         obj[converted_key] = convert_dict_key_case(obj.pop(key), converter)
  38 
  39     return obj

  42 class CamelSnakeSerializer(Serializer):
  43     """
  44     Allows parameters to be defined in snake case, but passed as camel case.
  45 
  46     Errors are output in camel case.
  47     """
  48 
  49     def __init__(self, instance=None, data=empty, **kwargs):
  50         if data is not empty:
  51             data = convert_dict_key_case(data, camel_to_snake_case)
  52         return super(CamelSnakeSerializer, self).__init__(instance=instance, data=data, **kwargs)
  53 
  54     @property
  55     def errors(self):
  56         errors = super(CamelSnakeSerializer, self).errors
  57         return convert_dict_key_case(errors, snake_to_camel_case)
