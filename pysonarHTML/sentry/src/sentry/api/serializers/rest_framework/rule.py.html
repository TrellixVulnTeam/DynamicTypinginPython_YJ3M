  15 class RuleNodeField(serializers.Field):
  16     def __init__(self, type):
  17         super(RuleNodeField, self).__init__()
  18         self.type_name = type
  19 
  20     def to_representation(self, value):
  21         return value

  23     def to_internal_value(self, data):
  24         if not isinstance(data, dict):
  25             msg = "Incorrect type. Expected a mapping, but got %s"
  26             raise ValidationError(msg % type(data).__name__)
  27 
  28         if "id" not in data:
  29             raise ValidationError("Missing attribute 'id'")
  30 
  31         cls = rules.get(data["id"], self.type_name)
  32         if cls is None:
  33             msg = "Invalid node. Could not find '%s'"
  34             raise ValidationError(msg % data["id"])
  35 
  36         node = cls(self.context["project"], data)
  37 
  38         if not node.form_cls:
  39             return data
  40 
  41         form = node.get_form_instance()
  42 
  43         if not form.is_valid():
  44             # XXX(epurkhiser): Very hacky, but we really just want validation
  45             # errors that are more specific, not just 'this wasn't filled out',
  46             # give a more generic error for those.
  47             first_error = next(six.itervalues(form.errors))[0]
  48 
  49             if first_error != "This field is required.":
  50                 raise ValidationError(first_error)
  51 
  52             raise ValidationError(
  53                 "Ensure at least one action is enabled and all required fields are filled in."
  54             )
  55 
  56         # Update data from cleaned form values
  57         data.update(form.cleaned_data)
  58 
  59         if getattr(form, "_pending_save", False):
  60             data["pending_save"] = True
  61         return data

  64 class RuleSerializer(serializers.Serializer):
  65     name = serializers.CharField(max_length=64)
  66     environment = serializers.CharField(max_length=64, required=False, allow_null=True)
  67     actionMatch = serializers.ChoiceField(
  68         choices=(("all", "all"), ("any", "any"), ("none", "none"))
  69     )
  70     actions = ListField(child=RuleNodeField(type="action/event"))
  71     conditions = ListField(child=RuleNodeField(type="condition/event"))
  72     frequency = serializers.IntegerField(min_value=5, max_value=60 * 24 * 30)
  73 
  74     def validate_environment(self, environment):
  75         if environment is None:
  76             return environment
  77 
  78         try:
  79             environment = Environment.get_for_organization_id(
  80                 self.context["project"].organization_id, environment
  81             ).id
  82         except Environment.DoesNotExist:
  83             raise serializers.ValidationError(u"This environment has not been created.")
  84 
  85         return environment
  86 
  87     def validate_conditions(self, value):
  88         if not value:
  89             raise serializers.ValidationError(u"Must select a condition")
  90         return value
  91 
  92     def validate_actions(self, value):
  93         if not value:
  94             raise serializers.ValidationError(u"Must select an action")
  95         return value
  96 
  97     def validate(self, attrs):
  98         # XXX(meredith): For rules that have the Slack integration as an action
  99         # we need to check if the channel_id needs to be looked up via an async task.
 100         # If the "pending_save" attribute is set we want to bubble that up to the
 101         # project_rule(_details) endpoints by setting it on attrs
 102         actions = attrs.get("actions", tuple())
 103         for action in actions:
 104             # remove this attribute because we don't want it to be saved in the rule
 105             if action.pop("pending_save", None):
 106                 attrs["pending_save"] = True
 107                 break
 108 
 109         return attrs
 110 
 111     def save(self, rule):
 112         rule.project = self.context["project"]
 113         if "environment" in self.validated_data:
 114             environment = self.validated_data["environment"]
 115             rule.environment_id = int(environment) if environment else environment
 116         if self.validated_data.get("name"):
 117             rule.label = self.validated_data["name"]
 118         if self.validated_data.get("actionMatch"):
 119             rule.data["action_match"] = self.validated_data["actionMatch"]
 120         if self.validated_data.get("actions") is not None:
 121             rule.data["actions"] = self.validated_data["actions"]
 122         if self.validated_data.get("conditions") is not None:
 123             rule.data["conditions"] = self.validated_data["conditions"]
 124         if self.validated_data.get("frequency"):
 125             rule.data["frequency"] = self.validated_data["frequency"]
 126         rule.save()
 127         return rule
