   9 def extract_user_ids_from_mentions(organization_id, mentions):
  10     """
  11     Extracts user ids from a set of mentions. Mentions should be a list of
  12     `Actor` instances. Returns a dictionary with 'users' and 'team_users' keys.
  13     'users' is the user ids for all explicitly mentioned users, and 'team_users'
  14     is all user ids from explicitly mentioned teams, excluding any already
  15     mentioned users.
  16     """
  17     actors = Actor.resolve_many(mentions)
  18     actor_mentions = seperate_resolved_actors(actors)
  19 
  20     mentioned_team_users = list(
  21         User.objects.get_from_teams(organization_id, actor_mentions["teams"])
  22         .exclude(id__in={u.id for u in actor_mentions["users"]})
  23         .values_list("id", flat=True)
  24     )
  25 
  26     return {
  27         "users": set([user.id for user in actor_mentions["users"]]),
  28         "team_users": set(mentioned_team_users),
  29     }

  32 def seperate_actors(actors):
  33     users = [actor for actor in actors if actor.type is User]
  34     teams = [actor for actor in actors if actor.type is Team]
  35 
  36     return {"users": users, "teams": teams}
  37

  39 def seperate_resolved_actors(actors):
  40     users = [actor for actor in actors if isinstance(actor, User)]
  41     teams = [actor for actor in actors if isinstance(actor, Team)]
  42 
  43     return {"users": users, "teams": teams}

  46 class MentionsMixin(object):
  47     def validate_mentions(self, mentions):
  48         if mentions and "projects" in self.context:
  49 
  50             seperated_actors = seperate_actors(mentions)
  51             # Validate that all mentioned users exist and are on the project.
  52             users = seperated_actors["users"]
  53 
  54             mentioned_user_ids = {user.id for user in users}
  55 
  56             projects = self.context["projects"]
  57             organization_id = self.context["organization_id"]
  58             users = User.objects.get_from_projects(organization_id, projects)
  59             user_ids = users.filter(id__in=mentioned_user_ids).values_list("id", flat=True)
  60 
  61             if len(mentioned_user_ids) > len(user_ids):
  62                 raise serializers.ValidationError("Cannot mention a non team member")
  63 
  64             # Validate that all mentioned teams exist and are on the project.
  65             teams = seperated_actors["teams"]
  66             mentioned_team_ids = {team.id for team in teams}
  67             if (
  68                 len(mentioned_team_ids)
  69                 > Team.objects.filter(
  70                     id__in=mentioned_team_ids, projectteam__project__in=projects
  71                 ).count()
  72             ):
  73                 raise serializers.ValidationError(
  74                     "Mentioned team not found or not associated with project"
  75                 )
  76 
  77         return mentions
