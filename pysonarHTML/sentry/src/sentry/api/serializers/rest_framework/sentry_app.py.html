  18 class ApiScopesField(serializers.Field):
  19     def to_internal_value(self, data):
  20         valid_scopes = ApiScopes()
  21 
  22         if data is None:
  23             return
  24 
  25         for scope in data:
  26             if scope not in valid_scopes:
  27                 raise ValidationError(u"{} not a valid scope".format(scope))
  28         return data

  31 class EventListField(serializers.Field):
  32     def to_internal_value(self, data):
  33         if data is None:
  34             return
  35 
  36         if not set(data).issubset(VALID_EVENT_RESOURCES):
  37             raise ValidationError(
  38                 u"Invalid event subscription: {}".format(
  39                     ", ".join(set(data).difference(VALID_EVENT_RESOURCES))
  40                 )
  41             )
  42         return data

  60 class URLField(serializers.URLField):
  61     def to_internal_value(self, url):
  62         # The Django URLField doesn't distinguish between different types of
  63         # invalid URLs, so do any manual checks here to give the User a better
  64         # error message.
  65         if url and not url.startswith("http"):
  66             raise ValidationError("URL must start with http[s]://")
  67         return url

  70 class SentryAppSerializer(Serializer):
  71     name = serializers.CharField()
  72     author = serializers.CharField(required=False, allow_null=True)
  73     scopes = ApiScopesField(allow_null=True)
  74     status = serializers.CharField(required=False, allow_null=True)
  75     events = EventListField(required=False, allow_null=True)
  76     schema = SchemaField(required=False, allow_null=True)
  77     webhookUrl = URLField(required=False, allow_null=True, allow_blank=True)
  78     redirectUrl = URLField(required=False, allow_null=True, allow_blank=True)
  79     isInternal = serializers.BooleanField(required=False, default=False)
  80     isAlertable = serializers.BooleanField(required=False, default=False)
  81     overview = serializers.CharField(required=False, allow_null=True)
  82     verifyInstall = serializers.BooleanField(required=False, default=True)
  83     allowedOrigins = ListField(child=serializers.CharField(max_length=255), required=False)
  84 
  85     def __init__(self, *args, **kwargs):
  86         self.access = kwargs["access"]
  87         del kwargs["access"]
  88         Serializer.__init__(self, *args, **kwargs)
  89 
  90     # an abstraction to pull fields from attrs if they are available or the sentry_app if not
  91     def get_current_value_wrapper(self, attrs):
  92         def get_current_value(field_name):
  93             if field_name in attrs:
  94                 return attrs[field_name]
  95             # params might be passed as camel case but we always store as snake case
  96             mapped_field_name = camel_to_snake_case(field_name)
  97             if hasattr(self.instance, mapped_field_name):
  98                 return getattr(self.instance, mapped_field_name)
  99             else:
 100                 return None
 101 
 102         return get_current_value
 103 
 104     def validate_name(self, value):
 105         max_length = 64 - UUID_CHARS_IN_SLUG - 1  # -1 comes from the - before the UUID bit
 106         if len(value) > max_length:
 107             raise ValidationError("Cannot exceed %d characters" % max_length)
 108         return value
 109 
 110     def validate_allowedOrigins(self, value):
 111         for allowed_origin in value:
 112             if "*" in allowed_origin:
 113                 raise ValidationError("'*' not allowed in origin")
 114         return value
 115 
 116     def validate_scopes(self, value):
 117         if not value:
 118             return value
 119 
 120         validation_errors = []
 121         for scope in value:
 122             # if the existing instance already has this scope, skip the check
 123             if self.instance and self.instance.has_scope(scope):
 124                 continue
 125             # add an error if the requester lacks permissions being requested
 126             if not self.access.has_scope(scope):
 127                 validation_errors.append(
 128                     "Requested permission of %s exceeds requester's permission. Please contact an administrator to make the requested change."
 129                     % (scope)
 130                 )
 131 
 132         if validation_errors:
 133             raise ValidationError(validation_errors)
 134 
 135         return value
 136 
 137     def validate(self, attrs):
 138         # validates events against scopes
 139         if attrs.get("scopes"):
 140             for resource in attrs.get("events", []):
 141                 needed_scope = REQUIRED_EVENT_PERMISSIONS[resource]
 142                 if needed_scope not in attrs["scopes"]:
 143                     raise ValidationError(
 144                         {
 145                             "events": u"{} webhooks require the {} permission.".format(
 146                                 resource, needed_scope
 147                             )
 148                         }
 149                     )
 150 
 151         get_current_value = self.get_current_value_wrapper(attrs)
 152         # validate if webhookUrl is missing that we don't have any webhook features enabled
 153         if not get_current_value("webhookUrl"):
 154             if get_current_value("isInternal"):
 155                 # for internal apps, make sure there aren't any events if webhookUrl is null
 156                 if get_current_value("events"):
 157                     raise ValidationError(
 158                         {"webhookUrl": "webhookUrl required if webhook events are enabled"}
 159                     )
 160                 # also check that we don't have the alert rule enabled
 161                 if get_current_value("isAlertable"):
 162                     raise ValidationError(
 163                         {"webhookUrl": "webhookUrl required if alert rule action is enabled"}
 164                     )
 165             else:
 166                 raise ValidationError({"webhookUrl": "webhookUrl required for public integrations"})
 167 
 168         # validate author for public integrations
 169         if not get_current_value("isInternal") and not get_current_value("author"):
 170             raise ValidationError({"author": "author required for public integrations"})
 171 
 172         return attrs
