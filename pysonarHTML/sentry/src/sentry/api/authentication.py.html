  21 class StandardAuthentication(QuietBasicAuthentication):
  22     token_name = None
  23 
  24     def authenticate(self, request):
  25         auth = get_authorization_header(request).split()
  26 
  27         if not auth or auth[0].lower() != self.token_name:
  28             return None
  29 
  30         if len(auth) == 1:
  31             msg = "Invalid token header. No credentials provided."
  32             raise AuthenticationFailed(msg)
  33         elif len(auth) > 2:
  34             msg = "Invalid token header. Token string should not contain spaces."
  35             raise AuthenticationFailed(msg)
  36 
  37         return self.authenticate_credentials(request, auth[1])

  40 class RelayAuthentication(BasicAuthentication):
  41     def authenticate(self, request):
  42         relay_id = get_header_relay_id(request)
  43         relay_sig = get_header_relay_signature(request)
  44         if not relay_id:
  45             raise AuthenticationFailed("Invalid relay ID")
  46         if not relay_sig:
  47             raise AuthenticationFailed("Missing relay signature")
  48         return self.authenticate_credentials(relay_id, relay_sig, request)
  49 
  50     def authenticate_credentials(self, relay_id, relay_sig, request):
  51         with configure_scope() as scope:
  52             scope.set_tag("relay_id", relay_id)
  53 
  54         try:
  55             relay = Relay.objects.get(relay_id=relay_id)
  56         except Relay.DoesNotExist:
  57             raise AuthenticationFailed("Unknown relay")
  58 
  59         try:
  60             data = relay.public_key_object.unpack(request.body, relay_sig, max_age=60 * 5)
  61             request.relay = relay
  62             request.relay_request_data = data
  63         except UnpackError:
  64             raise AuthenticationFailed("Invalid relay signature")
  65 
  66         # TODO(mitsuhiko): can we return the relay here?  would be nice if we
  67         # could find some common interface for it
  68         return (AnonymousUser(), None)

  71 class ApiKeyAuthentication(QuietBasicAuthentication):
  72     def authenticate_credentials(self, userid, password):
  73         if password:
  74             return None
  75 
  76         try:
  77             key = ApiKey.objects.get_from_cache(key=userid)
  78         except ApiKey.DoesNotExist:
  79             raise AuthenticationFailed("API key is not valid")
  80 
  81         if not key.is_active:
  82             raise AuthenticationFailed("Key is disabled")
  83 
  84         with configure_scope() as scope:
  85             scope.set_tag("api_key", key.id)
  86 
  87         return (AnonymousUser(), key)

  90 class ClientIdSecretAuthentication(QuietBasicAuthentication):
  91     """
  92     Authenticates a Sentry Application using its Client ID and Secret
  93 
  94     This will be the method by which we identify which Sentry Application is
  95     making the request, for any requests not scoped to an installation.
  96 
  97     For example, the request to exchange a Grant Code for an Api Token.
  98     """
  99 
 100     def authenticate(self, request):
 101         if not request.json_body:
 102             raise AuthenticationFailed("Invalid request")
 103 
 104         client_id = request.json_body.get("client_id")
 105         client_secret = request.json_body.get("client_secret")
 106 
 107         invalid_pair_error = AuthenticationFailed("Invalid Client ID / Secret pair")
 108 
 109         if not client_id or not client_secret:
 110             raise invalid_pair_error
 111 
 112         try:
 113             application = ApiApplication.objects.get(client_id=client_id)
 114         except ApiApplication.DoesNotExist:
 115             raise invalid_pair_error
 116 
 117         if not constant_time_compare(application.client_secret, client_secret):
 118             raise invalid_pair_error
 119 
 120         try:
 121             return (application.sentry_app.proxy_user, None)
 122         except Exception:
 123             raise invalid_pair_error

 126 class TokenAuthentication(StandardAuthentication):
 127     token_name = b"bearer"
 128 
 129     def authenticate_credentials(self, request, token_str):
 130         token = SystemToken.from_request(request, token_str)
 131         try:
 132             token = (
 133                 token
 134                 or ApiToken.objects.filter(token=token_str)
 135                 .select_related("user", "application")
 136                 .get()
 137             )
 138         except ApiToken.DoesNotExist:
 139             raise AuthenticationFailed("Invalid token")
 140 
 141         if token.is_expired():
 142             raise AuthenticationFailed("Token expired")
 143 
 144         if not token.user.is_active:
 145             raise AuthenticationFailed("User inactive or deleted")
 146 
 147         if token.application and not token.application.is_active:
 148             raise AuthenticationFailed("UserApplication inactive or deleted")
 149 
 150         with configure_scope() as scope:
 151             scope.set_tag("api_token_type", self.token_name)
 152             scope.set_tag("api_token", token.id)
 153 
 154         return (token.user, token)

 157 class DSNAuthentication(StandardAuthentication):
 158     token_name = b"dsn"
 159 
 160     def authenticate_credentials(self, request, token):
 161         try:
 162             key = ProjectKey.from_dsn(token)
 163         except ProjectKey.DoesNotExist:
 164             raise AuthenticationFailed("Invalid token")
 165 
 166         if not key.is_active:
 167             raise AuthenticationFailed("Invalid token")
 168 
 169         with configure_scope() as scope:
 170             scope.set_tag("api_token_type", self.token_name)
 171             scope.set_tag("api_project_key", key.id)
 172 
 173         return (AnonymousUser(), key)
