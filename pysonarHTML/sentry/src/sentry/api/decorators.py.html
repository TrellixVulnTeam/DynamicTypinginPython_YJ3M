   9 def is_considered_sudo(request):
  10     # Users without a password are assumed to always have sudo powers
  11     user = request.user
  12 
  13     return (
  14         request.is_sudo()
  15         or isinstance(request.auth, ApiKey)
  16         or isinstance(request.auth, ApiToken)
  17         or user.is_authenticated()
  18         and not user.has_usable_password()
  19     )
  20

  22 def sudo_required(func):
  23     @wraps(func)
  24     def wrapped(self, request, *args, **kwargs):
  25         # If we are already authenticated through an API key we do not
  26         # care about the sudo flag.
  27         if not is_considered_sudo(request):
  28             # TODO(dcramer): support some kind of auth flow to allow this
  29             # externally
  30             raise SudoRequired(request.user)
  31         return func(self, request, *args, **kwargs)
  32 
  33     return wrapped
