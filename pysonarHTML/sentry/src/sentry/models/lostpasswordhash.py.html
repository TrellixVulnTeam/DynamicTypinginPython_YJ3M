  73     def for_user(cls, user):
  74         # NOTE(mattrobenolt): Some security people suggest we invalidate
  75         # existing password hashes, but this opens up the possibility
  76         # of a DoS vector where then password resets are continually
  77         # requested, thus preventing someone from actually resetting
  78         # their password.
  79         # See: https://github.com/getsentry/sentry/pull/17299
  80         password_hash, created = cls.objects.get_or_create(user=user)
  81         if not password_hash.is_valid():
  82             password_hash.date_added = timezone.now()
  83             password_hash.set_hash()
  84             password_hash.save()
  85 
  86         return password_hash
