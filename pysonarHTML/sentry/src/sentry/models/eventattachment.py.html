  14 def get_crashreport_key(group_id):
  15     """
  16     Returns the ``django.core.cache`` key for groups that have exceeded their
  17     configured crash report limit.
  18     """
  19     return u"cr:%s" % (group_id,)

  22 class EventAttachment(Model):
  23     __core__ = False
  24 
  25     project_id = BoundedBigIntegerField()
  26     group_id = BoundedBigIntegerField(null=True, db_index=True)
  27     event_id = models.CharField(max_length=32, db_index=True)
  28     file = FlexibleForeignKey("sentry.File")
  29     name = models.TextField()
  30     date_added = models.DateTimeField(default=timezone.now, db_index=True)
  31 
  32     class Meta:
  33         app_label = "sentry"
  34         db_table = "sentry_eventattachment"
  35         index_together = (("project_id", "date_added"), ("project_id", "date_added", "file"))
  36         unique_together = (("project_id", "event_id", "file"),)
  37 
  38     __repr__ = sane_repr("event_id", "name", "file_id")
  39 
  40     def delete(self, *args, **kwargs):
  41         super(EventAttachment, self).delete(*args, **kwargs)
  42 
  43         # Always prune the group cache. Even if there are more crash reports
  44         # stored than the now configured limit, the cache will be repopulated
  45         # with the next incoming crash report.
  46         cache.delete(get_crashreport_key(self.group_id))
  47 
  48         self.file.delete()
