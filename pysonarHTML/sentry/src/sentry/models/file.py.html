  49 def _get_size_and_checksum(fileobj, logger=nooplogger):
  50     logger.debug("_get_size_and_checksum.start")
  51     size = 0
  52     checksum = sha1()
  53     while True:
  54         chunk = fileobj.read(65536)
  55         if not chunk:
  56             break
  57         size += len(chunk)
  58         checksum.update(chunk)
  59 
  60     logger.debug("_get_size_and_checksum.end")
  61     return size, checksum.hexdigest()

  64 @contextmanager
  65 def _locked_blob(checksum, logger=nooplogger):
  66     logger.debug("_locked_blob.start", extra={"checksum": checksum})
  67     lock = locks.get(u"fileblob:upload:{}".format(checksum), duration=UPLOAD_RETRY_TIME)
  68     with TimedRetryPolicy(UPLOAD_RETRY_TIME, metric_instance="lock.fileblob.upload")(lock.acquire):
  69         logger.debug("_locked_blob.acquired", extra={"checksum": checksum})
  70         # test for presence
  71         try:
  72             existing = FileBlob.objects.get(checksum=checksum)
  73         except FileBlob.DoesNotExist:
  74             existing = None
  75         yield existing
  76     logger.debug("_locked_blob.end", extra={"checksum": checksum})

 110     @classmethod
 111     def from_files(cls, files, organization=None, logger=nooplogger):
 112         """A faster version of `from_file` for multiple files at the time.
 113         If an organization is provided it will also create `FileBlobOwner`
 114         entries.  Files can be a list of files or tuples of file and checksum.
 115         If both are provided then a checksum check is performed.
 116 
 117         If the checksums mismatch an `IOError` is raised.
 118         """
 119         logger.debug("FileBlob.from_files.start")
 120 
 121         files_with_checksums = []
 122         for fileobj in files:
 123             if isinstance(fileobj, tuple):
 124                 files_with_checksums.append(fileobj)
 125             else:
 126                 files_with_checksums.append((fileobj, None))
 127 
 128         checksums_seen = set()
 129         blobs_created = []
 130         blobs_to_save = []
 131         locks = set()
 132         semaphore = Semaphore(value=MULTI_BLOB_UPLOAD_CONCURRENCY)
 133 
 134         def _upload_and_pend_chunk(fileobj, size, checksum, lock):
 135             logger.debug(
 136                 "FileBlob.from_files._upload_and_pend_chunk.start",
 137                 extra={"checksum": checksum, "size": size},
 138             )
 139             blob = cls(size=size, checksum=checksum)
 140             blob.path = cls.generate_unique_path()
 141             storage = get_storage()
 142             storage.save(blob.path, fileobj)
 143             blobs_to_save.append((blob, lock))
 144             metrics.timing("filestore.blob-size", size, tags={"function": "from_files"})
 145             logger.debug(
 146                 "FileBlob.from_files._upload_and_pend_chunk.end",
 147                 extra={"checksum": checksum, "path": blob.path},
 148             )
 149 
 150         def _ensure_blob_owned(blob):
 151             if organization is None:
 152                 return
 153             try:
 154                 with transaction.atomic():
 155                     FileBlobOwner.objects.create(organization=organization, blob=blob)
 156             except IntegrityError:
 157                 pass
 158 
 159         def _save_blob(blob):
 160             logger.debug("FileBlob.from_files._save_blob.start", extra={"path": blob.path})
 161             blob.save()
 162             _ensure_blob_owned(blob)
 163             logger.debug("FileBlob.from_files._save_blob.end", extra={"path": blob.path})
 164 
 165         def _flush_blobs():
 166             while True:
 167                 try:
 168                     blob, lock = blobs_to_save.pop()
 169                 except IndexError:
 170                     break
 171 
 172                 _save_blob(blob)
 173                 lock.__exit__(None, None, None)
 174                 locks.discard(lock)
 175                 semaphore.release()
 176 
 177         try:
 178             with ThreadPoolExecutor(max_workers=MULTI_BLOB_UPLOAD_CONCURRENCY) as exe:
 179                 for fileobj, reference_checksum in files_with_checksums:
 180                     logger.debug(
 181                         "FileBlob.from_files.executor_start", extra={"checksum": reference_checksum}
 182                     )
 183                     _flush_blobs()
 184 
 185                     # Before we go and do something with the files we calculate
 186                     # the checksums and compare it against the reference.  This
 187                     # also deduplicates duplicates uploaded in the same request.
 188                     # This is necessary because we acquire multiple locks in one
 189                     # go which would let us deadlock otherwise.
 190                     size, checksum = _get_size_and_checksum(fileobj)
 191                     if reference_checksum is not None and checksum != reference_checksum:
 192                         raise IOError("Checksum mismatch")
 193                     if checksum in checksums_seen:
 194                         continue
 195                     checksums_seen.add(checksum)
 196 
 197                     # Check if we need to lock the blob.  If we get a result back
 198                     # here it means the blob already exists.
 199                     lock = _locked_blob(checksum, logger=logger)
 200                     existing = lock.__enter__()
 201                     if existing is not None:
 202                         lock.__exit__(None, None, None)
 203                         blobs_created.append(existing)
 204                         _ensure_blob_owned(existing)
 205                         continue
 206 
 207                     # Remember the lock to force unlock all at the end if we
 208                     # encounter any difficulties.
 209                     locks.add(lock)
 210 
 211                     # Otherwise we leave the blob locked and submit the task.
 212                     # We use the semaphore to ensure we never schedule too
 213                     # many.  The upload will be done with a certain amount
 214                     # of concurrency controlled by the semaphore and the
 215                     # `_flush_blobs` call will take all those uploaded
 216                     # blobs and associate them with the database.
 217                     semaphore.acquire()
 218                     exe.submit(_upload_and_pend_chunk(fileobj, size, checksum, lock))
 219                     logger.debug("FileBlob.from_files.end", extra={"checksum": reference_checksum})
 220 
 221             _flush_blobs()
 222         finally:
 223             for lock in locks:
 224                 try:
 225                     lock.__exit__(None, None, None)
 226                 except Exception:
 227                     pass
 228             logger.debug("FileBlob.from_files.end")
 229 
 230     @classmethod
 231     def from_file(cls, fileobj, logger=nooplogger):
 232         """
 233         Retrieve a single FileBlob instances for the given file.
 234         """
 235         logger.debug("FileBlob.from_file.start")
 236 
 237         size, checksum = _get_size_and_checksum(fileobj)
 238 
 239         # TODO(dcramer): the database here is safe, but if this lock expires
 240         # and duplicate files are uploaded then we need to prune one
 241         with _locked_blob(checksum, logger=logger) as existing:
 242             if existing is not None:
 243                 return existing
 244 
 245             blob = cls(size=size, checksum=checksum)
 246             blob.path = cls.generate_unique_path()
 247             storage = get_storage()
 248             storage.save(blob.path, fileobj)
 249             blob.save()
 250 
 251         metrics.timing("filestore.blob-size", size)
 252         logger.debug("FileBlob.from_file.end")
 253         return blob

 255     @classmethod
 256     def generate_unique_path(cls):
 257         # We intentionally do not use checksums as path names to avoid concurrency issues
 258         # when we attempt concurrent uploads for any reason.
 259         uuid_hex = uuid4().hex
 260         pieces = [uuid_hex[:2], uuid_hex[2:6], uuid_hex[6:]]
 261         return u"/".join(pieces)

 263     def delete(self, *args, **kwargs):
 264         lock = locks.get(u"fileblob:upload:{}".format(self.checksum), duration=UPLOAD_RETRY_TIME)
 265         with TimedRetryPolicy(UPLOAD_RETRY_TIME, metric_instance="lock.fileblob.delete")(
 266             lock.acquire
 267         ):
 268             super(FileBlob, self).delete(*args, **kwargs)
 269         if self.path:
 270             self.deletefile(commit=False)
 271 
 272     def deletefile(self, commit=False):
 273         assert self.path
 274 
 275         # Defer this by 1 minute just to make sure
 276         # we avoid any transaction isolation where the
 277         # FileBlob row might still be visible by the
 278         # task before transaction is committed.
 279         delete_file_task.apply_async(
 280             kwargs={"path": self.path, "checksum": self.checksum}, countdown=60
 281         )
 282 
 283         self.path = None
 284 
 285         if commit:
 286             self.save()

 288     def getfile(self):

 290         Return a file-like object for this File's content.
 291 
 292         >>> with blob.getfile() as src, open('/tmp/localfile', 'wb') as dst:
 293         >>>     for chunk in src.chunks():
 294         >>>         dst.write(chunk)

 296         assert self.path

 299         return storage.open(self.path)

 333     def getfile(self, mode=None, prefetch=False):
 334         """Returns a file object.  By default the file is fetched on
 335         demand but if prefetch is enabled the file is fully prefetched
 336         into a tempfile before reading can happen.
 337         """
 338         impl = self._get_chunked_blob(mode, prefetch)
 339         return FileObj(impl, self.name)

 341     def save_to(self, path):
 342         """Fetches the file and emplaces it at a certain location.  The
 343         write is done atomically to a tempfile first and then moved over.
 344         If the directory does not exist it is created.
 345         """
 346         path = os.path.abspath(path)
 347         base = os.path.dirname(path)
 348         try:
 349             os.makedirs(base)
 350         except OSError:
 351             pass
 352 
 353         f = None
 354         try:
 355             f = self._get_chunked_blob(
 356                 prefetch=True, prefetch_to=base, delete=False
 357             ).detach_tempfile()
 358             os.rename(f.name, path)
 359             f.close()
 360             f = None
 361         finally:
 362             if f is not None:
 363                 f.close()
 364                 try:
 365                     os.remove(f.name)
 366                 except Exception:
 367                     pass

 369     def putfile(self, fileobj, blob_size=DEFAULT_BLOB_SIZE, commit=True, logger=nooplogger):
 370         """
 371         Save a fileobj into a number of chunks.
 372 
 373         Returns a list of `FileBlobIndex` items.
 374 
 375         >>> indexes = file.putfile(fileobj)
 376         """
 377         results = []
 378         offset = 0
 379         checksum = sha1(b"")
 380 
 381         while True:
 382             contents = fileobj.read(blob_size)
 383             if not contents:
 384                 break
 385             checksum.update(contents)
 386 
 387             blob_fileobj = ContentFile(contents)
 388             blob = FileBlob.from_file(blob_fileobj, logger=logger)
 389 
 390             results.append(FileBlobIndex.objects.create(file=self, blob=blob, offset=offset))
 391             offset += blob.size
 392         self.size = offset
 393         self.checksum = checksum.hexdigest()
 394         metrics.timing("filestore.file-size", offset)
 395         if commit:
 396             self.save()
 397         return results

 399     def assemble_from_file_blob_ids(self, file_blob_ids, checksum, commit=True):
 400         """
 401         This creates a file, from file blobs and returns a temp file with the
 402         contents.
 403         """
 404         tf = tempfile.NamedTemporaryFile()
 405         with transaction.atomic():
 406             file_blobs = FileBlob.objects.filter(id__in=file_blob_ids).all()
 407 
 408             # Ensure blobs are in the order and duplication as provided
 409             blobs_by_id = {blob.id: blob for blob in file_blobs}
 410             file_blobs = [blobs_by_id[blob_id] for blob_id in file_blob_ids]
 411 
 412             new_checksum = sha1(b"")
 413             offset = 0
 414             for blob in file_blobs:
 415                 FileBlobIndex.objects.create(file=self, blob=blob, offset=offset)
 416                 for chunk in blob.getfile().chunks():
 417                     new_checksum.update(chunk)
 418                     tf.write(chunk)
 419                 offset += blob.size
 420 
 421             self.size = offset
 422             self.checksum = new_checksum.hexdigest()
 423 
 424             if checksum != self.checksum:
 425                 raise AssembleChecksumMismatch("Checksum mismatch")
 426 
 427         metrics.timing("filestore.file-size", offset)
 428         if commit:
 429             self.save()
 430         tf.flush()
 431         tf.seek(0)
 432         return tf

 448 class ChunkedFileBlobIndexWrapper(object):
 449     def __init__(self, indexes, mode=None, prefetch=False, prefetch_to=None, delete=True):
 450         # eager load from database incase its a queryset
 451         self._indexes = list(indexes)
 452         self._curfile = None
 453         self._curidx = None
 454         if prefetch:
 455             self.prefetched = True
 456             self._prefetch(prefetch_to, delete)
 457         else:
 458             self.prefetched = False
 459         self.mode = mode
 460         self.open()
 461 
 462     def __enter__(self):
 463         return self
 464 
 465     def __exit__(self, exc_type, exc_value, tb):
 466         self.close()
 467 
 468     def detach_tempfile(self):
 469         if not self.prefetched:
 470             raise TypeError("Can only detech tempfiles in prefetch mode")
 471         rv = self._curfile
 472         self._curfile = None
 473         self.close()
 474         rv.seek(0)
 475         return rv
 476 
 477     def _nextidx(self):
 478         assert not self.prefetched, "this makes no sense"
 479         old_file = self._curfile
 480         try:
 481             try:
 482                 self._curidx = six.next(self._idxiter)
 483                 self._curfile = self._curidx.blob.getfile()
 484             except StopIteration:
 485                 self._curidx = None
 486                 self._curfile = None
 487         finally:
 488             if old_file is not None:
 489                 old_file.close()
 490 
 491     @property
 492     def size(self):
 493         return sum(i.blob.size for i in self._indexes)
 494 
 495     def open(self):
 496         self.closed = False
 497         self.seek(0)
 498 
 499     def _prefetch(self, prefetch_to=None, delete=True):
 500         size = self.size
 501         f = tempfile.NamedTemporaryFile(prefix="._prefetch-", dir=prefetch_to, delete=delete)
 502         if size == 0:
 503             self._curfile = f
 504             return
 505 
 506         # Zero out the file
 507         f.seek(size - 1)
 508         f.write(b"\x00")
 509         f.flush()
 510 
 511         mem = mmap.mmap(f.fileno(), size)
 512 
 513         def fetch_file(offset, getfile):
 514             with getfile() as sf:
 515                 while True:
 516                     chunk = sf.read(65535)
 517                     if not chunk:
 518                         break
 519                     mem[offset : offset + len(chunk)] = chunk
 520                     offset += len(chunk)
 521 
 522         with ThreadPoolExecutor(max_workers=4) as exe:
 523             for idx in self._indexes:
 524                 exe.submit(fetch_file, idx.offset, idx.blob.getfile)
 525 
 526         mem.flush()
 527         self._curfile = f
 528 
 529     def close(self):
 530         if self._curfile:
 531             self._curfile.close()
 532         self._curfile = None
 533         self._curidx = None
 534         self.closed = True
 535 
 536     def seek(self, pos):
 537         if self.closed:
 538             raise ValueError("I/O operation on closed file")
 539 
 540         if self.prefetched:
 541             return self._curfile.seek(pos)
 542 
 543         if pos < 0:
 544             raise IOError("Invalid argument")
 545         if pos == 0 and not self._indexes:
 546             # Empty file, there's no seeking to be done.
 547             return
 548 
 549         for n, idx in enumerate(self._indexes[::-1]):
 550             if idx.offset <= pos:
 551                 if idx != self._curidx:
 552                     self._idxiter = iter(self._indexes[-(n + 1) :])
 553                     self._nextidx()
 554                 break
 555         else:
 556             raise ValueError("Cannot seek to pos")
 557         self._curfile.seek(pos - self._curidx.offset)
 558 
 559     def tell(self):
 560         if self.closed:
 561             raise ValueError("I/O operation on closed file")
 562         if self.prefetched:
 563             return self._curfile.tell()
 564         if self._curfile is None:
 565             return self.size
 566         return self._curidx.offset + self._curfile.tell()
 567 
 568     def read(self, n=-1):
 569         if self.closed:
 570             raise ValueError("I/O operation on closed file")
 571 
 572         if self.prefetched:
 573             return self._curfile.read(n)
 574 
 575         result = bytearray()
 576 
 577         # Read to the end of the file
 578         if n < 0:
 579             while self._curfile is not None:
 580                 blob_result = self._curfile.read(32768)
 581                 if not blob_result:
 582                     self._nextidx()
 583                 else:
 584                     result.extend(blob_result)
 585 
 586         # Read until a certain number of bytes are read
 587         else:
 588             while n > 0 and self._curfile is not None:
 589                 blob_result = self._curfile.read(min(n, 32768))
 590                 if not blob_result:
 591                     self._nextidx()
 592                 else:
 593                     n -= len(blob_result)
 594                     result.extend(blob_result)
 595 
 596         return bytes(result)

 611 def clear_cached_files(cache_path):
 612     try:
 613         cache_folders = os.listdir(cache_path)
 614     except OSError:
 615         return
 616 
 617     cutoff = int(time.time()) - ONE_DAY_AND_A_HALF
 618 
 619     for cache_folder in cache_folders:
 620         cache_folder = os.path.join(cache_path, cache_folder)
 621         try:
 622             items = os.listdir(cache_folder)
 623         except OSError:
 624             continue
 625         for cached_file in items:
 626             cached_file = os.path.join(cache_folder, cached_file)
 627             try:
 628                 mtime = os.path.getmtime(cached_file)
 629             except OSError:
 630                 continue
 631             if mtime < cutoff:
 632                 try:
 633                     os.remove(cached_file)
 634                 except OSError:
 635                     pass
