  18 class SentryAppInstallationToken(Model):
  19     __core__ = False
  20 
  21     api_token = FlexibleForeignKey("sentry.ApiToken")
  22     sentry_app_installation = FlexibleForeignKey("sentry.SentryAppInstallation")
  23 
  24     class Meta:
  25         app_label = "sentry"
  26         db_table = "sentry_sentryappinstallationtoken"
  27         unique_together = (("sentry_app_installation", "api_token"),)
  28 
  29     @classmethod
  30     def has_organization_access(cls, token, organization):
  31         try:
  32             install_token = cls.objects.select_related("sentry_app_installation").get(
  33                 api_token=token
  34             )
  35         except cls.DoesNotExist:
  36             return False
  37 
  38         return install_token.sentry_app_installation.organization_id == organization.id
  39 
  40     @classmethod
  41     def get_projects(cls, token):
  42         try:
  43             install_token = cls.objects.select_related("sentry_app_installation").get(
  44                 api_token=token
  45             )
  46         except cls.DoesNotExist:
  47             return False
  48 
  49         return Project.objects.filter(
  50             organization_id=install_token.sentry_app_installation.organization_id
  51         )

  54 class SentryAppInstallation(ParanoidModel):
  55     __core__ = True
  56 
  57     sentry_app = FlexibleForeignKey("sentry.SentryApp", related_name="installations")
  58 
  59     # SentryApp's are installed and scoped to an Organization. They will have
  60     # access, defined by their scopes, to Teams, Projects, etc. under that
  61     # Organization, implicitly.
  62     organization = FlexibleForeignKey(
  63         "sentry.Organization", related_name="sentry_app_installations"
  64     )
  65 
  66     # Each installation has a Grant that the integration can exchange for an
  67     # Access Token.
  68     api_grant = models.OneToOneField(
  69         "sentry.ApiGrant",
  70         null=True,
  71         on_delete=models.SET_NULL,
  72         related_name="sentry_app_installation",
  73     )
  74 
  75     # Only use this token for public integrtions since each install has only token at a time
  76     # An installation gets an access token once the Grant has been exchanged,
  77     # and is updated when the token gets refreshed.
  78     #
  79     # Do NOT Use this token for internal integrations since there could be multiple
  80     # need to look at SentryAppInstallationToken which connects api_tokens to installations
  81     api_token = models.OneToOneField(
  82         "sentry.ApiToken",
  83         null=True,
  84         on_delete=models.SET_NULL,
  85         related_name="sentry_app_installation",
  86     )
  87 
  88     uuid = models.CharField(max_length=64, default=default_uuid)
  89 
  90     status = BoundedPositiveIntegerField(
  91         default=SentryAppInstallationStatus.PENDING,
  92         choices=SentryAppInstallationStatus.as_choices(),
  93         db_index=True,
  94     )
  95 
  96     date_added = models.DateTimeField(default=timezone.now)
  97     date_updated = models.DateTimeField(default=timezone.now)
  98 
  99     class Meta:
 100         app_label = "sentry"
 101         db_table = "sentry_sentryappinstallation"
 102 
 103     # Used when first creating an Installation to tell the serializer that the
 104     # grant code should be included in the serialization.
 105     is_new = False
 106 
 107     def save(self, *args, **kwargs):
 108         self.date_updated = timezone.now()
 109         return super(SentryAppInstallation, self).save(*args, **kwargs)
