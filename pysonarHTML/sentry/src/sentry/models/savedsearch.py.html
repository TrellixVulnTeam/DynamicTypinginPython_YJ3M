  20 class SavedSearch(Model):
  21     """
  22     A saved search query.
  23     """
  24 
  25     __core__ = True
  26     # TODO: Remove this column and rows where it's not null once we've
  27     # completely removed Sentry 9
  28     project = FlexibleForeignKey("sentry.Project", null=True)
  29     organization = FlexibleForeignKey("sentry.Organization", null=True)
  30     type = models.PositiveSmallIntegerField(default=SearchType.ISSUE.value, null=True)
  31     name = models.CharField(max_length=128)
  32     query = models.TextField()
  33     date_added = models.DateTimeField(default=timezone.now)
  34     # TODO: Remove this column once we've completely removed Sentry 9
  35     is_default = models.BooleanField(default=False)
  36     is_global = models.NullBooleanField(null=True, default=False, db_index=True)
  37     owner = FlexibleForeignKey("sentry.User", null=True)
  38 
  39     class Meta:
  40         app_label = "sentry"
  41         db_table = "sentry_savedsearch"
  42         # Note that we also have a partial unique constraint on:
  43         #   (organization_id, name, type) WHERE owner_id IS NULL
  44         #   (is_global, name) WHERE is_global
  45         unique_together = (
  46             ("project", "name"),
  47             # Each user can have one default search per org
  48             ("organization", "owner", "type"),
  49         )
  50 
  51     @property
  52     def is_pinned(self):
  53         if hasattr(self, "_is_pinned"):
  54             return self._is_pinned
  55         return self.owner is not None and self.organization is not None
  56 
  57     @is_pinned.setter
  58     def is_pinned(self, value):
  59         self._is_pinned = value
  60 
  61     @property
  62     def is_org_custom_search(self):
  63         return self.owner is None and self.organization is not None
  64 
  65     __repr__ = sane_repr("project_id", "name")
