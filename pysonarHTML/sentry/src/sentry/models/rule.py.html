  25 class Rule(Model):
  26     __core__ = True
  27 
  28     DEFAULT_ACTION_MATCH = "all"  # any, all
  29     DEFAULT_FREQUENCY = 30  # minutes
  30 
  31     project = FlexibleForeignKey("sentry.Project")
  32     environment_id = BoundedPositiveIntegerField(null=True)
  33     label = models.CharField(max_length=64)
  34     data = GzippedDictField()
  35     status = BoundedPositiveIntegerField(
  36         default=RuleStatus.ACTIVE,
  37         choices=((RuleStatus.ACTIVE, "Active"), (RuleStatus.INACTIVE, "Inactive")),
  38         db_index=True,
  39     )
  40 
  41     date_added = models.DateTimeField(default=timezone.now)
  42 
  43     objects = BaseManager(cache_fields=("pk",))
  44 
  45     class Meta:
  46         db_table = "sentry_rule"
  47         app_label = "sentry"
  48 
  49     __repr__ = sane_repr("project_id", "label")

  51     @classmethod
  52     def get_for_project(cls, project_id):
  53         cache_key = u"project:{}:rules".format(project_id)
  54         rules_list = cache.get(cache_key)
  55         if rules_list is None:
  56             rules_list = list(cls.objects.filter(project=project_id, status=RuleStatus.ACTIVE))
  57             cache.set(cache_key, rules_list, 60)
  58         return rules_list
  59 
  60     def delete(self, *args, **kwargs):
  61         rv = super(Rule, self).delete(*args, **kwargs)
  62         cache_key = u"project:{}:rules".format(self.project_id)
  63         cache.delete(cache_key)
  64         return rv
  65 
  66     def save(self, *args, **kwargs):
  67         rv = super(Rule, self).save(*args, **kwargs)
  68         cache_key = u"project:{}:rules".format(self.project_id)
  69         cache.delete(cache_key)
  70         return rv

  72     def get_audit_log_data(self):
