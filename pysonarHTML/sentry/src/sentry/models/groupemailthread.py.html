   9 class GroupEmailThread(Model):
  10     """
  11     Keep track of the original Message-Id that was sent
  12     unique per email destination and Group object.This allows
  13     the tracking of proper In-Reply-To and References headers
  14     for email threading.
  15     """
  16 
  17     __core__ = False
  18 
  19     email = models.EmailField(max_length=75)
  20     project = FlexibleForeignKey("sentry.Project", related_name="groupemail_set")
  21     group = FlexibleForeignKey("sentry.Group", related_name="groupemail_set")
  22     msgid = models.CharField(max_length=100)
  23     date = models.DateTimeField(default=timezone.now, db_index=True)
  24 
  25     objects = BaseManager()
  26 
  27     class Meta:
  28         app_label = "sentry"
  29         db_table = "sentry_groupemailthread"
  30         unique_together = (("email", "group"), ("email", "msgid"))
  31 
  32     __repr__ = sane_repr("email", "group_id", "msgid")
