  40     @classmethod
  41     def as_choices(cls):
  42         result = []
  43         for name, member in six.iteritems(cls.__members__):
  44             # an alias
  45             if name != member.name:
  46                 continue
  47             # realistically Enum shouldn't even creating these, but alas
  48             if name.startswith("_"):
  49                 continue
  50             result.append((member.value, member.label))
  51         return tuple(result)

  65     def get_for_user(self, user, scope=None, only_visible=True):
  66         """
  67         Returns a set of all organizations a user has access to.
  68         """
  69         from sentry.models import OrganizationMember
  70 
  71         if not user.is_authenticated():
  72             return []
  73 
  74         if settings.SENTRY_PUBLIC and scope is None:
  75             if only_visible:
  76                 return list(self.filter(status=OrganizationStatus.ACTIVE))
  77             else:
  78                 return list(self.filter())
  79 
  80         qs = OrganizationMember.objects.filter(user=user).select_related("organization")
  81         if only_visible:
  82             qs = qs.filter(organization__status=OrganizationStatus.ACTIVE)
  83 
  84         results = list(qs)
  85 
  86         if scope is not None:
  87             return [r.organization for r in results if scope in r.get_scopes()]
  88         return [r.organization for r in results]

 161     def save(self, *args, **kwargs):
 162         if not self.slug:
 163             lock = locks.get("slug:organization", duration=5)
 164             with TimedRetryPolicy(10)(lock.acquire):
 165                 slugify_instance(self, self.name, reserved=RESERVED_ORGANIZATION_SLUGS)
 166             super(Organization, self).save(*args, **kwargs)
 167         else:
 168             super(Organization, self).save(*args, **kwargs)

 221     def merge_to(from_org, to_org):
 222         from sentry.models import (
 223             ApiKey,
 224             AuditLogEntry,
 225             AuthProvider,
 226             Commit,
 227             OrganizationAvatar,
 228             OrganizationIntegration,
 229             OrganizationMember,
 230             OrganizationMemberTeam,
 231             Project,
 232             Release,
 233             ReleaseCommit,
 234             ReleaseEnvironment,
 235             ReleaseFile,
 236             ReleaseHeadCommit,
 237             Repository,
 238             Team,
 239             Environment,
 240         )
 241 
 242         for from_member in OrganizationMember.objects.filter(
 243             organization=from_org, user__isnull=False
 244         ):
 245             logger = logging.getLogger("sentry.merge")
 246             try:
 247                 to_member = OrganizationMember.objects.get(
 248                     organization=to_org, user=from_member.user
 249                 )
 250             except OrganizationMember.DoesNotExist:
 251                 from_member.update(organization=to_org)
 252                 to_member = from_member
 253             else:
 254                 qs = OrganizationMemberTeam.objects.filter(
 255                     organizationmember=from_member, is_active=True
 256                 ).select_related()
 257                 for omt in qs:
 258                     OrganizationMemberTeam.objects.create_or_update(
 259                         organizationmember=to_member, team=omt.team, defaults={"is_active": True}
 260                     )
 261             logger.info(
 262                 "user.migrate",
 263                 extra={
 264                     "instance_id": from_member.id,
 265                     "new_member_id": to_member.id,
 266                     "from_organization_id": from_org.id,
 267                     "to_organization_id": to_org.id,
 268                 },
 269             )
 270 
 271         for from_team in Team.objects.filter(organization=from_org):
 272             try:
 273                 with transaction.atomic():
 274                     from_team.update(organization=to_org)
 275             except IntegrityError:
 276                 slugify_instance(from_team, from_team.name, organization=to_org)
 277                 from_team.update(organization=to_org, slug=from_team.slug)
 278             logger.info(
 279                 "team.migrate",
 280                 extra={
 281                     "instance_id": from_team.id,
 282                     "new_slug": from_team.slug,
 283                     "from_organization_id": from_org.id,
 284                     "to_organization_id": to_org.id,
 285                 },
 286             )
 287 
 288         for from_project in Project.objects.filter(organization=from_org):
 289             try:
 290                 with transaction.atomic():
 291                     from_project.update(organization=to_org)
 292             except IntegrityError:
 293                 slugify_instance(
 294                     from_project,
 295                     from_project.name,
 296                     organization=to_org,
 297                     reserved=RESERVED_PROJECT_SLUGS,
 298                 )
 299                 from_project.update(organization=to_org, slug=from_project.slug)
 300             logger.info(
 301                 "project.migrate",
 302                 extra={
 303                     "instance_id": from_project.id,
 304                     "new_slug": from_project.slug,
 305                     "from_organization_id": from_org.id,
 306                     "to_organization_id": to_org.id,
 307                 },
 308             )
 309 
 310         # TODO(jess): update this when adding unique constraint
 311         # on version, organization for releases
 312         for from_release in Release.objects.filter(organization=from_org):
 313             try:
 314                 to_release = Release.objects.get(version=from_release.version, organization=to_org)
 315             except Release.DoesNotExist:
 316                 Release.objects.filter(id=from_release.id).update(organization=to_org)
 317             else:
 318                 Release.merge(to_release, [from_release])
 319             logger.info(
 320                 "release.migrate",
 321                 extra={
 322                     "instance_id": from_release.id,
 323                     "from_organization_id": from_org.id,
 324                     "to_organization_id": to_org.id,
 325                 },
 326             )
 327 
 328         def do_update(queryset, params):
 329             model_name = queryset.model.__name__.lower()
 330             try:
 331                 with transaction.atomic():
 332                     queryset.update(**params)
 333             except IntegrityError:
 334                 for instance in queryset:
 335                     try:
 336                         with transaction.atomic():
 337                             instance.update(**params)
 338                     except IntegrityError:
 339                         logger.info(
 340                             "{}.migrate-skipped".format(model_name),
 341                             extra={
 342                                 "from_organization_id": from_org.id,
 343                                 "to_organization_id": to_org.id,
 344                             },
 345                         )
 346                     else:
 347                         logger.info(
 348                             "{}.migrate".format(model_name),
 349                             extra={
 350                                 "instance_id": instance.id,
 351                                 "from_organization_id": from_org.id,
 352                                 "to_organization_id": to_org.id,
 353                             },
 354                         )
 355             else:
 356                 logger.info(
 357                     "{}.migrate".format(model_name),
 358                     extra={"from_organization_id": from_org.id, "to_organization_id": to_org.id},
 359                 )
 360 
 361         INST_MODEL_LIST = (
 362             AuthProvider,
 363             ApiKey,
 364             AuditLogEntry,
 365             OrganizationAvatar,
 366             OrganizationIntegration,
 367             ReleaseEnvironment,
 368             ReleaseFile,
 369         )
 370 
 371         ATTR_MODEL_LIST = (Commit, ReleaseCommit, ReleaseHeadCommit, Repository, Environment)
 372 
 373         for model in INST_MODEL_LIST:
 374             queryset = model.objects.filter(organization=from_org)
 375             do_update(queryset, {"organization": to_org})
 376 
 377         for model in ATTR_MODEL_LIST:
 378             queryset = model.objects.filter(organization_id=from_org.id)
 379             do_update(queryset, {"organization_id": to_org.id})

 397     def send_delete_confirmation(self, audit_log_entry, countdown):
 398         from sentry import options
 399         from sentry.utils.email import MessageBuilder
 400 
 401         owners = self.get_owners()
 402 
 403         context = {
 404             "organization": self,
 405             "audit_log_entry": audit_log_entry,
 406             "eta": timezone.now() + timedelta(seconds=countdown),
 407             "url": absolute_uri(reverse("sentry-restore-organization", args=[self.slug])),
 408         }
 409 
 410         MessageBuilder(
 411             subject="%sOrganization Queued for Deletion" % (options.get("mail.subject-prefix"),),
 412             template="sentry/emails/org_delete_confirm.txt",
 413             html_template="sentry/emails/org_delete_confirm.html",
 414             type="org.confirm_delete",
 415             context=context,
 416         ).send_async([o.email for o in owners])

 422     def handle_2fa_required(self, request):
 423         from sentry.models import ApiKey
 424         from sentry.tasks.auth import remove_2fa_non_compliant_members
 425 
 426         actor_id = request.user.id if request.user and request.user.is_authenticated() else None
 427         api_key_id = (
 428             request.auth.id
 429             if hasattr(request, "auth") and isinstance(request.auth, ApiKey)
 430             else None
 431         )
 432         ip_address = request.META["REMOTE_ADDR"]
 433 
 434         remove_2fa_non_compliant_members.delay(
 435             self.id, actor_id=actor_id, actor_key_id=api_key_id, ip_address=ip_address
 436         )
