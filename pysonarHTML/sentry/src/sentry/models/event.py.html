   9 class EventDict(CanonicalKeyDict):

  11     Creating an instance of this dictionary will send the event through basic
  12     (Rust-based) type/schema validation called "re-normalization".
  13 
  14     This is used as a wrapper type for `Event.data` such that creating an event
  15     object (or loading it from the DB) will ensure the data fits the type
  16     schema.
  17     """
  18 
  19     def __init__(self, data, skip_renormalization=False, **kwargs):
  20         is_renormalized = isinstance(data, EventDict) or (
  21             isinstance(data, NodeData) and isinstance(data.data, EventDict)
  22         )
  23 
  24         if not skip_renormalization and not is_renormalized:
  25             normalizer = StoreNormalizer(is_renormalize=True, enable_trimming=False)
  26             data = normalizer.normalize_event(dict(data))
  27 
  28         CanonicalKeyDict.__init__(self, data, **kwargs)
