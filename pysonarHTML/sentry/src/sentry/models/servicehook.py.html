  47 class ServiceHook(Model):

  49 
  50     guid = models.CharField(max_length=32, unique=True, null=True)
  51     # hooks may be bound to an api application, or simply registered by a user

  53     actor_id = BoundedPositiveIntegerField(db_index=True)

  59     status = BoundedPositiveIntegerField(

  61     )
  62     version = BoundedPositiveIntegerField(default=0, choices=((0, "0"),))
  63     date_added = models.DateTimeField(default=timezone.now)
