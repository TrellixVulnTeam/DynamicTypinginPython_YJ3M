  40 class ProjectDebugFileManager(BaseManager):
  41     def find_missing(self, checksums, project):
  42         if not checksums:
  43             return []
  44 
  45         checksums = [x.lower() for x in checksums]
  46         missing = set(checksums)
  47 
  48         found = ProjectDebugFile.objects.filter(
  49             file__checksum__in=checksums, project=project
  50         ).values("file__checksum")
  51 
  52         for values in found:
  53             missing.discard(values.values()[0])
  54 
  55         return sorted(missing)
  56 
  57     def find_by_checksums(self, checksums, project):
  58         if not checksums:
  59             return []
  60         checksums = [x.lower() for x in checksums]
  61         return ProjectDebugFile.objects.filter(file__checksum__in=checksums, project=project)
  62 
  63     def find_by_debug_ids(self, project, debug_ids, features=None):
  64         """Finds debug information files matching the given debug identifiers.
  65 
  66         If a set of features is specified, only files that satisfy all features
  67         will be returned. This does not apply to legacy debug files that were
  68         not tagged with features.
  69 
  70         Returns a dict of debug files keyed by their debug identifier.
  71         """
  72         features = frozenset(features) if features is not None else frozenset()
  73 
  74         difs = (
  75             ProjectDebugFile.objects.filter(project=project, debug_id__in=debug_ids)
  76             .select_related("file")
  77             .order_by("-id")
  78         )
  79 
  80         difs_by_id = {}
  81         for dif in difs:
  82             difs_by_id.setdefault(dif.debug_id, []).append(dif)
  83 
  84         rv = {}
  85         for debug_id, group in six.iteritems(difs_by_id):
  86             with_features = [dif for dif in group if "features" in (dif.data or ())]
  87 
  88             # In case we've never computed features for any of these files, we
  89             # just take the first one and assume that it matches.
  90             if not with_features:
  91                 rv[debug_id] = group[0]
  92                 continue
  93 
  94             # There's at least one file with computed features. Older files are
  95             # considered redundant and will be deleted. We search for the first
  96             # file matching the given feature set. This might not resolve if no
  97             # DIF matches the given feature set.
  98             for dif in with_features:
  99                 if dif.features >= features:
 100                     rv[debug_id] = dif
 101                     break
 102 
 103         return rv

 163 def clean_redundant_difs(project, debug_id):
 164     """Deletes redundant debug files from the database and file storage. A debug
 165     file is considered redundant if there is a newer file with the same debug
 166     identifier and the same or a superset of its features.
 167     """
 168     difs = (
 169         ProjectDebugFile.objects.filter(project=project, debug_id=debug_id)
 170         .select_related("file")
 171         .order_by("-id")
 172     )
 173 
 174     all_features = set()
 175     for i, dif in enumerate(difs):
 176         # We always keep the latest file. If it has no features, likely the
 177         # previous files did not have features either and will be removed, or we
 178         # keep both. Subsequent uploads will remove this file later.
 179         if i > 0 and dif.features <= all_features:
 180             dif.delete()
 181         else:
 182             all_features.update(dif.features)

 185 def create_dif_from_id(project, meta, fileobj=None, file=None):
 186     """This creates a mach dsym file or proguard mapping from the given
 187     debug id and open file object to a debug file.  This will not verify the
 188     debug id (intentionally so).  Use `detect_dif_from_path` to do that.
 189     """
 190     if meta.file_format == "proguard":
 191         object_name = "proguard-mapping"
 192     elif meta.file_format in ("macho", "elf", "pdb", "pe", "sourcebundle"):
 193         object_name = meta.name
 194     elif meta.file_format == "breakpad":
 195         object_name = meta.name[:-4] if meta.name.endswith(".sym") else meta.name
 196     else:
 197         raise TypeError("unknown dif type %r" % (meta.file_format,))
 198 
 199     if file is not None:
 200         checksum = file.checksum
 201     elif fileobj is not None:
 202         h = hashlib.sha1()
 203         while True:
 204             chunk = fileobj.read(16384)
 205             if not chunk:
 206                 break
 207             h.update(chunk)
 208         checksum = h.hexdigest()
 209         fileobj.seek(0, 0)
 210     else:
 211         raise RuntimeError("missing file object")
 212 
 213     dif = (
 214         ProjectDebugFile.objects.select_related("file")
 215         .filter(
 216             project=project, debug_id=meta.debug_id, file__checksum=checksum, data__isnull=False
 217         )
 218         .order_by("-id")
 219         .first()
 220     )
 221 
 222     if dif is not None:
 223         return dif, False
 224 
 225     if file is None:
 226         file = File.objects.create(
 227             name=meta.debug_id,
 228             type="project.dif",
 229             headers={"Content-Type": DIF_MIMETYPES[meta.file_format]},
 230         )
 231         file.putfile(fileobj)
 232     else:
 233         file.type = "project.dif"
 234         file.headers["Content-Type"] = DIF_MIMETYPES[meta.file_format]
 235         file.save()
 236 
 237     dif = ProjectDebugFile.objects.create(
 238         file=file,
 239         debug_id=meta.debug_id,
 240         code_id=meta.code_id,
 241         cpu_name=meta.arch,
 242         object_name=object_name,
 243         project=project,
 244         data=meta.data,
 245     )
 246 
 247     # The DIF we've just created might actually be removed here again. But since
 248     # this can happen at any time in near or distant future, we don't care and
 249     # assume a successful upload. The DIF will be reported to the uploader and
 250     # reprocessing can start.
 251     clean_redundant_difs(project, meta.debug_id)
 252 
 253     resolve_processing_issue(project=project, scope="native", object="dsym:%s" % meta.debug_id)
 254 
 255     return dif, True

 258 def _analyze_progard_filename(filename):
 259     match = _proguard_file_re.search(filename)
 260     if match is None:
 261         return None
 262 
 263     ident = match.group(1)
 264 
 265     try:
 266         return six.text_type(uuid.UUID(ident))
 267     except Exception:
 268         pass

 271 class DifMeta(object):
 272     def __init__(self, file_format, arch, debug_id, path, code_id=None, name=None, data=None):
 273         self.file_format = file_format
 274         self.arch = arch
 275         self.debug_id = debug_id
 276         self.code_id = code_id
 277         self.path = path
 278         self.data = data
 279 
 280         if name is not None:
 281             self.name = os.path.basename(name)
 282         elif path is not None:
 283             self.name = os.path.basename(path)
 284 
 285     @classmethod
 286     def from_object(cls, obj, path, name=None, debug_id=None):
 287         if debug_id is not None:
 288             try:
 289                 debug_id = normalize_debug_id(debug_id)
 290             except SymbolicError:
 291                 debug_id = None
 292 
 293         # Only allow overrides in the debug_id's age if the rest of the debug id
 294         # matches with what we determine from the object file. We generally
 295         # trust the server more than the client.
 296         obj_id = obj.debug_id
 297         if obj_id and debug_id and obj_id[:36] == debug_id[:36]:
 298             obj_id = debug_id
 299 
 300         return cls(
 301             file_format=obj.file_format,
 302             arch=obj.arch,
 303             debug_id=obj_id,
 304             code_id=obj.code_id,
 305             path=path,
 306             # TODO: Extract the object name from the object
 307             name=name,
 308             data={"type": obj.kind, "features": list(obj.features)},
 309         )
 310 
 311     @property
 312     def basename(self):
 313         return os.path.basename(self.path)

 316 def detect_dif_from_path(path, name=None, debug_id=None):
 317     """This detects which kind of dif(Debug Information File) the path
 318     provided is. It returns an array since an Archive can contain more than
 319     one Object.
 320     """
 321     # proguard files (proguard/UUID.txt) or
 322     # (proguard/mapping-UUID.txt).
 323     proguard_id = _analyze_progard_filename(path)
 324     if proguard_id is not None:
 325         data = {"features": ["mapping"]}
 326         return [
 327             DifMeta(
 328                 file_format="proguard",
 329                 arch="any",
 330                 debug_id=proguard_id,
 331                 code_id=None,
 332                 path=path,
 333                 name=name,
 334                 data=data,
 335             )
 336         ]
 337 
 338     # native debug information files (MachO, ELF or Breakpad)
 339     try:
 340         archive = Archive.open(path)
 341     except ObjectErrorUnsupportedObject as e:
 342         raise BadDif("Unsupported debug information file: %s" % e)
 343     except SymbolicError as e:
 344         logger.warning("dsymfile.bad-fat-object", exc_info=True)
 345         raise BadDif("Invalid debug information file: %s" % e)
 346     else:
 347         objs = []
 348         for obj in archive.iter_objects():
 349             objs.append(DifMeta.from_object(obj, path, name=name, debug_id=debug_id))
 350         return objs

 353 def create_debug_file_from_dif(to_create, project):
 354     """Create a ProjectDebugFile from a dif (Debug Information File) and
 355     return an array of created objects.
 356     """
 357     rv = []
 358     for meta in to_create:
 359         with open(meta.path, "rb") as f:
 360             dif, created = create_dif_from_id(project, meta, fileobj=f)
 361             if created:
 362                 rv.append(dif)
 363     return rv

 366 def create_files_from_dif_zip(fileobj, project):
 367     """Creates all missing debug files from the given zip file.  This
 368     returns a list of all files created.
 369     """
 370     scratchpad = tempfile.mkdtemp()
 371     try:
 372         safe_extract_zip(fileobj, scratchpad, strip_toplevel=False)
 373         to_create = []
 374 
 375         for dirpath, dirnames, filenames in os.walk(scratchpad):
 376             for fn in filenames:
 377                 fn = os.path.join(dirpath, fn)
 378                 try:
 379                     difs = detect_dif_from_path(fn)
 380                 except BadDif:
 381                     difs = None
 382 
 383                 if difs is None:
 384                     difs = []
 385                 to_create = to_create + difs
 386 
 387         rv = create_debug_file_from_dif(to_create, project)
 388 
 389         # Uploading new dsysm changes the reprocessing revision
 390         bump_reprocessing_revision(project)
 391 
 392         return rv
 393     finally:
 394         shutil.rmtree(scratchpad)

 397 class DIFCache(object):
 398     @property
 399     def cache_path(self):
 400         return options.get("dsym.cache-path")
 401 
 402     def get_project_path(self, project):
 403         return os.path.join(self.cache_path, six.text_type(project.id))
 404 
 405     def fetch_difs(self, project, debug_ids, features=None):
 406         """Given some ids returns an id to path mapping for where the
 407         debug symbol files are on the FS.
 408         """
 409         debug_ids = [six.text_type(debug_id).lower() for debug_id in debug_ids]
 410         difs = ProjectDebugFile.objects.find_by_debug_ids(project, debug_ids, features)
 411 
 412         rv = {}
 413         for debug_id, dif in six.iteritems(difs):
 414             dif_path = os.path.join(self.get_project_path(project), debug_id)
 415             try:
 416                 os.stat(dif_path)
 417             except OSError as e:
 418                 if e.errno != errno.ENOENT:
 419                     raise
 420                 dif.file.save_to(dif_path)
 421             rv[debug_id] = dif_path
 422 
 423         return rv
 424 
 425     def clear_old_entries(self):
 426         clear_cached_files(self.cache_path)
