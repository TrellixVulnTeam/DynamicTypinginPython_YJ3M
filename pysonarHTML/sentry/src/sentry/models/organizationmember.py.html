  78 class OrganizationMember(Model):
  79     """
  80     Identifies relationships between organizations and users.
  81 
  82     Users listed as team members are considered to have access to all projects
  83     and could be thought of as team owners (though their access level may not)
  84     be set to ownership.

 183     @property
 184     def token_expired(self):
 185         # Old tokens don't expire to preserve compatibility and not require
 186         # a backfill migration.
 187         if self.token_expires_at is None:
 188             return False
 189         if self.token_expires_at > timezone.now():
 190             return False
 191         return True

 193     @property
 194     def legacy_token(self):
 195         checksum = md5()
 196         checksum.update(six.text_type(self.organization_id).encode("utf-8"))
 197         checksum.update(self.get_email().encode("utf-8"))
 198         checksum.update(force_bytes(settings.SECRET_KEY))
 199         return checksum.hexdigest()

 204     def get_invite_link(self):

 214     def send_invite_email(self):
 215         from sentry.utils.email import MessageBuilder
 216 
 217         context = {
 218             "email": self.email,
 219             "organization": self.organization,
 220             "url": self.get_invite_link(),
 221         }
 222 
 223         msg = MessageBuilder(
 224             subject="Join %s in using Sentry" % self.organization.name,
 225             template="sentry/emails/member-invite.txt",
 226             html_template="sentry/emails/member-invite.html",
 227             type="organization.invite",
 228             context=context,
 229         )
 230 
 231         try:
 232             msg.send_async([self.get_email()])

 234             logger = get_logger(name="sentry.mail")

 258     def send_sso_unlink_email(self, actor, provider):
 259         from sentry.utils.email import MessageBuilder
 260         from sentry.models import LostPasswordHash
 261 
 262         email = self.get_email()
 263 
 264         recover_uri = u"{path}?{query}".format(
 265             path=reverse("sentry-account-recover"), query=urlencode({"email": email})
 266         )
 267 
 268         # Nothing to send if this member isn't associated to a user
 269         if not self.user_id:
 270             return
 271 
 272         context = {
 273             "email": email,
 274             "recover_url": absolute_uri(recover_uri),
 275             "has_password": self.user.password,
 276             "organization": self.organization,
 277             "actor": actor,
 278             "provider": provider,
 279         }
 280 
 281         if not self.user.password:
 282             password_hash = LostPasswordHash.for_user(self.user)
 283             context["set_password_url"] = password_hash.get_absolute_url(mode="set_password")
 284 
 285         msg = MessageBuilder(
 286             subject="Action Required for %s" % (self.organization.name,),
 287             template="sentry/emails/auth-sso-disabled.txt",
 288             html_template="sentry/emails/auth-sso-disabled.html",
 289             type="organization.auth_sso_disabled",
 290             context=context,
 291         )
 292         msg.send_async([email])

 314     def get_audit_log_data(self):
 315         from sentry.models import Team
 316 
 317         teams = list(
 318             Team.objects.filter(
 319                 id__in=OrganizationMemberTeam.objects.filter(
 320                     organizationmember=self, is_active=True
 321                 ).values_list("team", flat=True)
 322             ).values("id", "slug")
 323         )
 324 
 325         return {
 326             "email": self.get_email(),
 327             "user": self.user_id,
 328             "teams": [t["id"] for t in teams],
 329             "teams_slugs": [t["slug"] for t in teams],
 330             "has_global_access": self.has_global_access,
 331             "role": self.role,
 332             "invite_status": invite_status_names[self.invite_status],
 333         }

 335     def get_teams(self):

 341                 organizationmember=self, is_active=True

 345     def get_scopes(self):
 346         scopes = roles.get(self.role).scopes
 347 
 348         if self.role == "member" and not self.organization.get_option(
 349             "sentry:events_member_admin", EVENTS_MEMBER_ADMIN_DEFAULT
 350         ):
 351             scopes = frozenset(s for s in scopes if s != "event:admin")
 352 
 353         return scopes

 355     @classmethod
 356     def delete_expired(cls, threshold):
 357         """
 358         Delete un-accepted member invitations that expired
 359         ``threshold`` days ago.
 360         """
 361         cls.objects.filter(token_expires_at__lt=threshold, user_id__exact=None).exclude(
 362             email__exact=None
 363         ).delete()
