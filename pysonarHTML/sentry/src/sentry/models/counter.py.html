  27 def increment_project_counter(project, delta=1):
  28     """This method primarily exists so that south code can use it."""
  29     if delta <= 0:
  30         raise ValueError("There is only one way, and that's up.")
  31 
  32     cur = connection.cursor()
  33     try:
  34         cur.execute(
  35             """
  36             select sentry_increment_project_counter(%s, %s)
  37         """,
  38             [project.id, delta],
  39         )
  40         return cur.fetchone()[0]
  41     finally:
  42         cur.close()

  45 # this must be idempotent because it seems to execute twice
  46 # (at least during test runs)
  47 def create_counter_function(app_config, using, **kwargs):
  48     if app_config and app_config.name != "sentry":
  49         return
  50 
  51     try:
  52         app_config.get_model("Counter")
  53     except LookupError:
  54         return
  55 
  56     cursor = connections[using].cursor()
  57     try:
  58         cursor.execute(
  59             """
  60             create or replace function sentry_increment_project_counter(
  61                 project bigint, delta int) returns int as $$
  62             declare
  63             new_val int;
  64             begin
  65             loop
  66                 update sentry_projectcounter set value = value + delta
  67                 where project_id = project
  68                 returning value into new_val;
  69                 if found then
  70                 return new_val;
  71                 end if;
  72                 begin
  73                 insert into sentry_projectcounter(project_id, value)
  74                     values (project, delta)
  75                     returning value into new_val;
  76                 return new_val;
  77                 exception when unique_violation then
  78                 end;
  79             end loop;
  80             end
  81             $$ language plpgsql;
  82         """
  83         )
  84     finally:
  85         cursor.close()
