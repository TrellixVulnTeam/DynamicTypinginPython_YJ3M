  18 class ReleaseFile(Model):
  19     r"""
  20     A ReleaseFile is an association between a Release and a File.
  21 
  22     The ident of the file should be sha1(name) or
  23     sha1(name '\x00\x00' dist.name) and must be unique per release.
  24     """
  25     __core__ = False
  26 
  27     organization = FlexibleForeignKey("sentry.Organization")
  28     # DEPRECATED
  29     project_id = BoundedPositiveIntegerField(null=True)
  30     release = FlexibleForeignKey("sentry.Release")
  31     file = FlexibleForeignKey("sentry.File")
  32     ident = models.CharField(max_length=40)
  33     name = models.TextField()
  34     dist = FlexibleForeignKey("sentry.Distribution", null=True)
  35 
  36     __repr__ = sane_repr("release", "ident")
  37 
  38     class Meta:
  39         unique_together = (("release", "ident"),)
  40         index_together = (("release", "name"),)
  41         app_label = "sentry"
  42         db_table = "sentry_releasefile"

  44     def save(self, *args, **kwargs):
  45         if not self.ident and self.name:
  46             dist = self.dist_id and self.dist.name or None
  47             self.ident = type(self).get_ident(self.name, dist)
  48         return super(ReleaseFile, self).save(*args, **kwargs)
  49 
  50     def update(self, *args, **kwargs):
  51         # If our name is changing, we must also change the ident
  52         if "name" in kwargs and "ident" not in kwargs:
  53             dist = kwargs.get("dist") or self.dist
  54             kwargs["ident"] = self.ident = type(self).get_ident(
  55                 kwargs["name"], dist and dist.name or dist
  56             )
  57         return super(ReleaseFile, self).update(*args, **kwargs)
  58 
  59     @classmethod
  60     def get_ident(cls, name, dist=None):
  61         if dist is not None:
  62             return sha1_text(name + "\x00\x00" + dist).hexdigest()
  63         return sha1_text(name).hexdigest()

  65     @classmethod
  66     def normalize(cls, url):
  67         """Transforms a full absolute url into 2 or 4 generalized options
  68 
  69         * the original url as input
  70         * (optional) original url without querystring
  71         * the full url, but stripped of scheme and netloc
  72         * (optional) full url without scheme and netloc or querystring
  73         """
  74         # Always ignore the fragment
  75         scheme, netloc, path, query, _ = urlsplit(url)
  76 
  77         uri_without_fragment = (scheme, netloc, path, query, None)
  78         uri_relative = (None, None, path, query, None)
  79         uri_without_query = (scheme, netloc, path, None, None)
  80         uri_relative_without_query = (None, None, path, None, None)
  81         urls = [urlunsplit(uri_without_fragment)]
  82         if query:
  83             urls.append(urlunsplit(uri_without_query))
  84         urls.append("~" + urlunsplit(uri_relative))
  85         if query:
  86             urls.append("~" + urlunsplit(uri_relative_without_query))
  87         return urls

  90 class ReleaseFileCache(object):
  91     @property
  92     def cache_path(self):
  93         return options.get("releasefile.cache-path")
  94 
  95     def getfile(self, releasefile):
  96         cutoff = options.get("releasefile.cache-limit")
  97         file_size = releasefile.file.size
  98         if file_size < cutoff:
  99             metrics.timing("release_file.cache.get.size", file_size, tags={"cutoff": True})
 100             return releasefile.file.getfile()
 101 
 102         file_id = six.text_type(releasefile.file_id)
 103         organization_id = six.text_type(releasefile.organization_id)
 104         file_path = os.path.join(self.cache_path, organization_id, file_id)
 105 
 106         hit = True
 107         try:
 108             os.stat(file_path)
 109         except OSError as e:
 110             if e.errno != errno.ENOENT:
 111                 raise
 112             releasefile.file.save_to(file_path)
 113             hit = False
 114 
 115         metrics.timing("release_file.cache.get.size", file_size, tags={"hit": hit, "cutoff": False})
 116         return FileObj(open(file_path, "rb"))
 117 
 118     def clear_old_entries(self):
 119         clear_cached_files(self.cache_path)
 120

 122 ReleaseFile.cache = ReleaseFileCache()
