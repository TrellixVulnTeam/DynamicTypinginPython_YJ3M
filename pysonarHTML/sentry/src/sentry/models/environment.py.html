  46     @classmethod
  47     def is_valid_name(cls, value):
  48         """Limit length and reject problematic bytes
  49 
  50         If you change the rules here also update the event ingestion schema
  51         in sentry.interfaces.schemas
  52         """
  53         if len(value) > ENVIRONMENT_NAME_MAX_LENGTH:
  54             return False
  55         return OK_NAME_PATTERN.match(value) is not None

  65     @classmethod
  66     def get_for_organization_id(cls, organization_id, name):
  67         name = cls.get_name_or_default(name)
  68 
  69         cache_key = cls.get_cache_key(organization_id, name)
  70 
  71         env = cache.get(cache_key)
  72         if env is None:
  73             env = cls.objects.get(name=name, organization_id=organization_id)

  78     @classmethod
  79     def get_or_create(cls, project, name):
  80         with metrics.timer("models.environment.get_or_create") as metrics_tags:
  81             name = cls.get_name_or_default(name)
  82 
  83             cache_key = cls.get_cache_key(project.organization_id, name)
  84 
  85             env = cache.get(cache_key)
  86             if env is None:

  89                     0

  99     def add_project(self, project, is_hidden=None):
 100         cache_key = "envproj:c:%s:%s" % (self.id, project.id)
 101 
 102         if cache.get(cache_key) is None:
 103             try:
 104                 with transaction.atomic():
 105                     EnvironmentProject.objects.create(
 106                         project=project, environment=self, is_hidden=is_hidden
 107                     )
 108                 cache.set(cache_key, 1, 3600)
 109             except IntegrityError:
 110                 # We've already created the object, should still cache the action.
 111                 cache.set(cache_key, 1, 3600)

 113     @staticmethod
 114     def get_name_from_path_segment(segment):
 115         # In cases where the environment name is passed as a URL path segment,
 116         # the (case-sensitive) string "none" represents the empty string
 117         # environment name for historic reasons (see commit b09858f.) In all
 118         # other contexts (incl. request query string parameters), the empty
 119         # string should be used.
 120         return segment if segment != "none" else ""
