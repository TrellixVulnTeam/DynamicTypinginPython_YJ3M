  57 class AuthenticatorManager(BaseManager):
  58     def all_interfaces_for_user(self, user, return_missing=False, ignore_backup=False):
  59         """Returns a correctly sorted list of all interfaces the user
  60         has enabled.  If `return_missing` is set to `True` then all
  61         interfaces are returned even if not enabled.
  62         """
  63 
  64         def _sort(x):
  65             return sorted(x, key=lambda x: (x.type == 0, x.type))
  66 
  67         # Collect interfaces user is enrolled in
  68         ifaces = [
  69             x.interface
  70             for x in Authenticator.objects.filter(
  71                 user=user,
  72                 type__in=[a.type for a in available_authenticators(ignore_backup=ignore_backup)],
  73             )
  74         ]
  75 
  76         if return_missing:
  77             # Collect additional interfaces that the user
  78             # is not enrolled in
  79             rvm = dict(AUTHENTICATOR_INTERFACES)
  80             for iface in ifaces:
  81                 rvm.pop(iface.interface_id, None)
  82             for iface_cls in six.itervalues(rvm):
  83                 if iface_cls.is_available:
  84                     ifaces.append(iface_cls())
  85 
  86         return _sort(ifaces)
  87 
  88     def auto_add_recovery_codes(self, user, force=False):
  89         """This automatically adds the recovery code backup interface in
  90         case no backup interface is currently set for the user.  Returns
  91         the interface that was added.
  92         """
  93         has_authenticators = False
  94 
  95         # If we're not forcing, check for a backup interface already setup
  96         # or if it's missing, we'll need to set it.
  97         if not force:
  98             for authenticator in Authenticator.objects.filter(
  99                 user=user, type__in=[a.type for a in available_authenticators()]
 100             ):
 101                 iface = authenticator.interface
 102                 if iface.is_backup_interface:
 103                     return
 104                 has_authenticators = True
 105 
 106         if has_authenticators or force:
 107             interface = RecoveryCodeInterface()
 108             interface.enroll(user)
 109             return interface
 110 
 111     def get_interface(self, user, interface_id):
 112         """Looks up an interface by interface ID for a user.  If the
 113         interface is not available but configured a
 114         `Authenticator.DoesNotExist` will be raised just as if the
 115         authenticator was not configured at all.
 116         """
 117         interface = AUTHENTICATOR_INTERFACES.get(interface_id)
 118         if interface is None or not interface.is_available:
 119             raise LookupError("No such interface %r" % interface_id)
 120         try:
 121             return Authenticator.objects.get(user=user, type=interface.type).interface
 122         except Authenticator.DoesNotExist:
 123             return interface()
 124 
 125     def user_has_2fa(self, user):
 126         """Checks if the user has any 2FA configured.
 127         """
 128         return Authenticator.objects.filter(
 129             user=user, type__in=[a.type for a in available_authenticators(ignore_backup=True)]
 130         ).exists()
 131 
 132     def bulk_users_have_2fa(self, user_ids):
 133         """Checks if a list of user ids have 2FA configured.
 134         Returns a dict of {<id>: <has_2fa>}
 135         """
 136         authenticators = set(
 137             Authenticator.objects.filter(
 138                 user__in=user_ids,
 139                 type__in=[a.type for a in available_authenticators(ignore_backup=True)],
 140             )
 141             .distinct()
 142             .values_list("user_id", flat=True)
 143         )
 144         return {id: id in authenticators for id in user_ids}

 159 def available_authenticators(ignore_backup=False):
 160     interfaces = six.itervalues(AUTHENTICATOR_INTERFACES)

 162         return [v for v in interfaces if v.is_available]

 166 class AuthenticatorInterface(object):
 167     type = -1
 168     interface_id = None
 169     name = None
 170     description = None
 171     is_backup_interface = False
 172     enroll_button = _("Enroll")
 173     configure_button = _("Info")
 174     remove_button = _("Remove")
 175     is_available = True
 176     allow_multi_enrollment = False
 177 
 178     def __init__(self, authenticator=None):
 179         if authenticator is None:
 180             self.authenticator = None
 181         else:
 182             self.authenticator = authenticator
 183 
 184     @property
 185     def is_enrolled(self):
 186         """Returns `True` if the interfaces is enrolled (eg: has an
 187         authenticator for a user attached).
 188         """
 189         return self.authenticator is not None
 190 
 191     @property
 192     def requires_activation(self):
 193         """If the interface has an activation method that needs to be
 194         called this returns `True`.
 195         """
 196         return self.activate.im_func is not AuthenticatorInterface.activate.im_func
 197 
 198     @property
 199     def can_validate_otp(self):
 200         """If the interface is able to validate OTP codes then this returns
 201         `True`.
 202         """
 203         return self.validate_otp.im_func is not AuthenticatorInterface.validate_otp.im_func
 204 
 205     @property
 206     def config(self):
 207         """Returns the configuration dictionary for this interface.  If
 208         the interface is registered with an authenticator (eg: it is
 209         enrolled) then the authenticator's config is returned, otherwise
 210         a new config is used on first access.
 211         """
 212         if self.authenticator is not None:
 213             return self.authenticator.config
 214         rv = getattr(self, "_unbound_config", None)
 215         if rv is None:
 216             # Prevent bad recursion if stuff wants to access the default
 217             # config
 218             self._unbound_config = {}
 219             rv = self._unbound_config = self.generate_new_config()
 220         return rv
 221 
 222     def generate_new_config(self):
 223         """This method is invoked if a new config is required."""
 224         return {}
 225 
 226     def activate(self, request):
 227         """If an authenticator overrides this then the method is called
 228         when the dialog for authentication is brought up.  The returned string
 229         is then rendered in the UI.
 230         """
 231         # This method needs to be empty for the default
 232         # `requires_activation` property to make sense.
 233         pass
 234 
 235     def enroll(self, user):
 236         """Invoked to enroll a user for this interface.  If already enrolled
 237         an error is raised.
 238         """
 239         if self.authenticator is None:
 240             self.authenticator = Authenticator.objects.create(
 241                 user=user, type=self.type, config=self.config
 242             )
 243         else:
 244             if not self.allow_multi_enrollment:
 245                 raise Authenticator.AlreadyEnrolled()
 246             self.authenticator.config = self.config
 247             self.authenticator.save()
 248 
 249     def validate_otp(self, otp):
 250         """This method is invoked for an OTP response and has to return
 251         `True` or `False` based on the validity of the OTP response.  Note
 252         that this can be called with otp responses from other interfaces.
 253         """
 254         return False
 255 
 256     def validate_response(self, request, challenge, response):
 257         """If the activation generates a challenge that needs to be
 258         responded to this validates the response for that challenge.  This
 259         is only ever called for challenges emitted by the activation of this
 260         activation interface.
 261         """
 262         return False

 266 class RecoveryCodeInterface(AuthenticatorInterface):
 267     """A backup interface that is based on static recovery codes."""
 268 
 269     type = 0
 270     interface_id = "recovery"
 271     name = _("Recovery Codes")
 272     description = _(
 273         "Recovery codes are the only way to access your account "
 274         "if you lose your device and cannot receive two factor "
 275         "authentication codes."
 276     )
 277     enroll_button = _("Activate")
 278     configure_button = _("View Codes")
 279     remove_button = None
 280     is_backup_interface = True
 281 
 282     def __init__(self, authenticator=None):
 283         AuthenticatorInterface.__init__(self, authenticator)
 284 
 285     def get_codes(self):
 286         rv = []
 287         if self.is_enrolled:
 288             h = hmac.new(key=self.config["salt"].encode("utf-8"), msg=None, digestmod=hashlib.sha1)
 289             for x in range(10):
 290                 h.update("%s|" % x)
 291                 rv.append(base64.b32encode(h.digest())[:8])
 292         return rv
 293 
 294     def generate_new_config(self):
 295         salt = hexlify(os.urandom(16))
 296         return {"salt": salt, "used": 0}
 297 
 298     def regenerate_codes(self, save=True):
 299         if not self.is_enrolled:
 300             raise RuntimeError("Interface is not enrolled")
 301         self.config.update(self.generate_new_config())
 302         self.authenticator.reset_fields(save=False)
 303         if save:
 304             self.authenticator.save()
 305 
 306     def validate_otp(self, otp):
 307         mask = self.config["used"]
 308         code = otp.strip().replace("-", "").upper()
 309         for idx, ref_code in enumerate(self.get_codes()):
 310             if code == ref_code:
 311                 if mask & (1 << idx):
 312                     break
 313                 self.config["used"] = mask | (1 << idx)
 314                 return True
 315         return False
 316 
 317     def get_unused_codes(self):
 318         mask = self.config["used"]
 319         rv = []
 320         for idx, code in enumerate(self.get_codes()):
 321             if not mask & (1 << idx):
 322                 rv.append(code[:4] + "-" + code[4:])
 323         return rv

 326 class OtpMixin(object):
 327     def generate_new_config(self):
 328         return {"secret": generate_secret_key()}
 329 
 330     def _get_secret(self):
 331         return self.config["secret"]
 332 
 333     def _set_secret(self, secret):
 334         self.config["secret"] = secret
 335 
 336     secret = property(_get_secret, _set_secret)
 337     del _get_secret, _set_secret
 338 
 339     def make_otp(self):
 340         return TOTP(self.secret)
 341 
 342     def _get_otp_counter_cache_key(self, counter):
 343         if self.authenticator is not None:
 344             return "used-otp-counters:%s:%s" % (self.authenticator.user_id, counter)
 345 
 346     def check_otp_counter(self, counter):
 347         # OTP uses an internal counter that increments every 30 seconds.
 348         # A hash function generates a six digit code based on the counter
 349         # and a secret key.  If the generated PIN was used it is marked in
 350         # redis as used by remembering which counter it was generated
 351         # from.  This is what we check for here.
 352         cache_key = self._get_otp_counter_cache_key(counter)
 353         return cache_key is None or cache.get(cache_key) != "1"
 354 
 355     def mark_otp_counter_used(self, counter):
 356         cache_key = self._get_otp_counter_cache_key(counter)
 357         if cache_key is not None:
 358             # Mark us used for three windows
 359             cache.set(cache_key, "1", timeout=120)
 360 
 361     def validate_otp(self, otp):
 362         if not otp:
 363             return False
 364         otp = otp.strip().replace("-", "").replace(" ", "")
 365         used_counter = self.make_otp().verify(
 366             otp, return_counter=True, check_counter_func=self.check_otp_counter
 367         )
 368         if used_counter is not None:
 369             self.mark_otp_counter_used(used_counter)
 370             return True
 371         return False

 393 class SmsInterface(OtpMixin, AuthenticatorInterface):
 394     """This interface sends OTP codes via text messages to the user."""
 395 
 396     type = 2
 397     interface_id = "sms"
 398     name = _("Text Message")
 399     description = _(
 400         "This authenticator sends you text messages for "
 401         "verification.  It's useful as a backup method "
 402         "or when you do not have a phone that supports "
 403         "an authenticator application."
 404     )
 405     code_ttl = 45
 406 
 407     @classproperty
 408     def is_available(cls):
 409         return sms_available()
 410 
 411     def generate_new_config(self):
 412         config = super(SmsInterface, self).generate_new_config()
 413         config["phone_number"] = None
 414         return config
 415 
 416     def make_otp(self):
 417         return TOTP(self.config["secret"], digits=6, interval=self.code_ttl, default_window=1)
 418 
 419     def _get_phone_number(self):
 420         return self.config["phone_number"]
 421 
 422     def _set_phone_number(self, value):
 423         self.config["phone_number"] = value
 424 
 425     phone_number = property(_get_phone_number, _set_phone_number)
 426     del _get_phone_number, _set_phone_number
 427 
 428     def activate(self, request):
 429         phone_number = self.config["phone_number"]
 430         if len(phone_number) == 10:
 431             mask = "(***) ***-**%s" % (phone_number[-2:])
 432         else:
 433             mask = "%s%s" % ((len(phone_number) - 2) * "*", phone_number[-2:])
 434 
 435         if self.send_text(request=request):
 436             return ActivationMessageResult(
 437                 _(
 438                     "A confirmation code was sent to %(phone_mask)s. "
 439                     "It is valid for %(ttl)d seconds."
 440                 )
 441                 % {"phone_mask": "<strong>%s</strong>" % mask, "ttl": self.code_ttl}
 442             )
 443         return ActivationMessageResult(
 444             _(
 445                 "Error: we failed to send a text message to you. You "
 446                 "can try again later or sign in with a different method."
 447             ),
 448             type="error",
 449         )
 450 
 451     def send_text(self, for_enrollment=False, request=None):
 452         ctx = {"code": self.make_otp().generate_otp()}
 453 
 454         if for_enrollment:
 455             text = _(
 456                 "%(code)s is your Sentry two-factor enrollment code. "
 457                 "You are about to set up text message based two-factor "
 458                 "authentication."
 459             )
 460         else:
 461             text = _("%(code)s is your Sentry authentication code.")
 462 
 463         if request is not None:
 464             text = u"%s\n\n%s" % (text, _("Requested from %(ip)s"))
 465             ctx["ip"] = request.META["REMOTE_ADDR"]
 466 
 467         return send_sms(text % ctx, to=self.phone_number)

 471 class U2fInterface(AuthenticatorInterface):
 472     type = 3
 473     interface_id = "u2f"
 474     configure_button = _("Configure")
 475     name = _("U2F (Universal 2nd Factor)")
 476     description = _(
 477         "Authenticate with a U2F hardware device. This is a "
 478         "device like a Yubikey or something similar which "
 479         "supports FIDO's U2F specification. This also requires "
 480         "a browser which supports this system (like Google "
 481         "Chrome)."
 482     )
 483     allow_multi_enrollment = True
 484 
 485     @classproperty
 486     def u2f_app_id(cls):
 487         rv = options.get("u2f.app-id")
 488         return rv or absolute_uri(reverse("sentry-u2f-app-id"))
 489 
 490     @classproperty
 491     def u2f_facets(cls):
 492         facets = options.get("u2f.facets")
 493         if not facets:
 494             return [options.get("system.url-prefix")]
 495         return [x.rstrip("/") for x in facets]
 496 
 497     @classproperty
 498     def is_available(cls):
 499         url_prefix = options.get("system.url-prefix")
 500         return url_prefix and url_prefix.startswith("https://")
 501 
 502     def generate_new_config(self):
 503         return {}
 504 
 505     def start_enrollment(self):
 506         return u2f.begin_registration(self.u2f_app_id, self.get_u2f_devices()).data_for_client
 507 
 508     def get_u2f_devices(self):
 509         rv = []
 510         for data in self.config.get("devices") or ():
 511             # XXX: The previous version of python-u2flib-server didn't store
 512             # the `version` in the device binding. Defaulting to `U2F_V2` here
 513             # so that we don't break existing u2f registrations.
 514             data["binding"].setdefault("version", "U2F_V2")
 515             rv.append(DeviceRegistration(data["binding"]))
 516         return rv
 517 
 518     def remove_u2f_device(self, key):
 519         """Removes a U2F device but never removes the last one.  This returns
 520         False if the last device would be removed.
 521         """
 522         devices = [x for x in self.config.get("devices") or () if x["binding"]["keyHandle"] != key]
 523         if devices:
 524             self.config["devices"] = devices
 525             return True
 526         return False
 527 
 528     def get_device_name(self, key):
 529         for device in self.config.get("devices") or ():
 530             if device["binding"]["keyHandle"] == key:
 531                 return device["name"]
 532 
 533     def get_registered_devices(self):
 534         rv = []
 535         for device in self.config.get("devices") or ():
 536             rv.append(
 537                 {
 538                     "timestamp": to_datetime(device["ts"]),
 539                     "name": device["name"],
 540                     "key_handle": device["binding"]["keyHandle"],
 541                     "app_id": device["binding"]["appId"],
 542                 }
 543             )
 544         rv.sort(key=lambda x: x["name"])
 545         return rv
 546 
 547     def try_enroll(self, enrollment_data, response_data, device_name=None):
 548         binding, cert = u2f.complete_registration(enrollment_data, response_data, self.u2f_facets)
 549         devices = self.config.setdefault("devices", [])
 550         devices.append(
 551             {
 552                 "name": device_name or "Security Key",
 553                 "ts": int(time.time()),
 554                 "binding": dict(binding),
 555             }
 556         )
 557 
 558     def activate(self, request):
 559         challenge = dict(u2f.begin_authentication(self.u2f_app_id, self.get_u2f_devices()))
 560         # XXX: Upgrading python-u2flib-server to 5.0.0 changes the response
 561         # format. Our current js u2f library expects the old format, so
 562         # massaging the data to include the old `authenticateRequests` key here.
 563         authenticate_requests = []
 564         for registered_key in challenge["registeredKeys"]:
 565             authenticate_requests.append(
 566                 {
 567                     "challenge": challenge["challenge"],
 568                     "version": registered_key["version"],
 569                     "keyHandle": registered_key["keyHandle"],
 570                     "appId": registered_key["appId"],
 571                 }
 572             )
 573         challenge["authenticateRequests"] = authenticate_requests
 574 
 575         return ActivationChallengeResult(challenge=challenge)
 576 
 577     def validate_response(self, request, challenge, response):
 578         try:
 579             u2f.complete_authentication(challenge, response, self.u2f_facets)
 580         except (InvalidSignature, InvalidKey, StopIteration):
 581             return False
 582         return True
