 196     def save(self, *args, **kwargs):
 197         if not self.actor_label:
 198             assert self.actor or self.actor_key
 199             if self.actor:
 200                 self.actor_label = self.actor.username
 201             else:
 202                 self.actor_label = self.actor_key.key
 203         super(AuditLogEntry, self).save(*args, **kwargs)

 205     def get_actor_name(self):
 206         if self.actor:
 207             return self.actor.get_display_name()
 208         elif self.actor_key:
 209             return self.actor_key.key + " (api key)"
 210         return self.actor_label

 212     def get_note(self):
 213         if self.event == AuditLogEntryEvent.MEMBER_INVITE:
 214             return "invited member %s" % (self.data["email"],)
 215         elif self.event == AuditLogEntryEvent.MEMBER_ADD:
 216             if self.target_user == self.actor:
 217                 return "joined the organization"
 218             return "added member %s" % (self.target_user.get_display_name(),)
 219         elif self.event == AuditLogEntryEvent.MEMBER_ACCEPT:
 220             return "accepted the membership invite"
 221         elif self.event == AuditLogEntryEvent.MEMBER_REMOVE:
 222             if self.target_user == self.actor:
 223                 return "left the organization"
 224             return "removed member %s" % (
 225                 self.data.get("email") or self.target_user.get_display_name(),
 226             )
 227         elif self.event == AuditLogEntryEvent.MEMBER_EDIT:
 228             return "edited member %s (role: %s, teams: %s)" % (
 229                 self.data.get("email") or self.target_user.get_display_name(),
 230                 self.data.get("role") or "N/A",
 231                 ", ".join(six.text_type(x) for x in self.data.get("team_slugs", [])) or "N/A",
 232             )
 233         elif self.event == AuditLogEntryEvent.MEMBER_JOIN_TEAM:
 234             if self.target_user == self.actor:
 235                 return "joined team %s" % (self.data["team_slug"],)
 236             return "added %s to team %s" % (
 237                 self.data.get("email") or self.target_user.get_display_name(),
 238                 self.data["team_slug"],
 239             )
 240         elif self.event == AuditLogEntryEvent.MEMBER_LEAVE_TEAM:
 241             if self.target_user == self.actor:
 242                 return "left team %s" % (self.data["team_slug"],)
 243             return "removed %s from team %s" % (
 244                 self.data.get("email") or self.target_user.get_display_name(),
 245                 self.data["team_slug"],
 246             )
 247         elif self.event == AuditLogEntryEvent.MEMBER_PENDING:
 248             return "required member %s to setup 2FA" % (
 249                 self.data.get("email") or self.target_user.get_display_name(),
 250             )
 251 
 252         elif self.event == AuditLogEntryEvent.ORG_ADD:
 253             return "created the organization"
 254         elif self.event == AuditLogEntryEvent.ORG_EDIT:
 255             return "edited the organization setting: " + (
 256                 ", ".join(u"{} {}".format(k, v) for k, v in self.data.items())
 257             )
 258         elif self.event == AuditLogEntryEvent.ORG_REMOVE:
 259             return "removed the organization"
 260         elif self.event == AuditLogEntryEvent.ORG_RESTORE:
 261             return "restored the organization"
 262 
 263         elif self.event == AuditLogEntryEvent.TEAM_ADD:
 264             return "created team %s" % (self.data["slug"],)
 265         elif self.event == AuditLogEntryEvent.TEAM_EDIT:
 266             return "edited team %s" % (self.data["slug"],)
 267         elif self.event == AuditLogEntryEvent.TEAM_REMOVE:
 268             return "removed team %s" % (self.data["slug"],)
 269 
 270         elif self.event == AuditLogEntryEvent.PROJECT_ADD:
 271             return "created project %s" % (self.data["slug"],)
 272         elif self.event == AuditLogEntryEvent.PROJECT_EDIT:
 273             return "edited project settings " + (
 274                 " ".join(
 275                     [" in %s to %s" % (key, value) for (key, value) in six.iteritems(self.data)]
 276                 )
 277             )
 278         elif self.event == AuditLogEntryEvent.PROJECT_REMOVE:
 279             return "removed project %s" % (self.data["slug"],)
 280         elif self.event == AuditLogEntryEvent.PROJECT_REQUEST_TRANSFER:
 281             return "requested to transfer project %s" % (self.data["slug"],)
 282         elif self.event == AuditLogEntryEvent.PROJECT_ACCEPT_TRANSFER:
 283             return "accepted transfer of project %s" % (self.data["slug"],)
 284         elif self.event in [AuditLogEntryEvent.PROJECT_ENABLE, AuditLogEntryEvent.PROJECT_DISABLE]:
 285             verb = "enabled" if self.event == AuditLogEntryEvent.PROJECT_ENABLE else "disabled"
 286 
 287             # Most logs will just be name of the filter, but legacy browser changes can be bool, str or sets
 288             filter_name = self.data["state"]
 289             if (
 290                 filter_name in ("0", "1")
 291                 or isinstance(filter_name, set)
 292                 or isinstance(filter_name, bool)
 293             ):
 294                 message = "%s project filter legacy-browsers" % (verb,)
 295                 if isinstance(filter_name, set):
 296                     message += ": %s" % (", ".join(filter_name),)
 297                 return message
 298             else:
 299                 return "%s project filter %s" % (verb, filter_name)
 300 
 301         elif self.event == AuditLogEntryEvent.TAGKEY_REMOVE:
 302             return "removed tags matching %s = *" % (self.data["key"],)
 303 
 304         elif self.event == AuditLogEntryEvent.PROJECTKEY_ADD:
 305             return "added project key %s" % (self.data["public_key"],)
 306         elif self.event == AuditLogEntryEvent.PROJECTKEY_EDIT:
 307             return "edited project key %s" % (self.data["public_key"],)
 308         elif self.event == AuditLogEntryEvent.PROJECTKEY_REMOVE:
 309             return "removed project key %s" % (self.data["public_key"],)
 310         elif self.event == AuditLogEntryEvent.PROJECTKEY_ENABLE:
 311             return "enabled project key %s" % (self.data["public_key"],)
 312         elif self.event == AuditLogEntryEvent.PROJECTKEY_DISABLE:
 313             return "disabled project key %s" % (self.data["public_key"],)
 314 
 315         elif self.event == AuditLogEntryEvent.SSO_ENABLE:
 316             return "enabled sso (%s)" % (self.data["provider"],)
 317         elif self.event == AuditLogEntryEvent.SSO_DISABLE:
 318             return "disabled sso (%s)" % (self.data["provider"],)
 319         elif self.event == AuditLogEntryEvent.SSO_EDIT:
 320             return "edited sso settings: " + (
 321                 ", ".join(u"{} {}".format(k, v) for k, v in self.data.items())
 322             )
 323         elif self.event == AuditLogEntryEvent.SSO_IDENTITY_LINK:
 324             return "linked their account to a new identity"
 325 
 326         elif self.event == AuditLogEntryEvent.APIKEY_ADD:
 327             return "added api key %s" % (self.data["label"],)
 328         elif self.event == AuditLogEntryEvent.APIKEY_EDIT:
 329             return "edited api key %s" % (self.data["label"],)
 330         elif self.event == AuditLogEntryEvent.APIKEY_REMOVE:
 331             return "removed api key %s" % (self.data["label"],)
 332 
 333         elif self.event == AuditLogEntryEvent.RULE_ADD:
 334             return 'added rule "%s"' % (self.data["label"],)
 335         elif self.event == AuditLogEntryEvent.RULE_EDIT:
 336             return 'edited rule "%s"' % (self.data["label"],)
 337         elif self.event == AuditLogEntryEvent.RULE_REMOVE:
 338             return 'removed rule "%s"' % (self.data["label"],)
 339 
 340         elif self.event == AuditLogEntryEvent.SET_ONDEMAND:
 341             if self.data["ondemand"] == -1:
 342                 return "changed on-demand spend to unlimited"
 343             return "changed on-demand max spend to $%d" % (self.data["ondemand"] / 100,)
 344         elif self.event == AuditLogEntryEvent.TRIAL_STARTED:
 345             return "started trial"
 346         elif self.event == AuditLogEntryEvent.PLAN_CHANGED:
 347             return "changed plan to %s" % (self.data["plan_name"],)
 348         elif self.event == AuditLogEntryEvent.PLAN_CANCELLED:
 349             return "cancelled plan"
 350 
 351         elif self.event == AuditLogEntryEvent.SERVICEHOOK_ADD:
 352             return 'added a service hook for "%s"' % (truncatechars(self.data["url"], 64),)
 353         elif self.event == AuditLogEntryEvent.SERVICEHOOK_EDIT:
 354             return 'edited the service hook for "%s"' % (truncatechars(self.data["url"], 64),)
 355         elif self.event == AuditLogEntryEvent.SERVICEHOOK_REMOVE:
 356             return 'removed the service hook for "%s"' % (truncatechars(self.data["url"], 64),)
 357         elif self.event == AuditLogEntryEvent.SERVICEHOOK_ENABLE:
 358             return 'enabled theservice hook for "%s"' % (truncatechars(self.data["url"], 64),)
 359         elif self.event == AuditLogEntryEvent.SERVICEHOOK_DISABLE:
 360             return 'disabled the service hook for "%s"' % (truncatechars(self.data["url"], 64),)
 361 
 362         elif self.event == AuditLogEntryEvent.INTEGRATION_ADD:
 363             if self.data.get("provider"):
 364                 return "installed %s for the %s integration" % (
 365                     self.data["name"],
 366                     self.data["provider"],
 367                 )
 368             else:
 369                 return "enabled integration %s for project %s" % (
 370                     self.data["integration"],
 371                     self.data["project"],
 372                 )
 373         elif self.event == AuditLogEntryEvent.INTEGRATION_EDIT:
 374             if self.data.get("provider"):
 375                 return "edited the %s for the %s integration" % (
 376                     self.data["name"],
 377                     self.data["provider"],
 378                 )
 379             return "edited integration %s for project %s" % (
 380                 self.data["integration"],
 381                 self.data["project"],
 382             )
 383         elif self.event == AuditLogEntryEvent.INTEGRATION_REMOVE:
 384             if self.data.get("provider"):
 385                 return "uninstalled %s for the %s integration" % (
 386                     self.data["name"],
 387                     self.data["provider"],
 388                 )
 389             return "disabled integration %s from project %s" % (
 390                 self.data["integration"],
 391                 self.data["project"],
 392             )
 393         elif self.event == AuditLogEntryEvent.INTEGRATION_UPGRADE:
 394             if self.data.get("provider"):
 395                 return "upgraded %s for the %s integration" % (
 396                     self.data["name"],
 397                     self.data["provider"],
 398                 )
 399 
 400         elif self.event == AuditLogEntryEvent.SENTRY_APP_ADD:
 401             return "created sentry app %s" % (self.data["sentry_app"])
 402         elif self.event == AuditLogEntryEvent.SENTRY_APP_REMOVE:
 403             return "removed sentry app %s" % (self.data["sentry_app"])
 404         elif self.event == AuditLogEntryEvent.SENTRY_APP_INSTALL:
 405             return "installed sentry app %s" % (self.data["sentry_app"])
 406         elif self.event == AuditLogEntryEvent.SENTRY_APP_UNINSTALL:
 407             return "uninstalled sentry app %s" % (self.data["sentry_app"])
 408         elif self.event == AuditLogEntryEvent.INTERNAL_INTEGRATION_ADD:
 409             return "created internal integration %s" % (self.data.get("name", ""))
 410         elif self.event == AuditLogEntryEvent.INTERNAL_INTEGRATION_ADD_TOKEN:
 411             return "created a token for internal integration %s" % (self.data["sentry_app"])
 412         elif self.event == AuditLogEntryEvent.INTERNAL_INTEGRATION_REMOVE_TOKEN:
 413             return "revoked a token for internal integration %s" % (self.data["sentry_app"])
 414         elif self.event == AuditLogEntryEvent.INVITE_REQUEST_ADD:
 415             return "request added to invite %s" % (self.data["email"],)
 416         elif self.event == AuditLogEntryEvent.INVITE_REQUEST_REMOVE:
 417             return "removed the invite request for %s" % (self.data["email"],)
 418 
 419         return ""
