  30 class IdentityProvider(Model):
  31     """
  32     An IdentityProvider is an instance of a provider.
  33 
  34     The IdentityProvider is unique on the type of provider (eg github, slack,
  35     google, etc).
  36 
  37     A SAML identity provide might look like this, type: onelogin, instance:
  38     acme-org.onelogin.com.
  39     """
  40 
  41     __core__ = False
  42 
  43     type = models.CharField(max_length=64)
  44     config = EncryptedJsonField()
  45     date_added = models.DateTimeField(default=timezone.now, null=True)
  46     external_id = models.CharField(max_length=64, null=True)
  47 
  48     class Meta:
  49         app_label = "sentry"
  50         db_table = "sentry_identityprovider"
  51         unique_together = (("type", "external_id"),)
  52 
  53     def get_provider(self):
  54         from sentry.identity import get
  55 
  56         return get(self.type)

  59 class Identity(Model):
  60     """
  61     A verified link between a user and a third party identity.
  62     """
  63 
  64     __core__ = False
  65 
  66     idp = FlexibleForeignKey("sentry.IdentityProvider")
  67     user = FlexibleForeignKey(settings.AUTH_USER_MODEL)
  68     external_id = models.CharField(max_length=64)
  69     data = EncryptedJsonField()
  70     status = BoundedPositiveIntegerField(default=IdentityStatus.UNKNOWN)
  71     scopes = ArrayField()
  72     date_verified = models.DateTimeField(default=timezone.now)
  73     date_added = models.DateTimeField(default=timezone.now)
  74 
  75     class Meta:
  76         app_label = "sentry"
  77         db_table = "sentry_identity"
  78         unique_together = (("idp", "external_id"), ("idp", "user"))
  79 
  80     def get_provider(self):
  81         from sentry.identity import get
  82 
  83         return get(self.idp.type)
  84 
  85     @classmethod
  86     def reattach(cls, idp, external_id, user, defaults):
  87         """
  88         Removes identities under `idp` associated with either `external_id` or `user`
  89         and creates a new identity linking them.
  90         """
  91         lookup = Q(external_id=external_id) | Q(user=user)
  92         Identity.objects.filter(lookup, idp=idp).delete()
  93         logger.info(
  94             "deleted-identity",
  95             extra={"external_id": external_id, "idp_id": idp.id, "user_id": user.id},
  96         )
  97 
  98         identity_model = Identity.objects.create(
  99             idp=idp, user=user, external_id=external_id, **defaults
 100         )
 101         logger.info(
 102             "created-identity",
 103             extra={
 104                 "idp_id": idp.id,
 105                 "external_id": external_id,
 106                 "object_id": identity_model.id,
 107                 "user_id": user.id,
 108             },
 109         )
 110         return identity_model
