  23 class TeamManager(BaseManager):
  24     def get_for_user(self, organization, user, scope=None, with_projects=False):
  25         """
  26         Returns a list of all teams a user has some level of access to.
  27         """
  28         from sentry.auth.superuser import is_active_superuser
  29         from sentry.models import (
  30             OrganizationMemberTeam,
  31             Project,
  32             ProjectStatus,
  33             ProjectTeam,
  34             OrganizationMember,
  35         )
  36 
  37         if not user.is_authenticated():
  38             return []
  39 
  40         base_team_qs = self.filter(organization=organization, status=TeamStatus.VISIBLE)
  41 
  42         if env.request and is_active_superuser(env.request) or settings.SENTRY_PUBLIC:
  43             team_list = list(base_team_qs)
  44         else:
  45             try:
  46                 om = OrganizationMember.objects.get(user=user, organization=organization)
  47             except OrganizationMember.DoesNotExist:
  48                 # User is not a member of the organization at all
  49                 return []
  50 
  51             # If a scope is passed through, make sure this scope is
  52             # available on the OrganizationMember object.
  53             if scope is not None and scope not in om.get_scopes():
  54                 return []
  55 
  56             team_list = list(
  57                 base_team_qs.filter(
  58                     id__in=OrganizationMemberTeam.objects.filter(
  59                         organizationmember=om, is_active=True
  60                     ).values_list("team")
  61                 )
  62             )
  63 
  64         results = sorted(team_list, key=lambda x: x.name.lower())
  65 
  66         if with_projects:
  67             project_list = sorted(
  68                 Project.objects.filter(teams__in=team_list, status=ProjectStatus.VISIBLE),
  69                 key=lambda x: x.name.lower(),
  70             )
  71 
  72             teams_by_project = defaultdict(set)
  73             for project_id, team_id in ProjectTeam.objects.filter(
  74                 project__in=project_list, team__in=team_list
  75             ).values_list("project_id", "team_id"):
  76                 teams_by_project[project_id].add(team_id)
  77 
  78             projects_by_team = {t.id: [] for t in team_list}
  79             for project in project_list:
  80                 for team_id in teams_by_project[project.id]:
  81                     projects_by_team[team_id].append(project)
  82 
  83             # these kinds of queries make people sad :(
  84             for idx, team in enumerate(results):
  85                 team_projects = projects_by_team[team.id]
  86                 results[idx] = (team, team_projects)
  87 
  88         return results

  98 class Team(Model):
  99     """
 100     A team represents a group of individuals which maintain ownership of projects.
 101     """

 105     organization = FlexibleForeignKey("sentry.Organization")
 106     slug = models.SlugField()
 107     name = models.CharField(max_length=64)
 108     status = BoundedPositiveIntegerField(
 109         choices=(
 110             (TeamStatus.VISIBLE, _("Active")),

 112             (TeamStatus.DELETION_IN_PROGRESS, _("Deletion in Progress")),
 113         ),
 114         default=TeamStatus.VISIBLE,
 115     )
 116     date_added = models.DateTimeField(default=timezone.now, null=True)

 127     def __unicode__(self):
 128         return u"%s (%s)" % (self.name, self.slug)

 130     def save(self, *args, **kwargs):
 131         if not self.slug:
 132             lock = locks.get("slug:team", duration=5)
 133             with TimedRetryPolicy(10)(lock.acquire):
 134                 slugify_instance(self, self.name, organization=self.organization)
 135             super(Team, self).save(*args, **kwargs)
 136         else:
 137             super(Team, self).save(*args, **kwargs)

 147     def has_access(self, user, access=None):
 148         from sentry.models import AuthIdentity, OrganizationMember
 149 
 150         warnings.warn("Team.has_access is deprecated.", DeprecationWarning)
 151 
 152         queryset = self.member_set.filter(user=user)
 153         if access is not None:
 154             queryset = queryset.filter(type__lte=access)
 155 
 156         try:
 157             member = queryset.get()
 158         except OrganizationMember.DoesNotExist:
 159             return False
 160 
 161         try:
 162             auth_identity = AuthIdentity.objects.get(
 163                 auth_provider__organization=self.organization_id, user=member.user_id
 164             )
 165         except AuthIdentity.DoesNotExist:
 166             return True
 167 
 168         return auth_identity.is_valid(member)

 170     def transfer_to(self, organization):
 171         """
 172         Transfers a team and all projects under it to the given organization.
 173         """
 174         from sentry.models import (
 175             OrganizationAccessRequest,
 176             OrganizationMember,
 177             OrganizationMemberTeam,
 178             Project,
 179             ProjectTeam,
 180             ReleaseProject,
 181             ReleaseProjectEnvironment,
 182         )
 183 
 184         try:
 185             with transaction.atomic():
 186                 self.update(organization=organization)
 187         except IntegrityError:
 188             # likely this means a team already exists, let's try to coerce to
 189             # it instead of a blind transfer
 190             new_team = Team.objects.get(organization=organization, slug=self.slug)
 191         else:
 192             new_team = self
 193 
 194         project_ids = list(
 195             Project.objects.filter(teams=self)
 196             .exclude(organization=organization)
 197             .values_list("id", flat=True)
 198         )
 199 
 200         # remove associations with releases from other org
 201         ReleaseProject.objects.filter(project_id__in=project_ids).delete()
 202         ReleaseProjectEnvironment.objects.filter(project_id__in=project_ids).delete()
 203 
 204         Project.objects.filter(id__in=project_ids).update(organization=organization)
 205 
 206         ProjectTeam.objects.filter(project_id__in=project_ids).update(team=new_team)
 207 
 208         # remove any pending access requests from the old organization
 209         if self != new_team:
 210             OrganizationAccessRequest.objects.filter(team=self).delete()
 211 
 212         # identify shared members and ensure they retain team access
 213         # under the new organization
 214         old_memberships = OrganizationMember.objects.filter(teams=self).exclude(
 215             organization=organization
 216         )
 217         for member in old_memberships:
 218             try:
 219                 new_member = OrganizationMember.objects.get(
 220                     user=member.user, organization=organization
 221                 )
 222             except OrganizationMember.DoesNotExist:
 223                 continue
 224 
 225             try:
 226                 with transaction.atomic():
 227                     OrganizationMemberTeam.objects.create(
 228                         team=new_team, organizationmember=new_member
 229                     )
 230             except IntegrityError:
 231                 pass
 232 
 233         OrganizationMemberTeam.objects.filter(team=self).exclude(
 234             organizationmember__organization=organization
 235         ).delete()
 236 
 237         if new_team != self:
 238             cursor = connections[router.db_for_write(Team)].cursor()
 239             # we use a cursor here to avoid automatic cascading of relations
 240             # in Django
 241             try:
 242                 cursor.execute("DELETE FROM sentry_team WHERE id = %s", [self.id])
 243             finally:
 244                 cursor.close()
