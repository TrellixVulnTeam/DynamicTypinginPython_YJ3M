  15 class RecentSearch(Model):
  16     """
  17     Searches run by users recently.
  18     """
  19 
  20     __core__ = True
  21 
  22     organization = FlexibleForeignKey("sentry.Organization")

  38 def remove_excess_recent_searches(organization, user, search_type):
  39     """
  40     Remove any excess recent searches. We do this by sorting by `last_seen`
  41     descending and removing any rows after the `MAX_RECENT_SEARCHES` row. In
  42     practice this should only be removing a single row at most.
  43     """
  44     recent_searches_to_remove = RecentSearch.objects.filter(
  45         organization=organization, user=user, type=search_type
  46     ).order_by("-last_seen")[MAX_RECENT_SEARCHES:]
  47     RecentSearch.objects.filter(id__in=recent_searches_to_remove).delete()
  48

  50 @receiver(pre_save, sender=RecentSearch)
  51 def issue_saved(sender, instance, **kwargs):
  52     if not instance.query_hash:
  53         instance.query_hash = md5_text(instance.query).hexdigest()
