  41     @classmethod
  42     def _get_or_create_impl(cls, release, project, environment, datetime, metrics_tags, **kwargs):
  43         cache_key = cls.get_cache_key(project.id, release.id, environment.id)
  44 
  45         instance = cache.get(cache_key)
  46         if instance is None:
  47             metrics_tags["cache_hit"] = "false"
  48             instance, created = cls.objects.get_or_create(
  49                 release=release,
  50                 project=project,
  51                 environment=environment,
  52                 defaults={"first_seen": datetime, "last_seen": datetime},
  53             )
  54             cache.set(cache_key, instance, 3600)

  56         else:
  57             created = False
  58 
  59         metrics_tags["created"] = "true" if created else "false"
  60 
  61         # Same as releaseenvironment model. Minimizes last_seen updates to once a minute
  62         if not created and instance.last_seen < datetime - timedelta(seconds=60):
  63             cls.objects.filter(
  64                 id=instance.id, last_seen__lt=datetime - timedelta(seconds=60)
  65             ).update(last_seen=datetime)
  66             instance.last_seen = datetime
  67             cache.set(cache_key, instance, 3600)
  68             metrics_tags["bumped"] = "true"
  69         else:
  70             metrics_tags["bumped"] = "false"
  71 
  72         return instance
