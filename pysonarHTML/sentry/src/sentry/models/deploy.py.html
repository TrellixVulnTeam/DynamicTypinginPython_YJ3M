   6 from __future__ import absolute_import
   7 
   8 from django.db import models
   9 from django.utils import timezone
  10 
  11 from sentry.app import locks
  12 from sentry.db.models import BoundedPositiveIntegerField, FlexibleForeignKey, Model
  13 from sentry.utils.retries import TimedRetryPolicy
  14 
  15 
  16 class Deploy(Model):
  17     __core__ = False
  18 
  19     organization_id = BoundedPositiveIntegerField(db_index=True)
  20     release = FlexibleForeignKey("sentry.Release")
  21     environment_id = BoundedPositiveIntegerField(db_index=True)
  22     date_finished = models.DateTimeField(default=timezone.now)
  23     date_started = models.DateTimeField(null=True, blank=True)
  24     name = models.CharField(max_length=64, null=True, blank=True)
  25     url = models.URLField(null=True, blank=True)
  26     notified = models.NullBooleanField(null=True, db_index=True, default=False)
  27 
  28     class Meta:
  29         app_label = "sentry"
  30         db_table = "sentry_deploy"
  31 
  32     @staticmethod
  33     def get_lock_key(deploy_id):
  34         return "deploy-notify:%s" % deploy_id
  35 
  36     @classmethod
  37     def notify_if_ready(cls, deploy_id, fetch_complete=False):
  38         """
  39         create activity and send deploy notifications
  40         if they haven't been sent
  41         """
  42         from sentry.models import Activity, Environment, ReleaseCommit, ReleaseHeadCommit
  43 
  44         lock_key = cls.get_lock_key(deploy_id)
  45         lock = locks.get(lock_key, duration=30)
  46         with TimedRetryPolicy(10)(lock.acquire):
  47             deploy = cls.objects.filter(id=deploy_id).select_related("release").get()
  48             if deploy.notified:
  49                 return
  50 
  51             release = deploy.release
  52             environment = Environment.objects.get(
  53                 organization_id=deploy.organization_id, id=deploy.environment_id
  54             )
  55 
  56             if not fetch_complete:
  57                 release_has_commits = ReleaseCommit.objects.filter(
  58                     organization_id=release.organization_id, release=release
  59                 ).exists()
  60 
  61                 if not release_has_commits:
  62                     # check if we have head commits, which
  63                     # would indicate that we're waiting for
  64                     # fetch_commits to complete
  65                     if ReleaseHeadCommit.objects.filter(
  66                         organization_id=release.organization_id, release=release
  67                     ).exists():
  68                         return
  69 
  70             activity = None
  71             for project in deploy.release.projects.all():
  72                 activity = Activity.objects.create(
  73                     type=Activity.DEPLOY,
  74                     project=project,
  75                     ident=Activity.get_version_ident(release.version),
  76                     data={
  77                         "version": release.version,
  78                         "deploy_id": deploy.id,
  79                         "environment": environment.name,
  80                     },
  81                     datetime=deploy.date_finished,
  82                 )
  83             # Somewhat hacky, only send notification for one
  84             # Deploy Activity record because it will cover all projects
  85             if activity is not None:
  86                 activity.send_notification()
  87                 deploy.update(notified=True)
