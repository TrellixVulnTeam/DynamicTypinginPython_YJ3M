  40 def parse_short_id(short_id):
  41     match = _short_id_re.match(short_id.strip())
  42     if match is None:
  43         return None
  44     slug, id = match.groups()
  45     slug = slug.lower()
  46     try:
  47         short_id = base32_decode(id)
  48         # We need to make sure the short id is not overflowing the
  49         # field's max or the lookup will fail with an assertion error.
  50         max_id = Group._meta.get_field("short_id").MAX_VALUE
  51         if short_id > max_id:
  52             return None
  53     except ValueError:
  54         return None
  55     return ShortId(slug, short_id)

  58 def looks_like_short_id(value):
  59     return _short_id_re.match((value or "").strip()) is not None

  62 def get_group_with_redirect(id_or_qualified_short_id, queryset=None, organization=None):
  63     """
  64     Retrieve a group by ID, checking the redirect table if the requested group
  65     does not exist. Returns a two-tuple of ``(object, redirected)``.
  66     """
  67     if queryset is None:
  68         if organization:
  69             queryset = Group.objects.filter(project__organization=organization)
  70             getter = queryset.get
  71         else:
  72             queryset = Group.objects.all()
  73             # When not passing a queryset, we want to read from cache
  74             getter = Group.objects.get_from_cache
  75     else:
  76         if organization:
  77             queryset = queryset.filter(project__organization=organization)
  78         getter = queryset.get
  79 
  80     if not (
  81         isinstance(id_or_qualified_short_id, six.integer_types)  # noqa
  82         or id_or_qualified_short_id.isdigit()
  83     ):  # NOQA
  84         short_id = parse_short_id(id_or_qualified_short_id)
  85         if not short_id or not organization:
  86             raise Group.DoesNotExist()
  87         params = {
  88             "project__slug": short_id.project_slug,
  89             "short_id": short_id.short_id,
  90             "project__organization": organization,
  91         }
  92     else:
  93         short_id = None
  94         params = {"id": id_or_qualified_short_id}
  95 
  96     try:
  97         return getter(**params), False
  98     except Group.DoesNotExist as error:
  99         from sentry.models import GroupRedirect
 100 
 101         if short_id:
 102             params = {
 103                 "id": GroupRedirect.objects.filter(
 104                     organization_id=organization.id,
 105                     previous_short_id=short_id.short_id,
 106                     previous_project_slug=short_id.project_slug,
 107                 ).values_list("group_id", flat=True)
 108             }
 109         else:
 110             params["id"] = GroupRedirect.objects.filter(previous_group_id=params["id"]).values_list(
 111                 "group_id", flat=True
 112             )
 113         try:
 114             return queryset.get(**params), True
 115         except Group.DoesNotExist:
 116             raise error  # raise original `DoesNotExist`

 137 def get_oldest_or_latest_event_for_environments(
 138     ordering, environments=(), issue_id=None, project_id=None
 139 ):
 140     conditions = []
 141 
 142     if len(environments) > 0:
 143         conditions.append(["environment", "IN", environments])
 144 
 145     events = eventstore.get_events(
 146         filter=eventstore.Filter(
 147             conditions=conditions, project_ids=[project_id], group_ids=[issue_id]
 148         ),
 149         limit=1,
 150         orderby=ordering.value,
 151         referrer="Group.get_latest",
 152     )
 153 
 154     if events:
 155         return events[0]
 156 
 157     return None

 193     def from_event_id(self, project, event_id):
 194         """
 195         Resolves the 32 character event_id string into
 196         a Group for which it is found.
 197         """
 198         group_id = None
 199 
 200         event = eventstore.get_event_by_id(project.id, event_id)
 201 
 202         if event:
 203             group_id = event.group_id
 204 
 205         if group_id is None:
 206             # Raise a Group.DoesNotExist here since it makes
 207             # more logical sense since this is intending to resolve
 208             # a Group.
 209             raise Group.DoesNotExist()
 210 
 211         return Group.objects.get(id=group_id)

 213     def filter_by_event_id(self, project_ids, event_id):
 214         event_ids = [event_id]
 215         conditions = [["group_id", "IS NOT NULL", None]]
 216         data = eventstore.get_events(
 217             filter=eventstore.Filter(
 218                 event_ids=event_ids, project_ids=project_ids, conditions=conditions
 219             ),
 220             limit=max(len(project_ids), 100),
 221             referrer="Group.filter_by_event_id",
 222         )
 223 
 224         group_ids = set([evt.group_id for evt in data])
 225 
 226         return Group.objects.filter(id__in=group_ids)

 301     def save(self, *args, **kwargs):
 302         if not self.last_seen:
 303             self.last_seen = timezone.now()
 304         if not self.first_seen:
 305             self.first_seen = self.last_seen
 306         if not self.active_at:
 307             self.active_at = self.first_seen
 308         # We limit what we store for the message body
 309         self.message = strip(self.message)
 310         if self.message:
 311             self.message = truncatechars(self.message.splitlines()[0], 255)
 312         if self.times_seen is None:
 313             self.times_seen = 1
 314         self.score = type(self).calculate_score(
 315             times_seen=self.times_seen, last_seen=self.last_seen
 316         )
 317         super(Group, self).save(*args, **kwargs)

 319     def get_absolute_url(self, params=None):

 352     def get_status(self):
 353         # XXX(dcramer): GroupSerializer reimplements this logic
 354         from sentry.models import GroupSnooze
 355 
 356         status = self.status
 357 
 358         if status == GroupStatus.IGNORED:
 359             try:
 360                 snooze = GroupSnooze.objects.get_from_cache(group=self)
 361             except GroupSnooze.DoesNotExist:
 362                 pass
 363             else:
 364                 if not snooze.is_valid(group=self):
 365                     status = GroupStatus.UNRESOLVED
 366 
 367         if status == GroupStatus.UNRESOLVED and self.is_over_resolve_age():
 368             return GroupStatus.RESOLVED
 369         return status
