  45     @classmethod
  46     def get_ownership_cached(cls, project_id):
  47         """
  48         Cached read access to projectownership.
  49 
  50         This method implements a negative cache which saves us
  51         a pile of read queries in post_processing as most projects
  52         don't have ownership rules.
  53 
  54         See the post_save and post_delete signals below for additional
  55         cache updates.
  56         """
  57         cache_key = cls.get_cache_key(project_id)
  58         ownership = cache.get(cache_key)
  59         if ownership is None:
  60             try:
  61                 ownership = cls.objects.get(project_id=project_id)
  62             except cls.DoesNotExist:
  63                 ownership = False
  64             cache.set(cache_key, ownership, READ_CACHE_DURATION)
  65         return ownership or None
  66 
  67     @classmethod
  68     def get_owners(cls, project_id, data):
  69         """
  70         For a given project_id, and event data blob.
  71 
  72         If Everyone is returned, this means we implicitly are
  73         falling through our rules and everyone is responsible.
  74 
  75         If an empty list is returned, this means there are explicitly
  76         no owners.
  77         """
  78         ownership = cls.get_ownership_cached(project_id)
  79         if not ownership:
  80             ownership = cls(project_id=project_id)
  81 
  82         rules = cls._matching_ownership_rules(ownership, project_id, data)
  83         if not rules:
  84             return cls.Everyone if ownership.fallthrough else [], None
  85 
  86         owners = {o for rule in rules for o in rule.owners}
  87         owners_to_actors = resolve_actors(owners, project_id)
  88         ordered_actors = []
  89         for rule in rules:
  90             for o in rule.owners:
  91                 if o in owners and owners_to_actors.get(o) is not None:
  92                     ordered_actors.append(owners_to_actors[o])
  93                     owners.remove(o)
  94 
  95         return ordered_actors, rules

  97     @classmethod
  98     def get_autoassign_owner(cls, project_id, data):
  99         """
 100         Get the auto-assign owner for a project if there are any.
 101 
 102         Will return None if there are no owners, or a list of owners.
 103         """
 104         ownership = cls.get_ownership_cached(project_id)
 105         if not ownership or not ownership.auto_assignment:
 106             return None
 107 
 108         rules = cls._matching_ownership_rules(ownership, project_id, data)
 109         if not rules:
 110             return None
 111 
 112         score = 0
 113         owners = None
 114         # Automatic assignment prefers the owner with the longest
 115         # matching pattern as the match is more specific.
 116         for rule in rules:
 117             candidate = len(rule.matcher.pattern)
 118             if candidate > score:
 119                 score = candidate
 120                 owners = rule.owners
 121         actors = [_f for _f in resolve_actors(owners, project_id).values() if _f]
 122 
 123         # Can happen if the ownership rule references a user/team that no longer
 124         # is assigned to the project or has been removed from the org.
 125         if not actors:
 126             return None
 127         return actors[0].resolve()
 128 
 129     @classmethod
 130     def _matching_ownership_rules(cls, ownership, project_id, data):
 131         rules = []
 132         if ownership.schema is not None:
 133             for rule in load_schema(ownership.schema):
 134                 if rule.test(data):
 135                     rules.append(rule)
 136 
 137         return rules

 140 def resolve_actors(owners, project_id):
 141     """ Convert a list of Owner objects into a dictionary
 142     of {Owner: Actor} pairs. Actors not identified are returned
 143     as None. """
 144     from sentry.api.fields.actor import Actor
 145     from sentry.models import User, Team
 146 
 147     if not owners:
 148         return {}
 149 
 150     users, teams = [], []
 151     owners_lookup = {}
 152 
 153     for owner in owners:
 154         # teams aren't technical case insensitive, but teams also
 155         # aren't allowed to have non-lowercase in slugs, so
 156         # this kinda works itself out correctly since they won't match
 157         owners_lookup[(owner.type, owner.identifier.lower())] = owner
 158         if owner.type == "user":
 159             users.append(owner)
 160         elif owner.type == "team":
 161             teams.append(owner)
 162 
 163     actors = {}
 164     if users:
 165         actors.update(
 166             {
 167                 ("user", email.lower()): Actor(u_id, User)
 168                 for u_id, email in User.objects.filter(
 169                     reduce(operator.or_, [Q(emails__email__iexact=o.identifier) for o in users]),
 170                     # We don't require verified emails
 171                     # emails__is_verified=True,
 172                     is_active=True,
 173                     sentry_orgmember_set__organizationmemberteam__team__projectteam__project_id=project_id,
 174                 )
 175                 .distinct()
 176                 .values_list("id", "emails__email")
 177             }
 178         )
 179 
 180     if teams:
 181         actors.update(
 182             {
 183                 ("team", slug): Actor(t_id, Team)
 184                 for t_id, slug in Team.objects.filter(
 185                     slug__in=[o.identifier for o in teams], projectteam__project_id=project_id
 186                 ).values_list("id", "slug")
 187             }
 188         )
 189 
 190     return {o: actors.get((o.type, o.identifier.lower())) for o in owners}
