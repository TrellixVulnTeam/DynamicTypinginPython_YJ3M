  46 def get_user_options(key, user_ids, project, default):
  47     from sentry.models import UserOption
  48 
  49     options = {
  50         (option.user_id, option.project_id): option.value
  51         for option in UserOption.objects.filter(
  52             Q(project__isnull=True) | Q(project=project),
  53             user_id__in=user_ids,
  54             key="workflow:notifications",
  55         )
  56     }
  57 
  58     results = {}
  59 
  60     for user_id in user_ids:
  61         results[user_id] = options.get((user_id, project.id), options.get((user_id, None), default))
  62 
  63     return results

  66 class GroupSubscriptionManager(BaseManager):
  67     def subscribe(self, group, user, reason=GroupSubscriptionReason.unknown):
  68         """
  69         Subscribe a user to an issue, but only if the user has not explicitly
  70         unsubscribed.
  71         """
  72         try:
  73             with transaction.atomic():
  74                 self.create(
  75                     user=user, group=group, project=group.project, is_active=True, reason=reason
  76                 )
  77         except IntegrityError:
  78             pass
  79 
  80     def subscribe_actor(self, group, actor, reason=GroupSubscriptionReason.unknown):
  81         from sentry.models import User, Team
  82 
  83         if isinstance(actor, User):
  84             return self.subscribe(group, actor, reason)
  85         if isinstance(actor, Team):
  86             # subscribe the members of the team
  87             team_users_ids = list(actor.member_set.values_list("user_id", flat=True))
  88             return self.bulk_subscribe(group, team_users_ids, reason)
  89 
  90         raise NotImplementedError("Unknown actor type: %r" % type(actor))
  91 
  92     def bulk_subscribe(self, group, user_ids, reason=GroupSubscriptionReason.unknown):
  93         """
  94         Subscribe a list of user ids to an issue, but only if the users are not explicitly
  95         unsubscribed.
  96         """
  97         user_ids = set(user_ids)
  98 
  99         # 5 retries for race conditions where
 100         # concurrent subscription attempts cause integrity errors
 101         for i in range(4, -1, -1):  # 4 3 2 1 0
 102 
 103             existing_subscriptions = set(
 104                 GroupSubscription.objects.filter(
 105                     user_id__in=user_ids, group=group, project=group.project
 106                 ).values_list("user_id", flat=True)
 107             )
 108 
 109             subscriptions = [
 110                 GroupSubscription(
 111                     user_id=user_id,
 112                     group=group,
 113                     project=group.project,
 114                     is_active=True,
 115                     reason=reason,
 116                 )
 117                 for user_id in user_ids
 118                 if user_id not in existing_subscriptions
 119             ]
 120 
 121             try:
 122                 with transaction.atomic():
 123                     self.bulk_create(subscriptions)
 124                     return True
 125             except IntegrityError as e:
 126                 if i == 0:
 127                     raise e

 129     def get_participants(self, group):
 130         """
 131         Identify all users who are participating with a given issue.
 132         """
 133         from sentry.models import User, UserOptionValue
 134 
 135         users = {
 136             user.id: user
 137             for user in User.objects.filter(
 138                 sentry_orgmember_set__teams__in=group.project.teams.all(), is_active=True
 139             )
 140         }
 141 
 142         excluded_ids = set()
 143 
 144         subscriptions = {
 145             subscription.user_id: subscription
 146             for subscription in GroupSubscription.objects.filter(
 147                 group=group, user_id__in=users.keys()
 148             )
 149         }
 150 
 151         for user_id, subscription in subscriptions.items():
 152             if not subscription.is_active:
 153                 excluded_ids.add(user_id)
 154 
 155         options = get_user_options(
 156             "workflow:notifications",
 157             users.keys(),
 158             group.project,
 159             UserOptionValue.participating_only,
 160         )
 161 
 162         for user_id, option in options.items():
 163             if option == UserOptionValue.no_conversations:
 164                 excluded_ids.add(user_id)
 165             elif option == UserOptionValue.participating_only:
 166                 if user_id not in subscriptions:
 167                     excluded_ids.add(user_id)
 168 
 169         results = {}
 170 
 171         for user_id, user in users.items():
 172             if user_id in excluded_ids:
 173                 continue
 174 
 175             subscription = subscriptions.get(user_id)
 176             if subscription is not None:
 177                 results[user] = subscription.reason
 178             else:
 179                 results[user] = GroupSubscriptionReason.implicit
 180 
 181         return results

 184 class GroupSubscription(Model):
 185     """
 186     Identifies a subscription relationship between a user and an issue.
 187     """
 188 
 189     __core__ = False
 190 
 191     project = FlexibleForeignKey("sentry.Project", related_name="subscription_set")
 192     group = FlexibleForeignKey("sentry.Group", related_name="subscription_set")
 193     # namespace related_name on User since we don't own the model
 194     user = FlexibleForeignKey(settings.AUTH_USER_MODEL)
 195     is_active = models.BooleanField(default=True)
 196     reason = BoundedPositiveIntegerField(default=GroupSubscriptionReason.unknown)
 197     date_added = models.DateTimeField(default=timezone.now, null=True)
 198 
 199     objects = GroupSubscriptionManager()
 200 
 201     class Meta:
 202         app_label = "sentry"
 203         db_table = "sentry_groupsubscription"
 204         unique_together = (("group", "user"),)
 205 
 206     __repr__ = sane_repr("project_id", "group_id", "user_id")
