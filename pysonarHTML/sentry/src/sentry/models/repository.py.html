  13 class Repository(Model, PendingDeletionMixin):

  40     def has_integration_provider(self):
  41         return self.provider and self.provider.startswith("integrations:")
  42 
  43     def get_provider(self):
  44         from sentry.plugins.base import bindings
  45 
  46         if self.has_integration_provider():
  47             provider_cls = bindings.get("integration-repository.provider").get(self.provider)
  48             return provider_cls(self.provider)
  49 
  50         provider_cls = bindings.get("repository.provider").get(self.provider)
  51         return provider_cls(self.provider)
  52 
  53     def generate_delete_fail_email(self, error_message):
  54         from sentry.utils.email import MessageBuilder
  55 
  56         new_context = {
  57             "repo": self,
  58             "error_message": error_message,
  59             "provider_name": self.get_provider().name,
  60         }
  61 
  62         return MessageBuilder(
  63             subject="Unable to Delete Repository Webhooks",
  64             context=new_context,
  65             template="sentry/emails/unable-to-delete-repo.txt",
  66             html_template="sentry/emails/unable-to-delete-repo.html",
  67         )

  69     def rename_on_pending_deletion(self, fields=None):
  70         # Due to the fact that Repository is shown to the user
  71         # as it is pending deletion, this is added to display the fields
  72         # correctly to the user.
  73         self.config["pending_deletion_name"] = self.name
  74         super(Repository, self).rename_on_pending_deletion(fields, ["config"])
  75 
  76     def reset_pending_deletion_field_names(self):
  77         del self.config["pending_deletion_name"]
  78         super(Repository, self).reset_pending_deletion_field_names(["config"])

  81 def on_delete(instance, actor=None, **kwargs):
  82     # If there is no provider, we don't have any webhooks, etc to delete
  83     if not instance.provider:
  84         return
  85 
  86     # TODO(lb): I'm assuming that this is used by integrations... is it?
  87     def handle_exception(e):
  88         import six
  89         from sentry.exceptions import InvalidIdentity, PluginError
  90         from sentry.shared_integrations.exceptions import IntegrationError
  91 
  92         if isinstance(e, (IntegrationError, PluginError, InvalidIdentity)):
  93             error = six.text_type(e)
  94         else:
  95             error = "An unknown error occurred"
  96         if actor is not None:
  97             msg = instance.generate_delete_fail_email(error)
  98             msg.send_async(to=[actor.email])
  99 
 100     if instance.has_integration_provider():
 101         try:
 102             instance.get_provider().on_delete_repository(repo=instance)
 103         except Exception as exc:
 104             handle_exception(exc)
 105     else:
 106         try:
 107             instance.get_provider().delete_repository(repo=instance, actor=actor)
 108         except Exception as exc:
 109             handle_exception(exc)
