  16 # The order of these keys are significant to also indicate priority

  54     @classmethod

  56         return md5_text(value.split(":", 1)[-1]).hexdigest()

  58     @classmethod
  59     def for_tags(cls, project_id, values):
  60         """
  61         Finds matching EventUser objects from a list of tag values.
  62 
  63         Return a dictionary of {tag_value: event_user}.
  64         """
  65         hashes = [cls.hash_from_tag(v) for v in values]
  66         return {e.tag_value: e for e in cls.objects.filter(project_id=project_id, hash__in=hashes)}

 106     def find_similar_users(self, user):
 107         from sentry.models import OrganizationMemberTeam, Project
 108 
 109         # limit to only teams user has opted into
 110         project_ids = list(
 111             Project.objects.filter(
 112                 teams__in=OrganizationMemberTeam.objects.filter(
 113                     organizationmember__user=user,
 114                     organizationmember__organization__project=self.project_id,
 115                     is_active=True,
 116                 ).values("team")
 117             ).values_list("id", flat=True)[:1000]
 118         )
 119         if not project_ids:
 120             return type(self).objects.none()
 121 
 122         filters = []
 123         if self.email:
 124             filters.append(models.Q(email=self.email))
 125         if self.ip_address:
 126             filters.append(models.Q(ip_address=self.ip_address))
 127         if not filters:
 128             return type(self).objects.none()
 129         return (
 130             type(self)
 131             .objects.exclude(id=self.id)
 132             .filter(reduce(or_, filters), project_id__in=project_ids)
 133         )
