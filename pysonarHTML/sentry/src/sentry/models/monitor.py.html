  37 def get_next_schedule(base_datetime, schedule_type, schedule):
  38     if schedule_type == ScheduleType.CRONTAB:
  39         itr = croniter(schedule, base_datetime)
  40         next_schedule = itr.get_next(datetime)
  41     elif schedule_type == ScheduleType.INTERVAL:
  42         count, unit_name = schedule
  43         # count is the "number of units" and unit_name is the "unit name of interval"
  44         # which is inverse from what rrule calls them
  45         rule = rrule.rrule(
  46             freq=SCHEDULE_INTERVAL_MAP[unit_name], interval=count, dtstart=base_datetime, count=2
  47         )
  48         if rule[0] > base_datetime:
  49             next_schedule = rule[0]
  50         else:
  51             next_schedule = rule[1]
  52     else:
  53         raise NotImplementedError("unknown schedule_type")
  54 
  55     return next_schedule

 159     def get_next_scheduled_checkin(self, last_checkin=None):
 160         if last_checkin is None:
 161             last_checkin = self.last_checkin
 162         tz = pytz.timezone(self.config.get("timezone") or "UTC")
 163         schedule_type = self.config.get("schedule_type", ScheduleType.CRONTAB)
 164         base_datetime = last_checkin.astimezone(tz)
 165         next_checkin = get_next_schedule(base_datetime, schedule_type, self.config["schedule"])
 166         return next_checkin + timedelta(minutes=int(self.config.get("checkin_margin") or 0))

 168     def mark_failed(self, last_checkin=None, reason=MonitorFailure.UNKNOWN):
 169         from sentry.coreapi import ClientApiHelper
 170         from sentry.event_manager import EventManager
 171         from sentry.models import Project
 172         from sentry.signals import monitor_failed
 173 
 174         if last_checkin is None:
 175             next_checkin_base = timezone.now()
 176             last_checkin = self.last_checkin or timezone.now()
 177         else:
 178             next_checkin_base = last_checkin
 179 
 180         affected = (
 181             type(self)
 182             .objects.filter(
 183                 Q(last_checkin__lte=last_checkin) | Q(last_checkin__isnull=True), id=self.id
 184             )
 185             .update(
 186                 next_checkin=self.get_next_scheduled_checkin(next_checkin_base),
 187                 status=MonitorStatus.ERROR,
 188                 last_checkin=last_checkin,
 189             )
 190         )
 191         if not affected:
 192             return False
 193 
 194         event_manager = EventManager(
 195             {
 196                 "logentry": {"message": "Monitor failure: %s (%s)" % (self.name, reason)},
 197                 "contexts": {"monitor": get_monitor_context(self)},
 198                 "fingerprint": ["monitor", six.text_type(self.guid), reason],
 199             },
 200             project=Project(id=self.project_id),
 201         )
 202         event_manager.normalize()
 203         data = event_manager.get_data()
 204         helper = ClientApiHelper(project_id=self.project_id)
 205         helper.insert_data_to_database(data)
 206         monitor_failed.send(monitor=self, sender=type(self))
 207         return True
