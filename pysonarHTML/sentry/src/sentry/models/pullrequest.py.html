  11 class PullRequest(Model):

  14     organization_id = BoundedPositiveIntegerField(db_index=True)

  34     def find_referenced_groups(self):
  35         text = u"{} {}".format(self.message, self.title)
  36         return find_referenced_groups(text, self.organization_id)

  38     @classmethod
  39     def create_or_save(cls, organization_id, repository_id, key, values):
  40         """
  41         Wraps create_or_update and ensures post_save signals are fired
  42         for updated records as GroupLink functionality is dependent
  43         on signals being fired.
  44         """
  45         affected, created = cls.objects.create_or_update(
  46             organization_id=organization_id, repository_id=repository_id, key=key, values=values
  47         )
  48         if created is False:
  49             instance = cls.objects.get(
  50                 organization_id=organization_id, repository_id=repository_id, key=key
  51             )
  52             post_save.send(sender=cls, instance=instance, created=created)
  53         return affected, created
