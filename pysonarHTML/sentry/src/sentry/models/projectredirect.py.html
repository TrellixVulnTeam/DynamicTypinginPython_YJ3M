   9 class ProjectRedirect(Model):
  10     __core__ = True
  11 
  12     redirect_slug = models.SlugField(db_index=True)
  13     project = FlexibleForeignKey("sentry.Project")
  14     organization = FlexibleForeignKey("sentry.Organization")
  15     date_added = models.DateTimeField(default=timezone.now)
  16 
  17     class Meta:
  18         app_label = "sentry"
  19         db_table = "sentry_projectredirect"
  20         unique_together = (("organization", "redirect_slug"),)
  21 
  22     @classmethod
  23     def record(cls, project, historic_slug):
  24         """
  25         Records a historic slug used for redirect purposes. Overwrites
  26         historic redirect slugs from previous projects
  27         """
  28         redirect, created = cls.objects.get_or_create(
  29             redirect_slug=historic_slug,
  30             organization=project.organization,
  31             defaults={"project": project},
  32         )
  33 
  34         if not created:
  35             redirect.update(project=project)
