   9 class GroupResolution(Model):
  10     """
  11     Describes when a group was marked as resolved.
  12     """
  13 
  14     __core__ = False
  15 
  16     class Type:
  17         in_release = 0
  18         in_next_release = 1
  19 
  20     class Status:
  21         pending = 0
  22         resolved = 1
  23 
  24     group = FlexibleForeignKey("sentry.Group", unique=True)
  25     # the release in which its suggested this was resolved
  26     # which allows us to indicate if it still happens in newer versions
  27     release = FlexibleForeignKey("sentry.Release")
  28     type = BoundedPositiveIntegerField(
  29         choices=((Type.in_next_release, "in_next_release"), (Type.in_release, "in_release")),
  30         null=True,
  31     )
  32     actor_id = BoundedPositiveIntegerField(null=True)
  33     datetime = models.DateTimeField(default=timezone.now, db_index=True)
  34     status = BoundedPositiveIntegerField(
  35         default=Status.pending,
  36         choices=((Status.pending, _("Pending")), (Status.resolved, _("Resolved"))),
  37     )
  38 
  39     class Meta:
  40         db_table = "sentry_groupresolution"
  41         app_label = "sentry"
  42 
  43     __repr__ = sane_repr("group_id", "release_id")
  44 
  45     @classmethod
  46     def has_resolution(cls, group, release):
  47         """
  48         Determine if a resolution exists for the given group and release.
  49 
  50         This is used to suggest if a regression has occurred.
  51         """
  52         try:
  53             res_type, res_release, res_release_datetime = (
  54                 cls.objects.filter(group=group)
  55                 .select_related("release")
  56                 .values_list("type", "release__id", "release__date_added")[0]
  57             )
  58         except IndexError:
  59             return False
  60 
  61         # if no release is present, we assume we've gone from "no release" to "some release"
  62         # in application configuration, and thus this must be older
  63         if not release:
  64             return True
  65 
  66         if res_type in (None, cls.Type.in_next_release):
  67             if res_release == release.id:
  68                 return True
  69             elif res_release_datetime > release.date_added:
  70                 return True
  71             return False
  72         elif res_type == cls.Type.in_release:
  73             if res_release == release.id:
  74                 return False
  75             if res_release_datetime < release.date_added:
  76                 return False
  77             return True
  78         else:
  79             raise NotImplementedError
