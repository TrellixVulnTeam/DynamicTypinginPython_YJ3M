  18     environment = models.CharField(max_length=64, default="")

  27     __repr__ = sane_repr("group_id", "release_id")

  29     @classmethod
  30     def get_cache_key(cls, group_id, release_id, environment):
  31         return u"grouprelease:1:{}:{}".format(
  32             group_id, md5_text(u"{}:{}".format(release_id, environment)).hexdigest()
  33         )
  34 
  35     @classmethod
  36     def get_or_create(cls, group, release, environment, datetime, **kwargs):
  37         cache_key = cls.get_cache_key(group.id, release.id, environment.name)
  38 
  39         instance = cache.get(cache_key)
  40         if instance is None:
  41             try:
  42                 with transaction.atomic():
  43                     instance, created = (
  44                         cls.objects.create(
  45                             release_id=release.id,
  46                             group_id=group.id,
  47                             environment=environment.name,
  48                             project_id=group.project_id,
  49                             first_seen=datetime,
  50                             last_seen=datetime,
  51                         ),
  52                         True,
  53                     )
  54             except IntegrityError:
  55                 instance, created = (
  56                     cls.objects.get(
  57                         release_id=release.id, group_id=group.id, environment=environment.name
  58                     ),
  59                     False,
  60                 )
  61             cache.set(cache_key, instance, 3600)
  62         else:
  63             created = False
  64 
  65         # TODO(dcramer): this would be good to buffer, but until then we minimize
  66         # updates to once a minute, and allow Postgres to optimistically skip
  67         # it even if we can't
  68         if not created and instance.last_seen < datetime - timedelta(seconds=60):
  69             cls.objects.filter(
  70                 id=instance.id, last_seen__lt=datetime - timedelta(seconds=60)
  71             ).update(last_seen=datetime)
  72             instance.last_seen = datetime
  73             cache.set(cache_key, instance, 3600)
  74         return instance
