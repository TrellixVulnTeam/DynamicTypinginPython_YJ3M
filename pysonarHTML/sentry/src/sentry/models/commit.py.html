  11 class Commit(Model):
  12     __core__ = False
  13 
  14     organization_id = BoundedPositiveIntegerField(db_index=True)

  17     date_added = models.DateTimeField(default=timezone.now)

  23     class Meta:
  24         app_label = "sentry"

  26         index_together = (("repository_id", "date_added"),)

  28 
  29     __repr__ = sane_repr("organization_id", "repository_id", "key")
  30 
  31     @memoize
  32     def title(self):
  33         if not self.message:
  34             return ""
  35         return self.message.splitlines()[0]
  36 
  37     @memoize
  38     def short_id(self):
  39         if len(self.key) == 40:
  40             return self.key[:7]
  41         return self.key
  42 
  43     def find_referenced_groups(self):
  44         return find_referenced_groups(self.message, self.organization_id)
