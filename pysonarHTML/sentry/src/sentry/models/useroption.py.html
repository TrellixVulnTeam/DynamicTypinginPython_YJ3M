  25 class UserOptionManager(OptionManager):
  26     def _make_key(self, user, project=None, organization=None):
  27         if project:
  28             metakey = u"%s:%s:project" % (user.pk, project.id)
  29         elif organization:
  30             metakey = u"%s:%s:organization" % (user.pk, organization.id)
  31         else:
  32             metakey = u"%s:user" % (user.pk)
  33 
  34         return super(UserOptionManager, self)._make_key(metakey)
  35 
  36     def get_value(self, user, key, default=None, **kwargs):
  37         project = kwargs.get("project")
  38         organization = kwargs.get("organization")
  39 
  40         if organization and project:
  41             raise NotImplementedError(option_scope_error)
  42         if organization:
  43             result = self.get_all_values(user, None, organization)
  44         else:
  45             result = self.get_all_values(user, project)
  46         return result.get(key, default)
  47 
  48     def unset_value(self, user, project, key):
  49         # this isn't implemented for user-organization scoped options yet, because
  50         # it hasn't been needed
  51         self.filter(user=user, project=project, key=key).delete()
  52         if not hasattr(self, "_metadata"):
  53             return
  54 
  55         metakey = self._make_key(user, project=project)
  56 
  57         if metakey not in self._option_cache:
  58             return
  59         self._option_cache[metakey].pop(key, None)
  60 
  61     def set_value(self, user, key, value, **kwargs):
  62         project = kwargs.get("project")
  63         organization = kwargs.get("organization")
  64 
  65         if organization and project:
  66             raise NotImplementedError(option_scope_error)
  67 
  68         inst, created = self.get_or_create(
  69             user=user,
  70             project=project,
  71             organization=organization,
  72             key=key,
  73             defaults={"value": value},
  74         )
  75         if not created and inst.value != value:
  76             inst.update(value=value)
  77 
  78         metakey = self._make_key(user, project=project, organization=organization)
  79 
  80         if metakey not in self._option_cache:
  81             return
  82         self._option_cache[metakey][key] = value
  83 
  84     def get_all_values(self, user, project=None, organization=None, force_reload=False):
  85         if organization and project:
  86             raise NotImplementedError(option_scope_error)
  87 
  88         metakey = self._make_key(user, project=project, organization=organization)
  89 
  90         if metakey not in self._option_cache or force_reload:
  91             result = dict(
  92                 (i.key, i.value)
  93                 for i in self.filter(user=user, project=project, organization=organization)
  94             )
  95             self._option_cache[metakey] = result
  96         return self._option_cache.get(metakey, {})
  97 
  98     def post_save(self, instance, **kwargs):
  99         self.get_all_values(
 100             instance.user, instance.project, instance.organization, force_reload=True
 101         )
 102 
 103     def post_delete(self, instance, **kwargs):
 104         self.get_all_values(
 105             instance.user, instance.project, instance.organization, force_reload=True
 106         )

 111 class UserOption(Model):
 112     """
 113     User options apply only to a user, and optionally a project OR an organization.
 114 
 115     Options which are specific to a plugin should namespace
 116     their key. e.g. key='myplugin:optname'
 117 
 118     Keeping user feature state
 119     key: "feature:assignment"
 120     value: { updated: datetime, state: bool }
 121     """
 122 
 123     __core__ = True
 124 
 125     user = FlexibleForeignKey(settings.AUTH_USER_MODEL)
 126     project = FlexibleForeignKey("sentry.Project", null=True)
 127     organization = FlexibleForeignKey("sentry.Organization", null=True)
 128     key = models.CharField(max_length=64)
 129     value = EncryptedPickledObjectField()
 130 
 131     objects = UserOptionManager()
 132 
 133     class Meta:
 134         app_label = "sentry"
 135         db_table = "sentry_useroption"
 136         unique_together = (("user", "project", "key"), ("user", "organization", "key"))
 137 
 138     __repr__ = sane_repr("user_id", "project_id", "organization_id", "key", "value")
