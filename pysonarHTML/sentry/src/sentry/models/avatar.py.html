  49     def get_cached_photo(self, size):
  50         if not self.file:
  51             return
  52         if size not in self.ALLOWED_SIZES:
  53             size = min(self.ALLOWED_SIZES, key=lambda x: abs(x - size))
  54         cache_key = self.get_cache_key(size)
  55         photo = cache.get(cache_key)
  56         if photo is None:
  57             photo_file = self.file.getfile()
  58             with Image.open(photo_file) as image:
  59                 image = image.resize((size, size), Image.LANCZOS)
  60                 image_file = BytesIO()
  61                 image.save(image_file, "PNG")
  62                 photo = image_file.getvalue()
  63                 cache.set(cache_key, photo)
  64         return photo
