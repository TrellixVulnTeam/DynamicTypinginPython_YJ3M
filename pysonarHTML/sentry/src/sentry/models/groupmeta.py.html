  49     def populate_cache(self, instance_list):
  50         for group in instance_list:
  51             self.__cache.setdefault(group.id, {})
  52 
  53         results = self.filter(group__in=instance_list).values_list("group", "key", "value")
  54         for group_id, key, value in results:
  55             self.__cache[group_id][key] = value

  57     def get_value_bulk(self, instance_list, key, default=None):
  58         results = {}
  59         for instance in instance_list:
  60             try:
  61                 inst_cache = self.__cache[instance.id]
  62             except KeyError:
  63                 raise self.model.CacheNotPopulated(ERR_CACHE_MISISNG % (instance.id,))
  64             results[instance] = inst_cache.get(key, default)
