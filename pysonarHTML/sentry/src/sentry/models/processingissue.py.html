  11 def get_processing_issue_checksum(scope, object):
  12     h = sha1()
  13     h.update(scope.encode("utf-8") + b"\x00")
  14     h.update(object.encode("utf-8") + b"\x00")
  15     return h.hexdigest()

  18 class ProcessingIssueManager(BaseManager):
  19     def with_num_events(self):
  20         return self.annotate(num_events=Count("eventprocessingissue"))
  21 
  22     def resolve_processing_issue(self, project, scope, object, type=None):
  23         """Resolves the given processing issues.  If not type is given
  24         all processing issues for scope and object are resolved regardless
  25         of the type.
  26         """
  27         checksum = get_processing_issue_checksum(scope, object)
  28         q = ProcessingIssue.objects.filter(project=project, checksum=checksum)
  29         if type is not None:
  30             q = q.filter(type=type)
  31         q.delete()
  32 
  33     def resolve_all_processing_issue(self, project):
  34         """
  35         Resolves all processing issues.
  36         """
  37         q = ProcessingIssue.objects.filter(project=project)
  38         q.delete()
  39 
  40     def discard_all_processing_issue(self, project):
  41         """
  42         Resolves all processing issues.
  43         """
  44         self.resolve_all_processing_issue(project)
  45         from sentry.models import RawEvent, ReprocessingReport
  46 
  47         RawEvent.objects.filter(project_id=project.id).delete()
  48         ReprocessingReport.objects.filter(project_id=project.id).delete()
  49 
  50     def find_resolved_queryset(self, project_ids):
  51         from sentry.models import RawEvent
  52 
  53         return RawEvent.objects.filter(
  54             project_id__in=project_ids, eventprocessingissue__isnull=True
  55         )

  57     def find_resolved(self, project_id, limit=100):
  58         """Returns a list of raw events that generally match the given
  59         processing issue and no longer have any issues remaining.  Returns
  60         a list of raw events that are now resolved and a bool that indicates
  61         if there are more.
  62         """
  63         from sentry import eventstore
  64 
  65         rv = list(self.find_resolved_queryset([project_id])[:limit])
  66         if len(rv) > limit:
  67             rv = rv[:limit]
  68             has_more = True
  69         else:
  70             has_more = False
  71 
  72         rv = list(rv)
  73         eventstore.bind_nodes(rv, "data")
  74         return rv, has_more

  76     def record_processing_issue(self, raw_event, scope, object, type, data=None):
  77         """Records a new processing issue for the given raw event."""
  78         data = dict(data or {})
  79         checksum = get_processing_issue_checksum(scope, object)
  80         data["_scope"] = scope
  81         data["_object"] = object
  82         issue, _ = ProcessingIssue.objects.get_or_create(
  83             project_id=raw_event.project_id, checksum=checksum, type=type, defaults=dict(data=data)
  84         )
  85         ProcessingIssue.objects.filter(pk=issue.id).update(datetime=timezone.now())
  86         # In case the issue moved away from unresolved we want to make
  87         # sure it's back to unresolved
  88         EventProcessingIssue.objects.get_or_create(raw_event=raw_event, processing_issue=issue)
