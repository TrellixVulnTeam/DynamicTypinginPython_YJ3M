  11 class GroupEnvironment(Model):
  12     __core__ = False
  13 
  14     group = FlexibleForeignKey("sentry.Group", db_constraint=False)
  15     environment = FlexibleForeignKey("sentry.Environment", db_constraint=False)
  16     first_release = FlexibleForeignKey(
  17         "sentry.Release",
  18         null=True,
  19         db_constraint=False,
  20         # We have no index here, so we don't want to use the ORM's cascade
  21         # delete functionality
  22         on_delete=DO_NOTHING,
  23     )
  24     first_seen = DateTimeField(default=timezone.now, db_index=True, null=True)
  25 
  26     class Meta:
  27         app_label = "sentry"
  28         db_table = "sentry_groupenvironment"
  29         index_together = [("environment", "first_release")]
  30         unique_together = [("group", "environment")]
  31 
  32     __repr__ = sane_repr("group_id", "environment_id")
  33 
  34     @classmethod
  35     def _get_cache_key(self, group_id, environment_id):
  36         return u"groupenv:1:{}:{}".format(group_id, environment_id)
  37 
  38     @classmethod
  39     def get_or_create(cls, group_id, environment_id, defaults=None):
  40         cache_key = cls._get_cache_key(group_id, environment_id)
  41         instance = cache.get(cache_key)
  42         if instance is None:
  43             instance, created = cls.objects.get_or_create(
  44                 group_id=group_id, environment_id=environment_id, defaults=defaults
  45             )
  46             cache.set(cache_key, instance, 3600)
  47         else:
  48             created = False
  49 
  50         return instance, created
  51
