   6 from sentry.db.models import BoundedPositiveIntegerField, FlexibleForeignKey, Model

   9 class Feature(object):
  10     API = 0
  11     ISSUE_LINK = 1
  12     STACKTRACE_LINK = 2
  13     EVENT_HOOKS = 3
  14     PROJECT_MANAGEMENT = 4
  15     INCIDENT_MANAGEMENT = 5
  16     FEATURE_FLAG = 6
  17 
  18     @classmethod
  19     def as_choices(cls):
  20         return (
  21             (cls.API, "integrations-api"),
  22             (cls.ISSUE_LINK, "integrations-issue-link"),
  23             (cls.STACKTRACE_LINK, "integrations-stacktrace-link"),
  24             (cls.EVENT_HOOKS, "integrations-event-hooks"),
  25             (cls.PROJECT_MANAGEMENT, "integrations-project-management"),
  26             (cls.INCIDENT_MANAGEMENT, "integrations-incident-management"),
  27             (cls.FEATURE_FLAG, "integrations-feature-flag"),
  28         )
  29 
  30     @classmethod
  31     def as_str(cls, feature):
  32         if feature == cls.ISSUE_LINK:
  33             return "integrations-issue-link"
  34         if feature == cls.STACKTRACE_LINK:
  35             return "integrations-stacktrace-link"
  36         if feature == cls.EVENT_HOOKS:
  37             return "integrations-event-hooks"
  38         if feature == cls.PROJECT_MANAGEMENT:
  39             return "integrations-project-management"
  40         if feature == cls.INCIDENT_MANAGEMENT:
  41             return "integrations-incident-management"
  42         if feature == cls.FEATURE_FLAG:
  43             return "integrations-feature-flag"
  44         return "integrations-api"
  45 
  46     @classmethod
  47     def description(cls, feature, name):
  48         if feature == cls.PROJECT_MANAGEMENT:
  49             return "Create or link issues in %s from Sentry issue groups." % name
  50         if feature == cls.INCIDENT_MANAGEMENT:
  51             return "Manage incidents and outages by sending Sentry notifications to %s." % name
  52         if feature == cls.FEATURE_FLAG:
  53             return "Improve visibility into feature flagging by sending Sentry errors to %s." % name
  54         if feature == cls.ISSUE_LINK:
  55             return "Organizations can **create or link Sentry issues** to another service."
  56         if feature == cls.STACKTRACE_LINK:
  57             return "Organizations can **open a line to Sentry's stack trace** in another service."
  58         if feature == cls.EVENT_HOOKS:
  59             return "%s allows organizations to **forward events to another service**." % name
  60         # default
  61         return (
  62             "%s can **utilize the Sentry API** to pull data or update resources in Sentry (with permissions granted, of course)."
  63             % name
  64         )

  67 class IntegrationFeature(Model):
  68     __core__ = False
  69 
  70     sentry_app = FlexibleForeignKey("sentry.SentryApp")
  71     user_description = models.TextField(null=True)
  72     feature = BoundedPositiveIntegerField(default=0, choices=Feature.as_choices())
  73     date_added = models.DateTimeField(default=timezone.now)
  74 
  75     class Meta:
  76         app_label = "sentry"
  77         db_table = "sentry_integrationfeature"
  78         unique_together = (("sentry_app", "feature"),)
  79 
  80     def feature_str(self):
  81         return Feature.as_str(self.feature)
  82 
  83     @property
  84     def description(self):
  85         if self.user_description:
  86             return self.user_description
  87         return Feature.description(self.feature, self.sentry_app.name)
