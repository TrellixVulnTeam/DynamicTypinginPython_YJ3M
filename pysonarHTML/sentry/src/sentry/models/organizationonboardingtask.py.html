  37 # NOTE: data fields for some event types are as follows:
  38 #
  39 #   FIRST_EVENT:      { 'platform':  'flask', }
  40 #   INVITE_MEMBER:    { 'invited_member': user.id, 'teams': [team.id] }
  41 #   ISSUE_TRACKER:    { 'plugin': 'plugin_name' }
  42 #   ISSUE_ASSIGNMENT: { 'assigned_member': user.id }
  43 #   SECOND_PLATFORM:  { 'platform': 'javascript' }
  44 #
  45 # NOTE: Currently the `PENDING` status is applicable for the following
  46 # onboarding tasks:
  47 #
  48 #   FIRST_EVENT:     User confirms that sdk has been installed
  49 #   INVITE_MEMBER:   Until the member has successfully joined org
  50 #   SECOND_PLATFORM: User confirms that sdk has been installed
  51 #   USER_CONTEXT:    User has added user context to sdk
  52 #   ISSUE_TRACKER:   Tracker added, issue not yet created
  53

  55 class OrganizationOnboardingTaskManager(BaseManager):
  56     def record(self, organization_id, task, **kwargs):
  57         cache_key = "organizationonboardingtask:%s:%s" % (organization_id, task)
  58         if cache.get(cache_key) is None:
  59             try:
  60                 with transaction.atomic():
  61                     self.create(organization_id=organization_id, task=task, **kwargs)
  62                     return True
  63             except IntegrityError:
  64                 pass
  65 
  66             # Store marker to prevent running all the time
  67             cache.set(cache_key, 1, 3600)
  68 
  69         return False
  70

  72 class OrganizationOnboardingTask(Model):
  73     """
  74     Onboarding tasks walk new Sentry orgs through basic features of Sentry.
  75     """
  76 
  77     __core__ = False
  78 
  79     TASK_CHOICES = (
  80         (OnboardingTask.FIRST_PROJECT, "create_project"),
  81         (OnboardingTask.FIRST_EVENT, "send_first_event"),
  82         (OnboardingTask.INVITE_MEMBER, "invite_member"),
  83         (OnboardingTask.SECOND_PLATFORM, "setup_second_platform"),
  84         (OnboardingTask.USER_CONTEXT, "setup_user_context"),
  85         (OnboardingTask.RELEASE_TRACKING, "setup_release_tracking"),
  86         (OnboardingTask.SOURCEMAPS, "setup_sourcemaps"),
  87         (OnboardingTask.USER_REPORTS, "setup_user_reports"),
  88         (OnboardingTask.ISSUE_TRACKER, "setup_issue_tracker"),
  89         (OnboardingTask.ALERT_RULE, "setup_alert_rules"),
  90     )
  91 
  92     STATUS_CHOICES = (
  93         (OnboardingTaskStatus.COMPLETE, "complete"),
  94         (OnboardingTaskStatus.PENDING, "pending"),
  95         (OnboardingTaskStatus.SKIPPED, "skipped"),
  96     )
  97 
  98     # Used in the API to map IDs to string keys. This keeps things
  99     # a bit more maintainable on the frontend.
 100     TASK_KEY_MAP = dict(TASK_CHOICES)
 101     TASK_LOOKUP_BY_KEY = {v: k for k, v in TASK_CHOICES}
 102 
 103     STATUS_KEY_MAP = dict(STATUS_CHOICES)
 104     STATUS_LOOKUP_BY_KEY = {v: k for k, v in STATUS_CHOICES}
 105 
 106     # Tasks which must be completed for the onboarding to be considered
 107     # complete.
 108     REQUIRED_ONBOARDING_TASKS = frozenset(
 109         [
 110             OnboardingTask.FIRST_PROJECT,
 111             OnboardingTask.FIRST_EVENT,
 112             OnboardingTask.INVITE_MEMBER,
 113             OnboardingTask.SECOND_PLATFORM,
 114             OnboardingTask.USER_CONTEXT,
 115             OnboardingTask.RELEASE_TRACKING,
 116             OnboardingTask.SOURCEMAPS,
 117             OnboardingTask.ISSUE_TRACKER,
 118             OnboardingTask.ALERT_RULE,
 119         ]
 120     )
 121 
 122     SKIPPABLE_TASKS = frozenset(
 123         [
 124             OnboardingTask.INVITE_MEMBER,
 125             OnboardingTask.SECOND_PLATFORM,
 126             OnboardingTask.USER_CONTEXT,
 127             OnboardingTask.RELEASE_TRACKING,
 128             OnboardingTask.SOURCEMAPS,
 129             OnboardingTask.USER_REPORTS,
 130             OnboardingTask.ISSUE_TRACKER,
 131             OnboardingTask.ALERT_RULE,
 132         ]
 133     )
 134 
 135     organization = FlexibleForeignKey("sentry.Organization")
 136     user = FlexibleForeignKey(settings.AUTH_USER_MODEL, null=True)  # user that completed
 137     task = BoundedPositiveIntegerField(choices=TASK_CHOICES)
