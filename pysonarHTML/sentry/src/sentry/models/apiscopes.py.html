  23     def __init__(self):
  24         self.scopes = (

  45 class HasApiScopes(models.Model):
  46     """
  47     Mixin for models that hold a list of OAuth Scopes.
  48     """
  49 
  50     class Meta:
  51         abstract = True
  52 
  53     # List of scopes in bit form
  54     scopes = BitField(flags=ApiScopes().to_bitfield())
  55 
  56     # Human readable list of scopes
  57     scope_list = ArrayField(of=models.TextField)

  59     def get_scopes(self):
  60         if self.scope_list:
  61             return self.scope_list
  62         return [k for k, v in six.iteritems(self.scopes) if v]
  63 
  64     def has_scope(self, scope):
