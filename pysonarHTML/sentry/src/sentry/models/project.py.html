  50 class ProjectManager(BaseManager):
  51     # TODO(dcramer): we might want to cache this per user
  52     def get_for_user(self, team, user, scope=None, _skip_team_check=False):
  53         from sentry.models import Team
  54 
  55         if not (user and user.is_authenticated()):
  56             return []
  57 
  58         if not _skip_team_check:
  59             team_list = Team.objects.get_for_user(
  60                 organization=team.organization, user=user, scope=scope
  61             )
  62 
  63             try:
  64                 team = team_list[team_list.index(team)]
  65             except ValueError:
  66                 logging.info("User does not have access to team: %s", team.id)
  67                 return []
  68 
  69         base_qs = self.filter(teams=team, status=ProjectStatus.VISIBLE)
  70 
  71         project_list = []
  72         for project in base_qs:
  73             project_list.append(project)
  74 
  75         return sorted(project_list, key=lambda x: x.name.lower())

 201     def has_access(self, user, access=None):
 202         from sentry.models import AuthIdentity, OrganizationMember
 203 
 204         warnings.warn("Project.has_access is deprecated.", DeprecationWarning)
 205 
 206         queryset = self.member_set.filter(user=user)
 207 
 208         if access is not None:
 209             queryset = queryset.filter(type__lte=access)
 210 
 211         try:
 212             member = queryset.get()
 213         except OrganizationMember.DoesNotExist:
 214             return False
 215 
 216         try:
 217             auth_identity = AuthIdentity.objects.get(
 218                 auth_provider__organization=self.organization_id, user=member.user_id
 219             )
 220         except AuthIdentity.DoesNotExist:
 221             return True

 223         return auth_identity.is_valid(member)

 237     def get_member_alert_settings(self, user_option):
 238         """
 239         Returns a list of users who have alert notifications explicitly
 240         enabled/disabled.
 241         :param user_option: alert option key, typically 'mail:alert'
 242         :return: A dictionary in format {<user_id>: <int_alert_value>}
 243         """
 244         from sentry.models import UserOption
 245 
 246         return {
 247             o.user_id: int(o.value)
 248             for o in UserOption.objects.filter(project=self, key=user_option)
 249         }

 251     def get_notification_recipients(self, user_option):
 252         from sentry.models import UserOption
 253 
 254         alert_settings = self.get_member_alert_settings(user_option)
 255         disabled = set(u for u, v in six.iteritems(alert_settings) if v == 0)
 256 
 257         member_set = set(self.member_set.exclude(user__in=disabled).values_list("user", flat=True))
 258 
 259         # determine members default settings
 260         members_to_check = set(u for u in member_set if u not in alert_settings)
 261         if members_to_check:
 262             disabled = set(
 263                 (
 264                     uo.user_id
 265                     for uo in UserOption.objects.filter(
 266                         key="subscribe_by_default", user__in=members_to_check
 267                     )
 268                     if uo.value == "0"
 269                 )
 270             )
 271             member_set = [x for x in member_set if x not in disabled]
 272 
 273         return member_set

 293     def filter_to_subscribed_users(self, users):
 294         """
 295         Filters a list of users down to the users who are subscribed to email alerts. We
 296         check both the project level settings and global default settings.
 297         """
 298         from sentry.models import UserOption
 299 
 300         project_options = UserOption.objects.filter(
 301             user__in=users, project=self, key="mail:alert"
 302         ).values_list("user_id", "value")
 303 
 304         user_settings = {user_id: value for user_id, value in project_options}
 305         users_without_project_setting = [user for user in users if user.id not in user_settings]
 306         if users_without_project_setting:
 307             user_default_settings = {
 308                 user_id: value
 309                 for user_id, value in UserOption.objects.filter(
 310                     user__in=users_without_project_setting,
 311                     key="subscribe_by_default",
 312                     project__isnull=True,
 313                 ).values_list("user_id", "value")
 314             }
 315             for user in users_without_project_setting:
 316                 user_settings[user.id] = int(user_default_settings.get(user.id, "1"))
 317 
 318         return [user for user in users if bool(user_settings[user.id])]

 320     def transfer_to(self, team=None, organization=None):
 321         # NOTE: this will only work properly if the new team is in a different
 322         # org than the existing one, which is currently the only use case in
 323         # production
 324         # TODO(jess): refactor this to make it an org transfer only
 325         from sentry.models import (
 326             Environment,
 327             EnvironmentProject,
 328             ProjectTeam,
 329             ReleaseProject,
 330             ReleaseProjectEnvironment,
 331             Rule,
 332         )
 333 
 334         if organization is None:
 335             organization = team.organization
 336 
 337         old_org_id = self.organization_id
 338         org_changed = old_org_id != organization.id
 339 
 340         self.organization = organization
 341 
 342         try:
 343             with transaction.atomic():
 344                 self.update(organization=organization)
 345         except IntegrityError:
 346             slugify_instance(self, self.name, organization=organization)
 347             self.update(slug=self.slug, organization=organization)
 348 
 349         # Both environments and releases are bound at an organization level.
 350         # Due to this, when you transfer a project into another org, we have to
 351         # handle this behavior somehow. We really only have two options here:
 352         # * Copy over all releases/environments into the new org and handle de-duping
 353         # * Delete the bindings and let them reform with new data.
 354         # We're generally choosing to just delete the bindings since new data
 355         # flowing in will recreate links correctly. The tradeoff is that
 356         # historical data is lost, but this is a compromise we're willing to
 357         # take and a side effect of allowing this feature. There are exceptions
 358         # to this however, such as rules, which should maintain their
 359         # configuration when moved across organizations.
 360         if org_changed:
 361             for model in ReleaseProject, ReleaseProjectEnvironment, EnvironmentProject:
 362                 model.objects.filter(project_id=self.id).delete()
 363             # this is getting really gross, but make sure there aren't lingering associations
 364             # with old orgs or teams
 365             ProjectTeam.objects.filter(project=self, team__organization_id=old_org_id).delete()
 366 
 367         rules_by_environment_id = defaultdict(set)
 368         for rule_id, environment_id in Rule.objects.filter(
 369             project_id=self.id, environment_id__isnull=False
 370         ).values_list("id", "environment_id"):
 371             rules_by_environment_id[environment_id].add(rule_id)
 372 
 373         environment_names = dict(
 374             Environment.objects.filter(id__in=rules_by_environment_id).values_list("id", "name")
 375         )
 376 
 377         for environment_id, rule_ids in rules_by_environment_id.items():
 378             Rule.objects.filter(id__in=rule_ids).update(
 379                 environment_id=Environment.get_or_create(self, environment_names[environment_id]).id
 380             )
 381 
 382         # ensure this actually exists in case from team was null
 383         if team is not None:
 384             self.add_team(team)
 385

 398     def get_security_token(self):
 399         lock = locks.get(self.get_lock_key(), duration=5)
 400         with TimedRetryPolicy(10)(lock.acquire):
 401             security_token = self.get_option("sentry:token", None)
 402             if security_token is None:
 403                 security_token = uuid1().hex
 404                 self.update_option("sentry:token", security_token)

 410     def copy_settings_from(self, project_id):
 411         """
 412         Copies project level settings of the inputted project
 413         - General Settings
 414         - ProjectTeams
 415         - Alerts Settings and Rules
 416         - EnvironmentProjects
 417         - ProjectOwnership Rules and settings
 418         - Project Inbound Data Filters
 419 
 420         Returns True if the settings have successfully been copied over
 421         Returns False otherwise
 422         """
 423         from sentry.models import EnvironmentProject, ProjectOption, ProjectOwnership, Rule
 424 
 425         model_list = [EnvironmentProject, ProjectOwnership, ProjectTeam, Rule]
 426 
 427         project = Project.objects.get(id=project_id)
 428         try:
 429             with transaction.atomic():
 430                 for model in model_list:
 431                     # remove all previous project settings
 432                     model.objects.filter(project_id=self.id).delete()
 433 
 434                     # add settings from other project to self
 435                     for setting in model.objects.filter(project_id=project_id):
 436                         setting.pk = None
 437                         setting.project_id = self.id
 438                         setting.save()
 439 
 440                 options = ProjectOption.objects.get_all_values(project=project)
 441                 for key, value in six.iteritems(options):
 442                     self.update_option(key, value)
 443 
 444         except IntegrityError as e:
 445             logging.exception(
 446                 "Error occurred during copy project settings.",
 447                 extra={
 448                     "error": six.text_type(e),
 449                     "project_to": self.id,
 450                     "project_from": project_id,
 451                 },
 452             )
 453             return False
 454         return True
