  16 class TypesClass(object):
  17     TYPES = []
  18 
  19     @classmethod
  20     def as_choices(cls):
  21         return cls.TYPES
  22 
  23     @classmethod
  24     def get_type_name(cls, num):
  25         for id, name in cls.TYPES:
  26             if id == num:
  27                 return name
  28 
  29     @classmethod
  30     def get_id_for_type_name(cls, type_name):
  31         for id, name in cls.TYPES:
  32             if type_name == name:
  33                 return id
