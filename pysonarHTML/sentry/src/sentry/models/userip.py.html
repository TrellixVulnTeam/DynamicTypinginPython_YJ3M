  12 class UserIP(Model):

  22     class Meta:

  28 
  29     @classmethod
  30     def log(cls, user, ip_address):
  31         # Only log once every 5 minutes for the same user/ip_address pair
  32         # since this is hit pretty frequently by all API calls in the UI, etc.
  33         cache_key = "userip.log:%d:%s" % (user.id, ip_address)
  34         if cache.get(cache_key):
  35             return
  36 
  37         try:
  38             geo = geo_by_addr(ip_address)
  39         except Exception:
  40             geo = None
  41 
  42         values = {"last_seen": timezone.now()}
  43         if geo:
  44             values.update({"country_code": geo["country_code"], "region_code": geo["region"]})
  45 
  46         UserIP.objects.create_or_update(user=user, ip_address=ip_address, values=values)
  47         cache.set(cache_key, 1, 300)
