  24 # When a developer selects to receive "<Resource> Webhooks" it really means
  25 # listening to a list of specific events. This is a mapping of what those
  26 # specific events are for each resource.
  27 EVENT_EXPANSION = {
  28     "issue": ["issue.created", "issue.resolved", "issue.ignored", "issue.assigned"],
  29     "error": ["error.created"],
  30 }
  31 
  32 # We present Webhook Subscriptions per-resource (Issue, Project, etc.), not
  33 # per-event-type (issue.created, project.deleted, etc.). These are valid
  34 # resources a Sentry App may subscribe to.
  35 VALID_EVENT_RESOURCES = ("issue", "error")
  36 
  37 REQUIRED_EVENT_PERMISSIONS = {
  38     "issue": "event:read",
  39     "error": "event:read",
  40     "project": "project:read",
  41     "member": "member:read",
  42     "organization": "org:read",
  43     "team": "team:read",
  44 }
  45 
  46 # The only events valid for Sentry Apps are the ones listed in the values of
  47 # EVENT_EXPANSION above. This list is likely a subset of all valid ServiceHook
  48 # events.
  49 VALID_EVENTS = tuple(itertools.chain(*EVENT_EXPANSION.values()))

  56 def default_uuid():

  60 def generate_slug(name, is_internal=False):
  61     slug = slugify(name)
  62     # for internal, add some uuid to make it unique
  63     if is_internal:
  64         slug = u"{}-{}".format(slug, default_uuid()[:UUID_CHARS_IN_SLUG])
  65 
  66     return slug

  77 class SentryApp(ParanoidModel, HasApiScopes):
  78     __core__ = True
  79 
  80     application = models.OneToOneField(
  81         "sentry.ApiApplication", null=True, on_delete=models.SET_NULL, related_name="sentry_app"
  82     )
  83 
  84     # Much of the OAuth system in place currently depends on a User existing.
  85     # This "proxy user" represents the SentryApp in those cases.
  86     proxy_user = models.OneToOneField(
  87         "sentry.User", null=True, on_delete=models.SET_NULL, related_name="sentry_app"
  88     )
  89 
  90     # The Organization the Sentry App was created in "owns" it. Members of that
  91     # Org have differing access, dependent on their role within the Org.
  92     owner = FlexibleForeignKey("sentry.Organization", related_name="owned_sentry_apps")
  93 
  94     name = models.TextField()
  95     slug = models.CharField(max_length=SENTRY_APP_SLUG_MAX_LENGTH, unique=True)
  96     author = models.TextField(null=True)
  97     status = BoundedPositiveIntegerField(
  98         default=SentryAppStatus.UNPUBLISHED, choices=SentryAppStatus.as_choices(), db_index=True
  99     )
 100     uuid = models.CharField(max_length=64, default=default_uuid)
 101 
 102     redirect_url = models.URLField(null=True)
 103     webhook_url = models.URLField(max_length=512, null=True)
 104     # does the application subscribe to `event.alert`,
 105     # meaning can it be used in alert rules as a {service} ?
 106     is_alertable = models.BooleanField(default=False)
 107 
 108     # does the application need to wait for verification
 109     # on behalf of the external service to know if its installations
 110     # are successfully installed ?
 111     verify_install = models.BooleanField(default=True)
 112 
 113     events = ArrayField(of=models.TextField, null=True)
 114 
 115     overview = models.TextField(null=True)
 116     schema = EncryptedJsonField(default=dict)
 117 
 118     date_added = models.DateTimeField(default=timezone.now)
 119     date_updated = models.DateTimeField(default=timezone.now)
 120     date_published = models.DateTimeField(null=True, blank=True)

 126     @classmethod
 127     def visible_for_user(cls, request):
 128         from sentry.auth.superuser import is_active_superuser
 129 
 130         if is_active_superuser(request):
 131             return cls.objects.all()
 132 
 133         return cls.objects.filter(status=SentryAppStatus.PUBLISHED)
 134 
 135     # this method checks if a user from a sentry app has permission to a specific project
 136     # for now, only checks if app is installed on the org of the project
 137     @classmethod
 138     def check_project_permission_for_sentry_app_user(cls, user, project):
 139         assert user.is_sentry_app
 140         # if the user exists, so should the sentry_app
 141         sentry_app = cls.objects.get(proxy_user=user)
 142         return sentry_app.is_installed_on(project.organization)
 143 
 144     @property
 145     def is_published(self):
 146         return self.status == SentryAppStatus.PUBLISHED
 147 
 148     @property
 149     def is_unpublished(self):
 150         return self.status == SentryAppStatus.UNPUBLISHED
 151 
 152     @property
 153     def is_internal(self):
 154         return self.status == SentryAppStatus.INTERNAL
 155 
 156     @property
 157     def slug_for_metrics(self):
 158         if self.is_internal:
 159             return "internal"
 160         if self.is_unpublished:
 161             return "unpublished"
 162         return self.slug

 164     def save(self, *args, **kwargs):
 165         self.date_updated = timezone.now()
 166         return super(SentryApp, self).save(*args, **kwargs)
 167 
 168     def is_installed_on(self, organization):
 169         return SentryAppInstallation.objects.filter(organization=organization).exists()
 170 
 171     def build_signature(self, body):
 172         secret = self.application.client_secret
 173         return hmac.new(
 174             key=secret.encode("utf-8"), msg=body.encode("utf-8"), digestmod=sha256
 175         ).hexdigest()
 176 
 177     def show_auth_info(self, access):
 178         encoded_scopes = set({u"%s" % scope for scope in list(access.scopes)})
 179         return set(self.scope_list).issubset(encoded_scopes)
