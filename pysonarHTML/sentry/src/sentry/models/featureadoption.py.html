 112 class FeatureAdoptionManager(BaseManager):
 113     def in_cache(self, organization_id, feature_id):
 114         org_key = FEATURE_ADOPTION_REDIS_KEY.format(organization_id)
 115         feature_matches = []
 116         with redis.clusters.get("default").map() as client:
 117             feature_matches.append(client.sismember(org_key, feature_id))
 118 
 119         return any([p.value for p in feature_matches])
 120 
 121     def set_cache(self, organization_id, feature_id):
 122         org_key = FEATURE_ADOPTION_REDIS_KEY.format(organization_id)
 123         with redis.clusters.get("default").map() as client:
 124             client.sadd(org_key, feature_id)
 125         return True
 126 
 127     def get_all_cache(self, organization_id):
 128         org_key = FEATURE_ADOPTION_REDIS_KEY.format(organization_id)
 129         result = []
 130         with redis.clusters.get("default").map() as client:
 131             result.append(client.smembers(org_key))
 132 
 133         return {int(x) for x in set.union(*[p.value for p in result])}
 134 
 135     def bulk_set_cache(self, organization_id, *args):
 136         if not args:
 137             return False
 138 
 139         org_key = FEATURE_ADOPTION_REDIS_KEY.format(organization_id)
 140         with redis.clusters.get("default").map() as client:
 141             client.sadd(org_key, *args)
 142         return True
 143 
 144     def record(self, organization_id, feature_slug, **kwargs):
 145         try:
 146             feature_id = manager.get_by_slug(feature_slug).id
 147         except UnknownFeature as e:
 148             logger.exception(e)
 149             return False
 150 
 151         if not self.in_cache(organization_id, feature_id):
 152             row, created = self.create_or_update(
 153                 organization_id=organization_id, feature_id=feature_id, complete=True
 154             )
 155             self.set_cache(organization_id, feature_id)
 156             return created
 157 
 158         return False
 159 
 160     def bulk_record(self, organization_id, feature_slugs, **kwargs):
 161         features = []
 162 
 163         try:
 164             feature_ids = set([manager.get_by_slug(slug).id for slug in feature_slugs])
 165         except UnknownFeature as e:
 166             logger.exception(e)
 167             return False
 168 
 169         incomplete_feature_ids = feature_ids - self.get_all_cache(organization_id)
 170 
 171         if not incomplete_feature_ids:
 172             return False
 173 
 174         for feature_id in incomplete_feature_ids:
 175             features.append(
 176                 FeatureAdoption(
 177                     organization_id=organization_id, feature_id=feature_id, complete=True
 178                 )
 179             )
 180         try:
 181             with transaction.atomic():
 182                 self.bulk_create(features)
 183                 return True
 184 
 185         except IntegrityError:
 186             # This can occur if redis somehow loses the set of complete features and
 187             # we attempt to insert duplicate (org_id, feature_id) rows
 188             # This also will happen if we get parallel processes running `bulk_record` and
 189             # `get_all_cache` returns in the second process before the first process
 190             # can `bulk_set_cache`.
 191             return False
 192         finally:
 193             return self.bulk_set_cache(organization_id, *incomplete_feature_ids)
 194 
 195     def get_by_slug(self, organization, slug):
 196         return self.filter(
 197             organization=organization, feature_id=manager.get_by_slug(slug).id
 198         ).first()

 201 class FeatureAdoption(Model):
 202     __core__ = False
 203 
 204     organization = FlexibleForeignKey("sentry.Organization")
 205     feature_id = models.PositiveIntegerField(choices=[(f.id, f.name) for f in manager.all()])
 206     date_completed = models.DateTimeField(default=timezone.now)
 207     complete = models.BooleanField(default=False)
 208     applicable = models.BooleanField(default=True)  # Is this feature applicable to this team?
 209     data = JSONField()
 210 
 211     objects = FeatureAdoptionManager()
 212 
 213     __repr__ = sane_repr("organization_id", "feature_id", "complete", "applicable")
 214 
 215     class Meta:
 216         app_label = "sentry"
 217         db_table = "sentry_featureadoption"
