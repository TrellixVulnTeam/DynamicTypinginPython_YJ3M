  24 class ApiGrant(Model):
  25     """
  26     A grant represents a token with a short lifetime that can
  27     be swapped for an access token, as described in :rfc:`4.1.2`
  28     of the OAuth 2 spec.
  29     """
  30 
  31     __core__ = False
  32 
  33     user = FlexibleForeignKey("sentry.User")
  34     application = FlexibleForeignKey("sentry.ApiApplication")
  35     code = models.CharField(max_length=64, db_index=True, default=generate_code)
  36     expires_at = models.DateTimeField(db_index=True, default=default_expiration)
  37     redirect_uri = models.CharField(max_length=255)
  38     scopes = BitField(
  39         flags=(
  40             ("project:read", "project:read"),
  41             ("project:write", "project:write"),
  42             ("project:admin", "project:admin"),
  43             ("project:releases", "project:releases"),
  44             ("team:read", "team:read"),
  45             ("team:write", "team:write"),
  46             ("team:admin", "team:admin"),
  47             ("event:read", "event:read"),
  48             ("event:write", "event:write"),
  49             ("event:admin", "event:admin"),
  50             ("org:read", "org:read"),
  51             ("org:write", "org:write"),
  52             ("org:admin", "org:admin"),
  53             ("member:read", "member:read"),
  54             ("member:write", "member:write"),
  55             ("member:admin", "member:admin"),
  56         )
  57     )
  58     scope_list = ArrayField(of=models.TextField)
  59 
  60     class Meta:
  61         app_label = "sentry"
  62         db_table = "sentry_apigrant"
  63 
  64     def get_scopes(self):
  65         if self.scope_list:
  66             return self.scope_list
  67         return [k for k, v in six.iteritems(self.scopes) if v]
  68 
  69     def has_scope(self, scope):

  72     def is_expired(self):

  75     def redirect_uri_allowed(self, uri):
