 116     @classmethod
 117     def from_dsn(cls, dsn):
 118         urlparts = urlparse(dsn)
 119 
 120         public_key = urlparts.username
 121         project_id = urlparts.path.rsplit("/", 1)[-1]
 122 
 123         try:
 124             return ProjectKey.objects.get(public_key=public_key, project=project_id)
 125         except ValueError:
 126             # ValueError would come from a non-integer project_id,
 127             # which is obviously a DoesNotExist. We catch and rethrow this
 128             # so anything downstream expecting DoesNotExist works fine
 129             raise ProjectKey.DoesNotExist("ProjectKey matching query does not exist.")
 130

 158     def get_dsn(self, domain=None, secure=True, public=False):
 159         urlparts = urlparse(self.get_endpoint(public=public))
 160 
 161         if not public:
 162             key = "%s:%s" % (self.public_key, self.secret_key)
 163         else:
 164             key = self.public_key
 165 
 166         # If we do not have a scheme or domain/hostname, dsn is never valid
 167         if not urlparts.netloc or not urlparts.scheme:
 168             return ""
 169 
 170         return "%s://%s@%s/%s" % (
 171             urlparts.scheme,
 172             key,
 173             urlparts.netloc + urlparts.path,
 174             self.project_id,
 175         )

 240     def get_endpoint(self, public=True):
 241         if public:
 242             endpoint = settings.SENTRY_PUBLIC_ENDPOINT or settings.SENTRY_ENDPOINT
 243         else:
 244             endpoint = settings.SENTRY_ENDPOINT
 245 
 246         if not endpoint:
 247             endpoint = options.get("system.url-prefix")
 248 
 249         if features.has("organizations:org-subdomains", self.project.organization):
 250             urlparts = urlparse(endpoint)
 251             if urlparts.scheme and urlparts.netloc:
 252                 endpoint = "%s://%s.%s%s" % (
 253                     urlparts.scheme,
 254                     settings.SENTRY_ORG_SUBDOMAIN_TEMPLATE.format(
 255                         organization_id=self.project.organization_id
 256                     ),
 257                     urlparts.netloc,
 258                     urlparts.path,
 259                 )
 260 
 261         return endpoint
