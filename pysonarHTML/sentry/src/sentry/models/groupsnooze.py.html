  20 class GroupSnooze(Model):
  21     """
  22     A snooze marks an issue as ignored until a condition is hit.
  23 
  24     - If ``until`` is set, the snooze is lifted at the given datetime.
  25     - If ``count`` is set, the snooze is lifted when total occurrences match.
  26     - If ``window`` is set (in addition to count), the snooze is lifted when
  27       the rate of events matches.
  28     - If ``user_count`` is set, the snooze is lfited when unique users match.
  29     - If ``user_window`` is set (in addition to count), the snooze is lifted
  30       when the rate unique users matches.
  31 
  32     NOTE: `window` and `user_window` are specified in minutes
  33     """
  34 
  35     __core__ = False
  36 
  37     group = FlexibleForeignKey("sentry.Group", unique=True)
  38     until = models.DateTimeField(null=True)
  39     count = BoundedPositiveIntegerField(null=True)
  40     window = BoundedPositiveIntegerField(null=True)
  41     user_count = BoundedPositiveIntegerField(null=True)
  42     user_window = BoundedPositiveIntegerField(null=True)
  43     state = JSONField(null=True)
  44     actor_id = BoundedPositiveIntegerField(null=True)
  45 
  46     objects = BaseManager(cache_fields=("group",))
  47 
  48     class Meta:
  49         db_table = "sentry_groupsnooze"
  50         app_label = "sentry"
  51 
  52     __repr__ = sane_repr("group_id")
  53 
  54     @classmethod
  55     def get_cache_key(cls, group_id):
  56         return "groupsnooze_group_id:1:%s" % (group_id)
  57 
  58     def is_valid(self, group=None, test_rates=False):
  59         if group is None:
  60             group = self.group
  61         elif group.id != self.group_id:
  62             raise ValueError
  63 
  64         if self.until:
  65             if self.until <= timezone.now():
  66                 return False
  67 
  68         if self.count:
  69             if self.window:
  70                 if test_rates:
  71                     if not self.test_frequency_rates():
  72                         return False
  73             elif self.count <= group.times_seen - self.state["times_seen"]:
  74                 return False
  75 
  76         if self.user_count and test_rates:
  77             if self.user_window:
  78                 if not self.test_user_rates():
  79                     return False
  80             elif self.user_count <= group.count_users_seen() - self.state["users_seen"]:
  81                 return False
  82         return True
  83 
  84     def test_frequency_rates(self):
  85         from sentry import tsdb
  86 
  87         end = timezone.now()
  88         start = end - timedelta(minutes=self.window)
  89 
  90         rate = tsdb.get_sums(model=tsdb.models.group, keys=[self.group_id], start=start, end=end)[
  91             self.group_id
  92         ]
  93         if rate >= self.count:
  94             return False
  95 
  96         return True
  97 
  98     def test_user_rates(self):
  99         from sentry import tsdb
 100 
 101         end = timezone.now()
 102         start = end - timedelta(minutes=self.user_window)
 103 
 104         rate = tsdb.get_distinct_counts_totals(
 105             model=tsdb.models.users_affected_by_group, keys=[self.group_id], start=start, end=end
 106         )[self.group_id]
 107 
 108         if rate >= self.user_count:
 109             return False
 110 
 111         return True
