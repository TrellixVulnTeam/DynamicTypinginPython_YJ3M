  13 class ProjectOptionManager(OptionManager):
  14     def get_value_bulk(self, instances, key):
  15         instance_map = dict((i.id, i) for i in instances)
  16         queryset = self.filter(project__in=instances, key=key)
  17         result = dict((i, None) for i in instances)
  18         for obj in queryset:
  19             result[instance_map[obj.project_id]] = obj.value
  20         return result
  21 
  22     def get_value(self, project, key, default=None, validate=None):
  23         result = self.get_all_values(project)
  24         if key in result:
  25             if validate is None or validate(result[key]):
  26                 return result[key]
  27         if default is None:
  28             well_known_key = projectoptions.lookup_well_known_key(key)
  29             if well_known_key is not None:
  30                 return well_known_key.get_default(project)
  31         return default
  32 
  33     def unset_value(self, project, key):
  34         self.filter(project=project, key=key).delete()
  35         self.reload_cache(project.id, "projectoption.unset_value")
  36 
  37     def set_value(self, project, key, value):
  38         inst, created = self.create_or_update(project=project, key=key, values={"value": value})
  39         self.reload_cache(project.id, "projectoption.set_value")
  40         return created or inst > 0
  41 
  42     def get_all_values(self, project):
  43         if isinstance(project, models.Model):
  44             project_id = project.id
  45         else:
  46             project_id = project
  47         cache_key = self._make_key(project_id)
  48 
  49         if cache_key not in self._option_cache:
  50             result = cache.get(cache_key)
  51             if result is None:
  52                 result = self.reload_cache(project_id, "projectoption.get_all_values")
  53             else:
  54                 self._option_cache[cache_key] = result
  55         return self._option_cache.get(cache_key, {})
  56 
  57     def reload_cache(self, project_id, update_reason):
  58         if update_reason != "projectoption.get_all_values":
  59             schedule_update_config_cache(
  60                 project_id=project_id, generate=True, update_reason=update_reason
  61             )
  62         cache_key = self._make_key(project_id)
  63         result = dict((i.key, i.value) for i in self.filter(project=project_id))
  64         cache.set(cache_key, result)
  65         self._option_cache[cache_key] = result
  66         return result
  67 
  68     def post_save(self, instance, **kwargs):
  69         self.reload_cache(instance.project_id, "projectoption.post_save")
  70 
  71     def post_delete(self, instance, **kwargs):
  72         self.reload_cache(instance.project_id, "projectoption.post_delete")

  75 class ProjectOption(Model):
  76     """
  77     Project options apply only to an instance of a project.
  78 
  79     Options which are specific to a plugin should namespace
  80     their key. e.g. key='myplugin:optname'
  81     """
  82 
  83     __core__ = True
  84 
  85     project = FlexibleForeignKey("sentry.Project")
  86     key = models.CharField(max_length=64)
  87     value = EncryptedPickledObjectField()
  88 
  89     objects = ProjectOptionManager()
  90 
  91     class Meta:
  92         app_label = "sentry"
  93         db_table = "sentry_projectoptions"
  94         unique_together = (("project", "key"),)
  95 
  96     __repr__ = sane_repr("project_id", "key", "value")
