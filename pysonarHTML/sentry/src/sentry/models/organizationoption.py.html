  12 class OrganizationOptionManager(OptionManager):
  13     def get_value_bulk(self, instances, key):
  14         instance_map = dict((i.id, i) for i in instances)
  15         queryset = self.filter(organization__in=instances, key=key)
  16         result = dict((i, None) for i in instances)
  17         for obj in queryset:
  18             result[instance_map[obj.organization_id]] = obj.value
  19         return result

  21     def get_value(self, organization, key, default=None):

  25     def unset_value(self, organization, key):
  26         try:
  27             inst = self.get(organization=organization, key=key)
  28         except self.model.DoesNotExist:
  29             return
  30         inst.delete()
  31         self.reload_cache(organization.id, "organizationoption.unset_value")
  32 
  33     def set_value(self, organization, key, value):
  34         self.create_or_update(organization=organization, key=key, values={"value": value})
  35         self.reload_cache(organization.id, "organizationoption.set_value")

  37     def get_all_values(self, organization):
  38         if isinstance(organization, models.Model):
  39             organization_id = organization.id
  40         else:
  41             organization_id = organization
  42         cache_key = self._make_key(organization_id)
  43 
  44         if cache_key not in self._option_cache:
  45             result = cache.get(cache_key)
  46             if result is None:
  47                 result = self.reload_cache(organization_id, "organizationoption.get_all_values")
  48             else:
  49                 self._option_cache[cache_key] = result
  50         return self._option_cache.get(cache_key, {})
  51 
  52     def reload_cache(self, organization_id, update_reason):
  53         if update_reason != "organizationoption.get_all_values":
  54             schedule_update_config_cache(
  55                 organization_id=organization_id, generate=False, update_reason=update_reason
  56             )
  57 
  58         cache_key = self._make_key(organization_id)
  59         result = dict((i.key, i.value) for i in self.filter(organization=organization_id))
  60         cache.set(cache_key, result)
  61         self._option_cache[cache_key] = result
  62         return result
  63 
  64     def post_save(self, instance, **kwargs):

  71 class OrganizationOption(Model):
  72     """
  73     Organization options apply only to an instance of a organization.
  74 
  75     Options which are specific to a plugin should namespace
  76     their key. e.g. key='myplugin:optname'
  77 
  78     key: onboarding:complete
  79     value: { updated: datetime }
  80     """
  81 
  82     __core__ = True
  83 
  84     organization = FlexibleForeignKey("sentry.Organization")

  88     objects = OrganizationOptionManager()

  90     class Meta:
