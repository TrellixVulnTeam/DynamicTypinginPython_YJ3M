 120     def add_organization(self, organization, user=None, default_auth_id=None):
 121         """
 122         Add an organization to this integration.
 123 
 124         Returns False if the OrganizationIntegration was not created
 125         """
 126         try:
 127             org_integration, created = OrganizationIntegration.objects.get_or_create(
 128                 organization_id=organization.id,
 129                 integration_id=self.id,
 130                 defaults={"default_auth_id": default_auth_id, "config": {}},
 131             )
 132             if not created and default_auth_id:
 133                 org_integration.update(default_auth_id=default_auth_id)
 134         except IntegrityError:
 135             logger.info(
 136                 "add-organization-integrity-error",
 137                 extra={
 138                     "organization_id": organization.id,
 139                     "integration_id": self.id,
 140                     "default_auth_id": default_auth_id,
 141                 },
 142             )
 143             return False
 144         else:
 145             integration_added.send_robust(
 146                 integration=self, organization=organization, user=user, sender=self.__class__
 147             )
 148 
 149             return org_integration

 151     def reauthorize(self, data):
 152         """
 153         The structure of `data` depends on the `build_integration`
 154         method on the integration provider.
 155 
 156         Each provider may have their own way of reauthorizing the
 157         integration.
 158         """
 159         if self.provider == "slack":
 160             metadata = data.get("metadata", {})
 161             metadata["old_access_token"] = self.metadata["access_token"]
 162             self.update(metadata=metadata)
