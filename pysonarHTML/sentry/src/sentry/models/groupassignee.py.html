  17 def get_user_project_ids(users):
  18     """
  19     Given a list of users, return a dict where keys are user_ids
  20     and values are a set of the project_ids the user is a member of
  21     """
  22     from sentry.models import OrganizationMemberTeam, ProjectTeam
  23 
  24     user_teams = list(
  25         OrganizationMemberTeam.objects.filter(
  26             organizationmember__user__in=users, is_active=True
  27         ).values("organizationmember__user", "team")
  28     )
  29 
  30     # team_id to list of projects
  31     projects_by_team = defaultdict(set)
  32     for tp in ProjectTeam.objects.filter(team__in=[ut["team"] for ut in user_teams]):
  33         projects_by_team[tp.team_id].add(tp.project_id)
  34 
  35     # user_id to projects
  36     projects_by_user = defaultdict(set)
  37     for ut in user_teams:
  38         projects_by_user[ut["organizationmember__user"]].update(projects_by_team[ut["team"]])
  39 
  40     return projects_by_user

  43 def sync_group_assignee_inbound(integration, email, external_issue_key, assign=True):
  44     """
  45     Given an integration, user email address and an external issue key,
  46     assign linked groups to matching users. Checks project membership.
  47     Returns a list of groups that were successfully assigned.
  48     """
  49     from sentry import features
  50     from sentry.models import Group, UserEmail, User
  51 
  52     logger = logging.getLogger("sentry.integrations.%s" % integration.provider)
  53 
  54     orgs_with_sync_enabled = []
  55     for org in integration.organizations.all():
  56         has_issue_sync = features.has("organizations:integrations-issue-sync", org)
  57         if not has_issue_sync:
  58             continue
  59 
  60         installation = integration.get_installation(org.id)
  61         if installation.should_sync("inbound_assignee"):
  62             orgs_with_sync_enabled.append(org.id)
  63 
  64     affected_groups = list(
  65         Group.objects.get_groups_by_external_issue(integration, external_issue_key).filter(
  66             project__organization_id__in=orgs_with_sync_enabled
  67         )
  68     )
  69 
  70     if not affected_groups:
  71         return []
  72 
  73     if not assign:
  74         for group in affected_groups:
  75             GroupAssignee.objects.deassign(group)
  76         return affected_groups
  77 
  78     users = {
  79         u.id: u
  80         for u in User.objects.filter(
  81             id__in=UserEmail.objects.filter(is_verified=True, email=email).values_list(
  82                 "user_id", flat=True
  83             )
  84         )
  85     }
  86 
  87     projects_by_user = get_user_project_ids(users.values())
  88 
  89     groups_assigned = []
  90     for group in affected_groups:
  91         try:
  92             user_id = [
  93                 user_id
  94                 for user_id, projects in projects_by_user.items()
  95                 if group.project_id in projects
  96             ][0]
  97         except IndexError:
  98             logger.info(
  99                 "assignee-not-found-inbound",
 100                 extra={
 101                     "integration_id": integration.id,
 102                     "email": email,
 103                     "issue_key": external_issue_key,
 104                 },
 105             )
 106         else:
 107             user = users[user_id]
 108             GroupAssignee.objects.assign(group, user)
 109             groups_assigned.append(group)
 110 
 111     return groups_assigned

 114 def sync_group_assignee_outbound(group, user_id, assign=True):
 115     from sentry.tasks.integrations import sync_assignee_outbound
 116     from sentry.models import GroupLink
 117 
 118     external_issue_ids = GroupLink.objects.filter(
 119         project_id=group.project_id, group_id=group.id, linked_type=GroupLink.LinkedType.issue
 120     ).values_list("linked_id", flat=True)
 121 
 122     for external_issue_id in external_issue_ids:
 123         sync_assignee_outbound.apply_async(
 124             kwargs={"external_issue_id": external_issue_id, "user_id": user_id, "assign": assign}
 125         )

 128 class GroupAssigneeManager(BaseManager):
 129     def assign(self, group, assigned_to, acting_user=None):
 130         from sentry import features
 131         from sentry.models import User, Team, GroupSubscription, GroupSubscriptionReason
 132 
 133         GroupSubscription.objects.subscribe_actor(
 134             group=group, actor=assigned_to, reason=GroupSubscriptionReason.assigned
 135         )
 136 
 137         if isinstance(assigned_to, User):
 138             assignee_type = "user"
 139             other_type = "team"
 140         elif isinstance(assigned_to, Team):
 141             assignee_type = "team"
 142             other_type = "user"
 143         else:
 144             raise AssertionError("Invalid type to assign to: %r" % type(assigned_to))
 145 
 146         now = timezone.now()
 147         assignee, created = GroupAssignee.objects.get_or_create(
 148             group=group,
 149             defaults={"project": group.project, assignee_type: assigned_to, "date_added": now},
 150         )
 151 
 152         if not created:
 153             affected = (
 154                 GroupAssignee.objects.filter(group=group)
 155                 .exclude(**{assignee_type: assigned_to})
 156                 .update(**{assignee_type: assigned_to, other_type: None, "date_added": now})
 157             )
 158         else:
 159             affected = True
 160             issue_assigned.send_robust(
 161                 project=group.project, group=group, user=acting_user, sender=self.__class__
 162             )
 163 
 164         if affected:
 165             activity = Activity.objects.create(
 166                 project=group.project,
 167                 group=group,
 168                 type=Activity.ASSIGNED,
 169                 user=acting_user,
 170                 data={
 171                     "assignee": six.text_type(assigned_to.id),
 172                     "assigneeEmail": getattr(assigned_to, "email", None),
 173                     "assigneeType": assignee_type,
 174                 },
 175             )
 176             activity.send_notification()
 177             metrics.incr("group.assignee.change", instance="assigned", skip_internal=True)
 178             # sync Sentry assignee to external issues
 179             if assignee_type == "user" and features.has(
 180                 "organizations:integrations-issue-sync", group.organization, actor=acting_user
 181             ):
 182                 sync_group_assignee_outbound(group, assigned_to.id, assign=True)
 183 
 184     def deassign(self, group, acting_user=None):
 185         from sentry import features
 186 
 187         affected = GroupAssignee.objects.filter(group=group)[:1].count()
 188         GroupAssignee.objects.filter(group=group).delete()
 189 
 190         if affected > 0:
 191             activity = Activity.objects.create(
 192                 project=group.project, group=group, type=Activity.UNASSIGNED, user=acting_user
 193             )
 194             activity.send_notification()
 195             metrics.incr("group.assignee.change", instance="deassigned", skip_internal=True)
 196             # sync Sentry assignee to external issues
 197             if features.has(
 198                 "organizations:integrations-issue-sync", group.organization, actor=acting_user
 199             ):
 200                 sync_group_assignee_outbound(group, None, assign=False)

 203 class GroupAssignee(Model):
 204     """
 205     Identifies an assignment relationship between a user/team and an
 206     aggregated event (Group).
 207     """
 208 
 209     __core__ = False
 210 
 211     objects = GroupAssigneeManager()
 212 
 213     project = FlexibleForeignKey("sentry.Project", related_name="assignee_set")
 214     group = FlexibleForeignKey("sentry.Group", related_name="assignee_set", unique=True)
 215     user = FlexibleForeignKey(
 216         settings.AUTH_USER_MODEL, related_name="sentry_assignee_set", null=True
 217     )
 218     team = FlexibleForeignKey("sentry.Team", related_name="sentry_assignee_set", null=True)
 219     date_added = models.DateTimeField(default=timezone.now)
 220 
 221     class Meta:
 222         app_label = "sentry"
 223         db_table = "sentry_groupasignee"
 224         unique_together = [("project", "group")]
 225 
 226     __repr__ = sane_repr("group_id", "user_id", "team_id")
 227 
 228     def save(self, *args, **kwargs):
 229         assert not (self.user_id is not None and self.team_id is not None) and not (
 230             self.user_id is None and self.team_id is None
 231         ), "Must have Team or User, not both"
 232         super(GroupAssignee, self).save(*args, **kwargs)
 233 
 234     def assigned_actor_id(self):
 235         if self.user:
 236             return u"user:{}".format(self.user_id)
 237 
 238         if self.team:
 239             return u"team:{}".format(self.team_id)
 240 
 241         raise NotImplementedError("Unknown Assignee")
 242 
 243     def assigned_actor(self):
 244         from sentry.api.fields.actor import Actor
 245 
 246         return Actor.from_actor_identifier(self.assigned_actor_id())
