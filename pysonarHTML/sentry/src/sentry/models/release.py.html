 135     @classmethod
 136     def get(cls, project, version):
 137         cache_key = cls.get_cache_key(project.organization_id, version)
 138 
 139         release = cache.get(cache_key)
 140         if release is None:
 141             try:
 142                 release = cls.objects.get(
 143                     organization_id=project.organization_id, projects=project, version=version
 144                 )
 145             except cls.DoesNotExist:
 146                 release = -1
 147             cache.set(cache_key, release, 300)
 148 
 149         if release == -1:
 150             return
 151 
 152         return release

 159     @classmethod
 160     def _get_or_create_impl(cls, project, version, date_added, metric_tags):
 161         from sentry.models import Project
 162 
 163         if date_added is None:
 164             date_added = timezone.now()
 165 
 166         cache_key = cls.get_cache_key(project.organization_id, version)
 167 
 168         release = cache.get(cache_key)
 169 
 170         if release in (None, -1):
 171             # TODO(dcramer): if the cache result is -1 we could attempt a
 172             # default create here instead of default get
 173             project_version = ("%s-%s" % (project.slug, version))[:DB_VERSION_LENGTH]
 174             releases = list(
 175                 cls.objects.filter(
 176                     organization_id=project.organization_id,
 177                     version__in=[version, project_version],
 178                     projects=project,
 179                 )
 180             )
 181 
 182             if releases:
 183                 try:
 184                     release = [r for r in releases if r.version == project_version][0]
 185                 except IndexError:
 186                     release = releases[0]
 187                 metric_tags["created"] = "false"
 188             else:
 189                 try:
 190                     with transaction.atomic():
 191                         release = cls.objects.create(
 192                             organization_id=project.organization_id,
 193                             version=version,
 194                             date_added=date_added,
 195                             total_deploys=0,
 196                         )
 197 
 198                     metric_tags["created"] = "true"
 199                 except IntegrityError:
 200                     metric_tags["created"] = "false"
 201                     release = cls.objects.get(
 202                         organization_id=project.organization_id, version=version
 203                     )
 204 
 205                 release.add_project(project)
 206                 if not project.flags.has_releases:
 207                     project.flags.has_releases = True
 208                     project.update(flags=F("flags").bitor(Project.flags.has_releases))
 209 
 210             # TODO(dcramer): upon creating a new release, check if it should be
 211             # the new "latest release" for this project
 212             cache.set(cache_key, release, 3600)
 213             metric_tags["cache_hit"] = "false"
 214         else:
 215             metric_tags["cache_hit"] = "true"
 216 
 217         return release

 229         # The following models reference release:
 230         # ReleaseCommit.release
 231         # ReleaseEnvironment.release_id
 232         # ReleaseProject.release
 233         # GroupRelease.release_id
 234         # GroupResolution.release
 235         # Group.first_release
 236         # ReleaseFile.release
 237 
 238         from sentry.models import (
 239             ReleaseCommit,
 240             ReleaseEnvironment,
 241             ReleaseFile,
 242             ReleaseProject,
 243             ReleaseProjectEnvironment,
 244             Group,
 245             GroupRelease,
 246             GroupResolution,
 247         )
 248 
 249         model_list = (
 250             ReleaseCommit,
 251             ReleaseEnvironment,
 252             ReleaseFile,
 253             ReleaseProject,
 254             ReleaseProjectEnvironment,
 255             GroupRelease,
 256             GroupResolution,
 257         )
 258         for release in from_releases:
 259             for model in model_list:
 260                 if hasattr(model, "release"):
 261                     update_kwargs = {"release": to_release}
 262                 else:
 263                     update_kwargs = {"release_id": to_release.id}
 264                 try:
 265                     with transaction.atomic():
 266                         model.objects.filter(release_id=release.id).update(**update_kwargs)
 267                 except IntegrityError:
 268                     for item in model.objects.filter(release_id=release.id):
 269                         try:
 270                             with transaction.atomic():
 271                                 model.objects.filter(id=item.id).update(**update_kwargs)
 272                         except IntegrityError:
 273                             item.delete()
 274 
 275             Group.objects.filter(first_release=release).update(first_release=to_release)
 276 
 277             release.delete()

 317     def handle_commit_ranges(self, refs):
 318         """
 319         Takes commit refs of the form:
 320         [
 321             {
 322                 'previousCommit': None,
 323                 'commit': 'previous_commit..commit',
 324             }
 325         ]
 326         Note: Overwrites 'previousCommit' and 'commit'
 327         """
 328         for ref in refs:
 329             if COMMIT_RANGE_DELIMITER in ref["commit"]:
 330                 ref["previousCommit"], ref["commit"] = ref["commit"].split(COMMIT_RANGE_DELIMITER)

 332     def set_refs(self, refs, user, fetch=False):
 333         from sentry.api.exceptions import InvalidRepository
 334         from sentry.models import Commit, ReleaseHeadCommit, Repository
 335         from sentry.tasks.commits import fetch_commits
 336 
 337         # TODO: this does the wrong thing unless you are on the most
 338         # recent release.  Add a timestamp compare?
 339         prev_release = (
 340             type(self)
 341             .objects.filter(organization_id=self.organization_id, projects__in=self.projects.all())
 342             .extra(select={"sort": "COALESCE(date_released, date_added)"})
 343             .exclude(version=self.version)
 344             .order_by("-sort")
 345             .first()
 346         )
 347 
 348         names = {r["repository"] for r in refs}
 349         repos = list(
 350             Repository.objects.filter(organization_id=self.organization_id, name__in=names)
 351         )
 352         repos_by_name = {r.name: r for r in repos}
 353         invalid_repos = names - set(repos_by_name.keys())
 354         if invalid_repos:
 355             raise InvalidRepository("Invalid repository names: %s" % ",".join(invalid_repos))
 356 
 357         self.handle_commit_ranges(refs)
 358 
 359         for ref in refs:
 360             repo = repos_by_name[ref["repository"]]
 361 
 362             commit = Commit.objects.get_or_create(
 363                 organization_id=self.organization_id, repository_id=repo.id, key=ref["commit"]
 364             )[0]
 365             # update head commit for repo/release if exists
 366             ReleaseHeadCommit.objects.create_or_update(
 367                 organization_id=self.organization_id,
 368                 repository_id=repo.id,
 369                 release=self,
 370                 values={"commit": commit},
 371             )
 372         if fetch:
 373             fetch_commits.apply_async(
 374                 kwargs={
 375                     "release_id": self.id,
 376                     "user_id": user.id,
 377                     "refs": refs,
 378                     "prev_release_id": prev_release and prev_release.id,
 379                 }
 380             )

 382     def set_commits(self, commit_list):
 383         """
 384         Bind a list of commits to this release.
 385 
 386         This will clear any existing commit log and replace it with the given
 387         commits.
 388         """
 389 
 390         # Sort commit list in reverse order
 391         commit_list.sort(key=lambda commit: commit.get("timestamp"), reverse=True)
 392 
 393         # TODO(dcramer): this function could use some cleanup/refactoring as it's a bit unwieldy
 394         from sentry.models import (
 395             Commit,
 396             CommitAuthor,
 397             Group,
 398             GroupLink,
 399             GroupResolution,
 400             GroupStatus,
 401             ReleaseCommit,
 402             ReleaseHeadCommit,
 403             Repository,
 404             PullRequest,
 405         )
 406         from sentry.plugins.providers.repository import RepositoryProvider
 407         from sentry.tasks.integrations import kick_off_status_syncs
 408 
 409         # todo(meredith): implement for IntegrationRepositoryProvider
 410         commit_list = [
 411             c
 412             for c in commit_list
 413             if not RepositoryProvider.should_ignore_commit(c.get("message", ""))
 414         ]
 415         lock_key = type(self).get_lock_key(self.organization_id, self.id)
 416         lock = locks.get(lock_key, duration=10)
 417         with TimedRetryPolicy(10)(lock.acquire):
 418             start = time()
 419             with transaction.atomic():
 420                 # TODO(dcramer): would be good to optimize the logic to avoid these
 421                 # deletes but not overly important
 422                 ReleaseCommit.objects.filter(release=self).delete()
 423 
 424                 authors = {}
 425                 repos = {}
 426                 commit_author_by_commit = {}
 427                 head_commit_by_repo = {}
 428                 latest_commit = None
 429                 for idx, data in enumerate(commit_list):
 430                     repo_name = data.get("repository") or u"organization-{}".format(
 431                         self.organization_id
 432                     )
 433                     if repo_name not in repos:
 434                         repos[repo_name] = repo = Repository.objects.get_or_create(
 435                             organization_id=self.organization_id, name=repo_name
 436                         )[0]
 437                     else:
 438                         repo = repos[repo_name]
 439 
 440                     author_email = data.get("author_email")
 441                     if author_email is None and data.get("author_name"):
 442                         author_email = (
 443                             re.sub(r"[^a-zA-Z0-9\-_\.]*", "", data["author_name"]).lower()
 444                             + "@localhost"
 445                         )
 446 
 447                     author_email = truncatechars(author_email, 75)
 448 
 449                     if not author_email:
 450                         author = None
 451                     elif author_email not in authors:
 452                         author_data = {"name": data.get("author_name")}
 453                         author, created = CommitAuthor.objects.get_or_create(
 454                             organization_id=self.organization_id,
 455                             email=author_email,
 456                             defaults=author_data,
 457                         )
 458                         if author.name != author_data["name"]:
 459                             author.update(name=author_data["name"])
 460                         authors[author_email] = author
 461                     else:
 462                         author = authors[author_email]
 463 
 464                     commit_data = {}
 465 
 466                     # Update/set message and author if they are provided.
 467                     if author is not None:
 468                         commit_data["author"] = author
 469                     if "message" in data:
 470                         commit_data["message"] = data["message"]
 471                     if "timestamp" in data:
 472                         commit_data["date_added"] = data["timestamp"]
 473 
 474                     commit, created = Commit.objects.get_or_create(
 475                         organization_id=self.organization_id,
 476                         repository_id=repo.id,
 477                         key=data["id"],
 478                         defaults=commit_data,
 479                     )
 480                     if not created:
 481                         commit_data = {
 482                             key: value
 483                             for key, value in six.iteritems(commit_data)
 484                             if getattr(commit, key) != value
 485                         }
 486                         if commit_data:
 487                             commit.update(**commit_data)
 488 
 489                     if author is None:
 490                         author = commit.author
 491 
 492                     commit_author_by_commit[commit.id] = author
 493 
 494                     patch_set = data.get("patch_set", [])
 495                     for patched_file in patch_set:
 496                         try:
 497                             with transaction.atomic():
 498                                 CommitFileChange.objects.create(
 499                                     organization_id=self.organization.id,
 500                                     commit=commit,
 501                                     filename=patched_file["path"],
 502                                     type=patched_file["type"],
 503                                 )
 504                         except IntegrityError:
 505                             pass
 506 
 507                     try:
 508                         with transaction.atomic():
 509                             ReleaseCommit.objects.create(
 510                                 organization_id=self.organization_id,
 511                                 release=self,
 512                                 commit=commit,
 513                                 order=idx,
 514                             )
 515                     except IntegrityError:
 516                         pass
 517 
 518                     if latest_commit is None:
 519                         latest_commit = commit
 520 
 521                     head_commit_by_repo.setdefault(repo.id, commit.id)
 522 
 523                 self.update(
 524                     commit_count=len(commit_list),
 525                     authors=[
 526                         six.text_type(a_id)
 527                         for a_id in ReleaseCommit.objects.filter(
 528                             release=self, commit__author_id__isnull=False
 529                         )
 530                         .values_list("commit__author_id", flat=True)
 531                         .distinct()
 532                     ],
 533                     last_commit_id=latest_commit.id if latest_commit else None,
 534                 )
 535                 metrics.timing("release.set_commits.duration", time() - start)
 536 
 537         # fill any missing ReleaseHeadCommit entries
 538         for repo_id, commit_id in six.iteritems(head_commit_by_repo):
 539             try:
 540                 with transaction.atomic():
 541                     ReleaseHeadCommit.objects.create(
 542                         organization_id=self.organization_id,
 543                         release_id=self.id,
 544                         repository_id=repo_id,
 545                         commit_id=commit_id,
 546                     )
 547             except IntegrityError:
 548                 pass
 549 
 550         release_commits = list(
 551             ReleaseCommit.objects.filter(release=self)
 552             .select_related("commit")
 553             .values("commit_id", "commit__key")
 554         )
 555 
 556         commit_resolutions = list(
 557             GroupLink.objects.filter(
 558                 linked_type=GroupLink.LinkedType.commit,
 559                 linked_id__in=[rc["commit_id"] for rc in release_commits],
 560             ).values_list("group_id", "linked_id")
 561         )
 562 
 563         commit_group_authors = [
 564             (cr[0], commit_author_by_commit.get(cr[1])) for cr in commit_resolutions  # group_id
 565         ]
 566 
 567         pr_ids_by_merge_commit = list(
 568             PullRequest.objects.filter(
 569                 merge_commit_sha__in=[rc["commit__key"] for rc in release_commits],
 570                 organization_id=self.organization_id,
 571             ).values_list("id", flat=True)
 572         )
 573 
 574         pull_request_resolutions = list(
 575             GroupLink.objects.filter(
 576                 relationship=GroupLink.Relationship.resolves,
 577                 linked_type=GroupLink.LinkedType.pull_request,
 578                 linked_id__in=pr_ids_by_merge_commit,
 579             ).values_list("group_id", "linked_id")
 580         )
 581 
 582         pr_authors = list(
 583             PullRequest.objects.filter(
 584                 id__in=[prr[1] for prr in pull_request_resolutions]
 585             ).select_related("author")
 586         )
 587 
 588         pr_authors_dict = {pra.id: pra.author for pra in pr_authors}
 589 
 590         pull_request_group_authors = [
 591             (prr[0], pr_authors_dict.get(prr[1])) for prr in pull_request_resolutions
 592         ]
 593 
 594         user_by_author = {None: None}
 595 
 596         commits_and_prs = list(itertools.chain(commit_group_authors, pull_request_group_authors))
 597 
 598         group_project_lookup = dict(
 599             Group.objects.filter(id__in=[group_id for group_id, _ in commits_and_prs]).values_list(
 600                 "id", "project_id"
 601             )
 602         )
 603 
 604         for group_id, author in commits_and_prs:
 605             if author not in user_by_author:
 606                 try:
 607                     user_by_author[author] = author.find_users()[0]
 608                 except IndexError:
 609                     user_by_author[author] = None
 610             actor = user_by_author[author]
 611 
 612             with transaction.atomic():
 613                 GroupResolution.objects.create_or_update(
 614                     group_id=group_id,
 615                     values={
 616                         "release": self,
 617                         "type": GroupResolution.Type.in_release,
 618                         "status": GroupResolution.Status.resolved,
 619                         "actor_id": actor.id if actor else None,
 620                     },
 621                 )
 622                 group = Group.objects.get(id=group_id)
 623                 group.update(status=GroupStatus.RESOLVED)
 624                 metrics.incr("group.resolved", instance="in_commit", skip_internal=True)
 625 
 626             issue_resolved.send_robust(
 627                 organization_id=self.organization_id,
 628                 user=actor,
 629                 group=group,
 630                 project=group.project,
 631                 resolution_type="with_commit",
 632                 sender=type(self),
 633             )
 634 
 635             kick_off_status_syncs.apply_async(
 636                 kwargs={"project_id": group_project_lookup[group_id], "group_id": group_id}
 637             )

 639     def safe_delete(self):
 640         """Deletes a release if possible or raises a `UnsafeReleaseDeletion`
 641         exception.
 642         """
 643         from sentry.models import Group, ReleaseFile
 644         from sentry.snuba.sessions import check_has_health_data
 645 
 646         # we don't want to remove the first_release metadata on the Group, and
 647         # while people might want to kill a release (maybe to remove files),
 648         # removing the release is prevented
 649         if Group.objects.filter(first_release=self).exists():
 650             raise UnsafeReleaseDeletion(ERR_RELEASE_REFERENCED)
 651 
 652         # We do not allow releases with health data to be deleted because
 653         # the upserting from snuba data would create the release again.
 654         # We would need to be able to delete this data from snuba which we
 655         # can't do yet.
 656         project_ids = list(self.projects.values_list("id").all())
 657         if check_has_health_data([(p[0], self.version) for p in project_ids]):
 658             raise UnsafeReleaseDeletion(ERR_RELEASE_HEALTH_DATA)
 659 
 660         # TODO(dcramer): this needs to happen in the queue as it could be a long
 661         # and expensive operation
 662         file_list = ReleaseFile.objects.filter(release=self).select_related("file")
 663         for releasefile in file_list:
