  28 class Map(Attribute):
  29     def __init__(self, name, attributes=None, required=True):
  30         self.name = name
  31         self.required = required
  32         self.attributes = attributes
  33 
  34     def extract(self, value):
  35         """
  36         If passed a non dictionary we assume we can pull attributes from it.
  37 
  38         This will hard error in some situations if you're passing a bad type
  39         (like an int).
  40         """
  41         if value is None:
  42             return value
  43 
  44         if not isinstance(value, Mapping):
  45             new_value = {}
  46             for attr in self.attributes:
  47                 new_value[attr.name] = attr.extract(getattr(value, attr.name, None))
  48             items = new_value
  49         else:
  50             # ensure we dont mutate the original
  51             # we dont need to deepcopy as if it recurses into another Map it
  52             # will once again copy itself
  53             items = value.copy()
  54 
  55         data = {}
  56         for attr in self.attributes:
  57             nv = items.pop(attr.name, None)
  58             if attr.required and nv is None:
  59                 raise ValueError(u"{} is required (cannot be None)".format(attr.name))
  60 
  61             data[attr.name] = attr.extract(nv)
  62 
  63         if items:
  64             raise ValueError(
  65                 u"Unknown attributes: {}".format(", ".join(map(six.text_type, six.iterkeys(items))))
  66             )
  67 
  68         return data

  71 class Event(object):
  72     __slots__ = ["uuid", "data", "datetime"]
  73 
  74     type = None
  75 
  76     attributes = ()
  77 
  78     def __init__(self, type=None, datetime=None, **items):
  79         self.uuid = uuid1()
  80 
  81         self.datetime = datetime or timezone.now()
  82         if type is not None:
  83             self.type = type
  84 
  85         if self.type is None:
  86             raise ValueError("Event is missing type")
  87 
  88         data = {}
  89         for attr in self.attributes:
  90             nv = items.pop(attr.name, None)
  91             if attr.required and nv is None:
  92                 raise ValueError(u"{} is required (cannot be None)".format(attr.name))
  93             data[attr.name] = attr.extract(nv)
  94 
  95         if items:
  96             raise ValueError(u"Unknown attributes: {}".format(", ".join(six.iterkeys(items))))
  97 
  98         self.data = data
  99 
 100     def serialize(self):
 101         return {
 102             "uuid": b64encode(self.uuid.bytes),
 103             "timestamp": to_timestamp(self.datetime),
 104             "type": self.type,
 105             "data": self.data,
 106         }

 108     @classmethod
 109     def from_instance(cls, instance, **kwargs):
 110         values = {}
 111         for attr in cls.attributes:
 112             values[attr.name] = kwargs.get(attr.name, getattr(instance, attr.name, None))
 113         return cls(**values)
