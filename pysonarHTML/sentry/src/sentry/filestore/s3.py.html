  75 def safe_join(base, *paths):
  76     """
  77     A version of django.utils._os.safe_join for S3 paths.
  78 
  79     Joins one or more path components to the base path component
  80     intelligently. Returns a normalized version of the final path.
  81 
  82     The final path must be located inside of the base path component
  83     (otherwise a ValueError is raised).
  84 
  85     Paths outside the base path indicate a possible security
  86     sensitive operation.
  87     """
  88     base_path = force_text(base)
  89     base_path = base_path.rstrip("/")
  90     paths = [force_text(p) for p in paths]
  91 
  92     final_path = base_path
  93     for path in paths:
  94         final_path = urlparse.urljoin(final_path.rstrip("/") + "/", path)
  95 
  96     # Ensure final_path starts with base_path and that the next character after
  97     # the final path is '/' (or nothing, in which case final_path must be
  98     # equal to base_path).
  99     base_path_len = len(base_path)
 100     if not final_path.startswith(base_path) or final_path[
 101         base_path_len : base_path_len + 1
 102     ] not in ("", "/"):
 103         raise ValueError("the joined path is located outside of the base path" " component")
 104 
 105     return final_path.lstrip("/")

 108 class S3Boto3StorageFile(File):
 109     """
 110     The default file object used by the S3Boto3Storage backend.
 111 
 112     This file implements file streaming using boto's multipart
 113     uploading functionality. The file can be opened in read or
 114     write mode.
 115 
 116     This class extends Django's File class. However, the contained
 117     data is only the data contained in the current buffer. So you
 118     should not access the contained file object directly. You should
 119     access the data via this class.
 120 
 121     Warning: This file *must* be closed using the close() method in
 122     order to properly write the file to S3. Be sure to close the file
 123     in your application.
 124     """
 125 
 126     # TODO: Read/Write (rw) mode may be a bit undefined at the moment. Needs testing.
 127     # TODO: When Django drops support for Python 2.5, rewrite to use the
 128     #       BufferedIO streams in the Python 2.6 io module.
 129     buffer_size = 5242880
 130 
 131     def __init__(self, name, mode, storage, buffer_size=None):
 132         self._storage = storage
 133         self.name = name[len(self._storage.location) :].lstrip("/")
 134         self._mode = mode
 135         self.obj = storage.bucket.Object(storage._encode_name(name))
 136         # NOTE(mattrobenolt): This is an explicit deviation from
 137         # django-storages. This adds an extra HEAD request before
 138         # every GET. This effectively doubles the time it takes for
 139         # every chunk in our filestore. We explicitly are opting
 140         # out of this behavior to avoid this overhead.
 141         #
 142         # if 'w' not in mode:
 143         #     # Force early RAII-style exception if object does not exist
 144         #     self.obj.load()
 145         self._is_dirty = False
 146         self._file = None
 147         self._multipart = None
 148         # 5 MB is the minimum part size (if there is more than one part).
 149         # Amazon allows up to 10,000 parts.  The default supports uploads
 150         # up to roughly 50 GB.  Increase the part size to accommodate
 151         # for files larger than this.
 152         if buffer_size is not None:
 153             self.buffer_size = buffer_size
 154         self._write_counter = 0
 155 
 156     @property
 157     def size(self):
 158         return self.obj.content_length
 159 
 160     def _get_file(self):
 161         if self._file is None:
 162             with metrics.timer("filestore.read", instance="s3"):
 163                 self._file = BytesIO()
 164                 if "r" in self._mode:
 165                     self._is_dirty = False
 166                     self._file.write(self.obj.get()["Body"].read())
 167                     self._file.seek(0)
 168                 if self._storage.gzip and self.obj.content_encoding == "gzip":
 169                     self._file = GzipFile(mode=self._mode, fileobj=self._file, mtime=0.0)
 170         return self._file
 171 
 172     def _set_file(self, value):
 173         self._file = value
 174 
 175     file = property(_get_file, _set_file)
 176 
 177     def read(self, *args, **kwargs):
 178         if "r" not in self._mode:
 179             raise AttributeError("File was not opened in read mode.")
 180         return super(S3Boto3StorageFile, self).read(*args, **kwargs)
 181 
 182     def write(self, content):
 183         if "w" not in self._mode:
 184             raise AttributeError("File was not opened in write mode.")
 185         self._is_dirty = True
 186         if self._multipart is None:
 187             parameters = self._storage.object_parameters.copy()
 188             parameters["ACL"] = self._storage.default_acl
 189             parameters["ContentType"] = (
 190                 mimetypes.guess_type(self.obj.key)[0] or self._storage.default_content_type
 191             )
 192             if self._storage.reduced_redundancy:
 193                 parameters["StorageClass"] = "REDUCED_REDUNDANCY"
 194             if self._storage.encryption:
 195                 parameters["ServerSideEncryption"] = "AES256"
 196             self._multipart = self.obj.initiate_multipart_upload(**parameters)
 197         if self.buffer_size <= self._buffer_file_size:
 198             self._flush_write_buffer()
 199         return super(S3Boto3StorageFile, self).write(force_bytes(content))
 200 
 201     @property
 202     def _buffer_file_size(self):
 203         pos = self.file.tell()
 204         self.file.seek(0, os.SEEK_END)
 205         length = self.file.tell()
 206         self.file.seek(pos)
 207         return length
 208 
 209     def _flush_write_buffer(self):
 210         """
 211         Flushes the write buffer.
 212         """
 213         if self._buffer_file_size:
 214             self._write_counter += 1
 215             self.file.seek(0)
 216             part = self._multipart.Part(self._write_counter)
 217             part.upload(Body=self.file.read())
 218 
 219     def close(self):
 220         if self._is_dirty:
 221             self._flush_write_buffer()
 222             # TODO: Possibly cache the part ids as they're being uploaded
 223             # instead of requesting parts from server. For now, emulating
 224             # s3boto's behavior.
 225             parts = [
 226                 {"ETag": part.e_tag, "PartNumber": part.part_number}
 227                 for part in self._multipart.parts.all()
 228             ]
 229             self._multipart.complete(MultipartUpload={"Parts": parts})
 230         else:
 231             if self._multipart is not None:
 232                 self._multipart.abort()
 233         if self._file is not None:
 234             self._file.close()
 235             self._file = None

 238 class S3Boto3Storage(Storage):
 239     """
 240     Amazon Simple Storage Service using Boto3
 241 
 242     This storage backend supports opening files in read or write
 243     mode and supports streaming(buffering) data in chunks to S3
 244     when writing.
 245     """
 246 
 247     # XXX: note that this file reads entirely into memory before the first
 248     # read happens.  This means that it should only be used for small
 249     # files (eg: see how sentry.models.file works with it through the
 250     # ChunkedFileBlobIndexWrapper.
 251     connection_class = staticmethod(resource)
 252     connection_service_name = "s3"
 253     default_content_type = "application/octet-stream"
 254     connection_response_error = ClientError
 255     file_class = S3Boto3StorageFile
 256     # If config provided in init, signature_version and addressing_style settings/args are ignored.
 257     config = None
 258 
 259     # used for looking up the access and secret key from env vars
 260     access_key_names = ["AWS_S3_ACCESS_KEY_ID", "AWS_ACCESS_KEY_ID"]
 261     secret_key_names = ["AWS_S3_SECRET_ACCESS_KEY", "AWS_SECRET_ACCESS_KEY"]
 262 
 263     access_key = None
 264     secret_key = None
 265     file_overwrite = True
 266     object_parameters = {}
 267     bucket_name = None
 268     auto_create_bucket = False
 269     default_acl = "public-read"
 270     bucket_acl = default_acl
 271     querystring_auth = True
 272     querystring_expire = 3600
 273     signature_version = None
 274     reduced_redundancy = False
 275     location = ""
 276     encryption = False
 277     custom_domain = None
 278     addressing_style = None
 279     secure_urls = True
 280     file_name_charset = "utf-8"
 281     gzip = False
 282     preload_metadata = False
 283     gzip_content_types = (
 284         "text/css",
 285         "text/javascript",
 286         "application/javascript",
 287         "application/x-javascript",
 288         "image/svg+xml",
 289     )
 290     url_protocol = "https:"
 291     endpoint_url = None
 292     region_name = None
 293     use_ssl = True
 294 
 295     def __init__(self, acl=None, bucket=None, **settings):
 296         # check if some of the settings we've provided as class attributes
 297         # need to be overwritten with values passed in here
 298         for name, value in settings.items():
 299             if hasattr(self, name):
 300                 setattr(self, name, value)
 301 
 302         # For backward-compatibility of old differing parameter names
 303         if acl is not None:
 304             self.default_acl = acl
 305         if bucket is not None:
 306             self.bucket_name = bucket
 307 
 308         self.location = (self.location or "").lstrip("/")
 309         # Backward-compatibility: given the anteriority of the SECURE_URL setting
 310         # we fall back to https if specified in order to avoid the construction
 311         # of unsecure urls.
 312         if self.secure_urls:
 313             self.url_protocol = "https:"
 314 
 315         self._entries = {}
 316         self._bucket = None
 317         self._connection = None
 318 
 319         if not self.access_key and not self.secret_key:
 320             self.access_key, self.secret_key = self._get_access_keys()
 321 
 322         if not self.config:
 323             self.config = Config(
 324                 s3={"addressing_style": self.addressing_style},
 325                 signature_version=self.signature_version,
 326             )

 328     @property
 329     def connection(self):
 330         # TODO: Support host, port like in s3boto
 331         # Note that proxies are handled by environment variables that the underlying
 332         # urllib/requests libraries read. See https://github.com/boto/boto3/issues/338
 333         # and http://docs.python-requests.org/en/latest/user/advanced/#proxies
 334         if self._connection is None:
 335 
 336             # If this is running on an ec2 instance, allow boto to connect using an IAM role
 337             # instead of explicitly provided an access key and secret
 338             # http://boto3.readthedocs.io/en/latest/guide/configuration.html#iam-role
 339             kwargs = {}
 340             if self.access_key and self.secret_key:
 341                 kwargs["aws_access_key_id"] = self.access_key
 342                 kwargs["aws_secret_access_key"] = self.secret_key
 343 
 344             self._connection = self.connection_class(
 345                 self.connection_service_name,
 346                 region_name=self.region_name,
 347                 use_ssl=self.use_ssl,
 348                 endpoint_url=self.endpoint_url,
 349                 config=self.config,
 350                 **kwargs
 351             )
 352         return self._connection

 354     @property
 355     def bucket(self):
 356         """
 357         Get the current bucket. If there is no current bucket object
 358         create it.
 359         """
 360         if self._bucket is None:
 361             self._bucket = self._get_or_create_bucket(self.bucket_name)
 362         return self._bucket
 363 
 364     @property
 365     def entries(self):
 366         """
 367         Get the locally cached files for the bucket.
 368         """
 369         if self.preload_metadata and not self._entries:
 370             self._entries = dict(
 371                 (self._decode_name(entry.key), entry)
 372                 for entry in self.bucket.objects.filter(Prefix=self.location)
 373             )
 374         return self._entries

 376     def _get_access_keys(self):
 377         """
 378         Gets the access keys to use when accessing S3. If none
 379         are provided to the class in the constructor or in the
 380         settings then get them from the environment variables.
 381         """
 382 
 383         def lookup_env(names):
 384             for name in names:
 385                 value = os.environ.get(name)
 386                 if value:
 387                     return value
 388 
 389         access_key = self.access_key or lookup_env(self.access_key_names)
 390         secret_key = self.secret_key or lookup_env(self.secret_key_names)
 391         return access_key, secret_key
 392 
 393     def _get_or_create_bucket(self, name):
 394         """
 395         Retrieves a bucket if it exists, otherwise creates it.
 396         """
 397         bucket = self.connection.Bucket(name)
 398         if self.auto_create_bucket:
 399             try:
 400                 # Directly call head_bucket instead of bucket.load() because head_bucket()
 401                 # fails on wrong region, while bucket.load() does not.
 402                 bucket.meta.client.head_bucket(Bucket=name)
 403             except self.connection_response_error as err:
 404                 if err.response["ResponseMetadata"]["HTTPStatusCode"] == 301:
 405                     raise ImproperlyConfigured(
 406                         "Bucket %s exists, but in a different "
 407                         "region than we are connecting to. Set "
 408                         "the region to connect to by setting "
 409                         "AWS_S3_REGION_NAME to the correct region." % name
 410                     )
 411                     # Notes: When using the us-east-1 Standard endpoint, you can create
 412                     # buckets in other regions. The same is not true when hitting region specific
 413                     # endpoints. However, when you create the bucket not in the same region, the
 414                     # connection will fail all future requests to the Bucket after the creation
 415                     # (301 Moved Permanently).
 416                     #
 417                     # For simplicity, we enforce in S3Boto3Storage that any auto-created
 418                     # bucket must match the region that the connection is for.
 419                     #
 420                     # Also note that Amazon specifically disallows "us-east-1" when passing bucket
 421                     # region names; LocationConstraint *must* be blank to create in US Standard.
 422                     bucket_params = {"ACL": self.bucket_acl}
 423                     region_name = self.connection.meta.client.meta.region_name
 424                     if region_name != "us-east-1":
 425                         bucket_params["CreateBucketConfiguration"] = {
 426                             "LocationConstraint": region_name
 427                         }
 428                     bucket.create(ACL=self.bucket_acl)
 429                 else:
 430                     raise ImproperlyConfigured(
 431                         "Bucket %s does not exist. Buckets "
 432                         "can be automatically created by "
 433                         "setting AWS_AUTO_CREATE_BUCKET to "
 434                         "``True``." % name
 435                     )
 436         return bucket

 438     def _clean_name(self, name):
 439         """
 440         Cleans the name so that Windows style paths work
 441         """
 442         # Normalize Windows style paths
 443         clean_name = posixpath.normpath(name).replace("\\", "/")
 444 
 445         # os.path.normpath() can strip trailing slashes so we implement
 446         # a workaround here.
 447         if name.endswith("/") and not clean_name.endswith("/"):
 448             # Add a trailing slash as it was stripped.
 449             return clean_name + "/"
 450         else:
 451             return clean_name
 452 
 453     def _normalize_name(self, name):
 454         """
 455         Normalizes the name so that paths like /path/to/ignored/../something.txt
 456         work. We check to make sure that the path pointed to is not outside
 457         the directory specified by the LOCATION setting.
 458         """
 459         try:
 460             return safe_join(self.location, name)
 461         except ValueError:
 462             raise SuspiciousOperation("Attempted access to '%s' denied." % name)
 463 
 464     def _encode_name(self, name):
 465         return smart_str(name, encoding=self.file_name_charset)
 466 
 467     def _decode_name(self, name):
 468         return force_text(name, encoding=self.file_name_charset)
 469 
 470     def _compress_content(self, content):
 471         """Gzip a given string content."""
 472         zbuf = BytesIO()
 473         zfile = GzipFile(mode="wb", compresslevel=6, fileobj=zbuf)
 474         try:
 475             zfile.write(force_bytes(content.read()))
 476         finally:
 477             zfile.close()
 478         zbuf.seek(0)
 479         # Boto 2 returned the InMemoryUploadedFile with the file pointer replaced,
 480         # but Boto 3 seems to have issues with that. No need for fp.name in Boto3
 481         # so just returning the BytesIO directly
 482         return zbuf
 483 
 484     def _open(self, name, mode="rb"):
 485         name = self._normalize_name(self._clean_name(name))
 486         try:
 487             f = self.file_class(name, mode, self)
 488         except self.connection_response_error as err:
 489             if err.response["ResponseMetadata"]["HTTPStatusCode"] == 404:
 490                 raise IOError("File does not exist: %s" % name)
 491             raise  # Let it bubble up if it was some other error
 492         return f

 494     def _save(self, name, content):
 495         with metrics.timer("filestore.save", instance="s3"):
 496             cleaned_name = self._clean_name(name)
 497             name = self._normalize_name(cleaned_name)
 498             parameters = self.object_parameters.copy()
 499             content_type = getattr(
 500                 content, "content_type", mimetypes.guess_type(name)[0] or self.default_content_type
 501             )
 502 
 503             # setting the content_type in the key object is not enough.
 504             parameters.update({"ContentType": content_type})
 505 
 506             if self.gzip and content_type in self.gzip_content_types:
 507                 content = self._compress_content(content)
 508                 parameters.update({"ContentEncoding": "gzip"})
 509 
 510             encoded_name = self._encode_name(name)
 511             obj = self.bucket.Object(encoded_name)
 512             if self.preload_metadata:
 513                 self._entries[encoded_name] = obj
 514 
 515             self._save_content(obj, content, parameters=parameters)
 516             # Note: In boto3, after a put, last_modified is automatically reloaded
 517             # the next time it is accessed; no need to specifically reload it.
 518         return cleaned_name
 519 
 520     def _save_content(self, obj, content, parameters):
 521         # only pass backwards incompatible arguments if they vary from the default
 522         put_parameters = parameters.copy() if parameters else {}
 523         if self.encryption:
 524             put_parameters["ServerSideEncryption"] = "AES256"
 525         if self.reduced_redundancy:
 526             put_parameters["StorageClass"] = "REDUCED_REDUNDANCY"
 527         if self.default_acl:
 528             put_parameters["ACL"] = self.default_acl
 529         content.seek(0, os.SEEK_SET)
 530         obj.upload_fileobj(content, ExtraArgs=put_parameters)
 531 
 532     def delete(self, name):
 533         name = self._normalize_name(self._clean_name(name))
 534         self.bucket.Object(self._encode_name(name)).delete()
 535 
 536     def exists(self, name):
 537         if not name:
 538             try:
 539                 self.bucket
 540                 return True
 541             except ImproperlyConfigured:
 542                 return False
 543         name = self._normalize_name(self._clean_name(name))
 544         if self.entries:
 545             return name in self.entries
 546         obj = self.bucket.Object(self._encode_name(name))
 547         try:
 548             obj.load()
 549             return True
 550         except self.connection_response_error:
 551             return False
 552 
 553     def listdir(self, name):
 554         name = self._normalize_name(self._clean_name(name))
 555         # for the bucket.objects.filter and logic below name needs to end in /
 556         # But for the root path "" we leave it as an empty string
 557         if name and not name.endswith("/"):
 558             name += "/"
 559 
 560         files = []
 561         dirs = set()
 562         base_parts = name.split("/")[:-1]
 563         for item in self.bucket.objects.filter(Prefix=self._encode_name(name)):
 564             parts = item.key.split("/")
 565             parts = parts[len(base_parts) :]
 566             if len(parts) == 1:
 567                 # File
 568                 files.append(parts[0])
 569             elif len(parts) > 1:
 570                 # Directory
 571                 dirs.add(parts[0])
 572         return list(dirs), files
 573 
 574     def size(self, name):
 575         name = self._normalize_name(self._clean_name(name))
 576         if self.entries:
 577             entry = self.entries.get(name)
 578             if entry:
 579                 return entry.content_length
 580             return 0
 581         return self.bucket.Object(self._encode_name(name)).content_length

 583     def get_modified_time(self, name):
 584         """
 585         Returns an (aware) datetime object containing the last modified time if
 586         USE_TZ is True, otherwise returns a naive datetime in the local timezone.
 587         """
 588         name = self._normalize_name(self._clean_name(name))
 589         entry = self.entries.get(name)
 590         # only call self.bucket.Object() if the key is not found
 591         # in the preloaded metadata.
 592         if entry is None:
 593             entry = self.bucket.Object(self._encode_name(name))
 594         if settings.USE_TZ:
 595             # boto3 returns TZ aware timestamps
 596             return entry.last_modified
 597         else:
 598             return localtime(entry.last_modified).replace(tzinfo=None)
 599 
 600     def modified_time(self, name):
 601         """Returns a naive datetime object containing the last modified time."""
 602         return localtime(self.get_modified_time(name)).replace(tzinfo=None)

 604     def _strip_signing_parameters(self, url):
 605         # Boto3 does not currently support generating URLs that are unsigned. Instead we
 606         # take the signed URLs and strip any querystring params related to signing and expiration.
 607         # Note that this may end up with URLs that are still invalid, especially if params are
 608         # passed in that only work with signed URLs, e.g. response header params.
 609         # The code attempts to strip all query parameters that match names of known parameters
 610         # from v2 and v4 signatures, regardless of the actual signature version used.
 611         split_url = urlparse.urlsplit(url)
 612         qs = urlparse.parse_qsl(split_url.query, keep_blank_values=True)
 613         blacklist = set(
 614             [
 615                 "x-amz-algorithm",
 616                 "x-amz-credential",
 617                 "x-amz-date",
 618                 "x-amz-expires",
 619                 "x-amz-signedheaders",
 620                 "x-amz-signature",
 621                 "x-amz-security-token",
 622                 "awsaccesskeyid",
 623                 "expires",
 624                 "signature",
 625             ]
 626         )
 627         filtered_qs = ((key, val) for key, val in qs if key.lower() not in blacklist)
 628         # Note: Parameters that did not have a value in the original query string will have
 629         # an '=' sign appended to it, e.g ?foo&bar becomes ?foo=&bar=
 630         joined_qs = ("=".join(keyval) for keyval in filtered_qs)
 631         split_url = split_url._replace(query="&".join(joined_qs))
 632         return split_url.geturl()
 633 
 634     def url(self, name, parameters=None, expire=None):
 635         # Preserve the trailing slash after normalizing the path.
 636         # TODO: Handle force_http=not self.secure_urls like in s3boto
 637         name = self._normalize_name(self._clean_name(name))
 638         if self.custom_domain:
 639             return "%s//%s/%s" % (self.url_protocol, self.custom_domain, filepath_to_uri(name))
 640         if expire is None:
 641             expire = self.querystring_expire
 642 
 643         params = parameters.copy() if parameters else {}
 644         params["Bucket"] = self.bucket.name
 645         params["Key"] = self._encode_name(name)
 646         url = self.bucket.meta.client.generate_presigned_url(
 647             "get_object", Params=params, ExpiresIn=expire
 648         )
 649         if self.querystring_auth:
 650             return url
 651         return self._strip_signing_parameters(url)
 652 
 653     def get_available_name(self, name, max_length=None):
 654         """Overwrite existing file with the same name."""
 655         if self.file_overwrite:
 656             name = self._clean_name(name)
 657             return name
 658         return super(S3Boto3Storage, self).get_available_name(name, max_length)
