  41 def try_repeated(func):
  42     """
  43     Runs a function a few times ignoring errors we see from GCS
  44     due to what appears to be network issues.  This is a temporary workaround
  45     until we can find the root cause.
  46     """
  47     if hasattr(func, "__name__"):
  48         func_name = func.__name__
  49     elif hasattr(func, "func"):
  50         # Partials
  51         func_name = getattr(func.func, "__name__", "__unknown__")
  52     else:
  53         func_name = "__unknown__"
  54 
  55     metrics_key = "filestore.gcs.retry"
  56     metrics_tags = {"function": func_name}
  57     idx = 0
  58     while True:
  59         try:
  60             result = func()
  61             metrics_tags.update({"success": "1"})
  62             metrics.timing(metrics_key, idx, tags=metrics_tags)
  63             return result
  64         except (DataCorruption, TransportError, RefreshError, RequestException, OpenSSLError) as e:
  65             if idx >= GCS_RETRIES:
  66                 metrics_tags.update({"success": "0", "exception_class": e.__class__.__name__})
  67                 metrics.timing(metrics_key, idx, tags=metrics_tags)
  68                 raise
  69         idx += 1

  72 def get_client(project_id, credentials):
  73     global _client
  74     if _client[2] is None or (project_id, credentials) != (_client[0], _client[1]):
  75         client = Client(project=project_id, credentials=credentials)
  76         session = client._http
  77         adapter = TimeoutAdapter(timeout=GCS_TIMEOUT)
  78         session.mount("http://", adapter)
  79         session.mount("https://", adapter)
  80         _client = (project_id, credentials, client)
  81     return _client[2]

  84 def clean_name(name):
  85     """
  86     Cleans the name so that Windows style paths work
  87     """
  88     # Normalize Windows style paths
  89     clean_name = posixpath.normpath(name).replace("\\", "/")
  90 
  91     # os.path.normpath() can strip trailing slashes so we implement
  92     # a workaround here.
  93     if name.endswith("/") and not clean_name.endswith("/"):
  94         # Add a trailing slash as it was stripped.
  95         clean_name = clean_name + "/"
  96 
  97     # Given an empty string, os.path.normpath() will return ., which we don't want
  98     if clean_name == ".":
  99         clean_name = ""
 100 
 101     return clean_name

 104 def safe_join(base, *paths):
 105     """
 106     A version of django.utils._os.safe_join for S3 paths.
 107     Joins one or more path components to the base path component
 108     intelligently. Returns a normalized version of the final path.
 109     The final path must be located inside of the base path component
 110     (otherwise a ValueError is raised).
 111     Paths outside the base path indicate a possible security
 112     sensitive operation.
 113     """
 114     base_path = force_text(base)
 115     base_path = base_path.rstrip("/")
 116     paths = [force_text(p) for p in paths]
 117 
 118     final_path = base_path + "/"
 119     for path in paths:
 120         _final_path = posixpath.normpath(posixpath.join(final_path, path))
 121         # posixpath.normpath() strips the trailing /. Add it back.
 122         if path.endswith("/") or _final_path + "/" == final_path:
 123             _final_path += "/"
 124         final_path = _final_path
 125     if final_path == base_path:
 126         final_path += "/"
 127 
 128     # Ensure final_path starts with base_path and that the next character after
 129     # the base path is /.
 130     base_path_len = len(base_path)
 131     if not final_path.startswith(base_path) or final_path[base_path_len] != "/":
 132         raise ValueError("the joined path is located outside of the base path" " component")
 133 
 134     return final_path.lstrip("/")

 137 class FancyBlob(Blob):

 142     def _get_download_url(self):
 143         if self.media_link is None:
 144             download_url = u"{download_url}/download/storage/v1{path}?alt=media".format(
 145                 download_url=self.download_url, path=self.path
 146             )
 147             if self.generation is not None:
 148                 download_url += u"&generation={:d}".format(self.generation)

 150         else:
 151             return self.media_link

 154 class GoogleCloudFile(File):
 155     def __init__(self, name, mode, storage):
 156         self.name = name
 157         self.mime_type = mimetypes.guess_type(name)[0]
 158         self._mode = mode
 159         self._storage = storage
 160         # NOTE(mattrobenolt): This is the same change in behavior as in
 161         # the s3 backend. We're opting now to load the file
 162         # or metadata at this step. This means we won't actually
 163         # know a file doesn't exist until we try to read it.
 164         self.blob = FancyBlob(storage.download_url, self.name, storage.bucket)
 165         self._file = None
 166         self._is_dirty = False
 167 
 168     @property
 169     def size(self):
 170         return self.blob.size
 171 
 172     def _get_file(self):
 173         def _try_download():
 174             self.blob.download_to_file(self._file)
 175             self._file.seek(0)
 176 
 177         if self._file is None:
 178             with metrics.timer("filestore.read", instance="gcs"):
 179                 self._file = SpooledTemporaryFile(
 180                     max_size=self._storage.max_memory_size, suffix=".GSStorageFile", dir=None
 181                 )
 182                 if "r" in self._mode:
 183                     self._is_dirty = False
 184                     try_repeated(_try_download)
 185         return self._file
 186 
 187     def _set_file(self, value):
 188         self._file = value
 189 
 190     file = property(_get_file, _set_file)
 191 
 192     def read(self, num_bytes=None):
 193         if "r" not in self._mode:
 194             raise AttributeError("File was not opened in read mode.")
 195 
 196         if num_bytes is None:
 197             num_bytes = -1
 198 
 199         return super(GoogleCloudFile, self).read(num_bytes)
 200 
 201     def write(self, content):
 202         if "w" not in self._mode:
 203             raise AttributeError("File was not opened in write mode.")
 204         self._is_dirty = True
 205         return super(GoogleCloudFile, self).write(force_bytes(content))
 206 
 207     def close(self):
 208         def _try_upload():
 209             self.file.seek(0)
 210             self.blob.upload_from_file(self.file, content_type=self.mime_type)
 211 
 212         if self._file is not None:
 213             if self._is_dirty:
 214                 try_repeated(_try_upload)
 215             self._file.close()
 216             self._file = None

 219 class GoogleCloudStorage(Storage):
 220     project_id = None
 221     credentials = None
 222     bucket_name = None
 223     file_name_charset = "utf-8"
 224     file_overwrite = True
 225     download_url = "https://www.googleapis.com"
 226     # The max amount of memory a returned file can take up before being
 227     # rolled over into a temporary file on disk. Default is 0: Do not roll over.
 228     max_memory_size = 0
 229 
 230     def __init__(self, **settings):
 231         # check if some of the settings we've provided as class attributes
 232         # need to be overwritten with values passed in here
 233         for name, value in settings.items():
 234             if hasattr(self, name):
 235                 setattr(self, name, value)
 236 
 237         self._bucket = None
 238         self._client = None
 239 
 240     @property
 241     def client(self):
 242         if self._client is None:
 243             self._client = get_client(self.project_id, self.credentials)
 244         return self._client
 245 
 246     @property
 247     def bucket(self):
 248         if self._bucket is None:
 249             self._bucket = Bucket(self.client, name=self.bucket_name)
 250         return self._bucket
 251 
 252     def _normalize_name(self, name):
 253         """
 254         Normalizes the name so that paths like /path/to/ignored/../something.txt
 255         and ./file.txt work.  Note that clean_name adds ./ to some paths so
 256         they need to be fixed here.
 257         """
 258         return safe_join("", name)
 259 
 260     def _encode_name(self, name):
 261         return smart_str(name, encoding=self.file_name_charset)
 262 
 263     def _open(self, name, mode="rb"):
 264         name = self._normalize_name(clean_name(name))
 265         return GoogleCloudFile(name, mode, self)
 266 
 267     def _save(self, name, content):
 268         def _try_upload():
 269             content.seek(0, os.SEEK_SET)
 270             file.blob.upload_from_file(content, size=content.size, content_type=file.mime_type)
 271 
 272         with metrics.timer("filestore.save", instance="gcs"):
 273             cleaned_name = clean_name(name)
 274             name = self._normalize_name(cleaned_name)
 275 
 276             content.name = cleaned_name
 277             encoded_name = self._encode_name(name)
 278             file = GoogleCloudFile(encoded_name, "w", self)
 279             try_repeated(_try_upload)
 280         return cleaned_name
 281 
 282     def delete(self, name):
 283         def _try_delete():
 284             normalized_name = self._normalize_name(clean_name(name))
 285             self.bucket.delete_blob(self._encode_name(normalized_name))
 286 
 287         try:
 288             try_repeated(_try_delete)
 289         except NotFound:
 290             pass
 291 
 292     def exists(self, name):
 293         if not name:  # root element aka the bucket
 294             try:
 295                 self.bucket
 296                 return True
 297             except ImproperlyConfigured:
 298                 return False
 299 
 300         name = self._normalize_name(clean_name(name))
 301         return bool(self.bucket.get_blob(self._encode_name(name)))
 302 
 303     def listdir(self, name):
 304         name = self._normalize_name(clean_name(name))
 305         # for the bucket.list and logic below name needs to end in /
 306         # But for the root path "" we leave it as an empty string
 307         if name and not name.endswith("/"):
 308             name += "/"
 309 
 310         files_list = list(self.bucket.list_blobs(prefix=self._encode_name(name)))
 311         files = []
 312         dirs = set()
 313 
 314         base_parts = name.split("/")[:-1]
 315         for item in files_list:
 316             parts = item.name.split("/")
 317             parts = parts[len(base_parts) :]
 318             if len(parts) == 1 and parts[0]:
 319                 # File
 320                 files.append(parts[0])
 321             elif len(parts) > 1 and parts[0]:
 322                 # Directory
 323                 dirs.add(parts[0])
 324         return list(dirs), files
 325 
 326     def _get_blob(self, name):
 327         # Wrap google.cloud.storage's blob to raise if the file doesn't exist
 328         blob = self.bucket.get_blob(name)
 329 
 330         if blob is None:
 331             raise NotFound(u"File does not exist: {}".format(name))
 332 
 333         return blob
 334 
 335     def size(self, name):
 336         name = self._normalize_name(clean_name(name))
 337         blob = self._get_blob(self._encode_name(name))
 338         return blob.size
 339 
 340     def modified_time(self, name):
 341         name = self._normalize_name(clean_name(name))
 342         blob = self._get_blob(self._encode_name(name))
 343         return timezone.make_naive(blob.updated)
 344 
 345     def get_modified_time(self, name):
 346         name = self._normalize_name(clean_name(name))
 347         blob = self._get_blob(self._encode_name(name))
 348         updated = blob.updated
 349         return updated if settings.USE_TZ else timezone.make_naive(updated)
 350 
 351     def url(self, name):
 352         # Preserve the trailing slash after normalizing the path.
 353         name = self._normalize_name(clean_name(name))
 354         blob = self._get_blob(self._encode_name(name))
 355         return blob.public_url
 356 
 357     def get_available_name(self, name, max_length=None):
 358         if self.file_overwrite:
 359             name = clean_name(name)
 360             return name
 361         return super(GoogleCloudStorage, self).get_available_name(name, max_length)
