  13 def get_package_version(module_name, app):
  14     version = None
  15 
  16     # Try to pull version from pkg_resource first
  17     # as it is able to detect version tagged with egg_info -b
  18     if pkg_resources is not None:
  19         # pull version from pkg_resources if distro exists
  20         try:
  21             return pkg_resources.get_distribution(module_name).version
  22         except Exception:
  23             pass
  24 
  25     if hasattr(app, "get_version"):
  26         version = app.get_version
  27     elif hasattr(app, "__version__"):
  28         version = app.__version__
  29     elif hasattr(app, "VERSION"):
  30         version = app.VERSION
  31     elif hasattr(app, "version"):
  32         version = app.version
  33 
  34     if callable(version):
  35         try:
  36             version = version()
  37         except Exception:
  38             return None
  39 
  40     if not isinstance(version, six.string_types + (list, tuple)):
  41         version = None
  42 
  43     if version is None:
  44         return None
  45 
  46     if isinstance(version, (list, tuple)):
  47         version = ".".join(map(six.text_type, version))
  48 
  49     return six.text_type(version)

  52 def get_all_package_versions():
  53     packages = {}
  54     for module_name, app in sys.modules.items():
  55         # ignore items that look like submodules
  56         if "." in module_name:
  57             continue
  58 
  59         if "sys" == module_name:
  60             continue
  61 
  62         version = get_package_version(module_name, app)
  63 
  64         if version is None:
  65             continue
  66 
  67         packages[module_name] = version
  68 
  69     packages["sys"] = u"{0}.{1}.{2}".format(*sys.version_info)
  70 
  71     return packages
