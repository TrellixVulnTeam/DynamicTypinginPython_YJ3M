  26 def split_urls(value):
  27     if not value:
  28         return ()
  29     return filter(bool, (url.strip() for url in value.splitlines()))
  30 
  31 
  32 def validate_urls(value, **kwargs):
  33     urls = split_urls(value)
  34     if any((not u.startswith(("http://", "https://")) or not is_valid_url(u)) for u in urls):
  35         raise PluginError("Not a valid URL.")
  36     return "\n".join(urls)

  97     def get_group_data(self, group, event, triggering_rules):
  98         data = {
  99             "id": six.text_type(group.id),
 100             "project": group.project.slug,
 101             "project_name": group.project.name,
 102             "project_slug": group.project.slug,
 103             "logger": event.get_tag("logger"),
 104             "level": event.get_tag("level"),
 105             "culprit": group.culprit,
 106             "message": event.message,
 107             "url": group.get_absolute_url(params={"referrer": "webhooks_plugin"}),
 108             "triggering_rules": triggering_rules,
 109         }
 110         data["event"] = dict(event.data or {})
 111         data["event"]["tags"] = event.tags
 112         data["event"]["event_id"] = event.event_id
 113         data["event"]["id"] = event.event_id
 114         return data

 116     def get_webhook_urls(self, project):

 119     def send_webhook(self, url, payload):
 120         return safe_urlopen(url=url, json=payload, timeout=self.timeout, verify_ssl=False)

 122     def notify_users(self, group, event, triggering_rules, fail_silently=False, **kwargs):
 123         payload = self.get_group_data(group, event, triggering_rules)
 124         for url in self.get_webhook_urls(group.project):
 125             # TODO: Use API client with raise_error
 126             safe_execute(self.send_webhook, url, payload, _with_transaction=False)
