  24 class ConfigValidator(object):
  25     def __init__(self, config, data=None, initial=None, context=None):
  26         self.errors = {}
  27         self.result = {}
  28         self.context = context or {}
  29 
  30         self.config = OrderedDict(((f["name"], f) for f in config))
  31 
  32         self._data = data or {}
  33         self._initial = initial or {}
  34         self._validated = False
  35 
  36     def is_valid(self):
  37         data = self._data
  38         initial = self._initial
  39         cleaned = self.result
  40         errors = self.errors
  41         for field in six.itervalues(self.config):
  42             key = field["name"]
  43             value = data.get(key, initial.get(key))
  44 
  45             if field.get("required") and not value:
  46                 errors[key] = ERR_FIELD_REQUIRED
  47 
  48             try:
  49                 value = self.validate_field(name=key, value=value)
  50             except (forms.ValidationError, serializers.ValidationError, PluginError) as e:
  51                 errors[key] = six.text_type(e)
  52 
  53             if not errors.get(key):
  54                 cleaned[key] = value
  55 
  56         self._validated = True
  57         return not errors
  58 
  59     def validate_field(self, name, value):
  60         """
  61         ```
  62         if name == 'foo' and value != 'bar':
  63             raise PluginError('foo must be bar')
  64         return value
  65         ```
  66         """
  67         field = self.config[name]
  68         if value is None:
  69             if field.get("required"):
  70                 raise PluginError("Field is required")
  71             return value
  72 
  73         if isinstance(value, six.string_types):
  74             value = value.strip()
  75             # TODO(dcramer): probably should do something with default
  76             # validations here, though many things will end up bring string
  77             # based
  78             if not value and field.get("required"):
  79                 raise PluginError("Field is required")
  80 
  81         for validator in DEFAULT_VALIDATORS.get(field["type"], ()):
  82             value = validator(value=value)
  83 
  84         for validator in field.get("validators", ()):
  85             value = validator(value=value, **self.context)
  86         return value

 111     def validate_config_field(self, project, name, value, actor=None):
 112         """
 113         ```
 114         if name == 'foo' and value != 'bar':
 115             raise PluginError('foo must be bar')
 116         return value
 117         ```
 118         """
 119         for config in self.get_config(project=project, user=actor):
 120             if config["name"] != name:
 121                 continue
 122 
 123             if value is None:
 124                 if config.get("required"):
 125                     raise PluginError("Field is required")
 126                 if config.get("type") == "secret":
 127                     value = self.get_option(name, project)
 128                 return value
 129 
 130             if isinstance(value, six.string_types):
 131                 value = value.strip()
 132                 # TODO(dcramer): probably should do something with default
 133                 # validations here, though many things will end up bring string
 134                 # based
 135                 if not value:
 136                     if config.get("required"):
 137                         raise PluginError("Field is required")
 138                     if config.get("type") == "secret":
 139                         value = self.get_option(name, project)
 140 
 141             for validator in DEFAULT_VALIDATORS.get(config["type"], ()):
 142                 value = validator(project=project, value=value, actor=actor)
 143 
 144             for validator in config.get("validators", ()):
 145                 value = validator(value, project=project, actor=actor)
 146             return value
 147         return value
