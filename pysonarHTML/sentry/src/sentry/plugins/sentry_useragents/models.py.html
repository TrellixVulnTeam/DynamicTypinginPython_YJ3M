   9 class UserAgentPlugin(TagPlugin):
  10     version = sentry.VERSION
  11     author = "Sentry Team"
  12     author_url = "https://github.com/getsentry/sentry"
  13     project_default_enabled = True
  14 
  15     def get_tag_values(self, event):
  16         contexts = event.interfaces.get("contexts")
  17         # disable tagging if contexts are present
  18         if contexts:
  19             return []
  20 
  21         http = event.interfaces.get("request")
  22         if not http:
  23             return []
  24         if not http.headers:
  25             return []
  26 
  27         headers = http.headers
  28         # XXX: transitional support for workers
  29         if isinstance(headers, dict):
  30             headers = headers.items()
  31 
  32         output = []
  33         for key, value in headers:
  34             if key != "User-Agent":
  35                 continue
  36             ua = Parse(value)
  37             if not ua:
  38                 continue
  39             result = self.get_tag_from_ua(ua)
  40             if result:
  41                 output.append(result)
  42         return output

  45 class BrowserPlugin(UserAgentPlugin):
  46     """
  47     Automatically adds the 'browser' tag from events containing interface data
  48     from ``request``.
  49     """
  50 
  51     slug = "browsers"
  52     title = "Auto Tag: Browsers"
  53     tag = "browser"
  54 
  55     def get_tag_from_ua(self, ua):
  56         ua = ua["user_agent"]
  57 
  58         if not ua["family"]:
  59             return
  60 
  61         version = ".".join(value for value in [ua["major"], ua["minor"]] if value)
  62         tag = ua["family"]
  63         if version:
  64             tag += " " + version
  65 
  66         return tag
