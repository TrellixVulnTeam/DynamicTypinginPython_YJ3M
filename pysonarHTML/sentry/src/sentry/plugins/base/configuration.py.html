  47 def default_plugin_config(plugin, project, request):
  48     if plugin.can_enable_for_projects() and not plugin.can_configure_for_project(project):
  49         raise Http404()
  50 
  51     plugin_key = plugin.get_conf_key()
  52     form_class = plugin.get_conf_form(project)
  53     template = plugin.get_conf_template(project)
  54 
  55     if form_class is None:
  56         return HttpResponseRedirect(
  57             reverse("sentry-manage-project", args=[project.organization.slug, project.slug])
  58         )
  59 
  60     test_results = None
  61 
  62     form = form_class(
  63         request.POST if request.POST.get("plugin") == plugin.slug else None,
  64         initial=plugin.get_conf_options(project),
  65         prefix=plugin_key,
  66     )
  67     if form.is_valid():
  68         if "action_test" in request.POST and plugin.is_testable():
  69             test_results = plugin.test_configuration_and_get_test_results(project)
  70         else:
  71             for field, value in six.iteritems(form.cleaned_data):
  72                 key = "%s:%s" % (plugin_key, field)
  73                 if project:
  74                     ProjectOption.objects.set_value(project, key, value)
  75                 else:
  76                     options.set(key, value)
  77 
  78             messages.add_message(
  79                 request, messages.SUCCESS, _("Your settings were saved successfully.")
  80             )
  81             return HttpResponseRedirect(request.path)
  82 
  83     # TODO(mattrobenolt): Reliably determine if a plugin is configured
  84     # if hasattr(plugin, 'is_configured'):
  85     #     is_configured = plugin.is_configured(project)
  86     # else:
  87     #     is_configured = True
  88     is_configured = True
  89 
  90     return mark_safe(
  91         render_to_string(
  92             template=template,
  93             context={
  94                 "form": form,
  95                 "plugin": plugin,
  96                 "plugin_description": plugin.get_description() or "",
  97                 "plugin_test_results": test_results,
  98                 "plugin_is_configured": is_configured,
  99             },
 100             request=request,
 101         )
 102     )

 115 def default_plugin_options(plugin, project):
 116     form_class = plugin.get_conf_form(project)
 117     if form_class is None:
 118         return {}
 119 
 120     NOTSET = object()
 121     plugin_key = plugin.get_conf_key()
 122     initials = plugin.get_form_initial(project)
 123     for field in form_class.base_fields:
 124         key = "%s:%s" % (plugin_key, field)
 125         if project is not None:
 126             value = ProjectOption.objects.get_value(project, key, NOTSET)
 127         else:
 128             value = options.get(key)
 129         if value is not NOTSET:
 130             initials[field] = value
 131     return initials
