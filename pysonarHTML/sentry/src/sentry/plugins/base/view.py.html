  10 class PluggableViewMixin(object):
  11     """
  12     A mix-in which provides a render method which returns a special object to
  13     enable embedding of content within base-views.
  14     """
  15 
  16     def redirect(self, url):
  17         """
  18         Returns a redirect response type.
  19         """
  20         return HttpResponseRedirect(url)
  21 
  22     def render(self, template, context=None):
  23         """
  24         Given a template name, and an optional context (dictionary), returns a
  25         ready-to-render response.
  26 
  27         Default context includes the plugin instance.
  28 
  29         >>> self.render('template.html', {'hello': 'world'})
  30         """
  31         if context is None:
  32             context = {}
  33         context["plugin"] = self
  34         return Response(template, context)
