  11 class PluginManager(InstanceManager):
  12     def __iter__(self):
  13         return iter(self.all())
  14 
  15     def __len__(self):
  16         return sum(1 for i in self.all())

  18     def all(self, version=1):
  19         for plugin in sorted(super(PluginManager, self).all(), key=lambda x: x.get_title()):
  20             if not plugin.is_enabled():
  21                 continue
  22             if version is not None and plugin.__version__ != version:
  23                 continue
  24             yield plugin
  25 
  26     def plugin_that_can_be_configured(self):
  27         for plugin in self.all(version=None):
  28             if plugin.has_project_conf():
  29                 yield plugin
  30 
  31     def configurable_for_project(self, project, version=1):
  32         for plugin in self.all(version=version):
  33             if not safe_execute(plugin.can_configure_for_project, project, _with_transaction=False):
  34                 continue
  35             yield plugin
  36 
  37     def exists(self, slug):
  38         for plugin in self.all(version=None):
  39             if plugin.slug == slug:
  40                 return True
  41         return False
  42 
  43     def for_project(self, project, version=1):
  44         for plugin in self.all(version=version):
  45             if not safe_execute(plugin.is_enabled, project, _with_transaction=False):
  46                 continue
  47             yield plugin
  48 
  49     def for_site(self, version=1):
  50         for plugin in self.all(version=version):
  51             if not plugin.has_site_conf():
  52                 continue
  53             yield plugin
  54 
  55     def get(self, slug):
  56         for plugin in self.all(version=None):
  57             if plugin.slug == slug:
  58                 return plugin
  59         raise KeyError(slug)
  60 
  61     def first(self, func_name, *args, **kwargs):
  62         version = kwargs.pop("version", 1)
  63         for plugin in self.all(version=version):
  64             try:
  65                 result = getattr(plugin, func_name)(*args, **kwargs)
  66             except Exception as e:
  67                 logger = logging.getLogger("sentry.plugins.%s" % (type(plugin).slug,))
  68                 logger.error("%s.process_error", func_name, exc_info=True, extra={"exception": e})
  69                 continue
  70 
  71             if result is not None:
  72                 return result

  74     def register(self, cls):
  75         self.add("%s.%s" % (cls.__module__, cls.__name__))
  76         return cls
  77 
  78     def unregister(self, cls):
  79         self.remove("%s.%s" % (cls.__module__, cls.__name__))
  80         return cls
