  13 def load_plugin_urls(plugins):
  14     urlpatterns = []
  15     for plugin in plugins:
  16         urls = plugin.get_project_urls()
  17         if not urls:
  18             continue
  19         try:
  20             # a plugin's get_project_urls should return an iterable of url()'s,
  21             # which can either be RegexURLResolver or RegexURLPattern
  22             for u in urls:
  23                 if not isinstance(u, (RegexURLResolver, RegexURLPattern)):
  24                     raise TypeError(
  25                         "url must be RegexURLResolver or RegexURLPattern, not %r: %r"
  26                         % (type(u).__name__, u)
  27                     )
  28         except Exception:
  29             logger.exception("routes.failed", extra={"plugin": type(plugin).__name__})
  30         else:
  31             urlpatterns.append(url(r"^%s/" % re.escape(plugin.slug), include(urls)))
  32 
  33     return urlpatterns
