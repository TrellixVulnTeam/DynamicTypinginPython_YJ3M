  19 class PluginMount(type):
  20     def __new__(cls, name, bases, attrs):
  21         new_cls = type.__new__(cls, name, bases, attrs)
  22         if IPlugin2 in bases:
  23             return new_cls
  24         if new_cls.title is None:
  25             new_cls.title = new_cls.__name__
  26         if not new_cls.slug:
  27             new_cls.slug = new_cls.title.replace(" ", "-").lower()
  28         if not hasattr(new_cls, "logger"):
  29             new_cls.logger = logging.getLogger("sentry.plugins.%s" % (new_cls.slug,))
  30         return new_cls

  33 class IPlugin2(local, PluginConfigMixin, PluginStatusMixin):
  34     """
  35     Plugin interface. Should not be inherited from directly.
  36 
  37     A plugin should be treated as if it were a singleton. The owner does not
  38     control when or how the plugin gets instantiated, nor is it guaranteed that
  39     it will happen, or happen more than once.
  40 
  41     >>> from sentry.plugins.base.v2 import Plugin2
  42     >>>
  43     >>> class MyPlugin(Plugin2):
  44     >>>     def get_title(self):
  45     >>>         return 'My Plugin'
  46 
  47     As a general rule all inherited methods should allow ``**kwargs`` to ensure
  48     ease of future compatibility.
  49     """

  84     def is_enabled(self, project=None):
  85         """
  86         Returns a boolean representing if this plugin is enabled.
  87 
  88         If ``project`` is passed, it will limit the scope to that project.
  89 
  90         >>> plugin.is_enabled()
  91         """
  92         if not self.enabled:
  93             return False
  94 
  95         if not self.can_disable:
  96             return True
  97 
  98         if project:
  99             project_enabled = self.get_option("enabled", project)
 100             if project_enabled is not None:
 101                 return project_enabled
 102             else:
 103                 return self.project_default_enabled
 104 
 105         return True

 201     def get_conf_version(self, project):
 202         """
 203         Returns a version string that represents the current configuration state.
 204 
 205         If any option changes or new options added, the version will change.
 206 
 207         >>> plugin.get_conf_version(project)
 208         """
 209         options = self.get_conf_options(project)
 210         return md5_text("&".join(sorted("%s=%s" % o for o in six.iteritems(options)))).hexdigest()[
 211             :3

 232     def can_configure_for_project(self, project):
 233         """
 234         Checks if the plugin can be configured for a specific project.
 235         """
 236         from sentry import features
 237 
 238         if not self.enabled:
 239             return False
 240 
 241         if not features.has("projects:plugins", project, self, actor=None):
 242             return False
 243 
 244         if not self.can_disable:
 245             return True
 246 
 247         return True

 257     def render(self, template, context=None):
 258         """
 259         Given a template name, and an optional context (dictionary), returns a
 260         ready-to-render response.
 261

 403     def get_event_enhancers(self, data, **kwargs):
 404         """
 405         Return a list of enhancers to apply to the given event.

 408         input and returns modified data as output. If no changes to the data are

 410 
 411         As opposed to event (pre)processors, enhancers run **before** stacktrace
