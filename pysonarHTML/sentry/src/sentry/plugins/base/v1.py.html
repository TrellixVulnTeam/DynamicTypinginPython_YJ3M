  22 class PluginMount(type):
  23     def __new__(cls, name, bases, attrs):
  24         new_cls = type.__new__(cls, name, bases, attrs)
  25         if IPlugin in bases:
  26             return new_cls
  27         if new_cls.title is None:
  28             new_cls.title = new_cls.__name__
  29         if not new_cls.slug:
  30             new_cls.slug = new_cls.title.replace(" ", "-").lower()
  31         if not hasattr(new_cls, "logger") or new_cls.logger in [
  32             getattr(b, "logger", None) for b in bases
  33         ]:
  34             new_cls.logger = logging.getLogger("sentry.plugins.%s" % (new_cls.slug,))
  35         return new_cls

  38 class IPlugin(local, PluggableViewMixin, PluginConfigMixin, PluginStatusMixin):

  40     Plugin interface. Should not be inherited from directly.
  41 
  42     A plugin should be treated as if it were a singleton. The owner does not
  43     control when or how the plugin gets instantiated, nor is it guaranteed that
  44     it will happen, or happen more than once.
  45 
  46     >>> from sentry.plugins import Plugin  # NOQA
  47     >>> class MyPlugin(Plugin):
  48     >>>     title = 'My Plugin'
  49     >>>
  50     >>>     def widget(self, request, group, **kwargs):
  51     >>>         return self.render('myplugin/widget.html')
  52 
  53     All children should allow ``**kwargs`` on all inherited methods.
  54     """

  92     def is_enabled(self, project=None):
  93         """
  94         Returns a boolean representing if this plugin is enabled.
  95 
  96         If ``project`` is passed, it will limit the scope to that project.
  97 
  98         >>> plugin.is_enabled()
  99         """
 100         if not self.enabled:
 101             return False
 102         if not self.can_disable:
 103             return True
 104         if not self.can_enable_for_projects():
 105             return True
 106 
 107         if project:
 108             project_enabled = self.get_option("enabled", project)
 109             if project_enabled is not None:
 110                 return project_enabled
 111             else:
 112                 return self.project_default_enabled
 113 
 114         return True

 251     def can_configure_for_project(self, project):
 252         """
 253         Returns a boolean describing whether this plugin can be enabled on
 254         a per project basis
 255         """
 256         from sentry import features
 257 
 258         if not self.enabled:
 259             return False
 260         if not self.can_enable_for_projects():
 261             return False
 262 
 263         if not features.has("projects:plugins", project, self, actor=None):
 264             return False
 265 
 266         if not self.can_disable:
 267             return True
 268 
 269         return True

 308     def get_view_response(self, request, group):
 309         self.selected = request.path == self.get_url(group)
 310 
 311         if not self.selected:
 312             return
 313 
 314         response = self.view(request, group)
 315 
 316         if not response:
 317             return
 318 
 319         if isinstance(response, HttpResponseRedirect):
 320             return response
 321 
 322         if not isinstance(response, Response):
 323             raise NotImplementedError("Use self.render() when returning responses.")
 324 
 325         event = group.get_latest_event()
 326         if event:
 327             event.group = group
 328 
 329         request.access = access.from_request(request, group.organization)
 330 
 331         return response.respond(
 332             request,
 333             {
 334                 "plugin": self,
 335                 "project": group.project,
 336                 "group": group,
 337                 "event": event,
 338                 "can_admin_event": request.access.has_scope("event:write"),
 339                 "can_remove_event": request.access.has_scope("event:admin"),
 340             },
 341         )

 351     def before_events(self, request, group_list, **kwargs):
 352         """
 353         Allows preprocessing of groups in the list view.
 354 
 355         This is generally useful if you need to cache lookups
 356         for something like ``tags`` which would otherwise do
 357         multiple queries.
 358 
 359         If you use this **at all** you should ensure it's already
 360         reset on each execution.
 361 
 362         As an example, here's how we might get a reference to ticket ids we were
 363         storing per event, in an efficient O(1) manner.
 364 
 365         >>> def before_events(self, request, event_list, **kwargs):
 366         >>>     prefix = self.get_conf_key()
 367         >>>     GroupMeta.objects.get_value_bulk(event_list, '%s:tid' % prefix)
 368         """
