   1 from __future__ import absolute_import

   3 import sentry
   4 
   5 from sentry.plugins.bases.tag import TagPlugin

   7 
   8 class UrlsPlugin(TagPlugin):
   9     """
  10     Automatically adds the 'url' tag from events containing interface data
  11     from ``request``.
  12     """
  13 
  14     slug = "urls"
  15     title = "Auto Tag: URLs"
  16     version = sentry.VERSION
  17     author = "Sentry Team"
  18     author_url = "https://github.com/getsentry/sentry"
  19     tag = "url"
  20     project_default_enabled = True
  21 
  22     def get_tag_values(self, event):
  23         http = event.interfaces.get("request")
  24         if not http:
  25             return []
  26         if not http.url:
  27             return []
  28         return [http.url]
