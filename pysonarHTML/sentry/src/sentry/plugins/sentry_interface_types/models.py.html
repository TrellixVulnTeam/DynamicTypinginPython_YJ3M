  10 class InterfaceTypePlugin(TagPlugin):
  11     """
  12     Automatically adds the 'interface_type' tag from events containing referencing
  13     the class name of each interface (e.g. Http, Stacktrace, Exception).
  14     """
  15 
  16     description = __doc__
  17     slug = "interface_types"
  18     title = "Auto Tag: Interface Types"
  19     version = sentry.VERSION
  20     author = "Sentry Team"
  21     author_url = "https://github.com/getsentry/sentry"
  22     tag = "interface_type"
  23     project_default_enabled = False
  24 
  25     def get_tag_values(self, event):
  26         return [i.rsplit(".", 1)[-1] for i in six.iterkeys(event.interfaces)]
