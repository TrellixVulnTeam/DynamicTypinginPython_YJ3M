  29     def dispatch(self, request, organization, **kwargs):
  30         try:
  31             config = self.get_repository_data(organization, request.data)
  32             result = self.build_repository_config(organization=organization, data=config)
  33         except Exception as e:
  34             return self.handle_api_error(e)
  35 
  36         try:
  37             with transaction.atomic():
  38                 repo = Repository.objects.create(
  39                     organization_id=organization.id,
  40                     name=result["name"],
  41                     external_id=result.get("external_id"),
  42                     url=result.get("url"),
  43                     config=result.get("config") or {},
  44                     provider=self.id,
  45                     integration_id=result.get("integration_id"),
  46                 )
  47         except IntegrityError:
  48             # Try to delete webhook we just created
  49             try:
  50                 repo = Repository(
  51                     organization_id=organization.id,
  52                     name=result["name"],
  53                     external_id=result.get("external_id"),
  54                     url=result.get("url"),
  55                     config=result.get("config") or {},
  56                     provider=self.id,
  57                     integration_id=result.get("integration_id"),
  58                 )
  59                 self.on_delete_repository(repo)
  60             except IntegrationError:
  61                 pass
  62             return Response(
  63                 {"errors": {"__all__": "A repository with that name already exists"}}, status=400
  64             )
  65         else:
  66             repo_linked.send_robust(repo=repo, user=request.user, sender=self.__class__)
  67 
  68         analytics.record(
  69             "integration.repo.added",
  70             provider=self.id,
  71             id=result.get("integration_id"),
  72             organization_id=organization.id,
  73         )
  74         return Response(serialize(repo, request.user), status=201)

  76     def handle_api_error(self, e):
  77         context = {"error_type": "unknown"}
  78 
  79         if isinstance(e, IntegrationError):
  80             if "503" in six.text_type(e):
  81                 context.update(
  82                     {"error_type": "service unavailable", "errors": {"__all__": six.text_type(e)}}
  83                 )
  84                 status = 503
  85             else:
  86                 # TODO(dcramer): we should have a proper validation error
  87                 context.update(
  88                     {"error_type": "validation", "errors": {"__all__": six.text_type(e)}}
  89                 )
  90                 status = 400
  91         elif isinstance(e, Integration.DoesNotExist):
  92             context.update({"error_type": "not found", "errors": {"__all__": six.text_type(e)}})
  93             status = 404
  94         else:
  95             if self.logger:
  96                 self.logger.exception(six.text_type(e))
  97             status = 500
  98         return Response(context, status=status)
