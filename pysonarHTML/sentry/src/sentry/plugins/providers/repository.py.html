  22 class RepositoryProvider(ProviderMixin):
  23     """
  24     Plugin Repository Provider
  25     Includes all plugins such as those in sentry-plugins repo
  26     as well as any outside plugin respoitories (i.e. Trello, Youtrack).
  27     Does not include the integrations in the sentry repository.
  28     """
  29 
  30     name = None
  31 
  32     def __init__(self, id):
  33         self.id = id
  34 
  35     def dispatch(self, request, organization, **kwargs):
  36         if self.needs_auth(request.user):
  37             # TODO(dcramer): this should be a 401
  38             return Response(
  39                 {
  40                     "error_type": "auth",
  41                     "auth_url": reverse("socialauth_associate", args=[self.auth_provider]),
  42                 },
  43                 status=400,
  44             )
  45 
  46         try:
  47             fields = self.get_config()
  48         except Exception as e:
  49             return self.handle_api_error(e)
  50 
  51         if request.method == "GET":
  52             return Response(fields)
  53 
  54         validator = ConfigValidator(fields, request.data)
  55         if not validator.is_valid():
  56             return Response({"error_type": "validation", "errors": validator.errors}, status=400)
  57 
  58         try:
  59             config = self.validate_config(organization, validator.result, actor=request.user)
  60         except Exception as e:
  61             return self.handle_api_error(e)
  62 
  63         try:
  64             result = self.create_repository(
  65                 organization=organization, data=config, actor=request.user
  66             )
  67         except PluginError as e:
  68             logger.exception("repo.create-error")
  69             return Response({"errors": {"__all__": six.text_type(e)}}, status=400)
  70 
  71         try:
  72             with transaction.atomic():
  73                 repo = Repository.objects.create(
  74                     organization_id=organization.id,
  75                     name=result["name"],
  76                     external_id=result.get("external_id"),
  77                     url=result.get("url"),
  78                     config=result.get("config") or {},
  79                     provider=self.id,
  80                 )
  81         except IntegrityError:
  82             # Try to delete webhook we just created
  83             try:
  84                 repo = Repository(
  85                     organization_id=organization.id,
  86                     name=result["name"],
  87                     external_id=result.get("external_id"),
  88                     url=result.get("url"),
  89                     config=result.get("config") or {},
  90                     provider=self.id,
  91                 )
  92                 self.delete_repository(repo, actor=request.user)
  93             except PluginError:
  94                 pass
  95             return Response(
  96                 {"errors": {"__all__": "A repository with that name already exists"}}, status=400
  97             )
  98         else:
  99             repo_linked.send_robust(repo=repo, user=request.user, sender=self.__class__)
 100 
 101         return Response(serialize(repo, request.user), status=201)
