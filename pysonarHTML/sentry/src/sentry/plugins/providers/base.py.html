  18     def link_auth(self, user, organization, data):
  19         try:
  20             usa = UserSocialAuth.objects.get(
  21                 user=user, id=data["default_auth_id"], provider=self.auth_provider
  22             )
  23         except UserSocialAuth.DoesNotExist:
  24             raise PluginError
  25 
  26         integration = Integration.objects.get_or_create(
  27             provider=self.auth_provider, external_id=usa.uid
  28         )[0]
  29         integration.add_organization(organization, user, default_auth_id=usa.id)

  31     def get_available_auths(self, user, organization, integrations, social_auths, **kwargs):
  32         if self.auth_provider is None:
  33             return []
  34 
  35         social_auths_by_id = {usa.id: usa for usa in social_auths}
  36         linked_social_auths = set()
  37 
  38         auths = []
  39         for i in integrations:
  40             associated_auth = i.default_auth_id and social_auths_by_id[i.default_auth_id]
  41             if associated_auth:
  42                 linked_social_auths.add(associated_auth.id)
  43             auths.append(
  44                 {
  45                     "defaultAuthId": i.default_auth_id,
  46                     "user": associated_auth and {"email": associated_auth.user.email},
  47                     "externalId": i.external_id,
  48                     "integrationId": six.text_type(i.id),
  49                     "linked": True,
  50                 }
  51             )
  52         auths.extend(
  53             [
  54                 {
  55                     "defaultAuthId": sa.id,
  56                     "user": {"email": sa.user.email},
  57                     "externalId": sa.uid,
  58                     "integrationId": None,
  59                     "linked": False,
  60                 }
  61                 for sa in social_auths
  62                 if sa.id not in linked_social_auths
  63             ]
  64         )
  65         return auths

  67     def get_auth_url(self, user, **kwargs):
  68         if self.auth_provider is None:
  69             return
  70 
  71         return reverse("socialauth_associate", args=[self.auth_provider])

  73     def needs_auth(self, user, **kwargs):
  74         """
  75         Return ``True`` if the authenticated user needs to associate an auth
  76         service before performing actions with this provider.
  77         """
  78         if self.auth_provider is None:
  79             return False
  80 
  81         organization = kwargs.get("organization")
  82         if organization:
  83             has_auth = OrganizationIntegration.objects.filter(
  84                 integration__provider=self.auth_provider, organization=organization
  85             ).exists()
  86             if has_auth:
  87                 return False
  88 
  89         if not user.is_authenticated():
  90             return True
  91 
  92         return not UserSocialAuth.objects.filter(user=user, provider=self.auth_provider).exists()
  93 
  94     def get_auth(self, user, **kwargs):
  95         if self.auth_provider is None:
  96             return None
  97 
  98         organization = kwargs.get("organization")
  99         if organization:
 100             try:
 101                 auth = UserSocialAuth.objects.get(
 102                     id=OrganizationIntegration.objects.filter(
 103                         organization=organization, integration__provider=self.auth_provider
 104                     ).values_list("default_auth_id", flat=True)[0]
 105                 )
 106             except UserSocialAuth.DoesNotExist:
 107                 pass
 108             else:
 109                 return auth
 110 
 111         if not user.is_authenticated():
 112             return None
 113 
 114         return UserSocialAuth.objects.filter(user=user, provider=self.auth_provider).first()

 116     def handle_api_error(self, e):
 117         context = {"error_type": "unknown"}
 118         if isinstance(e, InvalidIdentity):
 119             if self.auth_provider is None:
 120                 context.update(
 121                     {
 122                         "message": "Your authentication credentials are invalid. Please check your project settings."
 123                     }
 124                 )
 125             else:
 126                 context.update(
 127                     {
 128                         "error_type": "auth",
 129                         "auth_url": reverse("socialauth_associate", args=[self.auth_provider]),
 130                     }
 131                 )
 132             status = 400
 133         elif isinstance(e, PluginError):
 134             # TODO(dcramer): we should have a proper validation error
 135             context.update({"error_type": "validation", "errors": {"__all__": six.text_type(e)}})
 136             status = 400
 137         else:
 138             if self.logger:
 139                 self.logger.exception(six.text_type(e))
 140             status = 500
 141         return Response(context, status=status)
