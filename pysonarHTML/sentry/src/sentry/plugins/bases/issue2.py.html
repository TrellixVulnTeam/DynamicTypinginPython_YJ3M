  57     def get_group_body(self, request, group, event, **kwargs):
  58         result = []
  59         for interface in six.itervalues(event.interfaces):
  60             output = safe_execute(interface.to_string, event, _with_transaction=False)
  61             if output:
  62                 result.append(output)
  63         return "\n\n".join(result)

  68         body = self.get_group_body(request, group, event)
  69         if body:
  70             output.extend(["", "```", body, "```"])

  91     def get_auth_for_user(self, user, **kwargs):
  92         """
  93         Return a ``UserSocialAuth`` object for the given user based on this plugins ``auth_provider``.
  94         """
  95         assert self.auth_provider, "There is no auth provider configured for this plugin."
  96 
  97         if not user.is_authenticated():
  98             return None
  99 
 100         try:
 101             return UserSocialAuth.objects.filter(user=user, provider=self.auth_provider)[0]
 102         except IndexError:
 103             return None

 105     def needs_auth(self, request, project, **kwargs):
 106         """
 107         Return ``True`` if the authenticated user needs to associate an auth service before
 108         performing actions with this plugin.
 109         """
 110         if self.auth_provider is None:
 111             return False
 112 
 113         if not request.user.is_authenticated():
 114             return True
 115 
 116         return not UserSocialAuth.objects.filter(
 117             user=request.user, provider=self.auth_provider
 118         ).exists()

 192     def validate_form(self, fields, form_data):
 193         errors = {}
 194         for field in fields:
 195             if field.get("required", True) and not field.get("readonly"):
 196                 value = form_data.get(field["name"])
 197                 if value is None or value == "":
 198                     errors[field["name"]] = u"%s is a required field." % field["label"]
 199         return errors

 208     def build_issue(self, group):
 209         issue_field_map = self.get_issue_field_map()

 226     def view_create(self, request, group, **kwargs):
 227         auth_errors = self.check_config_and_auth(request, group)
 228         if auth_errors:
 229             return Response(auth_errors, status=400)
 230 
 231         event = group.get_latest_event()
 232         if event is None:
 233             return Response(
 234                 {
 235                     "message": "Unable to create issues: there are "
 236                     "no events associated with this group"
 237                 },
 238                 status=400,
 239             )
 240         try:
 241             fields = self.get_new_issue_fields(request, group, event, **kwargs)
 242         except Exception as e:
 243             return self.handle_api_error(e)
 244         if request.method == "GET":
 245             return Response(fields)
 246 
 247         errors = self.validate_form(fields, request.data)
 248         if errors:
 249             return Response({"error_type": "validation", "errors": errors}, status=400)
 250 
 251         try:
 252             issue = self.create_issue(group=group, form_data=request.data, request=request)
 253         except Exception as e:
 254             return self.handle_api_error(e)
 255 
 256         if not isinstance(issue, dict):
 257             issue = {"id": issue}
 258 
 259         issue_field_map = self.get_issue_field_map()
 260         for key, meta_name in six.iteritems(issue_field_map):
 261             if key in issue:
 262                 GroupMeta.objects.set_value(group, meta_name, issue[key])
 263             else:
 264                 GroupMeta.objects.unset_value(group, meta_name)
 265 
 266         issue_information = {
 267             "title": issue.get("title")
 268             or request.data.get("title")
 269             or self._get_issue_label_compat(group, issue),
 270             "provider": self.get_title(),
 271             "location": self._get_issue_url_compat(group, issue),
 272             "label": self._get_issue_label_compat(group, issue),
 273         }
 274         Activity.objects.create(
 275             project=group.project,
 276             group=group,
 277             type=Activity.CREATE_ISSUE,
 278             user=request.user,
 279             data=issue_information,
 280         )
 281 
 282         issue_tracker_used.send_robust(
 283             plugin=self, project=group.project, user=request.user, sender=type(self)
 284         )
 285         return Response(
 286             {
 287                 "issue_url": self.get_issue_url(group, issue),
 288                 "link": self._get_issue_url_compat(group, issue),
 289                 "label": self._get_issue_label_compat(group, issue),
 290                 "id": issue["id"],
 291             }
 292         )

 294     def view_link(self, request, group, **kwargs):
 295         auth_errors = self.check_config_and_auth(request, group)
 296         if auth_errors:
 297             return Response(auth_errors, status=400)
 298 
 299         event = group.get_latest_event()
 300         if event is None:
 301             return Response(
 302                 {
 303                     "message": "Unable to create issues: there are "
 304                     "no events associated with this group"
 305                 },
 306                 status=400,
 307             )
 308 
 309         try:
 310             fields = self.get_link_existing_issue_fields(request, group, event, **kwargs)
 311         except Exception as e:
 312             return self.handle_api_error(e)
 313         if request.method == "GET":
 314             return Response(fields)
 315         errors = self.validate_form(fields, request.data)
 316         if errors:
 317             return Response({"error_type": "validation", "errors": errors}, status=400)
 318 
 319         try:
 320             issue = self.link_issue(group=group, form_data=request.data, request=request) or {}
 321         except Exception as e:
 322             return self.handle_api_error(e)
 323 
 324         # HACK(dcramer): maintain data for legacy issues
 325         if "id" not in issue and "issue_id" in request.data:
 326             issue["id"] = request.data["issue_id"]
 327 
 328         issue_field_map = self.get_issue_field_map()
 329         for key, meta_name in six.iteritems(issue_field_map):
 330             if key in issue:
 331                 GroupMeta.objects.set_value(group, meta_name, issue[key])
 332             else:
 333                 GroupMeta.objects.unset_value(group, meta_name)
 334 
 335         issue_information = {
 336             "title": issue.get("title") or self._get_issue_label_compat(group, issue),
 337             "provider": self.get_title(),
 338             "location": self._get_issue_url_compat(group, issue),
 339             "label": self._get_issue_label_compat(group, issue),
 340         }
 341         Activity.objects.create(
 342             project=group.project,
 343             group=group,
 344             type=Activity.CREATE_ISSUE,
 345             user=request.user,
 346             data=issue_information,
 347         )
 348         return Response(
 349             {
 350                 "message": "Successfully linked issue.",
 351                 "link": self._get_issue_url_compat(group, issue),
 352                 "label": self._get_issue_label_compat(group, issue),
 353                 "id": issue["id"],
 354             }
 355         )

 357     def view_unlink(self, request, group, **kwargs):
 358         auth_errors = self.check_config_and_auth(request, group)
 359         if auth_errors:
 360             return Response(auth_errors, status=400)
 361         issue = self.build_issue(group)
 362         if issue and "unlink" in self.allowed_actions:
 363             self.unlink_issue(request, group, issue)
 364             return Response({"message": "Successfully unlinked issue."})
 365         return Response({"message": "No issues to unlink."}, status=400)

 367     def plugin_issues(self, request, group, plugin_issues, **kwargs):
 368         if not self.is_configured(request=request, project=group.project):
 369             return plugin_issues

 371         item = {
 372             "slug": self.slug,
 373             "allowed_actions": self.allowed_actions,

 375         }
 376         issue = self.build_issue(group)
 377         if issue:
 378             item["issue"] = {

 395     def check_config_and_auth(self, request, group):
 396         has_auth_configured = self.has_auth_configured()
 397         if not (has_auth_configured and self.is_configured(project=group.project, request=request)):
 398             if self.auth_provider:
 399                 required_auth_settings = settings.AUTH_PROVIDERS[self.auth_provider]
 400             else:
 401                 required_auth_settings = None
 402 
 403             return {
 404                 "error_type": "config",
 405                 "has_auth_configured": has_auth_configured,
 406                 "auth_provider": self.auth_provider,
 407                 "required_auth_settings": required_auth_settings,
 408             }
 409 
 410         if self.needs_auth(project=group.project, request=request):
 411             return {
 412                 "error_type": "auth",
 413                 "auth_url": reverse("socialauth_associate", args=[self.auth_provider]),
 414             }
