  53     def notify(self, notification, raise_exception=False):
  54         """
  55         This calls the notify_users method of the plugin.
  56         Normally this method eats the error and logs it but if we
  57         set raise_exception=True like we do for the test plugin buttion,
  58         the exception is raised
  59         """
  60         event = notification.event
  61         try:
  62             return self.notify_users(
  63                 event.group, event, triggering_rules=[r.label for r in notification.rules]
  64             )
  65         except (SSLError, HTTPError, ApiError, PluginError, UrllibHTTPError) as err:
  66             self.logger.info(
  67                 "notification-plugin.notify-failed",
  68                 extra={
  69                     "error": six.text_type(err),
  70                     "plugin": self.slug,
  71                     "project_id": event.group.project_id,
  72                     "organization_id": event.group.project.organization_id,
  73                 },
  74             )
  75             if raise_exception:
  76                 raise err
  77             return False

  79     def rule_notify(self, event, futures):
  80         rules = []
  81         extra = {"event_id": event.event_id, "group_id": event.group_id, "plugin": self.slug}
  82         log_event = "dispatched"
  83         for future in futures:
  84             rules.append(future.rule)
  85             extra["rule_id"] = future.rule.id
  86             if not future.kwargs:
  87                 continue
  88             raise NotImplementedError(
  89                 "The default behavior for notification de-duplication does not support args"
  90             )
  91 
  92         project = event.group.project
  93         extra["project_id"] = project.id
  94         notification = Notification(event=event, rules=rules)
  95         self.notify(notification)
  96         self.logger.info("notification.%s" % log_event, extra=extra)

 121     def should_notify(self, group, event):
 122         project = event.project
 123         if not self.is_configured(project=project):
 124             return False
 125 
 126         # If the plugin doesn't support digests or they are not enabled,
 127         # perform rate limit checks to support backwards compatibility with
 128         # older plugins.
 129         if not (
 130             hasattr(self, "notify_digest") and digests.enabled(project)
 131         ) and self.__is_rate_limited(group, event):
 132             logger = logging.getLogger(u"sentry.plugins.{0}".format(self.get_conf_key()))
 133             logger.info("notification.rate_limited", extra={"project_id": project.id})
 134             return False
 135 
 136         return True

 145     def test_configuration_and_get_test_results(self, project):
 146         try:
 147             test_results = self.test_configuration(project)
 148         except Exception as exc:
 149             if isinstance(exc, HTTPError) and hasattr(exc.response, "text"):
 150                 test_results = "%s\n%s" % (exc, exc.response.text[:256])
 151             elif hasattr(exc, "read") and callable(exc.read):
 152                 test_results = "%s\n%s" % (exc, exc.read()[:256])
 153             else:
 154                 logging.exception(
 155                     "Plugin(%s) raised an error during test, %s", self.slug, six.text_type(exc)
 156                 )
 157                 if six.text_type(exc).lower().startswith("error communicating with"):
 158                     test_results = six.text_type(exc)[:256]
 159                 else:
 160                     test_results = (
 161                         "There was an internal error with the Plugin, %s" % six.text_type(exc)[:256]
 162                     )
 163         if not test_results:
 164             test_results = "No errors returned"
 165         return test_results

 170     def add_notification_referrer_param(self, url):
 171         if self.slug:
 172             parsed_url = urlparse(url)
 173             query = parse_qs(parsed_url.query)
 174             query["referrer"] = self.slug
 175 
 176             url_list = list(parsed_url)
 177             url_list[4] = urlencode(query, doseq=True)
 178             return urlunparse(url_list)
 179 
 180         return url
