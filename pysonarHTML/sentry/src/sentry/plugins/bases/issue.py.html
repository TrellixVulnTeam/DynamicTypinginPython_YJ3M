  38     def _get_group_body(self, request, group, event, **kwargs):
  39         result = []
  40         for interface in six.itervalues(event.interfaces):
  41             output = safe_execute(interface.to_string, event, _with_transaction=False)
  42             if output:
  43                 result.append(output)
  44         return "\n\n".join(result)

  46     def _get_group_description(self, request, group, event):
  47         referrer = self.get_conf_key() + "_plugin"
  48         output = [absolute_uri(group.get_absolute_url(params={"referrer": referrer}))]
  49         body = self._get_group_body(request, group, event)
  50         if body:
  51             output.extend(["", "```", body, "```"])
  52         return "\n".join(output)

  60     def get_auth_for_user(self, user, **kwargs):
  61         """
  62         Return a ``UserSocialAuth`` object for the given user based on this plugins ``auth_provider``.
  63         """
  64         assert self.auth_provider, "There is no auth provider configured for this plugin."
  65 
  66         if not user.is_authenticated():
  67             return None
  68 
  69         try:
  70             return UserSocialAuth.objects.filter(user=user, provider=self.auth_provider)[0]
  71         except IndexError:
  72             return None
  73 
  74     def needs_auth(self, request, project, **kwargs):
  75         """
  76         Return ``True`` if the authenticated user needs to associate an auth service before
  77         performing actions with this plugin.
  78         """
  79         if self.auth_provider is None:
  80             return False
  81 
  82         if not request.user.is_authenticated():
  83             return True
  84 
  85         return bool(
  86             not UserSocialAuth.objects.filter(
  87                 user=request.user, provider=self.auth_provider
  88             ).exists()
  89         )

 178     def view(self, request, group, **kwargs):
 179         has_auth_configured = self.has_auth_configured()
 180         if not (has_auth_configured and self.is_configured(project=group.project, request=request)):
 181             if self.auth_provider:
 182                 required_auth_settings = settings.AUTH_PROVIDERS[self.auth_provider]
 183             else:
 184                 required_auth_settings = None
 185 
 186             project = group.project
 187 
 188             return self.render(
 189                 self.not_configured_template,
 190                 {
 191                     "title": self.get_title(),
 192                     "project": group.project,
 193                     "has_auth_configured": has_auth_configured,
 194                     "required_auth_settings": required_auth_settings,
 195                     "plugin_link": u"/settings/{}/projects/{}/plugins/{}/".format(
 196                         project.organization.slug, project.slug, self.slug
 197                     ),
 198                 },
 199             )
 200 
 201         if self.needs_auth(project=group.project, request=request):
 202             return self.render(
 203                 self.needs_auth_template, {"title": self.get_title(), "project": group.project}
 204             )
 205 
 206         if GroupMeta.objects.get_value(group, "%s:tid" % self.get_conf_key(), None):
 207             if self.can_unlink_issues and request.GET.get("unlink"):
 208                 return self.handle_unlink_issue(request, group, **kwargs)
 209             return None
 210 
 211         prefix = self.get_conf_key()
 212         event = group.get_latest_event()
 213 
 214         op = request.POST.get("op", "create")
 215 
 216         create_form = self.get_new_issue_form(request, group, event)
 217         link_form = None
 218         if self.can_link_existing_issues:
 219             link_form = self.get_link_existing_issue_form(request, group, event)
 220 
 221         if op == "create":
 222             if create_form.is_valid():
 223                 try:
 224                     issue_id = self.create_issue(
 225                         group=group, form_data=create_form.cleaned_data, request=request
 226                     )
 227                 except forms.ValidationError as e:
 228                     create_form.errors["__all__"] = [u"Error creating issue: %s" % e]
 229 
 230             if create_form.is_valid():
 231                 GroupMeta.objects.set_value(group, "%s:tid" % prefix, issue_id)
 232 
 233                 issue_information = {
 234                     "title": create_form.cleaned_data["title"],
 235                     "provider": self.get_title(),
 236                     "location": self.get_issue_url(group, issue_id),
 237                     "label": self.get_issue_label(group=group, issue_id=issue_id),
 238                 }
 239                 Activity.objects.create(
 240                     project=group.project,
 241                     group=group,
 242                     type=Activity.CREATE_ISSUE,
 243                     user=request.user,
 244                     data=issue_information,
 245                 )
 246 
 247                 issue_tracker_used.send_robust(
 248                     plugin=self,
 249                     project=group.project,
 250                     user=request.user,
 251                     sender=IssueTrackingPlugin,
 252                 )
 253                 return self.redirect(group.get_absolute_url())
 254 
 255         elif op == "link":
 256             if link_form.is_valid():
 257                 try:
 258                     self.link_issue(group=group, form_data=link_form.cleaned_data, request=request)
 259                 except forms.ValidationError as e:
 260                     link_form.errors["__all__"] = [u"Error creating issue: %s" % e]
 261 
 262             if link_form.is_valid():
 263                 issue_id = int(link_form.cleaned_data["issue_id"])
 264                 GroupMeta.objects.set_value(group, "%s:tid" % prefix, issue_id)
 265                 issue_information = {
 266                     "title": self.get_issue_title_by_id(request, group, issue_id),
 267                     "provider": self.get_title(),
 268                     "location": self.get_issue_url(group, issue_id),
 269                     "label": self.get_issue_label(group=group, issue_id=issue_id),
 270                 }
 271                 Activity.objects.create(
 272                     project=group.project,
 273                     group=group,
 274                     type=Activity.CREATE_ISSUE,
 275                     user=request.user,
 276                     data=issue_information,
 277                 )
 278 
 279                 return self.redirect(group.get_absolute_url())
 280 
 281         context = {
 282             "create_form": create_form,
 283             # pass in 'form' for legacy compat
 284             "form": create_form,
 285             "title": self.get_new_issue_title(),
 286             "read_only_fields": self.get_new_issue_read_only_fields(group=group),
 287             "can_link_existing_issues": self.can_link_existing_issues,
 288             "link_form": link_form,
 289             "op": op,
 290         }
 291 
 292         return self.render(self.create_issue_template, context)

 294     def actions(self, request, group, action_list, **kwargs):
 295         if not self.is_configured(request=request, project=group.project):
 296             return action_list
 297         prefix = self.get_conf_key()
 298         if not GroupMeta.objects.get_value(group, "%s:tid" % prefix, None):
 299             action_list.append((self.get_new_issue_title(), self.get_url(group)))
 300         elif self.can_unlink_issues:
 301             action_list.append(
 302                 (self.get_unlink_issue_title(), "%s?unlink=1" % self.get_url(group).rstrip("/"))
 303             )
 304         return action_list
 305 
 306     def tags(self, request, group, tag_list, **kwargs):
 307         if not self.is_configured(request=request, project=group.project):
 308             return tag_list
 309 
 310         prefix = self.get_conf_key()
 311         issue_id = GroupMeta.objects.get_value(group, "%s:tid" % prefix)
 312         if not issue_id:
 313             return tag_list
 314 
 315         tag_list.append(
 316             format_html(
 317                 '<a href="{}" rel="noreferrer">{}</a>',
 318                 self.get_issue_url(group=group, issue_id=issue_id),
 319                 self.get_issue_label(group=group, issue_id=issue_id),
 320             )
 321         )
 322 
 323         return tag_list
