  61 class RedisBackedState(object):
  62     # Expire the pipeline after 10 minutes of inactivity.
  63     EXPIRATION_TTL = 10 * 60
  64 
  65     def __init__(self, request):
  66         self.__dict__["request"] = request
  67 
  68     @property
  69     def _client(self):
  70         return clusters.get("default").get_local_client_for_key(self.auth_key)
  71 
  72     @property
  73     def auth_key(self):
  74         return self.request.session.get("auth_key")
  75 
  76     def regenerate(self, initial_state):
  77         auth_key = u"auth:pipeline:{}".format(uuid4().hex)
  78 
  79         self.request.session["auth_key"] = auth_key
  80         self.request.session.modified = True
  81 
  82         value = json.dumps(initial_state)
  83         self._client.setex(auth_key, self.EXPIRATION_TTL, value)
  84 
  85     def clear(self):
  86         if not self.auth_key:
  87             return
  88 
  89         self._client.delete(self.auth_key)
  90         del self.request.session["auth_key"]
  91         self.request.session.modified = True
  92 
  93     def is_valid(self):
  94         return self.auth_key and self._client.get(self.auth_key)
  95 
  96     def get_state(self):
  97         if not self.auth_key:
  98             return None
  99 
 100         state_json = self._client.get(self.auth_key)
 101         if not state_json:
 102             return None
 103 
 104         return json.loads(state_json)
 105 
 106     def __getattr__(self, key):
 107         state = self.get_state()
 108         return state[key] if state else None
 109 
 110     def __setattr__(self, key, value):
 111         state = self.get_state()
 112         if not state:
 113             return
 114 
 115         state[key] = value
 116         self._client.setex(self.auth_key, self.EXPIRATION_TTL, json.dumps(state))

 119 def handle_existing_identity(
 120     auth_provider, provider, organization, request, state, auth_identity, identity
 121 ):
 122     # TODO(dcramer): this is very similar to attach
 123     now = timezone.now()
 124     auth_identity.update(
 125         data=provider.update_identity(
 126             new_data=identity.get("data", {}), current_data=auth_identity.data
 127         ),
 128         last_verified=now,
 129         last_synced=now,
 130     )
 131 
 132     try:
 133         member = OrganizationMember.objects.get(user=auth_identity.user, organization=organization)
 134     except OrganizationMember.DoesNotExist:
 135         # this is likely the case when someone was removed from the org
 136         # but still has access to rejoin
 137         member = handle_new_membership(auth_provider, organization, request, auth_identity)
 138     else:
 139         if getattr(member.flags, "sso:invalid") or not getattr(member.flags, "sso:linked"):
 140             setattr(member.flags, "sso:invalid", False)
 141             setattr(member.flags, "sso:linked", True)
 142             member.save()
 143 
 144     user = auth_identity.user
 145     user.backend = settings.AUTHENTICATION_BACKENDS[0]
 146 
 147     if not auth.login(
 148         request, user, after_2fa=request.build_absolute_uri(), organization_id=organization.id
 149     ):
 150         return HttpResponseRedirect(auth.get_login_redirect(request))
 151 
 152     state.clear()
 153     metrics.incr(
 154         "sso.login-success", tags={"provider": provider.key}, skip_internal=False, sample_rate=1.0
 155     )
 156 
 157     return HttpResponseRedirect(auth.get_login_redirect(request))

 160 def handle_new_membership(auth_provider, organization, request, auth_identity):
 161     user = auth_identity.user
 162 
 163     # If the user is either currently *pending* invite acceptance (as indicated
 164     # from the pending-invite cookie) OR an existing invite exists on this
 165     # organziation for the email provided by the identity provider.
 166     invite_helper = ApiInviteHelper.from_cookie_or_email(
 167         request=request, organization=organization, email=user.email
 168     )
 169 
 170     # If we are able to accept an existing invite for the user for this
 171     # organization, do so, otherwise handle new membership
 172     if invite_helper:
 173         if invite_helper.invite_approved:
 174             invite_helper.accept_invite(user)
 175             return
 176 
 177         # It's possible the user has an _invite request_ that hasn't been approved yet,
 178         # and is able to join the organization without an invite through the SSO flow.
 179         # In that case, delete the invite request and create a new membership.
 180         invite_helper.handle_invite_not_approved()
 181 
 182     # Otherwise create a new membership
 183     om = OrganizationMember.objects.create(
 184         organization=organization,
 185         role=organization.default_role,
 186         user=user,
 187         flags=OrganizationMember.flags["sso:linked"],
 188     )
 189 
 190     default_teams = auth_provider.default_teams.all()
 191     for team in default_teams:
 192         OrganizationMemberTeam.objects.create(team=team, organizationmember=om)
 193 
 194     AuditLogEntry.objects.create(
 195         organization=organization,
 196         actor=user,
 197         ip_address=request.META["REMOTE_ADDR"],
 198         target_object=om.id,
 199         target_user=om.user,
 200         event=AuditLogEntryEvent.MEMBER_ADD,
 201         data=om.get_audit_log_data(),
 202     )
 203 
 204     return om

 207 @transaction.atomic
 208 def handle_attach_identity(auth_provider, request, organization, provider, identity, member=None):
 209     """
 210     Given an already authenticated user, attach or re-attach an identity.
 211     """
 212     user = request.user
 213 
 214     try:
 215         try:
 216             # prioritize identifying by the SSO provider's user ID
 217             auth_identity = AuthIdentity.objects.get(
 218                 auth_provider=auth_provider, ident=identity["id"]
 219             )
 220         except AuthIdentity.DoesNotExist:
 221             # otherwise look for an already attached identity
 222             # this can happen if the SSO provider's internal ID changes
 223             auth_identity = AuthIdentity.objects.get(auth_provider=auth_provider, user=user)
 224     except AuthIdentity.DoesNotExist:
 225         auth_identity = AuthIdentity.objects.create(
 226             auth_provider=auth_provider,
 227             user=user,
 228             ident=identity["id"],
 229             data=identity.get("data", {}),
 230         )
 231         auth_is_new = True
 232     else:
 233         now = timezone.now()
 234 
 235         # TODO(dcramer): this might leave the user with duplicate accounts,
 236         # and in that kind of situation its very reasonable that we could
 237         # test email addresses + is_managed to determine if we can auto
 238         # merge
 239         if auth_identity.user != user:
 240             # it's possible the user has an existing identity, let's wipe it out
 241             # so that the new identifier gets used (other we'll hit a constraint)
 242             # violation since one might exist for (provider, user) as well as
 243             # (provider, ident)
 244             AuthIdentity.objects.exclude(id=auth_identity.id).filter(
 245                 auth_provider=auth_provider, user=user
 246             ).delete()
 247 
 248             # since we've identify an identity which is no longer valid
 249             # lets preemptively mark it as such
 250             try:
 251                 other_member = OrganizationMember.objects.get(
 252                     user=auth_identity.user_id, organization=organization
 253                 )
 254             except OrganizationMember.DoesNotExist:
 255                 pass
 256             else:
 257                 other_member.flags["sso:invalid"] = True
 258                 other_member.flags["sso:linked"] = False
 259                 other_member.save()
 260 
 261         auth_identity.update(
 262             user=user,
 263             ident=identity["id"],
 264             data=provider.update_identity(
 265                 new_data=identity.get("data", {}), current_data=auth_identity.data
 266             ),
 267             last_verified=now,
 268             last_synced=now,
 269         )
 270         auth_is_new = False
 271 
 272     if member is None:
 273         try:
 274             member = OrganizationMember.objects.get(user=user, organization=organization)
 275         except OrganizationMember.DoesNotExist:
 276             member = OrganizationMember.objects.create(
 277                 organization=organization,
 278                 role=organization.default_role,
 279                 user=user,
 280                 flags=OrganizationMember.flags["sso:linked"],
 281             )
 282 
 283             default_teams = auth_provider.default_teams.all()
 284             for team in default_teams:
 285                 OrganizationMemberTeam.objects.create(team=team, organizationmember=member)
 286 
 287             AuditLogEntry.objects.create(
 288                 organization=organization,
 289                 actor=user,
 290                 ip_address=request.META["REMOTE_ADDR"],
 291                 target_object=member.id,
 292                 target_user=user,
 293                 event=AuditLogEntryEvent.MEMBER_ADD,
 294                 data=member.get_audit_log_data(),
 295             )
 296     if getattr(member.flags, "sso:invalid") or not getattr(member.flags, "sso:linked"):
 297         setattr(member.flags, "sso:invalid", False)
 298         setattr(member.flags, "sso:linked", True)
 299         member.save()
 300 
 301     if auth_is_new:
 302         AuditLogEntry.objects.create(
 303             organization=organization,
 304             actor=user,
 305             ip_address=request.META["REMOTE_ADDR"],
 306             target_object=auth_identity.id,
 307             event=AuditLogEntryEvent.SSO_IDENTITY_LINK,
 308             data=auth_identity.get_audit_log_data(),
 309         )
 310 
 311         messages.add_message(request, messages.SUCCESS, OK_LINK_IDENTITY)
 312 
 313     return auth_identity

 342 def handle_unknown_identity(request, organization, auth_provider, provider, state, identity):
 343     """
 344     Flow is activated upon a user logging in to where an AuthIdentity is
 345     not present.
 346 
 347     XXX(dcramer): this docstring is out of date
 348 
 349     The flow will attempt to answer the following:
 350 
 351     - Is there an existing user with the same email address? Should they be
 352       merged?
 353 
 354     - Is there an existing user (via authentication) that should be merged?
 355 
 356     - Should I create a new user based on this identity?
 357     """
 358     op = request.POST.get("op")
 359     if not request.user.is_authenticated():
 360         # TODO(dcramer): its possible they have multiple accounts and at
 361         # least one is managed (per the check below)
 362         try:
 363             acting_user = User.objects.filter(
 364                 id__in=UserEmail.objects.filter(email__iexact=identity["email"]).values("user"),
 365                 is_active=True,
 366             ).first()
 367         except IndexError:
 368             acting_user = None
 369         login_form = AuthenticationForm(
 370             request,
 371             request.POST if request.POST.get("op") == "login" else None,
 372             initial={"username": acting_user.username if acting_user else None},
 373         )
 374     else:
 375         acting_user = request.user
 376 
 377     # If they already have an SSO account and the identity provider says
 378     # the email matches we go ahead and let them merge it. This is the
 379     # only way to prevent them having duplicate accounts, and because
 380     # we trust identity providers, its considered safe.
 381     # Note: we do not trust things like SAML, so the SSO implementation needs
 382     # to consider if 'email_verified' can be trusted or not
 383     if acting_user and identity.get("email_verified"):
 384         # we only allow this flow to happen if the existing user has
 385         # membership, otherwise we short circuit because it might be
 386         # an attempt to hijack membership of another organization
 387         has_membership = OrganizationMember.objects.filter(
 388             user=acting_user, organization=organization
 389         ).exists()
 390         if has_membership:
 391             if not auth.login(
 392                 request,
 393                 acting_user,
 394                 after_2fa=request.build_absolute_uri(),
 395                 organization_id=organization.id,
 396             ):
 397                 if acting_user.has_usable_password():
 398                     return post_login_redirect(request)
 399                 else:
 400                     acting_user = None
 401             else:
 402                 # assume they've confirmed they want to attach the identity
 403                 op = "confirm"
 404         else:
 405             # force them to create a new account
 406             acting_user = None
 407     # without a usable password they cant login, so let's clear the acting_user
 408     elif acting_user and not acting_user.has_usable_password():
 409         acting_user = None
 410 
 411     if op == "confirm" and request.user.is_authenticated():
 412         auth_identity = handle_attach_identity(
 413             auth_provider, request, organization, provider, identity
 414         )
 415     elif op == "newuser":
 416         auth_identity = handle_new_user(auth_provider, organization, request, identity)
 417     elif op == "login" and not request.user.is_authenticated():
 418         # confirm authentication, login
 419         op = None
 420         if login_form.is_valid():
 421             # This flow is special.  If we are going through a 2FA
 422             # flow here (login returns False) we want to instruct the
 423             # system to return upon completion of the 2fa flow to the
 424             # current URL and continue with the dialog.
 425             #
 426             # If there is no 2fa we don't need to do this and can just
 427             # go on.
 428             if not auth.login(
 429                 request,
 430                 login_form.get_user(),
 431                 after_2fa=request.build_absolute_uri(),
 432                 organization_id=organization.id,
 433             ):
 434                 return post_login_redirect(request)
 435         else:
 436             auth.log_auth_failure(request, request.POST.get("username"))
 437     else:
 438         op = None
 439 
 440     if not op:
 441         if request.user.is_authenticated():
 442             return respond(
 443                 "sentry/auth-confirm-link.html",
 444                 organization,
 445                 request,
 446                 {
 447                     "identity": identity,
 448                     "existing_user": request.user,
 449                     "identity_display_name": get_display_name(identity),
 450                     "identity_identifier": get_identifier(identity),
 451                 },
 452             )
 453 
 454         return respond(
 455             "sentry/auth-confirm-identity.html",
 456             organization,
 457             request,
 458             {
 459                 "existing_user": acting_user,
 460                 "identity": identity,
 461                 "login_form": login_form,
 462                 "identity_display_name": get_display_name(identity),
 463                 "identity_identifier": get_identifier(identity),
 464             },
 465         )
 466 
 467     user = auth_identity.user
 468     user.backend = settings.AUTHENTICATION_BACKENDS[0]
 469 
 470     # XXX(dcramer): this is repeated from above
 471     if not auth.login(
 472         request, user, after_2fa=request.build_absolute_uri(), organization_id=organization.id
 473     ):
 474         return post_login_redirect(request)
 475 
 476     state.clear()
 477 
 478     return post_login_redirect(request)

 481 def handle_new_user(auth_provider, organization, request, identity):
 482     user = User.objects.create(
 483         username=uuid4().hex, email=identity["email"], name=identity.get("name", "")[:200]
 484     )
 485 
 486     if settings.TERMS_URL and settings.PRIVACY_URL:
 487         user.update(flags=F("flags").bitor(User.flags.newsletter_consent_prompt))
 488 
 489     try:
 490         with transaction.atomic():
 491             auth_identity = AuthIdentity.objects.create(
 492                 auth_provider=auth_provider,
 493                 user=user,
 494                 ident=identity["id"],
 495                 data=identity.get("data", {}),
 496             )
 497     except IntegrityError:
 498         auth_identity = AuthIdentity.objects.get(auth_provider=auth_provider, ident=identity["id"])
 499         auth_identity.update(user=user, data=identity.get("data", {}))
 500 
 501     user.send_confirm_emails(is_new_user=True)
 502     provider = auth_provider.provider if auth_provider else None
 503     user_signup.send_robust(
 504         sender=handle_new_user, user=user, source="sso", provider=provider, referrer="in-app"
 505     )
 506 
 507     handle_new_membership(auth_provider, organization, request, auth_identity)
 508 
 509     return auth_identity

 512 class AuthHelper(object):
 513     """
 514     Helper class which is passed into AuthView's.
 515 
 516     Designed to link provider and views as well as manage the state and
 517     pipeline. The pipeline is optional, in case can be done
 518 
 519     Auth has several flows:
 520 
 521     1. The user is going through provider setup, thus enforcing that they link
 522        their current account to the new auth identity.
 523     2. The user is anonymous and creating a brand new account.
 524     3. The user is anonymous and logging into an existing account.
 525     4. The user is anonymous and creating a brand new account, but may have an
 526        existing account that should/could be merged.
 527     5. The user is authenticated and creating a new identity, thus associating
 528        it with their current account.
 529     6. The user is authenticated and creating a new identity, but not linking
 530        it with their account (thus creating a new account).
 531     """
 532 
 533     # logging in or registering
 534     FLOW_LOGIN = 1
 535     # configuring the provider
 536     FLOW_SETUP_PROVIDER = 2
 537 
 538     @classmethod
 539     def get_for_request(cls, request):
 540         state = RedisBackedState(request)
 541         if not state.is_valid():
 542             return None
 543 
 544         organization_id = state.org_id
 545         if not organization_id:
 546             logging.info("Invalid SSO data found")
 547             return None
 548 
 549         flow = state.flow
 550         auth_provider_id = state.auth_provider
 551         provider_key = state.provider_key
 552         if auth_provider_id:
 553             auth_provider = AuthProvider.objects.get(id=auth_provider_id)
 554         elif provider_key:
 555             auth_provider = None
 556 
 557         organization = Organization.objects.get(id=state.org_id)
 558 
 559         return cls(
 560             request, organization, flow, auth_provider=auth_provider, provider_key=provider_key
 561         )
 562 
 563     def __init__(self, request, organization, flow, auth_provider=None, provider_key=None):
 564         assert provider_key or auth_provider
 565 
 566         self.request = request
 567         self.auth_provider = auth_provider
 568         self.organization = organization
 569         self.flow = flow
 570         self.state = RedisBackedState(request)
 571 
 572         if auth_provider:
 573             provider = auth_provider.get_provider()
 574         elif provider_key:
 575             provider = manager.get(provider_key)
 576         else:
 577             raise NotImplementedError
 578 
 579         self.provider = provider
 580         if flow == self.FLOW_LOGIN:
 581             self.pipeline = provider.get_auth_pipeline()
 582         elif flow == self.FLOW_SETUP_PROVIDER:
 583             self.pipeline = provider.get_setup_pipeline()
 584         else:
 585             raise NotImplementedError
 586 
 587         # we serialize the pipeline to be [AuthView().get_ident(), ...] which
 588         # allows us to determine if the pipeline has changed during the auth
 589         # flow or if the user is somehow circumventing a chunk of it
 590         self.signature = md5_text(" ".join(av.get_ident() for av in self.pipeline)).hexdigest()
 591 
 592     def pipeline_is_valid(self):
 593         if not self.state.is_valid():
 594             return False
 595         if self.state.flow not in (self.FLOW_LOGIN, self.FLOW_SETUP_PROVIDER):
 596             return False
 597         return self.state.signature == self.signature
 598 
 599     def init_pipeline(self):
 600         self.state.regenerate(
 601             {
 602                 "uid": self.request.user.id if self.request.user.is_authenticated() else None,
 603                 "auth_provider": self.auth_provider.id if self.auth_provider else None,
 604                 "provider_key": self.provider.key,
 605                 "org_id": self.organization.id,
 606                 "step_index": 0,
 607                 "signature": self.signature,
 608                 "flow": self.flow,
 609                 "data": {},
 610             }
 611         )
 612 
 613     def get_redirect_url(self):
 614         return absolute_uri(reverse("sentry-auth-sso"))
 615 
 616     def clear_session(self):
 617         self.state.clear()
 618 
 619     def current_step(self):
 620         """
 621         Render the current step.
 622         """
 623         step_index = self.state.step_index
 624 
 625         if step_index == len(self.pipeline):
 626             return self.finish_pipeline()
 627 
 628         return self.pipeline[step_index].dispatch(request=self.request, helper=self)
 629 
 630     def next_step(self):
 631         """
 632         Render the next step.
 633         """
 634         self.state.step_index += 1
 635         return self.current_step()
 636 
 637     def finish_pipeline(self):
 638         data = self.fetch_state()
 639 
 640         # The state data may have expried, in which case the state data will
 641         # simply be None.
 642         if not data:
 643             return self.error(ERR_INVALID_IDENTITY)
 644 
 645         try:
 646             identity = self.provider.build_identity(data)
 647         except IdentityNotValid as error:
 648             return self.error(six.text_type(error) or ERR_INVALID_IDENTITY)
 649 
 650         if self.state.flow == self.FLOW_LOGIN:
 651             # create identity and authenticate the user
 652             response = self._finish_login_pipeline(identity)
 653         elif self.state.flow == self.FLOW_SETUP_PROVIDER:
 654             response = self._finish_setup_pipeline(identity)
 655 
 656         return response
 657 
 658     @transaction.atomic
 659     def _finish_login_pipeline(self, identity):
 660         """
 661         The login flow executes both with anonymous and authenticated users.
 662 
 663         Upon completion a few branches exist:
 664 
 665         If the identity is already linked, the user should be logged in
 666         and redirected immediately.
 667 
 668         Otherwise, the user is presented with a confirmation window. That window
 669         will show them the new account that will be created, and if they're
 670         already authenticated an optional button to associate the identity with
 671         their account.
 672         """
 673         auth_provider = self.auth_provider
 674         user_id = identity["id"]
 675 
 676         lock = locks.get(
 677             u"sso:auth:{}:{}".format(auth_provider.id, md5_text(user_id).hexdigest()), duration=5
 678         )
 679         with TimedRetryPolicy(5)(lock.acquire):
 680             try:
 681                 auth_identity = AuthIdentity.objects.select_related("user").get(
 682                     auth_provider=auth_provider, ident=user_id
 683                 )
 684             except AuthIdentity.DoesNotExist:
 685                 auth_identity = None
 686 
 687             # Handle migration of identity keys
 688             if not auth_identity and isinstance(user_id, MigratingIdentityId):
 689                 try:
 690                     auth_identity = AuthIdentity.objects.select_related("user").get(
 691                         auth_provider=auth_provider, ident=user_id.legacy_id
 692                     )
 693                     auth_identity.update(ident=user_id.id)
 694                 except AuthIdentity.DoesNotExist:
 695                     auth_identity = None
 696 
 697             if not auth_identity:
 698                 return handle_unknown_identity(
 699                     self.request,
 700                     self.organization,
 701                     self.auth_provider,
 702                     self.provider,
 703                     self.state,
 704                     identity,
 705                 )
 706 
 707             # If the User attached to this AuthIdentity is not active,
 708             # we want to clobber the old account and take it over, rather than
 709             # getting logged into the inactive account.
 710             if not auth_identity.user.is_active:
 711                 # Current user is also not logged in, so we have to
 712                 # assume unknown.
 713                 if not self.request.user.is_authenticated():
 714                     return handle_unknown_identity(
 715                         self.request,
 716                         self.organization,
 717                         self.auth_provider,
 718                         self.provider,
 719                         self.state,
 720                         identity,
 721                     )
 722                 auth_identity = handle_attach_identity(
 723                     self.auth_provider, self.request, self.organization, self.provider, identity
 724                 )
 725 
 726             return handle_existing_identity(
 727                 self.auth_provider,
 728                 self.provider,
 729                 self.organization,
 730                 self.request,
 731                 self.state,
 732                 auth_identity,
 733                 identity,
 734             )
 735 
 736     @transaction.atomic
 737     def _finish_setup_pipeline(self, identity):
 738         """
 739         The setup flow creates the auth provider as well as an identity linked
 740         to the active user.
 741         """
 742         request = self.request
 743         if not request.user.is_authenticated():
 744             return self.error(ERR_NOT_AUTHED)
 745 
 746         if request.user.id != self.state.uid:
 747             return self.error(ERR_UID_MISMATCH)
 748 
 749         data = self.fetch_state()
 750         config = self.provider.build_config(data)
 751 
 752         try:
 753             om = OrganizationMember.objects.get(user=request.user, organization=self.organization)
 754         except OrganizationMember.DoesNotExist:
 755             return self.error(ERR_UID_MISMATCH)
 756 
 757         # disable require 2FA for the organization
 758         # since only SSO or require 2FA can be enabled
 759         self.disable_2fa_required()
 760 
 761         self.auth_provider = AuthProvider.objects.create(
 762             organization=self.organization, provider=self.provider.key, config=config
 763         )
 764 
 765         handle_attach_identity(
 766             self.auth_provider, self.request, self.organization, self.provider, identity, om
 767         )
 768 
 769         auth.mark_sso_complete(request, self.organization.id)
 770 
 771         sso_enabled.send_robust(
 772             organization=self.organization,
 773             user=request.user,
 774             provider=self.provider.key,
 775             sender=self.__class__,
 776         )
 777 
 778         AuditLogEntry.objects.create(
 779             organization=self.organization,
 780             actor=request.user,
 781             ip_address=request.META["REMOTE_ADDR"],
 782             target_object=self.auth_provider.id,
 783             event=AuditLogEntryEvent.SSO_ENABLE,
 784             data=self.auth_provider.get_audit_log_data(),
 785         )
 786 
 787         email_missing_links.delay(self.organization.id, request.user.id, self.provider.key)
 788 
 789         messages.add_message(self.request, messages.SUCCESS, OK_SETUP_SSO)
 790 
 791         self.clear_session()
 792 
 793         next_uri = reverse(
 794             "sentry-organization-auth-provider-settings", args=[self.organization.slug]
 795         )
 796         return HttpResponseRedirect(next_uri)
 797 
 798     def error(self, message):
 799         redirect_uri = "/"
 800 
 801         if self.state.flow == self.FLOW_LOGIN:
 802             # create identity and authenticate the user
 803             redirect_uri = reverse("sentry-auth-organization", args=[self.organization.slug])
 804 
 805         elif self.state.flow == self.FLOW_SETUP_PROVIDER:
 806             redirect_uri = reverse(
 807                 "sentry-organization-auth-settings", args=[self.organization.slug]
 808             )
 809 
 810         metrics.incr(
 811             "sso.error",
 812             tags={"provider": self.provider.key, "flow": self.state.flow},
 813             skip_internal=False,
 814             sample_rate=1.0,
 815         )
 816 
 817         messages.add_message(
 818             self.request, messages.ERROR, u"Authentication error: {}".format(message)
 819         )
 820 
 821         return HttpResponseRedirect(redirect_uri)

 823     def bind_state(self, key, value):
 824         data = self.state.data or {}
 825         data[key] = value
 826 
 827         self.state.data = data
 828 
 829     def fetch_state(self, key=None):
 830         return self.state.data if key is None else self.state.data.get(key)

 832     def disable_2fa_required(self):
 833         require_2fa = self.organization.flags.require_2fa
 834 
 835         if not require_2fa or not require_2fa.is_set:
 836             return
 837 
 838         self.organization.update(flags=F("flags").bitand(~Organization.flags.require_2fa))
 839 
 840         logger.info(
 841             "Require 2fa disabled during sso setup", extra={"organization_id": self.organization.id}
 842         )
 843         create_audit_entry(
 844             request=self.request,
 845             organization=self.organization,
 846             target_object=self.organization.id,
 847             event=AuditLogEntryEvent.ORG_EDIT,
 848             data={"require_2fa": u"to False when enabling SSO"},
 849         )
