   1 """
   2 Superuser in Sentry works differently than the native Django implementation.
   3 
   4 In Sentry a user must achieve the following to be treated as a superuser:
   5 
   6 - ``User.is_superuser`` must be True
   7 - If configured, the user must be accessing Sentry from a privileged IP (``SUPERUSER_ALLOWED_IPS``)
   8 - The user must have a valid 'superuser session', which is a secondary session on top of their
   9   standard auth. This session has a shorter lifespan.
  10 """
  11

  56 def is_active_superuser(request):
  57     if is_system_auth(getattr(request, "auth", None)):
  58         return True
  59     su = getattr(request, "superuser", None) or Superuser(request)
  60     return su.is_active

  63 class Superuser(object):
  64     allowed_ips = [ipaddress.ip_network(six.text_type(v), strict=False) for v in ALLOWED_IPS]
  65 
  66     org_id = ORG_ID
  67 
  68     def __init__(self, request, allowed_ips=UNSET, org_id=UNSET, current_datetime=None):
  69         self.request = request
  70         if allowed_ips is not UNSET:
  71             self.allowed_ips = frozenset(
  72                 ipaddress.ip_network(six.text_type(v), strict=False) for v in allowed_ips or ()
  73             )
  74         if org_id is not UNSET:
  75             self.org_id = org_id
  76         self._populate(current_datetime=current_datetime)
  77 
  78     @property
  79     def is_active(self):
  80         # if we've been logged out
  81         if not self.request.user.is_authenticated():
  82             return False
  83         # if superuser status was changed
  84         if not self.request.user.is_superuser:
  85             return False
  86         # if the user has changed
  87         if six.text_type(self.request.user.id) != self.uid:
  88             return False
  89         return self._is_active
  90 
  91     def is_privileged_request(self):
  92         """
  93         Returns ``(bool is_privileged, str reason)``
  94         """
  95         allowed_ips = self.allowed_ips
  96         # if we've bound superuser to an organization they must
  97         # have completed SSO to gain status
  98         if self.org_id and not has_completed_sso(self.request, self.org_id):
  99             return False, "incomplete-sso"
 100         # if there's no IPs configured, we allow assume its the same as *
 101         if not allowed_ips:
 102             return True, None
 103         ip = ipaddress.ip_address(six.text_type(self.request.META["REMOTE_ADDR"]))
 104         if not any(ip in addr for addr in allowed_ips):
 105             return False, "invalid-ip"
 106         return True, None

 108     def get_session_data(self, current_datetime=None):
 109         """
 110         Return the current session data, with native types coerced.
 111         """
 112         request = self.request
 113         data = request.session.get(SESSION_KEY)
 114 
 115         try:
 116             cookie_token = request.get_signed_cookie(
 117                 key=COOKIE_NAME, default=None, salt=COOKIE_SALT, max_age=MAX_AGE.total_seconds()
 118             )
 119         except BadSignature:
 120             logger.exception(
 121                 "superuser.bad-cookie-signature",
 122                 extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 123             )
 124             return
 125 
 126         if not cookie_token:
 127             if data:
 128                 logger.warn(
 129                     "superuser.missing-cookie-token",
 130                     extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 131                 )
 132             return False
 133         elif not data:
 134             logger.warn(
 135                 "superuser.missing-session-data",
 136                 extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 137             )
 138             return
 139 
 140         session_token = data.get("tok")
 141         if not session_token:
 142             logger.warn(
 143                 "superuser.missing-session-token",
 144                 extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 145             )
 146             return
 147 
 148         if not constant_time_compare(cookie_token, session_token):
 149             logger.warn(
 150                 "superuser.invalid-token",
 151                 extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 152             )
 153             return
 154 
 155         if data["uid"] != six.text_type(request.user.id):
 156             logger.warn(
 157                 "superuser.invalid-uid",
 158                 extra={
 159                     "ip_address": request.META["REMOTE_ADDR"],
 160                     "user_id": request.user.id,
 161                     "expected_user_id": data["uid"],
 162                 },
 163             )
 164             return
 165 
 166         if current_datetime is None:
 167             current_datetime = timezone.now()
 168 
 169         try:
 170             data["idl"] = datetime.utcfromtimestamp(float(data["idl"])).replace(tzinfo=timezone.utc)
 171         except (TypeError, ValueError):
 172             logger.warn(
 173                 "superuser.invalid-idle-expiration",
 174                 extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 175                 exc_info=True,
 176             )
 177             return
 178 
 179         if data["idl"] < current_datetime:
 180             logger.info(
 181                 "superuser.session-expired",
 182                 extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 183             )
 184             return
 185 
 186         try:
 187             data["exp"] = datetime.utcfromtimestamp(float(data["exp"])).replace(tzinfo=timezone.utc)
 188         except (TypeError, ValueError):
 189             logger.warn(
 190                 "superuser.invalid-expiration",
 191                 extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 192                 exc_info=True,
 193             )
 194             return
 195 
 196         if data["exp"] < current_datetime:
 197             logger.info(
 198                 "superuser.session-expired",
 199                 extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 200             )
 201             return
 202 
 203         return data

 205     def _populate(self, current_datetime=None):
 206         if current_datetime is None:
 207             current_datetime = timezone.now()
 208 
 209         request = self.request
 210         user = getattr(request, "user", None)
 211         if not hasattr(request, "session"):
 212             data = None
 213         elif not (user and user.is_superuser):
 214             data = None
 215         else:
 216             data = self.get_session_data(current_datetime=current_datetime)
 217 
 218         if not data:
 219             self._set_logged_out()
 220         else:
 221             self._set_logged_in(expires=data["exp"], token=data["tok"], user=user)
 222 
 223             if not self.is_active:
 224                 if self._inactive_reason:
 225                     logger.warn(
 226                         u"superuser.{}".format(self._inactive_reason),
 227                         extra={
 228                             "ip_address": request.META["REMOTE_ADDR"],
 229                             "user_id": request.user.id,
 230                         },
 231                     )
 232                 else:
 233                     logger.warn(
 234                         "superuser.inactive-unknown-reason",
 235                         extra={
 236                             "ip_address": request.META["REMOTE_ADDR"],
 237                             "user_id": request.user.id,
 238                         },
 239                     )

 241     def _set_logged_in(self, expires, token, user, current_datetime=None):
 242         # we bind uid here, as if you change users in the same request
 243         # we wouldn't want to still support superuser auth (given
 244         # the superuser check happens right here)
 245         assert user.is_superuser
 246         if current_datetime is None:
 247             current_datetime = timezone.now()
 248         self.token = token
 249         self.uid = six.text_type(user.id)
 250         # the absolute maximum age of this session
 251         self.expires = expires
 252         # do we have a valid superuser session?
 253         self.is_valid = True
 254         # is the session active? (it could be valid, but inactive)
 255         self._is_active, self._inactive_reason = self.is_privileged_request()
 256         self.request.session[SESSION_KEY] = {
 257             "exp": self.expires.strftime("%s"),
 258             "idl": (current_datetime + IDLE_MAX_AGE).strftime("%s"),
 259             "tok": self.token,
 260             # XXX(dcramer): do we really need the uid safety m echanism
 261             "uid": self.uid,
 262         }
 263 
 264     def _set_logged_out(self):
 265         self.uid = None
 266         self.expires = None
 267         self.token = None
 268         self._is_active = False
 269         self._inactive_reason = None
 270         self.is_valid = False
 271         self.request.session.pop(SESSION_KEY, None)
 272 
 273     def set_logged_in(self, user, current_datetime=None):
 274         """
 275         Mark a session as superuser-enabled.
 276         """
 277         request = self.request
 278         if current_datetime is None:
 279             current_datetime = timezone.now()
 280         self._set_logged_in(
 281             expires=current_datetime + MAX_AGE,
 282             token=get_random_string(12),
 283             user=user,
 284             current_datetime=current_datetime,
 285         )
 286         logger.info(
 287             "superuser.logged-in",
 288             extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": user.id},
 289         )
 290 
 291     def set_logged_out(self):
 292         """
 293         Mark a session as superuser-disabled.
 294         """
 295         request = self.request
 296         self._set_logged_out()
 297         logger.info(
 298             "superuser.logged-out",
 299             extra={"ip_address": request.META["REMOTE_ADDR"], "user_id": request.user.id},
 300         )

 302     def on_response(self, response, current_datetime=None):
 303         request = self.request
 304 
 305         if current_datetime is None:
 306             current_datetime = timezone.now()
 307 
 308         # always re-bind the cookie to update the idle expiration window
 309         if self.is_active:
 310             response.set_signed_cookie(
 311                 COOKIE_NAME,
 312                 self.token,
 313                 salt=COOKIE_SALT,
 314                 # set max_age to None, as we want this cookie to expire on browser close
 315                 max_age=None,
 316                 secure=request.is_secure() if COOKIE_SECURE is None else COOKIE_SECURE,
 317                 httponly=COOKIE_HTTPONLY,
 318                 path=COOKIE_PATH,
 319                 domain=COOKIE_DOMAIN,
 320             )
 321         # otherwise, if the session is invalid and there's a cookie set, clear it
 322         elif not self.is_valid and request.COOKIES.get(COOKIE_NAME):
 323             response.delete_cookie(COOKIE_NAME)
