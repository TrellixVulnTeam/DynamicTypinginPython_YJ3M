  10 # Ideally this and PluginManager abstracted from the same base, but
  11 # InstanceManager has become convoluted and wasteful
  12 class ProviderManager(object):
  13     def __init__(self):
  14         self.__values = {}
  15 
  16     def __iter__(self):
  17         return six.iteritems(self.__values)
  18 
  19     def get(self, key, **kwargs):
  20         try:
  21             cls = self.__values[key]
  22         except KeyError:
  23             raise ProviderNotRegistered(key)
  24         return cls(key=key, **kwargs)
  25 
  26     def exists(self, key):
  27         return key in self.__values
  28 
  29     def register(self, key, cls):
  30         self.__values[key] = cls
  31 
  32     def unregister(self, key, cls):
  33         try:
  34             if self.__values[key] != cls:
  35                 # dont allow unregistering of arbitrary provider
  36                 raise ProviderNotRegistered(key)
  37         except KeyError:
  38             # we gracefully handle a missing provider
  39             return
  40         del self.__values[key]
