  27 def _sso_params(member):
  28     """
  29     Return a tuple of (requires_sso, sso_is_valid) for a given member.
  30     """
  31     # TODO(dcramer): we want to optimize this access pattern as its several
  32     # network hops and needed in a lot of places
  33     try:
  34         auth_provider = AuthProvider.objects.get(organization=member.organization_id)
  35     except AuthProvider.DoesNotExist:
  36         sso_is_valid = True
  37         requires_sso = False
  38     else:
  39         if auth_provider.flags.allow_unlinked:
  40             requires_sso = False
  41             sso_is_valid = True
  42         else:
  43             requires_sso = True
  44             try:
  45                 auth_identity = AuthIdentity.objects.get(
  46                     auth_provider=auth_provider, user=member.user_id
  47                 )
  48             except AuthIdentity.DoesNotExist:
  49                 sso_is_valid = False
  50                 # If an owner is trying to gain access,
  51                 # allow bypassing SSO if there are no other
  52                 # owners with SSO enabled.
  53                 if member.role == roles.get_top_dog().id:
  54                     requires_sso = AuthIdentity.objects.filter(
  55                         auth_provider=auth_provider,
  56                         user__in=OrganizationMember.objects.filter(
  57                             organization=member.organization_id,
  58                             role=roles.get_top_dog().id,
  59                             user__is_active=True,
  60                         )
  61                         .exclude(id=member.id)
  62                         .values_list("user_id"),
  63                     ).exists()
  64             else:
  65                 sso_is_valid = auth_identity.is_valid(member)
  66     return requires_sso, sso_is_valid

  69 class BaseAccess(object):
  70     is_active = False
  71     sso_is_valid = False
  72     requires_sso = False
  73     organization_id = None
  74     # teams with membership
  75     teams = ()
  76     # projects with membership
  77     projects = ()
  78     # if has_global_access is True, then any project
  79     # matching organization_id is valid. This is used for
  80     # both `organization.allow_joinleave` and to indicate
  81     # that the role is global / a user is an active superuser
  82     has_global_access = False
  83     scopes = frozenset()
  84     permissions = frozenset()
  85     role = None
  86 
  87     def has_permission(self, permission):
  88         """
  89         Return bool representing if the user has the given permission.
  90 
  91         >>> access.has_permission('broadcasts.admin')
  92         """
  93         if not self.is_active:
  94             return False
  95         return permission in self.permissions
  96 
  97     def has_scope(self, scope):
  98         """
  99         Return bool representing if the user has the given scope.
 100 
 101         >>> access.has_project('org:read')
 102         """
 103         if not self.is_active:
 104             return False
 105         return scope in self.scopes
 106 
 107     def has_team(self, team):
 108         warnings.warn("has_team() is deprecated in favor of has_team_access", DeprecationWarning)
 109         return self.has_team_access(team)
 110 
 111     def has_team_access(self, team):
 112         """
 113         Return bool representing if a user should have access to information for the given team.
 114 
 115         >>> access.has_team_access(team)
 116         """
 117         if not self.is_active:
 118             return False
 119         if self.has_global_access and self.organization_id == team.organization_id:
 120             return True
 121         return team in self.teams
 122 
 123     def has_team_scope(self, team, scope):
 124         """
 125         Return bool representing if a user should have access with the given scope to information
 126         for the given team.
 127 
 128         >>> access.has_team_scope(team, 'team:read')
 129         """
 130         return self.has_team_access(team) and self.has_scope(scope)
 131 
 132     def has_project_access(self, project):
 133         """
 134         Return bool representing if a user should have access to information for the given project.
 135 
 136         >>> access.has_project_access(project)
 137         """
 138         if not self.is_active:
 139             return False
 140         if self.has_global_access and self.organization_id == project.organization_id:
 141             return True
 142         return project in self.projects
 143 
 144     def has_projects_access(self, projects):
 145         """
 146         Returns bool representing if a user should have access to every requested project
 147         """
 148         return all([self.has_project_access(project) for project in projects])
 149 
 150     def has_project_membership(self, project):
 151         """
 152         Return bool representing if a user has explicit membership for the given project.
 153 
 154         >>> access.has_project_membership(project)
 155         """
 156         if not self.is_active:
 157             return False
 158         return project in self.projects
 159 
 160     def has_project_scope(self, project, scope):
 161         """
 162         Return bool representing if a user should have access with the given scope to information
 163         for the given project.
 164 
 165         >>> access.has_project_scope(project, 'project:read')
 166         """
 167         return self.has_project_access(project) and self.has_scope(scope)
 168 
 169     def to_django_context(self):
 170         return {s.replace(":", "_"): self.has_scope(s) for s in settings.SENTRY_SCOPES}

 273 def from_request(request, organization=None, scopes=None):
 274     if not organization:
 275         return from_user(request.user, organization=organization, scopes=scopes)
 276 
 277     if getattr(request.user, "is_sentry_app", False):
 278         return from_sentry_app(request.user, organization=organization)
 279 
 280     if is_active_superuser(request):
 281         role = None
 282         # we special case superuser so that if they're a member of the org
 283         # they must still follow SSO checks, but they gain global access
 284         try:
 285             member = OrganizationMember.objects.get(user=request.user, organization=organization)
 286         except OrganizationMember.DoesNotExist:
 287             requires_sso, sso_is_valid = False, True
 288         else:
 289             requires_sso, sso_is_valid = _sso_params(member)
 290             role = member.role
 291 
 292         team_list = ()
 293 
 294         project_list = ()
 295         return Access(
 296             scopes=scopes if scopes is not None else settings.SENTRY_SCOPES,
 297             is_active=True,
 298             organization_id=organization.id if organization else None,
 299             teams=team_list,
 300             projects=project_list,
 301             sso_is_valid=sso_is_valid,
 302             requires_sso=requires_sso,
 303             has_global_access=True,
 304             permissions=UserPermission.for_user(request.user.id),
 305             role=role,
 306         )
 307 
 308     if hasattr(request, "auth") and not request.user.is_authenticated():
 309         return from_auth(request.auth, scopes=scopes)
 310 
 311     return from_user(request.user, organization, scopes=scopes)

 314 def from_sentry_app(user, organization=None):
 315     if not organization:
 316         return NoAccess()
 317 
 318     sentry_app = SentryApp.objects.get(proxy_user=user)
 319 
 320     if not sentry_app.is_installed_on(organization):
 321         return NoAccess()
 322 
 323     team_list = list(Team.objects.filter(organization=organization))
 324     project_list = list(
 325         Project.objects.filter(status=ProjectStatus.VISIBLE, teams__in=team_list).distinct()
 326     )
 327 
 328     return Access(
 329         scopes=sentry_app.scope_list,
 330         is_active=True,
 331         organization_id=organization.id,
 332         teams=team_list,
 333         projects=project_list,
 334         permissions=(),
 335         has_global_access=False,
 336         sso_is_valid=True,
 337         requires_sso=False,
 338     )

 341 def from_user(user, organization=None, scopes=None):
 342     if not user or user.is_anonymous() or not user.is_active:
 343         return DEFAULT
 344 
 345     if not organization:
 346         return OrganizationlessAccess(permissions=UserPermission.for_user(user.id))
 347 
 348     try:
 349         om = OrganizationMember.objects.get(user=user, organization=organization)
 350     except OrganizationMember.DoesNotExist:
 351         return OrganizationlessAccess(permissions=UserPermission.for_user(user.id))
 352 
 353     # ensure cached relation
 354     om.organization = organization
 355 
 356     return from_member(om, scopes=scopes)

 359 def from_member(member, scopes=None):
 360     # TODO(dcramer): we want to optimize this access pattern as its several
 361     # network hops and needed in a lot of places
 362     requires_sso, sso_is_valid = _sso_params(member)
 363 
 364     team_list = member.get_teams()
 365     with sentry_sdk.start_span(op="get_project_access_in_teams") as span:
 366         project_list = list(
 367             Project.objects.filter(status=ProjectStatus.VISIBLE, teams__in=team_list).distinct()
 368         )
 369         span.set_data("Project Count", len(project_list))
 370         span.set_data("Team Count", len(team_list))
 371 
 372     if scopes is not None:
 373         scopes = set(scopes) & member.get_scopes()
 374     else:
 375         scopes = member.get_scopes()
 376 
 377     return Access(
 378         is_active=True,
 379         requires_sso=requires_sso,
 380         sso_is_valid=sso_is_valid,
 381         scopes=scopes,
 382         organization_id=member.organization_id,
 383         teams=team_list,
 384         projects=project_list,
 385         has_global_access=bool(member.organization.flags.allow_joinleave)
 386         or roles.get(member.role).is_global,
 387         permissions=UserPermission.for_user(member.user_id),
 388         role=member.role,
 389     )

 392 def from_auth(auth, organization):
 393     if is_system_auth(auth):
 394         return SystemAccess()
 395     if auth.organization_id == organization.id:
 396         return OrganizationGlobalAccess(auth.organization)
 397     else:
 398         return DEFAULT
