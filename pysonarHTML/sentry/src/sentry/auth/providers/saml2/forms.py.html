  13 def extract_idp_data_from_parsed_data(data):
  14     """
  15     Transform data returned by the OneLogin_Saml2_IdPMetadataParser into the
  16     expected IdP dict shape.
  17     """
  18     idp = data.get("idp", {})
  19 
  20     # In some scenarios the IDP sticks the x509cert in the x509certMulti
  21     # parameter
  22     cert = idp.get("x509cert", idp.get("x509certMulti", {}).get("signing", [None])[0])
  23 
  24     return {
  25         "entity_id": idp.get("entityId"),
  26         "sso_url": idp.get("singleSignOnService", {}).get("url"),
  27         "slo_url": idp.get("singleLogoutService", {}).get("url"),
  28         "x509cert": cert,
  29     }

  32 def process_url(form):
  33     url = form.cleaned_data["metadata_url"]
  34     response = safe_urlopen(url)
  35     data = OneLogin_Saml2_IdPMetadataParser.parse(response.content)
  36     return extract_idp_data_from_parsed_data(data)

  39 def process_xml(form):
  40     # cast unicode xml to byte string so lxml won't complain when trying to
  41     # parse a xml document with a type declaration.
  42     xml = form.cleaned_data["metadata_xml"].encode("utf8")
  43     data = OneLogin_Saml2_IdPMetadataParser.parse(xml)
  44     return extract_idp_data_from_parsed_data(data)

  65 def process_metadata(form_cls, request, helper):
  66     form = form_cls()
  67 
  68     if "action_save" not in request.POST:
  69         return form
  70 
  71     form = form_cls(request.POST)
  72 
  73     if not form.is_valid():
  74         return form
  75 
  76     try:
  77         data = form_cls.processor(form)
  78     except Exception:
  79         errors = form._errors.setdefault("__all__", ErrorList())
  80         errors.append("Failed to parse provided SAML2 metadata")
  81         return form
  82 
  83     saml_form = SAMLForm(data)
  84     if not saml_form.is_valid():
  85         field_errors = [
  86             "%s: %s" % (k, ", ".join([force_text(i) for i in v]))
  87             for k, v in saml_form.errors.items()
  88         ]
  89         error_list = ", ".join(field_errors)
  90 
  91         errors = form._errors.setdefault("__all__", ErrorList())
  92         errors.append(u"Invalid metadata: {}".format(error_list))
  93         return form
  94 
  95     helper.bind_state("idp", data)
  96 
  97     # Data is bound, do not respond with a form to signal the nexts steps
  98     return None
