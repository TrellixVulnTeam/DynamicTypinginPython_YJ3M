  54 def get_provider(organization_slug):
  55     try:
  56         organization = Organization.objects.get(slug=organization_slug)
  57     except Organization.DoesNotExist:
  58         return None
  59 
  60     if organization.status != OrganizationStatus.VISIBLE:
  61         return None
  62 
  63     try:
  64         provider = AuthProvider.objects.get(organization=organization).get_provider()
  65     except AuthProvider.DoesNotExist:
  66         return None
  67 
  68     if not isinstance(provider, SAML2Provider):
  69         return None
  70 
  71     return provider

  74 class SAML2LoginView(AuthView):
  75     def dispatch(self, request, helper):
  76         if "SAMLResponse" in request.POST:
  77             return helper.next_step()
  78 
  79         provider = helper.provider
  80 
  81         # During the setup pipeline, the provider will not have been configured yet,
  82         # so build the config first from the state.
  83         if not provider.config:
  84             provider.config = provider.build_config(helper.fetch_state())
  85 
  86         saml_config = build_saml_config(provider.config, helper.organization.slug)
  87         auth = build_auth(request, saml_config)
  88 
  89         return self.redirect(auth.login())

  92 # With SAML, the SSO request can be initiated by both the Service Provider
  93 # (sentry) (the typical case) and the Identity Provider. In the second case,
  94 # the auth assertion is directly posted to the ACS URL. Because the user will
  95 # not have initiated their SSO flow we must provide a endpoint similar to
  96 # auth_provider_login, but with support for initing the auth flow.
  97 class SAML2AcceptACSView(BaseView):
  98     @method_decorator(csrf_exempt)
  99     def dispatch(self, request, organization_slug):
 100         from sentry.auth.helper import AuthHelper
 101 
 102         helper = AuthHelper.get_for_request(request)
 103 
 104         # SP initiated authentication, request helper is provided
 105         if helper:
 106             from sentry.web.frontend.auth_provider_login import AuthProviderLoginView
 107 
 108             sso_login = AuthProviderLoginView()
 109             return sso_login.handle(request)
 110 
 111         # IdP initiated authentication. The organization_slug must be valid and
 112         # an auth provider must exist for this organization to proceed with
 113         # IdP initiated SAML auth.
 114         try:
 115             organization = Organization.objects.get(slug=organization_slug)
 116         except Organization.DoesNotExist:
 117             messages.add_message(request, messages.ERROR, ERR_NO_SAML_SSO)
 118             return self.redirect(reverse("sentry-login"))
 119 
 120         try:
 121             auth_provider = AuthProvider.objects.get(organization=organization)
 122         except AuthProvider.DoesNotExist:
 123             messages.add_message(request, messages.ERROR, ERR_NO_SAML_SSO)
 124             return self.redirect(reverse("sentry-login"))
 125 
 126         helper = AuthHelper(
 127             request=request,
 128             organization=organization,
 129             auth_provider=auth_provider,
 130             flow=AuthHelper.FLOW_LOGIN,
 131         )
 132 
 133         helper.init_pipeline()
 134         return helper.current_step()

 137 class SAML2ACSView(AuthView):
 138     @method_decorator(csrf_exempt)
 139     def dispatch(self, request, helper):
 140         provider = helper.provider
 141 
 142         # If we're authenticating during the setup pipeline the provider will
 143         # not have been configured yet, build the config first from the state
 144         if not provider.config:
 145             provider.config = provider.build_config(helper.fetch_state())
 146 
 147         saml_config = build_saml_config(provider.config, helper.organization.slug)
 148 
 149         auth = build_auth(request, saml_config)
 150         auth.process_response()
 151 
 152         # SSO response verification failed
 153         if auth.get_errors():
 154             return helper.error(ERR_SAML_FAILED.format(reason=auth.get_last_error_reason()))
 155 
 156         helper.bind_state("auth_attributes", auth.get_attributes())
 157 
 158         # Not all providers send a session expiration value, but if they do,
 159         # we should respect it and set session cookies to expire at the given time.
 160         if auth.get_session_expiration() is not None:
 161             session_expiration = datetime.fromtimestamp(auth.get_session_expiration()).replace(
 162                 tzinfo=timezone.utc
 163             )
 164             request.session.set_expiry(session_expiration)
 165 
 166         return helper.next_step()

 169 class SAML2SLSView(BaseView):
 170     @method_decorator(csrf_exempt)
 171     def dispatch(self, request, organization_slug):
 172         provider = get_provider(organization_slug)
 173         if provider is None:
 174             messages.add_message(request, messages.ERROR, ERR_NO_SAML_SSO)
 175             return self.redirect("/")
 176 
 177         saml_config = build_saml_config(provider.config, organization_slug)
 178         auth = build_auth(request, saml_config)
 179 
 180         # No need to logout an anonymous user.
 181         should_logout = request.user.is_authenticated()
 182 
 183         def force_logout():
 184             logout(request)
 185 
 186         redirect_to = auth.process_slo(
 187             delete_session_cb=force_logout, keep_local_session=not should_logout
 188         )
 189 
 190         if not redirect_to:
 191             redirect_to = get_login_url()
 192 
 193         return self.redirect(redirect_to)

 196 class SAML2MetadataView(BaseView):
 197     def dispatch(self, request, organization_slug):
 198         provider = get_provider(organization_slug)
 199         config = provider.config if provider else {}
 200 
 201         saml_config = build_saml_config(config, organization_slug)
 202         saml_settings = OneLogin_Saml2_Settings(settings=saml_config, sp_validation_only=True)
 203         metadata = saml_settings.get_sp_metadata()
 204         errors = saml_settings.validate_metadata(metadata)
 205 
 206         if len(errors) > 0:
 207             message = "\n".join(errors)
 208             return HttpResponseServerError(content=message, content_type="plain/text")
 209 
 210         return HttpResponse(content=metadata, content_type="text/xml")

 220 class SAML2Provider(Provider):
 221     """
 222     Base SAML2 Authentication provider. SAML style authentication plugins
 223     should implement this.
 224 
 225     - The provider must implement the `get_configure_view`.
 226 
 227     - The provider must implement the `get_saml_setup_pipeline`. The
 228       AuthView(s) passed in this method MUST bind the `idp` configuration
 229       object. The dict should match the shape:
 230 
 231       >>> state.get('idp')
 232       {
 233         'entity_id': # Identity Provider entity ID. Usually a URL
 234         'x509cert':  # Identity Provider x509 public certificate
 235         'sso_url':   # Identity Provider Single Sign-On URL
 236         'slo_url':   # identity Provider Single Sign-Out URL
 237       }
 238 
 239       The provider may also bind the `advanced` configuration. This dict
 240       provides advanced SAML configurations. The dict should match the shape:
 241 
 242       HINT: You *probably* don't need this.
 243 
 244       >>> state.get('advanced')
 245       {
 246         'authn_request_signed':     # Sign the authentication request?
 247         'logout_request_signed':    # Sign the logout request?
 248         'logout_response_signed':   # Sign the logout response?
 249         'metadata_signed':          # Sign the metadata?
 250         'want_message_signed':      # Expect signed message
 251         'want_assertion_signed':    # Expect signed assertions
 252         'want_assertion_encrypted': # Expect encrypted assertions
 253         'signature_algorithm':      # Algorithm used to sign / verify requests / responses
 254         'digest_algorithm':         # Algorithm used to generate / verify digests
 255         'x509cert':                 # Public Service Provider key
 256         'private_key':              # Private Key used for signing / encryption
 257       }
 258 
 259     - The provider must EITHER specify an attribute mapping by implementing the
 260       `attribute_mapping` method OR bind the `attribute_mapping` key to the
 261       state during setup. The attribute mapping should map the `Attributes`
 262       constants to the Identity Provider attribute keys.
 263     """
 264 
 265     required_feature = "organizations:sso-saml2"
 266 
 267     def get_auth_pipeline(self):
 268         return [SAML2LoginView(), SAML2ACSView()]
 269 
 270     def get_setup_pipeline(self):
 271         return self.get_saml_setup_pipeline() + self.get_auth_pipeline()
 272 
 273     def get_saml_setup_pipeline(self):
 274         """
 275         Return a list of AuthViews to setup the SAML provider.
 276 
 277         The setup AuthView(s) must bind the `idp` parameter into the helper
 278         state.
 279         """
 280         raise NotImplementedError
 281 
 282     def attribute_mapping(self):
 283         """
 284         Returns the default Attribute Key -> IdP attribute key mapping.
 285 
 286         This value will be merged into the configuration by self.build_config,
 287         however, should a attribute_mapping exist in the helper state at
 288         configuration build time, these may be overridden.
 289         """
 290         return {}
 291 
 292     def build_config(self, state):
 293         config = state
 294 
 295         # Default attribute mapping if none bound
 296         if "attribute_mapping" not in config:
 297             config["attribute_mapping"] = self.attribute_mapping()
 298 
 299         return config
 300 
 301     def build_identity(self, state):
 302         raw_attributes = state["auth_attributes"]
 303         attributes = {}
 304 
 305         # map configured provider attributes
 306         for key, provider_key in iteritems(self.config["attribute_mapping"]):
 307             attributes[key] = raw_attributes.get(provider_key, [""])[0]
 308 
 309         # Email and identifier MUST be correctly mapped
 310         if not attributes[Attributes.IDENTIFIER] or not attributes[Attributes.USER_EMAIL]:
 311             raise IdentityNotValid(
 312                 _(
 313                     "Failed to map SAML attributes. Assertion returned the following attribute keys: %(keys)s"
 314                 )
 315                 % {"keys": raw_attributes.keys()}
 316             )
 317 
 318         name = (attributes[k] for k in (Attributes.FIRST_NAME, Attributes.LAST_NAME))
 319         name = " ".join([_f for _f in name if _f])
 320 
 321         return {
 322             "id": attributes[Attributes.IDENTIFIER],
 323             "email": attributes[Attributes.USER_EMAIL],
 324             "name": name,
 325         }
 326 
 327     def refresh_identity(self, auth_identity):
 328         # Nothing to refresh
 329         return

 332 def build_saml_config(provider_config, org):
 333     """
 334     Construct the SAML configuration dict to be passed into the OneLogin SAML
 335     library.
 336 
 337     For more details about the structure of this object see the
 338     SAML2Provider.build_config method.
 339     """
 340     avd = provider_config.get("advanced", {})
 341 
 342     security_config = {
 343         "authnRequestsSigned": avd.get("authn_request_signed", False),
 344         "logoutRequestSigned": avd.get("logout_request_signed", False),
 345         "logoutResponseSigned": avd.get("logout_response_signed", False),
 346         "signMetadata": avd.get("metadata_signed", False),
 347         "wantMessagesSigned": avd.get("want_message_signed", False),
 348         "wantAssertionsSigned": avd.get("want_assertion_signed", False),
 349         "wantAssertionsEncrypted": avd.get("want_assertion_encrypted", False),
 350         "signatureAlgorithm": avd.get("signature_algorithm", OneLogin_Saml2_Constants.RSA_SHA256),
 351         "digestAlgorithm": avd.get("digest_algorithm", OneLogin_Saml2_Constants.SHA256),
 352         "wantNameId": False,
 353         "requestedAuthnContext": False,
 354     }
 355 
 356     # TODO(epurkhiser): This is also available in the helper and should probably come from there.
 357     acs_url = absolute_uri(reverse("sentry-auth-organization-saml-acs", args=[org]))
 358     sls_url = absolute_uri(reverse("sentry-auth-organization-saml-sls", args=[org]))
 359     metadata_url = absolute_uri(reverse("sentry-auth-organization-saml-metadata", args=[org]))
 360 
 361     saml_config = {
 362         "strict": True,
 363         "sp": {
 364             "entityId": metadata_url,
 365             "assertionConsumerService": {
 366                 "url": acs_url,
 367                 "binding": OneLogin_Saml2_Constants.BINDING_HTTP_POST,
 368             },
 369             "singleLogoutService": {
 370                 "url": sls_url,
 371                 "binding": OneLogin_Saml2_Constants.BINDING_HTTP_REDIRECT,
 372             },
 373         },
 374         "security": security_config,
 375     }
 376 
 377     idp = provider_config.get("idp")
 378 
 379     if idp is not None:
 380         saml_config["idp"] = {
 381             "entityId": idp["entity_id"],
 382             "x509cert": idp["x509cert"],
 383             "singleSignOnService": {"url": idp["sso_url"]},
 384             "singleLogoutService": {"url": idp["slo_url"]},
 385         }
 386 
 387     if avd.get("x509cert") is not None:
 388         saml_config["sp"]["x509cert"] = avd["x509cert"]
 389 
 390     if avd.get("private_key") is not None:
 391         saml_config["sp"]["privateKey"] = avd["private_key"]
 392 
 393     return saml_config

 396 def build_auth(request, saml_config):
 397     """
 398     Construct a OneLogin_Saml2_Auth object for the current request.
 399     """
 400     url = urlparse(options.get("system.url-prefix"))
 401     saml_request = {
 402         "https": "on" if url.scheme == "https" else "off",
 403         "http_host": url.hostname,
 404         "script_name": request.META["PATH_INFO"],
 405         "server_port": url.port,
 406         "get_data": request.GET.copy(),
 407         "post_data": request.POST.copy(),
 408     }
 409 
 410     return OneLogin_Saml2_Auth(saml_request, saml_config)
