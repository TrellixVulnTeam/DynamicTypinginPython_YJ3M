  15 class FetchUser(AuthView):
  16     def __init__(self, domains, version, *args, **kwargs):
  17         self.domains = domains
  18         self.version = version
  19         super(FetchUser, self).__init__(*args, **kwargs)

  21     def dispatch(self, request, helper):
  22         data = helper.fetch_state("data")
  23 
  24         try:
  25             id_token = data["id_token"]
  26         except KeyError:
  27             logger.error("Missing id_token in OAuth response: %s" % data)
  28             return helper.error(ERR_INVALID_RESPONSE)
  29 
  30         try:
  31             _, payload, _ = map(urlsafe_b64decode, id_token.split(".", 2))
  32         except Exception as exc:
  33             logger.error(u"Unable to decode id_token: %s" % exc, exc_info=True)
  34             return helper.error(ERR_INVALID_RESPONSE)
  35 
  36         try:
  37             payload = json.loads(payload)
  38         except Exception as exc:
  39             logger.error(u"Unable to decode id_token payload: %s" % exc, exc_info=True)
  40             return helper.error(ERR_INVALID_RESPONSE)
  41 
  42         if not payload.get("email"):
  43             logger.error("Missing email in id_token payload: %s" % id_token)
  44             return helper.error(ERR_INVALID_RESPONSE)
  45 
  46         # support legacy style domains with pure domain regexp
  47         if self.version is None:
  48             domain = extract_domain(payload["email"])
  49         else:
  50             domain = payload.get("hd")
  51 
  52         if domain is None:
  53             return helper.error(ERR_INVALID_DOMAIN % (domain,))
  54 
  55         if domain in DOMAIN_BLOCKLIST:
  56             return helper.error(ERR_INVALID_DOMAIN % (domain,))
  57 
  58         if self.domains and domain not in self.domains:
  59             return helper.error(ERR_INVALID_DOMAIN % (domain,))
  60 
  61         helper.bind_state("domain", domain)
  62         helper.bind_state("user", payload)
  63 
  64         return helper.next_step()
