  32     def __init__(self, domain=None, domains=None, version=None, **config):
  33         if domain:
  34             if domains:
  35                 domains.append(domain)
  36             else:
  37                 domains = [domain]
  38         self.domains = domains
  39         # if a domain is not configured this is part of the setup pipeline
  40         # this is a bit complex in Sentry's SSO implementation as we don't
  41         # provide a great way to get initial state for new setup pipelines
  42         # vs missing state in case of migrations.
  43         if domains is None:
  44             version = DATA_VERSION
  45         else:
  46             version = None
  47         self.version = version
  48         super(GoogleOAuth2Provider, self).__init__(**config)

  76     def build_identity(self, state):
  77         # https://developers.google.com/identity/protocols/OpenIDConnect#server-flow
  78         # data.user => {
  79         #      "iss":"accounts.google.com",
  80         #      "at_hash":"HK6E_P6Dh8Y93mRNtsDB1Q",
  81         #      "email_verified":"true",
  82         #      "sub":"10769150350006150715113082367",
  83         #      "azp":"1234987819200.apps.googleusercontent.com",
  84         #      "email":"jsmith@example.com",
  85         #      "aud":"1234987819200.apps.googleusercontent.com",
  86         #      "iat":1353601026,
  87         #      "exp":1353604926,
  88         #      "hd":"example.com"
  89         # }
  90         data = state["data"]
  91         user_data = state["user"]
  92 
  93         # XXX(epurkhiser): We initially were using the email as the id key.
  94         # This caused account dupes on domain changes. Migrate to the
  95         # account-unique sub key.
  96         user_id = MigratingIdentityId(id=user_data["sub"], legacy_id=user_data["email"])
  97 
  98         return {
  99             "id": user_id,
 100             "email": user_data["email"],
 101             "name": user_data["email"],
 102             "data": self.get_oauth_data(data),
 103             "email_verified": user_data["email_verified"],
 104         }
