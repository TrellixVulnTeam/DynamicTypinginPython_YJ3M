  19 def _get_name_from_email(email):
  20     """
  21     Given an email return a capitalized name. Ex. john.smith@example.com would return John Smith.
  22     """
  23     name = email.rsplit("@", 1)[0]
  24     name = " ".join([n_part.capitalize() for n_part in name.split(".")])
  25     return name

  28 class FetchUser(AuthView):
  29     def __init__(self, org=None, *args, **kwargs):
  30         self.org = org
  31         super(FetchUser, self).__init__(*args, **kwargs)
  32 
  33     def handle(self, request, helper):
  34         client = GitHubClient(helper.fetch_state("data")["access_token"])
  35 
  36         if self.org is not None:
  37             if not client.is_org_member(self.org["id"]):
  38                 return helper.error(ERR_NO_ORG_ACCESS)
  39 
  40         user = client.get_user()
  41 
  42         if not user.get("email"):
  43             emails = client.get_user_emails()
  44             email = [
  45                 e["email"]
  46                 for e in emails
  47                 if ((not REQUIRE_VERIFIED_EMAIL) | e["verified"]) and e["primary"]
  48             ]
  49             if len(email) == 0:
  50                 if REQUIRE_VERIFIED_EMAIL:
  51                     msg = ERR_NO_VERIFIED_PRIMARY_EMAIL
  52                 else:
  53                     msg = ERR_NO_PRIMARY_EMAIL
  54                 return helper.error(msg)
  55             elif len(email) > 1:
  56                 if REQUIRE_VERIFIED_EMAIL:
  57                     msg = ERR_NO_SINGLE_VERIFIED_PRIMARY_EMAIL
  58                 else:
  59                     msg = ERR_NO_SINGLE_PRIMARY_EMAIL
  60                 return helper.error(msg)
  61             else:
  62                 user["email"] = email[0]
  63 
  64         # A user hasn't set their name in their Github profile so it isn't
  65         # populated in the response
  66         if not user.get("name"):
  67             user["name"] = _get_name_from_email(user["email"])
  68 
  69         helper.bind_state("user", user)
  70 
  71         return helper.next_step()

  78 class ConfirmEmail(AuthView):
  79     def handle(self, request, helper):
  80         user = helper.fetch_state("user")
  81 
  82         # TODO(dcramer): this isnt ideal, but our current flow doesnt really
  83         # support this behavior;
  84         try:
  85             auth_identity = AuthIdentity.objects.select_related("user").get(
  86                 auth_provider=helper.auth_provider, ident=user["id"]
  87             )
  88         except AuthIdentity.DoesNotExist:
  89             pass
  90         else:
  91             user["email"] = auth_identity.user.email
  92 
  93         if user.get("email"):
  94             return helper.next_step()
  95 
  96         form = ConfirmEmailForm(request.POST or None)
  97         if form.is_valid():
  98             user["email"] = form.cleaned_data["email"]
  99             helper.bind_state("user", user)
 100             return helper.next_step()
 101 
 102         return self.respond("sentry_auth_github/enter-email.html", {"form": form})

 115 class SelectOrganization(AuthView):
 116     def __init__(self, *args, **kwargs):
 117         super(SelectOrganization, self).__init__(*args, **kwargs)
 118 
 119     def handle(self, request, helper):
 120         client = GitHubClient(helper.fetch_state("data")["access_token"])
 121         org_list = client.get_org_list()
 122 
 123         form = SelectOrganizationForm(org_list, request.POST or None)
 124         if form.is_valid():
 125             org_id = form.cleaned_data["org"]
 126             org = [o for o in org_list if org_id == six.text_type(o["id"])][0]
 127             helper.bind_state("org", org)
 128             return helper.next_step()
 129 
 130         return self.respond(
 131             "sentry_auth_github/select-organization.html", {"form": form, "org_list": org_list}
 132         )
