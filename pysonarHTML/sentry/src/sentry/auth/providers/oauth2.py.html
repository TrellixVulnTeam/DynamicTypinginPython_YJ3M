  19 class OAuth2Login(AuthView):
  20     authorize_url = None
  21     client_id = None
  22     scope = ""
  23 
  24     def __init__(self, authorize_url=None, client_id=None, scope=None, *args, **kwargs):
  25         super(OAuth2Login, self).__init__(*args, **kwargs)
  26         if authorize_url is not None:
  27             self.authorize_url = authorize_url
  28         if client_id is not None:
  29             self.client_id = client_id
  30         if scope is not None:
  31             self.scope = scope
  32 
  33     def get_scope(self):
  34         return self.scope
  35 
  36     def get_authorize_url(self):
  37         return self.authorize_url
  38 
  39     def get_authorize_params(self, state, redirect_uri):
  40         return {
  41             "client_id": self.client_id,
  42             "response_type": "code",
  43             "scope": self.get_scope(),
  44             "state": state,
  45             "redirect_uri": redirect_uri,
  46         }
  47 
  48     def dispatch(self, request, helper):
  49         if "code" in request.GET:
  50             return helper.next_step()
  51 
  52         state = uuid4().hex
  53 
  54         params = self.get_authorize_params(
  55             state=state, redirect_uri=absolute_uri(helper.get_redirect_url())
  56         )
  57         redirect_uri = u"{}?{}".format(self.get_authorize_url(), urlencode(params))
  58 
  59         helper.bind_state("state", state)
  60 
  61         return self.redirect(redirect_uri)

  64 class OAuth2Callback(AuthView):
  65     access_token_url = None
  66     client_id = None
  67     client_secret = None
  68 
  69     def __init__(self, access_token_url=None, client_id=None, client_secret=None, *args, **kwargs):
  70         super(OAuth2Callback, self).__init__(*args, **kwargs)
  71         if access_token_url is not None:
  72             self.access_token_url = access_token_url
  73         if client_id is not None:
  74             self.client_id = client_id
  75         if client_secret is not None:
  76             self.client_secret = client_secret
  77 
  78     def get_token_params(self, code, redirect_uri):
  79         return {
  80             "grant_type": "authorization_code",
  81             "code": code,
  82             "redirect_uri": redirect_uri,
  83             "client_id": self.client_id,
  84             "client_secret": self.client_secret,
  85         }
  86 
  87     def exchange_token(self, request, helper, code):
  88         # TODO: this needs the auth yet
  89         data = self.get_token_params(
  90             code=code, redirect_uri=absolute_uri(helper.get_redirect_url())
  91         )
  92         req = safe_urlopen(self.access_token_url, data=data)
  93         body = safe_urlread(req)
  94         if req.headers["Content-Type"].startswith("application/x-www-form-urlencoded"):
  95             return dict(parse_qsl(body))
  96         return json.loads(body)
  97 
  98     def dispatch(self, request, helper):
  99         error = request.GET.get("error")
 100         state = request.GET.get("state")
 101         code = request.GET.get("code")
 102 
 103         if error:
 104             return helper.error(error)
 105 
 106         if state != helper.fetch_state("state"):
 107             return helper.error(ERR_INVALID_STATE)
 108 
 109         data = self.exchange_token(request, helper, code)
 110 
 111         if "error_description" in data:
 112             return helper.error(data["error_description"])
 113 
 114         if "error" in data:
 115             logging.info("Error exchanging token: %s", data["error"])
 116             return helper.error("Unable to retrieve your token")
 117 
 118         # we can either expect the API to be implicit and say "im looking for
 119         # blah within state data" or we need to pass implementation + call a
 120         # hook here
 121         helper.bind_state("data", data)
 122 
 123         return helper.next_step()

 126 class OAuth2Provider(Provider):
 127     client_id = None
 128     client_secret = None
 129 
 130     def get_client_id(self):
 131         return self.client_id
 132 
 133     def get_client_secret(self):
 134         return self.client_secret
 135 
 136     def get_auth_pipeline(self):
 137         return [
 138             OAuth2Login(client_id=self.get_client_id()),
 139             OAuth2Callback(client_id=self.get_client_id(), client_secret=self.get_client_secret()),
 140         ]
 141 
 142     def get_refresh_token_url(self):
 143         raise NotImplementedError
 144 
 145     def get_refresh_token_params(self, refresh_token):
 146         return {
 147             "client_id": self.get_client_id(),
 148             "client_secret": self.get_client_secret(),
 149             "grant_type": "refresh_token",
 150             "refresh_token": refresh_token,
 151         }
 152 
 153     def get_oauth_data(self, payload):
 154         data = {"access_token": payload["access_token"], "token_type": payload["token_type"]}
 155         if "expires_in" in payload:
 156             data["expires"] = int(time()) + int(payload["expires_in"])
 157         if "refresh_token" in payload:
 158             data["refresh_token"] = payload["refresh_token"]
 159         return data
 160

 162         # data = state['data']
 163         # return {
 164         #     'id': '',
 165         #     'email': '',
 166         #     'name': '',
 167         #     'data': self.get_oauth_data(data),
 168         # }
 169         raise NotImplementedError
 170 
 171     def update_identity(self, new_data, current_data):
 172         # we want to maintain things like refresh_token that might not
 173         # exist on a refreshed state
 174         if "refresh_token" in current_data:
 175             new_data.setdefault("refresh_token", current_data["refresh_token"])
 176         return new_data
 177 
 178     def refresh_identity(self, auth_identity):
 179         refresh_token = auth_identity.data.get("refresh_token")
 180 
 181         if not refresh_token:
 182             raise IdentityNotValid("Missing refresh token")
 183 
 184         data = self.get_refresh_token_params(refresh_token=refresh_token)
 185         req = safe_urlopen(self.get_refresh_token_url(), data=data)
 186 
 187         try:
 188             body = safe_urlread(req)
 189             payload = json.loads(body)
 190         except Exception:
 191             payload = {}
 192 
 193         error = payload.get("error", "unknown_error")
 194         error_description = payload.get("error_description", "no description available")
 195 
 196         formatted_error = u"HTTP {} ({}): {}".format(req.status_code, error, error_description)
 197 
 198         if req.status_code == 401:
 199             raise IdentityNotValid(formatted_error)
 200 
 201         if req.status_code == 400:
 202             # this may not be common, but at the very least Google will return
 203             # an invalid grant when a user is suspended
 204             if error == "invalid_grant":
 205                 raise IdentityNotValid(formatted_error)
 206 
 207         if req.status_code != 200:
 208             raise Exception(formatted_error)
 209 
 210         auth_identity.data.update(self.get_oauth_data(payload))
 211         auth_identity.update(data=auth_identity.data)
