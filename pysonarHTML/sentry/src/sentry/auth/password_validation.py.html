  15 def get_default_password_validators():
  16     global _default_password_validators
  17     if _default_password_validators is None:

  22 def get_password_validators(validator_config):
  23     validators = []
  24     for validator in validator_config:
  25         try:
  26             klass = import_string(validator["NAME"])
  27         except ImportError:
  28             msg = "The module in NAME could not be imported: %s. Check your AUTH_PASSWORD_VALIDATORS setting."
  29             raise ImproperlyConfigured(msg % validator["NAME"])
  30         validators.append(klass(**validator.get("OPTIONS", {})))
  31 
  32     return validators

  35 def validate_password(password, password_validators=None):
  36     """
  37     Validate whether the password meets all validator requirements.
  38 
  39     If the password is valid, return ``None``.
  40     If the password is invalid, raise ValidationError with all error messages.
  41     """
  42     errors = []
  43     if password_validators is None:
  44         password_validators = get_default_password_validators()
  45     for validator in password_validators:
  46         try:
  47             validator.validate(password)
  48         except ValidationError as error:
  49             errors.append(error)
  50     if errors:
  51         raise ValidationError(errors)

  54 def password_validators_help_texts(password_validators=None):
  55     """
  56     Return a list of all help texts of all configured validators.
  57     """
  58     help_texts = []
  59     if password_validators is None:
  60         password_validators = get_default_password_validators()
  61     for validator in password_validators:
  62         help_texts.append(validator.get_help_text())
  63     return help_texts

  66 def _password_validators_help_text_html(password_validators=None):
  67     """
  68     Return an HTML string with all help texts of all configured validators
  69     in an <ul>.
  70     """
  71     help_texts = password_validators_help_texts(password_validators)
  72     help_items = [format_html("<li>{}</li>", help_text) for help_text in help_texts]
  73     return "<ul>%s</ul>" % "".join(help_items) if help_items else ""

  79 class MinimumLengthValidator(object):
  80     """
  81     Validate whether the password is of a minimum length.
  82     """
  83 
  84     def __init__(self, min_length=8):
  85         self.min_length = min_length
  86 
  87     def validate(self, password):
  88         if len(password) < self.min_length:
  89             raise ValidationError(
  90                 ungettext(
  91                     "This password is too short. It must contain at least %(min_length)d character.",
  92                     "This password is too short. It must contain at least %(min_length)d characters.",
  93                     self.min_length,
  94                 ),
  95                 code="password_too_short",
  96                 params={"min_length": self.min_length},
  97             )
  98 
  99     def get_help_text(self):
 100         return ungettext(
 101             "Your password must contain at least %(min_length)d character.",
 102             "Your password must contain at least %(min_length)d characters.",
 103             self.min_length,
 104         ) % {"min_length": self.min_length}
