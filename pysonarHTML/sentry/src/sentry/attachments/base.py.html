  21 class CachedAttachment(object):
  22     def __init__(
  23         self,
  24         key=None,
  25         id=None,
  26         name=None,
  27         content_type=None,
  28         type=None,
  29         data=UNINITIALIZED_DATA,
  30         chunks=None,
  31         cache=None,
  32     ):
  33         self.key = key
  34         self.id = id
  35 
  36         self.name = name
  37         self.content_type = content_type
  38         self.type = type or "event.attachment"
  39         assert isinstance(self.type, string_types), self.type
  40 
  41         self._data = data
  42         self.chunks = chunks
  43         self._cache = cache
  44 
  45     @classmethod
  46     def from_upload(cls, file, **kwargs):
  47         return CachedAttachment(
  48             name=file.name, content_type=file.content_type, data=file.read(), **kwargs
  49         )
  50 
  51     @property
  52     def data(self):
  53         if self._data is UNINITIALIZED_DATA and self._cache is not None:
  54             self._data = self._cache.get_data(self)
  55 
  56         assert self._data is not UNINITIALIZED_DATA
  57         return self._data
  58 
  59     def delete(self):
  60         for key in self.chunk_keys:
  61             self._cache.inner.delete(key)
  62 
  63     @property
  64     def chunk_keys(self):
  65         assert self.key is not None
  66         assert self.id is not None
  67 
  68         if self.chunks is None:
  69             yield ATTACHMENT_UNCHUNKED_DATA_KEY.format(key=self.key, id=self.id)
  70             return
  71 
  72         for chunk_index in range(self.chunks):
  73             yield ATTACHMENT_DATA_CHUNK_KEY.format(
  74                 key=self.key, id=self.id, chunk_index=chunk_index
  75             )
  76 
  77     def meta(self):
  78         return prune_empty_keys(
  79             {
  80                 "id": self.id,
  81                 "name": self.name,
  82                 "content_type": self.content_type,
  83                 "type": self.type,
  84                 "chunks": self.chunks,
  85             }
  86         )

  89 class BaseAttachmentCache(object):
  90     def __init__(self, inner):
  91         self.inner = inner
  92 
  93     def set(self, key, attachments, timeout=None):
  94         for id, attachment in enumerate(attachments):
  95             if attachment.chunks is not None:
  96                 continue
  97             # TODO(markus): We need to get away from sequential IDs, they
  98             # are risking collision when using Relay.
  99             if attachment.id is None:
 100                 attachment.id = id
 101 
 102             if attachment.key is None:
 103                 attachment.key = key
 104 
 105             metrics_tags = {"type": attachment.type}
 106             self.set_unchunked_data(
 107                 key=key,
 108                 id=attachment.id,
 109                 data=attachment.data,
 110                 timeout=timeout,
 111                 metrics_tags=metrics_tags,
 112             )
 113 
 114         meta = []
 115 
 116         for attachment in attachments:
 117             attachment._cache = self
 118             meta.append(attachment.meta())
 119 
 120         self.inner.set(ATTACHMENT_META_KEY.format(key=key), meta, timeout, raw=False)
 121 
 122     def set_chunk(self, key, id, chunk_index, chunk_data, timeout=None):
 123         key = ATTACHMENT_DATA_CHUNK_KEY.format(key=key, id=id, chunk_index=chunk_index)
 124         self.inner.set(key, zlib.compress(chunk_data), timeout, raw=True)
 125 
 126     def set_unchunked_data(self, key, id, data, timeout=None, metrics_tags=None):
 127         key = ATTACHMENT_UNCHUNKED_DATA_KEY.format(key=key, id=id)
 128         compressed = zlib.compress(data)
 129         metrics.timing("attachments.blob-size.raw", len(data), tags=metrics_tags)
 130         metrics.timing("attachments.blob-size.compressed", len(compressed), tags=metrics_tags)
 131         metrics.incr("attachments.received", tags=metrics_tags, skip_internal=False)
 132         self.inner.set(key, compressed, timeout, raw=True)
 133 
 134     def get_from_chunks(self, key, **attachment):
 135         return CachedAttachment(key=key, cache=self, **attachment)
 136 
 137     def get(self, key):
 138         result = self.inner.get(ATTACHMENT_META_KEY.format(key=key), raw=False)
 139 
 140         for id, attachment in enumerate(result or ()):
 141             attachment.setdefault("id", id)
 142             attachment.setdefault("key", key)
 143             yield CachedAttachment(cache=self, **attachment)
 144 
 145     def get_data(self, attachment):
 146         data = []
 147 
 148         for key in attachment.chunk_keys:
 149             raw_data = self.inner.get(key, raw=True)
 150             if raw_data is None:
 151                 raise MissingAttachmentChunks()
 152             data.append(zlib.decompress(raw_data))
 153 
 154         return b"".join(data)
 155 
 156     def delete(self, key):
 157         for attachment in self.get(key):
 158             attachment.delete()
 159 
 160         self.inner.delete(ATTACHMENT_META_KEY.format(key=key))
