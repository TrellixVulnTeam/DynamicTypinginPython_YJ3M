  32 class AmqpBackend(object):
  33     def __init__(self, broker_url):
  34         dsn = urlparse(broker_url)
  35         self.conn_info = dict(
  36             host=dsn.hostname,
  37             port=dsn.port,
  38             userid=dsn.username,
  39             password=dsn.password,
  40             virtual_host=dsn.path[1:],
  41         )
  42 
  43     def get_conn(self):
  44         from librabbitmq import Connection
  45 
  46         return Connection(**self.conn_info)
  47 
  48     def _get_size_from_channel(self, channel, queue):
  49         # In AMQP, the way to do this is to attempt to create a queue passively.
  50         # which is basically checking for it's existence (passive=True), this also
  51         # returns back the queue size.
  52         try:
  53             _, size, _ = channel.queue_declare(queue, passive=True)
  54         except Exception:
  55             return 0
  56         return size
  57 
  58     def bulk_get_sizes(self, queues):
  59         sizes = []
  60         with self.get_conn() as conn:
  61             with conn.channel() as channel:
  62                 for queue in queues:
  63                     sizes.append((queue, self._get_size_from_channel(channel, queue)))
  64                 return sizes
  65 
  66     def get_size(self, queue):
  67         with self.get_conn() as conn:
  68             with conn.channel() as channel:
  69                 return self._get_size_from_channel(channel, queue)
  70 
  71     def purge_queue(self, queue):
  72         with self.get_conn() as conn:
  73             with conn.channel() as channel:
  74                 return channel.queue_purge(queue)
