   9 from sentry import analytics
  10 from sentry.mediators import Mediator, Param
  11 from sentry.models import (
  12     AuditLogEntryEvent,

  15     SentryApp,
  16     SentryAppComponent,
  17     User,
  18 )
  19 from sentry.constants import SentryAppStatus
  20 from sentry.models.sentryapp import generate_slug, default_uuid

  23 class Creator(Mediator):
  24     name = Param(six.string_types)
  25     author = Param(six.string_types)
  26     organization = Param("sentry.models.Organization")
  27     scopes = Param(Iterable, default=lambda self: [])
  28     events = Param(Iterable, default=lambda self: [])
  29     webhook_url = Param(
  30         six.string_types, required=False
  31     )  # only not required for internal integrations but internalCreator calls this
  32     redirect_url = Param(six.string_types, required=False)
  33     is_alertable = Param(bool, default=False)
  34     verify_install = Param(bool, default=True)
  35     schema = Param(dict, default=lambda self: {})
  36     overview = Param(six.string_types, required=False)
  37     allowed_origins = Param(Iterable, default=lambda self: [])
  38     request = Param("rest_framework.request.Request", required=False)
  39     user = Param("sentry.models.User")
  40     is_internal = Param(bool)
  41 
  42     def call(self):
  43         self.slug = self._generate_and_validate_slug()
  44         self.proxy = self._create_proxy_user()
  45         self.api_app = self._create_api_application()
  46         self.sentry_app = self._create_sentry_app()
  47         self._create_ui_components()
  48         self._create_integration_feature()
  49         return self.sentry_app
  50 
  51     def _generate_and_validate_slug(self):
  52         slug = generate_slug(self.name, is_internal=self.is_internal)
  53 
  54         # validate globally unique slug
  55         queryset = SentryApp.with_deleted.filter(slug=slug)
  56 
  57         if queryset.exists():
  58             # In reality, the slug is taken but it's determined by the name field
  59             raise ValidationError(
  60                 {"name": [u"Name {} is already taken, please use another.".format(self.name)]}
  61             )
  62         return slug
  63 
  64     def _create_proxy_user(self):
  65         # need a proxy user name that will always be unique
  66         return User.objects.create(
  67             username=u"{}-{}".format(self.slug, default_uuid()), is_sentry_app=True
  68         )
  69 
  70     def _create_api_application(self):
  71         return ApiApplication.objects.create(
  72             owner_id=self.proxy.id, allowed_origins="\n".join(self.allowed_origins)
  73         )
  74 
  75     def _create_sentry_app(self):
  76         from sentry.mediators.service_hooks.creator import expand_events
  77 
  78         kwargs = {
  79             "name": self.name,
  80             "slug": self.slug,
  81             "author": self.author,
  82             "application_id": self.api_app.id,
  83             "owner_id": self.organization.id,
  84             "proxy_user_id": self.proxy.id,
  85             "scope_list": self.scopes,
  86             "events": expand_events(self.events),
  87             "schema": self.schema or {},
  88             "webhook_url": self.webhook_url,
  89             "redirect_url": self.redirect_url,
  90             "is_alertable": self.is_alertable,
  91             "verify_install": self.verify_install,
  92             "overview": self.overview,
  93         }
  94 
  95         if self.is_internal:
  96             kwargs["status"] = SentryAppStatus.INTERNAL
  97 
  98         return SentryApp.objects.create(**kwargs)
  99 
 100     def _create_ui_components(self):
 101         schema = self.schema or {}
 102 
 103         for element in schema.get("elements", []):
 104             SentryAppComponent.objects.create(
 105                 type=element["type"], sentry_app_id=self.sentry_app.id, schema=element
 106             )
 107 
 108     def _create_integration_feature(self):
 109         # sentry apps must have at least one feature
 110         # defaults to 'integrations-api'
 111         try:
 112             with transaction.atomic():
 113                 IntegrationFeature.objects.create(sentry_app=self.sentry_app)
 114         except IntegrityError as e:
 115             self.log(sentry_app=self.sentry_app.slug, error_message=six.text_type(e))
 116 
 117     def audit(self):
 118         from sentry.utils.audit import create_audit_entry
 119 
 120         if self.request:
 121             create_audit_entry(
 122                 request=self.request,
 123                 organization=self.organization,
 124                 target_object=self.organization.id,
 125                 event=AuditLogEntryEvent.SENTRY_APP_ADD,
 126                 data={"sentry_app": self.sentry_app.name},
 127             )
 128 
 129     def record_analytics(self):
 130         analytics.record(
 131             "sentry_app.created",
 132             user_id=self.user.id,
 133             organization_id=self.organization.id,
 134             sentry_app=self.sentry_app.slug,
 135         )
