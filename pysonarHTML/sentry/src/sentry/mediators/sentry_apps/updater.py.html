  19 class Updater(Mediator):
  20     sentry_app = Param("sentry.models.SentryApp")
  21     name = Param(six.string_types, required=False)
  22     status = Param(six.string_types, required=False)
  23     scopes = Param(Iterable, required=False)
  24     events = Param(Iterable, required=False)
  25     webhook_url = Param(six.string_types, required=False)
  26     redirect_url = Param(six.string_types, required=False)
  27     is_alertable = Param(bool, required=False)
  28     verify_install = Param(bool, required=False)
  29     schema = Param(dict, required=False)
  30     overview = Param(six.string_types, required=False)
  31     allowed_origins = Param(Iterable, required=False)
  32     user = Param("sentry.models.User")
  33 
  34     def call(self):
  35         self._update_name()
  36         self._update_author()
  37         self._update_status()
  38         self._update_scopes()
  39         self._update_events()
  40         self._update_webhook_url()
  41         self._update_redirect_url()
  42         self._update_is_alertable()
  43         self._update_verify_install()
  44         self._update_overview()
  45         self._update_allowed_origins()
  46         self._update_schema()
  47         self._update_service_hooks()
  48         self.sentry_app.save()
  49         return self.sentry_app
  50 
  51     @if_param("name")
  52     def _update_name(self):
  53         self.sentry_app.name = self.name
  54 
  55     @if_param("author")
  56     def _update_author(self):
  57         self.sentry_app.author = self.author
  58 
  59     @if_param("status")
  60     def _update_status(self):
  61         if self.user.is_superuser:
  62             if self.status == SentryAppStatus.PUBLISHED_STR:
  63                 self.sentry_app.status = SentryAppStatus.PUBLISHED
  64                 self.sentry_app.date_published = timezone.now()
  65             if self.status == SentryAppStatus.UNPUBLISHED_STR:
  66                 self.sentry_app.status = SentryAppStatus.UNPUBLISHED
  67 
  68     @if_param("scopes")
  69     def _update_scopes(self):
  70         if (
  71             self.sentry_app.status == SentryAppStatus.PUBLISHED
  72             and self.sentry_app.scope_list != self.scopes
  73         ):
  74             raise APIError("Cannot update permissions on a published integration.")
  75         self.sentry_app.scope_list = self.scopes
  76         # update the scopes of active tokens tokens
  77         ApiToken.objects.filter(
  78             Q(expires_at__isnull=True) | Q(expires_at__gt=timezone.now()),
  79             application=self.sentry_app.application,
  80         ).update(scope_list=list(self.scopes))
  81 
  82     @if_param("events")
  83     def _update_events(self):
  84         for event in self.events:
  85             needed_scope = REQUIRED_EVENT_PERMISSIONS[event]
  86             if needed_scope not in self.sentry_app.scope_list:
  87                 raise APIError(
  88                     u"{} webhooks require the {} permission.".format(event, needed_scope)
  89                 )
  90 
  91         from sentry.mediators.service_hooks.creator import expand_events
  92 
  93         self.sentry_app.events = expand_events(self.events)
  94 
  95     def _update_service_hooks(self):
  96         hooks = ServiceHook.objects.filter(application=self.sentry_app.application)
  97         # sentry_app.webhook_url will be updated at this point
  98         webhook_url = self.sentry_app.webhook_url
  99         for hook in hooks:
 100             # update the url and events
 101             if webhook_url:
 102                 service_hooks.Updater.run(
 103                     service_hook=hook, events=self.sentry_app.events, url=webhook_url
 104                 )
 105             # if no url, then the service hook is no longer active in which case we need to delete it
 106             else:
 107                 service_hooks.Destroyer.run(service_hook=hook)
 108         # if we don't have hooks but we have a webhook url now, need to create it for an internal integration
 109         if webhook_url and self.sentry_app.is_internal and not hooks:
 110             installation = SentryAppInstallation.objects.get(sentry_app_id=self.sentry_app.id)
 111             service_hooks.Creator.run(
 112                 application=self.sentry_app.application,
 113                 actor=installation,
 114                 projects=[],
 115                 organization=self.sentry_app.owner,
 116                 events=self.sentry_app.events,
 117                 url=webhook_url,
 118             )

 132     @if_param("verify_install")
 133     def _update_verify_install(self):
 134         if self.sentry_app.is_internal and self.verify_install:
 135             raise APIError(u"Internal integrations cannot have verify_install=True.")
 136         self.sentry_app.verify_install = self.verify_install
 137 
 138     @if_param("overview")
 139     def _update_overview(self):
 140         self.sentry_app.overview = self.overview
 141 
 142     @if_param("allowed_origins")
 143     def _update_allowed_origins(self):
 144         self.sentry_app.application.allowed_origins = "\n".join(self.allowed_origins)
 145         self.sentry_app.application.save()
 146 
 147     @if_param("schema")
 148     def _update_schema(self):
 149         self.sentry_app.schema = self.schema
 150         self._delete_old_ui_components()
 151         self._create_ui_components()
 152 
 153     def _delete_old_ui_components(self):
 154         SentryAppComponent.objects.filter(sentry_app_id=self.sentry_app.id).delete()
 155 
 156     def _create_ui_components(self):
 157         for element in self.schema.get("elements", []):
 158             SentryAppComponent.objects.create(
 159                 type=element["type"], sentry_app_id=self.sentry_app.id, schema=element
 160             )
 161 
 162     def record_analytics(self):
 163         analytics.record(
 164             "sentry_app.updated", user_id=self.user.id, sentry_app=self.sentry_app.slug
 165         )
