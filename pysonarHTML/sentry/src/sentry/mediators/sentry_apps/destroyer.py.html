  10 class Destroyer(Mediator):

  12     request = Param("rest_framework.request.Request", required=False)
  13     user = Param("sentry.models.User")
  14 
  15     def call(self):
  16         self._destroy_api_application()

  18         self._destroy_proxy_user()
  19         self._destroy_sentry_app()
  20         return self.sentry_app

  22     def _destroy_sentry_app_installations(self):
  23         for install in self.sentry_app.installations.all():
  24             notify = False if self.sentry_app.is_internal else True
  25             sentry_app_installations.Destroyer.run(
  26                 install=install, user=self.sentry_app.proxy_user, notify=notify
  27             )
  28 
  29     def _destroy_api_application(self):
  30         self.sentry_app.application.delete()
  31 
  32     def _destroy_proxy_user(self):
  33         self.sentry_app.proxy_user.delete()
  34 
  35     def _destroy_sentry_app(self):
  36         self.sentry_app.delete()
  37 
  38     def audit(self):
  39         if self.request:
  40             create_audit_entry(
  41                 request=self.request,
  42                 organization=self.sentry_app.owner,
  43                 target_object=self.sentry_app.owner.id,
  44                 event=AuditLogEntryEvent.SENTRY_APP_REMOVE,
  45                 data={"sentry_app": self.sentry_app.name},
  46             )
  47 
  48     def record_analytics(self):
  49         analytics.record(
  50             "sentry_app.deleted",
  51             user_id=self.user.id,
  52             organization_id=self.sentry_app.owner.id,
  53             sentry_app=self.sentry_app.slug,
  54         )
