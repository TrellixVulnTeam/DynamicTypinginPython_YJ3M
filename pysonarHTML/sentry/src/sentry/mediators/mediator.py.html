  16 class Mediator(object):
  17     """
  18     Objects that encapsulate domain logic.
  19 
  20     Mediators provide a layer between User accessible components like Endpoints
  21     and the database. They encapsulate the logic necessary to create domain
  22     objects, including all dependant objects, cross-object validations, etc.
  23 
  24     Mediators are intended to be composable and make it obvious where a piece
  25     of domain logic resides.
  26 
  27     Invocation:
  28         Invoke Mediators through their ``run`` class method. This essentially
  29         just wraps ``__init__(**kwargs).call()`` with some useful stuff,
  30         namely a DB transaction.
  31 
  32         >>> Mediator.run(**kwargs)
  33 
  34     Declaration:
  35         Mediators should define two things - a set of ``Param``s and a ``call``
  36         function.
  37 
  38         >>> class Creator(Mediator):
  39         >>>     name = Param(six.binary_type)
  40         >>>
  41         >>>     def call(self):
  42         >>>         with self.log():
  43         >>>             Thing.objects.create(name=self.name)
  44         >>>
  45         >>> Creator.run(name='thing')
  46         >>>
  47 
  48     Transactions
  49         Mediators are automatically wrapped in a transaction. As long as you
  50         invoke them via their ``run`` class method, you don't need to worry
  51         about declaring one yourself.
  52 
  53     Naming & Organization Conventions
  54         Mediators are organized by domain object and describe some domain
  55         process relevant to that object. For example:
  56 
  57             sentry.mediators.sentry_apps.Creator
  58             sentry.mediators.sentry_apps.Deactivator
  59 
  60     Params:
  61         Mediators declare the params they need similarly to Models. On
  62         instantiation, the Mediator will validate using the ``**kwargs``
  63         passed in.
  64 
  65         >>> from sentry.mediators import Mediator, Param
  66         >>>
  67         >>> class Creator(Mediator):
  68         >>>     name = Param(six.binary_type, default='example')
  69         >>>     user = Param('sentry.models.user.User', none=True)
  70 
  71         See ``sentry.mediators.param`` for more in-depth docs.
  72 
  73     Interface:
  74         Mediators have two main functions you should be aware of.
  75 
  76         ``run``:
  77             Convenience function for ``__init__(**kwargs).call()``
  78 
  79         ``call``:
  80             Instance method where you should implement your logic.
  81 
  82         >>> class Creator(Mediator):
  83         >>>     name = Param(six.binary_type, default='example')
  84         >>>
  85         >>>     def call(self):
  86         >>>         Thing.objects.create(name=self.name)
  87 
  88     Logging:
  89         Mediators have a ``log`` function available to them that will write to
  90         a logger with a standardized name. The name will be the full module
  91         path and class of the Mediator.
  92 
  93         When invoked via ``run``, the Mediator will automaticallyt log the
  94         start and end of its run.
  95 
  96         >>> class Creator(Mediator):
  97         >>>     def call(self):
  98         >>>         self.log(at='step')
  99         >>>
 100         >>> Creator.run()
 101         18:14:26 [INFO] sentry.mediators.sentry_apps.creator.Creator:  (at=u'step')  # NOQA
 102 
 103     Measuring:
 104         Mediators will automatically log the start and finish time of
 105         execution. If it ends via an Exception, it will log that instead of the
 106         finish time.
 107 
 108         The exception logging it just meant to augment what you get in Sentry
 109         itself, not replace anything about it. It's sometimes useful to see the
 110         progression of a request through many Mediators.
 111 
 112         You can manually use ``log`` to either write a set of attributes to
 113         stdout
 114 
 115         >>> self.log(at='a-special-step')
 116         18:14:26 [INFO] sentry.mediators.things.creator.Creator:  (at=u'a-special-step')  # NOQA
 117 
 118         or as a generator do log the start and finish of a block of
 119         code.
 120 
 121         >>> with self.log():
 122         >>>     do_a_thing()
 123         18:14:26 [INFO] sentry.mediators.things.creator.Creator:  (at=u'start')  # NOQA
 124         18:14:27 [INFO] sentry.mediators.things.creator.Creator:  (at=u'finish', elapsed=1634)  # NOQA
 125     """
 126 
 127     # Have we processed the Param declarations yet. Should happen once per
 128     # class.
 129     _params_prepared = False
 130 
 131     @classmethod
 132     def _prepare_params(cls):
 133         if sentry.mediators.mediator.Mediator in cls.__bases__ and not cls._params_prepared:
 134             params = [(k, v) for k, v in six.iteritems(cls.__dict__) if isinstance(v, Param)]
 135             for name, param in params:
 136                 param.setup(cls, name)
 137             cls._params_prepared = True
 138 
 139     @classmethod
 140     def run(cls, *args, **kwargs):
 141         with transaction.atomic():
 142             obj = cls(*args, **kwargs)
 143 
 144             with obj.log():
 145                 result = obj.call()
 146                 obj.audit()
 147                 obj.record_analytics()
 148                 return result
 149 
 150     def __init__(self, *args, **kwargs):
 151         self.kwargs = kwargs
 152         self.logger = kwargs.get("logger", logging.getLogger(self._logging_name))
 153         self._prepare_params()
 154         self._validate_params(**kwargs)
 155 
 156     def audit(self):
 157         # used for creating audit log entries
 158         pass
 159 
 160     def record_analytics(self):
 161         # used to record data to Amplitude
 162         pass
 163 
 164     def call(self):
 165         raise NotImplementedError
 166 
 167     def log(self, **kwargs):
 168         if any(kwargs):
 169             extra = {}
 170             extra.update(self._default_logging)
 171             extra.update(self._logging_context)
 172             extra.update(kwargs)
 173             self.logger.info(None, extra=extra)
 174         else:
 175             return self._measured(self)
 176 
 177     def _validate_params(self, **kwargs):
 178         for name, param in six.iteritems(self._params):
 179             if param.is_required:
 180                 param.validate(self, name, self.__getattr__(name))
 181 
 182     def __getattr__(self, key):
 183         if key in self.kwargs:
 184             return self.kwargs.get(key)
 185 
 186         param = self._params.get(key)
 187 
 188         if param and param.has_default:
 189             return param.default(self)
 190 
 191         if param and not param.is_required and not param.has_default:
 192             return None
 193 
 194         try:
 195             return self.__getattribute__(key)
 196         except AttributeError:
 197             return None
 198 
 199     @property
 200     def _params(self):
 201         # These will be named ``_<name>`` on the class, so remove the ``_`` so
 202         # that it matches the name we'll be invoking on the Mediator instance.
 203         return dict(
 204             (k[1:], v) for k, v in six.iteritems(self.__class__.__dict__) if isinstance(v, Param)
 205         )
 206 
 207     @memoize
 208     def _logging_name(self):
 209         return ".".join([self.__class__.__module__, self.__class__.__name__])
 210 
 211     @property
 212     def _default_logging(self):
 213         from sentry.app import env
 214 
 215         if (
 216             not env.request
 217             or not hasattr(env.request, "resolver_match")
 218             or not hasattr(env.request.resolver_match, "kwargs")
 219         ):
 220             return {}
 221 
 222         request_params = env.request.resolver_match.kwargs
 223 
 224         return compact(
 225             {
 226                 "org": request_params.get("organization_slug"),
 227                 "team": request_params.get("team_slug"),
 228                 "project": request_params.get("project_slug"),
 229             }
 230         )
 231 
 232     @property
 233     def _logging_context(self):
 234         """
 235         Overwrite this function to add attributes to automatic log lines.
 236         """
 237         return {}
 238 
 239     @contextmanager
 240     def _measured(self, context):
 241         start = datetime.datetime.now()
 242         context.log(at="start")
 243 
 244         try:
 245             yield
 246         except Exception as e:
 247             context.log(at="exception", elapsed=self._milliseconds_since(start))
 248             raise e
 249 
 250         context.log(at="finish", elapsed=self._milliseconds_since(start))
 251 
 252     def _milliseconds_since(self, start):
 253         now = datetime.datetime.now()
 254         elapsed = now - start
 255         return (elapsed.seconds * 1000) + (elapsed.microseconds / 1000)
