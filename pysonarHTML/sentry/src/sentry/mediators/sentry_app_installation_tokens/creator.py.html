  21     def call(self):
  22         self._check_token_limit()

  27     def _check_token_limit(self):
  28         curr_count = SentryAppInstallationToken.objects.filter(
  29             sentry_app_installation=self.sentry_app_installation
  30         ).count()
  31         if curr_count >= INTERNAL_INTEGRATION_TOKEN_COUNT_MAX:
  32             raise ApiTokenLimitError(
  33                 "Cannot generate more than %d tokens for a single integration"
  34                 % INTERNAL_INTEGRATION_TOKEN_COUNT_MAX
  35             )
  36 
  37     def _create_api_token(self):
  38         self.api_token = ApiToken.objects.create(
  39             user=self.sentry_app.proxy_user,
  40             application_id=self.sentry_app.application.id,
  41             scope_list=self.sentry_app.scope_list,
  42             expires_at=self.expires_at,
  43         )
  44 
  45     def _create_sentry_app_installation_token(self):
  46         self.sentry_app_installation_token = SentryAppInstallationToken.objects.create(
  47             api_token=self.api_token, sentry_app_installation=self.sentry_app_installation
  48         )
  49 
  50     def audit(self):

  53         if self.request and self.generate_audit:

  56                 organization=self.organization,
  57                 target_object=self.api_token.id,
  58                 event=AuditLogEntryEvent.INTERNAL_INTEGRATION_ADD_TOKEN,
  59                 data={"sentry_app": self.sentry_app.name},

  62     def record_analytics(self):

  70             sentry_app=self.sentry_app.slug,
