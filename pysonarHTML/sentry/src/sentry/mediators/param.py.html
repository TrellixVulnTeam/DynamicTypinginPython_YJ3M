  10 class Param(object):
  11     """
  12     Argument declarations for Mediators.
  13 
  14     Params offer a way to validate the arguments passed to a Mediator as well
  15     as set defaults.
  16 
  17     Example Usage:
  18         >>> class Creator(Mediator):
  19         >>>     name = Param(six.binary_type, default='example')
  20         >>>
  21         >>> c = Creator(name='foo')
  22         >>> c.name
  23         'foo'
  24 
  25         >>> c = Creator()
  26         >>> c.name
  27         'example'
  28 
  29         >>> c = Creator(name=False)
  30         Traceback (most recent call last):
  31             ...
  32         TypeError: `name` must be a <type 'six.binary_type'>
  33 
  34     Type Validation:
  35         When a Mediator is instantiated, it validates each of it's Params. This
  36         mainly checks that the type of object passed in matches what we
  37         expected.
  38 
  39         >>> class Creator(Mediator):
  40         >>>     name = Param(six.binary_type)
  41         >>>
  42         >>> c = Creator(name=False)
  43         Traceback (most recent call last):
  44             ...
  45         TypeError: `name` must be a <type 'six.binary_type'>
  46 
  47     Presence Validation:
  48         Without specifying otherwise, Params are assumed to be required. If
  49         it's okay for specific param to be None or not passed at all, you can
  50         do so by declaring ``required=False``.
  51 
  52         >>> class Creator(Mediator):
  53         >>>     size = Param(int, required=False)
  54         >>>
  55         >>> c = Creator()
  56         >>> c.size
  57         None
  58 
  59     Default Value:
  60         You can set a default value using the ``default`` argument. Default
  61         values can be static ones like an int, string, etc. that get evaluated
  62         at import. Or they can be a ``lambda`` that gets evaluated when the
  63         Mediator is instantiated.
  64 
  65         Declaration order DOES matter.
  66 
  67         >>> class Creator(Mediator):
  68         >>>     name = Param(six.binary_type, default='Pete')
  69         >>>
  70         >>> c = Creator()
  71         >>> c.name
  72         'Pete'
  73 
  74         >>> class Creator(Mediator):
  75         >>>     user = Param(dict)
  76         >>>     name = Param(six.binary_type, default=lambda self: self.user['name'])
  77     """
  78 
  79     def __init__(self, type, **kwargs):
  80         self._type = type
  81         self.kwargs = kwargs
  82 
  83     def setup(self, target, name):
  84         delattr(target, name)
  85         setattr(target, u"_{}".format(name), self)
  86 
  87     def validate(self, target, name, value):
  88         """
  89         Ensure the value evaluated is present (when required) and of the
  90         correct type.
  91         """
  92         if value is None:
  93             value = self.default(target)
  94 
  95         if self._missing_value(value):
  96             raise AttributeError(u"Missing required param: `{}`".format(name))
  97 
  98         if self.is_required and not isinstance(value, self.type):
  99             raise TypeError(u"`{}` must be a {}, received {}".format(name, self.type, type(value)))
 100 
 101         return True
 102 
 103     def default(self, target):
 104         """
 105         Evaluated default value, when given.
 106         """
 107         default = value = self.kwargs.get("default")
 108 
 109         if self.is_lambda_default:
 110             value = default(target)
 111 
 112         return value
 113 
 114     @memoize
 115     def type(self):
 116         if isinstance(self._type, six.string_types):
 117             return self._eval_string_type()
 118         return self._type
 119 
 120     @memoize
 121     def has_default(self):
 122         return "default" in self.kwargs
 123 
 124     @memoize
 125     def is_lambda_default(self):
 126         return isinstance(self.kwargs.get("default"), types.LambdaType)
 127 
 128     @memoize
 129     def is_required(self):
 130         if self.kwargs.get("required") is False:
 131             return False
 132         return True
 133 
 134     def _eval_string_type(self):
 135         """
 136         Converts a class path in string form to the actual class object.
 137 
 138         Example:
 139             >>> self._type = 'sentry.models.Project'
 140             >>> self._eval_string_type()
 141             sentry.models.project.Project
 142         """
 143         mod, klass = self._type.rsplit(".", 1)
 144         return getattr(sys.modules[mod], klass)
 145 
 146     def _missing_value(self, value):
 147         return self.is_required and value is None and not self.has_default
 148

 150 def if_param(name):
 151     def _if_param(func):
 152         def wrapper(self, *args):
 153             if not hasattr(self, name) or getattr(self, name) is None:
 154                 return
 155             return func(self, *args)
 156 
 157         return wrapper
 158 
 159     return _if_param
