   3 from sentry.mediators import Mediator, Param
   4 from sentry.models import Repository
   5 from sentry.plugins.base import plugins
   6 from sentry.utils.cache import memoize

   9 class Migrator(Mediator):
  10     integration = Param("sentry.models.integration.Integration")
  11     organization = Param("sentry.models.organization.Organization")
  12 
  13     def call(self):
  14         for project in self.projects:
  15             for plugin in plugins.for_project(project):
  16                 if plugin.slug != self.integration.provider:
  17                     continue
  18 
  19                 if self.all_repos_migrated(plugin.slug):
  20                     # Since repos are Org-level, if they're all migrated, we
  21                     # can disable the Plugin for all Projects. There'd be no
  22                     # Repos left, associated with the Plugin.
  23                     self.disable_for_all_projects(plugin)
  24 
  25     def all_repos_migrated(self, provider):
  26         provider = "visualstudio" if provider == "vsts" else provider
  27 
  28         return all(r.integration_id is not None for r in self.repos_for_provider(provider))
  29 
  30     def disable_for_all_projects(self, plugin):
  31         for project in self.projects:
  32             try:
  33                 self.log(at="disable", project=project.slug, plugin=plugin.slug)
  34                 plugin.disable(project=project)
  35             except NotImplementedError:
  36                 pass
  37 
  38     def repos_for_provider(self, provider):
  39         return [r for r in self.repositories if r.provider == provider]
  40 
  41     @property
  42     def repositories(self):
  43         return Repository.objects.filter(organization_id=self.organization.id)
  44 
  45     @memoize
  46     def projects(self):
  47         return list(self.organization.project_set.all())
  48 
  49     @property
  50     def plugins(self):
  51         return [plugins.configurable_for_project(project) for project in self.projects]
  52 
  53     @property
  54     def _logging_context(self):
  55         return {
  56             "org": self.organization.slug,
  57             "integration_id": self.integration.id,
  58             "integration_provider": self.integration.provider,
  59         }
