  13 class Destroyer(Mediator):
  14     install = Param("sentry.models.SentryAppInstallation")
  15     user = Param("sentry.models.User")
  16     request = Param("rest_framework.request.Request", required=False)
  17     notify = Param(bool, default=True)
  18 
  19     @property
  20     def _logging_context(self):
  21         return {"install_id": self.install.id, "install_uuid": self.install.uuid}
  22 
  23     def call(self):
  24         self._destroy_grant()
  25         self._destroy_service_hooks()
  26         self._destroy_api_tokens()
  27         self._destroy_installation()
  28         return self.install
  29 
  30     def _destroy_grant(self):
  31         if self.install.api_grant_id:
  32             self.install.api_grant.delete()
  33 
  34     def _destroy_service_hooks(self):
  35         hooks = ServiceHook.objects.filter(
  36             application_id=self.install.sentry_app.application_id, actor_id=self.install.id
  37         )
  38         for hook in hooks:
  39             service_hooks.Destroyer.run(service_hook=hook)
  40 
  41     def _destroy_api_tokens(self):
  42         tokens = ApiToken.objects.filter(
  43             id__in=SentryAppInstallationToken.objects.filter(
  44                 sentry_app_installation_id=self.install.id
  45             ).values_list("api_token_id", flat=True)
  46         )
  47 
  48         for token in tokens:
  49             sentry_app_installation_tokens.Destroyer.run(
  50                 api_token=token, user=self.user, request=self.request
  51             )
  52 
  53     def _destroy_installation(self):
  54         if self.notify:
  55             try:
  56                 InstallationNotifier.run(install=self.install, user=self.user, action="deleted")
  57             # if the error is from a request exception, log the error and continue
  58             except RequestException as exc:
  59                 self.log(error=exc)
  60         self.install.delete()
  61 
  62     def audit(self):
  63         if self.request:
  64             create_audit_entry(
  65                 request=self.request,
  66                 organization=self.install.organization,
  67                 target_object=self.install.organization.id,
  68                 event=AuditLogEntryEvent.SENTRY_APP_UNINSTALL,
  69                 data={"sentry_app": self.install.sentry_app.name},
  70             )
  71 
  72     def record_analytics(self):
  73         analytics.record(
  74             "sentry_app.uninstalled",
  75             user_id=self.user.id,
  76             organization_id=self.install.organization_id,
  77             sentry_app=self.install.sentry_app.slug,
  78         )
