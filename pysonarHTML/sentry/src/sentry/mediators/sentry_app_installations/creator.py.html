   3 import six
   4 
   5 from sentry import analytics
   6 from sentry.constants import SentryAppInstallationStatus
   7 from sentry.mediators import Mediator, Param, service_hooks
   8 from sentry.models import AuditLogEntryEvent, ApiGrant, SentryApp, SentryAppInstallation
   9 from sentry.utils.cache import memoize
  10 from sentry.tasks.sentry_apps import installation_webhook
  11

  13 class Creator(Mediator):
  14     organization = Param("sentry.models.Organization")
  15     slug = Param(six.string_types)
  16     user = Param("sentry.models.User")
  17     request = Param("rest_framework.request.Request", required=False)
  18     notify = Param(bool, default=True)
  19 
  20     def call(self):
  21         self._create_api_grant()
  22         self._create_install()
  23         self._create_service_hooks()
  24         self._notify_service()
  25         self.install.is_new = True
  26         return self.install
  27 
  28     def _create_install(self):
  29         status = SentryAppInstallationStatus.PENDING
  30         if not self.sentry_app.verify_install:
  31             status = SentryAppInstallationStatus.INSTALLED
  32 
  33         self.install = SentryAppInstallation.objects.create(
  34             organization_id=self.organization.id,
  35             sentry_app_id=self.sentry_app.id,
  36             api_grant_id=self.api_grant.id,
  37             status=status,
  38         )
  39 
  40     def _create_api_grant(self):
  41         self.api_grant = ApiGrant.objects.create(
  42             user_id=self.sentry_app.proxy_user.id, application_id=self.api_application.id
  43         )
  44 
  45     def _create_service_hooks(self):
  46         # only make the service hook if there is a webhook url
  47         if self.sentry_app.webhook_url:
  48             service_hooks.Creator.run(
  49                 application=self.api_application,
  50                 actor=self.install,
  51                 projects=[],
  52                 organization=self.organization,
  53                 events=self.sentry_app.events,
  54                 url=self.sentry_app.webhook_url,
  55             )
  56 
  57     def _notify_service(self):
  58         if self.notify:
  59             installation_webhook.delay(self.install.id, self.user.id)
  60 
  61     def audit(self):
  62         from sentry.utils.audit import create_audit_entry
  63 
  64         if self.request:
  65             create_audit_entry(
  66                 request=self.request,
  67                 organization=self.install.organization,
  68                 target_object=self.install.organization.id,
  69                 event=AuditLogEntryEvent.SENTRY_APP_INSTALL,
  70                 data={"sentry_app": self.sentry_app.name},
  71             )
  72 
  73     def record_analytics(self):
  74         analytics.record(
  75             "sentry_app.installed",
  76             user_id=self.user.id,
  77             organization_id=self.organization.id,
  78             sentry_app=self.slug,
  79         )
  80 
  81     @memoize
  82     def api_application(self):
  83         return self.sentry_app.application
  84 
  85     @memoize
  86     def sentry_app(self):
  87         return SentryApp.objects.get(slug=self.slug)
