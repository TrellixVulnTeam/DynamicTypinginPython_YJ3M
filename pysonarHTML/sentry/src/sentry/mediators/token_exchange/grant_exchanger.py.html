  17 class GrantExchanger(Mediator):
  18     """
  19     Exchanges a Grant Code for an Access Token
  20     """
  21 
  22     install = Param("sentry.models.SentryAppInstallation")
  23     code = Param(six.string_types)
  24     client_id = Param(six.string_types)
  25     user = Param("sentry.models.User")
  26 
  27     def call(self):
  28         self._validate()
  29         self._create_token()
  30 
  31         # Once it's exchanged it's no longer valid and should not be
  32         # exchangable, so we delete it.
  33         self._delete_grant()
  34 
  35         return self.token
  36 
  37     def record_analytics(self):
  38         analytics.record(
  39             "sentry_app.token_exchanged",
  40             sentry_app_installation_id=self.install.id,
  41             exchange_type="authorization",
  42         )
  43 
  44     def _validate(self):
  45         Validator.run(install=self.install, client_id=self.client_id, user=self.user)
  46 
  47         if not self._grant_belongs_to_install() or not self._sentry_app_user_owns_grant():
  48             raise APIUnauthorized
  49 
  50         if not self._grant_is_active():
  51             raise APIUnauthorized("Grant has already expired.")
  52 
  53     def _grant_belongs_to_install(self):
  54         return self.grant.sentry_app_installation == self.install
  55 
  56     def _sentry_app_user_owns_grant(self):
  57         return self.grant.application.owner == self.user
  58 
  59     def _grant_is_active(self):
  60         return self.grant.expires_at > datetime.now(pytz.UTC)
  61 
  62     def _delete_grant(self):
  63         self.grant.delete()
  64 
  65     def _create_token(self):
  66         self.token = ApiToken.objects.create(
  67             user=self.user,
  68             application=self.application,
  69             scope_list=self.sentry_app.scope_list,
  70             expires_at=token_expiration(),
  71         )
  72         self.install.api_token = self.token
  73         self.install.save()
  74 
  75     @memoize
  76     def grant(self):
  77         try:
  78             return (
  79                 ApiGrant.objects.select_related("sentry_app_installation")
  80                 .select_related("application")
  81                 .select_related("application__sentry_app")
  82                 .get(code=self.code)
  83             )
  84         except ApiGrant.DoesNotExist:
  85             raise APIUnauthorized
  86 
  87     @property
  88     def application(self):
  89         try:
  90             return self.grant.application
  91         except ApiApplication.DoesNotExist:
  92             raise APIUnauthorized
  93 
  94     @property
  95     def sentry_app(self):
  96         try:
  97             return self.application.sentry_app
  98         except SentryApp.DoesNotExist:
  99             raise APIUnauthorized
