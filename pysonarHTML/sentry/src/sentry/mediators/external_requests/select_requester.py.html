  18 class SelectRequester(Mediator):
  19     """
  20     1. Makes a GET request to another service to fetch data needed to populate
  21     the SelectField dropdown in the UI.
  22 
  23     `installationId` and `project` are included in the params of the request
  24 
  25     2. Validates and formats the response.
  26     """
  27 
  28     install = Param("sentry.models.SentryAppInstallation")
  29     project = Param("sentry.models.Project", required=False)
  30     uri = Param(six.string_types)
  31     query = Param(six.string_types, required=False)
  32     dependent_data = Param(six.string_types, required=False)
  33 
  34     def call(self):
  35         return self._make_request()
  36 
  37     def _build_url(self):
  38         urlparts = list(urlparse(self.sentry_app.webhook_url))
  39         urlparts[2] = self.uri
  40 
  41         query = {"installationId": self.install.uuid}
  42 
  43         if self.project:
  44             query["projectSlug"] = self.project.slug
  45 
  46         if self.query:
  47             query["query"] = self.query
  48 
  49         if self.dependent_data:
  50             query["dependentData"] = self.dependent_data
  51 
  52         urlparts[4] = urlencode(query)
  53         return urlunparse(urlparts)
  54 
  55     def _make_request(self):
  56         try:
  57             body = safe_urlread(
  58                 send_and_save_sentry_app_request(
  59                     self._build_url(),
  60                     self.sentry_app,
  61                     self.install.organization_id,
  62                     "select_options.requested",
  63                     headers=self._build_headers(),
  64                 )
  65             )
  66 
  67             response = json.loads(body)
  68         except Exception as e:
  69             logger.info(
  70                 "select-requester.error",
  71                 extra={
  72                     "sentry_app": self.sentry_app.slug,
  73                     "install": self.install.uuid,
  74                     "project": self.project and self.project.slug,
  75                     "uri": self.uri,
  76                     "error_message": six.text_type(e),
  77                 },
  78             )
  79             response = {}
  80 
  81         if not self._validate_response(response):
  82             raise APIError()
  83 
  84         return self._format_response(response)
  85 
  86     def _validate_response(self, resp):
  87         return validate(instance=resp, schema_type="select")
  88 
  89     def _format_response(self, resp):
  90         # the UI expects the following form:
  91         # choices: [[label, value]]
  92         # default: [label, value]
  93         response = {}
  94         choices = []
  95 
  96         for option in resp:
  97             choices.append([option["value"], option["label"]])
  98             if option.get("default"):
  99                 response["defaultValue"] = option["value"]
 100 
 101         response["choices"] = choices
 102         return response
 103 
 104     def _build_headers(self):
 105         request_uuid = uuid4().hex
 106 
 107         return {
 108             "Content-Type": "application/json",
 109             "Request-ID": request_uuid,
 110             "Sentry-App-Signature": self.sentry_app.build_signature(""),
 111         }
 112 
 113     @memoize
 114     def sentry_app(self):
 115         return self.install.sentry_app
