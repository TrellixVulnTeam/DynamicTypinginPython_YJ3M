  18 class IssueLinkRequester(Mediator):
  19     """
  20     1. Makes a POST request to another service with data used for creating or
  21     linking a Sentry issue to an issue in the other service.
  22 
  23     The data sent to the other service is always in the following format:
  24         {
  25             'installationId': <install_uuid>,
  26             'issueId': <sentry_group_id>,
  27             'webUrl': <sentry_group_web_url>,
  28             <fields>,
  29         }
  30 
  31     <fields> are any of the 'create' or 'link' form fields (determined by
  32     the schema for that particular service)
  33 
  34     2. Validates the response format from the other service and returns the
  35     payload.
  36 
  37     The data sent to the other service is always in the following format:
  38         {
  39             'identifier': <some_identifier>,
  40             'webUrl': <external_issue_web_url>,
  41             'project': <top_level_identifier>,
  42         }
  43 
  44     The project and identifier are use to generate the display text for the linked
  45     issue in the UI (i.e. <project>#<identifier>)
  46     """
  47 
  48     install = Param("sentry.models.SentryAppInstallation")
  49     uri = Param(six.string_types)
  50     group = Param("sentry.models.Group")
  51     fields = Param(object)
  52     user = Param("sentry.models.User")
  53     action = Param(six.string_types)
  54 
  55     def call(self):
  56         return self._make_request()
  57 
  58     def _build_url(self):
  59         urlparts = urlparse(self.sentry_app.webhook_url)
  60         return u"{}://{}{}".format(urlparts.scheme, urlparts.netloc, self.uri)
  61 
  62     def _make_request(self):
  63         action_to_past_tense = {"create": "created", "link": "linked"}
  64 
  65         try:
  66             req = send_and_save_sentry_app_request(
  67                 self._build_url(),
  68                 self.sentry_app,
  69                 self.install.organization_id,
  70                 "external_issue.{}".format(action_to_past_tense[self.action]),
  71                 headers=self._build_headers(),
  72                 method="POST",
  73                 data=self.body,
  74             )
  75             body = safe_urlread(req)
  76             response = json.loads(body)
  77         except Exception as e:
  78             logger.info(
  79                 "issue-link-requester.error",
  80                 extra={
  81                     "sentry_app": self.sentry_app.slug,
  82                     "install": self.install.uuid,
  83                     "project": self.group.project.slug,
  84                     "group": self.group.id,
  85                     "uri": self.uri,
  86                     "error_message": six.text_type(e),
  87                 },
  88             )
  89             response = {}
  90 
  91         if not self._validate_response(response):
  92             raise APIError()
  93 
  94         return response
  95 
  96     def _validate_response(self, resp):
  97         return validate(instance=resp, schema_type="issue_link")
  98 
  99     def _build_headers(self):
 100         request_uuid = uuid4().hex
 101 
 102         return {
 103             "Content-Type": "application/json",
 104             "Request-ID": request_uuid,
 105             "Sentry-App-Signature": self.sentry_app.build_signature(self.body),
 106         }
 107 
 108     @memoize
 109     def body(self):
 110         body = {"fields": {}}
 111         for name, value in six.iteritems(self.fields):
 112             body["fields"][name] = value
 113 
 114         body["issueId"] = self.group.id
 115         body["installationId"] = self.install.uuid
 116         body["webUrl"] = self.group.get_absolute_url()
 117         project = self.group.project
 118         body["project"] = {"slug": project.slug, "id": project.id}
 119         body["actor"] = {"type": "user", "id": self.user.id, "name": self.user.name}
 120         return json.dumps(body)
 121 
 122     @memoize
 123     def sentry_app(self):
 124         return self.install.sentry_app
