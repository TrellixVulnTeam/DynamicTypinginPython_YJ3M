  19 _version_regexp = re.compile(r"^\d+\.\d+\.\d+$")  # We really only want stable releases
  20 LOADER_FOLDER = os.path.abspath(os.path.join(os.path.dirname(sentry.__file__), "loader"))

  23 @functools.lru_cache(maxsize=10)
  24 def load_registry(path):
  25     if "/" in path:
  26         return None
  27     fn = os.path.join(LOADER_FOLDER, path + ".json")
  28     try:
  29         with open(fn, "rb") as f:
  30             return json.load(f)
  31     except IOError:
  32         return None

  35 def get_highest_browser_sdk_version(versions):
  36     full_versions = [x for x in versions if _version_regexp.match(x)]
  37     return (
  38         six.binary_type(max(map(parse_version, full_versions)))
  39         if full_versions
  40         else settings.JS_SDK_LOADER_SDK_VERSION
  41     )

  62 def get_highest_selected_browser_sdk_version(selected_version):
  63     versions = load_version_from_file()
  64     if selected_version == "latest":
  65         return get_highest_browser_sdk_version(versions)
  66     return get_highest_browser_sdk_version(
  67         [x for x in versions if x.startswith(selected_version[0])]
