   6 class BaseVariant(object):
   7     # The type of the variant that is reported to the UI.
   8     type = None
   9 
  10     # This is true if `get_hash` does not return `None`.
  11     contributes = True
  12 
  13     def get_hash(self):
  14         return None
  15 
  16     @property
  17     def description(self):
  18         return self.type
  19 
  20     def _get_metadata_as_dict(self):
  21         return {}
  22 
  23     def as_dict(self):
  24         rv = {"type": self.type, "description": self.description, "hash": self.get_hash()}
  25         rv.update(self._get_metadata_as_dict())
  26         return rv
  27 
  28     def __repr__(self):
  29         return "<%s %r (%s)>" % (self.__class__.__name__, self.get_hash(), self.type)

 104 class SaltedComponentVariant(ComponentVariant):
 105     """A salted version of a component."""
 106 
 107     type = "salted-component"
 108 
 109     def __init__(self, values, component, config):
 110         ComponentVariant.__init__(self, component, config)
 111         self.values = values
 112 
 113     @property
 114     def description(self):
 115         return "modified " + self.component.description
 116 
 117     def get_hash(self):
 118         if not self.component.contributes:
 119             return None
 120         final_values = []
 121         for value in self.values:
 122             if value in DEFAULT_FINGERPRINT_VALUES:
 123                 final_values.extend(self.component.iter_values())
 124             else:
 125                 final_values.append(value)
 126         return hash_from_values(final_values)
 127 
 128     def _get_metadata_as_dict(self):
 129         rv = ComponentVariant._get_metadata_as_dict(self)
 130         rv["values"] = self.values
 131         return rv
