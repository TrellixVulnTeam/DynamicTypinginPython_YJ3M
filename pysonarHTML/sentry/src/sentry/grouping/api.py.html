  30 def get_grouping_config_dict_for_project(project, silent=True):
  31     """Fetches all the information necessary for grouping from the project
  32     settings.  The return value of this is persisted with the event on
  33     ingestion so that the grouping algorithm can be re-run later.
  34 
  35     This is called early on in normalization so that everything that is needed
  36     to group the project is pulled into the event.
  37     """
  38     config_id = project.get_option("sentry:grouping_config", validate=lambda x: x in CONFIGURATIONS)
  39 
  40     # At a later point we might want to store additional information here
  41     # such as frames that mark the end of a stacktrace and more.
  42     return {"id": config_id, "enhancements": _get_project_enhancements_config(project)}

  50 def _get_project_enhancements_config(project):
  51     enhancements = project.get_option("sentry:grouping_enhancements")
  52     enhancements_base = project.get_option(
  53         "sentry:grouping_enhancements_base", validate=lambda x: x in ENHANCEMENT_BASES
  54     )
  55 
  56     # Instead of parsing and dumping out config here, we can make a
  57     # shortcut
  58     from sentry.utils.cache import cache
  59     from sentry.utils.hashlib import md5_text
  60 
  61     cache_key = (
  62         "grouping-enhancements:" + md5_text("%s|%s" % (enhancements_base, enhancements)).hexdigest()
  63     )
  64     rv = cache.get(cache_key)
  65     if rv is not None:
  66         return rv
  67 
  68     try:
  69         rv = Enhancements.from_config_string(enhancements, bases=[enhancements_base]).dumps()
  70     except InvalidEnhancerConfig:
  71         rv = get_default_enhancements()
  72     cache.set(cache_key, rv)
  73     return rv

  91 def load_grouping_config(config_dict=None):
  92     """Loads the given grouping config."""
  93     if config_dict is None:
  94         config_dict = get_default_grouping_config_dict()
  95     elif "id" not in config_dict:
  96         raise ValueError("Malformed configuration dictionary")
  97     config_dict = dict(config_dict)
  98     config_id = config_dict.pop("id")
  99     if config_id not in CONFIGURATIONS:
 100         raise GroupingConfigNotFound(config_id)
 101     return CONFIGURATIONS[config_id](**config_dict)

 108 def get_fingerprinting_config_for_project(project):
 109     from sentry.grouping.fingerprinting import FingerprintingRules, InvalidFingerprintingConfig
 110 
 111     rules = project.get_option("sentry:fingerprinting_rules")
 112     if not rules:
 113         return FingerprintingRules([])
 114 
 115     from sentry.utils.cache import cache
 116     from sentry.utils.hashlib import md5_text
 117 
 118     cache_key = "fingerprinting-rules:" + md5_text(rules).hexdigest()
 119     rv = cache.get(cache_key)
 120     if rv is not None:
 121         return FingerprintingRules.from_json(rv)
 122 
 123     try:
 124         rv = FingerprintingRules.from_config_string(rules)
 125     except InvalidFingerprintingConfig:
 126         rv = FingerprintingRules([])
 127     cache.set(cache_key, rv.to_json())
 128     return rv

 137 def _get_calculated_grouping_variants_for_event(event, config):
 138     winning_strategy = None
 139     precedence_hint = None
 140     per_variant_components = {}
 141 
 142     for strategy in config.iter_strategies():
 143         rv = strategy.get_grouping_component_variants(event, config=config)
 144         for (variant, component) in six.iteritems(rv):
 145             per_variant_components.setdefault(variant, []).append(component)
 146 
 147             if winning_strategy is None:
 148                 if component.contributes:
 149                     winning_strategy = strategy.name
 150                     precedence_hint = "%s takes precedence" % (
 151                         "%s of %s" % (strategy.name, variant)
 152                         if variant != "default"
 153                         else strategy.name
 154                     )
 155             elif component.contributes and winning_strategy != strategy.name:
 156                 component.update(contributes=False, hint=precedence_hint)
 157 
 158     rv = {}
 159     for (variant, components) in six.iteritems(per_variant_components):
 160         component = GroupingComponent(id=variant, values=components)
 161         if not component.contributes and precedence_hint:
 162             component.update(hint=precedence_hint)
 163         rv[variant] = component
 164 
 165     return rv

 168 def get_grouping_variants_for_event(event, config=None):
 169     """Returns a dict of all grouping variants for this event."""
 170     # If a checksum is set the only variant that comes back from this
 171     # event is the checksum variant.
 172     checksum = event.data.get("checksum")
 173     if checksum:
 174         if HASH_RE.match(checksum):
 175             return {"checksum": ChecksumVariant(checksum)}
 176 
 177         rv = {
 178             "hashed-checksum": ChecksumVariant(hash_from_values(checksum), hashed=True),
 179         }
 180 
 181         # The legacy code path also supported arbitrary values here but
 182         # it will blow up if it results in more than 32 bytes of data
 183         # as this cannot be inserted into the database.  (See GroupHash.hash)
 184         if len(checksum) <= 32:
 185             rv["checksum"] = ChecksumVariant(checksum)
 186 
 187         return rv
 188 
 189     # Otherwise we go to the various forms of fingerprint handling.
 190     fingerprint = event.data.get("fingerprint") or ["{{ default }}"]
 191     defaults_referenced = sum(1 if d in DEFAULT_FINGERPRINT_VALUES else 0 for d in fingerprint)
 192 
 193     # If no defaults are referenced we produce a single completely custom
 194     # fingerprint.
 195     if defaults_referenced == 0:
 196         fingerprint = resolve_fingerprint_values(fingerprint, event)
 197         return {"custom-fingerprint": CustomFingerprintVariant(fingerprint)}
 198 
 199     if config is None:
 200         config = load_default_grouping_config()
 201 
 202     # At this point we need to calculate the default event values.  If the
 203     # fingerprint is salted we will wrap it.
 204     components = _get_calculated_grouping_variants_for_event(event, config)
 205     rv = {}
 206 
 207     # If the fingerprints are unsalted, we can return them right away.
 208     if defaults_referenced == 1 and len(fingerprint) == 1:
 209         for (key, component) in six.iteritems(components):
 210             rv[key] = ComponentVariant(component, config)
 211 
 212     # Otherwise we need to salt each of the components.
 213     else:
 214         fingerprint = resolve_fingerprint_values(fingerprint, event)
 215         for (key, component) in six.iteritems(components):
 216             rv[key] = SaltedComponentVariant(fingerprint, component, config)
 217 
 218     # Ensure we have a fallback hash if nothing else works out
 219     if not any(x.contributes for x in six.itervalues(rv)):
 220         rv["fallback"] = FallbackVariant()
 221 
 222     return rv
