  18 # Grammar is defined in EBNF syntax.
  19 fingerprinting_grammar = Grammar(
  20     r"""
  21 
  22 fingerprinting_rules = line+
  23 
  24 line = _ (comment / rule / empty) newline?
  25 
  26 rule = _ matchers _ follow _ fingerprint
  27 
  28 matchers       = matcher+
  29 matcher        = _ matcher_type sep argument
  30 matcher_type   = "path" / "function" / "module" / "family" / "type" / "value" / "message" / "package" / "app"
  31 argument       = quoted / unquoted
  32 
  33 fingerprint    = fp_value+
  34 fp_value        = _ fp_argument _ ","?
  35 fp_argument    = quoted / unquoted_no_comma
  36 
  37 comment        = ~r"#[^\r\n]*"
  38 
  39 quoted         = ~r'"([^"\\]*(?:\\.[^"\\]*)*)"'
  40 unquoted       = ~r"\S+"
  41 unquoted_no_comma = ~r"((?:\{\{\s*\S+\s*\}\})|(?:[^\s,]+))"
  42 
  43 follow  = "->"
  44 sep     = ":"
  45 space   = " "
  46 empty   = ""
  47 newline = ~r"[\r\n]"
  48 _       = space*
  49 
  50 """
  51 )

  58 class EventAccess(object):
  59     def __init__(self, event):
  60         self.event = event
  61         self._exceptions = None
  62         self._frames = None
  63         self._messages = None
  64 
  65     def get_messages(self):
  66         if self._messages is None:
  67             self._messages = []
  68             message = get_path(self.event, "logentry", "formatted", filter=True)
  69             if message:
  70                 self._messages.append(
  71                     {
  72                         "message": message,
  73                         "family": get_behavior_family_for_platform(self.event.get("platform")),
  74                     }
  75                 )
  76         return self._messages
  77 
  78     def get_exceptions(self):
  79         if self._exceptions is None:
  80             self._exceptions = []
  81             for exc in get_path(self.event, "exception", "values", filter=True) or ():
  82                 self._exceptions.append(
  83                     {
  84                         "type": exc.get("type"),
  85                         "value": exc.get("value"),
  86                         "family": get_behavior_family_for_platform(self.event.get("platform")),
  87                     }
  88                 )
  89         return self._exceptions
  90 
  91     def get_frames(self, with_functions=False):
  92         from sentry.stacktraces.functions import get_function_name_for_frame
  93 
  94         if self._frames is None:
  95             self._frames = []
  96 
  97             def _push_frame(frame):
  98                 platform = frame.get("platform") or self.event.get("platform")
  99                 func = get_function_name_for_frame(frame, platform)
 100                 self._frames.append(
 101                     {
 102                         "function": func or "<unknown>",
 103                         "path": frame.get("abs_path") or frame.get("filename"),
 104                         "module": frame.get("module"),
 105                         "family": get_behavior_family_for_platform(platform),
 106                         "package": frame.get("package"),
 107                         "app": frame.get("in_app"),
 108                     }
 109                 )
 110 
 111             have_errors = False
 112             for exc in get_path(self.event, "exception", "values", filter=True) or ():
 113                 for frame in get_path(exc, "stacktrace", "frames", filter=True) or ():
 114                     _push_frame(frame)
 115                 have_errors = True
 116 
 117             if not have_errors:
 118                 frames = get_path(self.event, "stacktrace", "frames", filter=True)
 119                 if not frames:
 120                     threads = get_path(self.event, "threads", "values", filter=True)
 121                     if threads and len(threads) == 1:
 122                         frames = get_path(threads, 0, "stacktrace", "frames")
 123                 for frame in frames or ():
 124                     _push_frame(frame)
 125 
 126         return self._frames
 127 
 128     def get_values(self, interface):
 129         if interface == "message":
 130             return self.get_messages()
 131         elif interface == "exception":
 132             return self.get_exceptions()
 133         elif interface == "frame":
 134             return self.get_frames()
 135         return []

 138 class FingerprintingRules(object):
 139     def __init__(self, rules, changelog=None, version=None):
 140         if version is None:
 141             version = VERSION
 142         self.version = version
 143         self.rules = rules
 144         self.changelog = changelog
 145 
 146     def iter_rules(self):
 147         return iter(self.rules)
 148 
 149     def get_fingerprint_values_for_event(self, event):
 150         if not self.rules:
 151             return
 152         access = EventAccess(event)
 153         for rule in self.iter_rules():
 154             new_values = rule.get_fingerprint_values_for_event_access(access)
 155             if new_values is not None:
 156                 return new_values
 157 
 158     @classmethod
 159     def _from_config_structure(cls, data):
 160         version = data["version"]
 161         if version != VERSION:
 162             raise ValueError("Unknown version")
 163         return cls(rules=[Rule._from_config_structure(x) for x in data["rules"]], version=version)
 164 
 165     def _to_config_structure(self):
 166         return {"version": self.version, "rules": [x._to_config_structure() for x in self.rules]}
 167 
 168     def to_json(self):
 169         return self._to_config_structure()
 170 
 171     @classmethod
 172     def from_json(cls, value):
 173         try:
 174             return cls._from_config_structure(value)
 175         except (LookupError, AttributeError, TypeError, ValueError) as e:
 176             raise ValueError("invalid fingerprinting config: %s" % e)
 177 
 178     @classmethod
 179     def from_config_string(self, s):
 180         try:
 181             tree = fingerprinting_grammar.parse(s)
 182         except ParseError as e:
 183             context = e.text[e.pos : e.pos + 33]
 184             if len(context) == 33:
 185                 context = context[:-1] + "..."
 186             raise InvalidFingerprintingConfig(
 187                 'Invalid syntax near "%s" (line %s, column %s)' % (context, e.line(), e.column())
 188             )
 189         return FingerprintingVisitor().visit(tree)

 192 class Match(object):
 193     def __init__(self, key, pattern):
 194         self.key = key
 195         self.pattern = pattern
 196 
 197     @property
 198     def interface(self):
 199         if self.key == "message":
 200             return "message"
 201         elif self.key in ("type", "value"):
 202             return "exception"
 203         return "frame"
 204 
 205     def matches_value(self, value):
 206         if value is None:
 207             return False
 208         if self.key in ("path", "package"):
 209             if glob_match(
 210                 value, self.pattern, ignorecase=True, doublestar=True, path_normalize=True
 211             ):
 212                 return True
 213             if not value.startswith("/") and glob_match(
 214                 "/" + value, self.pattern, ignorecase=True, doublestar=True, path_normalize=True
 215             ):
 216                 return True
 217         elif self.key == "family":
 218             flags = self.pattern.split(",")
 219             if "all" in flags or value in flags:
 220                 return True
 221         elif self.key == "app":
 222             ref_val = get_rule_bool(self.pattern)
 223             if ref_val is not None and ref_val == value:
 224                 return True
 225         elif glob_match(value, self.pattern, ignorecase=self.key in ("message", "value")):
 226             return True
 227         return False
 228 
 229     def _to_config_structure(self):
 230         return [self.key, self.pattern]
 231 
 232     @classmethod
 233     def _from_config_structure(cls, obj):
 234         return cls(obj[0], obj[1])

 237 class Rule(object):
 238     def __init__(self, matchers, fingerprint):
 239         self.matchers = matchers
 240         self.fingerprint = fingerprint
 241 
 242     def get_fingerprint_values_for_event_access(self, access):
 243         by_interface = {}
 244         for matcher in self.matchers:
 245             by_interface.setdefault(matcher.interface, []).append(matcher)
 246 
 247         for interface, matchers in six.iteritems(by_interface):
 248             for values in access.get_values(interface):
 249                 if all(x.matches_value(values.get(x.key)) for x in matchers):
 250                     break
 251             else:
 252                 return
 253 
 254         return self.fingerprint
 255 
 256     def _to_config_structure(self):
 257         return {
 258             "matchers": [x._to_config_structure() for x in self.matchers],
 259             "fingerprint": self.fingerprint,
 260         }
 261 
 262     @classmethod
 263     def _from_config_structure(cls, obj):
 264         return cls([Match._from_config_structure(x) for x in obj["matchers"]], obj["fingerprint"])

 267 class FingerprintingVisitor(NodeVisitor):
 268     visit_comment = visit_empty = lambda *a: None
 269 
 270     def visit_comment(self, node, children):
 271         return node.text
 272 
 273     def visit_fingerprinting_rules(self, node, children):
 274         changelog = []
 275         rules = []
 276         in_header = True
 277         for child in children:
 278             if isinstance(child, six.string_types):
 279                 if in_header and child[:2] == "##":
 280                     changelog.append(child[2:].rstrip())
 281                 else:
 282                     in_header = False
 283             elif child is not None:
 284                 rules.append(child)
 285                 in_header = False
 286         return FingerprintingRules(rules, inspect.cleandoc("\n".join(changelog)).rstrip() or None)
 287 
 288     def visit_line(self, node, children):
 289         _, line, _ = children
 290         comment_or_rule_or_empty = line[0]
 291         if comment_or_rule_or_empty:
 292             return comment_or_rule_or_empty
 293 
 294     def visit_rule(self, node, children):
 295         _, matcher, _, _, _, fingerprint = children
 296         return Rule(matcher, fingerprint)
 297 
 298     def visit_matcher(self, node, children):
 299         _, ty, _, argument = children
 300         return Match(ty, argument)
 301 
 302     def visit_matcher_type(self, node, children):
 303         return node.text
 304 
 305     def visit_argument(self, node, children):
 306         return children[0]
 307 
 308     visit_fp_argument = visit_argument
 309 
 310     def visit_fingerprint(self, node, children):
 311         return children
 312 
 313     def visit_fp_value(self, node, children):
 314         _, argument, _, _ = children
 315         return argument
 316 
 317     def visit_quoted(self, node, children):
 318         return node.text[1:-1].encode("ascii", "backslashreplace").decode("unicode-escape")
 319 
 320     def visit_unquoted(self, node, children):
 321         return node.text
 322 
 323     visit_unquoted_no_comma = visit_unquoted
 324 
 325     def generic_visit(self, node, children):
 326         return children
