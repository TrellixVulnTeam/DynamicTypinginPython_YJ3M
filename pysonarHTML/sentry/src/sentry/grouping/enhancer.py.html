  25 # Grammar is defined in EBNF syntax.
  26 enhancements_grammar = Grammar(
  27     r"""
  28 
  29 enhancements = line+
  30 
  31 line = _ (comment / rule / empty) newline?
  32 
  33 rule = _ matchers actions
  34 
  35 matchers         = matcher+
  36 matcher          = _ matcher_type sep argument
  37 matcher_type     = "path" / "function" / "module" / "family" / "package" / "app"
  38 
  39 actions          = action+
  40 action           = flag_action / var_action
  41 var_action       = _ var_name _ "=" _ expr
  42 var_name         = "max-frames" / "min-frames"
  43 flag_action      = _ range? flag flag_action_name
  44 flag_action_name = "group" / "app"
  45 flag             = "+" / "-"
  46 range            = "^" / "v"
  47 expr             = int
  48 int              = ~r"[0-9]+"
  49 
  50 comment          = ~r"#[^\r\n]*"
  51 
  52 argument         = quoted / unquoted
  53 quoted           = ~r'"([^"\\]*(?:\\.[^"\\]*)*)"'
  54 unquoted         = ~r"\S+"
  55 
  56 sep     = ":"
  57 space   = " "
  58 empty   = ""
  59 newline = ~r"[\r\n]"
  60 _       = space*
  61 
  62 """

  96 class Match(object):
  97     def __init__(self, key, pattern):
  98         self.key = key
  99         self.pattern = pattern
 100 
 101     @property
 102     def description(self):
 103         return "%s:%s" % (
 104             self.key,
 105             self.pattern.split() != [self.pattern] and '"%s"' % self.pattern or self.pattern,
 106         )
 107 
 108     def matches_frame(self, frame_data, platform):
 109         # Path matches are always case insensitive
 110         if self.key in ("path", "package"):
 111             if self.key == "package":
 112                 value = frame_data.get("package") or ""
 113             else:
 114                 value = frame_data.get("abs_path") or frame_data.get("filename") or ""
 115             if glob_match(
 116                 value, self.pattern, ignorecase=True, doublestar=True, path_normalize=True
 117             ):
 118                 return True
 119             if not value.startswith("/") and glob_match(
 120                 "/" + value, self.pattern, ignorecase=True, doublestar=True, path_normalize=True
 121             ):
 122                 return True
 123             return False
 124 
 125         # families need custom handling as well
 126         if self.key == "family":
 127             flags = self.pattern.split(",")
 128             if "all" in flags:
 129                 return True
 130             family = get_behavior_family_for_platform(frame_data.get("platform") or platform)
 131             return family in flags
 132 
 133         # in-app matching is just a bool
 134         if self.key == "app":
 135             ref_val = get_rule_bool(self.pattern)
 136             return ref_val is not None and ref_val == frame_data.get("in_app")
 137 
 138         # all other matches are case sensitive
 139         if self.key == "function":
 140             from sentry.stacktraces.functions import get_function_name_for_frame
 141 
 142             value = get_function_name_for_frame(frame_data, platform) or "<unknown>"
 143         elif self.key == "module":
 144             value = frame_data.get("module") or "<unknown>"
 145         else:
 146             # should not happen :)
 147             value = "<unknown>"
 148         return glob_match(value, self.pattern)
 149 
 150     def _to_config_structure(self):
 151         if self.key == "family":
 152             arg = "".join([_f for _f in [FAMILIES.get(x) for x in self.pattern.split(",")] if _f])
 153         elif self.key == "app":
 154             arg = {True: "1", False: "0"}.get(get_rule_bool(self.pattern), "")
 155         else:
 156             arg = self.pattern
 157         return MATCH_KEYS[self.key] + arg
 158 
 159     @classmethod
 160     def _from_config_structure(cls, obj):
 161         key = SHORT_MATCH_KEYS[obj[0]]
 162         if key == "family":
 163             arg = ",".join([_f for _f in [REVERSE_FAMILIES.get(x) for x in obj[1:]] if _f])
 164         else:
 165             arg = obj[1:]
 166         return cls(key, arg)

 179     @classmethod
 180     def _from_config_structure(cls, val):
 181         if isinstance(val, list):
 182             return VarAction(val[0], val[1])
 183         flag, range = REVERSE_ACTION_FLAGS[val >> 4]
 184         return FlagAction(ACTIONS[val & 0xF], flag, range)

 187 @implements_to_string
 188 class FlagAction(Action):
 189     def __init__(self, key, flag, range):
 190         self.key = key
 191         self.flag = flag
 192         self.range = range
 193 
 194     def __str__(self):
 195         return "%s%s%s" % (
 196             {"up": "^", "down": "v"}.get(self.range, ""),
 197             self.flag and "+" or "-",
 198             self.key,
 199         )
 200 
 201     def _to_config_structure(self):
 202         return ACTIONS.index(self.key) | (ACTION_FLAGS[self.flag, self.range] << 4)
 203 
 204     def _slice_to_range(self, seq, idx):
 205         if self.range is None:
 206             return [seq[idx]]
 207         elif self.range == "down":
 208             return seq[:idx]
 209         elif self.range == "up":
 210             return seq[idx + 1 :]
 211         return []
 212 
 213     def _in_app_changed(self, frame, component):
 214         orig_in_app = get_path(frame, "data", "orig_in_app")
 215 
 216         if orig_in_app is not None:
 217             if orig_in_app == -1:
 218                 orig_in_app = None
 219             return orig_in_app != frame.get("in_app")
 220         else:
 221             return self.flag == component.contributes
 222 
 223     def apply_modifications_to_frame(self, frames, idx):
 224         # Grouping is not stored on the frame
 225         if self.key == "group":
 226             return
 227         for frame in self._slice_to_range(frames, idx):
 228             if self.key == "app":
 229                 set_in_app(frame, self.flag)
 230 
 231     def update_frame_components_contributions(self, components, frames, idx, rule=None):
 232         rule_hint = "grouping enhancement rule"
 233         if rule:
 234             rule_hint = "%s (%s)" % (rule_hint, rule.matcher_description)
 235 
 236         sliced_components = self._slice_to_range(components, idx)
 237         sliced_frames = self._slice_to_range(frames, idx)
 238         for component, frame in zip(sliced_components, sliced_frames):
 239             if self.key == "group" and self.flag != component.contributes:
 240                 component.update(
 241                     contributes=self.flag,
 242                     hint="%s by %s" % (self.flag and "un-ignored" or "ignored", rule_hint),
 243                 )
 244             # The in app flag was set by `apply_modifications_to_frame`
 245             # but we want to add a hint if there is none yet.
 246             elif self.key == "app" and self._in_app_changed(frame, component):
 247                 component.update(
 248                     hint="marked %s by %s" % (self.flag and "in-app" or "out of app", rule_hint)
 249                 )

 270 class StacktraceState(object):
 271     def __init__(self):
 272         self.vars = {"max-frames": 0, "min-frames": 0}
 273         self.setters = {}
 274 
 275     def set(self, var, value, rule=None):
 276         self.vars[var] = value
 277         if rule is not None:
 278             self.setters[var] = rule
 279 
 280     def get(self, var):
 281         return self.vars.get(var)
 282 
 283     def describe_var_rule(self, var):
 284         rule = self.setters.get(var)
 285         if rule is not None:
 286             return rule.matcher_description
 287 
 288     def add_to_hint(self, hint, var):
 289         description = self.describe_var_rule(var)
 290         if description is None:
 291             return hint
 292         return "%s by grouping enhancement rule (%s)" % (hint, description)

 295 class Enhancements(object):
 296     def __init__(self, rules, changelog=None, version=None, bases=None, id=None):
 297         self.id = id
 298         self.rules = rules
 299         self.changelog = changelog
 300         if version is None:
 301             version = VERSION
 302         self.version = version
 303         if bases is None:
 304             bases = []
 305         self.bases = bases
 306 
 307     def apply_modifications_to_frame(self, frames, platform):
 308         """This applies the frame modifications to the frames itself.  This
 309         does not affect grouping.
 310         """
 311         for rule in self.iter_rules():
 312             for idx, frame in enumerate(frames):
 313                 actions = rule.get_matching_frame_actions(frame, platform)
 314                 for action in actions or ():
 315                     action.apply_modifications_to_frame(frames, idx)
 316 
 317     def update_frame_components_contributions(self, components, frames, platform):
 318         stacktrace_state = StacktraceState()
 319 
 320         # Apply direct frame actions and update the stack state alongside
 321         for rule in self.iter_rules():
 322             for idx, (component, frame) in enumerate(zip(components, frames)):
 323                 actions = rule.get_matching_frame_actions(frame, platform)
 324                 for action in actions or ():
 325                     action.update_frame_components_contributions(components, frames, idx, rule=rule)
 326                     action.modify_stacktrace_state(stacktrace_state, rule)
 327 
 328         # Use the stack state to update frame contributions again to trim
 329         # down to max-frames.  min-frames is handled on the other hand for
 330         # the entire stacktrace later.
 331         max_frames = stacktrace_state.get("max-frames")
 332         if max_frames > 0:
 333             ignored = 0
 334             for component in reversed(components):
 335                 if not component.contributes:
 336                     continue
 337                 ignored += 1
 338                 if ignored <= max_frames:
 339                     continue
 340                 hint = "ignored because only %d %s considered" % (
 341                     max_frames,
 342                     "frames are" if max_frames != 1 else "frame is",
 343                 )
 344                 hint = stacktrace_state.add_to_hint(hint, var="max-frames")
 345                 component.update(hint=hint, contributes=False)
 346 
 347         return stacktrace_state
 348 
 349     def assemble_stacktrace_component(self, components, frames, platform):
 350         """This assembles a stacktrace grouping component out of the given
 351         frame components and source frames.  Internally this invokes the
 352         `update_frame_components_contributions` method but also handles cases
 353         where the entire stacktrace should be discarded.
 354         """
 355         hint = None
 356         contributes = None
 357         stacktrace_state = self.update_frame_components_contributions(components, frames, platform)
 358 
 359         min_frames = stacktrace_state.get("min-frames")
 360         if min_frames > 0:
 361             total_contributes = sum(x.contributes for x in components)
 362             if 0 < total_contributes < min_frames:
 363                 hint = (
 364                     "discarded because stacktrace only contains %d "
 365                     "frame%s which is under the configured threshold"
 366                     % (total_contributes, "s" if total_contributes != 1 else "")
 367                 )
 368                 hint = stacktrace_state.add_to_hint(hint, var="min-frames")
 369                 contributes = False
 370 
 371         return GroupingComponent(
 372             id="stacktrace", values=components, hint=hint, contributes=contributes
 373         )
 374 
 375     def as_dict(self, with_rules=False):
 376         rv = {
 377             "id": self.id,
 378             "changelog": self.changelog,
 379             "bases": self.bases,
 380             "latest": projectoptions.lookup_well_known_key(
 381                 "sentry:grouping_enhancements_base"
 382             ).get_default(epoch=projectoptions.LATEST_EPOCH)
 383             == self.id,
 384         }
 385         if with_rules:
 386             rv["rules"] = [x.as_dict() for x in self.rules]
 387         return rv
 388 
 389     def _to_config_structure(self):
 390         return [self.version, self.bases, [x._to_config_structure() for x in self.rules]]
 391 
 392     def dumps(self):
 393         return (
 394             base64.urlsafe_b64encode(zlib.compress(msgpack.dumps(self._to_config_structure())))
 395             .decode("ascii")
 396             .strip(u"=")
 397         )
 398 
 399     def iter_rules(self):
 400         for base in self.bases:
 401             base = ENHANCEMENT_BASES.get(base)
 402             if base:
 403                 for rule in base.iter_rules():
 404                     yield rule
 405         for rule in self.rules:
 406             yield rule
 407 
 408     @classmethod
 409     def _from_config_structure(cls, data):
 410         version, bases, rules = data
 411         if version != VERSION:
 412             raise ValueError("Unknown version")
 413         return cls(
 414             rules=[Rule._from_config_structure(x) for x in rules], version=version, bases=bases
 415         )
 416 
 417     @classmethod
 418     def loads(cls, data):
 419         if isinstance(data, six.text_type):
 420             data = data.encode("ascii", "ignore")
 421         padded = data + b"=" * (4 - (len(data) % 4))
 422         try:
 423             return cls._from_config_structure(
 424                 msgpack.loads(zlib.decompress(base64.urlsafe_b64decode(padded)))
 425             )
 426         except (LookupError, AttributeError, TypeError, ValueError) as e:
 427             raise ValueError("invalid grouping enhancement config: %s" % e)
 428 
 429     @classmethod
 430     def from_config_string(self, s, bases=None, id=None):
 431         try:
 432             tree = enhancements_grammar.parse(s)
 433         except ParseError as e:
 434             context = e.text[e.pos : e.pos + 33]
 435             if len(context) == 33:
 436                 context = context[:-1] + "..."
 437             raise InvalidEnhancerConfig(
 438                 'Invalid syntax near "%s" (line %s, column %s)' % (context, e.line(), e.column())
 439             )
 440         return EnhancmentsVisitor(bases, id).visit(tree)

 443 class Rule(object):
 444     def __init__(self, matchers, actions):
 445         self.matchers = matchers
 446         self.actions = actions
 447 
 448     @property
 449     def matcher_description(self):
 450         rv = " ".join(x.description for x in self.matchers)
 451         for action in self.actions:
 452             rv = "%s %s" % (rv, action)
 453         return rv
 454 
 455     def as_dict(self):
 456         matchers = {}
 457         for matcher in self.matchers:
 458             matchers[matcher.key] = matcher.pattern
 459         return {"match": matchers, "actions": [six.text_type(x) for x in self.actions]}
 460 
 461     def get_matching_frame_actions(self, frame_data, platform):
 462         """Given a frame returns all the matching actions based on this rule.
 463         If the rule does not match `None` is returned.
 464         """
 465         if self.matchers and all(m.matches_frame(frame_data, platform) for m in self.matchers):
 466             return self.actions
 467 
 468     def _to_config_structure(self):
 469         return [
 470             [x._to_config_structure() for x in self.matchers],
 471             [x._to_config_structure() for x in self.actions],
 472         ]
 473 
 474     @classmethod
 475     def _from_config_structure(cls, tuple):
 476         return Rule(
 477             [Match._from_config_structure(x) for x in tuple[0]],
 478             [Action._from_config_structure(x) for x in tuple[1]],
 479         )

 482 class EnhancmentsVisitor(NodeVisitor):
 483     visit_comment = visit_empty = lambda *a: None
 484 
 485     def __init__(self, bases, id=None):
 486         self.bases = bases
 487         self.id = id
 488 
 489     def visit_comment(self, node, children):
 490         return node.text
 491 
 492     def visit_enhancements(self, node, children):
 493         changelog = []
 494         rules = []
 495         in_header = True
 496         for child in children:
 497             if isinstance(child, six.string_types):
 498                 if in_header and child[:2] == "##":
 499                     changelog.append(child[2:].rstrip())
 500                 else:
 501                     in_header = False
 502             elif child is not None:
 503                 rules.append(child)
 504                 in_header = False
 505         return Enhancements(
 506             rules,
 507             inspect.cleandoc("\n".join(changelog)).rstrip() or None,
 508             bases=self.bases,
 509             id=self.id,
 510         )
 511 
 512     def visit_line(self, node, children):
 513         _, line, _ = children
 514         comment_or_rule_or_empty = line[0]
 515         if comment_or_rule_or_empty:
 516             return comment_or_rule_or_empty
 517 
 518     def visit_rule(self, node, children):
 519         _, matcher, actions = children
 520         return Rule(matcher, actions)
 521 
 522     def visit_matcher(self, node, children):
 523         _, ty, _, argument = children
 524         return Match(ty, argument)
 525 
 526     def visit_matcher_type(self, node, children):
 527         return node.text
 528 
 529     def visit_argument(self, node, children):
 530         return children[0]
 531 
 532     def visit_var(self, node, children):
 533         _, var_name, _, _, _, arg = children
 534         return Action("set_var", (var_name, arg))
 535 
 536     def visit_action(self, node, children):
 537         return children[0]
 538 
 539     def visit_flag_action(self, node, children):
 540         _, rng, flag, action_name = children
 541         return FlagAction(action_name, flag, rng[0] if rng else None)
 542 
 543     def visit_flag_action_name(self, node, children):
 544         return node.text
 545 
 546     def visit_var_action(self, node, children):
 547         _, var_name, _, _, _, arg = children
 548         return VarAction(var_name, arg)
 549 
 550     def visit_var_name(self, node, children):
 551         return node.text
 552 
 553     def visit_flag(self, node, children):
 554         return node.text == "+"
 555 
 556     def visit_range(self, node, children):
 557         if node.text == "^":
 558             return "up"
 559         return "down"
 560 
 561     def visit_expr(self, node, children):
 562         return children[0]
 563 
 564     def visit_int(self, node, children):
 565         return int(node.text)
 566 
 567     def visit_quoted(self, node, children):
 568         return node.text[1:-1].encode("ascii", "backslashreplace").decode("unicode-escape")
 569 
 570     def visit_unquoted(self, node, children):
 571         return node.text
 572 
 573     def generic_visit(self, node, children):
 574         return children

 577 def _load_configs():
 578     rv = {}
 579     base = os.path.join(os.path.abspath(os.path.dirname(__file__)), "enhancement-configs")
 580     for fn in os.listdir(base):
 581         if fn.endswith(".txt"):
 582             with io.open(os.path.join(base, fn), "rt", encoding="utf-8") as f:
 583                 # We cannot use `:` in filenames on Windows but we already have ids with
 584                 # `:` in their names hence this trickery.
 585                 fn = fn.replace("@", ":")
 586                 rv[fn[:-4]] = Enhancements.from_config_string(f.read(), id=fn[:-4])
 587     return rv
