   8 # When a component ID appears here it has a human readable name which also
   9 # makes it a major component.  A major component is described as such for
  10 # the UI.
  11 KNOWN_MAJOR_COMPONENT_NAMES = {

  23 def _calculate_contributes(values):
  24     for value in values or ():
  25         if not isinstance(value, GroupingComponent) or value.contributes:
  26             return True
  27     return False

  30 class GroupingComponent(object):
  31     """A grouping component is a recursive structure that is flattened
  32     into components to make a hash for grouping purposes.
  33     """
  34 
  35     def __init__(self, id, hint=None, contributes=None, values=None):
  36         self.id = id
  37         if hint is None:
  38             hint = DEFAULT_HINTS.get(id)
  39         self.hint = hint
  40         if contributes is None:
  41             contributes = _calculate_contributes(values)
  42         self.contributes = contributes
  43         if values is None:
  44             values = []
  45         self.values = values
  46 
  47     @property
  48     def name(self):
  49         return KNOWN_MAJOR_COMPONENT_NAMES.get(self.id)
  50 
  51     @property
  52     def description(self):
  53         items = []
  54 
  55         def _walk_components(c, stack):
  56             stack.append(c.name)
  57             for value in c.values:
  58                 if isinstance(value, GroupingComponent) and value.contributes:
  59                     _walk_components(value, stack)
  60             parts = [_f for _f in stack if _f]
  61             items.append(parts)
  62             stack.pop()
  63 
  64         _walk_components(self, [])
  65         items.sort(key=lambda x: (len(x), x))
  66 
  67         if items and items[-1]:
  68             return " ".join(items[-1])
  69         return self.name or "others"
  70 
  71     def get_subcomponent(self, id, only_contributing=False):
  72         """Looks up a subcomponent by the id and returns the first or `None`."""
  73         return next(self.iter_subcomponents(id=id, only_contributing=only_contributing), None)
  74 
  75     def iter_subcomponents(self, id, recursive=False, only_contributing=False):
  76         """Finds all subcomponents matching an id, optionally recursively."""
  77         for value in self.values:
  78             if isinstance(value, GroupingComponent):
  79                 if only_contributing and not value.contributes:
  80                     continue
  81                 if value.id == id:
  82                     yield value
  83                 if recursive:
  84                     for subcomponent in value.iter_subcomponents(id, recursive=True):
  85                         yield subcomponent
  86 
  87     def update(self, hint=None, contributes=None, values=None):
  88         """Updates an already existing component with new values."""
  89         if hint is not None:
  90             self.hint = hint
  91         if values is not None:
  92             if contributes is None:
  93                 contributes = _calculate_contributes(values)
  94             self.values = values
  95         if contributes is not None:
  96             self.contributes = contributes
  97 
  98     def iter_values(self):
  99         """Recursively walks the component and flattens it into a list of
 100         values.
 101         """
 102         if self.contributes:
 103             for value in self.values:
 104                 if isinstance(value, GroupingComponent):
 105                     for x in value.iter_values():
 106                         yield x
 107                 else:
 108                     yield value
 109 
 110     def get_hash(self):
 111         """Returns the hash of the values if it contributes."""
 112         if self.contributes:
 113             return hash_from_values(self.iter_values())
 114 
 115     def as_dict(self):
 116         """Converts the component tree into a dictionary."""
 117         rv = {
 118             "id": self.id,
 119             "name": self.name,
 120             "contributes": self.contributes,
 121             "hint": self.hint,
 122             "values": [],
 123         }
 124         for value in self.values:
 125             if isinstance(value, GroupingComponent):
 126                 rv["values"].append(value.as_dict())
 127             else:
 128                 # this basically assumes that a value is only a primitive
 129                 # and never an object or list.  This should be okay
 130                 # because we verify this.
 131                 rv["values"].append(value)
 132         return rv
 133 
 134     def __repr__(self):
 135         return "GroupingComponent(%r, hint=%r, contributes=%r, values=%r)" % (
 136             self.id,
 137             self.hint,
 138             self.contributes,
 139             self.values,
 140         )
