  52 def is_recursion_v1(frame1, frame2):
  53     "Returns a boolean indicating whether frames are recursive calls."
  54     for field in RECURSION_COMPARISON_FIELDS:
  55         if getattr(frame1, field, None) != getattr(frame2, field, None):
  56             return False
  57 
  58     return True

  61 def get_filename_component(abs_path, filename, platform, allow_file_origin=False):
  62     """Attempt to normalize filenames by detecting special filenames and by
  63     using the basename only.
  64     """
  65     if filename is None:
  66         return GroupingComponent(id="filename")
  67 
  68     # Only use the platform independent basename for grouping and
  69     # lowercase it
  70     filename = _basename_re.split(filename)[-1].lower()
  71     filename_component = GroupingComponent(id="filename", values=[filename])
  72 
  73     if has_url_origin(abs_path, allow_file_origin=allow_file_origin):
  74         filename_component.update(contributes=False, hint="ignored because frame points to a URL")
  75     elif filename == "<anonymous>":
  76         filename_component.update(contributes=False, hint="anonymous filename discarded")
  77     elif filename == "[native code]":
  78         filename_component.update(contributes=False, hint="native code indicated by filename")
  79     elif platform == "java":
  80         new_filename = _java_assist_enhancer_re.sub(r"\1<auto>", filename)
  81         if new_filename != filename:
  82             filename_component.update(values=[new_filename], hint="cleaned javassist parts")
  83 
  84     return filename_component

  87 def get_module_component(abs_path, module, platform):
  88     """Given an absolute path, module and platform returns the module component
  89     with some necessary cleaning performed.
  90     """
  91     if module is None:
  92         return GroupingComponent(id="module")
  93 
  94     module_component = GroupingComponent(id="module", values=[module])
  95 
  96     if platform == "javascript" and "/" in module and abs_path and abs_path.endswith(module):
  97         module_component.update(contributes=False, hint="ignored bad javascript module")
  98     elif platform == "java":
  99         if "$$Lambda$" in module:
 100             module_component.update(contributes=False, hint="ignored java lambda")
 101         if module[:35] == "sun.reflect.GeneratedMethodAccessor":
 102             module_component.update(
 103                 values=["sun.reflect.GeneratedMethodAccessor"], hint="removed reflection marker"
 104             )
 105         elif module[:44] == "jdk.internal.reflect.GeneratedMethodAccessor":
 106             module_component.update(
 107                 values=["jdk.internal.reflect.GeneratedMethodAccessor"],
 108                 hint="removed reflection marker",
 109             )
 110         else:
 111             old_module = module
 112             module = _java_reflect_enhancer_re.sub(r"\1<auto>", module)
 113             module = _java_cglib_enhancer_re.sub(r"\1<auto>", module)
 114             module = _java_assist_enhancer_re.sub(r"\1<auto>", module)
 115             module = _clojure_enhancer_re.sub(r"\1<auto>", module)
 116             if module != old_module:
 117                 module_component.update(values=[module], hint="removed codegen marker")
 118 
 119     return module_component

 122 def get_function_component(
 123     function,
 124     platform,
 125     legacy_function_logic,
 126     sourcemap_used=False,
 127     context_line_available=False,
 128     raw_function=None,
 129     javascript_fuzzing=False,
 130     php_detect_anonymous_classes=False,
 131 ):
 132     """
 133     Attempt to normalize functions by removing common platform outliers.
 134 
 135     - Ruby generates (random?) integers for various anonymous style functions
 136       such as in erb and the active_support library.
 137     - Block functions have metadata that we don't care about.
 138 
 139     The `legacy_function_logic` parameter controls if the system should
 140     use the frame v1 function name logic or the frame v2 logic.  The difference
 141     is that v2 uses the function name consistently and v1 prefers raw function
 142     or a trimmed version (of the truncated one) for native.
 143     """
 144     from sentry.stacktraces.functions import trim_function_name
 145 
 146     behavior_family = get_behavior_family_for_platform(platform)
 147 
 148     if legacy_function_logic:
 149         func = raw_function or function
 150     else:
 151         func = function or raw_function
 152         if not raw_function and function:
 153             func = trim_function_name(func, platform)
 154 
 155     if not func:
 156         return GroupingComponent(id="function")
 157 
 158     function_component = GroupingComponent(id="function", values=[func])
 159 
 160     if platform == "ruby":
 161         if func.startswith("block "):
 162             function_component.update(values=["block"], hint="ruby block")
 163         else:
 164             new_function = _ruby_erb_func.sub("", func)
 165             if new_function != func:
 166                 function_component.update(
 167                     values=[new_function], hint="removed generated erb template suffix"
 168                 )
 169 
 170     elif platform == "php":
 171         if func.startswith(("[Anonymous", "class@anonymous\x00")):
 172             function_component.update(contributes=False, hint="ignored anonymous function")
 173         if php_detect_anonymous_classes and func.startswith("class@anonymous"):
 174             new_function = func.rsplit("::", 1)[-1]
 175             if new_function != func:
 176                 function_component.update(values=[new_function], hint="anonymous class method")
 177 
 178     elif platform == "java":
 179         if func.startswith("lambda$"):
 180             function_component.update(contributes=False, hint="ignored lambda function")
 181 
 182     elif behavior_family == "native":
 183         if func in ("<redacted>", "<unknown>"):
 184             function_component.update(contributes=False, hint="ignored unknown function")
 185         elif legacy_function_logic:
 186             new_function = trim_function_name(func, platform, normalize_lambdas=False)
 187             if new_function != func:
 188                 function_component.update(values=[new_function], hint="isolated function")
 189 
 190     elif javascript_fuzzing and behavior_family == "javascript":
 191         # This changes Object.foo or Foo.foo into foo so that we can
 192         # resolve some common cross browser differences
 193         new_function = func.rsplit(".", 1)[-1]
 194         if new_function != func:
 195             function_component.update(values=[new_function], hint="trimmed javascript function")
 196 
 197         # if a sourcemap was used for this frame and we know that we can
 198         # use the context line information we no longer want to use the
 199         # function name.  The reason for this is that function names in
 200         # sourcemaps are unreliable by the nature of sourcemaps and thus a
 201         # bad indicator for grouping.
 202         if sourcemap_used and context_line_available:
 203             function_component.update(
 204                 contributes=False, hint="ignored because sourcemap used and context line available"
 205             )
 206 
 207     return function_component

 251 def get_contextline_component(frame, platform, function, with_context_line_file_origin_bug=False):
 252     """Returns a contextline component.  The caller's responsibility is to
 253     make sure context lines are only used for platforms where we trust the
 254     quality of the sourcecode.  It does however protect against some bad
 255     JavaScript environments based on origin checks.
 256     """
 257     line = " ".join((frame.context_line or "").expandtabs(2).split())
 258     if not line:
 259         return GroupingComponent(id="context-line")
 260 
 261     component = GroupingComponent(id="context-line", values=[line])
 262     if line:
 263         if len(frame.context_line) > 120:
 264             component.update(hint="discarded because line too long", contributes=False)
 265         elif get_behavior_family_for_platform(platform) == "javascript":
 266             if with_context_line_file_origin_bug:
 267                 if has_url_origin(frame.abs_path, allow_file_origin=True):
 268                     component.update(hint="discarded because from URL origin", contributes=False)
 269             elif not function and has_url_origin(frame.abs_path):
 270                 component.update(
 271                     hint="discarded because from URL origin and no function", contributes=False
 272                 )
 273 
 274     return component

 277 def get_frame_component(
 278     frame,
 279     event,
 280     meta,
 281     legacy_function_logic=False,
 282     use_contextline=False,
 283     javascript_fuzzing=False,
 284     with_context_line_file_origin_bug=False,
 285     php_detect_anonymous_classes=False,
 286 ):
 287     platform = frame.platform or event.platform
 288 
 289     # Safari throws [native code] frames in for calls like ``forEach``
 290     # whereas Chrome ignores these. Let's remove it from the hashing algo
 291     # so that they're more likely to group together
 292     filename_component = get_filename_component(
 293         frame.abs_path, frame.filename, platform, allow_file_origin=javascript_fuzzing
 294     )
 295 
 296     # if we have a module we use that for grouping.  This will always
 297     # take precedence over the filename if it contributes
 298     module_component = get_module_component(frame.abs_path, frame.module, platform)
 299     if module_component.contributes and filename_component.contributes:
 300         filename_component.update(contributes=False, hint="module takes precedence")
 301 
 302     context_line_component = None
 303 
 304     # If we are allowed to use the contextline we add it now.
 305     if use_contextline:
 306         context_line_component = get_contextline_component(
 307             frame,
 308             platform,
 309             function=frame.function,
 310             with_context_line_file_origin_bug=with_context_line_file_origin_bug,
 311         )
 312 
 313     function_component = get_function_component(
 314         function=frame.function,
 315         raw_function=frame.raw_function,
 316         platform=platform,
 317         sourcemap_used=frame.data and frame.data.get("sourcemap") is not None,
 318         context_line_available=context_line_component and context_line_component.contributes,
 319         legacy_function_logic=legacy_function_logic,
 320         javascript_fuzzing=javascript_fuzzing,
 321         php_detect_anonymous_classes=php_detect_anonymous_classes,
 322     )
 323 
 324     values = [module_component, filename_component, function_component]
 325     if context_line_component is not None:
 326         values.append(context_line_component)
 327 
 328     rv = GroupingComponent(id="frame", values=values)
 329 
 330     # if we are in javascript fuzzing mode we want to disregard some
 331     # frames consistently.  These force common bad stacktraces together
 332     # to have a common hash at the cost of maybe skipping over frames that
 333     # would otherwise be useful.
 334     if javascript_fuzzing and get_behavior_family_for_platform(platform) == "javascript":
 335         func = frame.raw_function or frame.function
 336         if func:
 337             func = func.rsplit(".", 1)[-1]
 338         # special case empty functions not to have a hint
 339         if not func:
 340             function_component.update(contributes=False)
 341         elif func in (
 342             "?",
 343             "<anonymous function>",
 344             "<anonymous>",
 345             "Anonymous function",
 346         ) or func.endswith("/<"):
 347             function_component.update(contributes=False, hint="ignored unknown function name")
 348         if (func == "eval") or frame.abs_path in (
 349             "[native code]",
 350             "native code",
 351             "eval code",
 352             "<anonymous>",
 353         ):
 354             rv.update(contributes=False, hint="ignored low quality javascript frame")
 355 
 356     return rv

 369 def get_stacktrace_component(stacktrace, config, variant, meta):
 370     frames = stacktrace.frames
 371     all_frames_considered_in_app = False
 372 
 373     values = []
 374     prev_frame = None
 375     frames_for_filtering = []
 376     for frame in frames:
 377         frame_component = config.get_grouping_component(frame, variant=variant, **meta)
 378         if variant == "app" and not frame.in_app and not all_frames_considered_in_app:
 379             frame_component.update(contributes=False, hint="non app frame")
 380         elif prev_frame is not None and is_recursion_v1(frame, prev_frame):
 381             frame_component.update(contributes=False, hint="ignored due to recursion")
 382         elif variant == "app" and not frame.in_app and all_frames_considered_in_app:
 383             frame_component.update(hint="frame considered in-app because no frame is in-app")
 384         values.append(frame_component)
 385         frames_for_filtering.append(frame.get_raw_data())
 386         prev_frame = frame
 387 
 388     # Special case for JavaScript where we want to ignore single frame
 389     # stacktraces in certain cases where those would be of too low quality
 390     # for grouping.
 391     if (
 392         len(frames) == 1
 393         and values[0].contributes
 394         and get_behavior_family_for_platform(frames[0].platform or meta["event"].platform)
 395         == "javascript"
 396         and not frames[0].function
 397         and frames[0].is_url()
 398     ):
 399         values[0].update(contributes=False, hint="ignored single non-URL JavaScript frame")
 400 
 401     return config.enhancements.assemble_stacktrace_component(
 402         values, frames_for_filtering, meta["event"].platform
 403     )

 406 def single_exception_common(exception, config, meta, with_value):
 407     if exception.stacktrace is not None:
 408         stacktrace_component = config.get_grouping_component(exception.stacktrace, **meta)
 409     else:
 410         stacktrace_component = GroupingComponent(id="stacktrace")
 411 
 412     type_component = GroupingComponent(id="type", values=[exception.type] if exception.type else [])
 413 
 414     if exception.mechanism and exception.mechanism.synthetic:
 415         type_component.update(contributes=False, hint="ignored because exception is synthetic")
 416 
 417     values = [stacktrace_component, type_component]
 418 
 419     if with_value:
 420         value_component = GroupingComponent(id="value")
 421 
 422         value_in = exception.value
 423         if value_in is not None:
 424             value_trimmed = trim_message_for_grouping(value_in)
 425             hint = "stripped common values" if value_in != value_trimmed else None
 426             if value_trimmed:
 427                 value_component.update(values=[value_trimmed], hint=hint)
 428 
 429         if stacktrace_component.contributes and value_component.contributes:
 430             value_component.update(
 431                 contributes=False, hint="ignored because stacktrace takes precedence"
 432             )
 433 
 434         values.append(value_component)
 435 
 436     return GroupingComponent(id="exception", values=values)

 450 @strategy(
 451     id="chained-exception:v1", interfaces=["exception"], variants=["!system", "app"], score=2000
 452 )
 453 def chained_exception(chained_exception, config, **meta):
 454     # Case 1: we have a single exception, use the single exception
 455     # component directly to avoid a level of nesting
 456     exceptions = chained_exception.exceptions()
 457     if len(exceptions) == 1:
 458         return config.get_grouping_component(exceptions[0], **meta)
 459 
 460     # Case 2: produce a component for each chained exception
 461     values = [config.get_grouping_component(exception, **meta) for exception in exceptions]
 462     return GroupingComponent(id="chained-exception", values=values)

 470 @strategy(id="threads:v1", interfaces=["threads"], variants=["!system", "app"], score=1900)
 471 def threads(threads_interface, config, **meta):
 472     thread_count = len(threads_interface.values)
 473     if thread_count != 1:
 474         return GroupingComponent(
 475             id="threads",
 476             contributes=False,
 477             hint="ignored because contains %d threads" % thread_count,
 478         )
 479 
 480     stacktrace = threads_interface.values[0].get("stacktrace")
 481     if not stacktrace:
 482         return GroupingComponent(id="threads", contributes=False, hint="thread has no stacktrace")
 483 
 484     return GroupingComponent(
 485         id="threads", values=[config.get_grouping_component(stacktrace, **meta)]
 486     )
