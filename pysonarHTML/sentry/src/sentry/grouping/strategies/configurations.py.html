  13 # The full mapping of all known configurations.
  14 CONFIGURATIONS = {}
  15 
  16 
  17 def register_strategy_config(id, **kwargs):
  18     if kwargs.get("base") is not None:
  19         kwargs["base"] = CONFIGURATIONS[kwargs["base"]]
  20     rv = create_strategy_configuration(id, **kwargs)
  21     if rv.config_class not in CLASSES:
  22         CLASSES.append(rv.config_class)
  23     CONFIGURATIONS[rv.id] = rv
  24     return rv
