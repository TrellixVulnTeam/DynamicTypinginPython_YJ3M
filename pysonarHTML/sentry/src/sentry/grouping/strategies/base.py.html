  19 def strategy(id=None, ids=None, variants=None, interfaces=None, name=None, score=None):
  20     """Registers a strategy"""
  21     if interfaces is None or variants is None:
  22         raise TypeError("interfaces and variants are required")
  23 
  24     if name is None:
  25         if len(interfaces) != 1:
  26             raise RuntimeError("%r requires a name" % id)
  27         name = interfaces[0]
  28 
  29     if id is not None:
  30         if ids is not None:
  31             raise TypeError("id and ids given")
  32         ids = [id]
  33 
  34     def decorator(f):
  35         for id in ids:
  36             STRATEGIES[id] = rv = Strategy(
  37                 id=id, name=name, interfaces=interfaces, variants=variants, score=score, func=f
  38             )
  39         return rv
  40 
  41     return decorator

  44 def lookup_strategy(strategy_id):
  45     """Looks up a strategy by id."""
  46     try:
  47         return STRATEGIES[strategy_id]
  48     except KeyError:
  49         raise LookupError("Unknown strategy %r" % strategy_id)

  52 class Strategy(object):
  53     """Baseclass for all strategies."""
  54 
  55     def __init__(self, id, name, interfaces, variants, score, func):
  56         self.id = id
  57         self.strategy_class = id.split(":", 1)[0]
  58         self.name = name
  59         self.interfaces = interfaces
  60         self.mandatory_variants = []
  61         self.optional_variants = []
  62         self.variants = []
  63         for variant in variants:
  64             if variant[:1] == "!":
  65                 self.mandatory_variants.append(variant[1:])
  66             else:
  67                 self.optional_variants.append(variant)
  68             self.variants.append(variant)
  69         self.score = score
  70         self.func = func
  71         self.variant_processor_func = None
  72 
  73     def __repr__(self):
  74         return "<%s id=%r variants=%r>" % (self.__class__.__name__, self.id, self.variants)
  75 
  76     def _invoke(self, func, *args, **kwargs):
  77         # We forcefully override strategy here.  This lets a strategy
  78         # function always access its metadata and directly forward it to
  79         # subcomponents without having to filter out strategy.
  80         kwargs["strategy"] = self
  81         return func(*args, **kwargs)
  82 
  83     def __call__(self, *args, **kwargs):
  84         return self._invoke(self.func, *args, **kwargs)
  85 
  86     def variant_processor(self, func):
  87         """Registers a variant reducer function that can be used to postprocess
  88         all variants created from this strategy.
  89         """
  90         self.variant_processor_func = func
  91         return func
  92 
  93     def get_grouping_component(self, event, variant, config):
  94         """Given a specific variant this calculates the grouping component.
  95         """
  96         args = []
  97         for iface_path in self.interfaces:
  98             iface = event.interfaces.get(iface_path)
  99             if iface is None:
 100                 return None
 101             args.append(iface)
 102         return self(event=event, variant=variant, config=config, *args)
 103 
 104     def get_grouping_component_variants(self, event, config):
 105         """This returns a dictionary of all components by variant that this
 106         strategy can produce.
 107         """
 108         rv = {}
 109         # trivial case: we do not have mandatory variants and can handle
 110         # them all the same.
 111         if not self.mandatory_variants:
 112             for variant in self.variants:
 113                 component = self.get_grouping_component(event, variant, config)
 114                 if component is not None:
 115                     rv[variant] = component
 116 
 117         else:
 118             mandatory_component_hashes = {}
 119             prevent_contribution = None
 120 
 121             for variant in self.mandatory_variants:
 122                 component = self.get_grouping_component(event, variant, config)
 123                 if component is None:
 124                     continue
 125                 if component.contributes:
 126                     mandatory_component_hashes[component.get_hash()] = variant
 127                 rv[variant] = component
 128 
 129             prevent_contribution = not mandatory_component_hashes
 130 
 131             for variant in self.optional_variants:
 132                 # We also only want to create another variant if it
 133                 # produces different results than the mandatory components
 134                 component = self.get_grouping_component(event, variant, config)
 135                 if component is None:
 136                     continue
 137 
 138                 # In case this variant contributes we need to check two things
 139                 # here: if we did not have a system match we need to prevent
 140                 # it from contributing.  Additionally if it matches the system
 141                 # component we also do not want the variant to contribute but
 142                 # with a different message.
 143                 if component.contributes:
 144                     if prevent_contribution:
 145                         component.update(
 146                             contributes=False,
 147                             hint="ignored because %s variant is not used"
 148                             % (
 149                                 mandatory_component_hashes.values()[0]
 150                                 if len(mandatory_component_hashes) == 1
 151                                 else "other mandatory"
 152                             ),
 153                         )
 154                     else:
 155                         hash = component.get_hash()
 156                         duplicate_of = mandatory_component_hashes.get(hash)
 157                         if duplicate_of is not None:
 158                             component.update(
 159                                 contributes=False,
 160                                 hint="ignored because hash matches %s variant" % duplicate_of,
 161                             )
 162                 rv[variant] = component
 163 
 164         if self.variant_processor_func is not None:
 165             rv = self._invoke(self.variant_processor_func, rv, event=event, config=config)
 166         return rv

 169 class StrategyConfiguration(object):
 170     id = None
 171     base = None
 172     config_class = None
 173     strategies = {}
 174     delegates = {}
 175     changelog = None
 176     hidden = False
 177     risk = RISK_LEVEL_LOW
 178 
 179     def __init__(self, enhancements=None, **extra):
 180         if enhancements is None:
 181             enhancements = Enhancements([])
 182         else:
 183             enhancements = Enhancements.loads(enhancements)
 184         self.enhancements = enhancements
 185 
 186     def __repr__(self):
 187         return "<%s %r>" % (self.__class__.__name__, self.id)
 188 
 189     def iter_strategies(self):
 190         """Iterates over all strategies by highest score to lowest."""
 191         return iter(sorted(self.strategies.values(), key=lambda x: -x.score))
 192 
 193     def get_grouping_component(self, interface, *args, **kwargs):
 194         """Invokes a delegate grouping strategy.  If no such delegate is
 195         configured a fallback grouping component is returned.
 196         """
 197         path = interface.path
 198         strategy = self.delegates.get(path)
 199         if strategy is not None:
 200             kwargs["config"] = self
 201             return strategy(interface, *args, **kwargs)
 202         return GroupingComponent(id=path, hint="grouping algorithm does not consider this value")
 203 
 204     @classmethod
 205     def as_dict(self):
 206         return {
 207             "id": self.id,
 208             "base": self.base.id if self.base else None,
 209             "strategies": sorted(self.strategies),
 210             "changelog": self.changelog,
 211             "delegates": sorted(x.id for x in self.delegates.values()),
 212             "hidden": self.hidden,
 213             "risk": self.risk,
 214             "latest": projectoptions.lookup_well_known_key("sentry:grouping_config").get_default(
 215                 epoch=projectoptions.LATEST_EPOCH
 216             )
 217             == self.id,
 218         }

 221 def create_strategy_configuration(
 222     id, strategies=None, delegates=None, changelog=None, hidden=False, base=None, risk=None
 223 ):
 224     """Declares a new strategy configuration.
 225 
 226     Values can be inherited from a base configuration.  For strategies if there is
 227     a strategy of the same class it's replaced.  For delegates if there is a
 228     delegation for the same interface it's replaced.
 229 
 230     It's impossible to remove a strategy of a class when a base is declared (same
 231     for delegates).
 232     """
 233 
 234     class NewStrategyConfiguration(StrategyConfiguration):
 235         pass
 236 
 237     NewStrategyConfiguration.id = id
 238     NewStrategyConfiguration.base = base
 239     NewStrategyConfiguration.config_class = id.split(":", 1)[0]
 240     NewStrategyConfiguration.strategies = dict(base.strategies) if base else {}
 241     NewStrategyConfiguration.delegates = dict(base.delegates) if base else {}
 242     if risk is None:
 243         risk = RISK_LEVEL_LOW
 244     NewStrategyConfiguration.risk = risk
 245     NewStrategyConfiguration.hidden = hidden
 246 
 247     by_class = {}
 248     for strategy in six.itervalues(NewStrategyConfiguration.strategies):
 249         by_class.setdefault(strategy.strategy_class, []).append(strategy.id)
 250 
 251     for strategy_id in strategies or {}:
 252         strategy = lookup_strategy(strategy_id)
 253         if strategy.score is None:
 254             raise RuntimeError("Unscored strategy %s added to %s" % (strategy_id, id))
 255         for old_id in by_class.get(strategy.strategy_class) or ():
 256             NewStrategyConfiguration.strategies.pop(old_id, None)
 257         NewStrategyConfiguration.strategies[strategy_id] = strategy
 258 
 259     new_delegates = set()
 260     for strategy_id in delegates or ():
 261         strategy = lookup_strategy(strategy_id)
 262         for interface in strategy.interfaces:
 263             if interface in new_delegates:
 264                 raise RuntimeError(
 265                     "duplicate interface match for "
 266                     "delegate %r (conflict on %r)" % (id, interface)
 267                 )
 268             NewStrategyConfiguration.delegates[interface] = strategy
 269             new_delegates.add(interface)
 270 
 271     NewStrategyConfiguration.changelog = inspect.cleandoc(changelog or "")
 272     NewStrategyConfiguration.__name__ = "StrategyConfiguration(%s)" % id
 273     return NewStrategyConfiguration
