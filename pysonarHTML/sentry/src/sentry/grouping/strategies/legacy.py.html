  58 def is_unhashable_module_legacy(frame, platform):
  59     # Fix for the case where module is a partial copy of the URL
  60     # and should not be hashed
  61     if (
  62         platform == "javascript"
  63         and "/" in frame.module
  64         and frame.abs_path
  65         and frame.abs_path.endswith(frame.module)
  66     ):
  67         return True
  68     elif platform == "java" and "$$Lambda$" in frame.module:
  69         return True
  70     return False

  80 def is_recursion_legacy(frame1, frame2):
  81     "Returns a boolean indicating whether frames are recursive calls."
  82     for field in RECURSION_COMPARISON_FIELDS:
  83         if getattr(frame1, field, None) != getattr(frame2, field, None):
  84             return False

  89 def remove_module_outliers_legacy(module, platform):
  90     """Remove things that augment the module but really should not."""
  91     if platform == "java":
  92         if module[:35] == "sun.reflect.GeneratedMethodAccessor":
  93             return "sun.reflect.GeneratedMethodAccessor", "removed reflection marker"
  94         if module[:44] == "jdk.internal.reflect.GeneratedMethodAccessor":
  95             return "jdk.internal.reflect.GeneratedMethodAccessor", "removed reflection marker"
  96         old_module = module
  97         module = _java_reflect_enhancer_re.sub(r"\1<auto>", module)
  98         module = _java_cglib_enhancer_re.sub(r"\1<auto>", module)
  99         module = _java_assist_enhancer_re.sub(r"\1<auto>", module)
 100         module = _clojure_enhancer_re.sub(r"\1<auto>", module)
 101         if old_module != module:
 102             return module, "removed codegen marker"
 103     return module, None

 106 def remove_filename_outliers_legacy(filename, platform):
 107     """
 108     Attempt to normalize filenames by removing common platform outliers.
 109 
 110     - Sometimes filename paths contain build numbers
 111     """
 112     # On cocoa we generally only want to use the last path component as
 113     # the filename.  The reason for this is that the chances are very high
 114     # that full filenames contain information we do want to strip but
 115     # currently can't (for instance because the information we get from
 116     # the dwarf files does not contain prefix information) and that might
 117     # contain things like /Users/foo/Dropbox/...
 118     if platform == "cocoa":
 119         return posixpath.basename(filename), "stripped to basename"
 120 
 121     removed = []
 122     if platform == "java":
 123         new_filename = _java_assist_enhancer_re.sub(r"\1<auto>", filename)
 124         if new_filename != filename:
 125             removed.append("javassist parts")
 126             filename = new_filename
 127 
 128     new_filename = _filename_version_re.sub("<version>/", filename)
 129     if new_filename != filename:
 130         removed.append("version")
 131         filename = new_filename
 132 
 133     if removed:
 134         return filename, "removed %s" % " and ".join(removed)
 135     return filename, None

 138 def remove_function_outliers_legacy(function):
 139     """
 140     Attempt to normalize functions by removing common platform outliers.
 141 
 142     - Ruby generates (random?) integers for various anonymous style functions
 143       such as in erb and the active_support library.
 144     - Block functions have metadata that we don't care about.
 145     """
 146     if function.startswith("block "):
 147         return "block", "ruby block"
 148     new_function = _ruby_anon_func.sub("_<anon>", function)
 149     if new_function != function:
 150         return new_function, "trimmed integer suffix"
 151     return new_function, None

 154 @strategy(id="single-exception:legacy", interfaces=["singleexception"], variants=["!system", "app"])
 155 def single_exception_legacy(exception, config, **meta):
 156     type_component = GroupingComponent(
 157         id="type", values=[exception.type] if exception.type else [], contributes=False
 158     )
 159     value_component = GroupingComponent(
 160         id="value", values=[exception.value] if exception.value else [], contributes=False
 161     )
 162     stacktrace_component = GroupingComponent(id="stacktrace")
 163 
 164     if exception.stacktrace is not None:
 165         stacktrace_component = config.get_grouping_component(exception.stacktrace, **meta)
 166         if stacktrace_component.contributes:
 167             if exception.type:
 168                 type_component.update(contributes=True)
 169                 if exception.value:
 170                     value_component.update(hint="stacktrace and type take precedence")
 171             elif exception.value:
 172                 value_component.update(hint="stacktrace takes precedence")
 173 
 174     if not stacktrace_component.contributes:
 175         if exception.type:
 176             type_component.update(contributes=True)
 177         if exception.value:
 178             value_component.update(contributes=True)
 179 
 180     return GroupingComponent(
 181         id="exception", values=[stacktrace_component, type_component, value_component]
 182     )

 185 @strategy(
 186     id="chained-exception:legacy", interfaces=["exception"], variants=["!system", "app"], score=2000
 187 )
 188 def chained_exception_legacy(chained_exception, config, **meta):
 189     # Case 1: we have a single exception, use the single exception
 190     # component directly
 191     exceptions = chained_exception.exceptions()
 192     if len(exceptions) == 1:
 193         return config.get_grouping_component(exceptions[0], **meta)
 194 
 195     # Case 2: try to build a new component out of the individual
 196     # errors however with a trick.  In case any exception has a
 197     # stacktrace we want to ignore all other exceptions.
 198     any_stacktraces = False
 199     values = []
 200     for exception in exceptions:
 201         exception_component = config.get_grouping_component(exception, **meta)
 202         stacktrace_component = exception_component.get_subcomponent("stacktrace")
 203         if stacktrace_component is not None and stacktrace_component.contributes:
 204             any_stacktraces = True
 205         values.append(exception_component)
 206 
 207     if any_stacktraces:
 208         for value in values:
 209             stacktrace_component = value.get_subcomponent("stacktrace")
 210             if stacktrace_component is None or not stacktrace_component.contributes:
 211                 value.update(contributes=False, hint="exception has no stacktrace")
 212 
 213     return GroupingComponent(id="chained-exception", values=values)

 221 @strategy(id="frame:legacy", interfaces=["frame"], variants=["!system", "app"])
 222 def frame_legacy(frame, event, **meta):
 223     platform = frame.platform or event.platform
 224 
 225     # In certain situations we want to disregard the entire frame.
 226     contributes = None
 227     hint = None
 228 
 229     # this requires some explanation: older sentry versions did not have
 230     # raw_function but only function.  For some platforms like native
 231     # we now instead store a trimmed function name in frame.function so
 232     # and the original value moved to raw_function.  This requires us to
 233     # prioritize raw_function over function in the legacy grouping code to
 234     # avoid creating new groups.
 235     func = frame.raw_function or frame.function
 236 
 237     # Safari throws [native code] frames in for calls like ``forEach``
 238     # whereas Chrome ignores these. Let's remove it from the hashing algo
 239     # so that they're more likely to group together
 240     filename_component = GroupingComponent(id="filename")
 241     if frame.filename == "<anonymous>":
 242         filename_component.update(
 243             contributes=False, values=[frame.filename], hint="anonymous filename discarded"
 244         )
 245     elif frame.filename == "[native code]":
 246         contributes = False
 247         hint = "native code indicated by filename"
 248     elif frame.filename:
 249         if has_url_origin(frame.abs_path):
 250             filename_component.update(
 251                 contributes=False, values=[frame.filename], hint="ignored because filename is a URL"
 252             )
 253         # XXX(dcramer): dont compute hash using frames containing the 'Caused by'
 254         # text as it contains an exception value which may may contain dynamic
 255         # values (see raven-java#125)
 256         elif frame.filename.startswith("Caused by: "):
 257             filename_component.update(
 258                 values=[frame.filename], contributes=False, hint="ignored because invalid"
 259             )
 260         else:
 261             hashable_filename, hashable_filename_hint = remove_filename_outliers_legacy(
 262                 frame.filename, platform
 263             )
 264             filename_component.update(values=[hashable_filename], hint=hashable_filename_hint)
 265 
 266     # if we have a module we use that for grouping.  This will always
 267     # take precedence over the filename, even if the module is
 268     # considered unhashable.
 269     module_component = GroupingComponent(id="module")
 270     if frame.module:
 271         if is_unhashable_module_legacy(frame, platform):
 272             module_component.update(
 273                 values=[
 274                     GroupingComponent(
 275                         id="salt", values=["<module>"], hint="normalized generated module name"
 276                     )
 277                 ],
 278                 hint="ignored module",
 279             )
 280         else:
 281             module_name, module_hint = remove_module_outliers_legacy(frame.module, platform)
 282             module_component.update(values=[module_name], hint=module_hint)
 283         if frame.filename:
 284             filename_component.update(
 285                 values=[frame.filename], contributes=False, hint="module takes precedence"
 286             )
 287 
 288     # Context line when available is the primary contributor
 289     context_line_component = GroupingComponent(id="context-line")
 290     if frame.context_line is not None:
 291         if len(frame.context_line) > 120:
 292             context_line_component.update(hint="discarded because line too long")
 293         elif has_url_origin(frame.abs_path) and not func:
 294             context_line_component.update(hint="discarded because from URL origin")
 295         else:
 296             context_line_component.update(values=[frame.context_line])
 297 
 298     symbol_component = GroupingComponent(id="symbol")
 299     function_component = GroupingComponent(id="function")
 300     lineno_component = GroupingComponent(id="lineno")
 301 
 302     # The context line grouping information is the most reliable one.
 303     # If we did not manage to find some information there, we want to
 304     # see if we can come up with some extra information.  We only want
 305     # to do that if we managed to get a module of filename.
 306     if not context_line_component.contributes and (
 307         module_component.contributes or filename_component.contributes
 308     ):
 309         if frame.symbol:
 310             symbol_component.update(values=[frame.symbol])
 311             if func:
 312                 function_component.update(
 313                     contributes=False, values=[func], hint="symbol takes precedence"
 314                 )
 315             if frame.lineno:
 316                 lineno_component.update(
 317                     contributes=False, values=[frame.lineno], hint="symbol takes precedence"
 318                 )
 319         elif func:
 320             if is_unhashable_function_legacy(func):
 321                 function_component.update(
 322                     values=[
 323                         GroupingComponent(
 324                             id="salt", values=["<function>"], hint="normalized lambda function name"
 325                         )
 326                     ]
 327                 )
 328             else:
 329                 function, function_hint = remove_function_outliers_legacy(func)
 330                 function_component.update(values=[function], hint=function_hint)
 331             if frame.lineno:
 332                 lineno_component.update(
 333                     contributes=False, values=[frame.lineno], hint="function takes precedence"
 334                 )
 335         elif frame.lineno:
 336             lineno_component.update(values=[frame.lineno])
 337     else:
 338         if context_line_component.contributes:
 339             fallback_hint = "is not used if context-line is available"
 340         else:
 341             fallback_hint = "is not used if module or filename are available"
 342         if frame.symbol:
 343             symbol_component.update(
 344                 contributes=False, values=[frame.symbol], hint="symbol " + fallback_hint
 345             )
 346         if func:
 347             function_component.update(
 348                 contributes=False, values=[func], hint="function name " + fallback_hint
 349             )
 350         if frame.lineno:
 351             lineno_component.update(
 352                 contributes=False, values=[frame.lineno], hint="line number " + fallback_hint
 353             )
 354 
 355     return GroupingComponent(
 356         id="frame",
 357         values=[
 358             module_component,
 359             filename_component,
 360             context_line_component,
 361             symbol_component,
 362             function_component,
 363             lineno_component,
 364         ],
 365         contributes=contributes,
 366         hint=hint,
 367     )

 370 @strategy(
 371     id="stacktrace:legacy", interfaces=["stacktrace"], variants=["!system", "app"], score=1800
 372 )
 373 def stacktrace_legacy(stacktrace, config, variant, **meta):
 374     frames = stacktrace.frames
 375     contributes = None
 376     hint = None
 377     all_frames_considered_in_app = False
 378 
 379     # TODO(dcramer): this should apply only to platform=javascript
 380     # Browser JS will often throw errors (from inlined code in an HTML page)
 381     # which contain only a single frame, no function name, and have the HTML
 382     # document as the filename. In this case the hash is often not usable as
 383     # the context cannot be trusted and the URL is dynamic (this also means
 384     # the line number cannot be trusted).
 385     if len(frames) == 1 and not frames[0].function and frames[0].is_url():
 386         contributes = False
 387         hint = "ignored single frame stack"
 388     elif variant == "app":
 389         total_frames = len(frames)
 390         in_app_count = sum(1 if f.in_app else 0 for f in frames)
 391         if in_app_count == 0:
 392             in_app_count = total_frames
 393             all_frames_considered_in_app = True
 394 
 395         # if app frames make up less than 10% of the stacktrace discard
 396         # the hash as invalid
 397         if total_frames > 0 and in_app_count / float(total_frames) < 0.10:
 398             contributes = False
 399             hint = "less than 10% of frames are in-app"
 400 
 401     values = []
 402     prev_frame = None
 403     frames_for_filtering = []
 404     for frame in frames:
 405         frame_component = config.get_grouping_component(frame, variant=variant, **meta)
 406         if variant == "app" and not frame.in_app and not all_frames_considered_in_app:
 407             frame_component.update(contributes=False, hint="non app frame")
 408         elif prev_frame is not None and is_recursion_legacy(frame, prev_frame):
 409             frame_component.update(contributes=False, hint="ignored due to recursion")
 410         elif variant == "app" and not frame.in_app and all_frames_considered_in_app:
 411             frame_component.update(hint="frame considered in-app because no frame is in-app")
 412         values.append(frame_component)
 413         frames_for_filtering.append(frame.get_raw_data())
 414         prev_frame = frame
 415 
 416     rv = config.enhancements.assemble_stacktrace_component(
 417         values, frames_for_filtering, meta["event"].platform
 418     )
 419     rv.update(contributes=contributes, hint=hint)
 420     return rv

 423 @strategy(id="threads:legacy", interfaces=["threads"], variants=["!system", "app"], score=1900)
 424 def threads_legacy(threads_interface, config, **meta):
 425     thread_count = len(threads_interface.values)
 426     if thread_count != 1:
 427         return GroupingComponent(
 428             id="threads",
 429             contributes=False,
 430             hint="ignored because contains %d threads" % thread_count,
 431         )
 432 
 433     stacktrace = threads_interface.values[0].get("stacktrace")
 434     if not stacktrace:
 435         return GroupingComponent(id="threads", contributes=False, hint="thread has no stacktrace")
 436 
 437     return GroupingComponent(
 438         id="threads", values=[config.get_grouping_component(stacktrace, **meta)]
 439     )
