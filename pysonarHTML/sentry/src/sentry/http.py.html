  70 def safe_urlopen(
  71     url,
  72     method=None,
  73     params=None,
  74     data=None,
  75     json=None,
  76     headers=None,
  77     allow_redirects=False,
  78     timeout=30,
  79     verify_ssl=True,
  80     user_agent=None,
  81 ):
  82     """
  83     A slightly safer version of ``urlib2.urlopen`` which prevents redirection
  84     and ensures the URL isn't attempting to hit a blacklisted IP range.
  85     """
  86     if user_agent is not None:
  87         warnings.warn("user_agent is no longer used with safe_urlopen")
  88 
  89     session = SafeSession()
  90 
  91     kwargs = {}
  92 
  93     if json:
  94         kwargs["json"] = json
  95         if not headers:
  96             headers = {}
  97         headers.setdefault("Content-Type", "application/json")
  98 
  99     if data:
 100         kwargs["data"] = data
 101 
 102     if params:
 103         kwargs["params"] = params
 104 
 105     if headers:
 106         kwargs["headers"] = headers
 107 
 108     if method is None:
 109         method = "POST" if (data or json) else "GET"
 110 
 111     response = session.request(
 112         method=method,
 113         url=url,
 114         allow_redirects=allow_redirects,
 115         timeout=timeout,
 116         verify=verify_ssl,
 117         **kwargs
 118     )
 119 
 120     return response

 127 def expose_url(url):
 128     if url is None:
 129         return u"<unknown>"
 130     if url[:5] == "data:":
 131         return u"<data url>"
 132     url = truncatechars(url, MAX_URL_LENGTH)
 133     if isinstance(url, six.binary_type):
 134         url = url.decode("utf-8", "replace")
 135     return url

 138 def fetch_file(
 139     url,
 140     domain_lock_enabled=True,
 141     outfile=None,
 142     headers=None,
 143     allow_redirects=True,
 144     verify_ssl=False,
 145     timeout=settings.SENTRY_SOURCE_FETCH_SOCKET_TIMEOUT,
 146     **kwargs
 147 ):
 148     """
 149     Pull down a URL, returning a UrlResult object.
 150     """
 151     # lock down domains that are problematic
 152     if domain_lock_enabled:
 153         domain = urlparse(url).netloc
 154         domain_key = "source:blacklist:v2:%s" % (md5_text(domain).hexdigest(),)
 155         domain_result = cache.get(domain_key)
 156         if domain_result:
 157             domain_result["url"] = url
 158             raise CannotFetch(domain_result)
 159 
 160     logger.debug("Fetching %r from the internet", url)
 161 
 162     http_session = SafeSession()
 163     response = None
 164 
 165     try:
 166         try:
 167             start = time.time()
 168             response = http_session.get(
 169                 url,
 170                 allow_redirects=allow_redirects,
 171                 verify=verify_ssl,
 172                 headers=headers,
 173                 timeout=timeout,
 174                 stream=True,
 175                 **kwargs
 176             )
 177 
 178             try:
 179                 cl = int(response.headers["content-length"])
 180             except (LookupError, ValueError):
 181                 cl = 0
 182             if cl > settings.SENTRY_SOURCE_FETCH_MAX_SIZE:
 183                 raise OverflowError()
 184 
 185             return_body = False
 186             if outfile is None:
 187                 outfile = six.BytesIO()
 188                 return_body = True
 189 
 190             cl = 0
 191 
 192             # Only need to even attempt to read the response body if we
 193             # got a 200 OK
 194             if response.status_code == 200:
 195                 for chunk in response.iter_content(16 * 1024):
 196                     if time.time() - start > settings.SENTRY_SOURCE_FETCH_TIMEOUT:
 197                         raise Timeout()
 198                     outfile.write(chunk)
 199                     cl += len(chunk)
 200                     if cl > settings.SENTRY_SOURCE_FETCH_MAX_SIZE:
 201                         raise OverflowError()
 202 
 203         except Exception as exc:
 204             logger.debug("Unable to fetch %r", url, exc_info=True)
 205             if isinstance(exc, RestrictedIPAddress):
 206                 error = {"type": EventError.RESTRICTED_IP, "url": expose_url(url)}
 207             elif isinstance(exc, SuspiciousOperation):
 208                 error = {"type": EventError.SECURITY_VIOLATION, "url": expose_url(url)}
 209             elif isinstance(exc, (Timeout, ReadTimeout)):
 210                 error = {
 211                     "type": EventError.FETCH_TIMEOUT,
 212                     "url": expose_url(url),
 213                     "timeout": settings.SENTRY_SOURCE_FETCH_TIMEOUT,
 214                 }
 215             elif isinstance(exc, OverflowError):
 216                 error = {
 217                     "type": EventError.FETCH_TOO_LARGE,
 218                     "url": expose_url(url),
 219                     # We want size in megabytes to format nicely
 220                     "max_size": float(settings.SENTRY_SOURCE_FETCH_MAX_SIZE) / 1024 / 1024,
 221                 }
 222             elif isinstance(exc, (RequestException, ZeroReturnError, OpenSSLError)):
 223                 error = {
 224                     "type": EventError.FETCH_GENERIC_ERROR,
 225                     "value": six.text_type(type(exc)),
 226                     "url": expose_url(url),
 227                 }
 228             else:
 229                 logger.exception(six.text_type(exc))
 230                 error = {"type": EventError.UNKNOWN_ERROR, "url": expose_url(url)}
 231 
 232             # TODO(dcramer): we want to be less aggressive on disabling domains
 233             if domain_lock_enabled:
 234                 cache.set(domain_key, error or "", 300)
 235                 logger.warning("source.disabled", extra=error)
 236             raise CannotFetch(error)
 237 
 238         headers = {k.lower(): v for k, v in response.headers.items()}
 239         encoding = response.encoding
 240 
 241         body = None
 242         if return_body:
 243             body = outfile.getvalue()
 244             outfile.close()  # we only want to close StringIO
 245 
 246         result = (headers, body, response.status_code, encoding)
 247     finally:
 248         if response is not None:
 249             response.close()
 250 
 251     return UrlResult(url, result[0], result[1], result[2], result[3])
