   6 from django.db.migrations.executor import MigrationExecutor

   8 logger = logging.getLogger(__name__)

  11 class SentryMigrationExecutor(MigrationExecutor):
  12     def _check_fake(self, migration, fake):
  13         if (
  14             os.environ.get("MIGRATION_SKIP_DANGEROUS", "0") == "1"
  15             or os.environ.get("SOUTH_SKIP_DANGEROUS", "0") == "1"
  16         ) and getattr(migration, "is_dangerous", False):
  17             # If we plan to skip migrations we just set `fake` to True here. This causes
  18             # Django to skip running the migration, but records the row as expected.
  19             fake = True
  20             logger.warning("(too dangerous)")
  21         return fake
  22 
  23     def apply_migration(self, state, migration, fake=False, fake_initial=False):
  24         fake = self._check_fake(migration, fake)
  25         return super(SentryMigrationExecutor, self).apply_migration(
  26             state, migration, fake=fake, fake_initial=fake_initial
  27         )
  28 
  29     def unapply_migration(self, state, migration, fake=False):
  30         fake = self._check_fake(migration, fake)
  31         return super(SentryMigrationExecutor, self).unapply_migration(state, migration, fake=fake)
