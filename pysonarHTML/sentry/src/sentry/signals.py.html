   8 class BetterSignal(Signal):
   9     def connect(self, receiver=None, **kwargs):
  10         """
  11         Support decorator syntax:
  12 
  13         >>> @signal.connect(sender=type)
  14         >>> def my_receiver(**kwargs):
  15         >>>     pass
  16 
  17         """
  18 
  19         def wrapped(func):
  20             return super(BetterSignal, self).connect(func, **kwargs)
  21 
  22         if receiver is None:
  23             return wrapped
  24 
  25         if hasattr(receiver, "__name__"):
  26             wrapped.__name__ = receiver.__name__
  27         if hasattr(receiver, "__module__"):
  28             wrapped.__module__ = receiver.__module__
  29         if hasattr(receiver, "__doc__"):
  30             wrapped.__doc__ = receiver.__doc__
  31         return wrapped(receiver)
  32 
  33     def send_robust(self, sender, **named):
  34         """
  35         A reimplementation of send_robust which logs failures, thus recovering stacktraces.
  36         """
  37         responses = []
  38         if not self.receivers or self.sender_receivers_cache.get(sender) is NO_RECEIVERS:
  39             return responses
  40 
  41         # Call each receiver with whatever arguments it can accept.
  42         # Return a list of tuple pairs [(receiver, response), ... ].
  43         for receiver in self._live_receivers(sender):
  44             try:
  45                 response = receiver(signal=self, sender=sender, **named)
  46             except Exception as err:
  47                 logging.error("signal.failure", extra={"receiver": repr(receiver)}, exc_info=True)
  48                 responses.append((receiver, err))
  49             else:
  50                 responses.append((receiver, response))
  51         return responses
