  24 class BgTask(object):
  25     def __init__(self, callback, roles=None, interval=60):
  26         self.callback = callback
  27         self.roles = roles or []
  28         self.interval = interval
  29         self.running = False
  30 
  31     @property
  32     def name(self):
  33         return "%s:%s" % (self.callback.__module__, self.callback.__name__)
  34 
  35     def run(self):
  36         if self.running:
  37             return
  38         self.running = True
  39 
  40         next_run = time.time() + self.interval * random.random()
  41         while self.running:
  42             now = time.time()
  43             if now >= next_run:
  44                 try:
  45                     self.callback()
  46                 except Exception:
  47                     logging.error("bgtask.failed", exc_info=True, extra=dict(task_name=self.name))
  48                 next_run = now + self.interval
  49             time.sleep(1.0)
  50 
  51     def reconfigure(self, cfg):
  52         if "roles" in cfg:
  53             self.roles = cfg["roles"]
  54         if "interval" in cfg:
  55             self.interval = cfg["interval"]
  56 
  57     def spawn_daemon(self):
  58         if self.running:
  59             return
  60         logger.info("bgtask.spawn", extra=dict(task_name=self.name))
  61         t = threading.Thread(target=self.run)
  62         t.setDaemon(True)
  63         t.start()
  64 
  65     def stop(self):
  66         logger.info("bgtask.stop", extra=dict(task_name=self.name))
  67         self.running = False

  70 def get_task(task_name):
  71     module, task_cls = task_name.split(":", 1)
  72     mod = __import__(module, None, None, [task_cls])
  73     return getattr(mod, task_cls)

  76 def spawn_bgtasks(role):
  77     for import_name, cfg in six.iteritems(settings.BGTASKS):
  78         task = get_task(import_name)
  79         # This is already running
  80         if task.name in tasks:
  81             continue
  82         task.reconfigure(cfg)
  83         if role not in task.roles:
  84             continue
  85         task.spawn_daemon()
  86         tasks[task.name] = task
  87 
  88 
  89 def shutdown_bgtasks():
  90     for task_name, task in list(six.iteritems(tasks)):
  91         task.stop()
  92         tasks.pop(task_name, None)

  95 @contextmanager
  96 def managed_bgtasks(role):
  97     spawn_bgtasks(role)
  98     try:
  99         yield
