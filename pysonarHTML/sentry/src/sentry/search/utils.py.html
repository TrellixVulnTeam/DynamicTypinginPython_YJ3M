  43 def parse_duration(value, interval):
  44     try:
  45         value = float(value)
  46     except ValueError:
  47         raise InvalidQuery(u"{} is not a valid duration value".format(value))
  48 
  49     if interval == "ms":
  50         delta = timedelta(milliseconds=value)
  51     elif interval == "s":
  52         delta = timedelta(seconds=value)
  53     elif interval in ["min", "m"]:
  54         delta = timedelta(minutes=value)
  55     elif interval in ["hr", "h"]:
  56         delta = timedelta(hours=value)
  57     elif interval in ["day", "d"]:
  58         delta = timedelta(days=value)
  59     elif interval in ["wk", "w"]:
  60         delta = timedelta(days=value * 7)
  61     else:
  62         raise InvalidQuery(
  63             u"{} is not a valid duration type, must be ms, s, min, m, hr, h, day, d, wk or w".format(
  64                 interval
  65             )
  66         )
  67 
  68     return delta.total_seconds() * 1000.0

  71 def parse_datetime_range(value):
  72     try:
  73         flag, count, interval = value[0], int(value[1:-1]), value[-1]
  74     except (ValueError, TypeError, IndexError):
  75         raise InvalidQuery(u"{} is not a valid datetime query".format(value))
  76 
  77     if flag not in ("+", "-"):
  78         raise InvalidQuery(u"{} is not a valid datetime query".format(value))
  79 
  80     if interval == "h":
  81         delta = timedelta(hours=count)
  82     elif interval == "w":
  83         delta = timedelta(days=count * 7)
  84     elif interval == "d":
  85         delta = timedelta(days=count)
  86     elif interval == "m":
  87         delta = timedelta(minutes=count)
  88     else:
  89         raise InvalidQuery(u"{} is not a valid datetime query".format(value))
  90 
  91     if flag == "-":
  92         return ((timezone.now() - delta, True), None)
  93     else:
  94         return (None, (timezone.now() - delta, True))

 106 def parse_datetime_string(value):
 107     # timezones are not supported and are assumed UTC
 108     if value[-1:] == "Z":
 109         value = value[:-1]
 110     if len(value) >= 6 and value[-6] == "+":
 111         value = value[:-6]
 112 
 113     for format in [DATETIME_FORMAT_MICROSECONDS, DATETIME_FORMAT, DATE_FORMAT]:
 114         try:
 115             return datetime.strptime(value, format).replace(tzinfo=timezone.utc)
 116         except ValueError:
 117             pass
 118 
 119     try:
 120         return parse_unix_timestamp(value)
 121     except ValueError:
 122         pass
 123 
 124     raise InvalidQuery(u"{} is not a valid ISO8601 date query".format(value))

 127 def parse_datetime_comparison(value):
 128     if value[:2] == ">=":
 129         return ((parse_datetime_string(value[2:]), True), None)
 130     if value[:2] == "<=":
 131         return (None, (parse_datetime_string(value[2:]), True))
 132     if value[:1] == ">":
 133         return ((parse_datetime_string(value[1:]), False), None)
 134     if value[:1] == "<":
 135         return (None, (parse_datetime_string(value[1:]), False))
 136 
 137     raise InvalidQuery(u"{} is not a valid datetime query".format(value))

 140 def parse_datetime_value(value):
 141     # timezones are not supported and are assumed UTC
 142     if value[-1:] == "Z":
 143         value = value[:-1]
 144     if len(value) >= 6 and value[-6] == "+":
 145         value = value[:-6]
 146 
 147     result = None
 148 
 149     # A value that only specifies the date (without a time component) should be
 150     # expanded to an interval that spans the entire day.
 151     try:
 152         result = datetime.strptime(value, DATE_FORMAT).replace(tzinfo=timezone.utc)
 153     except ValueError:
 154         pass
 155     else:
 156         return ((result, True), (result + timedelta(days=1), False))
 157 
 158     # A value that contains the time should converted to an interval.
 159     for format in [DATETIME_FORMAT, DATETIME_FORMAT_MICROSECONDS]:
 160         try:
 161             result = datetime.strptime(value, format).replace(tzinfo=timezone.utc)
 162         except ValueError:
 163             pass
 164         else:
 165             break  # avoid entering the else clause below
 166     else:
 167         try:
 168             result = parse_unix_timestamp(value)
 169         except ValueError:
 170             pass
 171 
 172     if result is None:
 173         raise InvalidQuery(u"{} is not a valid datetime query".format(value))
 174 
 175     return ((result - timedelta(minutes=5), True), (result + timedelta(minutes=6), False))

 178 def parse_datetime_expression(value):
 179     if value.startswith(("-", "+")):
 180         return parse_datetime_range(value)
 181     elif value.startswith((">", "<", "<=", ">=")):
 182         return parse_datetime_comparison(value)
 183     else:
 184         return parse_datetime_value(value)

 295 def get_numeric_field_value(field, raw_value, type=int):
 296     try:
 297         for modifier, function in numeric_modifiers:
 298             if raw_value.startswith(modifier):
 299                 return function(field, type(raw_value[len(modifier) :]))
 300         else:
 301             return {field: type(raw_value)}
 302     except ValueError:
 303         msg = u'"{}" could not be converted to a number.'.format(raw_value)
 304         raise InvalidQuery(msg)

 307 def tokenize_query(query):
 308     """
 309     Tokenizes a standard Sentry search query.
 310 
 311     Example:
 312     >>> query = 'is:resolved foo bar tag:value'
 313     >>> tokenize_query(query)
 314     {
 315         'is': ['resolved'],
 316         'query': ['foo', 'bar'],
 317         'tag': ['value'],
 318     }
 319 
 320     Has a companion implementation in static/app/utils/tokenizeSearch.tsx
 321     """
 322     result = defaultdict(list)
 323     query_params = defaultdict(list)
 324     tokens = split_query_into_tokens(query)
 325     for token in tokens:
 326         state = "query"
 327         for idx, char in enumerate(token):
 328             next_char = token[idx + 1] if idx < len(token) - 1 else None
 329             if idx == 0 and char in ('"', "'", ":"):
 330                 break
 331             if char == ":":
 332                 if next_char in (":", " "):
 333                     state = "query"
 334                 else:
 335                     state = "tags"
 336                 break
 337         query_params[state].append(token)
 338 
 339     if "query" in query_params:
 340         result["query"] = map(format_query, query_params["query"])
 341     for tag in query_params["tags"]:
 342         key, value = format_tag(tag)
 343         result[key].append(value)
 344     return dict(result)

 347 def format_tag(tag):
 348     """
 349     Splits tags on ':' and removes enclosing quotes if present and returns
 350     returns both sides of the split as strings
 351 
 352     Example:
 353     >>> format_tag('user:foo')
 354     'user', 'foo'
 355     >>>format_tag('user:"foo bar"'')
 356     'user', 'foo bar'
 357     """
 358     idx = tag.index(":")
 359     key = tag[:idx].strip('"')
 360     value = tag[idx + 1 :].strip('"')
 361     return key, value

 375 def split_query_into_tokens(query):
 376     """
 377     Splits query string into tokens for parsing by 'tokenize_query'.
 378     Returns list of strigs
 379     Rules:
 380     Split on whitespace
 381         Unless
 382         - inside enclosing quotes -> 'user:"foo    bar"'
 383         - end of last word is a ':' -> 'user:  foo'
 384 
 385     Example:
 386     >>> split_query_into_tokens('user:foo user: bar  user"foo bar' foo  bar) =>
 387     ['user:foo', 'user: bar', 'user"foo bar"', 'foo',  'bar']
 388     """
 389     tokens = []
 390     token = ""
 391     quote_enclosed = False
 392     quote_type = None
 393     end_of_prev_word = None
 394     for idx, char in enumerate(query):
 395         next_char = query[idx + 1] if idx < len(query) - 1 else None
 396         token += char
 397         if next_char and not char.isspace() and next_char.isspace():
 398             end_of_prev_word = char
 399         if char.isspace() and not quote_enclosed and end_of_prev_word != ":":
 400             if not token.isspace():
 401                 tokens.append(token.strip(" "))
 402                 token = ""
 403         if char in ("'", '"'):
 404             if not quote_enclosed or quote_type == char:
 405                 quote_enclosed = not quote_enclosed
 406                 if quote_enclosed:
 407                     quote_type = char
 408     if not token.isspace():
 409         tokens.append(token.strip(" "))
 410     return tokens

 413 def parse_query(projects, query, user, environments):
 414     # TODO(dcramer): handle query being wrapped in quotes
 415     tokens = tokenize_query(query)
 416 
 417     results = {"tags": {}, "query": []}
 418     for key, token_list in six.iteritems(tokens):
 419         for value in token_list:
 420             if key == "query":
 421                 results["query"].append(value)
 422             elif key == "is":
 423                 if value == "unassigned":
 424                     results["unassigned"] = True
 425                 elif value == "assigned":
 426                     results["unassigned"] = False
 427                 else:
 428                     try:
 429                         results["status"] = STATUS_CHOICES[value]
 430                     except KeyError:
 431                         raise InvalidQuery(u"'is:' had unknown status code '{}'.".format(value))
 432             elif key == "assigned":
 433                 results["assigned_to"] = parse_actor_value(projects, value, user)
 434             elif key == "bookmarks":
 435                 results["bookmarked_by"] = parse_user_value(value, user)
 436             elif key == "subscribed":
 437                 results["subscribed_by"] = parse_user_value(value, user)
 438             elif key in ("first-release", "firstRelease"):
 439                 results["first_release"] = parse_release(value, projects, environments)
 440             elif key == "release":
 441                 results["tags"]["sentry:release"] = parse_release(value, projects, environments)
 442             elif key == "dist":
 443                 results["tags"]["sentry:dist"] = value
 444             elif key == "user":
 445                 if ":" in value:
 446                     comp, value = value.split(":", 1)
 447                 else:
 448                     comp = "id"
 449                 results["tags"]["sentry:user"] = get_user_tag(projects, comp, value)
 450             elif key == "has":
 451                 if value == "user":
 452                     value = "sentry:user"
 453                 elif value == "release":
 454                     value = "sentry:release"
 455                 # `has:x` query should not take precedence over `x:value` queries
 456                 if value not in results["tags"]:
 457                     results["tags"][value] = ANY
 458             elif key in ("age", "firstSeen"):
 459                 results.update(get_date_params(value, "age_from", "age_to"))
 460             elif key in ("last_seen", "lastSeen"):
 461                 results.update(get_date_params(value, "last_seen_from", "last_seen_to"))
 462             elif key == "activeSince":
 463                 results.update(get_date_params(value, "active_at_from", "active_at_to"))
 464             elif key.startswith("user."):
 465                 results["tags"]["sentry:user"] = get_user_tag(projects, key.split(".", 1)[1], value)
 466             elif key == "event.timestamp":
 467                 results.update(get_date_params(value, "date_from", "date_to"))
 468             elif key == "timesSeen":
 469                 results.update(get_numeric_field_value("times_seen", value))
 470             else:
 471                 results["tags"][key] = value
 472 
 473     results["query"] = " ".join(results["query"])
 474 
 475     return results

 478 def convert_user_tag_to_query(key, value):
 479     """
 480     Converts a user tag to a query string that can be used to search for that
 481     user. Returns None if not a user tag.
 482     """
 483     if key == "user" and ":" in value:
 484         sub_key, value = value.split(":", 1)
 485         if KEYWORD_MAP.get_key(sub_key, None):
 486             return 'user.%s:"%s"' % (sub_key, value.replace('"', '\\"'))
