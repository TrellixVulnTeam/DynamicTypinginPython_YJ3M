  77 class QCallbackCondition(Condition):
  78     def __init__(self, callback):
  79         self.callback = callback
  80 
  81     def apply(self, queryset, search_filter):
  82         value = search_filter.value.raw_value
  83         q = self.callback(value)
  84         if search_filter.operator not in ("=", "!="):
  85             raise InvalidSearchQuery(
  86                 u"Operator {} not valid for search {}".format(search_filter.operator, search_filter)
  87             )
  88         queryset_method = queryset.filter if search_filter.operator == "=" else queryset.exclude
  89         queryset = queryset_method(q)
  90         return queryset

  93 class ScalarCondition(Condition):
  94     """
  95     Adds a scalar filter to a ``QuerySet`` object. Only accepts `SearchFilter`
  96     instances
  97     """
  98 
  99     OPERATOR_TO_DJANGO = {">=": "gte", "<=": "lte", ">": "gt", "<": "lt"}
 100 
 101     def __init__(self, field, extra=None):
 102         self.field = field
 103         self.extra = extra
 104 
 105     def _get_operator(self, search_filter):
 106         django_operator = self.OPERATOR_TO_DJANGO.get(search_filter.operator, "")
 107         if django_operator:
 108             django_operator = "__{}".format(django_operator)
 109         return django_operator
 110 
 111     def apply(self, queryset, search_filter):
 112         django_operator = self._get_operator(search_filter)
 113         qs_method = queryset.exclude if search_filter.operator == "!=" else queryset.filter
 114 
 115         q_dict = {"{}{}".format(self.field, django_operator): search_filter.value.raw_value}
 116         if self.extra:
 117             q_dict.update(self.extra)
 118 
 119         return qs_method(**q_dict)

 122 class QuerySetBuilder(object):
 123     def __init__(self, conditions):
 124         self.conditions = conditions
 125 
 126     def build(self, queryset, search_filters):
 127         for search_filter in search_filters:
 128             name = search_filter.key.name
 129             if name in self.conditions:
 130                 condition = self.conditions[name]
 131                 queryset = condition.apply(queryset, search_filter)
 132         return queryset

 137     def query(
 138         self,
 139         projects,
 140         environments=None,
 141         sort_by="date",
 142         limit=100,
 143         cursor=None,
 144         count_hits=False,
 145         paginator_options=None,
 146         search_filters=None,
 147         date_from=None,
 148         date_to=None,
 149     ):
 150         search_filters = search_filters if search_filters is not None else []
 151 
 152         # ensure projects are from same org
 153         if len({p.organization_id for p in projects}) != 1:
 154             raise RuntimeError("Cross organization search not supported")
 155 
 156         if paginator_options is None:
 157             paginator_options = {}
 158 
 159         # filter out groups which are beyond the retention period
 160         retention = quotas.get_event_retention(organization=projects[0].organization)
 161         if retention:
 162             retention_window_start = timezone.now() - timedelta(days=retention)
 163         else:
 164             retention_window_start = None
 165 
 166         group_queryset = self._build_group_queryset(
 167             projects=projects,
 168             environments=environments,
 169             search_filters=search_filters,
 170             retention_window_start=retention_window_start,
 171             date_from=date_from,
 172             date_to=date_to,
 173         )
 174 
 175         query_executor = self._get_query_executor(
 176             group_queryset=group_queryset,
 177             projects=projects,
 178             environments=environments,
 179             search_filters=search_filters,
 180             date_from=date_from,
 181             date_to=date_to,
 182         )
 183 
 184         return query_executor.query(
 185             projects=projects,
 186             retention_window_start=retention_window_start,
 187             group_queryset=group_queryset,
 188             environments=environments,
 189             sort_by=sort_by,
 190             limit=limit,
 191             cursor=cursor,
 192             count_hits=count_hits,
 193             paginator_options=paginator_options,
 194             search_filters=search_filters,
 195             date_from=date_from,
 196             date_to=date_to,
 197         )

 199     def _build_group_queryset(
 200         self, projects, environments, search_filters, retention_window_start, *args, **kwargs
 201     ):
 202         """This method should return a QuerySet of the Group model.
 203         How you implement it is up to you, but we generally take in the various search parameters
 204         and filter Group's down using the field's we want to query on in Postgres."""
 205 
 206         group_queryset = self._initialize_group_queryset(
 207             projects, environments, retention_window_start, search_filters
 208         )
 209         qs_builder_conditions = self._get_queryset_conditions(
 210             projects, environments, search_filters
 211         )
 212         group_queryset = QuerySetBuilder(qs_builder_conditions).build(
 213             group_queryset, search_filters
 214         )
 215         return group_queryset

 217     def _initialize_group_queryset(
 218         self, projects, environments, retention_window_start, search_filters
 219     ):
 220         group_queryset = Group.objects.filter(project__in=projects).exclude(
 221             status__in=[
 222                 GroupStatus.PENDING_DELETION,
 223                 GroupStatus.DELETION_IN_PROGRESS,
 224                 GroupStatus.PENDING_MERGE,
 225             ]
 226         )
 227 
 228         if retention_window_start:
 229             group_queryset = group_queryset.filter(last_seen__gte=retention_window_start)
 230 
 231         if environments is not None:
 232             environment_ids = [environment.id for environment in environments]
 233             group_queryset = group_queryset.filter(
 234                 groupenvironment__environment_id__in=environment_ids
 235             )
 236         return group_queryset

 256     def _get_queryset_conditions(self, projects, environments, search_filters):
 257         queryset_conditions = {
 258             "status": QCallbackCondition(lambda status: Q(status=status)),
 259             "bookmarked_by": QCallbackCondition(
 260                 lambda user: Q(bookmark_set__project__in=projects, bookmark_set__user=user)
 261             ),
 262             "assigned_to": QCallbackCondition(
 263                 functools.partial(assigned_to_filter, projects=projects)
 264             ),
 265             "unassigned": QCallbackCondition(
 266                 functools.partial(unassigned_filter, projects=projects)
 267             ),
 268             "subscribed_by": QCallbackCondition(
 269                 lambda user: Q(
 270                     id__in=GroupSubscription.objects.filter(
 271                         project__in=projects, user=user, is_active=True
 272                     ).values_list("group")
 273                 )
 274             ),
 275             "active_at": ScalarCondition("active_at"),
 276         }
 277 
 278         if environments is not None:
 279             environment_ids = [environment.id for environment in environments]
 280             queryset_conditions.update(
 281                 {
 282                     "first_release": QCallbackCondition(
 283                         lambda version: Q(
 284                             # if environment(s) are selected, we just filter on the group
 285                             # environment's first_release attribute.
 286                             groupenvironment__first_release__organization_id=projects[
 287                                 0
 288                             ].organization_id,
 289                             groupenvironment__first_release__version=version,
 290                             groupenvironment__environment_id__in=environment_ids,
 291                         )
 292                     ),
 293                     "first_seen": ScalarCondition(
 294                         "groupenvironment__first_seen",
 295                         {"groupenvironment__environment_id__in": environment_ids},
 296                     ),
 297                 }
 298             )
 299         else:
 300             queryset_conditions.update(
 301                 {
 302                     "first_release": QCallbackCondition(
 303                         functools.partial(first_release_all_environments_filter, projects=projects)
 304                     ),
 305                     "first_seen": ScalarCondition("first_seen"),
 306                 }
 307             )
 308         return queryset_conditions
