  44 @instrumented_task(
  45     name="sentry.data_export.tasks.assemble_download",
  46     queue="data_export",

  48     max_retries=3,

  51 def assemble_download(
  52     data_export_id,
  53     export_limit=EXPORTED_ROWS_LIMIT,
  54     batch_size=SNUBA_MAX_RESULTS,
  55     offset=0,
  56     bytes_written=0,
  57     environment_id=None,
  58     **kwargs
  59 ):
  60     first_page = offset == 0
  61 
  62     try:
  63         if first_page:
  64             logger.info("dataexport.start", extra={"data_export_id": data_export_id})
  65         data_export = ExportedData.objects.get(id=data_export_id)
  66         if first_page:
  67             metrics.incr("dataexport.start", tags={"success": True}, sample_rate=1.0)
  68         logger.info("dataexport.run", extra={"data_export_id": data_export_id, "offset": offset})
  69     except ExportedData.DoesNotExist as error:
  70         if first_page:
  71             metrics.incr("dataexport.start", tags={"success": False}, sample_rate=1.0)
  72         logger.exception(error)
  73         return
  74 
  75     try:
  76         # ensure that the export limit is set and capped at EXPORTED_ROWS_LIMIT
  77         if export_limit is None:
  78             export_limit = EXPORTED_ROWS_LIMIT
  79         else:
  80             export_limit = min(export_limit, EXPORTED_ROWS_LIMIT)
  81 
  82         processor = get_processor(data_export, environment_id)
  83 
  84         with tempfile.TemporaryFile() as tf:
  85             writer = csv.DictWriter(tf, processor.header_fields, extrasaction="ignore")
  86             if first_page:
  87                 writer.writeheader()
  88 
  89             # the position in the file at the end of the headers
  90             starting_pos = tf.tell()
  91 
  92             # the row offset relative to the start of the current task
  93             # this offset tells you the number of rows written during this batch fragment
  94             fragment_offset = 0
  95 
  96             # the absolute row offset from the beginning of the export
  97             next_offset = offset + fragment_offset
  98 
  99             while True:
 100                 # the number of rows to export in the next batch fragment
 101                 fragment_row_count = min(batch_size, max(export_limit - next_offset, 1))
 102 
 103                 rows = process_rows(processor, data_export, fragment_row_count, next_offset)
 104                 writer.writerows(rows)
 105 
 106                 fragment_offset += len(rows)
 107                 next_offset = offset + fragment_offset
 108 
 109                 if (
 110                     not rows
 111                     or len(rows) < batch_size
 112                     # the batch may exceed MAX_BATCH_SIZE but immediately stops
 113                     or tf.tell() - starting_pos >= MAX_BATCH_SIZE
 114                 ):
 115                     break
 116 
 117             tf.seek(0)
 118             new_bytes_written = store_export_chunk_as_blob(data_export, bytes_written, tf)
 119             bytes_written += new_bytes_written
 120     except ExportError as error:
 121         return data_export.email_failure(message=six.text_type(error))
 122     except Exception as error:
 123         metrics.incr("dataexport.error", tags={"error": six.text_type(error)}, sample_rate=1.0)
 124         logger.error(
 125             "dataexport.error: %s",
 126             six.text_type(error),
 127             extra={"query": data_export.payload, "org": data_export.organization_id},
 128         )
 129         capture_exception(error)
 130 
 131         try:
 132             current.retry()
 133         except MaxRetriesExceededError:
 134             metrics.incr(
 135                 "dataexport.end",
 136                 tags={"success": False, "error": six.text_type(error)},
 137                 sample_rate=1.0,
 138             )
 139             return data_export.email_failure(message="Internal processing failure")
 140     else:
 141         if rows and len(rows) >= batch_size and new_bytes_written and next_offset < export_limit:
 142             assemble_download.delay(
 143                 data_export_id,
 144                 export_limit=export_limit,
 145                 batch_size=batch_size,
 146                 offset=next_offset,
 147                 bytes_written=bytes_written,
 148                 environment_id=environment_id,
 149             )
 150         else:
 151             metrics.timing("dataexport.row_count", next_offset)
 152             metrics.timing("dataexport.file_size", bytes_written)
 153             merge_export_blobs.delay(data_export_id)

 156 def get_processor(data_export, environment_id):
 157     try:
 158         if data_export.query_type == ExportQueryType.ISSUES_BY_TAG:
 159             payload = data_export.query_info
 160             processor = IssuesByTagProcessor(
 161                 project_id=payload["project"][0],
 162                 group_id=payload["group"],
 163                 key=payload["key"],
 164                 environment_id=environment_id,
 165             )
 166         elif data_export.query_type == ExportQueryType.DISCOVER:
 167             processor = DiscoverProcessor(
 168                 discover_query=data_export.query_info, organization_id=data_export.organization_id,
 169             )
 170         return processor
 171     except ExportError as error:
 172         metrics.incr("dataexport.error", tags={"error": six.text_type(error)}, sample_rate=1.0)
 173         logger.info("dataexport.error: {}".format(six.text_type(error)))
 174         capture_exception(error)
 175         raise

 178 def process_rows(processor, data_export, batch_size, offset):
 179     try:
 180         if data_export.query_type == ExportQueryType.ISSUES_BY_TAG:
 181             rows = process_issues_by_tag(processor, batch_size, offset)
 182         elif data_export.query_type == ExportQueryType.DISCOVER:
 183             rows = process_discover(processor, batch_size, offset)
 184         return rows
 185     except ExportError as error:
 186         metrics.incr("dataexport.error", tags={"error": six.text_type(error)}, sample_rate=1.0)
 187         logger.info("dataexport.error: {}".format(six.text_type(error)))
 188         capture_exception(error)
 189         raise

 211 @transaction.atomic()
 212 def store_export_chunk_as_blob(data_export, bytes_written, fileobj, blob_size=DEFAULT_BLOB_SIZE):
 213     # adapted from `putfile` in  `src/sentry/models/file.py`
 214     bytes_offset = 0
 215     while True:
 216         contents = fileobj.read(blob_size)
 217         if not contents:
 218             return bytes_offset
 219 
 220         blob_fileobj = ContentFile(contents)
 221         blob = FileBlob.from_file(blob_fileobj, logger=logger)
 222         ExportedDataBlob.objects.create(
 223             data_export=data_export, blob=blob, offset=bytes_written + bytes_offset
 224         )
 225 
 226         bytes_offset += blob.size
 227 
 228         # there is a maximum file size allowed, so we need to make sure we don't exceed it
 229         # NOTE: there seems to be issues with downloading files larger than 1 GB on slower
 230         # networks, limit the export to 1 GB for now to improve reliability
 231         if bytes_written + bytes_offset >= min(MAX_FILE_SIZE, 2 ** 30):
 232             transaction.set_rollback(True)
 233             return 0

 236 @instrumented_task(name="sentry.data_export.tasks.merge_blobs", queue="data_export", acks_late=True)
 237 def merge_export_blobs(data_export_id, **kwargs):
 238     try:
 239         data_export = ExportedData.objects.get(id=data_export_id)
 240     except ExportedData.DoesNotExist as error:
 241         logger.exception(error)
 242         return
 243 
 244     # adapted from `putfile` in  `src/sentry/models/file.py`
 245     try:
 246         with transaction.atomic():
 247             file = File.objects.create(
 248                 name=data_export.file_name, type="export.csv", headers={"Content-Type": "text/csv"},
 249             )
 250             size = 0
 251             file_checksum = sha1(b"")
 252 
 253             for export_blob in ExportedDataBlob.objects.filter(data_export=data_export).order_by(
 254                 "offset"
 255             ):
 256                 blob = export_blob.blob
 257                 FileBlobIndex.objects.create(file=file, blob=blob, offset=size)
 258                 size += blob.size
 259                 blob_checksum = sha1(b"")
 260 
 261                 for chunk in blob.getfile().chunks():
 262                     blob_checksum.update(chunk)
 263                     file_checksum.update(chunk)
 264 
 265                 if blob.checksum != blob_checksum.hexdigest():
 266                     raise AssembleChecksumMismatch("Checksum mismatch")
 267 
 268             file.size = size
 269             file.checksum = file_checksum.hexdigest()
 270             file.save()
 271             data_export.finalize_upload(file=file)
 272 
 273             time_elapsed = (timezone.now() - data_export.date_added).total_seconds()
 274             metrics.timing("dataexport.duration", time_elapsed)
 275             logger.info("dataexport.end", extra={"data_export_id": data_export_id})
 276             metrics.incr("dataexport.end", tags={"success": True}, sample_rate=1.0)
 277     except Exception as error:
 278         metrics.incr("dataexport.error", tags={"error": six.text_type(error)}, sample_rate=1.0)
 279         metrics.incr(
 280             "dataexport.end",
 281             tags={"success": False, "error": six.text_type(error)},
 282             sample_rate=1.0,
 283         )
 284         logger.error(
 285             "dataexport.error: %s",
 286             six.text_type(error),
 287             extra={"query": data_export.payload, "org": data_export.organization_id},
 288         )
 289         capture_exception(error)
 290         if isinstance(error, IntegrityError):
 291             message = "Failed to save the assembled file."
 292         else:
 293             message = "Internal processing failure."
 294         return data_export.email_failure(message=message)
