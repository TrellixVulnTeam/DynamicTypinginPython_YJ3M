  28 class DataExportEndpoint(OrganizationEndpoint, EnvironmentMixin):
  29     permission_classes = (OrganizationDataExportPermission,)
  30 
  31     def post(self, request, organization):
  32         """
  33         Create a new asynchronous file export task, and
  34         email user upon completion,
  35         """
  36         # Ensure new data-export features are enabled
  37         if not features.has("organizations:data-export", organization):
  38             return Response(status=404)
  39 
  40         # Get environment_id and limit if available
  41         try:
  42             environment_id = self._get_environment_id_from_request(request, organization.id)
  43         except Environment.DoesNotExist as error:
  44             return Response(error, status=400)
  45         limit = request.data.get("limit")
  46 
  47         # Validate the data export payload
  48         serializer = DataExportQuerySerializer(data=request.data)
  49         if not serializer.is_valid():
  50             return Response(serializer.errors, status=400)
  51         data = serializer.validated_data
  52 
  53         # Validate the project field, if provided
  54         # A PermissionDenied error will be raised in `_get_projects_by_id` if the request is invalid
  55         project_query = data["query_info"].get("project")
  56         if project_query:
  57             # Coerce the query into a set
  58             if isinstance(project_query, list):
  59                 projects = self._get_projects_by_id(
  60                     set(map(int, project_query)), request, organization
  61                 )
  62             else:
  63                 projects = self._get_projects_by_id({int(project_query)}, request, organization)
  64             data["query_info"]["project"] = [project.id for project in projects]
  65 
  66         # Discover Pre-processing
  67         if data["query_type"] == ExportQueryType.DISCOVER_STR:
  68             if not features.has("organizations:discover-basic", organization, actor=request.user):
  69                 return Response(status=403)
  70 
  71             query_info = data["query_info"]
  72 
  73             if len(query_info.get("field", [])) > MAX_FIELDS:
  74                 detail = "You can export up to {0} fields at a time. Please delete some and try again.".format(
  75                     MAX_FIELDS
  76                 )
  77                 raise ParseError(detail=detail)
  78 
  79             if "project" not in query_info:
  80                 projects = self.get_projects(request, organization)
  81                 query_info["project"] = [project.id for project in projects]
  82 
  83             start, end = get_date_range_from_params(query_info)
  84             if "statsPeriod" in query_info:
  85                 del query_info["statsPeriod"]
  86             if "statsPeriodStart" in query_info:
  87                 del query_info["statsPeriodStart"]
  88             if "statsPeriodEnd" in query_info:
  89                 del query_info["statsPeriodEnd"]
  90             query_info["start"] = start.isoformat()
  91             query_info["end"] = end.isoformat()
  92 
  93         try:
  94             # If this user has sent a sent a request with the same payload and organization,
  95             # we return them the latest one that is NOT complete (i.e. don't start another)
  96             query_type = ExportQueryType.from_str(data["query_type"])
  97             data_export, created = ExportedData.objects.get_or_create(
  98                 organization=organization,
  99                 user=request.user,
 100                 query_type=query_type,
 101                 query_info=data["query_info"],
 102                 date_finished=None,
 103             )
 104             status = 200
 105             if created:
 106                 metrics.incr(
 107                     "dataexport.enqueue", tags={"query_type": data["query_type"]}, sample_rate=1.0
 108                 )
 109                 assemble_download.delay(
 110                     data_export_id=data_export.id, export_limit=limit, environment_id=environment_id
 111                 )
 112                 status = 201
 113         except ValidationError as e:
 114             # This will handle invalid JSON requests
 115             metrics.incr(
 116                 "dataexport.invalid", tags={"query_type": data.get("query_type")}, sample_rate=1.0
 117             )
 118             return Response({"detail": six.text_type(e)}, status=400)
 119         return Response(serialize(data_export, request.user), status=status)
