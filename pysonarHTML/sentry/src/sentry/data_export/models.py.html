  26 class ExportedData(Model):
  27     """
  28     Stores references to asynchronous data export jobs
  29     """
  30 
  31     __core__ = False
  32 
  33     organization = FlexibleForeignKey("sentry.Organization")
  34     user = FlexibleForeignKey(settings.AUTH_USER_MODEL, null=True, on_delete=models.SET_NULL)
  35     file = FlexibleForeignKey(
  36         "sentry.File", null=True, db_constraint=False, on_delete=models.SET_NULL
  37     )
  38     date_added = models.DateTimeField(default=timezone.now)
  39     date_finished = models.DateTimeField(null=True)
  40     date_expired = models.DateTimeField(null=True, db_index=True)
  41     query_type = BoundedPositiveIntegerField(choices=ExportQueryType.as_choices())
  42     query_info = JSONField()
  43 
  44     @property
  45     def status(self):
  46         if self.date_finished is None:
  47             return ExportStatus.Early
  48         elif self.date_expired < timezone.now():
  49             return ExportStatus.Expired
  50         else:
  51             return ExportStatus.Valid
  52 
  53     @property
  54     def payload(self):
  55         payload = self.query_info.copy()
  56         payload["export_type"] = ExportQueryType.as_str(self.query_type)
  57         return payload
  58 
  59     @property
  60     def file_name(self):
  61         date = self.date_added.strftime("%Y-%B-%d")
  62         export_type = ExportQueryType.as_str(self.query_type)
  63         # Example: Discover_2020-July-21_27.csv
  64         return "{}_{}_{}.csv".format(export_type, date, self.id)
  65 
  66     @staticmethod
  67     def format_date(date):
  68         # Example: 12:21 PM on July 21, 2020 (UTC)
  69         return None if date is None else date.strftime("%-I:%M %p on %B %d, %Y (%Z)")
  70 
  71     def delete_file(self):
  72         if self.file:
  73             self.file.delete()
  74 
  75     def delete(self, *args, **kwargs):
  76         self.delete_file()
  77         super(ExportedData, self).delete(*args, **kwargs)
  78 
  79     def finalize_upload(self, file, expiration=DEFAULT_EXPIRATION):
  80         self.delete_file()  # If a file is present, remove it
  81         current_time = timezone.now()
  82         expire_time = current_time + expiration
  83         self.update(file=file, date_finished=current_time, date_expired=expire_time)
  84         self.email_success()
  85 
  86     def email_success(self):
  87         from sentry.utils.email import MessageBuilder
  88 
  89         # The following condition should never be true, but it's a safeguard in case someone manually calls this method
  90         if self.date_finished is None or self.date_expired is None or self.file is None:
  91             logger.warning(
  92                 "Notification email attempted on incomplete dataset",
  93                 extra={"data_export_id": self.id, "organization_id": self.organization_id},
  94             )
  95             return
  96         url = absolute_uri(
  97             reverse("sentry-data-export-details", args=[self.organization.slug, self.id])
  98         )
  99         msg = MessageBuilder(
 100             subject="Your data is ready.",
 101             context={"url": url, "expiration": self.format_date(self.date_expired)},
 102             type="organization.export-data",
 103             template="sentry/emails/data-export-success.txt",
 104             html_template="sentry/emails/data-export-success.html",
 105         )
 106         msg.send_async([self.user.email])
 107         metrics.incr("dataexport.end", tags={"success": True}, sample_rate=1.0)
 108 
 109     def email_failure(self, message):
 110         from sentry.utils.email import MessageBuilder
 111 
 112         msg = MessageBuilder(
 113             subject="We couldn't export your data.",
 114             context={
 115                 "creation": self.format_date(self.date_added),
 116                 "error_message": message,
 117                 "payload": json.dumps(self.payload, indent=2, sort_keys=True),
 118             },
 119             type="organization.export-data",
 120             template="sentry/emails/data-export-failure.txt",
 121             html_template="sentry/emails/data-export-failure.html",
 122         )
 123         msg.send_async([self.user.email])
 124         metrics.incr("dataexport.end", tags={"success": False}, sample_rate=1.0)
 125         self.delete()
 126 
 127     class Meta:
 128         app_label = "sentry"
 129         db_table = "sentry_exporteddata"
 130 
 131     __repr__ = sane_repr("query_type", "query_info")
