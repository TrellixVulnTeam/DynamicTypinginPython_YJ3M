  16 class DiscoverProcessor(object):
  17     """
  18     Processor for exports of discover data based on a provided query
  19     """
  20 
  21     def __init__(self, organization_id, discover_query):
  22         self.projects = self.get_projects(organization_id, discover_query)
  23         self.start, self.end = get_date_range_from_params(discover_query)
  24         self.params = {
  25             "organization_id": organization_id,
  26             "project_id": [project.id for project in self.projects],
  27             "start": self.start,
  28             "end": self.end,
  29         }
  30         self.header_fields = map(lambda x: get_function_alias(x), discover_query["field"])
  31         self.data_fn = self.get_data_fn(
  32             fields=discover_query["field"], query=discover_query["query"], params=self.params
  33         )
  34 
  35     @staticmethod
  36     def get_projects(organization_id, query):
  37         projects = list(Project.objects.filter(id__in=query.get("project")))
  38         if len(projects) == 0:
  39             raise ExportError("Requested project does not exist")
  40         return projects
  41 
  42     @staticmethod
  43     def get_data_fn(fields, query, params):
  44         def data_fn(offset, limit):
  45             return discover.query(
  46                 selected_columns=fields,
  47                 query=query,
  48                 params=params,
  49                 offset=offset,
  50                 limit=limit,
  51                 referrer="data_export.tasks.discover",
  52                 auto_fields=True,
  53                 use_aggregate_conditions=True,
  54             )
  55 
  56         return data_fn
  57 
  58     def handle_fields(self, result_list):
  59         # Find issue short_id if present
  60         # (originally in `/api/bases/organization_events.py`)
  61         new_result_list = result_list[:]
  62         if "issue" in self.header_fields:
  63             issue_ids = set(result["issue.id"] for result in new_result_list)
  64             issues = {
  65                 i.id: i.qualified_short_id
  66                 for i in Group.objects.filter(
  67                     id__in=issue_ids,
  68                     project__in=self.params["project_id"],
  69                     project__organization_id=self.params["organization_id"],
  70                 )
  71             }
  72             for result in new_result_list:
  73                 if "issue.id" in result:
  74                     result["issue"] = issues.get(result["issue.id"], "unknown")
  75         return new_result_list
