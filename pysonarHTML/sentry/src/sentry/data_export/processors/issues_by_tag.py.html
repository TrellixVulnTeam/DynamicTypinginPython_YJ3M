  11 class IssuesByTagProcessor(object):
  12     """
  13     Processor for exports of issues data based on a provided tag

  16     def __init__(self, project_id, group_id, key, environment_id):
  17         self.project = self.get_project(project_id)
  18         self.group = self.get_group(group_id, self.project)
  19         self.key = key
  20         self.environment_id = environment_id
  21         self.header_fields = self.get_header_fields(self.key)
  22         self.lookup_key = self.get_lookup_key(self.key)
  23         # Ensure the tag key exists, as it may have been deleted
  24         try:
  25             tagstore.get_tag_key(self.project.id, environment_id, self.lookup_key)
  26         except tagstore.TagKeyNotFound:
  27             raise ExportError("Requested key does not exist")
  28         self.callbacks = self.get_callbacks(self.key, self.group.project_id)

  30     @staticmethod
  31     def get_project(project_id):
  32         try:
  33             project = Project.objects.get_from_cache(id=project_id)
  34             return project
  35         except Project.DoesNotExist:
  36             raise ExportError("Requested project does not exist")
  37 
  38     @staticmethod
  39     def get_group(group_id, project):
  40         try:
  41             group, _ = get_group_with_redirect(
  42                 group_id, queryset=Group.objects.filter(project=project)
  43             )
  44             return group
  45         except Group.DoesNotExist:
  46             raise ExportError("Requested issue does not exist")

  48     @staticmethod
  49     def get_header_fields(key):
  50         if key == "user":
  51             return [
  52                 "value",
  53                 "id",
  54                 "email",
  55                 "username",
  56                 "ip_address",
  57                 "times_seen",
  58                 "last_seen",
  59                 "first_seen",
  60             ]
  61         else:
  62             return ["value", "times_seen", "last_seen", "first_seen"]
  63 
  64     @staticmethod
  65     def get_lookup_key(key):
  66         return six.text_type("sentry:{}".format(key)) if tagstore.is_reserved_key(key) else key
  67 
  68     @staticmethod
  69     def get_eventuser_callback(project_id):
  70         def attach_eventuser(items):
  71             users = EventUser.for_tags(project_id, [i.value for i in items])
  72             for item in items:
  73                 item._eventuser = users.get(item.value)
  74 
  75         return attach_eventuser
  76 
  77     @staticmethod
  78     def get_callbacks(key, project_id):
  79         return [IssuesByTagProcessor.get_eventuser_callback(project_id)] if key == "user" else []
  80 
  81     @staticmethod
  82     def serialize_row(item, key):
  83         result = {
  84             "value": item.value,
  85             "times_seen": item.times_seen,
  86             "last_seen": item.last_seen.strftime("%Y-%m-%dT%H:%M:%S.%fZ"),
  87             "first_seen": item.first_seen.strftime("%Y-%m-%dT%H:%M:%S.%fZ"),
  88         }
  89         if key == "user":
  90             euser = item._eventuser
  91             result["id"] = euser.ident if euser else ""
  92             result["email"] = euser.email if euser else ""
  93             result["username"] = euser.username if euser else ""
  94             result["ip_address"] = euser.ip_address if euser else ""
  95         return result

  97     def get_raw_data(self, limit=1000, offset=0):
  98         """
  99         Returns list of GroupTagValues
 100         """
 101         return tagstore.get_group_tag_value_iter(
 102             project_id=self.group.project_id,
 103             group_id=self.group.id,
 104             environment_ids=[self.environment_id],
 105             key=self.lookup_key,
 106             callbacks=self.callbacks,
 107             limit=limit,
 108             offset=offset,
 109         )
 110 
 111     def get_serialized_data(self, limit=1000, offset=0):
 112         """
 113         Returns list of serialized GroupTagValue dictionaries
 114         """
 115         raw_data = self.get_raw_data(limit=limit, offset=offset)
 116         return [self.serialize_row(item, self.key) for item in raw_data]
