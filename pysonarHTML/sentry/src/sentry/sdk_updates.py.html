  44             modules=get_path(event_data, "modules"),

  64 class EnableIntegrationSuggestion(Suggestion):
  65     def __init__(self, integration_name, integration_url):
  66         self.integration_name = integration_name
  67         self.integration_url = integration_url
  68 
  69     def to_json(self):
  70         return {
  71             "type": "enableIntegration",
  72             "integrationName": self.integration_name,
  73             "integrationUrl": self.integration_url,
  74         }
  75 
  76     def get_new_state(self, old_state):
  77         if self.integration_name in old_state.integrations:
  78             return old_state
  79 
  80         new_state = old_state.copy()
  81         new_state.integrations.append(self.integration_name)
  82         return new_state

  85 class UpdateSDKSuggestion(Suggestion):
  86     def __init__(self, sdk_name, new_sdk_version):
  87         self.sdk_name = sdk_name
  88         self.new_sdk_version = new_sdk_version
  89 
  90     def to_json(self):
  91         return {
  92             "type": "updateSdk",
  93             "sdkName": self.sdk_name,
  94             "newSdkVersion": self.new_sdk_version,
  95             "sdkUrl": get_sdk_urls().get(self.sdk_name),
  96         }
  97 
  98     def get_new_state(self, old_state):
  99         if self.new_sdk_version is None:
 100             return old_state
 101 
 102         try:
 103             has_newer_version = LooseVersion(old_state.sdk_version) < LooseVersion(
 104                 self.new_sdk_version
 105             )
 106         except Exception:
 107             has_newer_version = False
 108 
 109         if not has_newer_version:
 110             return old_state
 111 
 112         new_state = old_state.copy()
 113         new_state.sdk_version = self.new_sdk_version
 114         return new_state

 117 class ChangeSDKSuggestion(Suggestion):
 118     """
 119     :param module_names: Hide this suggestion if any of the given modules is
 120         loaded. This list is used to weed out invalid suggestions when using
 121         multiple SDKs in e.g. .NET.
 122     """
 123 
 124     def __init__(self, new_sdk_name, module_names=None):
 125         self.new_sdk_name = new_sdk_name
 126         self.module_names = module_names
 127 
 128     def to_json(self):
 129         return {
 130             "type": "changeSdk",
 131             "newSdkName": self.new_sdk_name,
 132             "sdkUrl": get_sdk_urls().get(self.new_sdk_name),
 133         }
 134 
 135     def get_new_state(self, old_state):
 136         if old_state.sdk_name == self.new_sdk_name:
 137             return old_state
 138 
 139         if any(x in old_state.modules for x in self.module_names or ()):
 140             return old_state
 141 
 142         new_state = old_state.copy()
 143         new_state.sdk_name = self.new_sdk_name
 144         return new_state

 331 def get_sdk_index():
 332     value = cache.get(SDK_INDEX_CACHE_KEY)
 333     if value is not None:
 334         return value
 335 
 336     base_url = settings.SENTRY_RELEASE_REGISTRY_BASEURL
 337     if not base_url:
 338         return {}
 339 
 340     url = "%s/sdks" % (base_url,)
 341 
 342     try:
 343         with Session() as session:
 344             response = session.get(url, timeout=1)
 345             response.raise_for_status()
 346             json = response.json()
 347     except Exception:
 348         logger.exception("Failed to fetch version index from release registry")
 349         json = {}
 350 
 351     cache.set(SDK_INDEX_CACHE_KEY, json, 3600)
 352     return json

 355 def get_sdk_versions():
 356     try:
 357         rv = settings.SDK_VERSIONS
 358         rv.update((key, info["version"]) for (key, info) in get_sdk_index().items())
 359         return rv
 360     except Exception:
 361         logger.exception("sentry-release-registry.sdk-versions")
 362         return {}

 365 def get_sdk_urls():
 366     try:
 367         rv = dict(settings.SDK_URLS)
 368         rv.update((key, info["main_docs_url"]) for (key, info) in get_sdk_index().items())
 369         return rv
 370     except Exception:
 371         logger.exception("sentry-release-registry.sdk-urls")
 372         return {}

 375 def _get_suggested_updates_step(setup_state, index_state):
 376     if not setup_state.sdk_name or not setup_state.sdk_version:
 377         return
 378 
 379     yield UpdateSDKSuggestion(
 380         setup_state.sdk_name, index_state.sdk_versions.get(setup_state.sdk_name)
 381     )
 382 
 383     # If an SDK is both outdated and entirely deprecated, we want to inform
 384     # the user of both. It's unclear if they would want to upgrade the SDK
 385     # or migrate to the new one.
 386     newest_name = settings.DEPRECATED_SDKS.get(setup_state.sdk_name, setup_state.sdk_name)
 387     yield ChangeSDKSuggestion(newest_name)
 388 
 389     for support_info in SDK_SUPPORTED_MODULES:
 390         if support_info["sdk_name"] != setup_state.sdk_name and not setup_state.sdk_name.startswith(
 391             support_info["sdk_name"] + "."
 392         ):
 393             continue
 394 
 395         if support_info["module_name"] not in setup_state.modules:
 396             continue
 397 
 398         try:
 399             if LooseVersion(support_info["sdk_version_added"]) > LooseVersion(
 400                 setup_state.sdk_version
 401             ):
 402                 continue
 403         except Exception:
 404             continue
 405 
 406         try:
 407             if LooseVersion(support_info["module_version_min"]) > LooseVersion(
 408                 setup_state.modules[support_info["module_name"]]
 409             ):
 410                 # TODO(markus): Maybe we want to suggest people to upgrade their module?
 411                 #
 412                 # E.g. "please upgrade Django so you can get the Django
 413                 # integration"
 414                 continue
 415         except Exception:
 416             continue
 417 
 418         yield support_info["suggestion"]

 421 def get_suggested_updates(setup_state, index_state=None, parent_suggestions=None):
 422     if index_state is None:
 423         index_state = SdkIndexState()
 424 
 425     if parent_suggestions is None:
 426         parent_suggestions = []
 427 
 428     suggestions = list(_get_suggested_updates_step(setup_state, index_state))
 429 
 430     rv = []
 431     new_setup_states = []
 432 
 433     for suggestion in suggestions:
 434         if suggestion in parent_suggestions:
 435             continue
 436 
 437         new_setup_state = suggestion.get_new_state(setup_state)
 438         if new_setup_state == setup_state:
 439             continue
 440 
 441         rv.append(suggestion)
 442         new_setup_states.append(new_setup_state)
 443 
 444     for new_setup_state, suggestion in zip(new_setup_states, rv):
 445         json = suggestion.to_json()
 446         json["enables"] = list(
 447             get_suggested_updates(
 448                 new_setup_state, parent_suggestions=parent_suggestions + rv, index_state=index_state
 449             )
 450         )
 451 
 452         yield json
