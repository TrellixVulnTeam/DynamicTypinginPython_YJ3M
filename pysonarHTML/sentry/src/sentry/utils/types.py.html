  17 class Type(object):
  18     """Base Type that provides type coersion"""
  19 
  20     name = ""
  21     # Default value to be returned when initializing
  22     default = None
  23     # Types that do not need to be coerced
  24     expected_types = ()
  25     # Types that are acceptable for coersion
  26     compatible_types = six.string_types
  27 
  28     def __call__(self, value=None):
  29         if value is None:
  30             return self._default()
  31         if self.test(value):
  32             return value
  33         if isinstance(value, self.compatible_types):
  34             rv = self.convert(value)
  35             # Make sure convert was able to do the right thing
  36             # and give us the type we were expecting
  37             if self.test(rv):
  38                 return rv
  39         raise InvalidTypeError(u"{!r} is not a valid {}".format(value, repr(self)))
  40 
  41     def convert(self, value):
  42         return value
  43 
  44     def _default(self):
  45         return self.default
  46 
  47     def test(self, value):
  48         """Check if the value is the correct type or not"""
  49         return isinstance(value, self.expected_types)
  50 
  51     def __repr__(self):
  52         return self.name

  63 class BoolType(Type):
  64     "Coerce a boolean from a string"
  65     name = "boolean"
  66     default = False
  67     expected_types = (bool,)
  68     compatible_types = six.string_types + six.integer_types
  69 
  70     def convert(self, value):
  71         if isinstance(value, six.integer_types):
  72             return bool(value)
  73         value = value.lower()
  74         if value in ("y", "yes", "t", "true", "1", "on"):
  75             return True
  76         if value in ("n", "no", "f", "false", "0", "off"):
  77             return False

 118 class DictType(Type):
 119     """Coerce a dict out of a json/yaml string"""
 120 
 121     name = "dictionary"
 122     expected_types = (dict,)
 123 
 124     def _default(self):
 125         # make sure we create a fresh dict each time
 126         return {}
 127 
 128     def convert(self, value):
 129         try:
 130             return safe_load(value)
 131         except (AttributeError, ParserError, ScannerError):
 132             return

 135 class SequenceType(Type):
 136     """Coerce a tuple out of a json/yaml string or a list"""
 137 
 138     name = "sequence"
 139     default = ()
 140     expected_types = (tuple, list)
 141     compatible_types = six.string_types + (tuple, list)
 142 
 143     def convert(self, value):
 144         if isinstance(value, six.string_types):
 145             try:
 146                 value = safe_load(value)
 147             except (AttributeError, ParserError, ScannerError):
 148                 return
 149         if isinstance(value, list):
 150             value = tuple(value)
 151         return value
