  13 class RedisLockBackend(LockBackend):
  14     def __init__(self, cluster, prefix="l:", uuid=None):
  15         if uuid is None:
  16             uuid = uuid4().hex
  17 
  18         self.cluster = cluster
  19         self.prefix = prefix
  20         self.uuid = uuid
  21 
  22     def get_client(self, key, routing_key=None):
  23         # This is a bit of an abstraction leak, but if an integer is provided
  24         # we use that value to determine placement rather than the cluster
  25         # router. This leaking allows us us to have more fine-grained control
  26         # when data is already placed within partitions where the router
  27         # wouldn't have placed it based on the key hash, and maintain data
  28         # locality and failure isolation within those partitions. (For example,
  29         # the entirety of a digest is bound to a specific partition by the
  30         # *digest* key, even though a digest is composed of multiple values at
  31         # different keys that would otherwise be placed on different
  32         # partitions.)
  33         if isinstance(routing_key, six.integer_types):
  34             index = routing_key % len(self.cluster.hosts)
  35             return self.cluster.get_local_client(index)
  36 
  37         if routing_key is not None:
  38             key = routing_key
  39         else:
  40             key = self.prefix_key(key)
  41 
  42         return self.cluster.get_local_client_for_key(key)
  43 
  44     def prefix_key(self, key):
  45         return u"{}{}".format(self.prefix, key)
  46 
  47     def acquire(self, key, duration, routing_key=None):
  48         client = self.get_client(key, routing_key)
  49         full_key = self.prefix_key(key)
  50         if client.set(full_key, self.uuid, ex=duration, nx=True) is not True:
  51             raise Exception(u"Could not set key: {!r}".format(full_key))
  52 
  53     def release(self, key, routing_key=None):
  54         client = self.get_client(key, routing_key)
  55         delete_lock(client, (self.prefix_key(key),), (self.uuid,))
