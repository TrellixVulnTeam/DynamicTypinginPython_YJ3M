   1 from __future__ import absolute_import
   2 
   3 from collections import Hashable, MutableMapping
   4 
   5 __unset__ = object()
   6 
   7 
   8 class BidirectionalMapping(MutableMapping):
   9     """\
  10     An associative data structure in which the ``(key, value)`` pairs form a
  11     one-to-one correspondence in both directions.
  12 
  13     For example, when ``(a, b)`` is added to the mapping, ``b`` can be found
  14     when ``a`` is used as a key, and ``a`` can *also* be found when ``b`` is
  15     provided to ``get_key``.
  16     """
  17 
  18     def __init__(self, data):
  19         self.__data = data
  20         self.__inverse = {v: k for k, v in self.__data.items()}
  21         if len(self.__data) != len(self.__inverse):
  22             raise ValueError("duplicate value provided")
  23 
  24     def __getitem__(self, key):
  25         return self.__data[key]
  26 
  27     def __setitem__(self, key, value):
  28         if not isinstance(key, Hashable):
  29             raise TypeError("key must be hashable")
  30 
  31         if not isinstance(value, Hashable):
  32             raise TypeError("value must be hashable")
  33 
  34         if value in self.__inverse:
  35             raise ValueError("value already present")
  36 
  37         previous = self.__data.pop(key, __unset__)
  38         if previous is not __unset__:
  39             assert self.__inverse.pop(previous) == key
  40 
  41         self.__data[key] = value
  42         self.__inverse[value] = key
  43 
  44     def __delitem__(self, key):
  45         del self.__inverse[self.__data.pop(key)]
  46 
  47     def __iter__(self):
  48         return iter(self.__data)
  49 
  50     def __len__(self):
  51         return len(self.__data)
  52 
  53     def get_key(self, value, default=__unset__):
  54         try:
  55             return self.__inverse[value]
  56         except KeyError:
  57             if default is __unset__:
  58                 raise
  59             else:
  60                 return default
  61 
  62     def inverse(self):
  63         return self.__inverse.copy()
