  39 def _get_tsdb_cache_key(project_id, event_id):

  41     return "is-tsdb-incremented:{}:{}".format(project_id, event_id)

  44 def mark_tsdb_incremented_many(items):
  45     """
  46     Remembers that TSDB was already called for an outcome.
  47 
  48     Sets a boolean flag in memcached to remember that
  49     tsdb_increments_from_outcome was already called for a particular
  50     event/outcome.
  51 
  52     This is used by the outcomes consumer to avoid double-emission.
  53     """
  54     cache.set_many(
  55         dict((_get_tsdb_cache_key(project_id, event_id), True) for project_id, event_id in items),
  56         3600,
  57     )

  60 def mark_tsdb_incremented(project_id, event_id):
  61     mark_tsdb_incremented_many([(project_id, event_id)])

  64 def tsdb_increments_from_outcome(org_id, project_id, key_id, outcome, reason):
  65     if outcome != Outcome.INVALID:
  66         # This simply preserves old behavior. We never counted invalid events
  67         # (too large, duplicate, CORS) toward regular `received` counts.
  68         if project_id is not None:
  69             yield (tsdb.models.project_total_received, project_id)
  70         if org_id is not None:
  71             yield (tsdb.models.organization_total_received, org_id)
  72         if key_id is not None:
  73             yield (tsdb.models.key_total_received, key_id)
  74 
  75     if outcome == Outcome.FILTERED:
  76         if project_id is not None:
  77             yield (tsdb.models.project_total_blacklisted, project_id)
  78         if org_id is not None:
  79             yield (tsdb.models.organization_total_blacklisted, org_id)
  80         if key_id is not None:
  81             yield (tsdb.models.key_total_blacklisted, key_id)
  82 
  83     elif outcome == Outcome.RATE_LIMITED:
  84         if project_id is not None:
  85             yield (tsdb.models.project_total_rejected, project_id)
  86         if org_id is not None:
  87             yield (tsdb.models.organization_total_rejected, org_id)
  88         if key_id is not None:
  89             yield (tsdb.models.key_total_rejected, key_id)
  90 
  91     if reason in FILTER_STAT_KEYS_TO_VALUES:
  92         if project_id is not None:
  93             yield (FILTER_STAT_KEYS_TO_VALUES[reason], project_id)

  96 def track_outcome(
  97     org_id,
  98     project_id,
  99     key_id,
 100     outcome,
 101     reason=None,
 102     timestamp=None,
 103     event_id=None,
 104     category=None,
 105     quantity=None,
 106 ):
 107     """
 108     This is a central point to track org/project counters per incoming event.
 109     NB: This should only ever be called once per incoming event, which means
 110     it should only be called at the point we know the final outcome for the
 111     event (invalid, rate_limited, accepted, discarded, etc.)
 112 
 113     This increments all the relevant legacy RedisTSDB counters, as well as
 114     sending a single metric event to Kafka which can be used to reconstruct the
 115     counters with SnubaTSDB.
 116     """
 117     global outcomes_publisher
 118     if outcomes_publisher is None:
 119         outcomes_publisher = KafkaPublisher(settings.KAFKA_CLUSTERS[outcomes["cluster"]])
 120 
 121     if quantity is None:
 122         quantity = 1
 123 
 124     assert isinstance(org_id, six.integer_types)
 125     assert isinstance(project_id, six.integer_types)
 126     assert isinstance(key_id, (type(None), six.integer_types))
 127     assert isinstance(outcome, Outcome)
 128     assert isinstance(timestamp, (type(None), datetime))
 129     assert isinstance(category, (type(None), DataCategory))
 130     assert isinstance(quantity, int)
 131 
 132     timestamp = timestamp or to_datetime(time.time())
 133 
 134     tsdb_in_consumer = decide_tsdb_in_consumer()
 135 
 136     if not tsdb_in_consumer:
 137         increment_list = list(
 138             tsdb_increments_from_outcome(
 139                 org_id=org_id, project_id=project_id, key_id=key_id, outcome=outcome, reason=reason
 140             )
 141         )
 142 
 143         if increment_list:
 144             tsdb.incr_multi(increment_list, timestamp=timestamp)
 145 
 146         if project_id and event_id:
 147             mark_tsdb_incremented(project_id, event_id)
 148 
 149     # Send a snuba metrics payload.
 150     outcomes_publisher.publish(
 151         outcomes["topic"],
 152         json.dumps(
 153             {
 154                 "timestamp": timestamp,
 155                 "org_id": org_id,
 156                 "project_id": project_id,
 157                 "key_id": key_id,
 158                 "outcome": outcome.value,
 159                 "reason": reason,
 160                 "event_id": event_id,
 161                 "category": category,
 162                 "quantity": quantity,
 163             }
 164         ),
 165     )
 166 
 167     metrics.incr(
 168         "events.outcomes",
 169         skip_internal=True,
 170         tags={"outcome": outcome.name.lower(), "reason": reason},
 171     )
