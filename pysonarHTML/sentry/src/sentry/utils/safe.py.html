  16 def safe_execute(func, *args, **kwargs):
  17     # TODO: we should make smart savepoints (only executing the savepoint server
  18     # side if we execute a query)
  19     _with_transaction = kwargs.pop("_with_transaction", True)
  20     expected_errors = kwargs.pop("expected_errors", None)
  21     try:
  22         if _with_transaction:
  23             with transaction.atomic():
  24                 result = func(*args, **kwargs)
  25         else:
  26             result = func(*args, **kwargs)
  27     except Exception as e:
  28         if hasattr(func, "im_class"):
  29             cls = func.im_class
  30         else:
  31             cls = func.__class__
  32 
  33         func_name = getattr(func, "__name__", six.text_type(func))
  34         cls_name = cls.__name__
  35         logger = logging.getLogger("sentry.safe.%s" % (cls_name.lower(),))
  36 
  37         if expected_errors and isinstance(e, expected_errors):
  38             logger.info("%s.process_error_ignored", func_name, extra={"exception": e})
  39             return
  40         logger.error("%s.process_error", func_name, exc_info=True, extra={"exception": e})
  41     else:
  42         return result

  45 def trim(
  46     value,
  47     max_size=settings.SENTRY_MAX_VARIABLE_SIZE,
  48     max_depth=6,
  49     object_hook=None,
  50     _depth=0,
  51     _size=0,
  52     **kwargs
  53 ):
  54     """
  55     Truncates a value to ```MAX_VARIABLE_SIZE```.
  56 
  57     The method of truncation depends on the type of value.
  58     """
  59     options = {
  60         "max_depth": max_depth,
  61         "max_size": max_size,
  62         "object_hook": object_hook,
  63         "_depth": _depth + 1,
  64     }
  65 
  66     if _depth > max_depth:
  67         if not isinstance(value, six.string_types):
  68             value = json.dumps(value)
  69         return trim(value, _size=_size, max_size=max_size)
  70 
  71     elif isinstance(value, dict):
  72         result = {}
  73         _size += 2
  74         for k in sorted(value.keys(), key=lambda x: (len(force_text(value[x])), x)):
  75             v = value[k]
  76             trim_v = trim(v, _size=_size, **options)
  77             result[k] = trim_v
  78             _size += len(force_text(trim_v)) + 1
  79             if _size >= max_size:
  80                 break
  81 
  82     elif isinstance(value, (list, tuple)):
  83         result = []
  84         _size += 2
  85         for v in value:
  86             trim_v = trim(v, _size=_size, **options)
  87             result.append(trim_v)
  88             _size += len(force_text(trim_v))
  89             if _size >= max_size:
  90                 break
  91         if isinstance(value, tuple):
  92             result = tuple(result)
  93 
  94     elif isinstance(value, six.string_types):
  95         result = truncatechars(value, max_size - _size)
  96 
  97     else:
  98         result = value
  99 
 100     if object_hook is None:
 101         return result
 102     return object_hook(result)

 105 def trim_pairs(iterable, max_items=settings.SENTRY_MAX_DICTIONARY_ITEMS, **kwargs):
 106     max_items -= 1
 107     result = []
 108     for idx, item in enumerate(iterable):
 109         key, value = item
 110         result.append((key, trim(value, **kwargs)))
 111         if idx > max_items:
 112             return result
 113     return result

 116 def trim_dict(value, max_items=settings.SENTRY_MAX_DICTIONARY_ITEMS, **kwargs):
 117     max_items -= 1
 118     for idx, key in enumerate(list(iter(value))):
 119         value[key] = trim(value[key], **kwargs)
 120         if idx > max_items:
 121             del value[key]
 122     return value

 125 def get_path(data, *path, **kwargs):
 126     """
 127     Safely resolves data from a recursive data structure. A value is only
 128     returned if the full path exists, otherwise ``None`` is returned.
 129 
 130     If the ``default`` argument is specified, it is returned instead of ``None``.
 131 
 132     If the ``filter`` argument is specified and the value is a list, it is
 133     filtered with the given callback. Alternatively, pass ``True`` as filter to
 134     only filter ``None`` values.
 135     """
 136     default = kwargs.pop("default", None)
 137     f = kwargs.pop("filter", None)
 138     for k in kwargs:
 139         raise TypeError("set_path() got an undefined keyword argument '%s'" % k)
 140 
 141     for p in path:
 142         if isinstance(data, collections.Mapping) and p in data:
 143             data = data[p]
 144         elif isinstance(data, (list, tuple)) and -len(data) <= p < len(data):
 145             data = data[p]
 146         else:
 147             return default
 148 
 149     if f and data and isinstance(data, (list, tuple)):
 150         data = filter((lambda x: x is not None) if f is True else f, data)
 151 
 152     return data if data is not None else default

 155 def set_path(data, *path, **kwargs):
 156     """
 157     Recursively traverses or creates the specified path and sets the given value
 158     argument. `None` is treated like a missing value. If a non-mapping item is
 159     encountered while traversing, the value is not set.
 160 
 161     This function is equivalent to a recursive dict.__setitem__. Returns True if
 162     the value was set, otherwise False.
 163 
 164     If the ``overwrite` kwarg is set to False, the value is only set if there is
 165     no existing value or it is None. See ``setdefault_path``.
 166     """
 167 
 168     try:
 169         value = kwargs.pop("value")
 170     except KeyError:
 171         raise TypeError("set_path() requires a 'value' keyword argument")
 172 
 173     overwrite = kwargs.pop("overwrite", True)
 174     for k in kwargs:
 175         raise TypeError("set_path() got an undefined keyword argument '%s'" % k)
 176 
 177     for p in path[:-1]:
 178         if not isinstance(data, collections.Mapping):
 179             return False
 180         if data.get(p) is None:
 181             data[p] = {}
 182         data = data[p]
 183 
 184     if not isinstance(data, collections.Mapping):
 185         return False
 186 
 187     p = path[-1]
 188     if overwrite or data.get(p) is None:
 189         data[p] = value
 190         return True
 191 
 192     return False

 195 def setdefault_path(data, *path, **kwargs):
 196     """
 197     Recursively traverses or creates the specified path and sets the given value
 198     argument if it does not exist. `None` is treated like a missing value. If a
 199     non-mapping item is encountered while traversing, the value is not set.
 200 
 201     This function is equivalent to a recursive dict.setdefault, except for None
 202     values. Returns True if the value was set, otherwise False.
 203     """
 204     kwargs["overwrite"] = False
 205     return set_path(data, *path, **kwargs)
