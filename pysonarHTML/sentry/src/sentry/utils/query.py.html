  17 class RangeQuerySetWrapper(object):
  18     """
  19     Iterates through a queryset by chunking results by ``step`` and using GREATER THAN
  20     and LESS THAN queries on the primary key.
  21 
  22     Very efficient, but ORDER BY statements will not work.
  23     """
  24 
  25     def __init__(self, queryset, step=1000, limit=None, min_id=None, order_by="pk", callbacks=()):
  26         # Support for slicing
  27         if queryset.query.low_mark == 0 and not (
  28             queryset.query.order_by or queryset.query.extra_order_by
  29         ):
  30             if limit is None:
  31                 limit = queryset.query.high_mark
  32             queryset.query.clear_limits()
  33         else:
  34             raise InvalidQuerySetError
  35 
  36         self.limit = limit
  37         if limit:
  38             self.step = min(limit, abs(step))
  39             self.desc = step < 0
  40         else:
  41             self.step = abs(step)
  42             self.desc = step < 0
  43         self.queryset = queryset
  44         self.min_value = min_id
  45         self.order_by = order_by
  46         self.callbacks = callbacks
  47 
  48     def __iter__(self):
  49         max_value = None
  50         if self.min_value is not None:
  51             cur_value = self.min_value
  52         else:
  53             cur_value = None
  54 
  55         num = 0
  56         limit = self.limit
  57 
  58         queryset = self.queryset
  59         if self.desc:
  60             queryset = queryset.order_by("-%s" % self.order_by)
  61         else:
  62             queryset = queryset.order_by(self.order_by)
  63 
  64         # we implement basic cursor pagination for columns that are not unique
  65         last_object_pk = None
  66         has_results = True
  67         while has_results:
  68             if (max_value and cur_value >= max_value) or (limit and num >= limit):
  69                 break
  70 
  71             start = num
  72 
  73             if cur_value is None:
  74                 results = queryset
  75             elif self.desc:
  76                 results = queryset.filter(**{"%s__lte" % self.order_by: cur_value})
  77             elif not self.desc:
  78                 results = queryset.filter(**{"%s__gte" % self.order_by: cur_value})
  79 
  80             results = list(results[0 : self.step])
  81 
  82             for cb in self.callbacks:
  83                 cb(results)
  84 
  85             for result in results:
  86                 if last_object_pk is not None and result.pk == last_object_pk:
  87                     continue
  88 
  89                 # Need to bind value before yielding, because the caller
  90                 # may mutate the value and we're left with a bad value.
  91                 # This is commonly the case if iterating over and
  92                 # deleting, because a Model.delete() mutates the `id`
  93                 # to `None` causing the loop to exit early.
  94                 num += 1
  95                 last_object_pk = result.pk
  96                 cur_value = getattr(result, self.order_by)
  97 
  98                 yield result
  99 
 100             if cur_value is None:
 101                 break
 102 
 103             has_results = num > start

 106 class RangeQuerySetWrapperWithProgressBar(RangeQuerySetWrapper):
 107     def __iter__(self):
 108         total_count = self.queryset.count()
 109         if not total_count:
 110             return iter([])
 111         iterator = super(RangeQuerySetWrapperWithProgressBar, self).__iter__()
 112         label = self.queryset.model._meta.verbose_name_plural.title()
 113         return iter(WithProgressBar(iterator, total_count, label))

 116 class WithProgressBar(object):
 117     def __init__(self, iterator, count=None, caption=None):
 118         if count is None and hasattr(iterator, "__len__"):
 119             count = len(iterator)
 120         self.iterator = iterator
 121         self.count = count
 122         self.caption = six.text_type(caption or u"Progress")
 123 
 124     def __iter__(self):
 125         if self.count != 0:
 126             widgets = [
 127                 "%s: " % (self.caption,),
 128                 progressbar.Percentage(),
 129                 " ",
 130                 progressbar.Bar(),
 131                 " ",
 132                 progressbar.ETA(),
 133             ]
 134             pbar = progressbar.ProgressBar(widgets=widgets, maxval=self.count)
 135             pbar.start()
 136             for idx, item in enumerate(self.iterator):
 137                 yield item
 138                 # It's possible that we've exceeded the maxval, but instead
 139                 # of exploding on a ValueError, let's just cap it so we don't.
 140                 # this could happen if new rows were added between calculating `count()`
 141                 # and actually beginning iteration where we're iterating slightly more
 142                 # than we thought.
 143                 pbar.update(min(idx, self.count))
 144             pbar.finish()

 147 def bulk_delete_objects(
 148     model, limit=10000, transaction_id=None, logger=None, partition_key=None, **filters
 149 ):
 150     connection = connections[router.db_for_write(model)]
 151     quote_name = connection.ops.quote_name
 152 
 153     query = []
 154     params = []
 155     partition_query = []
 156 
 157     if partition_key:
 158         for column, value in partition_key.items():
 159             partition_query.append("%s = %%s" % (quote_name(column),))
 160             params.append(value)
 161 
 162     for column, value in filters.items():
 163         query.append("%s = %%s" % (quote_name(column),))
 164         params.append(value)
 165 
 166     query = """
 167         delete from %(table)s
 168         where %(partition_query)s id = any(array(
 169             select id
 170             from %(table)s
 171             where (%(query)s)
 172             limit %(limit)d
 173         ))
 174     """ % dict(
 175         partition_query=(" AND ".join(partition_query)) + (" AND " if partition_query else ""),
 176         query=" AND ".join(query),
 177         table=model._meta.db_table,
 178         limit=limit,
 179     )
 180 
 181     cursor = connection.cursor()
 182     cursor.execute(query, params)
 183 
 184     has_more = cursor.rowcount > 0
 185 
 186     if has_more and logger is not None and _leaf_re.search(model.__name__) is None:
 187         logger.info(
 188             "object.delete.bulk_executed",
 189             extra=dict(
 190                 filters.items() + [("model", model.__name__), ("transaction_id", transaction_id)]
 191             ),
 192         )
 193 
 194     return has_more
