  70 def get_login_url(reset=False):
  71     global _LOGIN_URL
  72 
  73     if _LOGIN_URL is None or reset:
  74         # if LOGIN_URL resolves force login_required to it instead of our own
  75         # XXX: this must be done as late as possible to avoid idempotent requirements
  76         try:
  77             resolve(settings.LOGIN_URL)
  78         except Exception:
  79             _LOGIN_URL = settings.SENTRY_LOGIN_URL
  80         else:
  81             _LOGIN_URL = settings.LOGIN_URL
  82 
  83         if _LOGIN_URL is None:
  84             _LOGIN_URL = reverse("sentry-login")
  85         # ensure type is coerced to string (to avoid lazy proxies)
  86         _LOGIN_URL = six.text_type(_LOGIN_URL)
  87     return _LOGIN_URL

 112 def get_login_redirect(request, default=None):
 113     if default is None:
 114         default = get_login_url()
 115 
 116     # If there is a pending 2fa authentication bound to the session then
 117     # we need to go to the 2fa dialog.
 118     if has_pending_2fa(request):
 119         return reverse("sentry-2fa-dialog")
 120 
 121     # If we have a different URL to go after the 2fa flow we want to go to
 122     # that now here.
 123     after_2fa = request.session.pop("_after_2fa", None)
 124     if after_2fa is not None:
 125         return after_2fa
 126 
 127     login_url = request.session.pop("_next", None)
 128     if not login_url:
 129         return default
 130 
 131     if not is_valid_redirect(login_url, host=request.get_host()):
 132         login_url = default
 133 
 134     return login_url

 137 def is_valid_redirect(url, host=None):
 138     if not url:
 139         return False
 140     if url.startswith(get_login_url()):
 141         return False
 142     return is_safe_url(url, host=host)

 145 def mark_sso_complete(request, organization_id):
 146     # TODO(dcramer): this needs to be bound based on SSO options (e.g. changing
 147     # or enabling SSO invalidates this)
 148     sso = request.session.get(SSO_SESSION_KEY, "")
 149     if sso:
 150         sso = sso.split(",")
 151     else:
 152         sso = []
 153     sso.append(six.text_type(organization_id))
 154     request.session[SSO_SESSION_KEY] = ",".join(sso)
 155     request.session.modified = True

 163 def find_users(username, with_valid_password=True, is_active=None):
 164     """
 165     Return a list of users that match a username
 166     and falling back to email
 167     """
 168     qs = User.objects
 169 
 170     if is_active is not None:
 171         qs = qs.filter(is_active=is_active)
 172 
 173     if with_valid_password:
 174         qs = qs.exclude(password="!")
 175 
 176     try:
 177         # First, assume username is an iexact match for username
 178         user = qs.get(username__iexact=username)
 179         return [user]
 180     except User.DoesNotExist:
 181         # If not, we can take a stab at guessing it's an email address
 182         if "@" in username:
 183             # email isn't guaranteed unique
 184             return list(qs.filter(email__iexact=username))
 185     return []

 188 def login(request, user, passed_2fa=None, after_2fa=None, organization_id=None, source=None):
 189     """
 190     This logs a user in for the session and current request.
 191 
 192     If 2FA is enabled this method will start the MFA flow and return False as
 193     required.  If `passed_2fa` is set to `True` then the 2FA flow is set to be
 194     finalized (user passed the flow).
 195 
 196     If the session has already resolved MFA in the past, it will automatically
 197     detect it from the session.
 198 
 199     Optionally `after_2fa` can be set to a URL which will be used to override
 200     the regular session redirect target directly after the 2fa flow.
 201 
 202     Returns boolean indicating if the user was logged in.
 203     """
 204     has_2fa = Authenticator.objects.user_has_2fa(user)
 205     if passed_2fa is None:
 206         passed_2fa = request.session.get(MFA_SESSION_KEY, "") == six.text_type(user.id)
 207 
 208     if has_2fa and not passed_2fa:
 209         request.session["_pending_2fa"] = [user.id, time(), organization_id]
 210         if after_2fa is not None:
 211             request.session["_after_2fa"] = after_2fa
 212         request.session.modified = True
 213         return False
 214 
 215     # TODO(dcramer): this needs to be bound based on MFA options
 216     if passed_2fa:
 217         request.session[MFA_SESSION_KEY] = six.text_type(user.id)
 218         request.session.modified = True
 219 
 220     mfa_state = request.session.pop("_pending_2fa", ())
 221     if organization_id is None and len(mfa_state) == 3:
 222         organization_id = mfa_state[2]
 223 
 224     # Check for expired passwords here after we cleared the 2fa flow.
 225     # While this means that users will have to pass 2fa before they can
 226     # figure out that their passwords are expired this is still the more
 227     # reasonable behavior.
 228     #
 229     # We also remember _after_2fa here so that we can continue the flow if
 230     # someone does it in the same browser.
 231     if user.is_password_expired:
 232         raise AuthUserPasswordExpired(user)
 233 
 234     # If this User has a nonce value, we need to bind into the session.
 235     if user.session_nonce is not None:
 236         request.session["_nonce"] = user.session_nonce
 237 
 238     # If there is no authentication backend, just attach the first
 239     # one and hope it goes through.  This apparently is a thing we
 240     # have been doing for a long time, just moved it to a more
 241     # reasonable place.
 242     if not hasattr(user, "backend"):
 243         user.backend = settings.AUTHENTICATION_BACKENDS[0]
 244     _login(request, user)
 245     if organization_id:
 246         mark_sso_complete(request, organization_id)
 247     log_auth_success(request, user.username, organization_id, source)
 248     return True
