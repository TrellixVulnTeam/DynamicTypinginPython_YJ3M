  22 def _pack_int(i):
  23     result = bytearray()
  24     while i != 0:
  25         result.append(i & 0xFF)
  26         i >>= 8
  27     return six.binary_type(bytearray(reversed(result)).rjust(8, b"\0"))

  38 class TOTP(object):
  39     def __init__(self, secret=None, digits=6, interval=30, default_window=2):
  40         if secret is None:
  41             secret = generate_secret_key()
  42         if len(secret) % 8 != 0:
  43             raise RuntimeError("Secret length needs to be a multiple of 8")
  44         self.secret = secret
  45         self.digits = digits
  46         self.interval = interval
  47         self.default_window = default_window
  48 
  49     def generate_otp(self, ts=None, offset=0, counter=None):
  50         if counter is None:
  51             ts = _get_ts(ts)
  52             counter = int(ts) // self.interval + offset
  53         h = bytearray(
  54             hmac.HMAC(
  55                 base64.b32decode(self.secret.encode("ascii"), casefold=True),
  56                 _pack_int(counter),
  57                 hashlib.sha1,
  58             ).digest()
  59         )
  60         offset = h[-1] & 0xF
  61         code = (
  62             (h[offset] & 0x7F) << 24
  63             | (h[offset + 1] & 0xFF) << 16
  64             | (h[offset + 2] & 0xFF) << 8
  65             | (h[offset + 3] & 0xFF)
  66         )
  67         str_code = six.text_type(code % 10 ** self.digits)
  68         return ("0" * (self.digits - len(str_code))) + str_code

  70     def verify(self, otp, ts=None, window=None, return_counter=False, check_counter_func=None):
  71         ts = _get_ts(ts)
  72         if window is None:
  73             window = self.default_window
  74         for i in range(-window, window + 1):
  75             counter = int(ts) // self.interval + i
  76             if constant_time_compare(otp, self.generate_otp(counter=counter)):
  77                 # Check for blacklisted counters after the constant time
  78                 # compare
  79                 if check_counter_func is not None and not check_counter_func(counter):
  80                     continue
  81                 if return_counter:
  82                     return counter
  83                 return True
  84         if return_counter:
  85             return None
  86         return False

  88     def get_provision_url(self, user, issuer=None):
  89         if issuer is None:
  90             issuer = "Sentry"
  91         rv = "otpauth://totp/%s?issuer=%s&secret=%s" % (
  92             quote(user.encode("utf-8")),
  93             quote(issuer.encode("utf-8")),
  94             self.secret,
  95         )
  96         if self.digits != 6:
  97             rv += "&digits=%d" % self.digits
  98         if self.interval != 30:
  99             rv += "&period=%d" % self.interval
 100         return rv
