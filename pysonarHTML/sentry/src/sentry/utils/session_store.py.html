  11 class RedisSessionStore(object):
  12     """
  13     RedisSessionStore provides a convenience object, which when initalized will
  14     store attributes assigned to it into redis. The redis key is stored into
  15     the request session. Useful for storing data too large to be stored into
  16     the session cookie.
  17 
  18     NOTE: Assigning attributes immediately saves their value back into the
  19           redis key assigned for this store. Be aware of the multiple
  20           round-trips implication of this.
  21 
  22     NOTE: This object is subject to race conditions on updating valeus as the
  23           entire object value is stored in one redis key.
  24 
  25     >>> store = RedisSessionStore(request, 'store-name')
  26     >>> store.regenerate()
  27     >>> store.some_value = 'my value'
  28 
  29     The value will be available across requests as long as the same same store
  30     name is used.
  31 
  32     >>> store.some_value
  33     'my value'
  34 
  35     The store may be destroyed before it expires using the ``clear`` method.
  36 
  37     >>> store.clear()
  38 
  39     It's important to note that the session store will expire if values are not
  40     modified within the provided ttl.
  41     """
  42 
  43     def __init__(self, request, prefix, ttl=EXPIRATION_TTL):
  44         self.__dict__["request"] = request
  45         self.__dict__["prefix"] = prefix
  46         self.__dict__["ttl"] = ttl
  47 
  48     @property
  49     def _client(self):
  50         return clusters.get("default").get_local_client_for_key(self.redis_key)
  51 
  52     @property
  53     def session_key(self):
  54         return u"store:{}".format(self.prefix)
  55 
  56     @property
  57     def redis_key(self):
  58         return self.request.session.get(self.session_key)
  59 
  60     def regenerate(self, initial_state=None):
  61         if initial_state is None:
  62             initial_state = {}
  63 
  64         redis_key = u"session-cache:{}:{}".format(self.prefix, uuid4().hex)
  65 
  66         self.request.session[self.session_key] = redis_key
  67 
  68         value = dumps(initial_state)
  69         self._client.setex(redis_key, self.ttl, value)
  70 
  71     def clear(self):
  72         if not self.redis_key:
  73             return
  74 
  75         self._client.delete(self.redis_key)
  76         del self.request.session[self.session_key]
  77 
  78     def is_valid(self):
  79         return self.redis_key and self._client.get(self.redis_key)
  80 
  81     def get_state(self):
  82         if not self.redis_key:
  83             return None
  84 
  85         state_json = self._client.get(self.redis_key)
  86         if not state_json:
  87             return None
  88 
  89         return loads(state_json)
  90 
  91     def __getattr__(self, key):
  92         state = self.get_state()
  93 
  94         try:
  95             return state[key] if state else None
  96         except KeyError as e:
  97             raise AttributeError(e)
  98 
  99     def __setattr__(self, key, value):
 100         state = self.get_state()
 101 
 102         if state is None:
 103             return
 104 
 105         state[key] = value
 106         self._client.setex(self.redis_key, self.ttl, dumps(state))
