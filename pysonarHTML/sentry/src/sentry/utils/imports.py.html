   7 class ModuleProxyCache(dict):
   8     def __missing__(self, key):
   9         if "." not in key:
  10             return __import__(key)
  11 
  12         module_name, class_name = key.rsplit(".", 1)
  13 
  14         module = __import__(module_name, {}, {}, [class_name])
  15         handler = getattr(module, class_name)
  16 
  17         # We cache a NoneType for missing imports to avoid repeated lookups
  18         self[key] = handler
  19 
  20         return handler
  21 
  22 
  23 _cache = ModuleProxyCache()

  26 def import_string(path):

  28     Path must be module.path.ClassName
  29 
  30     >>> cls = import_string('sentry.models.Group')
  31     """

  36 def import_submodules(context, root_module, path):
  37     """

  39 
  40     >>> import_submodules(locals(), __name__, __path__)
  41     """

  46         for k, v in six.iteritems(vars(module)):
  47             if not k.startswith("_"):
