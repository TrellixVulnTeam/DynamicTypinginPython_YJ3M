  44 def inline_css(value):
  45     tree = lxml.html.document_fromstring(value)
  46     toronado.inline(tree)
  47     # CSS media query support is inconsistent when the DOCTYPE declaration is
  48     # missing, so we force it to HTML5 here.
  49     return lxml.html.tostring(tree, doctype="<!DOCTYPE html>")

  52 class _CaseInsensitiveSigner(Signer):
  53     """
  54     Generate a signature that is comprised of only lowercase letters.
  55 
  56     WARNING: Do not use this for anything that needs to be cryptographically
  57     secure! This is losing entropy and has a much higher chance of collision
  58     due to dropping to lowercase letters. For our purposes, this lack of entropy
  59     is ok and doesn't pose a risk.
  60 
  61     NOTE: This is needed strictly for signatures used in email addresses. Some
  62     clients, coughAirmailcough, treat email addresses as being case-insensitive,
  63     and sends the value as all lowercase.
  64     """
  65 
  66     def signature(self, value):
  67         sig = super(_CaseInsensitiveSigner, self).signature(value)
  68         return sig.lower()
  69 
  70     def unsign(self, signed_value):
  71         # This unsign is identical to subclass except for the lowercasing
  72         # See: https://github.com/django/django/blob/1.6.11/django/core/signing.py#L165-L172
  73         signed_value = force_str(signed_value)
  74         if self.sep not in signed_value:
  75             raise BadSignature('No "%s" found in value' % self.sep)
  76         value, sig = signed_value.rsplit(self.sep, 1)
  77         if constant_time_compare(sig.lower(), self.signature(value)):
  78             return force_text(value)
  79         raise BadSignature('Signature "%s" does not match' % sig)

  85 def email_to_group_id(address):
  86     """
  87     Email address should be in the form of:
  88         {group_id}+{signature}@example.com
  89     """
  90     address = address.split("@", 1)[0]
  91     signed_data = address.replace("+", ":")
  92     return int(force_bytes(signer.unsign(signed_data)))

  95 def group_id_to_email(group_id):
  96     signed_data = signer.sign(six.text_type(group_id))
  97     return "@".join(
  98         (
  99             signed_data.replace(":", "+"),
 100             options.get("mail.reply-hostname") or get_from_email_domain(),
 101         )

 105 def domain_from_email(email):
 106     email = parseaddr(email)[1]
 107     try:
 108         return email.split("@", 1)[1]
 109     except IndexError:
 110         # The email address is likely malformed or something
 111         return email

 114 # Slightly modified version of Django's
 115 # `django.core.mail.message:make_msgid` because we need
 116 # to override the domain. If we ever upgrade to
 117 # django 1.8, we can/should replace this.
 118 def make_msgid(domain):
 119     """Returns a string suitable for RFC 2822 compliant Message-ID, e.g:
 120     <20020201195627.33539.96671@nightshade.la.mastaler.com>
 121     Optional idstring if given is a string used to strengthen the
 122     uniqueness of the message id.  Optional domain if given provides the
 123     portion of the message id after the '@'.  It defaults to the locally
 124     defined hostname.
 125     """
 126     timeval = time.time()
 127     utcdate = time.strftime("%Y%m%d%H%M%S", time.gmtime(timeval))
 128     pid = os.getpid()
 129     randint = randrange(100000)
 130     msgid = "<%s.%s.%s@%s>" % (utcdate, pid, randint, domain)
 131     return msgid

 139 def get_from_email_domain():
 140     global _from_email_domain_cache
 141     from_ = options.get("mail.from")

 147 def create_fake_email(unique_id, namespace):
 148     """
 149     Generate a fake email of the form: {unique_id}@{namespace}{FAKE_EMAIL_TLD}
 150 
 151     For example: c74e5b75-e037-4e75-ad27-1a0d21a6b203@cloudfoundry.sentry-fake
 152     """
 153     return u"{}@{}{}".format(unique_id, namespace, FAKE_EMAIL_TLD)

 163 def get_email_addresses(user_ids, project=None):
 164     pending = set(user_ids)
 165     results = {}
 166 
 167     if project:
 168         queryset = UserOption.objects.filter(project=project, user__in=pending, key="mail:email")
 169         for option in (o for o in queryset if o.value and not is_fake_email(o.value)):
 170             results[option.user_id] = option.value
 171             pending.discard(option.user_id)
 172 
 173     if pending:
 174         queryset = User.objects.filter(pk__in=pending, is_active=True)
 175         for (user_id, email) in queryset.values_list("id", "email"):
 176             if email and not is_fake_email(email):
 177                 results[user_id] = email
 178                 pending.discard(user_id)
 179 
 180     if pending:
 181         logger.warning(
 182             "Could not resolve email addresses for user IDs in %r, discarding...", pending
 183         )
 184 
 185     return results

 188 class ListResolver(object):
 189     """
 190     Manages the generation of RFC 2919 compliant list-id strings from varying
 191     objects types.
 192     """
 193 
 194     class UnregisteredTypeError(Exception):
 195         """
 196         Error raised when attempting to build a list-id from an unregisted object type.
 197         """
 198 
 199     def __init__(self, namespace, type_handlers):
 200         assert is_valid_dot_atom(namespace)
 201 
 202         # The list-id-namespace that will be used when generating the list-id
 203         # string. This should be a domain name under the control of the
 204         # generator (see RFC 2919.)
 205         self.__namespace = namespace
 206 
 207         # A mapping of classes to functions that accept an instance of that
 208         # class, returning a tuple of values that will be used to generate the
 209         # list label. Returned values must be valid RFC 2822 dot-atom-text
 210         # values.
 211         self.__type_handlers = type_handlers
 212 
 213     def __call__(self, instance):
 214         """
 215         Build a list-id string from an instance.
 216 
 217         Raises ``UnregisteredTypeError`` if there is no registered handler for
 218         the instance type. Raises ``AssertionError`` if a valid list-id string
 219         cannot be generated from the values returned by the type handler.
 220         """
 221         try:
 222             handler = self.__type_handlers[type(instance)]
 223         except KeyError:
 224             raise self.UnregisteredTypeError(
 225                 u"Cannot generate mailing list identifier for {!r}".format(instance)
 226             )
 227 
 228         label = ".".join(map(six.text_type, handler(instance)))
 229         assert is_valid_dot_atom(label)
 230 
 231         return u"{}.{}".format(label, self.__namespace)
 232 
 233 
 234 default_list_type_handlers = {
 235     Activity: attrgetter("project.slug", "project.organization.slug"),
 236     Project: attrgetter("slug", "organization.slug"),
 237     Group: attrgetter("project.slug", "organization.slug"),
 238 }
 239 
 240 make_listid_from_instance = ListResolver(
 241     options.get("mail.list-namespace"), default_list_type_handlers
 242 )

 245 class MessageBuilder(object):
 246     def __init__(
 247         self,
 248         subject,
 249         context=None,
 250         template=None,
 251         html_template=None,
 252         body="",
 253         html_body=None,
 254         headers=None,
 255         reference=None,
 256         reply_reference=None,
 257         from_email=None,
 258         type=None,
 259     ):
 260         assert not (body and template)
 261         assert not (html_body and html_template)
 262         assert context or not (template or html_template)
 263 
 264         if headers is None:
 265             headers = {}
 266 
 267         self.subject = subject
 268         self.context = context or {}
 269         self.template = template
 270         self.html_template = html_template
 271         self._txt_body = body
 272         self._html_body = html_body
 273         self.headers = headers
 274         self.reference = reference  # The object that generated this message
 275         self.reply_reference = reply_reference  # The object this message is replying about
 276         self.from_email = from_email or options.get("mail.from")
 277         self._send_to = set()
 278         self.type = type if type else "generic"
 279 
 280         if reference is not None and "List-Id" not in headers:
 281             try:
 282                 headers["List-Id"] = make_listid_from_instance(reference)
 283             except ListResolver.UnregisteredTypeError as error:
 284                 logger.debug(six.text_type(error))
 285             except AssertionError as error:
 286                 logger.warning(six.text_type(error))
 287 
 288     def __render_html_body(self):
 289         html_body = None
 290         if self.html_template:
 291             html_body = render_to_string(self.html_template, self.context)
 292         else:
 293             html_body = self._html_body
 294 
 295         if html_body is not None:
 296             return inline_css(html_body)
 297 
 298     def __render_text_body(self):
 299         if self.template:
 300             return render_to_string(self.template, self.context)
 301         return self._txt_body
 302 
 303     def add_users(self, user_ids, project=None):
 304         self._send_to.update(get_email_addresses(user_ids, project).values())
 305 
 306     def build(self, to, reply_to=None, cc=None, bcc=None):
 307         if self.headers is None:
 308             headers = {}
 309         else:
 310             headers = self.headers.copy()
 311 
 312         if options.get("mail.enable-replies") and "X-Sentry-Reply-To" in headers:
 313             reply_to = headers["X-Sentry-Reply-To"]
 314         else:
 315             reply_to = set(reply_to or ())
 316             reply_to.discard(to)
 317             reply_to = ", ".join(reply_to)
 318 
 319         if reply_to:
 320             headers.setdefault("Reply-To", reply_to)
 321 
 322         # Every message sent needs a unique message id
 323         message_id = make_msgid(get_from_email_domain())
 324         headers.setdefault("Message-Id", message_id)
 325 
 326         subject = self.subject
 327 
 328         if self.reply_reference is not None:
 329             reference = self.reply_reference
 330             subject = "Re: %s" % subject
 331         else:
 332             reference = self.reference
 333 
 334         if isinstance(reference, Group):
 335             thread, created = GroupEmailThread.objects.get_or_create(
 336                 email=to,
 337                 group=reference,
 338                 defaults={"project": reference.project, "msgid": message_id},
 339             )
 340             if not created:
 341                 headers.setdefault("In-Reply-To", thread.msgid)
 342                 headers.setdefault("References", thread.msgid)
 343 
 344         msg = EmailMultiAlternatives(
 345             subject=subject.splitlines()[0],
 346             body=self.__render_text_body(),
 347             from_email=self.from_email,
 348             to=(to,),
 349             cc=cc or (),
 350             bcc=bcc or (),
 351             headers=headers,
 352         )
 353 
 354         html_body = self.__render_html_body()
 355         if html_body:
 356             msg.attach_alternative(html_body.decode("utf-8"), "text/html")
 357 
 358         return msg
 359 
 360     def get_built_messages(self, to=None, cc=None, bcc=None):
 361         send_to = set(to or ())
 362         send_to.update(self._send_to)
 363         results = [
 364             self.build(to=email, reply_to=send_to, cc=cc, bcc=bcc) for email in send_to if email
 365         ]
 366         if not results:
 367             logger.debug("Did not build any messages, no users to send to.")
 368         return results

 370     def format_to(self, to):
 371         if not to:
 372             return ""
 373         if len(to) > MAX_RECIPIENTS:
 374             to = to[:MAX_RECIPIENTS] + [u"and {} more.".format(len(to[MAX_RECIPIENTS:]))]
 375         return ", ".join(to)
 376 
 377     def send(self, to=None, cc=None, bcc=None, fail_silently=False):
 378         return send_messages(
 379             self.get_built_messages(to, cc=cc, bcc=bcc), fail_silently=fail_silently
 380         )
 381 
 382     def send_async(self, to=None, cc=None, bcc=None):
 383         from sentry.tasks.email import send_email
 384 
 385         fmt = options.get("system.logging-format")
 386         messages = self.get_built_messages(to, cc=cc, bcc=bcc)
 387         extra = {"message_type": self.type}
 388         loggable = [v for k, v in six.iteritems(self.context) if hasattr(v, "id")]
 389         for context in loggable:
 390             extra["%s_id" % type(context).__name__.lower()] = context.id
 391 
 392         log_mail_queued = partial(logger.info, "mail.queued", extra=extra)
 393         for message in messages:
 394             safe_execute(send_email.delay, message=message, _with_transaction=False)
 395             extra["message_id"] = message.extra_headers["Message-Id"]
 396             metrics.incr("email.queued", instance=self.type, skip_internal=False)
 397             if fmt == LoggingFormat.HUMAN:
 398                 extra["message_to"] = (self.format_to(message.to),)
 399                 log_mail_queued()
 400             elif fmt == LoggingFormat.MACHINE:
 401                 for recipient in message.to:
 402                     extra["message_to"] = recipient
 403                     log_mail_queued()

 406 def send_messages(messages, fail_silently=False):
 407     connection = get_connection(fail_silently=fail_silently)
 408     sent = connection.send_messages(messages)
 409     metrics.incr("email.sent", len(messages), skip_internal=False)
 410     for message in messages:
 411         extra = {
 412             "message_id": message.extra_headers["Message-Id"],
 413             "size": len(message.message().as_bytes()),
 414         }
 415         logger.info("mail.sent", extra=extra)
 416     return sent

 443 def send_mail(subject, message, from_email, recipient_list, fail_silently=False, **kwargs):
 444     """
 445     Wrapper that forces sending mail through our connection.
 446     Uses EmailMessage class which has more options than the simple send_mail
 447     """
 448     email = mail.EmailMessage(
 449         subject,
 450         message,
 451         from_email,
 452         recipient_list,
 453         connection=get_connection(fail_silently=fail_silently),
 454         **kwargs
 455     )
 456     return email.send(fail_silently=fail_silently)

 468 class PreviewBackend(BaseEmailBackend):
 469     """
 470     Email backend that can be used in local development to open messages in the
 471     local mail client as they are sent.
 472 
 473     Probably only works on OS X.
 474     """
 475 
 476     def send_messages(self, email_messages):
 477         for message in email_messages:
 478             content = six.binary_type(message.message())
 479             preview = tempfile.NamedTemporaryFile(
 480                 delete=False, prefix="sentry-email-preview-", suffix=".eml"
 481             )
 482             try:
 483                 preview.write(content)
 484                 preview.flush()
 485             finally:
 486                 preview.close()
 487 
 488             subprocess.check_call(("open", preview.name))
 489 
 490         return len(email_messages)
