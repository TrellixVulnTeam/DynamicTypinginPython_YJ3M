   8 class ValueFilter(filters.TokenFilter):
   9     include = (T.String, T.Number)
  10     exclude = (T.String.Symbol,)
  11 
  12     def process(self, stack, stream):
  13         for ttype, value in stream:
  14             parent = ttype
  15             while parent:
  16                 if parent in self.exclude:
  17                     break
  18                 if parent in self.include:
  19                     value = "?"
  20                     break
  21                 if ttype.parent == parent:
  22                     parent = None
  23                 else:
  24                     parent = ttype.parent
  25             yield ttype, value
