  12 class DeprecatedSettingWarning(DeprecationWarning):
  13     def __init__(self, setting, replacement, url=None, removed_in_version=None):
  14         self.setting = setting
  15         self.replacement = replacement
  16         self.url = url
  17         self.removed_in_version = removed_in_version
  18         super(DeprecatedSettingWarning, self).__init__(setting, replacement, url)
  19 
  20     def __str__(self):
  21         chunks = [
  22             u"The {} setting is deprecated. Please use {} instead.".format(
  23                 self.setting, self.replacement
  24             )
  25         ]
  26 
  27         if self.removed_in_version:
  28             chunks.append(
  29                 u"This setting will be removed in Sentry {}.".format(self.removed_in_version)
  30             )
  31 
  32         # TODO(tkaemming): This will be removed from the message in the future
  33         # when it's added to the API payload separately.
  34         if self.url:
  35             chunks.append(u"See {} for more information.".format(self.url))
  36 
  37         return " ".join(chunks)

  40 class WarningManager(object):
  41     """
  42     Transforms warnings into a standard form and invokes handlers.
  43     """
  44 
  45     def __init__(self, handlers, default_category=Warning):
  46         self.__handlers = handlers
  47         self.__default_category = default_category
  48 
  49     def warn(self, message, category=None, stacklevel=None):
  50         if isinstance(message, Warning):
  51             # Maybe log if `category` was passed and isn't a subclass of
  52             # `type(message)`?
  53             warning = message
  54         else:
  55             if category is None:
  56                 category = self.__default_category
  57 
  58             assert issubclass(category, Warning)
  59             warning = category(message)
  60 
  61         kwargs = {}
  62         if stacklevel is not None:
  63             kwargs["stacklevel"] = stacklevel
  64 
  65         for handler in self.__handlers:
  66             handler(warning, **kwargs)

  69 class WarningSet(collections.Set):
  70     """
  71     Add-only set structure for storing unique warnings.
  72     """
  73 
  74     def __init__(self):
  75         self.__warnings = {}
  76 
  77     def __contains__(self, value):
  78         assert isinstance(value, Warning)
  79         return self.__get_key(value) in self.__warnings
  80 
  81     def __len__(self):
  82         return len(self.__warnings)
  83 
  84     def __iter__(self):
  85         return six.itervalues(self.__warnings)
  86 
  87     def __get_key(self, warning):
  88         return (type(warning), warning.args if hasattr(warning, "args") else six.text_type(warning))
  89 
  90     def add(self, warning, stacklevel=None):
  91         self.__warnings[self.__get_key(warning)] = warning

  97 manager = WarningManager(
