  17 def reraise_as(new_exception_or_type):
  18     """
  19     Obtained from https://github.com/dcramer/reraise/blob/master/src/reraise.py
  20     >>> try:
  21     >>>     do_something_crazy()
  22     >>> except Exception:
  23     >>>     reraise_as(UnhandledException)
  24     """
  25     __traceback_hide__ = True  # NOQA
  26 
  27     e_type, e_value, e_traceback = sys.exc_info()
  28 
  29     if inspect.isclass(new_exception_or_type):
  30         new_type = new_exception_or_type
  31         new_exception = new_exception_or_type()
  32     else:
  33         new_type = type(new_exception_or_type)
  34         new_exception = new_exception_or_type
  35 
  36     new_exception.__cause__ = e_value
  37 
  38     try:
  39         six.reraise(new_type, new_exception, e_traceback)
  40     finally:
  41         del e_traceback

  44 def validate_settings(settings):
  45     for key, engine_key, engine_type in [
  46         ("DATABASES", "ENGINE", "database engine"),
  47         ("CACHES", "BACKEND", "caching backend"),
  48     ]:
  49 
  50         value = getattr(settings, key, {})
  51         for alias in value:
  52             engine = value[alias][engine_key]
  53             if engine not in PACKAGES:
  54                 continue
  55             validate_dependency(settings, engine_type, engine, PACKAGES[engine])

  58 def validate_dependency(settings, dependency_type, dependency, package):
  59     try:
  60         import_string(package)
  61     except ImportError:
  62         msg = ConfigurationError.get_error_message("%s %s" % (dependency_type, dependency), package)
  63         reraise_as(ConfigurationError(msg))

  66 class ConfigurationError(ValueError):
  67     """
  68     This error is thrown whenever a sentry configuration is wrong, or requires a third-party library
  69     that's not installed properly or can't be found.
  70     """
  71 
  72     @classmethod
  73     def get_error_message(cls, dependency, package):
  74         return (
  75             """Python could not find %(package)s in your current environment (required by %(dependency)s). If you have it installed, maybe you are using the wrong python binary to run sentry?"""
  76             % {"dependency": dependency, "package": package}
  77         )
