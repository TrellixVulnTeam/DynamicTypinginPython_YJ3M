  27 class SentryAppWebhookRequestsBuffer(object):
  28     """
  29     Create a data structure to store basic information about Sentry App webhook requests in Redis
  30     This should store the last 100 requests and last 100 errors (in different keys) for each event type, for each Sentry App
  31     """
  32 
  33     def __init__(self, sentry_app):
  34         self.sentry_app = sentry_app
  35 
  36         cluster_id = getattr(settings, "SENTRY_WEBHOOK_LOG_REDIS_CLUSTER", "default")
  37         self.client = redis.redis_clusters.get(cluster_id)
  38 
  39     def _get_redis_key(self, event, error=False):
  40         sentry_app_id = self.sentry_app.id
  41 
  42         if error:
  43             return "sentry-app-webhook-error:{{{0}}}:{1}".format(sentry_app_id, event)
  44         else:
  45             return "sentry-app-webhook-request:{{{0}}}:{1}".format(sentry_app_id, event)
  46 
  47     def _convert_redis_request(self, redis_request, event):
  48         """
  49         Convert the request string stored in Redis to a python dict
  50         Add the event type to the dict so that the request can be identified correctly
  51         """
  52         request = json.loads(redis_request)
  53         request["event_type"] = event
  54 
  55         return request
  56 
  57     def _add_to_buffer_pipeline(self, buffer_key, item, pipeline):
  58         """
  59         Add the item to the buffer key specified, using the given pipeline.
  60         This does not execute the pipeline's commands.
  61         """
  62 
  63         pipeline.lpush(buffer_key, json.dumps(item))
  64         pipeline.ltrim(buffer_key, 0, BUFFER_SIZE - 1)
  65         pipeline.expire(buffer_key, KEY_EXPIRY)
  66 
  67     def _get_all_from_buffer(self, buffer_key, pipeline=None):
  68         """
  69         Get the list at the buffer key, using the given pipeline if available.
  70         If a pipeline is provided, this does not return a value as the pipeline must still be executed.
  71         """
  72 
  73         if pipeline is not None:
  74             pipeline.lrange(buffer_key, 0, BUFFER_SIZE - 1)
  75         else:
  76             return self.client.lrange(buffer_key, 0, BUFFER_SIZE - 1)
  77 
  78     def _get_requests(self, event=None, error=False):
  79         # If no event is specified, return the latest requests/errors for all event types
  80         if event is None:
  81             pipe = self.client.pipeline()
  82 
  83             all_requests = []
  84             for evt in EXTENDED_VALID_EVENTS:
  85                 self._get_all_from_buffer(self._get_redis_key(evt, error=error), pipeline=pipe)
  86 
  87             values = pipe.execute()
  88 
  89             for idx, evt in enumerate(EXTENDED_VALID_EVENTS):
  90                 event_requests = [
  91                     self._convert_redis_request(request, evt) for request in values[idx]
  92                 ]
  93                 all_requests.extend(event_requests)
  94 
  95             all_requests.sort(key=lambda x: parse_date(x["date"]), reverse=True)
  96             return all_requests[0:BUFFER_SIZE]
  97 
  98         else:
  99             return [
 100                 self._convert_redis_request(request, event)
 101                 for request in self._get_all_from_buffer(self._get_redis_key(event, error=error))
 102             ]
 103 
 104     def get_requests(self, event=None, errors_only=False):
 105         return self._get_requests(event=event, error=errors_only)
 106 
 107     def add_request(self, response_code, org_id, event, url, error_id=None, project_id=None):
 108         if event not in EXTENDED_VALID_EVENTS:
 109             return
 110 
 111         request_key = self._get_redis_key(event)
 112 
 113         time = timezone.now()
 114         request_data = {
 115             "date": six.text_type(time),
 116             "response_code": response_code,
 117             "webhook_url": url,
 118         }
 119 
 120         # Don't store the org id for internal apps because it will always be the org that owns the app anyway
 121         if not self.sentry_app.is_internal:
 122             request_data["organization_id"] = org_id
 123 
 124         # We need both the error ID and project ID to link the error
 125         if error_id is not None and project_id is not None:
 126             request_data["error_id"] = error_id
 127             request_data["project_id"] = project_id
 128 
 129         pipe = self.client.pipeline()
 130 
 131         self._add_to_buffer_pipeline(request_key, request_data, pipe)
 132 
 133         # If it's an error add it to the error buffer
 134         if 400 <= response_code <= 599 or response_code == 0:
 135             error_key = self._get_redis_key(event, error=True)
 136             self._add_to_buffer_pipeline(error_key, request_data, pipe)
 137 
 138         pipe.execute()
