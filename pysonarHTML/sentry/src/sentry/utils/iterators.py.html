  10 def advance(n, iterator):
  11     """Advances an iterator n places."""
  12     next(itertools.islice(iterator, n, n), None)
  13     return iterator

  16 def shingle(n, iterator):
  17     """\
  18     Shingle a token stream into N-grams.
  19 
  20     >>> list(shingle(2, ('foo', 'bar', 'baz')))
  21     [('foo', 'bar'), ('bar', 'baz')]
  22     """
  23     return zip(
  24         *map(
  25             lambda i__iterator: advance(i__iterator[0], i__iterator[1]),
  26             enumerate(itertools.tee(iterator, n)),
  27         )
  28     )
