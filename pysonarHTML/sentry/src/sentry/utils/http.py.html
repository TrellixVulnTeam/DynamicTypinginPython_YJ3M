  18 def absolute_uri(url=None):

  21     return urljoin(options.get("system.url-prefix").rstrip("/") + "/", url.lstrip("/"))

  24 def origin_from_url(url):
  25     if not url:
  26         return url
  27     url = urlparse(url)
  28     return "%s://%s" % (url.scheme, url.netloc)

  31 def safe_urlencode(params, doseq=0):
  32     """
  33     UTF-8-safe version of safe_urlencode
  34 
  35     The stdlib safe_urlencode prior to Python 3.x chokes on UTF-8 values
  36     which can't fail down to ascii.
  37     """
  38     # Snippet originally from pysolr: https://github.com/toastdriven/pysolr
  39 
  40     if hasattr(params, "items"):
  41         params = params.items()
  42 
  43     new_params = list()
  44 
  45     for k, v in params:
  46         k = k.encode("utf-8")
  47 
  48         if isinstance(v, six.string_types):
  49             new_params.append((k, v.encode("utf-8")))
  50         elif isinstance(v, (list, tuple)):
  51             new_params.append((k, [i.encode("utf-8") for i in v]))
  52         else:
  53             new_params.append((k, six.text_type(v)))
  54 
  55     return urlencode(new_params, doseq)

  58 def is_same_domain(url1, url2):
  59     """
  60     Returns true if the two urls should be treated as if they're from the same
  61     domain (trusted).
  62     """
  63     url1 = urlparse(url1)
  64     url2 = urlparse(url2)
  65     return url1.netloc == url2.netloc

  68 def get_origins(project=None):
  69     if settings.SENTRY_ALLOW_ORIGIN == "*":
  70         return frozenset(["*"])
  71 
  72     if settings.SENTRY_ALLOW_ORIGIN:
  73         result = settings.SENTRY_ALLOW_ORIGIN.split(" ")
  74     else:
  75         result = []
  76 
  77     if project:
  78         optval = project.get_option("sentry:origins", ["*"])
  79         if optval:
  80             result.extend(optval)
  81 
  82     # lowercase and strip the trailing slash from all origin values
  83     # filter out empty values
  84     return frozenset(filter(bool, map(lambda x: (x or "").lower().rstrip("/"), result)))

  87 def parse_uri_match(value):
  88     if "://" in value:
  89         scheme, value = value.split("://", 1)
  90     else:
  91         scheme = "*"
  92 
  93     if "/" in value:
  94         domain, path = value.split("/", 1)
  95     else:
  96         domain, path = value, "*"
  97 
  98     if ":" in domain:
  99         domain, port = value.split(":", 1)
 100     else:
 101         port = None
 102 
 103     # we need to coerce our unicode inputs into proper
 104     # idna/punycode encoded representation for normalization.
 105     if isinstance(domain, six.binary_type):
 106         domain = domain.decode("utf8")
 107     domain = domain.encode("idna")
 108 
 109     if port:
 110         domain = "%s:%s" % (domain, port)
 111 
 112     return ParsedUriMatch(scheme, domain, path)

 115 def is_valid_origin(origin, project=None, allowed=None):
 116     """
 117     Given an ``origin`` which matches a base URI (e.g. http://example.com)
 118     determine if a valid origin is present in the project settings.
 119 
 120     Origins may be defined in several ways:
 121 
 122     - http://domain.com[:port]: exact match for base URI (must include port)
 123     - *: allow any domain
 124     - *.domain.com: matches domain.com and all subdomains, on any port
 125     - domain.com: matches domain.com on any port
 126     - *:port: wildcard on hostname, but explicit match on port
 127     """
 128     if allowed is None:
 129         allowed = get_origins(project)
 130 
 131     if not allowed:
 132         return False
 133 
 134     if "*" in allowed:
 135         return True
 136 
 137     if not origin:
 138         return False
 139 
 140     # we always run a case insensitive check
 141     origin = origin.lower()
 142 
 143     # Fast check
 144     if origin in allowed:
 145         return True
 146 
 147     # XXX: In some cases origin might be localhost (or something similar) which causes a string value
 148     # of 'null' to be sent as the origin
 149     if origin == "null":
 150         return False
 151 
 152     if isinstance(origin, six.binary_type):
 153         try:
 154             origin = origin.decode("utf-8")
 155         except UnicodeDecodeError:
 156             try:
 157                 origin = origin.decode("windows-1252")
 158             except UnicodeDecodeError:
 159                 return False
 160 
 161     parsed = urlparse(origin)
 162 
 163     if parsed.hostname is None:
 164         parsed_hostname = ""
 165     else:
 166         try:
 167             parsed_hostname = parsed.hostname.encode("idna")
 168         except UnicodeError:
 169             # We sometimes shove in some garbage input here, so just opting to ignore and carry on
 170             parsed_hostname = parsed.hostname
 171 
 172     if parsed.port:
 173         domain_matches = (
 174             "*",
 175             parsed_hostname,
 176             # Explicit hostname + port name
 177             "%s:%d" % (parsed_hostname, parsed.port),
 178             # Wildcard hostname with explicit port
 179             "*:%d" % parsed.port,
 180         )
 181     else:
 182         domain_matches = ("*", parsed_hostname)
 183 
 184     for value in allowed:
 185         try:
 186             bits = parse_uri_match(value)
 187         except UnicodeError:
 188             # We hit a bad uri, so ignore this value
 189             continue
 190 
 191         # scheme supports exact and any match
 192         if bits.scheme not in ("*", parsed.scheme):
 193             continue
 194 
 195         # domain supports exact, any, and prefix match
 196         if bits.domain[:2] == "*.":
 197             if parsed_hostname.endswith(bits.domain[1:]) or parsed_hostname == bits.domain[2:]:
 198                 return True
 199             continue
 200         elif bits.domain not in domain_matches:
 201             continue
 202 
 203         # path supports exact, any, and suffix match (with or without *)
 204         path = bits.path
 205         if path == "*":
 206             return True
 207         if path.endswith("*"):
 208             path = path[:-1]
 209         if parsed.path.startswith(path):
 210             return True
 211     return False

 214 def origin_from_request(request):
 215     """
 216     Returns either the Origin or Referer value from the request headers,
 217     ignoring "null" Origins.
 218     """
 219     rv = request.META.get("HTTP_ORIGIN", "null")
 220     # In some situation, an Origin header may be the literal value
 221     # "null". This means that the Origin header was stripped for
 222     # privacy reasons, but we should ignore this value entirely.
 223     # Behavior is specified in RFC6454. In either case, we should
 224     # treat a "null" Origin as a nonexistent one and fallback to Referer.
 225     if rv in ("", "null"):
 226         rv = origin_from_url(request.META.get("HTTP_REFERER"))
 227     return rv

 230 def heuristic_decode(data, possible_content_type=None):
 231     """
 232     Attempt to decode a HTTP body by trying JSON and Form URL decoders,
 233     returning the decoded body (if decoding was successful) and the inferred
 234     content type.
 235     """
 236     inferred_content_type = possible_content_type
 237 
 238     form_encoded_parser = partial(parse_qs, strict_parsing=True, keep_blank_values=True)
 239 
 240     decoders = [
 241         ("application/json", json.loads),
 242         ("application/x-www-form-urlencoded", form_encoded_parser),
 243     ]
 244 
 245     # Prioritize the decoder which supports the possible content type first.
 246     decoders.sort(key=lambda d: d[0] == possible_content_type, reverse=True)
 247 
 248     for decoding_type, decoder in decoders:
 249         try:
 250             return (decoder(data), decoding_type)
 251         except Exception:
 252             # Try another decoder
 253             continue
 254 
 255     return (data, inferred_content_type)

 258 def percent_encode(val):
 259     # see https://en.wikipedia.org/wiki/Percent-encoding
 260     if isinstance(val, six.text_type):
 261         val = val.encode("utf8", errors="replace")
 262     return quote(val).replace("%7E", "~").replace("/", "%2F")
