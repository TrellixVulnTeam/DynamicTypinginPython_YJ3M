   8 class Meta(object):
   9     """
  10     A lazy view to detached validation and normalization meta data. It allows to
  11     safely traverse the meta tree and create a deep path lazily. Use ``enter``
  12     to get a view to the meta data inside a specific key.
  13 
  14     The ``Meta`` object is a shallow view onto the top-level meta structure and
  15     only traverses data when actually accessing attributes. Thus, constructing
  16     Meta or calling ``enter`` is relatively cheap.
  17 
  18     To modify data for a certain path, use ``create`` and modify the returned
  19     dict. Alternatively, use the ``merge`` or ``add_error`` convenience methods.
  20     """
  21 
  22     def __init__(self, meta=None, path=None):
  23         self._meta = {} if meta is None else meta
  24         self._path = path or []
  25 
  26     def enter(self, *path):
  27         """
  28         Enters into sub meta data at the specified path. This always returns a
  29         new ``Meta`` object, regardless whether the path already exists.
  30         """
  31         return Meta(self._meta, path=self._path + map(six.text_type, path))
  32 
  33     @property
  34     def path(self):
  35         """
  36         Returns the full path of this meta instance, joined with dots (".").
  37         """
  38         return ".".join(self._path)
  39 
  40     def raw(self):
  41         """
  42         Returns the raw meta tree at the current path, if it exists; otherwise
  43         an empty object. This will contain both the meta data of the key ("")
  44         and sub meta trees.
  45 
  46         It is not safe to mutate the return value since it might be detached
  47         from the actual meta tree.
  48         """
  49         meta = self._meta
  50         for key in self._path:
  51             meta = meta.get(key) or {}
  52         return meta
  53 
  54     def get(self):
  55         """
  56         Returns meta data of the item at the current path, or an empty dict.
  57 
  58         It is not safe to mutate the return value since it might be detached
  59         from the actual meta tree.
  60         """
  61         return self.raw().get("") or {}
  62 
  63     def create(self):
  64         """
  65         Creates an empty meta data entry corresponding to the current path. This
  66         recursively creates the entire parent tree.
  67         """
  68         meta = self._meta
  69         for key in self._path + [""]:
  70             if key not in meta or meta[key] is None:
  71                 meta[key] = {}
  72             meta = meta[key]
  73 
  74         return meta
  75 
  76     def merge(self, other):
  77         """
  78         Merges meta data of the given other ``Meta`` object into the current
  79         path.
  80 
  81         If no meta data entry exists for the current path, it is created, along
  82         with the entire parent tree.
  83         """
  84         other = other.get()
  85         if not other:
  86             return
  87 
  88         meta = self.create()
  89         err = meta.get("err")
  90         meta.update(other)
  91 
  92         if err and other.get("err"):
  93             meta["err"] = err + other["err"]
  94 
  95         return meta
  96 
  97     def iter_errors(self):
  98         """
  99         Iterates over meta errors of the item at the current path, if any.
 100 
 101         Each error is a tuple ``(type, data)``, where:
 102          - ``type`` is the error constant also used in EventError
 103          - ``data`` a dictionary of additional error infos
 104         """
 105         return (
 106             ([err, {}] if isinstance(err, six.string_types) else err)
 107             for err in self.get().get("err") or ()
 108         )
 109 
 110     def get_event_errors(self):
 111         """
 112         Returns all errors of the item at the current path in EventError schema,
 113         which can directly be stored to an event's "errors" list.
 114         """
 115 
 116         errors = []
 117         value = self.get().get("val")
 118 
 119         for error, data in self.iter_errors():
 120             eventerror = dict(data)
 121             eventerror["type"] = error
 122 
 123             if self._path:
 124                 eventerror["name"] = ".".join(self._path)
 125 
 126             if value is not None:
 127                 eventerror["value"] = value
 128                 value = None
 129 
 130             errors.append(eventerror)
 131 
 132         return errors
 133 
 134     def add_error(self, error, value=None, data=None):
 135         """
 136         Adds an error to the meta data at the current path. The ``error``
 137         argument is converted to string. If optional ``data`` is specified, the
 138         data hash is stored alongside the error.
 139 
 140         If an optional ``value`` is specified, it is attached as original value
 141         into the meta data. Note that there is only one original value, not one
 142         per error.
 143 
 144         If no meta data entry exists for the current path, it is created, along
 145         with the entire parent tree.
 146         """
 147         meta = self.create()
 148         if "err" not in meta or meta["err"] is None:
 149             meta["err"] = []
 150 
 151         error = six.text_type(error)
 152         if isinstance(data, collections.Mapping):
 153             error = [error, dict(data)]
 154         meta["err"].append(error)
 155 
 156         if value is not None:
 157             meta["val"] = value
 158 
 159     def __iter__(self):
 160         """
 161         Iterates all child meta entries that potentially have errors set.
 162         """
 163         for key in self.raw():
 164             if key != "":
 165                 yield self.enter(key)
