  40 def truncatechars(value, arg, ellipsis="..."):
  41     # TODO (alex) could use unicode ellipsis: u'\u2026'
  42     """
  43     Truncates a string after a certain number of chars.
  44 
  45     Argument: Number of chars to truncate after.
  46     """
  47     if value is None:
  48         return value
  49     try:
  50         length = int(arg)
  51     except ValueError:  # Invalid literal for int().
  52         return value  # Fail silently.
  53     if len(value) > length:
  54         return value[: max(0, length - len(ellipsis))] + ellipsis
  55     return value

  58 def compress(value):
  59     """
  60     Compresses a value for safe passage as a string.
  61 
  62     This returns a unicode string rather than bytes, as the Django ORM works

  86 def soft_break(value, length, process=lambda chunk: chunk):
  87     """
  88     Encourages soft breaking of text values above a maximum length by adding
  89     zero-width spaces after common delimeters, as well as soft-hyphenating long
  90     identifiers.
  91     """
  92     delimiters = re.compile(
  93         six.text_type(r"([{}]+)").format("".join(map(re.escape, ",.$:/+@!?()<>[]{}")))
  94     )
  95 
  96     def soft_break_delimiter(match):
  97         results = []
  98 
  99         value = match.group(0)
 100         chunks = delimiters.split(value)
 101         for i, chunk in enumerate(chunks):
 102             if i % 2 == 1:  # check if this is this a delimiter
 103                 results.extend([chunk, u"\u200b"])
 104             else:
 105                 results.append(process(chunk))
 106 
 107         return u"".join(results).rstrip(u"\u200b")
 108 
 109     return re.sub(six.text_type(r"\S{{{},}}").format(length), soft_break_delimiter, value)

 112 def to_unicode(value):
 113     try:
 114         value = six.text_type(force_text(value))
 115     except (UnicodeEncodeError, UnicodeDecodeError):
 116         value = "(Error decoding value)"
 117     except Exception:  # in some cases we get a different exception
 118         try:
 119             value = six.text_type(repr(type(value)))
 120         except Exception:
 121             value = "(Error decoding value)"
 122     return value

 125 def split_camelcase(word):
 126     pieces = _camelcase_re.findall(word)
 127 
 128     # Unicode characters or some stuff, ignore it.
 129     if sum(len(x) for x in pieces) != len(word):
 130         yield word
 131     else:
 132         for piece in pieces:
 133             yield piece

 136 def split_any_wordlike(value, handle_camelcase=False):
 137     for word in _word_sep_re.split(value):
 138         if handle_camelcase:
 139             for chunk in split_camelcase(word):
 140                 yield chunk
 141         else:
 142             yield word

 145 def tokens_from_name(value, remove_digits=False):
 146     for word in split_any_wordlike(value, handle_camelcase=True):
 147         if remove_digits:
 148             word = _digit_re.sub("", word)
 149         word = word.lower()
 150         if word:
 151             yield word

 157 def is_valid_dot_atom(value):
 158     """Validate an input string as an RFC 2822 dot-atom-text value."""
 159     return (
 160         isinstance(value, six.string_types)  # must be a string type
 161         and not value[0] == "."
 162         and not value[-1] == "."  # cannot start or end with a dot
 163         and set(value).issubset(valid_dot_atom_characters)
 164     )  # can only contain valid characters

 172 def codec_lookup(encoding, default="utf-8"):
 173     """Safely lookup a codec and ignore non-text codecs,
 174     falling back to a default on errors.
 175     Note: the default value is not sanity checked and would
 176     bypass these checks."""
 177 
 178     def _get_default():
 179         if default is not None:
 180             return codecs.lookup(default)
 181 
 182     if not encoding:
 183         return _get_default()
 184 
 185     try:
 186         info = codecs.lookup(encoding)
 187     except (LookupError, TypeError):
 188         return _get_default()
 189 
 190     try:
 191         # Check for `CodecInfo._is_text_encoding`.
 192         # If this attribute exists, we can assume we can operate
 193         # with this encoding value safely. This attribute was
 194         # introduced into 2.7.12, so versions prior to this will
 195         # raise, but this is the best we can do.
 196         if not info._is_text_encoding:
 197             return _get_default()
 198     except AttributeError:
 199         pass
 200 
 201     # `undefined` is special a special encoding in python that 100% of
 202     # the time will raise, so ignore it.
 203     if info.name == "undefined":
 204         return _get_default()
 205 
 206     return info

 209 def oxfordize_list(strings):
 210     """Given a list of strings, formats them correctly given the length of the
 211     list. For example:
 212 
 213         oxfordize_list(['A'])  =>  'A'
 214 
 215         oxfordize_list(['A', 'B'])  =>  'A and B'
 216 
 217         oxfordize_list(['A', 'B', 'C'])  =>  'A, B, and C'
 218     """
 219 
 220     if len(strings) == 0:
 221         return ""
 222     elif len(strings) == 1:
 223         return strings[0]
 224     elif len(strings) == 2:
 225         return "%s and %s" % (strings[0], strings[1])
 226     else:
 227         return "%s, and %s" % (", ".join(strings[:-1]), strings[-1])
