  24 def hash_value(h, value):
  25     if value is None:
  26         h.update(b"\x00")
  27     elif value is True:
  28         h.update(b"\x01")
  29     elif value is False:
  30         h.update(b"\x02")
  31     elif isinstance(value, six.integer_types):
  32         h.update(b"\x03" + six.text_type(value).encode("ascii") + b"\x00")
  33     elif isinstance(value, (tuple, list)):
  34         h.update(b"\x04" + six.text_type(len(value)).encode("utf-8"))
  35         for item in value:
  36             hash_value(h, item)
  37     elif isinstance(value, dict):
  38         h.update(b"\x05" + six.text_type(len(value)).encode("utf-8"))
  39         for k, v in six.iteritems(value):
  40             hash_value(h, k)
  41             hash_value(h, v)
  42     elif isinstance(value, bytes):
  43         h.update(b"\x06" + value + b"\x00")
  44     elif isinstance(value, six.text_type):
  45         h.update(b"\x07" + value.encode("utf-8") + b"\x00")
  46     else:
  47         raise TypeError("Invalid hash value")

  50 def hash_values(values, seed=None, algorithm=_md5):
  51     h = _md5()
