  11 class PromptsConfig(object):
  12     """
  13     Used to configure available 'prompts' (frontend modals or UI that may be
  14     dismissed or have some other action recorded about it). This config
  15     declares what prompts are available And what fields may be required.
  16 
  17     required_fields available: [organization_id, project_id]
  18     """
  19 
  20     def __init__(self, prompts):
  21         self.prompts = prompts
  22 
  23     def add(self, name, config):
  24         if self.has(name):
  25             raise Exception(u"Prompt key {} is already in use".format(name))
  26         if "required_fields" not in config:
  27             raise Exception("'required_fields' must be present in the config dict")
  28 
  29         self.prompts[name] = config
  30 
  31     def has(self, name):
  32         return name in self.prompts
  33 
  34     def get(self, name):
  35         return self.prompts[name]
  36 
  37     def required_fields(self, name):
  38         return self.prompts[name]["required_fields"]
