  80 class InternalMetrics(object):
  81     def __init__(self):
  82         self._started = False
  83 
  84     def _start(self):
  85         self.q = q = Queue()
  86 
  87         def worker():
  88             from sentry import tsdb
  89 
  90             while True:
  91                 key, instance, tags, amount, sample_rate = q.get()
  92                 amount = _sampled_value(amount, sample_rate)
  93                 if instance:
  94                     full_key = u"{}.{}".format(key, instance)
  95                 else:
  96                     full_key = key
  97                 try:
  98                     tsdb.incr(tsdb.models.internal, full_key, count=amount)
  99                 except Exception:
 100                     logger = logging.getLogger("sentry.errors")
 101                     logger.exception("Unable to incr internal metric")
 102                 finally:
 103                     q.task_done()
 104 
 105         t = Thread(target=worker)
 106         t.setDaemon(True)
 107         t.start()
 108 
 109         self._started = True
 110 
 111     def incr(
 112         self,
 113         key,
 114         instance=None,
 115         tags=None,
 116         amount=1,
 117         sample_rate=settings.SENTRY_METRICS_SAMPLE_RATE,
 118     ):
 119         if not self._started:
 120             self._start()
 121         self.q.put((key, instance, tags, amount, sample_rate))

 127 def incr(
 128     key,
 129     amount=1,
 130     instance=None,
 131     tags=None,
 132     skip_internal=True,
 133     sample_rate=settings.SENTRY_METRICS_SAMPLE_RATE,
 134 ):
 135     current_tags = _get_current_global_tags()
 136     if tags is not None:
 137         current_tags.update(tags)
 138 
 139     should_send_internal = (
 140         not metrics_skip_all_internal
 141         and not skip_internal
 142         and _should_sample(sample_rate)
 143         and not key.startswith(metrics_skip_internal_prefixes)
 144     )
 145 
 146     if should_send_internal:
 147         internal.incr(key, instance, current_tags, amount, sample_rate)
 148 
 149     try:
 150         backend.incr(key, instance, current_tags, amount, sample_rate)
 151         if should_send_internal:
 152             backend.incr("internal_metrics.incr", key, None, 1, sample_rate)
 153     except Exception:
 154         logger = logging.getLogger("sentry.errors")
 155         logger.exception("Unable to record backend metric")

 170 @contextmanager
 171 def timer(key, instance=None, tags=None, sample_rate=settings.SENTRY_METRICS_SAMPLE_RATE):
 172     current_tags = _get_current_global_tags()
 173     if tags is not None:
 174         current_tags.update(tags)
 175 
 176     start = time()
 177     try:
 178         yield current_tags
 179     except Exception:
 180         current_tags["result"] = "failure"
 181         raise
 182     else:
 183         current_tags["result"] = "success"
 184     finally:
 185         timing(key, time() - start, instance, current_tags, sample_rate)
