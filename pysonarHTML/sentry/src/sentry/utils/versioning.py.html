  10 class Version(tuple):
  11     def __str__(self):
  12         return ".".join(map(six.binary_type, self))

  15 def summarize(sequence, max=3):
  16     items = sequence[:max]
  17     remainder = len(sequence) - max
  18     if remainder == 1:
  19         items.append("and one other")
  20     elif remainder > 1:
  21         items.append("and %s others" % (remainder,))
  22     return items

  25 def make_upgrade_message(service, modality, version, hosts):
  26     return u"{service} {modality} be upgraded to {version} on {hosts}.".format(
  27         hosts=",".join(map(six.binary_type, summarize(hosts.keys(), 2))),
  28         modality=modality,
  29         service=service,
  30         version=version,
  31     )

  34 def check_versions(service, versions, required, recommended=None):
  35     """
  36     Check that hosts fulfill version requirements.
  37 
  38     :param service: service label, such as ``Redis``
  39     :param versions: mapping of host to ``Version``
  40     :param required: lowest supported ``Version``. If any host does not fulfill
  41         this requirement, an ``InvalidConfiguration`` exception is raised.
  42     :param recommended: recommended version. If any host does not fulfill this
  43         requirement, a ``PendingDeprecationWarning`` is raised.
  44     """
  45     # x = (host, version)
  46     must_upgrade = dict([x for x in versions.items() if required > x[1]])
  47     if must_upgrade:
  48         raise InvalidConfiguration(make_upgrade_message(service, "must", required, must_upgrade))
  49 
  50     if recommended:
  51         # x = (host, version)
  52         should_upgrade = dict([x for x in versions.items() if recommended > x[1]])
  53         if should_upgrade:
  54             warnings.warn(
  55                 make_upgrade_message(service, "should", recommended, should_upgrade),
  56                 PendingDeprecationWarning,
  57             )
