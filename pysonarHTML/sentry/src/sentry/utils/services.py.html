  46 class LazyServiceWrapper(LazyObject):
  47     """
  48     Lazyily instantiates a standard Sentry service class.
  49 
  50     >>> LazyServiceWrapper(BaseClass, 'path.to.import.Backend', {})
  51 
  52     Provides an ``expose`` method for dumping public APIs to a context, such as
  53     module locals:
  54 
  55     >>> service = LazyServiceWrapper(...)
  56     >>> service.expose(locals())
  57     """
  58 
  59     def __init__(self, backend_base, backend_path, options, dangerous=(), metrics_path=None):
  60         super(LazyServiceWrapper, self).__init__()
  61         self.__dict__.update(
  62             {
  63                 "_backend": backend_path,
  64                 "_options": options,
  65                 "_base": backend_base,
  66                 "_dangerous": dangerous,
  67                 "_metrics_path": metrics_path,
  68             }
  69         )
  70 
  71     def __getattr__(self, name):
  72         if self._wrapped is empty:
  73             self._setup()
  74 
  75         attr = getattr(self._wrapped, name)
  76 
  77         # If we want to wrap in metrics, we need to make sure it's some callable,
  78         # and within our list of exposed attributes. Then we can safely wrap
  79         # in our metrics decorator.
  80         if self._metrics_path and callable(attr) and name in self._base.__all__:
  81             return metrics.wraps(
  82                 self._metrics_path, instance=name, tags={"backend": self._backend}
  83             )(attr)
  84 
  85         return attr
  86 
  87     def _setup(self):
  88         backend = import_string(self._backend)
  89         assert issubclass(backend, Service)
  90         if backend in self._dangerous:
  91             warnings.warn(
  92                 warnings.UnsupportedBackend(
  93                     u"The {!r} backend for {} is not recommended "
  94                     "for production use.".format(self._backend, self._base)
  95                 )
  96             )
  97         instance = backend(**self._options)
  98         self._wrapped = instance
  99 
 100     def expose(self, context):
 101         base = self._base
 102         base_instance = base()
 103         for key in itertools.chain(base.__all__, ("validate", "setup")):
 104             if inspect.ismethod(getattr(base_instance, key)):
 105                 context[key] = (lambda f: lambda *a, **k: getattr(self, f)(*a, **k))(key)
 106             else:
 107                 context[key] = getattr(base_instance, key)

 110 def resolve_callable(value):
 111     if callable(value):
 112         return value
 113     elif isinstance(value, six.string_types):
 114         return import_string(value)
 115     else:
 116         raise TypeError("Expected callable or string")

 128 class ServiceDelegator(Service):
 129     """\
 130     This is backend that coordinates and delegates method execution to multiple
 131     backends. It can be used to route requests to different backends based on
 132     method arguments, as well as execute the same request against multiple
 133     backends in parallel for testing backend performance and data consistency.
 134 
 135     The backends are provided as mapping of backend name to configuration
 136     parameters:
 137 
 138         'redis': {
 139             'path': 'sentry.tsdb.redis.RedisTSDB',
 140             'executor': {
 141                 'path': 'sentry.utils.services.ThreadedExecutor',
 142                 'options': {
 143                     'worker_count': 1,
 144                 },
 145             },
 146         },
 147         'dummy': {
 148             'path': 'sentry.tsdb.dummy.DummyTSDB',
 149             'executor': {
 150                 'path': 'sentry.utils.services.ThreadedExecutor',
 151                 'options': {
 152                     'worker_count': 4,
 153                 },
 154             },
 155         },
 156         # ... etc ...
 157 
 158     The backends used for a method call are determined by a selector function
 159     which is provided with the current context, the method name (as a string)
 160     and arguments (in the form returned by ``inspect.getcallargs``) and
 161     expected to return a list of strings which correspond to names in the
 162     backend mapping. (This list should contain at least one member.) The first
 163     item in the result list is considered the "primary backend". The remainder
 164     of the items in the result list are considered "secondary backends". The
 165     result value of the primary backend will be the result value of the
 166     delegated method (to callers, this appears as a synchronous method call.)
 167     The secondary backends are called asynchronously in the background.  (To
 168     receive the result values of these method calls, provide a callback_func,
 169     described below.) If the primary backend name returned by the selector
 170     function doesn't correspond to any registered backend, the function will
 171     raise a ``InvalidBackend`` exception.  If any referenced secondary backends
 172     are not registered names, they will be discarded and logged.
 173 
 174     The members and ordering of the selector function result (and thus the
 175     primary and secondary backends for a method call) may vary from call to
 176     call based on the calling arguments or some other state. For example, some
 177     calls may use a different primary backend based on some piece of global
 178     state (e.g. some property of a web request), or a secondary backend
 179     undergoing testing may be included based on the result of a random number
 180     generator (essentially calling it in the background for a sample of calls.)
 181 
 182     The selector function and callback function can be provided as either:
 183 
 184     - A dotted import path string (``path.to.callable``) that will be
 185       imported at backend instantiation, or
 186     - A reference to a callable object.
 187 
 188     Implementation notes:
 189 
 190     - Only method access is delegated to the individual backends. Attribute
 191       values are returned from the base backend. Only methods that are defined
 192       on the base backend are eligible for delegation (since these methods are
 193       considered the public API.)
 194     - The backend makes no attempt to synchronize common backend option values
 195       between backends (e.g. TSDB rollup configuration) to ensure equivalency
 196       of request parameters based on configuration.
 197     - Each backend is associated with an executor pool which defaults to a
 198       thread pool implementation unless otherwise specified in the backend
 199       configuration. If the backend itself is not thread safe (due to socket
 200       access, etc.), it's recommended to specify a pool size of 1 to ensure
 201       exclusive access to resources. Each executor is started when the first
 202       task is submitted.
 203     - The request is added to the request queue of the primary backend using a
 204       blocking put. The request is added to the request queue(s) of the
 205       secondary backend(s) as a non-blocking put (if these queues are full, the
 206       request is rejected and the future will raise ``Queue.Full`` when
 207       attempting to retrieve the result.)
 208     - The ``callback_func`` is called after all futures have completed, either
 209       successfully or unsuccessfully. The function parameters are:
 210       - the context,
 211       - the method name (as a string),
 212       - the calling arguments (as returned by ``inspect.getcallargs``),
 213       - the backend names (as returned by the selector function),
 214       - a list of results (as either a ``Future``, or ``None`` if the backend
 215         was invalid) of the same length and ordering as the backend names.
 216     """
 217 
 218     class InvalidBackend(Exception):
 219         """\
 220         Exception raised when an invalid backend is returned by a selector
 221         function.
 222         """
 223 
 224     class State(threading.local):
 225         def __init__(self):
 226             self.context = None
 227 
 228     __state = State()
 229 
 230     def __init__(self, backend_base, backends, selector_func, callback_func=None):
 231         self.__backend_base = import_string(backend_base)
 232 
 233         def load_executor(options):
 234             path = options.get("path")
 235             if path is None:
 236                 executor_cls = ThreadedExecutor
 237             else:
 238                 executor_cls = import_string(path)
 239             return executor_cls(**options.get("options", {}))
 240 
 241         self.__backends = {}
 242         for name, options in backends.items():
 243             self.__backends[name] = (
 244                 import_string(options["path"])(**options.get("options", {})),
 245                 load_executor(options.get("executor", {})),
 246             )
 247 
 248         self.__selector_func = resolve_callable(selector_func)
 249 
 250         if callback_func is not None:
 251             self.__callback_func = resolve_callable(callback_func)
 252         else:
 253             self.__callback_func = None
 254 
 255     def validate(self):
 256         for backend, executor in self.__backends.values():
 257             backend.validate()
 258 
 259     def setup(self):
 260         for backend, executor in self.__backends.values():
 261             backend.setup()
 262 
 263     def __getattr__(self, attribute_name):
 264         # When deciding how to handle attribute accesses, we have three
 265         # different possible outcomes:
 266         # 1. If this is defined as a method on the base implementation, we are
 267         #    able delegate it to the backends based on the selector function.
 268         # 2. If this is defined as an attribute on the base implementation, we
 269         #    are able to (immediately) return that as the value. (This also
 270         #    mirrors the behavior of ``LazyServiceWrapper``, which will cache
 271         #    any attribute access during ``expose``, so we can't delegate
 272         #    attribute access anyway.)
 273         # 3. If this isn't defined at all on the base implementation, we let
 274         #    the ``AttributeError`` raised by ``getattr`` propagate (mirroring
 275         #    normal attribute access behavior for a missing/invalid name.)
 276         base_value = getattr(self.__backend_base, attribute_name)
 277         if not inspect.ismethod(base_value):
 278             return base_value
 279 
 280         def execute(*args, **kwargs):
 281             context = type(self).__state.context
 282 
 283             # If there is no context object already set in the thread local
 284             # state, we are entering the delegator for the first time and need
 285             # to create a new context.
 286             if context is None:
 287                 from sentry.app import env  # avoids a circular import
 288 
 289                 context = Context(env.request, {})
 290 
 291             # If this thread already has an active backend for this base class,
 292             # we can safely call that backend synchronously without delegating.
 293             if self.__backend_base in context.backends:
 294                 backend = context.backends[self.__backend_base]
 295                 return getattr(backend, attribute_name)(*args, **kwargs)
 296 
 297             # Binding the call arguments to named arguments has two benefits:
 298             # 1. These values always be passed in the same form to the selector
 299             #    function and callback, regardless of how they were passed to
 300             #    the method itself (as positional arguments, keyword arguments,
 301             #    etc.)
 302             # 2. This ensures that the given arguments are those supported by
 303             #    the base backend itself, which should be a common subset of
 304             #    arguments that are supported by all backends.
 305             callargs = inspect.getcallargs(base_value, None, *args, **kwargs)
 306 
 307             selected_backend_names = list(self.__selector_func(context, attribute_name, callargs))
 308             if not len(selected_backend_names) > 0:
 309                 raise self.InvalidBackend("No backends returned by selector!")
 310 
 311             # Ensure that the primary backend is actually registered -- we
 312             # don't want to schedule any work on the secondaries if the primary
 313             # request is going to fail anyway.
 314             if selected_backend_names[0] not in self.__backends:
 315                 raise self.InvalidBackend(
 316                     u"{!r} is not a registered backend.".format(selected_backend_names[0])
 317                 )
 318 
 319             def call_backend_method(context, backend, is_primary):
 320                 # Update the thread local state in the executor to the provided
 321                 # context object. This allows the context to be propagated
 322                 # across different threads.
 323                 assert type(self).__state.context is None
 324                 type(self).__state.context = context
 325 
 326                 # Ensure that we haven't somehow accidentally entered a context
 327                 # where the backend we're calling has already been marked as
 328                 # active (or worse, some other backend is already active.)
 329                 base = self.__backend_base
 330                 assert base not in context.backends
 331 
 332                 # Mark the backend as active.
 333                 context.backends[base] = backend
 334                 try:
 335                     return getattr(backend, attribute_name)(*args, **kwargs)
 336                 except Exception as e:
 337                     # If this isn't the primary backend, we log any unexpected
 338                     # exceptions so that they don't pass by unnoticed. (Any
 339                     # exceptions raised by the primary backend aren't logged
 340                     # here, since it's assumed that the caller will log them
 341                     # from the calling thread.)
 342                     if not is_primary:
 343                         expected_raises = getattr(base_value, "__raises__", [])
 344                         if not expected_raises or not isinstance(e, tuple(expected_raises)):
 345                             logger.warning(
 346                                 "%s caught in executor while calling %r on %s.",
 347                                 type(e).__name__,
 348                                 attribute_name,
 349                                 type(backend).__name__,
 350                                 exc_info=True,
 351                             )
 352                     raise
 353                 finally:
 354                     type(self).__state.context = None
 355 
 356             # Enqueue all of the secondary backend requests first since these
 357             # are non-blocking queue insertions. (Since the primary backend
 358             # executor queue insertion can block, if that queue was full the
 359             # secondary requests would have to wait unnecessarily to be queued
 360             # until the after the primary request can be enqueued.)
 361             # NOTE: If the same backend is both the primary backend *and* in
 362             # the secondary backend list -- this is unlikely, but possible --
 363             # this means that one of the secondary requests will be queued and
 364             # executed before the primary request is queued.  This is such a
 365             # strange usage pattern that I don't think it's worth optimizing
 366             # for.)
 367             results = [None] * len(selected_backend_names)
 368             for i, backend_name in enumerate(selected_backend_names[1:], 1):
 369                 try:
 370                     backend, executor = self.__backends[backend_name]
 371                 except KeyError:
 372                     logger.warning(
 373                         "%r is not a registered backend and will be ignored.",
 374                         backend_name,
 375                         exc_info=True,
 376                     )
 377                 else:
 378                     results[i] = executor.submit(
 379                         functools.partial(
 380                             call_backend_method, context.copy(), backend, is_primary=False
 381                         ),
 382                         priority=1,
 383                         block=False,
 384                     )
 385 
 386             # The primary backend is scheduled last since it may block the
 387             # calling thread. (We don't have to protect this from ``KeyError``
 388             # since we already ensured that the primary backend exists.)
 389             backend, executor = self.__backends[selected_backend_names[0]]
 390             results[0] = executor.submit(
 391                 functools.partial(call_backend_method, context.copy(), backend, is_primary=True),
 392                 priority=0,
 393                 block=True,
 394             )
 395 
 396             if self.__callback_func is not None:
 397                 FutureSet([_f for _f in results if _f]).add_done_callback(
 398                     lambda *a, **k: self.__callback_func(
 399                         context, attribute_name, callargs, selected_backend_names, results
 400                     )
 401                 )
 402 
 403             return results[0].result()
 404 
 405         return execute
