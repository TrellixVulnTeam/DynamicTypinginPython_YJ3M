  15     "core::",
  16     "alloc::",

  35 FRAME_RE = re.compile(
  36     r"""(?xm)
  37     ^
  38         (?!stacktrace\:[\ ]stack[\ ]backtrace\:)
  39 
  40         [\ ]*                             # leading whitespace
  41         (?:\d+:[\ ]*)?                    # leading frame number
  42 
  43         (?P<addr>0x[a-f0-9]+)?            # instruction address (osx)
  44         ([\ ]?-[\ ])?                     # dash (osx), can happen without addr if inline frame
  45 
  46         (?P<symbol>[^\r\n]+?)
  47 
  48         ([\ ]?\(
  49             (?P<addr_linux>0x[a-f0-9]+)   # instruction address (linux)
  50         \))?
  51 
  52         (?:
  53             \r?\n
  54             [\ \t]+at[\ ]
  55             (?P<path>[^\r\n]+?)
  56             (?::(?P<lineno>\d+))?
  57         )?
  58     $
  59 """
  60 )

 103 def get_filename(abs_path):
 104     """Returns the basename of the given absolute path."""
 105     return abs_path.rsplit("/", 1)[-1].rsplit("\\", 1)[-1]
 106

 108 def strip_symbol(symbol):
 109     """Strips rust function hashes of the given symbol."""
 110     if symbol:
 111         match = HASH_FUNC_RE.match(symbol)
 112         if match:
 113             return match.group(1)
 114 
 115     return symbol

 118 def demangle_rust(symbol):
 119     """Demangles common escapes in the given rust symbol."""
 120     return RUST_ESCAPES_RE.sub(lambda m: RUST_ESCAPES.get(m.group(1), ""), symbol)

 123 def starts_with(function, pattern):
 124     """
 125     Returns whether the given function name matches the pattern.
 126     This takes trait implementation and name mangling into account.
 127     """
 128     return PATTERN_MATCH_RE.sub("", function).replace(".", ":").startswith(pattern)
 129 
 130 
 131 def matches_frame(function, patterns):
 132     """Returns whether the given function name matches any of the patterns."""
 133     return any(starts_with(function, p) for p in patterns)

 136 def frame_from_match(match, last_addr):
 137     """Creates a sentry stack frame from a backtrace entry."""
 138     symbol = strip_symbol(match.group("symbol"))
 139     function = demangle_rust(symbol)
 140 
 141     frame = {
 142         "function": function,
 143         "in_app": not matches_frame(function, SYSTEM_FRAMES),
 144         "instruction_addr": match.group("addr") or match.group("addr_linux") or last_addr,
 145     }
 146 
 147     if symbol != function:
 148         frame["symbol"] = symbol
 149 
 150     package = RUST_CRATE_RE.search(function)
 151     if package and package.group(1):
 152         frame["package"] = package.group(1)
 153 
 154     path = match.group("path")
 155     if path:
 156         frame["abs_path"] = path
 157         frame["filename"] = get_filename(path)
 158 
 159     lineno = match.group("lineno")
 160     if lineno:
 161         lineno = int(lineno)
 162     if lineno:
 163         frame["lineno"] = lineno
 164 
 165     return frame

 168 def frames_from_rust_info(rust_info):
 169     """
 170     Extracts a list of frames from the given rust_info string.
 171 
 172     Border frames from the python interpreter, FFI calls, panic unwinding and
 173     backtrace generation are trimmed off.
 174     """
 175     frames = []
 176     last_addr = None
 177     for m in FRAME_RE.finditer(rust_info):
 178         frame = frame_from_match(m, last_addr)
 179         last_addr = frame["instruction_addr"]
 180         frames.append(frame)
 181 
 182     end = next(
 183         (i for i, f in enumerate(frames) if matches_frame(f["function"], OUTER_BORDER_FRAMES)),
 184         len(frames),
 185     )
 186 
 187     start = -next(
 188         (
 189             i
 190             for i, f in enumerate(reversed(frames))
 191             if matches_frame(f["function"], INNER_BORDER_FRAMES)
 192         ),
 193         0,
 194     )
 195 
 196     return frames[start:end]

 199 def strip_backtrace_message(target, field):
 200     """
 201     Strips the backtrace off a message, if it contains one.
 202     """
 203     if target and isinstance(target.get(field), six.string_types):
 204         target[field] = target[field].split("\n\nstacktrace:", 1)[0].strip()

 207 def merge_rust_info_frames(event, hint):
 208     """
 209     Adds rust exception backtraces to the python traceback.
 210 
 211     If there is no rust_info attribute on the exception or the event has no
 212     exception, this operation is a noop. Otherwise, it parses the rust backtrace
 213     and adds it on top of existing python frames.
 214 
 215     This changes the event's platform to "native" and patches existing frames.
 216     Additionally, the traceback is removed from the exception value and logentry
 217     interfaces.
 218     """
 219     if "exc_info" not in hint:
 220         return event
 221 
 222     exc_type, exc_value, tb = hint["exc_info"]
 223     if not hasattr(exc_value, "rust_info") or not exc_value.rust_info:
 224         return event
 225 
 226     exception = get_path(event, "exception", "values", 0)
 227     stacktrace = get_path(exception, "stacktrace", "frames")
 228     if not stacktrace:
 229         return event
 230 
 231     # Update the platform
 232     event["platform"] = "native"
 233     for frame in stacktrace:
 234         frame["platform"] = "python"
 235 
 236     # Remove rust_info from messages
 237     strip_backtrace_message(exception, "value")
 238     strip_backtrace_message(event.get("logentry"), "message")
 239     strip_backtrace_message(event.get("logentry"), "formatted")
 240 
 241     # Extend the stacktrace
 242     frames = frames_from_rust_info(exc_value.rust_info)
 243     if frames:
 244         stacktrace.extend(reversed(frames))
 245 
 246     return event

 249 class RustInfoIntegration(Integration):
 250     identifier = "rust_info"
 251 
 252     @staticmethod
 253     def setup_once():
 254         @add_global_event_processor
 255         def processor(event, hint):
 256             integration = Hub.current.get_integration(RustInfoIntegration)
 257             if integration is None:
 258                 return event
 259 
 260             return merge_rust_info_frames(event, hint)
