   8 class Cursor(object):
   9     def __init__(self, value, offset=0, is_prev=False, has_results=None):
  10         self.value = int(value)
  11         self.offset = int(offset)
  12         self.is_prev = bool(is_prev)
  13         self.has_results = has_results
  14 
  15     def __str__(self):
  16         return "%s:%s:%s" % (self.value, self.offset, int(self.is_prev))
  17 
  18     def __eq__(self, other):
  19         return all(
  20             getattr(self, attr) == getattr(other, attr)
  21             for attr in ("value", "offset", "is_prev", "has_results")
  22         )
  23 
  24     def __repr__(self):
  25         return "<%s: value=%s offset=%s is_prev=%s>" % (
  26             type(self).__name__,
  27             self.value,
  28             self.offset,
  29             int(self.is_prev),
  30         )
  31 
  32     def __nonzero__(self):
  33         return self.has_results

  35     @classmethod
  36     def from_string(cls, value):
  37         bits = value.split(":")
  38         if len(bits) != 3:
  39             raise ValueError
  40         try:
  41             bits = float(bits[0]), int(bits[1]), int(bits[2])
  42         except (TypeError, ValueError):
  43             raise ValueError
  44         return cls(*bits)

  68 def _build_next_values(cursor, results, key, limit, is_desc):
  69     value = cursor.value
  70     offset = cursor.offset
  71     is_prev = cursor.is_prev
  72 
  73     num_results = len(results)
  74 
  75     if not value and num_results:
  76         value = int(key(results[0]))
  77 
  78     # Next cursor for a prev-cursor simply starts from that prev cursors value
  79     # without an offset.
  80     if is_prev:
  81         return (value, 0, True)
  82 
  83     # No results means no more next
  84     if not num_results:
  85         return (value, offset, False)
  86 
  87     # Are there more results than whats on the current page?
  88     has_next = num_results > limit
  89 
  90     # Determine what our next cursor is by ensuring we have a unique offset
  91     next_value = int(key(results[-1]))
  92 
  93     # value has not changed, page forward by adjusting the offset
  94     if next_value == value:
  95         next_offset = offset + limit
  96         return (next_value, next_offset, has_next)
  97 
  98     # We have an absolute value to page from. If any of the items in
  99     # the current result set come *after* or *before* (depending on the
 100     # is_desc flag) we will want to increment the offset to account for
 101     # moving past them.
 102     #
 103     # This is required to account for loss of precision in the key value.
 104     next_offset = 0
 105     result_iter = reversed(results)
 106 
 107     # If we have more results the last item in the results should be
 108     # skipped, as we know we want to start from that item and do not
 109     # need to offset from it.
 110     if has_next:
 111         six.next(result_iter)
 112 
 113     for result in result_iter:
 114         result_value = int(key(result))
 115 
 116         is_larger = result_value >= next_value
 117         is_smaller = result_value <= next_value
 118 
 119         if (is_desc and is_smaller) or (not is_desc and is_larger):
 120             next_offset += 1
 121         else:
 122             break
 123 
 124     return (next_value, next_offset, has_next)

 127 def _build_prev_values(cursor, results, key, limit, is_desc):
 128     value = cursor.value
 129     offset = cursor.offset
 130     is_prev = cursor.is_prev
 131 
 132     num_results = len(results)
 133 
 134     if is_prev:
 135         has_prev = num_results > limit
 136     else:
 137         # It's likely that there's a previous page if they passed us either
 138         # offset values
 139         has_prev = value or offset
 140 
 141     # If the cursor contains previous results, the first item is the item that
 142     # indicates if we have more items later, and is *not* the first item in the
 143     # list, that should be used for the value.
 144     first_prev_index = 1 if is_prev and has_prev else 0
 145 
 146     # If we're paging back we need to calculate the key from the first result
 147     # with for_prev=True to ensure rounding of the key is correct.See
 148     # sentry.api.paginator.BasePaginator.get_item_key
 149     prev_value = int(key(results[first_prev_index], for_prev=True)) if results else 0
 150 
 151     # Prev only has an offset if the cursor we were dealing with was a
 152     # previous cursor. Otherwise we'd be taking the offset while moving forward.
 153     prev_offset = offset if is_prev else 0
 154 
 155     if not (is_prev and num_results):
 156         return (prev_value, prev_offset, has_prev)
 157 
 158     # Value has not changed, page back by adjusting the offset
 159     if prev_value == value:
 160         prev_offset = offset + limit
 161         return (prev_value, prev_offset, has_prev)
 162 
 163     # Just as in the next cursor builder, we may need to add an offset
 164     # if any of the results at the beginning are *before* or *after*
 165     # (depending on the is_desc flag).
 166     #
 167     # This is required to account for loss of precision in the key value.
 168     prev_offset = 0
 169     result_iter = iter(results)
 170 
 171     # If we know there are more previous results, we need to move past
 172     # the item indicating that more items exist.
 173     if has_prev:
 174         six.next(result_iter)
 175 
 176     # Always move past the first item, this is the prev_value item and will
 177     # already be offset in the next query.
 178     six.next(result_iter)
 179 
 180     for result in result_iter:
 181         result_value = int(key(result, for_prev=True))
 182 
 183         is_larger = result_value >= prev_value
 184         is_smaller = result_value <= prev_value
 185 
 186         # Note that the checks are reversed here as a prev query has
 187         # it's ordering reversed.
 188         if (is_desc and is_larger) or (not is_desc and is_smaller):
 189             prev_offset += 1
 190         else:
 191             break
 192 
 193     return (prev_value, prev_offset, has_prev)

 196 def build_cursor(
 197     results, key, limit=100, is_desc=False, cursor=None, hits=None, max_hits=None, on_results=None
 198 ):
 199     if cursor is None:
 200         cursor = Cursor(0, 0, 0)
 201 
 202     # Compute values for next cursor
 203     next_value, next_offset, has_next = _build_next_values(
 204         cursor=cursor, results=results, key=key, limit=limit, is_desc=is_desc
 205     )
 206 
 207     # Compute values for prev cursor
 208     prev_value, prev_offset, has_prev = _build_prev_values(
 209         cursor=cursor, results=results, key=key, limit=limit, is_desc=is_desc
 210     )
 211 
 212     if cursor.is_prev and has_prev:
 213         # A prev cursor with more results should have the first item chopped off
 214         # as this is the item that indicates we have more items before, and
 215         # should not be included on this page.
 216         results = results[1:]
 217     elif not cursor.is_prev:
 218         # For next page cursors we cut off the extra item that indicates there
 219         # are more items.
 220         results = results[:limit]
 221 
 222     next_cursor = Cursor(next_value or 0, next_offset, False, has_next)
 223     prev_cursor = Cursor(prev_value or 0, prev_offset, True, has_prev)
 224 
 225     if on_results:
 226         results = on_results(results)
 227 
 228     return CursorResult(
 229         results=results, next=next_cursor, prev=prev_cursor, hits=hits, max_hits=max_hits
 230     )
