   9 def extract_lazy_object(lo):
  10     """
  11     Unwrap a LazyObject and return the inner object. Whatever that may be.
  12 
  13     ProTip: This is relying on `django.utils.functional.empty`, which may
  14     or may not be removed in the future. It's 100% undocumented.
  15     """
  16     if not hasattr(lo, "_wrapped"):
  17         return lo
  18     if lo._wrapped is empty:
  19         lo._setup()
  20     return lo._wrapped

  23 def apply_values(function, mapping):
  24     """\
  25     Applies ``function`` to a sequence containing all of the values in the
  26     provided mapping, returing a new mapping with the values replaced with
  27     the results of the provided function.
  28 
  29     >>> apply_values(
  30     ...   lambda values: map(u'{} fish'.format, values),
  31     ...   {1: 'red', 2: 'blue'},
  32     ... )
  33     {1: u'red fish', 2: u'blue fish'}
  34     """
  35     if not mapping:
  36         return {}
  37 
  38     keys, values = zip(*mapping.items())
  39     return dict(zip(keys, function(values)))

  42 def compact(seq):
  43     """
  44     Removes ``None`` values from various sequence-based data structures.
  45 
  46     dict:
  47         Removes keys with a corresponding ``None`` value.
  48 
  49     list:
  50         Removes ``None`` values.
  51 
  52     >>> compact({'foo': 'bar', 'baz': None})
  53     {'foo': 'bar'}
  54 
  55     >>> compact([1, None, 2])
  56     [1, 2]
  57     """
  58     if isinstance(seq, dict):
  59         return {k: v for k, v in six.iteritems(seq) if v is not None}
  60 
  61     elif isinstance(seq, list):
  62         return [k for k in seq if k is not None]
