 104 def load_data(platform, default=None, sample_name=None, timestamp=None, start_timestamp=None):
 105     # NOTE: Before editing this data, make sure you understand the context
 106     # in which its being used. It is NOT only used for local development and
 107     # has production consequences.
 108     #   * bin/load-mocks to generate fake data for local testing
 109     #   * When a new project is created, a fake event is generated as a "starter"
 110     #     event so it's not an empty project.
 111     #   * When a user clicks Test Configuration from notification plugin settings page,
 112     #     a fake event is generated to go through the pipeline.
 113     data = None
 114     language = None
 115     platform_data = INTEGRATION_ID_TO_PLATFORM_DATA.get(platform)
 116 
 117     if platform_data is not None and platform_data["type"] != "language":
 118         language = platform_data["language"]
 119 
 120     samples_root = os.path.join(DATA_ROOT, "samples")
 121     all_samples = set(f for f in os.listdir(samples_root) if f.endswith(".json"))
 122 
 123     for platform in (platform, language, default):
 124         if not platform:
 125             continue
 126 
 127         # Verify by checking if the file is within our folder explicitly
 128         # avoids being able to have a name that invokes traversing directories.
 129         json_path = "%s.json" % platform.encode("utf-8")
 130         if json_path not in all_samples:
 131             continue
 132 
 133         if not sample_name:
 134             try:
 135                 sample_name = INTEGRATION_ID_TO_PLATFORM_DATA[platform]["name"]
 136             except KeyError:
 137                 pass
 138 
 139         # XXX: At this point, it's assumed that `json_path` was safely found
 140         # within `samples_root` due to the check above and cannot traverse
 141         # into paths.
 142         with open(os.path.join(samples_root, json_path)) as fp:
 143             data = json.load(fp)
 144             break
 145 
 146     if data is None:
 147         return
 148 
 149     data = CanonicalKeyDict(data)
 150     if platform in ("csp", "hkpk", "expectct", "expectstaple"):
 151         return data
 152 
 153     # Generate a timestamp in the present.
 154     if timestamp is None:
 155         timestamp = timezone.now()
 156     else:
 157         timestamp = timestamp.replace(tzinfo=pytz.utc)
 158     data.setdefault("timestamp", to_timestamp(timestamp))
 159 
 160     if data.get("type") == "transaction":
 161         if start_timestamp is None:
 162             start_timestamp = timestamp - timedelta(seconds=2)
 163         else:
 164             start_timestamp = start_timestamp.replace(tzinfo=pytz.utc)
 165         data["start_timestamp"] = to_timestamp(start_timestamp)
 166 
 167         for span in data.get("spans", []):
 168             # Use data to generate span timestamps consistently and based
 169             # on event timestamp
 170             duration = span.get("data", {}).get("duration", 10.0)
 171             offset = span.get("data", {}).get("offset", 0)
 172 
 173             span_start = data["start_timestamp"] + offset
 174             span.setdefault("start_timestamp", span_start)
 175             span.setdefault("timestamp", span_start + duration)
 176 
 177     data["platform"] = platform
 178     # XXX: Message is a legacy alias for logentry. Do not overwrite if set.
 179     if "message" not in data:
 180         data["message"] = "This is an example %s exception" % (sample_name or platform,)
 181     data.setdefault(
 182         "user",
 183         generate_user(ip_address="127.0.0.1", username="sentry", id=1, email="sentry@example.com"),
 184     )
 185     data.setdefault(
 186         "extra",
 187         {
 188             "session": {"foo": "bar"},
 189             "results": [1, 2, 3, 4, 5],
 190             "emptyList": [],
 191             "emptyMap": {},
 192             "length": 10837790,
 193             "unauthorized": False,
 194             "url": "http://example.org/foo/bar/",
 195         },
 196     )
 197     data.setdefault("modules", {"my.package": "1.0.0"})
 198     data.setdefault(
 199         "request",
 200         {
 201             "cookies": "foo=bar;biz=baz",
 202             "url": "http://example.com/foo",
 203             "headers": {
 204                 "Referer": "http://example.com",
 205                 "Content-Type": "application/json",
 206                 "User-Agent": "Mozilla/5.0 (Windows NT 6.2; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/28.0.1500.72 Safari/537.36",
 207             },
 208             "env": {"ENV": "prod"},
 209             "query_string": "foo=bar",
 210             "data": '{"hello": "world"}',
 211             "method": "GET",
 212         },
 213     )
 214 
 215     return data

 218 def create_sample_event(project, platform=None, default=None, raw=True, sample_name=None, **kwargs):
 219     if not platform and not default:
 220         return
 221 
 222     data = load_data(platform, default, sample_name)
 223 
 224     if not data:
 225         return
 226 
 227     data.update(kwargs)
 228 
 229     manager = EventManager(data)
 230     manager.normalize()
 231     return manager.save(project.id, raw=raw)
