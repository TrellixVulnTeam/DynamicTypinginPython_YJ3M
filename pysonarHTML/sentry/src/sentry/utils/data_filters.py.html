  51 def is_valid_ip(project_config, ip_address):
  52     """
  53     Verify that an IP address is not being blacklisted
  54     for the given project.
  55     """
  56     blacklist = get_path(project_config.config, "filterSettings", "clientIps", "blacklistedIps")
  57     if not blacklist:
  58         return True
  59 
  60     for addr in blacklist:
  61         # We want to error fast if it's an exact match
  62         if ip_address == addr:
  63             return False
  64 
  65         # Check to make sure it's actually a range before
  66         try:
  67             if "/" in addr and (
  68                 ipaddress.ip_address(six.text_type(ip_address))
  69                 in ipaddress.ip_network(six.text_type(addr), strict=False)
  70             ):
  71                 return False
  72         except ValueError:
  73             # Ignore invalid values here
  74             pass
  75 
  76     return True
  77

  79 def is_valid_release(project_config, release):
  80     """
  81     Verify that a release is not being filtered
  82     for the given project.
  83     """
  84     invalid_versions = get_path(project_config.config, "filterSettings", "releases", "releases")
  85 
  86     if not invalid_versions:
  87         return True
  88 
  89     release = force_text(release).lower()
  90 
  91     for version in invalid_versions:
  92         if fnmatch.fnmatch(release, version.lower()):
  93             return False
  94 
  95     return True
  96

  98 def is_valid_error_message(project_config, message):
  99     """
 100     Verify that an error message is not being filtered
 101     for the given project.
 102     """
 103     filtered_errors = get_path(project_config.config["filterSettings"], "errorMessages", "patterns")
 104 
 105     if not filtered_errors:
 106         return True
 107 
 108     message = force_text(message).lower()
 109 
 110     for error in filtered_errors:
 111         try:
 112             if fnmatch.fnmatch(message, error.lower()):
 113                 return False
 114         except Exception:
 115             # fnmatch raises a string when the pattern is bad.
 116             # Patterns come from end users and can be full of mistakes.
 117             pass
 118 
 119     return True
