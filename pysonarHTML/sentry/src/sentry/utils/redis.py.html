  31 def _shared_pool(**opts):
  32     if "host" in opts:
  33         key = "%s:%s/%s" % (opts["host"], opts["port"], opts["db"])
  34     else:
  35         key = "%s/%s" % (opts["path"], opts["db"])
  36     pool = _pool_cache.get(key)
  37     if pool is not None:
  38         return pool
  39     with _pool_lock:
  40         pool = _pool_cache.get(key)
  41         if pool is not None:
  42             return pool
  43         pool = ConnectionPool(**opts)
  44         _pool_cache[key] = pool
  45         return pool

  81 class RetryingStrictRedisCluster(StrictRedisCluster):
  82     """
  83     Execute a command with cluster reinitialization retry logic.
  84 
  85     Should a cluster respond with a ConnectionError or BusyLoadingError the
  86     cluster nodes list will be reinitialized and the command will be executed
  87     again with the most up to date view of the world.
  88     """
  89 
  90     def execute_command(self, *args, **kwargs):
  91         try:
  92             return super(self.__class__, self).execute_command(*args, **kwargs)
  93         except (
  94             ConnectionError,
  95             BusyLoadingError,
  96             KeyError,  # see: https://github.com/Grokzen/redis-py-cluster/issues/287
  97         ):
  98             self.connection_pool.nodes.reset()
  99             return super(self.__class__, self).execute_command(*args, **kwargs)

 102 class _RedisCluster(object):
 103     def supports(self, config):
 104         # _RedisCluster supports two configurations:
 105         #  * Explicitly configured with is_redis_cluster. This mode is for real redis-cluster.
 106         #  * No is_redis_cluster, but only 1 host. This represents a singular node Redis running
 107         #    in non-cluster mode.
 108         return config.get("is_redis_cluster", False) or len(config.get("hosts")) == 1
 109 
 110     def factory(self, **config):
 111         # StrictRedisCluster expects a list of { host, port } dicts. Coerce the
 112         # configuration into the correct format if necessary.
 113         hosts = config.get("hosts")
 114         # TODO(joshuarli): modernize dict_six fixer
 115         hosts = list(hosts.values()) if isinstance(hosts, dict) else hosts
 116 
 117         # Redis cluster does not wait to attempt to connect. We'd prefer to not
 118         # make TCP connections on boot. Wrap the client in a lazy proxy object.
 119         def cluster_factory():
 120             if config.get("is_redis_cluster", False):
 121                 return RetryingStrictRedisCluster(
 122                     startup_nodes=hosts,
 123                     decode_responses=True,
 124                     skip_full_coverage_check=True,
 125                     max_connections=16,
 126                     max_connections_per_node=True,
 127                 )
 128             else:
 129                 host = hosts[0].copy()
 130                 host["decode_responses"] = True
 131                 return StrictRedis(**host)
 132 
 133         return SimpleLazyObject(cluster_factory)
 134 
 135     def __str__(self):
 136         return "Redis Cluster"

 139 class ClusterManager(object):
 140     def __init__(self, options_manager, cluster_type=_RBCluster):
 141         self.__clusters = {}
 142         self.__options_manager = options_manager
 143         self.__cluster_type = cluster_type()
 144 
 145     def get(self, key):
 146         cluster = self.__clusters.get(key)
 147 
 148         if cluster:
 149             return cluster
 150 
 151         # TODO: This would probably be safer with a lock, but I'm not sure
 152         # that it's necessary.
 153         configuration = self.__options_manager.get("redis.clusters").get(key)
 154         if configuration is None:
 155             raise KeyError(u"Invalid cluster name: {}".format(key))
 156 
 157         if not self.__cluster_type.supports(configuration):
 158             raise KeyError(u"Invalid cluster type, expected: {}".format(self.__cluster_type))
 159 
 160         cluster = self.__clusters[key] = self.__cluster_type.factory(**configuration)
 161 
 162         return cluster
 163

 172 def get_cluster_from_options(setting, options, cluster_manager=clusters):
 173     cluster_option_name = "cluster"
 174     default_cluster_name = "default"
 175     cluster_constructor_option_names = frozenset(("hosts",))
 176 
 177     options = options.copy()
 178     cluster_options = {
 179         key: options.pop(key)
 180         for key in set(options.keys()).intersection(cluster_constructor_option_names)
 181     }
 182     if cluster_options:
 183         if cluster_option_name in options:
 184             raise InvalidConfiguration(
 185                 u"Cannot provide both named cluster ({!r}) and cluster configuration ({}) options.".format(
 186                     cluster_option_name, ", ".join(map(repr, cluster_constructor_option_names))
 187                 )
 188             )
 189         else:
 190             warnings.warn(
 191                 DeprecatedSettingWarning(
 192                     u"{} parameter of {}".format(
 193                         ", ".join(map(repr, cluster_constructor_option_names)), setting
 194                     ),
 195                     u'{}["{}"]'.format(setting, cluster_option_name),
 196                     removed_in_version="8.5",
 197                 ),
 198                 stacklevel=2,
 199             )
 200         cluster = rb.Cluster(pool_cls=_shared_pool, **cluster_options)
 201     else:
 202         cluster = cluster_manager.get(options.pop(cluster_option_name, default_cluster_name))
 203 
 204     return cluster, options

 207 def get_dynamic_cluster_from_options(setting, config):
 208     cluster_name = config.get("cluster", "default")
 209     cluster_opts = options.default_manager.get("redis.clusters").get(cluster_name)
 210     if cluster_opts is not None and cluster_opts.get("is_redis_cluster"):
 211         # RedisCluster
 212         return True, redis_clusters.get(cluster_name), config
 213 
 214     # RBCluster
 215     return (False,) + get_cluster_from_options(setting, config)

 218 def validate_dynamic_cluster(is_redis_cluster, cluster):
 219     try:
 220         if is_redis_cluster:
 221             cluster.ping()
 222         else:
 223             with cluster.all() as client:
 224                 client.ping()
 225     except Exception as e:
 226         raise InvalidConfiguration(six.text_type(e))

 229 def check_cluster_versions(cluster, required, recommended=None, label=None):
 230     try:
 231         with cluster.all() as client:
 232             results = client.info()
 233     except Exception as e:
 234         # Any connection issues should be caught here.
 235         raise InvalidConfiguration(six.text_type(e))
 236 
 237     versions = {}
 238     for id, info in results.value.items():
 239         host = cluster.hosts[id]
 240         # NOTE: This assumes there is no routing magic going on here, and
 241         # all requests to this host are being served by the same database.
 242         key = u"{host}:{port}".format(host=host.host, port=host.port)
 243         versions[key] = Version(map(int, info["redis_version"].split(".", 3)))
 244 
 245     check_versions(
 246         "Redis" if label is None else "Redis (%s)" % (label,), versions, required, recommended
 247     )

 250 def load_script(path):
 251     script = Script(None, resource_string("sentry", posixpath.join("scripts", path)))
 252 
 253     # This changes the argument order of the ``Script.__call__`` method to
 254     # encourage using the script with a specific Redis client, rather
 255     # than implicitly using the first client that the script was registered
 256     # with. (This can prevent lots of bizarre behavior when dealing with
 257     # clusters of Redis servers.)
 258     def call_script(client, keys, args):
 259         u"""
 260         Executes {!r} as a Lua script on a Redis server.
 261 
 262         Takes the client to execute the script on as the first argument,
 263         followed by the values that will be provided as ``KEYS`` and ``ARGV``
 264         to the script as two sequence arguments.
 265         """.format(
 266             path
 267         )
 268         return script(keys, args, client)
 269 
 270     return call_script
