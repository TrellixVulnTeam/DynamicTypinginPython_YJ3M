   6 # Absolute paths where iOS mounts application files.
   7 IOS_APP_PATHS = (
   8     "/var/containers/Bundle/Application/",
   9     "/private/var/containers/Bundle/Application/",
  10 )
  11 
  12 # Locations which usually contain MacOS apps.
  13 MACOS_APP_PATHS = (".app/Contents/", "/Users/", "/usr/local/")
  14 
  15 # Paths which usually contain linux system or third party libraries.
  16 LINUX_SYS_PATHS = ("/lib/", "/usr/lib/", "linux-gate.so")
  17 
  18 # Regex matching the Windows folder on any drive.
  19 WINDOWS_SYS_PATH_RE = re.compile(r"^[a-z]:\\windows", re.IGNORECASE)
  20 
  21 # Regex matching well-known iOS and macOS frameworks (and our own).
  22 SUPPORT_FRAMEWORK_RE = re.compile(
  23     r"""(?x)
  24     /Frameworks/(
  25             libswift([a-zA-Z0-9]+)\.dylib$
  26         |   (KSCrash|SentrySwift|Sentry)\.framework/
  27     )
  28     """
  29 )
  30 
  31 
  32 def _is_support_framework(package):
  33     return SUPPORT_FRAMEWORK_RE.search(package) is not None

  37 def is_known_third_party(package, sdk_info=None):
  38     """
  39     Checks whether this package matches one of the well-known system image
  40     locations across platforms. The given package must not be ``None``.
  41     """
  42 
  43     # Check for common iOS and MacOS support frameworks, like Swift.
  44     if _is_support_framework(package):
  45         return True
  46 
  47     # Check for iOS app bundles in well known locations. These are definitely
  48     # not third-party, as they contain the application.
  49     if package.startswith(IOS_APP_PATHS):
  50         return False
  51 
  52     # Check for app locations in the iOS simulator
  53     if (
  54         "/Developer/CoreSimulator/Devices/" in package
  55         and "/Containers/Bundle/Application/" in package
  56     ):
  57         return False
  58 
  59     # Check for OS-specific rules
  60     sdk_name = sdk_info["sdk_name"].lower() if sdk_info else ""
  61     if sdk_name == "macos":
  62         return not any(p in package for p in MACOS_APP_PATHS)
  63     if sdk_name == "linux":
  64         return package.startswith(LINUX_SYS_PATHS)
  65     if sdk_name == "windows":
  66         return WINDOWS_SYS_PATH_RE.match(package) is not None
  67 
  68     # Everything else we don't know is considered third_party
  69     return True

  73 def is_optional_package(package, sdk_info=None):
  74     """
  75     Determines whether the given package is considered optional.
  76 
  77     This indicates that no error should be emitted if this package is missing
  78     during symbolication. Also, reprocessing should not block for this image.
  79     """
  80 
  81     if not package:
  82         return True
  83 
  84     # Support frameworks have been bundled with iOS apps at times. These are
  85     # considered optional.
  86     if _is_support_framework(package):
  87         return True
  88 
  89     # Bundled frameworks on iOS are considered optional, even though they live
  90     # in the application folder. They are not considered third party, however.
  91     if package.startswith(IOS_APP_PATHS) and "/Frameworks/" in package:
  92         return True
  93 
  94     # All other images, whether from the app bundle or not, are considered
  95     # required.
  96     return False
