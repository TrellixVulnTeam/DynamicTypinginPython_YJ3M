  53     def finalize_options(self):
  54         # This requires some explanation.  Basically what we want to do
  55         # here is to control if we want to build in-place or into the
  56         # build-lib folder.  Traditionally this is set by the `inplace`
  57         # command line flag for build_ext.  However as we are a subcommand
  58         # we need to grab this information from elsewhere.
  59         #
  60         # An in-place build puts the files generated into the source
  61         # folder, a regular build puts the files into the build-lib
  62         # folder.
  63         #
  64         # The following situations we need to cover:
  65         #
  66         #   command                         default in-place
  67         #   setup.py build_js               0
  68         #   setup.py build_ext              value of in-place for build_ext
  69         #   setup.py build_ext --inplace    1
  70         #   pip install --editable .        1
  71         #   setup.py install                0
  72         #   setup.py sdist                  0
  73         #   setup.py bdist_wheel            0
  74         #
  75         # The way this is achieved is that build_js is invoked by two
  76         # subcommands: bdist_ext (which is in our case always executed
  77         # due to a custom distribution) or sdist.
  78         #
  79         # Note: at one point install was an in-place build but it's not
  80         # quite sure why.  In case a version of install breaks again:
  81         # installations via pip from git URLs definitely require the
  82         # in-place flag to be disabled.  So we might need to detect
  83         # that separately.
  84         #
  85         # To find the default value of the inplace flag we inspect the
  86         # sdist and build_ext commands.
  87         sdist = self.distribution.get_command_obj("sdist")
  88         build_ext = self.get_finalized_command("build_ext")
  89 
  90         # If we are not decided on in-place we are inplace if either
  91         # build_ext is inplace or we are invoked through the install
  92         # command (easiest check is to see if it's finalized).
  93         if self.inplace is None:
  94             self.inplace = (build_ext.inplace or sdist.finalized) and 1 or 0
  95 
  96         # If we're coming from sdist, clear the hell out of the dist
  97         # folder first.
  98         if sdist.finalized:
  99             for path in self.get_dist_paths():
 100                 try:
 101                     shutil.rmtree(path)
 102                 except (OSError, IOError):
 103                     pass
 104 
 105         # In place means build_lib is src.  We also log this.
 106         if self.inplace:
 107             log.debug("in-place js building enabled")
 108             self.build_lib = "src"
 109         # Otherwise we fetch build_lib from the build command.
 110         else:
 111             self.set_undefined_options("build", ("build_lib", "build_lib"))
 112             log.debug("regular js build: build path is %s" % self.build_lib)
 113 
 114         if self.work_path is None:
 115             self.work_path = self.get_root_path()
 116 
 117     def _needs_built(self):
 118         for path in self.get_dist_paths():
 119             if not os.path.isdir(path):
 120                 return True
 121         return False

 123     def _setup_js_deps(self):
 124         node_version = None
 125         try:
 126             node_version = self._run_command(["node", "--version"]).rstrip()
 127         except OSError:
 128             log.fatal(u"Cannot find node executable. Please install node" " and try again.")
 129             sys.exit(1)
 130 
 131         if node_version[2] is not None:
 132             log.info(u"using node ({0})".format(node_version))
 133             self._run_command(["yarn", "install", "--production", "--frozen-lockfile", "--quiet"])
 134 
 135     def _run_command(self, cmd, env=None):
 136         cmd_str = " ".join(cmd)
 137         log.debug("running [%s]", cmd_str)
 138         try:
 139             return check_output(cmd, cwd=self.work_path, env=env, stderr=STDOUT)
 140         except CalledProcessError as err:
 141             log.error(
 142                 "[%s] failed with exit code [%s] on [%s]:\n%s",
 143                 cmd_str,
 144                 err.returncode,
 145                 self.work_path,
 146                 err.output,
 147             )
 148             raise
 149         except Exception:
 150             log.error("command failed [%s] via [%s]", cmd_str, self.work_path)
 151             raise

 153     def update_manifests(self):
 154         # if we were invoked from sdist, we need to inform sdist about
 155         # which files we just generated.  Otherwise they will be missing
 156         # in the manifest.  This adds the files for what webpack generates
 157         # plus our own assets.json file.
 158         sdist = self.distribution.get_command_obj("sdist")
 159         if not sdist.finalized:
 160             return
 161 
 162         # The path down from here only works for sdist:
 163 
 164         # Use the underlying file list so that we skip the file-exists
 165         # check which we do not want here.
 166         files = sdist.filelist.files
 167         base = os.path.abspath(".")
 168 
 169         # We need to split off the local parts of the files relative to
 170         # the current folder.  This will chop off the right path for the
 171         # manifest.
 172         for path in self.get_dist_paths():
 173             for dirname, _, filenames in os.walk(os.path.abspath(path)):
 174                 for filename in filenames:
 175                     filename = os.path.join(dirname, filename)
 176                     files.append(filename[len(base) :].lstrip(os.path.sep))
 177 
 178         for file in self.get_manifest_additions():
 179             files.append(file)
 180 
 181     def run(self):
 182         if self.force or self._needs_built():
 183             self._setup_js_deps()
 184             self._build()
 185             self.update_manifests()
