  48     def _get_package_version(self):
  49         """
  50         Attempt to get the most correct current version of Sentry.
  51         """
  52         pkg_path = os.path.join(self.work_path, "src")
  53 
  54         sys.path.insert(0, pkg_path)
  55         try:
  56             import sentry
  57         except Exception:
  58             version = None
  59             build = None
  60         else:
  61             log.info(u"pulled version information from 'sentry' module".format(sentry.__file__))
  62             version = self.distribution.get_version()
  63             build = sentry.__build__
  64         finally:
  65             sys.path.pop(0)
  66 
  67         if not (version and build):
  68             json_path = self.get_asset_json_path()
  69             try:
  70                 with open(json_path) as fp:
  71                     data = json.loads(fp.read())
  72             except Exception:
  73                 pass
  74             else:
  75                 log.info(u"pulled version information from '{}'".format(json_path))
  76                 version, build = data["version"], data["build"]
  77 
  78         return {"version": version, "build": build}

  80     def _needs_static(self, version_info):
  81         json_path = self.get_asset_json_path()
  82         if not os.path.exists(json_path):
  83             return True
  84 
  85         with open(json_path) as fp:
  86             data = json.load(fp)
  87         if data.get("version") != version_info.get("version"):
  88             return True
  89         if data.get("build") != version_info.get("build"):
  90             return True
  91         return False

  93     def _needs_built(self):
  94         if BaseBuildCommand._needs_built(self):
  95             return True
  96         version_info = self._get_package_version()
  97         return self._needs_static(version_info)

  99     def _build(self):
 100         version_info = self._get_package_version()
 101         log.info(
 102             u"building assets for {} v{} (build {})".format(
 103                 self.distribution.get_name(),
 104                 version_info["version"] or "UNKNOWN",
 105                 version_info["build"] or "UNKNOWN",
 106             )
 107         )
 108         if not version_info["version"] or not version_info["build"]:
 109             log.fatal("Could not determine sentry version or build")
 110             sys.exit(1)
 111 
 112         try:
 113             self._build_static()
 114         except Exception:
 115             traceback.print_exc()
 116             log.fatal("unable to build Sentry's static assets!")
 117             sys.exit(1)
 118 
 119         log.info("writing version manifest")
 120         manifest = self._write_version_file(version_info)
 121         log.info(u"recorded manifest\n{}".format(json.dumps(manifest, indent=2)))
 122 
 123     def _build_static(self):
 124         # By setting NODE_ENV=production, a few things happen
 125         #   * React optimizes out certain code paths
 126         #   * Webpack will add version strings to built/referenced assets
 127         env = dict(os.environ)
 128         env["SENTRY_STATIC_DIST_PATH"] = self.sentry_static_dist_path
 129         env["NODE_ENV"] = "production"
 130         # TODO: Our JS builds should not require 4GB heap space
 131         env["NODE_OPTIONS"] = (
 132             (env.get("NODE_OPTIONS", "") + " --max-old-space-size=4096")
 133         ).lstrip()
 134         self._run_command(["yarn", "tsc"], env=env)
 135         self._run_command(["yarn", "webpack", "--bail"], env=env)
 136 
 137     def _write_version_file(self, version_info):
 138         manifest = {
 139             "createdAt": datetime.datetime.utcnow().isoformat() + "Z",
 140             "version": version_info["version"],
 141             "build": version_info["build"],
 142         }
 143         with open(self.get_asset_json_path(), "w") as fp:
 144             json.dump(manifest, fp)
 145         return manifest
