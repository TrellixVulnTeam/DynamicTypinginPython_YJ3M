  72     def element(self, selector=None, xpath=None):
  73         """
  74         Get an element from the page. This method will wait for the element to show up.
  75         """
  76 
  77         if xpath is not None:
  78             self.wait_until(xpath=xpath)
  79             return self.driver.find_element_by_xpath(xpath)

  81             self.wait_until(selector)

 111     def click_when_visible(self, selector=None, timeout=3):
 112         """
 113         Waits until ``selector`` is available to be clicked before attempting to click
 114         """
 115         if selector:
 116             self.wait_until_clickable(selector, timeout)
 117             self.click(selector)
 118         else:
 119             raise ValueError
 120

 135     def wait_until_clickable(self, selector=None, timeout=10):
 136         """
 137         Waits until ``selector`` is visible and enabled to be clicked, or until ``timeout``
 138         is hit, whichever happens first.
 139         """
 140         from selenium.webdriver.common.by import By
 141 
 142         if selector:
 143             condition = expected_conditions.element_to_be_clickable((By.CSS_SELECTOR, selector))
 144         else:
 145             raise ValueError
 146 
 147         WebDriverWait(self.driver, timeout).until(condition)
 148 
 149         return self
 150 
 151     def wait_until(self, selector=None, xpath=None, title=None, timeout=10):
 152         """
 153         Waits until ``selector`` is found in the browser, or until ``timeout``
 154         is hit, whichever happens first.
 155         """
 156         from selenium.webdriver.common.by import By
 157 
 158         if selector:
 159             condition = expected_conditions.presence_of_element_located((By.CSS_SELECTOR, selector))
 160         elif xpath:
 161             condition = expected_conditions.presence_of_element_located((By.XPATH, xpath))
 162         elif title:
 163             condition = expected_conditions.title_is(title)
 164         else:
 165             raise ValueError
 166 
 167         WebDriverWait(self.driver, timeout).until(condition)
 168 
 169         return self
 170 
 171     def wait_until_test_id(self, selector):
 172         return self.wait_until('[data-test-id="%s"]' % (selector))

 174     def wait_until_not(self, selector=None, title=None, timeout=10):
 175         """
 176         Waits until ``selector`` is NOT found in the browser, or until
 177         ``timeout`` is hit, whichever happens first.
 178         """
 179         from selenium.webdriver.common.by import By
 180 
 181         if selector:
 182             condition = expected_conditions.presence_of_element_located((By.CSS_SELECTOR, selector))
 183         elif title:
 184             condition = expected_conditions.title_is(title)
 185         else:
 186             raise
 187 
 188         WebDriverWait(self.driver, timeout).until_not(condition)
 189 
 190         return self

 196     def implicitly_wait(self, duration):
 197         """
 198         An implicit wait tells WebDriver to poll the DOM for a certain amount of
 199         time when trying to find any element (or elements) not immediately
 200         available. The default setting is 0. Once set, the implicit wait is set
 201         for the life of the WebDriver object.
 202         """
 203         self.driver.implicitly_wait(duration)

 241     def save_cookie(
 242         self,
 243         name,
 244         value,
 245         domain=None,
 246         path="/",
 247         expires="Tue, 20 Jun 2025 19:07:44 GMT",
 248         max_age=None,
 249         secure=None,
 250     ):
 251         domain = domain or self.domain
 252         # Recent changes to Chrome no longer allow us to explicitly set a cookie domain
 253         # to be localhost. If no domain is specified, the cookie will be created on
 254         # the host of the current url that the browser has visited.
 255         if domain == "localhost":
 256             domain = None
 257         cookie = {
 258             "name": name,
 259             "value": value,
 260             "expires": expires,
 261             "path": path,
 262             "max-age": max_age,
 263             "secure": secure,
 264         }
 265         if domain:
 266             cookie["domain"] = domain
 267 
 268         # XXX(dcramer): the cookie store must be initialized via a URL
 269         if not self._has_initialized_cookie_store:
 270             logger.info("selenium.initialize-cookies")
 271             self.get("/")
 272 
 273         # XXX(dcramer): PhantomJS does not let us add cookies with the native
 274         # selenium API because....
 275         # http://stackoverflow.com/questions/37103621/adding-cookies-working-with-firefox-webdriver-but-not-in-phantomjs
 276 
 277         # TODO(dcramer): this should be escaped, but idgaf
 278         logger.info(u"selenium.set-cookie.{}".format(name), extra={"value": value})
 279         if isinstance(self.driver, webdriver.PhantomJS):
 280             self.driver.execute_script(
 281                 u"document.cookie = '{name}={value}; path={path}; domain={domain}; expires={expires}'; max-age={max_age}\n".format(
 282                     **cookie
 283                 )
 284             )
 285         else:
 286             # XXX(dcramer): chromedriver (of certain versions) is complaining about this being
 287             # an invalid kwarg
 288             del cookie["secure"]
 289             self.driver.add_cookie(cookie)

 346     window_size = request.config.getoption("window_size")
 347     window_width, window_height = map(int, window_size.split("x", 1))
 348 
 349     driver_type = request.config.getoption("selenium_driver")
 350     headless = not request.config.getoption("no_headless")
 351     if driver_type == "chrome":
 352         options = webdriver.ChromeOptions()
 353         options.add_argument("no-sandbox")
 354         options.add_argument("disable-gpu")
 355         options.add_argument(u"window-size={}".format(window_size))
 356         if headless:
 357             options.add_argument("headless")
 358         chrome_path = request.config.getoption("chrome_path")
 359         if chrome_path:
 360             options.binary_location = chrome_path
 361         chromedriver_path = request.config.getoption("chromedriver_path")
 362         chrome_args = {"options": options}
 363         if chromedriver_path:
 364             chrome_args["executable_path"] = chromedriver_path
 365 
 366         driver = start_chrome(**chrome_args)
 367     elif driver_type == "firefox":
 368         driver = webdriver.Firefox()
 369     elif driver_type == "phantomjs":
 370         phantomjs_path = request.config.getoption("phantomjs_path")
 371         if not phantomjs_path:
 372             phantomjs_path = os.path.join("node_modules", "phantomjs-prebuilt", "bin", "phantomjs")
 373         driver = webdriver.PhantomJS(executable_path=phantomjs_path)
 374     else:
 375         raise pytest.UsageError("--driver must be specified")
 376 
 377     driver.set_window_size(window_width, window_height)

 379     def fin():

 430 def _gather_url(item, report, driver, summary, extra):

 467 def _gather_logs(item, report, driver, summary, extra):
 468     try:
 469         types = driver.log_types
 470     except Exception as e:
 471         # note that some drivers may not implement log types
 472         summary.append(u"WARNING: Failed to gather log types: {0}".format(e))
 473         return
 474     for name in types:
 475         try:
 476             log = driver.get_log(name)
 477         except Exception as e:
 478             summary.append(u"WARNING: Failed to gather {0} log: {1}".format(name, e))
 479             return
 480         pytest_html = item.config.pluginmanager.getplugin("html")
 481         if pytest_html is not None:
 482             extra.append(pytest_html.extras.text(format_log(log), "%s Log" % name.title()))

 485 def format_log(log):
 486     timestamp_format = "%Y-%m-%d %H:%M:%S.%f"
 487     entries = [
 488         u"{0} {1[level]} - {1[message]}".format(
 489             datetime.utcfromtimestamp(entry["timestamp"] / 1000.0).strftime(timestamp_format), entry
 490         ).rstrip()
 491         for entry in log
 492     ]
 493     log = "\n".join(entries)
 494     log = log.encode("utf-8")
 495     return log
