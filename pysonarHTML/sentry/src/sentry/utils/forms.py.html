   8 def field_to_config(name, field):
   9     config = {
  10         "name": name,
  11         "label": field.label or name.replace("_", " ").title(),
  12         "placeholder": field.widget.attrs.get("placeholder"),
  13         "help": field.help_text,
  14         "required": field.required,
  15     }
  16     if isinstance(field, forms.URLField):
  17         config["type"] = "url"
  18     elif isinstance(field, forms.IntegerField):
  19         config["type"] = "number"
  20     elif isinstance(field, forms.BooleanField):
  21         config["type"] = "bool"
  22     elif isinstance(field, forms.EmailField):
  23         config["type"] = "email"
  24     elif isinstance(field, forms.CharField):
  25         if isinstance(field.widget, forms.PasswordInput):
  26             config["type"] = "secret"
  27         elif isinstance(field.widget, forms.Textarea):
  28             config["type"] = "textarea"
  29         else:
  30             config["type"] = "text"
  31     elif isinstance(field, forms.ChoiceField):
  32         config["type"] = "select"
  33         config["choices"] = field.choices
  34     return config

  37 def form_to_config(form):
  38     config = []
  39     for name, field in six.iteritems(form.base_fields):
  40         row = field_to_config(name, field)
  41         row["default"] = field.initial
  42         config.append(row)
  43     return config
