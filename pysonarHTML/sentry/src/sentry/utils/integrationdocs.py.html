  59 def dump_doc(path, data):
  60     fn = os.path.join(DOC_FOLDER, path + ".json")
  61     directory = os.path.dirname(fn)
  62     try:
  63         os.makedirs(directory)
  64     except OSError:
  65         pass
  66     with io.open(fn, "wt", encoding="utf-8") as f:
  67         # XXX: ideally, we use six.text_type here, but we can't use six.
  68         f.write(unicode(json.dumps(data, indent=2)))  # NOQA
  69         f.write(u"\n")

  72 def load_doc(path):
  73     if "/" in path:
  74         return None
  75     fn = os.path.join(DOC_FOLDER, path + ".json")
  76     try:
  77         with io.open(fn, "rt", encoding="utf-8") as f:

  89 def urlopen_with_retries(url, timeout=5, retries=10):
  90     for i in range(retries):
  91         try:
  92             return urlopen(url, timeout=timeout)
  93         except Exception:
  94             if i == retries - 1:
  95                 raise
  96             time.sleep(i * 0.01)

  99 def sync_docs(quiet=False):
 100     if not quiet:
 101         echo("syncing documentation (platform index)")
 102     body = urlopen_with_retries(BASE_URL.format("_index.json")).read().decode("utf-8")
 103     data = json.loads(body)
 104     platform_list = []
 105     for platform_id, integrations in iteritems(data["platforms"]):
 106         platform_list.append(
 107             {
 108                 "id": platform_id,
 109                 "name": integrations["_self"]["name"],
 110                 "integrations": [
 111                     {
 112                         "id": get_integration_id(platform_id, i_id),
 113                         "name": i_data["name"],
 114                         "type": i_data["type"],
 115                         "link": i_data["doc_link"],
 116                     }
 117                     for i_id, i_data in sorted(iteritems(integrations), key=lambda x: x[1]["name"])
 118                 ],
 119             }
 120         )
 121 
 122     platform_list.sort(key=lambda x: x["name"])
 123 
 124     dump_doc("_platforms", {"platforms": platform_list})
 125 
 126     # This value is derived from https://docs.python.org/3/library/concurrent.futures.html#threadpoolexecutor
 127     MAX_THREADS = 32
 128     # TODO(python3): Migrate this to concurrent.futures.ThreadPoolExecutor context manager
 129     executor = multiprocessing.dummy.Pool(
 130         min(len(data["platforms"]), multiprocessing.cpu_count() * 5, MAX_THREADS)
 131     )
 132     for platform_id, platform_data in iteritems(data["platforms"]):
 133         for integration_id, integration in iteritems(platform_data):
 134             executor.apply_async(
 135                 sync_integration_docs, (platform_id, integration_id, integration["details"], quiet)
 136             )
 137     executor.close()
 138     executor.join()

 141 def sync_integration_docs(platform_id, integration_id, path, quiet=False):
 142     if not quiet:
 143         echo("  syncing documentation for %s.%s integration" % (platform_id, integration_id))

 145     data = json.load(urlopen_with_retries(BASE_URL.format(path)))

 152 def integration_doc_exists(integration_id):
 153     # We use listdir() here as integration_id comes from user data
 154     # and using os.path.join() would allow directory traversal vulnerabilities
 155     # which we don't want.
 156     docs = os.listdir(DOC_FOLDER)
 157     filename = u"{}.json".format(integration_id)
 158     return filename in docs
