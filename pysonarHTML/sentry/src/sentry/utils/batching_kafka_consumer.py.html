  28 class AbstractBatchWorker(object):
  29     """The `BatchingKafkaConsumer` requires an instance of this class to
  30     handle user provided work such as processing raw messages and flushing
  31     processed batches to a custom backend."""
  32 
  33     @abc.abstractmethod
  34     def process_message(self, message):
  35         """Called with each (raw) Kafka message, allowing the worker to do
  36         incremental (preferablly local!) work on events. The object returned
  37         is put into the batch maintained by the `BatchingKafkaConsumer`.
  38 
  39         If this method returns `None` it is not added to the batch.
  40 
  41         A simple example would be decoding the JSON value and extracting a few
  42         fields.
  43         """
  44         pass
  45 
  46     @abc.abstractmethod
  47     def flush_batch(self, batch):
  48         """Called with a list of pre-processed (by `process_message`) objects.
  49         The worker should write the batch of processed messages into whatever
  50         store(s) it is maintaining. Afterwards the Kafka offsets are committed.
  51 
  52         A simple example would be writing the batch to another Kafka topic.
  53         """
  54         pass
  55 
  56     @abc.abstractmethod
  57     def shutdown(self):
  58         """Called when the `BatchingKafkaConsumer` is shutting down (because it
  59         was signalled to do so). Provides the worker a chance to do any final
  60         cleanup.
  61 
  62         A simple example would be closing any remaining backend connections."""
  63         pass

  66 class BatchingKafkaConsumer(object):
  67     """The `BatchingKafkaConsumer` is an abstraction over most Kafka consumer's main event
  68     loops. For this reason it uses inversion of control: the user provides an implementation
  69     for the `AbstractBatchWorker` and then the `BatchingKafkaConsumer` handles the rest.
  70 
  71     Main differences from the default KafkaConsumer are as follows:
  72     * Messages are processed locally (e.g. not written to an external datastore!) as they are
  73       read from Kafka, then added to an in-memory batch
  74     * Batches are flushed based on the batch size or time sent since the first message
  75       in the batch was received (e.g. "500 items or 1000ms")
  76     * Kafka offsets are not automatically committed! If they were, offsets might be committed
  77       for messages that are still sitting in an in-memory batch, or they might *not* be committed
  78       when messages are sent to an external datastore right before the consumer process dies
  79     * Instead, when a batch of items is flushed they are written to the external datastore and
  80       then Kafka offsets are immediately committed (in the same thread/loop)
  81     * Users need only provide an implementation of what it means to process a raw message
  82       and flush a batch of events
  83     * Supports an optional "dead letter topic" where messages that raise an exception during
  84       `process_message` are sent so as not to block the pipeline.
  85 
  86     NOTE: This does not eliminate the possibility of duplicates if the consumer process
  87     crashes between writing to its backend and commiting Kafka offsets. This should eliminate
  88     the possibility of *losing* data though. An "exactly once" consumer would need to store
  89     offsets in the external datastore and reconcile them on any partition rebalance.
  90     """
  91 
  92     # Set of logical (not literal) offsets to not publish to the commit log.
  93     # https://github.com/confluentinc/confluent-kafka-python/blob/443177e1c83d9b66ce30f5eb8775e062453a738b/tests/test_enums.py#L22-L25
  94     LOGICAL_OFFSETS = frozenset([OFFSET_BEGINNING, OFFSET_END, OFFSET_STORED, OFFSET_INVALID])
  95 
  96     # Set of error codes that can be returned by ``consumer.poll`` calls which
  97     # are generally able to be recovered from after a series of retries.
  98     RECOVERABLE_ERRORS = frozenset(
  99         [KafkaError._PARTITION_EOF, KafkaError._TRANSPORT]  # Local: Broker transport failure

 124         self.max_batch_time = max_batch_time  # in milliseconds
 125         self.__metrics = metrics
 126         self.__metrics_sample_rates = (

 130         self.group_id = group_id

 143         if isinstance(topics, (tuple, set)):

 161     def __record_timing(self, metric, value, tags=None):
 162         if self.__metrics is None:
 163             return
 164 
 165         tags = dict(tags or ())
 166         tags.update(self.__metrics_default_tags)
 167 
 168         sample_rate = self.__metrics_sample_rates.get(metric, settings.SENTRY_METRICS_SAMPLE_RATE)
 169         return self.__metrics.timing(metric, value, tags=tags, sample_rate=sample_rate)

 171     def create_consumer(
 172         self,
 173         topics,
 174         bootstrap_servers,
 175         group_id,
 176         auto_offset_reset,
 177         queued_max_messages_kbytes,
 178         queued_min_messages,
 179     ):
 180 
 181         consumer_config = {
 182             "enable.auto.commit": False,
 183             "bootstrap.servers": ",".join(bootstrap_servers),
 184             "group.id": group_id,
 185             "default.topic.config": {"auto.offset.reset": auto_offset_reset},
 186             # overridden to reduce memory usage when there's a large backlog
 187             "queued.max.messages.kbytes": queued_max_messages_kbytes,
 188             "queued.min.messages": queued_min_messages,
 189         }
 190 
 191         consumer = Consumer(consumer_config)
 192 
 193         def on_partitions_assigned(consumer, partitions):
 194             logger.info("New partitions assigned: %r", partitions)
 195 
 196         def on_partitions_revoked(consumer, partitions):
 197             "Reset the current in-memory batch, letting the next consumer take over where we left off."
 198             logger.info("Partitions revoked: %r", partitions)
 199             self._flush(force=True)
 200 
 201         consumer.subscribe(
 202             topics, on_assign=on_partitions_assigned, on_revoke=on_partitions_revoked
 203         )
 204 
 205         return consumer

 207     def run(self):
 208         "The main run loop, see class docstring for more information."
 209 
 210         logger.debug("Starting")
 211         while not self.shutdown:
 212             self._run_once()
 213 
 214         self._shutdown()
 215 
 216     def _run_once(self):
 217         self._flush()
 218 
 219         if self.producer:
 220             self.producer.poll(0.0)
 221 
 222         msg = self.consumer.poll(timeout=1.0)
 223 
 224         if msg is None:
 225             return
 226         if msg.error():
 227             if msg.error().code() in self.RECOVERABLE_ERRORS:
 228                 return
 229             else:
 230                 raise Exception(msg.error())
 231 
 232         self._handle_message(msg)
 233 
 234     def signal_shutdown(self):
 235         """Tells the `BatchingKafkaConsumer` to shutdown on the next run loop iteration.
 236         Typically called from a signal handler."""
 237         logger.debug("Shutdown signalled")
 238 
 239         self.shutdown = True
 240 
 241     def _handle_message(self, msg):
 242         start = time.time()
 243 
 244         # set the deadline only after the first message for this batch is seen
 245         if not self.__batch_deadline:
 246             self.__batch_deadline = self.max_batch_time / 1000.0 + start
 247 
 248         try:
 249             result = self.worker.process_message(msg)
 250         except Exception:
 251             if self.dead_letter_topic:
 252                 logger.exception("Error handling message, sending to dead letter topic.")
 253                 self.producer.produce(
 254                     self.dead_letter_topic,
 255                     key=msg.key(),
 256                     value=msg.value(),
 257                     headers={
 258                         "partition": six.text_type(msg.partition()) if msg.partition() else None,
 259                         "offset": six.text_type(msg.offset()) if msg.offset() else None,
 260                         "topic": msg.topic(),
 261                     },
 262                     on_delivery=self._commit_message_delivery_callback,
 263                 )
 264             else:
 265                 raise
 266         else:
 267             if result is not None:
 268                 self.__batch_results.append(result)
 269         finally:
 270             duration = (time.time() - start) * 1000
 271             self.__batch_messages_processed_count += 1
 272             self.__batch_processing_time_ms += duration
 273             self.__record_timing("process_message", duration)
 274 
 275             topic_partition_key = (msg.topic(), msg.partition())
 276             if topic_partition_key in self.__batch_offsets:
 277                 self.__batch_offsets[topic_partition_key][1] = msg.offset()
 278             else:
 279                 self.__batch_offsets[topic_partition_key] = [msg.offset(), msg.offset()]
 280 
 281     def _shutdown(self):
 282         logger.debug("Stopping")
 283 
 284         # drop in-memory events, letting the next consumer take over where we left off
 285         self._reset_batch()
 286 
 287         # tell the consumer to shutdown, and close the consumer
 288         logger.debug("Stopping worker")
 289         self.worker.shutdown()
 290         logger.debug("Stopping consumer")
 291         self.consumer.close()
 292         logger.debug("Stopped")
 293 
 294     def _reset_batch(self):
 295         logger.debug("Resetting in-memory batch")
 296         self.__batch_results = []
 297         self.__batch_offsets = {}
 298         self.__batch_deadline = None
 299         self.__batch_messages_processed_count = 0
 300         self.__batch_processing_time_ms = 0.0
 301 
 302     def _flush(self, force=False):
 303         """Decides whether the `BatchingKafkaConsumer` should flush because of either
 304         batch size or time. If so, delegate to the worker, clear the current batch,
 305         and commit offsets to Kafka."""
 306         if not self.__batch_messages_processed_count > 0:
 307             return  # No messages were processed, so there's nothing to do.
 308 
 309         batch_by_size = len(self.__batch_results) >= self.max_batch_size
 310         batch_by_time = self.__batch_deadline and time.time() > self.__batch_deadline
 311         if not (force or batch_by_size or batch_by_time):
 312             return
 313 
 314         logger.info(
 315             "Flushing %s items (from %r): forced:%s size:%s time:%s",
 316             len(self.__batch_results),
 317             self.__batch_offsets,
 318             force,
 319             batch_by_size,
 320             batch_by_time,
 321         )
 322 
 323         self.__record_timing(
 324             "process_message.normalized",
 325             self.__batch_processing_time_ms / self.__batch_messages_processed_count,
 326         )
 327 
 328         batch_results_length = len(self.__batch_results)
 329         self.__record_timing("batching_consumer.batch.size", batch_results_length)
 330         if batch_results_length > 0:
 331             logger.debug("Flushing batch via worker")
 332             flush_start = time.time()
 333             self.worker.flush_batch(self.__batch_results)
 334             flush_duration = (time.time() - flush_start) * 1000
 335             logger.info("Worker flush took %dms", flush_duration)
 336             self.__record_timing("batching_consumer.batch.flush", flush_duration)
 337             self.__record_timing(
 338                 "batching_consumer.batch.flush.normalized", flush_duration / batch_results_length
 339             )
 340 
 341         logger.debug("Committing Kafka offsets")
 342         commit_start = time.time()
 343         self._commit()
 344         commit_duration = (time.time() - commit_start) * 1000
 345         logger.debug("Kafka offset commit took %dms", commit_duration)
 346 
 347         self._reset_batch()
 348 
 349     def _commit_message_delivery_callback(self, error, message):
 350         if error is not None:
 351             raise Exception(error.str())
 352 
 353     def _commit(self):
 354         retries = 3
 355         while True:
 356             try:
 357                 offsets = self.consumer.commit(asynchronous=False)
 358                 logger.debug("Committed offsets: %s", offsets)
 359                 break  # success
 360             except KafkaException as e:
 361                 if e.args[0].code() in (
 362                     KafkaError.REQUEST_TIMED_OUT,
 363                     KafkaError.NOT_COORDINATOR_FOR_GROUP,
 364                     KafkaError._WAIT_COORD,
 365                 ):
 366                     logger.warning("Commit failed: %s (%d retries)", e, retries)
 367                     if retries <= 0:
 368                         raise
 369                     retries -= 1
 370                     time.sleep(1)
 371                     continue
 372                 else:
 373                     raise
 374 
 375         if self.commit_log_topic:
 376             for item in offsets:
 377                 if item.offset in self.LOGICAL_OFFSETS:
 378                     logger.debug(
 379                         "Skipped publishing logical offset (%r) to commit log for %s/%s",
 380                         item.offset,
 381                         item.topic,
 382                         item.partition,
 383                     )
 384                     continue
 385                 elif item.offset < 0:
 386                     logger.warning(
 387                         "Found unexpected negative offset (%r) after commit for %s/%s",
 388                         item.offset,
 389                         item.topic,
 390                         item.partition,
 391                     )
 392 
 393                 self.producer.produce(
 394                     self.commit_log_topic,
 395                     key="{}:{}:{}".format(item.topic, item.partition, self.group_id).encode(
 396                         "utf-8"
 397                     ),
 398                     value="{}".format(item.offset).encode("utf-8"),
 399                     on_delivery=self._commit_message_delivery_callback,
 400                 )
