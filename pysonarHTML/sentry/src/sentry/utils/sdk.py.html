  34 def is_current_event_safe():
  35     """
  36     Tests the current stack for unsafe locations that would likely cause
  37     recursion if an attempt to send to Sentry was made.
  38     """
  39 
  40     with configure_scope() as scope:
  41 
  42         # Scope was explicitly marked as unsafe
  43         if scope._tags.get(UNSAFE_TAG):
  44             return False
  45 
  46         project_id = scope._tags.get("project")
  47 
  48         if project_id and project_id == settings.SENTRY_PROJECT:
  49             return False
  50 
  51     for _, filename, _, _, _, _ in inspect.stack():
  52         if filename.endswith(UNSAFE_FILES):
  53             return False
  54 
  55     return True

  58 def mark_scope_as_unsafe():
  59     """
  60     Set the unsafe tag on the SDK scope for outgoing crashe and transactions.
  61 
  62     Marking a scope explicitly as unsafe allows the recursion breaker to
  63     decide early, before walking the stack and checking for unsafe files.
  64     """
  65     with configure_scope() as scope:
  66         scope.set_tag(UNSAFE_TAG, True)

  69 def set_current_project(project_id):
  70     """
  71     Set the current project on the SDK scope for outgoing crash reports.
  72 
  73     This is a dedicated function because it is also important for the recursion
  74     breaker to work. You really should set the project in every task that is
  75     relevant to event processing, or that task may crash ingesting
  76     sentry-internal errors, causing infinite recursion.
  77     """
  78     with configure_scope() as scope:
  79         scope.set_tag("project", project_id)

  82 def get_project_key():
  83     from sentry.models import ProjectKey
  84 
  85     if not settings.SENTRY_PROJECT:
  86         return None
  87 
  88     key = None
  89     try:
  90         if settings.SENTRY_PROJECT_KEY is not None:
  91             key = ProjectKey.objects.get(
  92                 id=settings.SENTRY_PROJECT_KEY, project=settings.SENTRY_PROJECT
  93             )
  94         else:
  95             key = ProjectKey.get_default(settings.SENTRY_PROJECT)
  96     except Exception as exc:
  97         # if the relation fails to query or is missing completely, lets handle
  98         # it gracefully
  99         sdk_logger.warn(
 100             "internal-error.unable-to-fetch-project",
 101             extra={
 102                 "project_id": settings.SENTRY_PROJECT,
 103                 "project_key": settings.SENTRY_PROJECT_KEY,
 104                 "error_message": six.text_type(exc),
 105             },
 106         )
 107     if key is None:
 108         sdk_logger.warn(
 109             "internal-error.no-project-available",
 110             extra={
 111                 "project_id": settings.SENTRY_PROJECT,
 112                 "project_key": settings.SENTRY_PROJECT_KEY,
 113             },
 114         )
 115     return key

 125 def configure_sdk():
 126     from sentry_sdk.integrations.logging import LoggingIntegration
 127     from sentry_sdk.integrations.django import DjangoIntegration
 128     from sentry_sdk.integrations.celery import CeleryIntegration
 129     from sentry_sdk.integrations.redis import RedisIntegration
 130 
 131     assert sentry_sdk.Hub.main.client is None
 132 
 133     sdk_options = dict(settings.SENTRY_SDK_CONFIG)
 134 
 135     relay_dsn = sdk_options.pop("relay_dsn", None)
 136     internal_project_key = get_project_key()
 137     upstream_dsn = sdk_options.pop("dsn", None)
 138 
 139     if upstream_dsn:
 140         upstream_transport = make_transport(get_options(dsn=upstream_dsn, **sdk_options))
 141     else:
 142         upstream_transport = None
 143 
 144     if relay_dsn:
 145         relay_transport = make_transport(get_options(dsn=relay_dsn, **sdk_options))
 146     elif internal_project_key and internal_project_key.dsn_private:
 147         relay_transport = make_transport(
 148             get_options(dsn=internal_project_key.dsn_private, **sdk_options)
 149         )
 150     else:
 151         relay_transport = None
 152 
 153     def capture_event(event):
 154         if event.get("type") == "transaction" and options.get(
 155             "transaction-events.force-disable-internal-project"
 156         ):
 157             return
 158 
 159         # Upstream should get the event first because it is most isolated from
 160         # the this sentry installation.
 161         if upstream_transport:
 162             metrics.incr("internal.captured.events.upstream")
 163             # TODO(mattrobenolt): Bring this back safely.
 164             # from sentry import options
 165             # install_id = options.get('sentry:install-id')
 166             # if install_id:
 167             #     event.setdefault('tags', {})['install-id'] = install_id
 168             upstream_transport.capture_event(event)
 169 
 170         if relay_transport and options.get("store.use-relay-dsn-sample-rate") == 1:
 171             if is_current_event_safe():
 172                 metrics.incr("internal.captured.events.relay")
 173                 relay_transport.capture_event(event)
 174             else:
 175                 metrics.incr("internal.uncaptured.events.relay", skip_internal=False)
 176                 if event.get("type") != "transaction":
 177                     sdk_logger.warn("internal-error.unsafe-stacktrace.relay")
 178 
 179     sentry_sdk.init(
 180         transport=capture_event,
 181         integrations=[
 182             DjangoIntegration(),
 183             CeleryIntegration(),
 184             LoggingIntegration(event_level=None),
 185             RustInfoIntegration(),
 186             RedisIntegration(),
 187         ],
 188         traceparent_v2=True,
 189         **sdk_options
 190     )

 193 class RavenShim(object):
 194     """Wrapper around sentry-sdk in case people are writing their own
 195     integrations that rely on this being here."""
 196 
 197     def captureException(self, exc_info=None, **kwargs):
 198         with sentry_sdk.push_scope() as scope:
 199             self._kwargs_into_scope(scope, **kwargs)
 200             return capture_exception(exc_info)
 201 
 202     def captureMessage(self, msg, **kwargs):
 203         with sentry_sdk.push_scope() as scope:
 204             self._kwargs_into_scope(scope, **kwargs)
 205             return capture_message(msg)
 206 
 207     def tags_context(self, tags):
 208         with sentry_sdk.configure_scope() as scope:
 209             for k, v in tags.items():
 210                 scope.set_tag(k, v)
 211 
 212     def _kwargs_into_scope(self, scope, extra=None, tags=None, fingerprint=None, request=None):
 213         for key, value in extra.items() if extra else ():
 214             scope.set_extra(key, value)
 215         for key, value in tags.items() if tags else ():
 216             scope.set_tag(key, value)
 217         if fingerprint is not None:
 218             scope.fingerprint = fingerprint

 221 def bind_organization_context(organization):
 222     helper = settings.SENTRY_ORGANIZATION_CONTEXT_HELPER
 223 
 224     # XXX(dcramer): this is duplicated in organizationContext.jsx on the frontend
 225     with sentry_sdk.configure_scope() as scope:
 226         scope.set_tag("organization", organization.id)
 227         scope.set_tag("organization.slug", organization.slug)
 228         scope.set_context("organization", {"id": organization.id, "slug": organization.slug})
 229         if helper:
 230             try:
 231                 helper(scope=scope, organization=organization)
 232             except Exception:
 233                 sdk_logger.exception(
 234                     "internal-error.organization-context",
 235                     extra={"organization_id": organization.id},
 236                 )
