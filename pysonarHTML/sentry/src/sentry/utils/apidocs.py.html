  40 def simplify_regex(pattern):
  41     """Clean up urlpattern regexes into something somewhat readable by
  42     Mere Humans: turns something like
  43     "^(?P<sport_slug>\w+)/athletes/(?P<athlete_slug>\w+)/$" into
  44     "{sport_slug}/athletes/{athlete_slug}/"
  45     """
  46     pattern = optional_group_matcher.sub(lambda m: "[%s]" % m.group(1), pattern)
  47 
  48     # handle named groups first
  49     pattern = named_group_matcher.sub(lambda m: "{%s}" % m.group(1), pattern)
  50 
  51     # handle non-named groups
  52     pattern = non_named_group_matcher.sub("{var}", pattern)
  53 
  54     # handle optional params
  55     pattern = either_option_matcher.sub(lambda m: m.group(1), pattern)
  56 
  57     # clean up any outstanding regex-y characters.
  58     pattern = (
  59         pattern.replace("^", "")
  60         .replace("$", "")
  61         .replace("?", "")
  62         .replace("//", "/")
  63         .replace("\\", "")
  64     )
  65     if not pattern.startswith("/"):
  66         pattern = "/" + pattern
  67     return pattern

  95 def parse_doc_string(doc):
  96     """
  97     Parse a docstring into a tuple.
  98 
  99     The tuple contains:
 100 
 101     (title, lines, warning, params)
 102 
 103     `lines` is a list for backwards compatibility with
 104     the JSON formatter.
 105     """
 106     title = None
 107     current_param = ""
 108     in_warning = False
 109     param_lines = []
 110     lines = []
 111     warning = []
 112     iterable = iter((doc or u"").splitlines())
 113 
 114     for line in iterable:
 115         stripped = line.strip()
 116         if title is None:
 117             if not line:
 118                 continue
 119             title = line.strip()
 120         elif stripped and stripped[0] * len(stripped) == stripped:
 121             # is an RST underline
 122             continue
 123         elif rst_block_re.match(stripped):
 124             # Presently the only RST block we use is `caution` which
 125             # displays as a 'warning'
 126             in_warning = True
 127         elif line and stripped.startswith(":"):
 128             # Is a new parameter or other annotation
 129             if current_param:
 130                 param_lines.append(current_param)
 131             current_param = stripped
 132         elif current_param:
 133             # Adding to an existing parameter annotation
 134             current_param = current_param + " " + stripped
 135         else:
 136             if in_warning:
 137                 # If we're indented at least 2 spaces assume
 138                 # we're in the RST block
 139                 if rst_indent_re.match(line) or not line:
 140                     warning.append(stripped)
 141                     continue
 142                 # An un-indented non-empty line means we
 143                 # have other content.
 144                 elif line:
 145                     in_warning = False
 146             # Normal text. We want empty lines here so we can
 147             # preserve paragraph breaks.
 148             lines.append(stripped)
 149 
 150     if current_param:
 151         param_lines.append(current_param)
 152 
 153     if warning:
 154         warning = "\n".join(warning).strip()
 155     if not warning:
 156         warning = None
 157 
 158     return title, lines, warning, parse_params(param_lines)

 161 def get_node_text(nodes):
 162     """Recursively read text from a node tree."""
 163     text = []
 164     format_tags = {"literal": "`", "strong": "**", "emphasis": "*"}
 165     for node in nodes:
 166         # Handle inline formatting elements.
 167         if node.nodeType == node.ELEMENT_NODE and node.tagName in format_tags:
 168             wrap = format_tags[node.tagName]
 169             text.append(wrap + get_node_text(node.childNodes) + wrap)
 170             continue
 171         if node.nodeType == node.TEXT_NODE:
 172             text.append(node.data)
 173         if node.nodeType == node.ELEMENT_NODE:
 174             text.append(get_node_text(node.childNodes))
 175     return "".join(text)

 178 def parse_params(params):
 179     """
 180     Parse parameter annotations.
 181 
 182     docutils doesn't give us much to work with, but
 183     we can get a DomDocument and traverse that for path parameters
 184     and query parameters, and layer on some text munging to get
 185     enough data to make the output we need.
 186     """
 187     parsed = defaultdict(list)
 188     param_tree = publish_doctree("\n".join(params)).asdom()
 189     field_names = param_tree.getElementsByTagName("field_name")
 190     field_values = param_tree.getElementsByTagName("field_body")
 191 
 192     for i, field in enumerate(field_names):
 193         name = get_node_text(field.childNodes)
 194         value = ""
 195         field_value = field_values.item(i)
 196         if field_value:
 197             value = get_node_text(field_value.childNodes)
 198         if name.startswith("pparam"):
 199             field_type = "path"
 200             name = name[7:]
 201         elif name.startswith("qparam"):
 202             field_type = "query"
 203             name = name[7:]
 204         elif name.startswith("auth"):
 205             field_type = "auth"
 206             name = ""
 207         else:
 208             field_type = "param"
 209             _, name = name.split(" ", 1)
 210 
 211         # Split out the parameter type
 212         param_type = ""
 213         if " " in name:
 214             param_type, name = name.split(" ", 1)
 215         parsed[field_type].append(dict(name=name, type=param_type, description=value))
 216     return parsed

 219 def camelcase_to_dashes(string):
 220     def handler(match):
 221         camel, regular = match.groups()
 222         if len(camel) != 1:
 223             camel = camel[:-1].lower() + "-" + camel[-1].lower()
 224         else:
 225             camel = camel.lower()
 226         return "-" + camel + regular.lower()
 227 
 228     return camel_re.sub(handler, string).lstrip("-")

 231 def extract_endpoint_info(pattern, internal_endpoint, parents):
 232     path = simplify_regex(pattern.regex.pattern)
 233     from sentry.constants import HTTP_METHODS
 234 
 235     for method_name in HTTP_METHODS:
 236         if method_name in ("HEAD", "OPTIONS"):
 237             continue
 238         method = getattr(internal_endpoint, method_name.lower(), None)
 239         if method is None:
 240             continue
 241         doc = extract_documentation(method)
 242         if doc is None:
 243             continue
 244         section = getattr(internal_endpoint, "doc_section", None)
 245         if section is None:
 246             continue
 247         endpoint_name = method.__name__.title() + internal_endpoint.__name__
 248         if endpoint_name.endswith("Endpoint"):
 249             endpoint_name = endpoint_name[:-8]
 250         endpoint_name = camelcase_to_dashes(endpoint_name)
 251         title, text, warning, params = parse_doc_string(doc)
 252         if not parents:
 253             api_path = API_PREFIX + path.lstrip("/")
 254         else:
 255             parents_prefix = "".join(map(lambda x: x.lstrip("/"), parents))
 256             api_path = API_PREFIX + parents_prefix.lstrip("/") + path.lstrip("/")
 257 
 258         yield dict(
 259             path=api_path,
 260             method=method_name,
 261             title=title,
 262             text=text,
 263             warning=warning,
 264             params=params,
 265             scenarios=getattr(method, "api_scenarios", None) or [],
 266             section=section.name.lower(),
 267             internal_path="%s:%s" % (get_endpoint_path(internal_endpoint), method.__name__),
 268             endpoint_name=endpoint_name,
 269         )

 276 def resolve_nested_routes(pattern, current_parents=None):
 277     """
 278     Returns a list of tuples. The first element in the tuple is a RegexURLPattern. The second element is
 279     a string list of all the parents that url pattern has.
 280     """
 281     if current_parents:
 282         parents = copy(current_parents)
 283     else:
 284         parents = []
 285 
 286     if isinstance(pattern, RegexURLPattern):
 287         return [(pattern, current_parents)]
 288     elif isinstance(pattern, RegexURLResolver):
 289         parent = simplify_regex(pattern.regex.pattern)
 290         parents.append(parent)
 291         return flatten(map(lambda p: resolve_nested_routes(p, parents), pattern.url_patterns))
 292     elif isinstance(pattern, list):
 293         return flatten(map(lambda p: resolve_nested_routes(p, current_parents), pattern))

 296 def iter_endpoints():
 297     from sentry.api.urls import urlpatterns
 298 
 299     resolved_patterns = resolve_nested_routes(urlpatterns)
 300 
 301     for pattern, parents in resolved_patterns:
 302         internal_endpoint = get_internal_endpoint_from_pattern(pattern)
 303         if internal_endpoint is None:
 304             continue
 305         for endpoint in extract_endpoint_info(pattern, internal_endpoint, parents):

 565     def request(self, method, path, headers=None, data=None, api_token=None, format="json"):
 566         if api_token is None:
 567             api_token = self.api_token
 568         path = "/api/0/" + path.lstrip("/")
 569         headers = dict(headers or {})
 570 
 571         request_is_json = True
 572         body = None
 573         files = None
 574         was_multipart = False
 575         if data is not None:
 576             if format == "json":
 577                 body = json.dumps(data, sort_keys=True)
 578                 headers["Content-Type"] = "application/json"
 579             elif format == "multipart":
 580                 files = {}
 581                 for key, value in data.items():
 582                     if hasattr(value, "read") or isinstance(value, tuple):
 583                         files[key] = value
 584                         del data[key]
 585                         was_multipart = True
 586                 body = data
 587 
 588         req_headers = dict(headers)
 589         req_headers["Host"] = "sentry.io"
 590         req_headers["Authorization"] = "Bearer %s" % api_token.token
 591 
 592         url = "http://127.0.0.1:%s%s" % (settings.SENTRY_APIDOCS_WEB_PORT, path)
 593 
 594         response = requests.request(
 595             method=method, url=url, files=files, headers=req_headers, data=body
 596         )
 597         response_headers = dict(response.headers)
 598 
 599         # Don't want those
 600         response_headers.pop("server", None)
 601         response_headers.pop("date", None)
 602 
 603         if response.headers.get("Content-Type") == "application/json":
 604             response_data = response.json()
 605             is_json = True
 606         else:
 607             response_data = response.text
 608             is_json = False
 609 
 610         if was_multipart:
 611             headers["Content-Type"] = response.request.headers["content-type"]
 612             data = response.request.body
 613             request_is_json = False
 614 
 615         rv = {
 616             "request": {
 617                 "method": method,
 618                 "path": path,
 619                 "headers": headers,
 620                 "data": data,
 621                 "is_json": request_is_json,
 622             },
 623             "response": {
 624                 "headers": response_headers,
 625                 "status": response.status_code,
 626                 "reason": response.reason,
 627                 "data": response_data,
 628                 "is_json": is_json,
 629             },
 630         }
 631 
 632         self.requests.append(rv)
 633         return rv
