  38 class TimedFuture(Future):
  39     def __init__(self, *args, **kwargs):
  40         self.__timing = [None, None]  # [started, finished/cancelled]
  41         super(TimedFuture, self).__init__(*args, **kwargs)
  42 
  43     def get_timing(self):
  44         """\
  45         Return the timing data for this future in the form ``(started, finished)``.
  46 
  47         The ``started`` value can be either a timestamp or ``None`` (if the
  48         future has not been started.) The ``finished`` value can also be either
  49         a timestamp or ``None`` (if the future has not been either completed or
  50         cancelled.)
  51 
  52         There are some idiosyncracies with the way the timings are recorded:
  53 
  54         - The ``started`` value will generally not be ``None`` if the
  55           ``finished`` value is also not ``None``. However, for a future that
  56           was marked as cancelled and has yet to be attempted to be executed,
  57           the ``finished`` value may be set while the ``started`` value is
  58           ``None``.
  59         - Similarly, the ``started`` value will generally be equal to or less
  60           than the ``finished`` value (ignoring non-monotonic clock phenomena.)
  61           However, for a future was is marked as cancelled prior to execution,
  62           the ``finished`` time (when the future was cancelled) may be before
  63           the ``started`` time (when the future was attempted to be executed.)
  64         """
  65         return tuple(self.__timing)
  66 
  67     def set_running_or_notify_cancel(self, *args, **kwargs):
  68         result = super(TimedFuture, self).set_running_or_notify_cancel(*args, **kwargs)
  69         # This method can only be called once (the second invocation will raise
  70         # a ``RuntimeError``) so if we've gotten this far we can be reasonably
  71         # confident that the start time hasn't been set.
  72         self.__timing[0] = time()
  73         return result
  74 
  75     def cancel(self, *args, **kwargs):
  76         with self._condition:
  77             # Futures can only be marked as cancelled if they are neither
  78             # running or finished (we have to duplicate this check that is also
  79             # performed in the superclass to ensure the timing is set before
  80             # callbacks are invoked.) As long as the future is in the correct
  81             # state, this call is guaranteed to succeed. This method can be
  82             # called multiple times, but we only record the first time the
  83             # future was cancelled.
  84             if self._state not in [RUNNING, FINISHED] and self.__timing[1] is None:
  85                 self.__timing[1] = time()
  86             return super(TimedFuture, self).cancel(*args, **kwargs)
  87 
  88     def set_result(self, *args, **kwargs):
  89         with self._condition:
  90             # This method always overwrites the result, so we always overwrite
  91             # the timing, even if another timing was already recorded.
  92             self.__timing[1] = time()
  93             return super(TimedFuture, self).set_result(*args, **kwargs)
  94 
  95     def set_exception_info(self, *args, **kwargs):
  96         # XXX: This makes the potentially unsafe assumption that
  97         # ``set_exception`` will always continue to call this function.
  98         with self._condition:
  99             # This method always overwrites the result, so we always overwrite
 100             # the timing, even if another timing was already recorded.
 101             self.__timing[1] = time()
 102             return super(TimedFuture, self).set_exception_info(*args, **kwargs)

 105 class Executor(object):
 106     """
 107     This class provides an API for executing tasks in different contexts
 108     (immediately, or asynchronously.)
 109 
 110     NOTE: This is *not* compatible with the ``concurrent.futures.Executor``
 111     API! Rather than ``submit`` accepting the function arguments, the function
 112     must already have the argument values bound (via ``functools.partial`` or
 113     similar), and ``submit`` passes all additional arguments to ``queue.put``
 114     to allow controlling whether or not queue insertion should be blocking.
 115     """
 116 
 117     Future = TimedFuture
 118 
 119     def submit(self, callable, priority=0, block=True, timeout=None):
 120         """
 121         Enqueue a task to be executed, returning a ``TimedFuture``.
 122 
 123         All implementations *must* accept the ``callable`` parameter, but other
 124         parameters may or may not be implemented, depending on the specific
 125         implementation used.
 126         """
 127         raise NotImplementedError

 130 class SynchronousExecutor(Executor):
 131     """
 132     This executor synchronously executes callables in the current thread.
 133 
 134     This is primarily exists to provide API compatibility with
 135     ``ThreadedExecutor`` for calls that do not do significant I/O.
 136     """
 137 
 138     # TODO: The ``Future`` implementation here could be replaced with a
 139     # lock-free future for efficiency.
 140 
 141     def submit(self, callable, *args, **kwargs):
 142         """
 143         Immediately execute a callable, returning a ``TimedFuture``.
 144         """
 145         future = self.Future()
 146         assert future.set_running_or_notify_cancel()
 147         try:
 148             result = callable()
 149         except Exception:
 150             future.set_exception_info(*sys.exc_info()[1:])
 151         else:
 152             future.set_result(result)
 153         return future

 156 class ThreadedExecutor(Executor):
 157     """\
 158     This executor provides a method of executing callables in a threaded worker
 159     pool. The number of outstanding requests can be limited by the ``maxsize``
 160     parameter, which has the same behavior as the parameter of the same name
 161     for the ``PriorityQueue`` constructor.
 162 
 163     All threads are daemon threads and will remain alive until the main thread
 164     exits. Any items remaining in the queue at this point may not be executed!
 165     """
 166 
 167     def __init__(self, worker_count=1, maxsize=0):
 168         self.__worker_count = worker_count
 169         self.__workers = set([])
 170         self.__started = False
 171         self.__queue = PriorityQueue(maxsize)
 172         self.__lock = threading.Lock()
 173 
 174     def __worker(self):
 175         queue = self.__queue
 176         while True:
 177             priority, (function, future) = queue.get(True)
 178             if not future.set_running_or_notify_cancel():
 179                 continue
 180             try:
 181                 result = function()
 182             except Exception:
 183                 future.set_exception_info(*sys.exc_info()[1:])
 184             else:
 185                 future.set_result(result)
 186             queue.task_done()
 187 
 188     def start(self):
 189         with self.__lock:
 190             if self.__started:
 191                 return
 192 
 193             for i in xrange(self.__worker_count):
 194                 t = threading.Thread(target=self.__worker)
 195                 t.daemon = True
 196                 t.start()
 197                 self.__workers.add(t)
 198 
 199             self.__started = True
 200 
 201     def submit(self, callable, priority=0, block=True, timeout=None):
 202         """\
 203         Enqueue a task to be executed, returning a ``TimedFuture``.
 204 
 205         Tasks can be prioritized by providing a value for the ``priority``
 206         argument, which follows the same specification as the standard library
 207         ``Queue.PriorityQueue`` (lowest valued entries are retrieved first.)
 208 
 209         If the worker pool has not already been started, calling this method
 210         will cause all of the worker threads to start running.
 211         """
 212         if not self.__started:
 213             self.start()
 214 
 215         future = self.Future()
 216         task = (priority, (callable, future))
 217         try:
 218             self.__queue.put(task, block=block, timeout=timeout)
 219         except Full as error:
 220             if future.set_running_or_notify_cancel():
 221                 future.set_exception(error)
 222         return future

 225 class FutureSet(object):
 226     """\
 227     Coordinates a set of ``Future`` objects (either from
 228     ``concurrent.futures``, or otherwise API compatible), and allows for
 229     attaching a callback when all futures have completed execution.
 230     """
 231 
 232     def __init__(self, futures):
 233         self.__pending = set(futures)
 234         self.__completed = set()
 235         self.__callbacks = []
 236         self.__lock = threading.Lock()
 237 
 238         for future in futures:
 239             future.add_done_callback(self.__mark_completed)
 240 
 241     def __iter__(self):
 242         with self.__lock:
 243             futures = self.__pending | self.__completed
 244         return iter(futures)
 245 
 246     def __execute_callback(self, callback):
 247         try:
 248             callback(self)
 249         except Exception as error:
 250             logger.warning("Error when calling callback %r: %s", callback, error, exc_info=True)
 251 
 252     def __mark_completed(self, future):
 253         with self.__lock:
 254             self.__pending.remove(future)
 255             self.__completed.add(future)
 256             remaining = len(self.__pending)
 257 
 258         if remaining == 0:
 259             for callback in self.__callbacks:
 260                 self.__execute_callback(callback)
 261 
 262     def add_done_callback(self, callback):
 263         with self.__lock:
 264             remaining = len(self.__pending)
 265             if remaining > 0:
 266                 self.__callbacks.append(callback)
 267 
 268         if remaining == 0:
 269             self.__execute_callback(callback)
