  15 class State(threading.local):
  16     def __init__(self):
  17         self.count = 0
  18         self.query_hashes = defaultdict(int)
  19 
  20     def record_query(self, sql):
  21         self.count += 1
  22         self.query_hashes[hash(sql)] += 1
  23 
  24     def count_dupes(self):
  25         return sum(1 for n in six.itervalues(self.query_hashes) if n > 1)

  28 class CursorWrapper(object):
  29     def __init__(self, cursor, connection, state):
  30         self.cursor = cursor
  31         self.connection = connection
  32         self._state = state
  33 
  34     def execute(self, sql, params=()):
  35         try:
  36             return self.cursor.execute(sql, params)
  37         finally:
  38             self._state.record_query(sql)
  39 
  40     def executemany(self, sql, paramlist):
  41         try:
  42             return self.cursor.executemany(sql, paramlist)
  43         finally:
  44             self._state.record_query(sql)
  45 
  46     def __getattr__(self, attr):
  47         if attr in self.__dict__:
  48             return self.__dict__[attr]
  49         else:
  50             return getattr(self.cursor, attr)
  51 
  52     def __iter__(self):
  53         return iter(self.cursor)

  59

  97         num_dupes = self.state.count_dupes()

  99         if self.state.count > self.max_queries:

 106         state = self.state
 107 
 108         context = {"stack": True, "data": {"query_count": state.count, "num_dupes": num_dupes}}

 110         self.logger.warning(
 111             "%d duplicate queries executed in %s", num_dupes, self.context, extra=context
