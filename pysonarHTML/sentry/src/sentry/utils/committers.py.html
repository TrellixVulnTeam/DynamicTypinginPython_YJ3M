  23 def tokenize_path(path):
  24     for sep in PATH_SEPERATORS:
  25         if sep in path:
  26             # Exclude empty path segments as some repository integrations
  27             # start their paths with `/` which we want to ignore.
  28             return reversed([x for x in path.split(sep) if x != ""])
  29     else:
  30         return iter([path])

  33 def score_path_match_length(path_a, path_b):
  34     score = 0
  35     for a, b in zip(tokenize_path(path_a), tokenize_path(path_b)):
  36         if a.lower() != b.lower():
  37             break
  38         score += 1
  39     return score

  42 def _get_frame_paths(event):
  43     data = event.data
  44     frames = get_path(data, "stacktrace", "frames", filter=True)
  45     if frames:
  46         return frames
  47 
  48     return get_path(data, "exception", "values", 0, "stacktrace", "frames", filter=True) or []

  59 def _get_commit_file_changes(commits, path_name_set):
  60     # Get distinct file names and bail if there are no files.
  61     filenames = {next(tokenize_path(path), None) for path in path_name_set}
  62     filenames = {path for path in filenames if path is not None}
  63     if not len(filenames):
  64         return []
  65 
  66     # build a single query to get all of the commit file that might match the first n frames
  67     path_query = reduce(operator.or_, (Q(filename__iendswith=path) for path in filenames))
  68 
  69     commit_file_change_matches = CommitFileChange.objects.filter(path_query, commit__in=commits)
  70 
  71     return list(commit_file_change_matches)

  74 def _match_commits_path(commit_file_changes, path):
  75     # find commits that match the run time path the best.
  76     matching_commits = {}
  77     best_score = 1
  78     for file_change in commit_file_changes:
  79         score = score_path_match_length(file_change.filename, path)
  80         if score > best_score:
  81             # reset matches for better match.
  82             best_score = score
  83             matching_commits = {}
  84         if score == best_score:
  85             # skip 1-score matches when file change is longer than 1 token
  86             if score == 1 and len(list(tokenize_path(file_change.filename))) > 1:
  87                 continue
  88             #  we want a list of unique commits that tie for longest match
  89             matching_commits[file_change.commit.id] = (file_change.commit, score)
  90 
  91     return matching_commits.values()

  94 def _get_commits_committer(commits, author_id):
  95     result = serialize(
  96         [commit for commit, score in commits if commit.author.id == author_id],
  97         serializer=CommitSerializer(exclude=["author"]),
  98     )
  99     for idx, row in enumerate(result):
 100         row["score"] = commits[idx][1]
 101     return result

 104 def _get_committers(annotated_frames, commits):
 105     # extract the unique committers and return their serialized sentry accounts
 106     committers = defaultdict(int)
 107 
 108     limit = 5
 109     for annotated_frame in annotated_frames:
 110         if limit == 0:
 111             break
 112         for commit, score in annotated_frame["commits"]:
 113             committers[commit.author.id] += limit
 114             limit -= 1
 115             if limit == 0:
 116                 break
 117 
 118     # organize them by this heuristic (first frame is worth 5 points, second is worth 4, etc.)
 119     sorted_committers = sorted(committers, key=committers.get)
 120     users_by_author = get_users_for_commits([c for c, _ in commits])
 121 
 122     user_dicts = [
 123         {
 124             "author": users_by_author.get(six.text_type(author_id)),
 125             "commits": [
 126                 (commit, score) for (commit, score) in commits if commit.author.id == author_id
 127             ],
 128         }
 129         for author_id in sorted_committers
 130     ]
 131 
 132     return user_dicts

 135 def get_previous_releases(project, start_version, limit=5):
 136     # given a release version + project, return the previous
 137     # `limit` releases (includes the release specified by `version`)
 138     key = "get_previous_releases:1:%s" % hash_values([project.id, start_version, limit])
 139     rv = cache.get(key)
 140     if rv is None:
 141         try:
 142             release_dates = (
 143                 Release.objects.filter(
 144                     organization_id=project.organization_id, version=start_version, projects=project
 145                 )
 146                 .values("date_released", "date_added")
 147                 .get()
 148             )
 149         except Release.DoesNotExist:
 150             rv = []
 151         else:
 152             start_date = release_dates["date_released"] or release_dates["date_added"]
 153 
 154             rv = list(
 155                 Release.objects.filter(projects=project, organization_id=project.organization_id)
 156                 .extra(
 157                     select={"date": "COALESCE(date_released, date_added)"},
 158                     where=["COALESCE(date_released, date_added) <= %s"],
 159                     params=[start_date],
 160                 )
 161                 .extra(order_by=["-date"])[:limit]
 162             )
 163         cache.set(key, rv, 60)
 164     return rv

 167 def get_event_file_committers(project, event, frame_limit=25):
 168     group = Group.objects.get(id=event.group_id)
 169 
 170     first_release_version = group.get_first_release()
 171 
 172     if not first_release_version:
 173         raise Release.DoesNotExist
 174 
 175     releases = get_previous_releases(project, first_release_version)
 176     if not releases:
 177         raise Release.DoesNotExist
 178 
 179     commits = _get_commits(releases)
 180 
 181     if not commits:
 182         raise Commit.DoesNotExist
 183 
 184     frames = _get_frame_paths(event) or ()
 185     app_frames = [frame for frame in frames if frame["in_app"]][-frame_limit:]
 186     if not app_frames:
 187         app_frames = [frame for frame in frames][-frame_limit:]
 188 
 189     # Java stackframes don't have an absolute path in the filename key.
 190     # That property is usually just the basename of the file. In the future
 191     # the Java SDK might generate better file paths, but for now we use the module
 192     # path to approximate the file path so that we can intersect it with commit
 193     # file paths.
 194     if event.platform == "java":
 195         for frame in frames:
 196             if frame.get("filename") is None:
 197                 continue
 198             if "/" not in frame.get("filename") and frame.get("module"):
 199                 # Replace the last module segment with the filename, as the
 200                 # terminal element in a module path is the class
 201                 module = frame["module"].split(".")
 202                 module[-1] = frame["filename"]
 203                 frame["filename"] = "/".join(module)
 204 
 205     # TODO(maxbittker) return this set instead of annotated frames
 206     # XXX(dcramer): frames may not define a filepath. For example, in Java its common
 207     # to only have a module name/path
 208     path_set = {
 209         f for f in (frame.get("filename") or frame.get("abs_path") for frame in app_frames) if f
 210     }
 211 
 212     file_changes = []
 213     if path_set:
 214         file_changes = _get_commit_file_changes(commits, path_set)
 215 
 216     commit_path_matches = {path: _match_commits_path(file_changes, path) for path in path_set}
 217 
 218     annotated_frames = [
 219         {
 220             "frame": frame,
 221             "commits": commit_path_matches.get(frame.get("filename") or frame.get("abs_path"))
 222             or [],
 223         }
 224         for frame in app_frames
 225     ]
 226 
 227     relevant_commits = list(
 228         {match for match in commit_path_matches for match in commit_path_matches[match]}
 229     )
 230 
 231     return _get_committers(annotated_frames, relevant_commits)

 234 def get_serialized_event_file_committers(project, event, frame_limit=25):
 235     committers = get_event_file_committers(project, event, frame_limit=frame_limit)
 236     commits = [commit for committer in committers for commit in committer["commits"]]
 237     serialized_commits = serialize(
 238         [c for (c, score) in commits], serializer=CommitSerializer(exclude=["author"])
 239     )
 240 
 241     serialized_commits_by_id = {}
 242 
 243     for (commit, score), serialized_commit in zip(commits, serialized_commits):
 244         serialized_commit["score"] = score
 245         serialized_commits_by_id[commit.id] = serialized_commit
 246 
 247     for committer in committers:
 248         commit_ids = [commit.id for (commit, _) in committer["commits"]]
 249         commits_result = [serialized_commits_by_id[commit_id] for commit_id in commit_ids]
 250         committer["commits"] = dedupe_commits(commits_result)
 251 
 252     metrics.incr(
 253         "feature.owners.has-committers",
 254         instance="hit" if committers else "miss",
 255         skip_internal=False,
 256     )
 257     return committers

 260 def dedupe_commits(commits):
 261     return {c["id"]: c for c in commits}.values()
