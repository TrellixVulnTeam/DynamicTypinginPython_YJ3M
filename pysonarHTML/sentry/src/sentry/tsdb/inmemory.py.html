  67     def get_range(self, model, keys, start, end, rollup=None, environment_ids=None):
  68         self.validate_arguments([model], environment_ids if environment_ids is not None else [None])
  69 
  70         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
  71 
  72         results = []
  73         for timestamp in map(to_datetime, series):
  74             norm_epoch = self.normalize_to_rollup(timestamp, rollup)
  75 
  76             for key in keys:
  77                 if not environment_ids:
  78                     value = self.data[model][(key, None)][norm_epoch]
  79                 else:
  80                     value = sum(
  81                         int(self.data[model][(key, environment_id)][norm_epoch])
  82                         for environment_id in environment_ids
  83                     )
  84                 results.append((to_timestamp(timestamp), key, value))
  85 
  86         results_by_key = defaultdict(dict)
  87         for epoch, key, count in results:
  88             results_by_key[key][epoch] = int(count or 0)
  89 
  90         for key, points in six.iteritems(results_by_key):
  91             results_by_key[key] = sorted(points.items())
  92         return dict(results_by_key)

 202     def record_frequency_multi(self, requests, timestamp=None, environment_id=None):
 203         environment_ids = set([environment_id, None])
 204 
 205         self.validate_arguments([model for model, request in requests], [environment_id])
 206 
 207         if timestamp is None:
 208             timestamp = timezone.now()
 209 
 210         for model, request in requests:
 211             for key, items in request.items():
 212                 items = {k: float(v) for k, v in items.items()}
 213                 for environment_id in environment_ids:
 214                     source = self.frequencies[model][(key, environment_id)]
 215                     for rollup in self.rollups:
 216                         source[self.normalize_to_rollup(timestamp, rollup)].update(items)

 218     def get_most_frequent(
 219         self, model, keys, start, end=None, rollup=None, limit=None, environment_id=None
 220     ):
 221         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
 222

 224 
 225         results = {}
 226         for key in keys:
 227             result = results[key] = Counter()
 228             source = self.frequencies[model][(key, environment_id)]
