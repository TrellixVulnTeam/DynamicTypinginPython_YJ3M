 188     def normalize_to_epoch(self, timestamp, seconds):
 189         """
 190         Given a ``timestamp`` (datetime object) normalize to an epoch timestamp.
 191 
 192         i.e. if the rollup is minutes, the resulting timestamp would have
 193         the seconds and microseconds rounded down.
 194         """
 195         epoch = int(to_timestamp(timestamp))
 196         return epoch - (epoch % seconds)
 197 
 198     def normalize_ts_to_epoch(self, epoch, seconds):
 199         """
 200         Given a ``epoch`` normalize to an epoch rollup.
 201         """
 202         return epoch - (epoch % seconds)

 204     def normalize_to_rollup(self, timestamp, seconds):
 205         """
 206         Given a ``timestamp`` (datetime object) normalize to an epoch rollup.
 207         """
 208         epoch = int(to_timestamp(timestamp))
 209         return int(epoch / seconds)

 211     def normalize_ts_to_rollup(self, epoch, seconds):
 212         """
 213         Given a ``epoch`` normalize to an epoch rollup.
 214         """
 215         return int(epoch / seconds)

 217     def get_optimal_rollup(self, start_timestamp, end_timestamp):
 218         """
 219         Identify the lowest granularity rollup available within the given time
 220         range.
 221         """
 222         num_seconds = int(to_timestamp(end_timestamp)) - int(to_timestamp(start_timestamp))
 223 
 224         # This loop attempts to find the smallest possible rollup that will
 225         # contain both the start and end timestamps. ``self.rollups`` is
 226         # ordered from the highest resolution (smallest interval) to lowest
 227         # resolution (largest interval.)
 228         # XXX: There is a bug here, since this function assumes that the end
 229         # timestamp is always equal to or greater than the current time. If the
 230         # time range is shifted far enough into the past (e.g. a 30 second
 231         # window, retrieved several days after it's occurrence), this can
 232         # return a rollup that has already been evicted due to TTL, even if a
 233         # lower resolution representation of the range exists.
 234         for rollup, samples in six.iteritems(self.rollups):
 235             if rollup * samples >= num_seconds:
 236                 return rollup
 237 
 238         # If nothing actually matches the requested range, just return the
 239         # lowest resolution interval.
 240         return list(self.rollups)[-1]
 241 
 242     def get_optimal_rollup_series(self, start, end=None, rollup=None):
 243         if end is None:
 244             end = timezone.now()
 245 
 246         if rollup is None:
 247             rollup = self.get_optimal_rollup(start, end)
 248 
 249         # This attempts to create a range with a duration as close as possible
 250         # to the requested interval using the requested (or inferred) rollup
 251         # resolution. This result always includes the ``end`` timestamp, but
 252         # may not include the ``start`` timestamp.
 253         series = []
 254         timestamp = end
 255         while timestamp >= start:
 256             series.append(self.normalize_to_epoch(timestamp, rollup))
 257             timestamp = timestamp - timedelta(seconds=rollup)
 258 
 259         return rollup, sorted(series)
 260 
 261     def get_active_series(self, start=None, end=None, timestamp=None):
 262         rollups = {}
 263         for rollup, samples in self.rollups.items():
 264             _, series = self.get_optimal_rollup_series(
 265                 start
 266                 if start is not None
 267                 else to_datetime(self.get_earliest_timestamp(rollup, timestamp=timestamp)),
 268                 end,
 269                 rollup=rollup,
 270             )
 271             rollups[rollup] = map(to_datetime, series)
 272         return rollups

 274     def make_series(self, default, start, end=None, rollup=None):
 275         f = default if isinstance(default, collections.Callable) else lambda timestamp: default
 276         return [
 277             (timestamp, f(timestamp))
 278             for timestamp in self.get_optimal_rollup_series(start, end, rollup)[1]
 279         ]

 281     def calculate_expiry(self, rollup, samples, timestamp):
 282         """
 283         Calculate the expiration time for a rollup.
 284 
 285         :param rollup: rollup interval (in seconds)
 286         :param samples: number of samples to maintain
 287         :param timestamp: datetime used to calculate the rollup epoch
 288         """
 289         epoch = self.normalize_to_epoch(timestamp, rollup)
 290         return epoch + (rollup * samples)
 291 
 292     def get_earliest_timestamp(self, rollup, timestamp=None):
 293         """
 294         Calculate the earliest available timestamp for a rollup.
 295         """
 296         if timestamp is None:
 297             timestamp = timezone.now()
 298 
 299         samples = self.__legacy_rollups.get(rollup)
 300         if samples is None:
 301             samples = self.rollups[rollup]
 302 
 303         lifespan = timedelta(seconds=rollup * (samples - 1))
 304         return self.normalize_to_epoch(timestamp - lifespan, rollup)

 365     def get_sums(self, model, keys, start, end, rollup=None, environment_id=None):
 366         range_set = self.get_range(
 367             model,
 368             keys,
 369             start,
 370             end,
 371             rollup,
 372             environment_ids=[environment_id] if environment_id is not None else None,
 373         )
 374         sum_set = dict(
 375             (key, sum(p for _, p in points)) for (key, points) in six.iteritems(range_set)
 376         )
 377         return sum_set

 379     def rollup(self, values, rollup):
 380         """
 381         Given a set of values (as returned from ``get_range``), roll them up
 382         using the ``rollup`` time (in seconds).
 383         """
 384         normalize_ts_to_epoch = self.normalize_ts_to_epoch
 385         result = {}
 386         for key, points in six.iteritems(values):
 387             result[key] = []
 388             last_new_ts = None
 389             for (ts, count) in points:
 390                 new_ts = normalize_ts_to_epoch(ts, rollup)
 391                 if new_ts == last_new_ts:
 392                     result[key][-1][1] += count
 393                 else:
 394                     result[key].append([new_ts, count])
 395                     last_new_ts = new_ts
 396         return result
