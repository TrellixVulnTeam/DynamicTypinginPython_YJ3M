 205     def get_data(
 206         self,
 207         model,
 208         keys,
 209         start,
 210         end,
 211         rollup=None,
 212         environment_ids=None,
 213         aggregation="count()",
 214         group_on_model=True,
 215         group_on_time=False,
 216     ):
 217         """
 218         Normalizes all the TSDB parameters and sends a query to snuba.
 219 
 220         `group_on_time`: whether to add a GROUP BY clause on the 'time' field.
 221         `group_on_model`: whether to add a GROUP BY clause on the primary model.
 222         """
 223         # XXX: to counteract the hack in project_key_stats.py
 224         if model in [
 225             TSDBModel.key_total_received,
 226             TSDBModel.key_total_blacklisted,
 227             TSDBModel.key_total_rejected,
 228         ]:
 229             keys = list(set(map(lambda x: int(x), keys)))
 230 
 231         # 10s is the only rollup under an hour that we support
 232         if rollup and rollup == 10 and model in self.lower_rollup_query_settings.keys():
 233             model_query_settings = self.lower_rollup_query_settings.get(model)
 234         else:
 235             model_query_settings = self.model_query_settings.get(model)
 236 
 237         if model_query_settings is None:
 238             raise Exception(u"Unsupported TSDBModel: {}".format(model.name))
 239 
 240         model_group = model_query_settings.groupby
 241         model_aggregate = model_query_settings.aggregate
 242 
 243         groupby = []
 244         if group_on_model and model_group is not None:
 245             groupby.append(model_group)
 246         if group_on_time:
 247             groupby.append("time")
 248         if aggregation == "count()" and model_aggregate is not None:
 249             # Special case, because count has different semantics, we change:
 250             # `COUNT(model_aggregate)` to `COUNT() GROUP BY model_aggregate`
 251             groupby.append(model_aggregate)
 252             model_aggregate = None
 253 
 254         columns = (model_query_settings.groupby, model_query_settings.aggregate)
 255         keys_map = dict(zip(columns, self.flatten_keys(keys)))
 256         keys_map = {k: v for k, v in six.iteritems(keys_map) if k is not None and v is not None}
 257         if environment_ids is not None:
 258             keys_map["environment"] = environment_ids
 259 
 260         aggregations = [[aggregation, model_aggregate, "aggregate"]]
 261 
 262         # For historical compatibility with bucket-counted TSDB implementations
 263         # we grab the original bucketed series and add the rollup time to the
 264         # timestamp of the last bucket to get the end time.
 265         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
 266         start = to_datetime(series[0])
 267         end = to_datetime(series[-1] + rollup)
 268         limit = min(10000, int(len(keys) * ((end - start).total_seconds() / rollup)))
 269 
 270         if keys:
 271             result = snuba.query(
 272                 dataset=model_query_settings.dataset,
 273                 start=start,
 274                 end=end,
 275                 groupby=groupby,
 276                 conditions=deepcopy(
 277                     model_query_settings.conditions
 278                 ),  # copy because we modify the conditions in snuba.query
 279                 filter_keys=keys_map,
 280                 aggregations=aggregations,
 281                 rollup=rollup,
 282                 limit=limit,
 283                 referrer="tsdb",
 284                 is_grouprelease=(model == TSDBModel.frequent_releases_by_group),
 285             )
 286         else:
 287             result = {}
 288 
 289         if group_on_time:
 290             keys_map["time"] = series
 291 
 292         self.zerofill(result, groupby, keys_map)
 293         self.trim(result, groupby, keys)
 294 
 295         return result

 297     def zerofill(self, result, groups, flat_keys):
 298         """
 299         Fills in missing keys in the nested result with zeroes.
 300         `result` is the nested result
 301         `groups` is the order in which the result is nested, eg: ['project', 'time']
 302         `flat_keys` is a map from groups to lists of required keys for that group.
 303                     eg: {'project': [1,2]}
 304         """
 305         if len(groups) > 0:
 306             group, subgroups = groups[0], groups[1:]
 307             # Zerofill missing keys
 308             for k in flat_keys[group]:
 309                 if k not in result:
 310                     result[k] = 0 if len(groups) == 1 else {}
 311 
 312             if subgroups:
 313                 for v in result.values():
 314                     self.zerofill(v, subgroups, flat_keys)
 315 
 316     def trim(self, result, groups, keys):
 317         """
 318         Similar to zerofill, but removes keys that should not exist.
 319         Uses the non-flattened version of keys, so that different sets
 320         of keys can exist in different branches at the same nesting level.
 321         """
 322         if len(groups) > 0:
 323             group, subgroups = groups[0], groups[1:]
 324             if isinstance(result, dict):
 325                 for rk in result.keys():
 326                     if group == "time":  # Skip over time group
 327                         self.trim(result[rk], subgroups, keys)
 328                     elif rk in keys:
 329                         if isinstance(keys, dict):
 330                             self.trim(result[rk], subgroups, keys[rk])
 331                     else:
 332                         del result[rk]
 333 
 334     def get_range(self, model, keys, start, end, rollup=None, environment_ids=None):
 335         # 10s is the only rollup under an hour that we support
 336         if rollup and rollup == 10 and model in self.lower_rollup_query_settings.keys():
 337             model_query_settings = self.lower_rollup_query_settings.get(model)
 338         else:
 339             model_query_settings = self.model_query_settings.get(model)
 340 
 341         assert model_query_settings is not None, u"Unsupported TSDBModel: {}".format(model.name)
 342 
 343         if model_query_settings.dataset == snuba.Dataset.Outcomes:
 344             aggregate_function = "sum"
 345         else:
 346             aggregate_function = "count()"
 347 
 348         result = self.get_data(
 349             model,
 350             keys,
 351             start,
 352             end,
 353             rollup,
 354             environment_ids,
 355             aggregation=aggregate_function,
 356             group_on_time=True,
 357         )
 358         # convert
 359         #    {group:{timestamp:count, ...}}
 360         # into
 361         #    {group: [(timestamp, count), ...]}
 362         return {k: sorted(result[k].items()) for k in result}

 410     def get_most_frequent(
 411         self, model, keys, start, end=None, rollup=None, limit=10, environment_id=None
 412     ):
 413         aggregation = u"topK({})".format(limit)
 414         result = self.get_data(
 415             model,
 416             keys,
 417             start,
 418             end,
 419             rollup,
 420             [environment_id] if environment_id is not None else None,
 421             aggregation=aggregation,
 422         )
 423         # convert
 424         #    {group:[top1, ...]}
 425         # into
 426         #    {group: [(top1, score), ...]}
 427         for k, top in six.iteritems(result):
 428             item_scores = [(v, float(i + 1)) for i, v in enumerate(reversed(top or []))]
 429             result[k] = list(reversed(item_scores))
 430 
 431         return result

 433     def get_most_frequent_series(
 434         self, model, keys, start, end=None, rollup=None, limit=10, environment_id=None
 435     ):
 436         aggregation = u"topK({})".format(limit)
 437         result = self.get_data(
 438             model,
 439             keys,
 440             start,
 441             end,
 442             rollup,
 443             [environment_id] if environment_id is not None else None,
 444             aggregation=aggregation,
 445             group_on_time=True,
 446         )
 447         # convert
 448         #    {group:{timestamp:[top1, ...]}}
 449         # into
 450         #    {group: [(timestamp, {top1: score, ...}), ...]}
 451         for k in result:
 452             result[k] = sorted(
 453                 [
 454                     (timestamp, {v: float(i + 1) for i, v in enumerate(reversed(topk or []))})
 455                     for (timestamp, topk) in result[k].items()
 456                 ]
 457             )
 458 
 459         return result

 489     def flatten_keys(self, items):
 490         """
 491         Returns a normalized set of keys based on the various formats accepted
 492         by TSDB methods. The input is either just a plain list of keys for the
 493         top level or a `{level1_key: [level2_key, ...]}` dictionary->list map.
 494         The output is a 2-tuple of ([level_1_keys], [all_level_2_keys])
 495         """
 496         if isinstance(items, collections.Mapping):
 497             return (
 498                 items.keys(),
 499                 list(set.union(*(set(v) for v in items.values())) if items else []),
 500             )
 501         elif isinstance(items, (collections.Sequence, collections.Set)):
 502             return (items, None)
 503         else:
 504             raise ValueError("Unsupported type: %s" % (type(items)))
