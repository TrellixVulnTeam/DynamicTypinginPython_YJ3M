  70 def selector_func(method, callargs, switchover_timestamp=None):
  71     spec = method_specifications.get(method)
  72     if spec is None:
  73         return "redis"  # default backend (possibly invoke base directly instead?)
  74 
  75     if switchover_timestamp is not None and time.time() < switchover_timestamp:
  76         return "redis"  # snuba does not yet have all data
  77 
  78     operation_type, model_extractor = spec
  79     backends = {model_backends[model][operation_type] for model in model_extractor(callargs)}
  80 
  81     assert len(backends) == 1, "request was not directed to a single backend"
  82     return backends.pop()

  85 def make_method(key):
  86     def method(self, *a, **kw):
  87         callargs = inspect.getcallargs(getattr(BaseTSDB, key), self, *a, **kw)
  88         backend = selector_func(key, callargs, self.switchover_timestamp)
  89         return getattr(self.backends[backend], key)(*a, **kw)
  90 
  91     return method

  94 # We have to apply these methods into RedisSnubaTSDB through
  95 # a metaclass since we can't simply overload `__getattr__` due to
  96 # the fact that the subclass BaseTSDB already defines all the methods.
  97 # So we need to actually apply methods on top to override them.
  98 class RedisSnubaTSDBMeta(type):
  99     def __new__(cls, name, bases, attrs):
 100         for key in method_specifications.keys():
 101             attrs[key] = make_method(key)
 102         return type.__new__(cls, name, bases, attrs)

 105 @six.add_metaclass(RedisSnubaTSDBMeta)
 106 class RedisSnubaTSDB(BaseTSDB):
 107     def __init__(self, switchover_timestamp=None, **options):
 108         """
 109         A TSDB backend that uses the Snuba outcomes dataset for certain models
 110         instead of reading/writing to redis. Reading will trigger a Snuba
 111         query, while writing is a noop as Snuba reads from outcomes.
 112 
 113         Note: Using this backend requires you to start Snuba outcomes consumers
 114         (not to be confused with the outcomes consumers in Sentry itself).
 115 
 116         :param switchover_timestamp: When set, only start reading from snuba
 117             after this timestamp (as returned by `time.time()`). When this
 118             timestamp has not been reached yet, this backend just degrades to
 119             Redis for *all* keys.
 120 
 121             The default `None` will start reading from Snuba immediately and is
 122             equivalent to setting a past timestamp.
 123         """
 124         self.switchover_timestamp = switchover_timestamp
 125         self.backends = {
 126             "dummy": DummyTSDB(),
 127             "redis": RedisTSDB(**options.pop("redis", {})),
 128             "snuba": SnubaTSDB(**options.pop("snuba", {})),
 129         }
 130         super(RedisSnubaTSDB, self).__init__(**options)
