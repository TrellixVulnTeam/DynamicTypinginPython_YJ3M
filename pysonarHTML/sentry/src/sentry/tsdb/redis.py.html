  32 class SuppressionWrapper(object):
  33     """\
  34     Wraps a context manager and prevents any exceptions raised either during
  35     the managed block or the exiting of the wrapped manager from propagating.
  36 
  37     You probably shouldn't use this.
  38     """
  39 
  40     def __init__(self, wrapped):
  41         self.wrapped = wrapped
  42 
  43     def __enter__(self):
  44         return self.wrapped.__enter__()
  45 
  46     def __exit__(self, *args):
  47         try:
  48             # allow the wrapped manager to perform any cleanup tasks regardless
  49             # of whether or not we are suppressing an exception raised within
  50             # the managed block
  51             self.wrapped.__exit__(*args)
  52         except Exception:
  53             pass
  54 
  55         return True

  58 class RedisTSDB(BaseTSDB):
  59     """
  60     A time series storage backend for Redis.
  61 
  62     The time series API supports three data types:
  63 
  64         * simple counters
  65         * distinct counters (number of unique elements seen)
  66         * frequency tables (a set of items ranked by most frequently observed)
  67 
  68     The backend also supports virtual nodes (``vnodes``) which controls shard
  69     distribution. This value should be set to the anticipated maximum number of
  70     physical hosts and not modified after data has been written.
  71 
  72     Simple counters are stored in hashes. The key of the hash is composed of
  73     the model, epoch (which defines the start of the rollup period), and a
  74     shard identifier. This allows TTLs to be applied to the entire bucket,
  75     instead of having to be stored for every individual element in the rollup
  76     period. This results in a data layout that looks something like this::
  77 
  78         {
  79             "<model>:<epoch>:<shard id>": {
  80                 "<key>": value,
  81                 ...
  82             },
  83             ...
  84         }
  85 
  86     Distinct counters are stored using HyperLogLog, which provides a
  87     cardinality estimate with a standard error of 0.8%. The data layout looks
  88     something like this::
  89 
  90         {
  91             "<model>:<epoch>:<key>": value,
  92             ...
  93         }
  94 
  95     Frequency tables are modeled using two data structures:
  96 
  97         * top-N index: a sorted set containing the most frequently observed items,
  98         * estimation matrix: a hash table containing counters, used in a Count-Min sketch
  99 
 100     Member scores are 100% accurate until the index is filled (and no memory is
 101     used for the estimation matrix until this point), after which the data
 102     structure switches to a probabilistic implementation and accuracy begins to
 103     degrade for less frequently observed items, but remains accurate for more
 104     frequently observed items.
 105 
 106     Frequency tables are especially useful when paired with a (non-distinct)
 107     counter of the total number of observations so that scores of items of the
 108     frequency table can be displayed as percentages of the whole data set.
 109     (Additional documentation and the bulk of the logic for implementing the
 110     frequency table API can be found in the ``cmsketch.lua`` script.)
 111     """

 127     def get_cluster(self, environment_id):
 128         """\
 129         Returns a 2-tuple of the form ``(cluster, durable)``.
 130 
 131         When a cluster is marked as "durable", any exception raised while
 132         attempting to write data to the cluster is propagated. When the cluster
 133         is *not* marked as "durable", exceptions raised while attempting to
 134         write data to the cluster are *not* propagated. This flag does not have
 135         an effect on read operations.
 136         """
 137         return self.cluster, True

 158                 prefix=self.prefix,
 159                 model=model.value,

 166     def make_counter_key(self, model, rollup, timestamp, key, environment_id):
 167         """
 168         Make a key that is used for counter values.
 169 
 170         Returns a 2-tuple that contains the hash key and the hash field.
 171         """
 172         model_key = self.get_model_key(key)
 173 
 174         if isinstance(model_key, six.integer_types):
 175             vnode = model_key % self.vnodes
 176         else:
 177             if isinstance(model_key, six.text_type):
 178                 model_key = model_key.encode("utf-8")
 179             vnode = crc32(model_key) % self.vnodes
 180 
 181         return (
 182             u"{prefix}{model}:{epoch}:{vnode}".format(
 183                 prefix=self.prefix,
 184                 model=model.value,
 185                 epoch=self.normalize_to_rollup(timestamp, rollup),
 186                 vnode=vnode,
 187             ),
 188             self.add_environment_parameter(model_key, environment_id),
 189         )

 191     def get_model_key(self, key):
 192         # We specialize integers so that a pure int-map can be optimized by
 193         # Redis, whereas long strings (say tag values) will store in a more
 194         # efficient hashed format.
 195         if not isinstance(key, six.integer_types):
 196             # enforce utf-8 encoding
 197             if isinstance(key, six.text_type):
 198                 key = key.encode("utf-8")
 199             return md5(repr(key)).hexdigest()
 200         return key

 207     def incr_multi(self, items, timestamp=None, count=1, environment_id=None):
 208         """
 209         Increment project ID=1 and group ID=5:
 210 
 211         >>> incr_multi([(TimeSeriesModel.project, 1), (TimeSeriesModel.group, 5)])
 212 
 213         Increment individual timestamps:
 214 
 215         >>> incr_multi([(TimeSeriesModel.project, 1, {"timestamp": ...}),
 216         ...             (TimeSeriesModel.group, 5, {"timestamp": ...})])
 217         """
 218 
 219         default_timestamp = timestamp
 220         default_count = count
 221 
 222         self.validate_arguments([item[0] for item in items], [environment_id])
 223 
 224         if default_timestamp is None:
 225             default_timestamp = timezone.now()
 226 
 227         for (cluster, durable), environment_ids in self.get_cluster_groups(
 228             set([None, environment_id])
 229         ):
 230             manager = cluster.map()
 231             if not durable:
 232                 manager = SuppressionWrapper(manager)
 233 
 234             with manager as client:
 235                 # (hash_key, hash_field) -> count
 236                 key_operations = defaultdict(lambda: 0)
 237                 # (hash_key) -> "max expiration encountered"
 238                 key_expiries = defaultdict(lambda: 0.0)
 239 
 240                 for rollup, max_values in six.iteritems(self.rollups):
 241                     for item in items:
 242                         if len(item) == 2:
 243                             model, key = item
 244                             options = {}
 245                         else:
 246                             model, key, options = item
 247 
 248                         count = options.get("count", default_count)
 249                         timestamp = options.get("timestamp", default_timestamp)
 250 
 251                         expiry = self.calculate_expiry(rollup, max_values, timestamp)
 252 
 253                         for environment_id in environment_ids:
 254                             hash_key, hash_field = self.make_counter_key(
 255                                 model, rollup, timestamp, key, environment_id
 256                             )
 257 
 258                             if key_expiries[hash_key] < expiry:
 259                                 key_expiries[hash_key] = expiry
 260 
 261                             key_operations[(hash_key, hash_field)] += count
 262 
 263                 for (hash_key, hash_field), count in six.iteritems(key_operations):
 264                     client.hincrby(hash_key, hash_field, count)
 265                     if key_expiries.get(hash_key):
 266                         client.expireat(hash_key, key_expiries.pop(hash_key))
 267 
 268     def get_range(self, model, keys, start, end, rollup=None, environment_ids=None):
 269         """
 270         To get a range of data for group ID=[1, 2, 3]:
 271 
 272         >>> now = timezone.now()
 273         >>> get_keys(TimeSeriesModel.group, [1, 2, 3],
 274         >>>          start=now - timedelta(days=1),
 275         >>>          end=now)
 276         """
 277         # redis backend doesn't support multiple envs
 278         if environment_ids is not None and len(environment_ids) > 1:
 279             raise NotImplementedError
 280         environment_id = environment_ids[0] if environment_ids else None
 281 
 282         self.validate_arguments([model], [environment_id])
 283 
 284         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
 285         series = map(to_datetime, series)
 286 
 287         results = []
 288         cluster, _ = self.get_cluster(environment_id)
 289         with cluster.map() as client:
 290             for key in keys:
 291                 for timestamp in series:
 292                     hash_key, hash_field = self.make_counter_key(
 293                         model, rollup, timestamp, key, environment_id
 294                     )
 295                     results.append(
 296                         (to_timestamp(timestamp), key, client.hget(hash_key, hash_field))
 297                     )
 298 
 299         results_by_key = defaultdict(dict)
 300         for epoch, key, count in results:
 301             results_by_key[key][epoch] = int(count.value or 0)
 302 
 303         for key, points in six.iteritems(results_by_key):
 304             results_by_key[key] = sorted(points.items())
 305         return dict(results_by_key)

 307     def merge(self, model, destination, sources, timestamp=None, environment_ids=None):
 308         environment_ids = (set(environment_ids) if environment_ids is not None else set()).union(
 309             [None]
 310         )
 311 
 312         self.validate_arguments([model], environment_ids)
 313 
 314         rollups = self.get_active_series(timestamp=timestamp)
 315 
 316         for (cluster, durable), environment_ids in self.get_cluster_groups(environment_ids):
 317             manager = cluster.map()
 318             if not durable:
 319                 manager = SuppressionWrapper(manager)
 320 
 321             with manager as client:
 322                 data = {}
 323                 for rollup, series in rollups.items():
 324                     data[rollup] = {}
 325                     for timestamp in series:
 326                         results = data[rollup][timestamp] = defaultdict(list)
 327                         for source in sources:
 328                             for environment_id in environment_ids:
 329                                 source_hash_key, source_hash_field = self.make_counter_key(
 330                                     model, rollup, timestamp, source, environment_id
 331                                 )
 332                                 results[environment_id].append(
 333                                     client.hget(source_hash_key, source_hash_field)
 334                                 )
 335                                 client.hdel(source_hash_key, source_hash_field)
 336 
 337             with cluster.map() as client:
 338                 for rollup, series in data.items():
 339                     for timestamp, results in series.items():
 340                         for environment_id, promises in results.items():
 341                             total = sum([int(p.value) for p in promises if p.value])
 342                             if total:
 343                                 (
 344                                     destination_hash_key,
 345                                     destination_hash_field,
 346                                 ) = self.make_counter_key(
 347                                     model, rollup, timestamp, destination, environment_id
 348                                 )
 349                                 client.hincrby(destination_hash_key, destination_hash_field, total)
 350                                 client.expireat(
 351                                     destination_hash_key,
 352                                     self.calculate_expiry(rollup, self.rollups[rollup], timestamp),
 353                                 )

 386     def record_multi(self, items, timestamp=None, environment_id=None):
 387         """
 388         Record an occurrence of an item in a distinct counter.
 389         """
 390         self.validate_arguments([model for model, key, values in items], [environment_id])
 391 
 392         if timestamp is None:
 393             timestamp = timezone.now()
 394 
 395         ts = int(to_timestamp(timestamp))  # ``timestamp`` is not actually a timestamp :(
 396 
 397         for (cluster, durable), environment_ids in self.get_cluster_groups(
 398             set([None, environment_id])
 399         ):
 400             manager = cluster.fanout()
 401             if not durable:
 402                 manager = SuppressionWrapper(manager)
 403 
 404             with manager as client:
 405                 for model, key, values in items:
 406                     c = client.target_key(key)
 407                     for rollup, max_values in six.iteritems(self.rollups):
 408                         for environment_id in environment_ids:
 409                             k = self.make_key(model, rollup, ts, key, environment_id)
 410                             c.pfadd(k, *values)
 411                             c.expireat(k, self.calculate_expiry(rollup, max_values, timestamp))

 413     def get_distinct_counts_series(
 414         self, model, keys, start, end=None, rollup=None, environment_id=None
 415     ):
 416         """
 417         Fetch counts of distinct items for each rollup interval within the range.
 418         """
 419         self.validate_arguments([model], [environment_id])
 420 
 421         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
 422 
 423         responses = {}
 424         cluster, _ = self.get_cluster(environment_id)
 425         with cluster.fanout() as client:
 426             for key in keys:
 427                 c = client.target_key(key)
 428                 r = responses[key] = []
 429                 for timestamp in series:
 430                     r.append(
 431                         (
 432                             timestamp,
 433                             c.pfcount(self.make_key(model, rollup, timestamp, key, environment_id)),
 434                         )
 435                     )
 436 
 437         return {
 438             key: [(timestamp, promise.value) for timestamp, promise in value]
 439             for key, value in six.iteritems(responses)
 440         }

 442     def get_distinct_counts_totals(
 443         self, model, keys, start, end=None, rollup=None, environment_id=None
 444     ):
 445         """
 446         Count distinct items during a time range.
 447         """
 448         self.validate_arguments([model], [environment_id])
 449 
 450         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
 451 
 452         responses = {}
 453         cluster, _ = self.get_cluster(environment_id)
 454         with cluster.fanout() as client:
 455             for key in keys:
 456                 # XXX: The current versions of the Redis driver don't implement
 457                 # ``PFCOUNT`` correctly (although this is fixed in the Git

 459                 # supports a single key argument -- not the variadic signature
 460                 # supported by the protocol -- so we have to call the command

 467 
 468         return {key: value.value for key, value in six.iteritems(responses)}

 470     def get_distinct_counts_union(
 471         self, model, keys, start, end=None, rollup=None, environment_id=None
 472     ):
 473         self.validate_arguments([model], [environment_id])
 474 
 475         if not keys:
 476             return 0
 477 
 478         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
 479 
 480         temporary_id = uuid.uuid1().hex
 481 
 482         def make_temporary_key(key):
 483             return u"{}{}:{}".format(self.prefix, temporary_id, key)
 484 
 485         def expand_key(key):
 486             """
 487             Return a list containing all keys for each interval in the series for a key.
 488             """
 489             return [
 490                 self.make_key(model, rollup, timestamp, key, environment_id) for timestamp in series
 491             ]
 492 
 493         cluster, _ = self.get_cluster(environment_id)
 494         router = cluster.get_router()
 495 
 496         def map_key_to_host(hosts, key):
 497             """
 498             Identify the host where a key is located and add it to the host map.
 499             """
 500             hosts[router.get_host_for_key(key)].add(key)
 501             return hosts
 502 
 503         def get_partition_aggregate(value):
 504             """
 505             Fetch the HyperLogLog value (in its raw byte representation) that
 506             results from merging all HyperLogLogs at the provided keys.
 507             """
 508             (host, keys) = value
 509             destination = make_temporary_key(u"p:{}".format(host))
 510             client = cluster.get_local_client(host)
 511             with client.pipeline(transaction=False) as pipeline:
 512                 pipeline.execute_command(
 513                     "PFMERGE", destination, *itertools.chain.from_iterable(map(expand_key, keys))
 514                 )
 515                 pipeline.get(destination)
 516                 pipeline.delete(destination)
 517                 return (host, pipeline.execute()[1])
 518 
 519         def merge_aggregates(values):
 520             """
 521             Calculate the cardinality of the provided HyperLogLog values.
 522             """
 523             destination = make_temporary_key("a")  # all values will be merged into this key
 524             aggregates = {make_temporary_key(u"a:{}".format(host)): value for host, value in values}
 525 
 526             # Choose a random host to execute the reduction on. (We use a host
 527             # here that we've already accessed as part of this process -- this
 528             # way, we constrain the choices to only hosts that we know are
 529             # running.)
 530             client = cluster.get_local_client(random.choice(values)[0])
 531             with client.pipeline(transaction=False) as pipeline:
 532                 pipeline.mset(aggregates)
 533                 pipeline.execute_command("PFMERGE", destination, *aggregates.keys())
 534                 pipeline.execute_command("PFCOUNT", destination)
 535                 pipeline.delete(destination, *aggregates.keys())
 536                 return pipeline.execute()[2]
 537 
 538         # TODO: This could be optimized to skip the intermediate step for the
 539         # host that has the largest number of keys if the final merge and count
 540         # is performed on that host. If that host contains *all* keys, the
 541         # final reduction could be performed as a single PFCOUNT, skipping the
 542         # MSET and PFMERGE operations entirely.
 543 
 544         return merge_aggregates(
 545             [
 546                 get_partition_aggregate(x)
 547                 for x in reduce(map_key_to_host, keys, defaultdict(set)).items()
 548             ]
 549         )
 550 
 551     def merge_distinct_counts(
 552         self, model, destination, sources, timestamp=None, environment_ids=None
 553     ):
 554         environment_ids = (set(environment_ids) if environment_ids is not None else set()).union(
 555             [None]
 556         )
 557 
 558         self.validate_arguments([model], environment_ids)
 559 
 560         rollups = self.get_active_series(timestamp=timestamp)
 561 
 562         for (cluster, durable), environment_ids in self.get_cluster_groups(environment_ids):
 563             wrapper = SuppressionWrapper if not durable else lambda value: value
 564 
 565             temporary_id = uuid.uuid1().hex
 566 
 567             def make_temporary_key(key):
 568                 return u"{}{}:{}".format(self.prefix, temporary_id, key)
 569 
 570             data = {}
 571             for rollup, series in rollups.items():
 572                 data[rollup] = {timestamp: {e: [] for e in environment_ids} for timestamp in series}
 573 
 574             with wrapper(cluster.fanout()) as client:
 575                 for source in sources:
 576                     c = client.target_key(source)
 577                     for rollup, series in data.items():
 578                         for timestamp, results in series.items():
 579                             for environment_id in environment_ids:
 580                                 key = self.make_key(
 581                                     model, rollup, to_timestamp(timestamp), source, environment_id
 582                                 )
 583                                 results[environment_id].append(c.get(key))
 584                                 c.delete(key)
 585 
 586             with wrapper(cluster.fanout()) as client:
 587                 c = client.target_key(destination)
 588 
 589                 temporary_key_sequence = itertools.count()
 590 
 591                 for rollup, series in data.items():
 592                     for timestamp, results in series.items():
 593                         for environment_id, promises in results.items():
 594                             values = {}
 595                             for promise in promises:
 596                                 if promise.value is None:
 597                                     continue
 598                                 k = make_temporary_key(next(temporary_key_sequence))
 599                                 values[k] = promise.value
 600 
 601                             if values:
 602                                 key = self.make_key(
 603                                     model,
 604                                     rollup,
 605                                     to_timestamp(timestamp),
 606                                     destination,
 607                                     environment_id,
 608                                 )
 609                                 c.mset(values)
 610                                 c.pfmerge(key, key, *values.keys())
 611                                 c.delete(*values.keys())
 612                                 c.expireat(
 613                                     key,
 614                                     self.calculate_expiry(rollup, self.rollups[rollup], timestamp),
 615                                 )

 654     def record_frequency_multi(self, requests, timestamp=None, environment_id=None):
 655         self.validate_arguments([model for model, request in requests], [environment_id])
 656 
 657         if not self.enable_frequency_sketches:
 658             return
 659 
 660         if timestamp is None:
 661             timestamp = timezone.now()
 662 
 663         ts = int(to_timestamp(timestamp))  # ``timestamp`` is not actually a timestamp :(
 664 
 665         for (cluster, durable), environment_ids in self.get_cluster_groups(
 666             set([None, environment_id])
 667         ):
 668             commands = {}
 669 
 670             for model, request in requests:
 671                 for key, items in six.iteritems(request):
 672                     keys = []
 673                     expirations = {}
 674 
 675                     # Figure out all of the keys we need to be incrementing, as
 676                     # well as their expiration policies.
 677                     for rollup, max_values in six.iteritems(self.rollups):
 678                         for environment_id in environment_ids:
 679                             chunk = self.make_frequency_table_keys(
 680                                 model, rollup, ts, key, environment_id
 681                             )
 682                             keys.extend(chunk)
 683 
 684                         expiry = self.calculate_expiry(rollup, max_values, timestamp)
 685                         for k in chunk:
 686                             expirations[k] = expiry
 687 
 688                     arguments = ["INCR"] + list(self.DEFAULT_SKETCH_PARAMETERS)
 689                     for member, score in items.items():
 690                         arguments.extend((score, member))
 691 
 692                     # Since we're essentially merging dictionaries, we need to
 693                     # append this to any value that already exists at the key.
 694                     cmds = commands.setdefault(key, [])
 695                     cmds.append((CountMinScript, keys, arguments))
 696                     for k, t in expirations.items():
 697                         cmds.append(("EXPIREAT", k, t))
 698 
 699             try:
 700                 cluster.execute_commands(commands)
 701             except Exception:
 702                 if durable:
 703                     raise

 705     def get_most_frequent(
 706         self, model, keys, start, end=None, rollup=None, limit=None, environment_id=None
 707     ):
 708         self.validate_arguments([model], [environment_id])
 709 
 710         if not self.enable_frequency_sketches:
 711             raise NotImplementedError("Frequency sketches are disabled.")
 712 
 713         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
 714 
 715         arguments = ["RANKED"] + list(self.DEFAULT_SKETCH_PARAMETERS)
 716         if limit is not None:
 717             arguments.append(int(limit))
 718 
 719         commands = {}
 720         for key in keys:
 721             ks = []
 722             for timestamp in series:
 723                 ks.extend(
 724                     self.make_frequency_table_keys(model, rollup, timestamp, key, environment_id)
 725                 )
 726             commands[key] = [(CountMinScript, ks, arguments)]
 727 
 728         results = {}
 729         cluster, _ = self.get_cluster(environment_id)
 730         for key, responses in cluster.execute_commands(commands).items():
 731             results[key] = [(member, float(score)) for member, score in responses[0].value]
 732 
 733         return results

 735     def get_most_frequent_series(
 736         self, model, keys, start, end=None, rollup=None, limit=None, environment_id=None
 737     ):
 738         self.validate_arguments([model], [environment_id])
 739 
 740         if not self.enable_frequency_sketches:
 741             raise NotImplementedError("Frequency sketches are disabled.")
 742 
 743         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
 744 
 745         arguments = ["RANKED"] + list(self.DEFAULT_SKETCH_PARAMETERS)
 746         if limit is not None:
 747             arguments.append(int(limit))
 748 
 749         commands = {}
 750         for key in keys:
 751             commands[key] = [
 752                 (
 753                     CountMinScript,
 754                     self.make_frequency_table_keys(model, rollup, timestamp, key, environment_id),
 755                     arguments,
 756                 )
 757                 for timestamp in series
 758             ]
 759 
 760         def unpack_response(response):
 761             return {item: float(score) for item, score in response.value}
 762 
 763         results = {}
 764         cluster, _ = self.get_cluster(environment_id)
 765         for key, responses in cluster.execute_commands(commands).items():
 766             results[key] = zip(series, map(unpack_response, responses))
 767 
 768         return results

 770     def get_frequency_series(self, model, items, start, end=None, rollup=None, environment_id=None):
 771         self.validate_arguments([model], [environment_id])
 772 
 773         if not self.enable_frequency_sketches:
 774             raise NotImplementedError("Frequency sketches are disabled.")
 775 
 776         rollup, series = self.get_optimal_rollup_series(start, end, rollup)
 777 
 778         # Here we freeze ordering of the members, since we'll be passing these
 779         # as positional arguments to the Redis script and later associating the
 780         # results (which are returned in the same order that the arguments were
 781         # provided) with the original input values to compose the result.
 782         for key, members in items.items():
 783             items[key] = list(members)
 784 
 785         commands = {}
 786 
 787         arguments = ["ESTIMATE"] + list(self.DEFAULT_SKETCH_PARAMETERS)
 788         for key, members in items.items():
 789             ks = []
 790             for timestamp in series:
 791                 ks.extend(
 792                     self.make_frequency_table_keys(model, rollup, timestamp, key, environment_id)
 793                 )
 794 
 795             commands[key] = [(CountMinScript, ks, arguments + members)]
 796 
 797         results = {}
 798 
 799         cluster, _ = self.get_cluster(environment_id)
 800         for key, responses in cluster.execute_commands(commands).items():
 801             members = items[key]
 802 
 803             chunk = results[key] = []
 804             for timestamp, scores in zip(series, responses[0].value):
 805                 chunk.append((timestamp, dict(zip(members, map(float, scores)))))
 806 
 807         return results

 809     def get_frequency_totals(self, model, items, start, end=None, rollup=None, environment_id=None):
 810         self.validate_arguments([model], [environment_id])
 811 
 812         if not self.enable_frequency_sketches:
 813             raise NotImplementedError("Frequency sketches are disabled.")
 814 
 815         responses = {}
 816 
 817         for key, series in six.iteritems(
 818             self.get_frequency_series(model, items, start, end, rollup, environment_id)
 819         ):
 820             response = responses[key] = {}
 821             for timestamp, results in series:
 822                 for member, value in results.items():
 823                     response[member] = response.get(member, 0.0) + value
 824 
 825         return responses

 827     def merge_frequencies(self, model, destination, sources, timestamp=None, environment_ids=None):
 828         environment_ids = list(
 829             (set(environment_ids) if environment_ids is not None else set()).union([None])
 830         )
 831 
 832         self.validate_arguments([model], environment_ids)
 833 
 834         if not self.enable_frequency_sketches:
 835             return
 836 
 837         rollups = []
 838         for rollup, samples in self.rollups.items():
 839             _, series = self.get_optimal_rollup_series(
 840                 to_datetime(self.get_earliest_timestamp(rollup, timestamp=timestamp)),
 841                 end=None,
 842                 rollup=rollup,
 843             )
 844             rollups.append((rollup, map(to_datetime, series)))
 845 
 846         for (cluster, durable), environment_ids in self.get_cluster_groups(environment_ids):
 847             exports = defaultdict(list)
 848 
 849             for source in sources:
 850                 for rollup, series in rollups:
 851                     for timestamp in series:
 852                         keys = []
 853                         for environment_id in environment_ids:
 854                             keys.extend(
 855                                 self.make_frequency_table_keys(
 856                                     model, rollup, to_timestamp(timestamp), source, environment_id
 857                                 )
 858                             )
 859                         arguments = ["EXPORT"] + list(self.DEFAULT_SKETCH_PARAMETERS)
 860                         exports[source].extend([(CountMinScript, keys, arguments), ["DEL"] + keys])
 861 
 862             try:
 863                 responses = cluster.execute_commands(exports)
 864             except Exception:
 865                 if durable:
 866                     raise
 867                 else:
 868                     continue
 869 
 870             imports = []
 871 
 872             for source, results in responses.items():
 873                 results = iter(results)
 874                 for rollup, series in rollups:
 875                     for timestamp in series:
 876                         for environment_id, payload in zip(environment_ids, next(results).value):
 877                             imports.append(
 878                                 (
 879                                     CountMinScript,
 880                                     self.make_frequency_table_keys(
 881                                         model,
 882                                         rollup,
 883                                         to_timestamp(timestamp),
 884                                         destination,
 885                                         environment_id,
 886                                     ),
 887                                     ["IMPORT"] + list(self.DEFAULT_SKETCH_PARAMETERS) + [payload],
 888                                 )
 889                             )
 890                         next(results)  # pop off the result of DEL
 891 
 892             try:
 893                 cluster.execute_commands({destination: imports})
 894             except Exception:
 895                 if durable:
 896                     raise
