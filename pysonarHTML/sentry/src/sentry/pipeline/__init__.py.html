  58 class NestedPipelineView(PipelineView):
  59     """
  60     A NestedPipelineView can be used within other pipelines to process another
  61     pipeline within a pipeline. Note that the nested pipelines finish_pipeline
  62     will NOT be called, instead it's data will be bound into the parent
  63     pipeline and the parents pipeline moved to the next step.
  64 
  65     Useful for embedding an identity authentication pipeline.
  66     """
  67 
  68     def __init__(self, bind_key, pipeline_cls, provider_key, config=None):
  69         self.provider_key = provider_key
  70         self.config = config or {}
  71 
  72         class NestedPipeline(pipeline_cls):
  73             def set_parent_pipeline(self, parent_pipeline):
  74                 self.parent_pipeline = parent_pipeline
  75 
  76             def finish_pipeline(self):
  77                 self.parent_pipeline.bind_state(bind_key, self.fetch_state())
  78                 self.clear_session()
  79 
  80                 return self.parent_pipeline.next_step()
  81 
  82         self.pipeline_cls = NestedPipeline
  83 
  84     def dispatch(self, request, pipeline):
  85         nested_pipeline = self.pipeline_cls(
  86             organization=pipeline.organization,
  87             request=request,
  88             provider_key=self.provider_key,
  89             config=self.config,
  90         )
  91 
  92         nested_pipeline.set_parent_pipeline(pipeline)
  93         # nested_pipeline.bind_state('_parent', pipeline.fetch_state())
  94 
  95         if not nested_pipeline.is_valid():
  96             nested_pipeline.initialize()
  97 
  98         return nested_pipeline.current_step()

 101 class Pipeline(object):
 102     """
 103     Pipeline provides a mechanism to guide the user through a request
 104     'pipeline', where each view may be completed by calling the ``next_step``
 105     pipeline method to traverse through the pipe.
 106 
 107     The pipeline works with a PipelineProvider object which provides the
 108     pipeline views and is made available to the views through the passed in
 109     pipeline.
 110 
 111     :provider_manager:
 112     A class property that must be specified to allow for lookup of a provider
 113     implementation object given it's key.
 114 
 115     :provider_model_cls:
 116     The Provider model object represents the instance of an object implementing
 117     the PipelineProvider interface. This is used to look up the instance
 118     when constructing an in progress pipeline (get_for_request).
 119 
 120     :config:
 121     A object that specifies additional pipeline and provider runtime
 122     configurations. An example of usage is for OAuth Identity providers, for
 123     overriding the scopes. The config object will be passed into the provider
 124     using the ``set_config`` method.
 125     """
 126 
 127     pipeline_name = None
 128     provider_manager = None
 129     provider_model_cls = None
 130 
 131     @classmethod
 132     def get_for_request(cls, request):
 133         state = RedisSessionStore(request, cls.pipeline_name)
 134         if not state.is_valid():
 135             return None
 136 
 137         provider_model = None
 138         if state.provider_model_id:
 139             provider_model = cls.provider_model_cls.objects.get(id=state.provider_model_id)
 140 
 141         organization = None
 142         if state.org_id:
 143             organization = Organization.objects.get(id=state.org_id)
 144 
 145         provider_key = state.provider_key
 146         config = state.config
 147 
 148         return cls(
 149             request,
 150             organization=organization,
 151             provider_key=provider_key,
 152             provider_model=provider_model,
 153             config=config,
 154         )
 155 
 156     def __init__(self, request, provider_key, organization=None, provider_model=None, config=None):
 157         if config is None:
 158             config = {}
 159 
 160         self.request = request
 161         self.organization = organization
 162         self.state = RedisSessionStore(request, self.pipeline_name)
 163         self.provider = self.provider_manager.get(provider_key)
 164         self.provider_model = provider_model
 165 
 166         self.config = config
 167         self.provider.set_pipeline(self)
 168         self.provider.set_config(config)
 169 
 170         self.pipeline_views = self.get_pipeline_views()
 171 
 172         # we serialize the pipeline to be ['fqn.PipelineView', ...] which
 173         # allows us to determine if the pipeline has changed during the auth
 174         # flow or if the user is somehow circumventing a chunk of it
 175         pipe_ids = [
 176             u"{}.{}".format(type(v).__module__, type(v).__name__) for v in self.pipeline_views
 177         ]
 178         self.signature = md5_text(*pipe_ids).hexdigest()
 179 
 180     def get_pipeline_views(self):
 181         """
 182         Retrieve the pipeline views from the provider.
 183 
 184         You may wish to override this method to provide views that all
 185         providers should inherit, or customize the provider method called to
 186         retrieve the views.
 187         """
 188         return self.provider.get_pipeline_views()
 189 
 190     def is_valid(self):
 191         return self.state.is_valid() and self.state.signature == self.signature
 192 
 193     def initialize(self):
 194         self.state.regenerate(
 195             {
 196                 "uid": self.request.user.id if self.request.user.is_authenticated() else None,
 197                 "provider_model_id": self.provider_model.id if self.provider_model else None,
 198                 "provider_key": self.provider.key,
 199                 "org_id": self.organization.id if self.organization else None,
 200                 "step_index": 0,
 201                 "signature": self.signature,
 202                 "config": self.config,
 203                 "data": {},
 204             }
 205         )
 206 
 207     def clear_session(self):
 208         self.state.clear()
 209 
 210     def current_step(self):
 211         """
 212         Render the current step.
 213         """
 214         step_index = self.state.step_index
 215 
 216         if step_index == len(self.pipeline_views):
 217             return self.finish_pipeline()
 218 
 219         step = self.pipeline_views[step_index]
 220 
 221         # support late binding steps
 222         if isinstance(step, LambdaType):
 223             step = step()
 224 
 225         return step.dispatch(request=self.request, pipeline=self)
 226 
 227     def error(self, message):
 228         context = {"error": message}
 229         return render_to_response("sentry/pipeline-error.html", context, self.request)
 230 
 231     def next_step(self, step_size=1):
 232         """
 233         Render the next step.
 234         """
 235         self.state.step_index += step_size
 236         if self.organization:
 237             analytics.record(
 238                 "integrations.pipeline_step",
 239                 user_id=self.request.user.id,
 240                 organization_id=self.organization.id,
 241                 integration=self.provider.key,
 242                 step_index=self.state.step_index,
 243                 pipeline_type="reauth" if self.fetch_state("integration_id") else "install",
 244             )
 245         return self.current_step()
 246 
 247     def finish_pipeline(self):
 248         """
 249         Called when the pipeline completes the final step.
 250         """
 251         raise NotImplementedError
 252 
 253     def bind_state(self, key, value):
 254         data = self.state.data
 255         data[key] = value
 256 
 257         self.state.data = data
 258 
 259     def fetch_state(self, key=None):
 260         return self.state.data if key is None else self.state.data.get(key)
 261 
 262     def get_logger(self):
 263         return logging.getLogger("sentry.integration.%s" % (self.provider.key,))
