  13 class DatadogMetricsBackend(MetricsBackend):
  14     def __init__(self, prefix=None, **kwargs):
  15         # TODO(dcramer): it'd be nice if the initialize call wasn't a global
  16         self.tags = kwargs.pop("tags", None)
  17         if "host" in kwargs:
  18             self.host = kwargs.pop("host")
  19         else:
  20             self.host = get_hostname()
  21         initialize(**kwargs)
  22         super(DatadogMetricsBackend, self).__init__(prefix=prefix)
  23 
  24     def __del__(self):
  25         try:
  26             self.stats.stop()
  27         except TypeError:
  28             # TypeError: 'NoneType' object is not callable
  29             pass
  30 
  31     @memoize
  32     def stats(self):
  33         instance = ThreadStats()
  34         instance.start()
  35         return instance
  36 
  37     def incr(self, key, instance=None, tags=None, amount=1, sample_rate=1):
  38         if tags is None:
  39             tags = {}
  40         if self.tags:
  41             tags.update(self.tags)
  42         if instance:
  43             tags["instance"] = instance
  44         if tags:
  45             tags = [u"{}:{}".format(*i) for i in tags.items()]
  46         self.stats.increment(
  47             self._get_key(key), amount, sample_rate=sample_rate, tags=tags, host=self.host
  48         )
