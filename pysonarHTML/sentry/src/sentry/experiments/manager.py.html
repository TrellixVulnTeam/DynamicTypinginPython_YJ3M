   5 class ExperimentManager(object):
   6     """
   7     Allows loading of experiment assignments (done in getsentry) on the frontend by
   8     including them in the serialized org details via the org serializer which is in sentry.
   9     """
  10 
  11     def __init__(self):
  12         self._experiments = {}
  13 
  14     def add(self, experiment, param):
  15         """
  16         >>> ExperimentManager.add(ExperimentClass, param='name_of_param')
  17         """
  18         self._experiments[experiment.__name__] = {"experiment": experiment, "param": param}

  20     def all(self, org=None, actor=None, user=None):
  21         """
  22         Returns an object with all the experiment assignments for an organization or user.
  23 
  24         :param org: The organization for org based experiments
  25         :param actor: The actor for org based experiments
  26         :param user: The user for user based experiments
  27         """
  28         if not org and not user:
  29             return {}
  30 
  31         assignments = {}
  32 
  33         if org:
  34             kwargs = {"org": org, "actor": actor}
  35             unit = "org"
  36         else:
  37             kwargs = {"user": user}
  38             unit = "user"
  39 
  40         for k, v in six.iteritems(self._experiments):
  41             cls = v["experiment"]
  42             if hasattr(cls, "unit") and cls.unit == unit:
  43                 assignments[k] = cls(**kwargs).get_variant(v["param"], log_exposure=False)
  44         return assignments
  45 
  46     def get(self, experiment_name, org=None, actor=None, user=None):
  47         """
  48         Returns the assignment for an experiment.
  49 
  50         :param experiment_name:  The name of the experiment

  52         :param actor: The actor for org based experiments
  53         :param user: The user for user based experiments
  54         """
  55         if not org and not user:
  56             return None
  57 
  58         value = self._experiments.get(experiment_name)
  59         if not value:
  60             return None
  61 
  62         if org:
  63             kwargs = {"org": org, "actor": actor}
  64         else:
  65             kwargs = {"user": user}
  66 
  67         cls = value["experiment"]
  68         return cls(**kwargs).get_variant(value["param"], log_exposure=False)
