  24 def register_subscriber(subscriber_key):
  25     def inner(func):
  26         if subscriber_key in subscriber_registry:
  27             raise Exception("Handler already registered for %s" % subscriber_key)
  28         subscriber_registry[subscriber_key] = func
  29         return func
  30 
  31     return inner

  42 class QuerySubscriptionConsumer(object):
  43     """
  44     A Kafka consumer that processes query subscription update messages. Each message has
  45     a related subscription id and the latest values related to the subscribed query.
  46     These values are passed along to a callback associated with the subscription.
  47     """
  48 
  49     topic_to_dataset = {
  50         settings.KAFKA_EVENTS_SUBSCRIPTIONS_RESULTS: QueryDatasets.EVENTS,
  51         settings.KAFKA_TRANSACTIONS_SUBSCRIPTIONS_RESULTS: QueryDatasets.TRANSACTIONS,
  52     }
  53 
  54     def __init__(
  55         self, group_id, topic=None, commit_batch_size=100, initial_offset_reset="earliest"
  56     ):
  57         self.group_id = group_id
  58         if not topic:
  59             topic = settings.KAFKA_EVENTS_SUBSCRIPTIONS_RESULTS
  60         self.topic = topic
  61         cluster_name = settings.KAFKA_TOPICS[topic]["cluster"]
  62         self.bootstrap_servers = settings.KAFKA_CLUSTERS[cluster_name]["bootstrap.servers"]
  63         self.commit_batch_size = commit_batch_size
  64         self.initial_offset_reset = initial_offset_reset
  65         self.offsets = {}
  66         self.consumer = None
  67 
  68     def run(self):
  69         logger.debug("Starting snuba query subscriber")
  70         self.offsets.clear()
  71 
  72         conf = {
  73             "bootstrap.servers": self.bootstrap_servers,
  74             "group.id": self.group_id,
  75             "session.timeout.ms": 6000,
  76             "auto.offset.reset": self.initial_offset_reset,
  77             "enable.auto.commit": "false",
  78             "enable.auto.offset.store": "false",
  79             "enable.partition.eof": "false",
  80             "default.topic.config": {"auto.offset.reset": self.initial_offset_reset},
  81         }
  82 
  83         def on_assign(consumer, partitions):
  84             for partition in partitions:
  85                 if partition.offset == OFFSET_INVALID:
  86                     updated_offset = None
  87                 else:
  88                     updated_offset = partition.offset
  89                 self.offsets[partition.partition] = updated_offset
  90 
  91         def on_revoke(consumer, partitions):
  92             partition_numbers = [partition.partition for partition in partitions]
  93             self.commit_offsets(partition_numbers)
  94             for partition_number in partition_numbers:
  95                 self.offsets.pop(partition_number, None)
  96 
  97         self.consumer = Consumer(conf)
  98         self.consumer.subscribe([self.topic], on_assign=on_assign, on_revoke=on_revoke)
  99 
 100         try:
 101             i = 0
 102             while True:
 103                 message = self.consumer.poll(0.1)
 104                 if message is None:
 105                     continue
 106 
 107                 error = message.error()
 108                 if error is not None:
 109                     raise KafkaException(error)
 110 
 111                 i = i + 1
 112 
 113                 with sentry_sdk.start_span(
 114                     Span(
 115                         op="handle_message",
 116                         transaction="query_subscription_consumer_process_message",
 117                         sampled=True,
 118                     )
 119                 ), metrics.timer("snuba_query_subscriber.handle_message"):
 120                     self.handle_message(message)
 121 
 122                 # Track latest completed message here, for use in `shutdown` handler.
 123                 self.offsets[message.partition()] = message.offset() + 1
 124 
 125                 if i % self.commit_batch_size == 0:
 126                     logger.debug("Committing offsets")
 127                     self.commit_offsets()
 128         except KeyboardInterrupt:
 129             pass
 130 
 131         self.shutdown()
 132 
 133     def commit_offsets(self, partitions=None):
 134         if self.offsets and self.consumer:
 135             if partitions is None:
 136                 partitions = self.offsets.keys()
 137             to_commit = []
 138             for partition in partitions:
 139                 offset = self.offsets.get(partition)
 140                 if offset is None:
 141                     # Skip partitions that have no offset
 142                     continue
 143                 to_commit.append(TopicPartition(self.topic, partition, offset))
 144 
 145             self.consumer.commit(offsets=to_commit)
 146 
 147     def shutdown(self):
 148         logger.debug("Committing offsets and closing consumer")
 149         self.commit_offsets()
 150         self.consumer.close()

 152     def handle_message(self, message):
 153         """
 154         Parses the value from Kafka, and if valid passes the payload to the callback defined by the
 155         subscription. If the subscription has been removed, or no longer has a valid callback then
 156         just log metrics/errors and continue.
 157         :param message:
 158         :return:
 159         """
 160         with sentry_sdk.push_scope() as scope:
 161             try:
 162                 with metrics.timer("snuba_query_subscriber.parse_message_value"):
 163                     contents = self.parse_message_value(message.value())
 164             except InvalidMessageError:
 165                 # If the message is in an invalid format, just log the error
 166                 # and continue
 167                 logger.exception(
 168                     "Subscription update could not be parsed",
 169                     extra={
 170                         "offset": message.offset(),
 171                         "partition": message.partition(),
 172                         "value": message.value(),
 173                     },
 174                 )
 175                 return
 176             scope.set_tag("query_subscription_id", contents["subscription_id"])
 177 
 178             try:
 179                 with metrics.timer("snuba_query_subscriber.fetch_subscription"):
 180                     subscription = QuerySubscription.objects.get_from_cache(
 181                         subscription_id=contents["subscription_id"]
 182                     )
 183                     if subscription.status != QuerySubscription.Status.ACTIVE.value:
 184                         metrics.incr("snuba_query_subscriber.subscription_inactive")
 185                         return
 186             except QuerySubscription.DoesNotExist:
 187                 metrics.incr("snuba_query_subscriber.subscription_doesnt_exist")
 188                 logger.error(
 189                     "Received subscription update, but subscription does not exist",
 190                     extra={
 191                         "offset": message.offset(),
 192                         "partition": message.partition(),
 193                         "value": message.value(),
 194                     },
 195                 )
 196                 try:
 197                     _delete_from_snuba(
 198                         self.topic_to_dataset[message.topic()], contents["subscription_id"]
 199                     )
 200                 except Exception:
 201                     logger.exception("Failed to delete unused subscription from snuba.")
 202                 return
 203 
 204             if subscription.type not in subscriber_registry:
 205                 metrics.incr("snuba_query_subscriber.subscription_type_not_registered")
 206                 logger.error(
 207                     "Received subscription update, but no subscription handler registered",
 208                     extra={
 209                         "offset": message.offset(),
 210                         "partition": message.partition(),
 211                         "value": message.value(),
 212                     },
 213                 )
 214                 return
 215 
 216             logger.info(
 217                 "query-subscription-consumer.handle_message",
 218                 extra={
 219                     "timestamp": contents["timestamp"],
 220                     "query_subscription_id": contents["subscription_id"],
 221                     "project_id": subscription.project_id,
 222                     "subscription_dataset": subscription.snuba_query.dataset,
 223                     "subscription_query": subscription.snuba_query.query,
 224                     "subscription_aggregation": subscription.snuba_query.aggregate,
 225                     "subscription_time_window": subscription.snuba_query.time_window,
 226                     "subscription_resolution": subscription.snuba_query.resolution,
 227                     "offset": message.offset(),
 228                     "partition": message.partition(),
 229                     "value": message.value(),
 230                 },
 231             )
 232 
 233             callback = subscriber_registry[subscription.type]
 234             with sentry_sdk.start_span(op="process_message") as span, metrics.timer(
 235                 "snuba_query_subscriber.callback.duration", instance=subscription.type
 236             ):
 237                 span.set_data("payload", contents)
 238                 callback(contents, subscription)

 240     def parse_message_value(self, value):
 241         """
 242         Parses the value received via the Kafka consumer and verifies that it
 243         matches the expected schema.
 244         :param value: A json formatted string
 245         :return: A dict with the parsed message
 246         """
 247         with metrics.timer("snuba_query_subscriber.parse_message_value.json_parse"):
 248             wrapper = loads(value)
 249 
 250         with metrics.timer("snuba_query_subscriber.parse_message_value.json_validate_wrapper"):
 251             try:
 252                 jsonschema.validate(wrapper, SUBSCRIPTION_WRAPPER_SCHEMA)
 253             except jsonschema.ValidationError:
 254                 metrics.incr("snuba_query_subscriber.message_wrapper_invalid")
 255                 raise InvalidSchemaError("Message wrapper does not match schema")
 256 
 257         schema_version = wrapper["version"]
 258         if schema_version not in SUBSCRIPTION_PAYLOAD_VERSIONS:
 259             metrics.incr("snuba_query_subscriber.message_wrapper_invalid_version")
 260             raise InvalidMessageError("Version specified in wrapper has no schema")
 261 
 262         payload = wrapper["payload"]
 263         with metrics.timer("snuba_query_subscriber.parse_message_value.json_validate_payload"):
 264             try:
 265                 jsonschema.validate(payload, SUBSCRIPTION_PAYLOAD_VERSIONS[schema_version])
 266             except jsonschema.ValidationError:
 267                 metrics.incr("snuba_query_subscriber.message_payload_invalid")
 268                 raise InvalidSchemaError("Message payload does not match schema")
 269         # XXX: Since we just return the raw dict here, when the payload changes it'll
 270         # break things. This should convert the payload into a class rather than passing
 271         # the dict around, but until we get time to refactor we can keep things working
 272         # here.
 273         payload.setdefault("values", payload.get("result"))
 274 
 275         payload["timestamp"] = parse_date(payload["timestamp"]).replace(tzinfo=pytz.utc)
 276         return payload
