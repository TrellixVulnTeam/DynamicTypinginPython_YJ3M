  64 def is_real_column(col):
  65     """
  66     Return true if col corresponds to an actual column to be fetched
  67     (not an aggregate function or field alias)
  68     """
  69     if is_function(col):
  70         return False
  71 
  72     if col in FIELD_ALIASES:
  73         return False
  74 
  75     return True

  78 def find_reference_event(reference_event):
  79     try:
  80         project_slug, event_id = reference_event.slug.split(":")
  81     except ValueError:
  82         raise InvalidSearchQuery("Invalid reference event format")
  83 
  84     column_names = [
  85         resolve_discover_column(col) for col in reference_event.fields if is_real_column(col)
  86     ]
  87     # We don't need to run a query if there are no columns
  88     if not column_names:
  89         return None
  90 
  91     try:
  92         project = Project.objects.get(
  93             slug=project_slug,
  94             organization=reference_event.organization,
  95             status=ProjectStatus.VISIBLE,
  96         )
  97     except Project.DoesNotExist:
  98         raise InvalidSearchQuery("Invalid reference event project")
  99 
 100     start = None
 101     end = None
 102     if reference_event.start:
 103         start = reference_event.start - timedelta(seconds=5)
 104     if reference_event.end:
 105         end = reference_event.end + timedelta(seconds=5)
 106 
 107     # We use raw_query here because generating conditions from an eventstore
 108     # event requires non-trivial translation from the flat list of fields into
 109     # structured fields like message, stack, and tags.
 110     event = raw_query(
 111         selected_columns=column_names,
 112         filter_keys={"project_id": [project.id], "event_id": [event_id]},
 113         start=start,
 114         end=end,
 115         dataset=Dataset.Discover,
 116         limit=1,
 117         referrer="discover.find_reference_event",
 118     )
 119     if "error" in event or len(event["data"]) != 1:
 120         raise InvalidSearchQuery("Unable to find reference event")
 121 
 122     return event["data"][0]

 125 def create_reference_event_conditions(reference_event):
 126     """
 127     Create a list of conditions based on a Reference object.
 128 
 129     This is useful when you want to get results that match an exemplar
 130     event. A use case of this is generating pagination links for, or getting
 131     timeseries results of the records inside a single aggregated row.
 132 
 133     reference_event (ReferenceEvent) The reference event to build conditions from.
 134     """
 135     conditions = []
 136     event_data = find_reference_event(reference_event)
 137     if event_data is None:
 138         return conditions
 139 
 140     field_names = [resolve_discover_column(col) for col in reference_event.fields]
 141     for (i, field) in enumerate(reference_event.fields):
 142         value = event_data.get(field_names[i], None)
 143         # If the value is a sequence use the first element as snuba
 144         # doesn't support `=` or `IN` operations on fields like exception_frames.filename
 145         if isinstance(value, (list, set)) and value:
 146             value = value.pop()
 147         if value:
 148             conditions.append([field, "=", value])
 149 
 150     return conditions

 156 def find_histogram_buckets(field, params, conditions):
 157     match = is_function(field)
 158     if not match:
 159         raise InvalidSearchQuery(u"received {}, expected histogram function".format(field))
 160 
 161     columns = [c.strip() for c in match.group("columns").split(",") if len(c.strip()) > 0]
 162 
 163     if len(columns) != 2:
 164         raise InvalidSearchQuery(
 165             u"histogram(...) expects 2 column arguments, received {:g} arguments".format(
 166                 len(columns)
 167             )
 168         )
 169 
 170     column = columns[0]
 171     # TODO evanh: This can be expanded to more fields at a later date, for now keep this limited.
 172     if column != "transaction.duration":
 173         raise InvalidSearchQuery(
 174             "histogram(...) can only be used with the transaction.duration column"
 175         )
 176 
 177     try:
 178         num_buckets = int(columns[1])
 179         if num_buckets < 1 or num_buckets > 500:
 180             raise Exception()
 181     except Exception:
 182         raise InvalidSearchQuery(
 183             u"histogram(...) requires a bucket value between 1 and 500, not {}".format(columns[1])
 184         )
 185 
 186     max_alias = u"max_{}".format(column)
 187     min_alias = u"min_{}".format(column)
 188 
 189     conditions = deepcopy(conditions) if conditions else []
 190     found = False
 191     for cond in conditions:
 192         if len(cond) == 3 and (cond[0], cond[1], cond[2]) == ("event.type", "=", "transaction"):
 193             found = True
 194             break
 195     if not found:
 196         conditions.append(["event.type", "=", "transaction"])
 197     snuba_filter = eventstore.Filter(conditions=conditions)
 198     translated_args, _ = resolve_discover_aliases(snuba_filter)
 199 
 200     results = raw_query(
 201         filter_keys={"project_id": params.get("project_id")},
 202         start=params.get("start"),
 203         end=params.get("end"),
 204         dataset=Dataset.Discover,
 205         conditions=translated_args.conditions,
 206         aggregations=[["max", "duration", max_alias], ["min", "duration", min_alias]],
 207     )
 208     if len(results["data"]) != 1:
 209         # If there are no transactions, so no max duration, return one empty bucket
 210         return "histogram({}, 1, 1, 0)".format(column)
 211 
 212     bucket_min = results["data"][0][min_alias]
 213     bucket_max = results["data"][0][max_alias]
 214     if bucket_max == 0:
 215         raise InvalidSearchQuery(u"Cannot calculate histogram for {}".format(field))
 216     bucket_size = ceil((bucket_max - bucket_min) / float(num_buckets))
 217     if bucket_size == 0.0:
 218         bucket_size = 1.0
 219 
 220     # Determine the first bucket that will show up in our results so that we can
 221     # zerofill correctly.
 222     offset = int(floor(bucket_min / bucket_size) * bucket_size)
 223 
 224     return "histogram({}, {:g}, {:.0f}, {:.0f})".format(column, num_buckets, bucket_size, offset)

 227 def zerofill_histogram(results, column_meta, orderby, sentry_function_alias, snuba_function_alias):
 228     parts = snuba_function_alias.split("_")
 229     if len(parts) < 2:
 230         raise Exception(u"{} is not a valid histogram alias".format(snuba_function_alias))
 231 
 232     bucket_offset, bucket_size, num_buckets = int(parts[-1]), int(parts[-2]), int(parts[-3])
 233     if len(results) == num_buckets:
 234         return results
 235 
 236     dummy_data = {}
 237     for column in column_meta:
 238         dummy_data[column["name"]] = "" if column.get("type") == "String" else 0
 239 
 240     def build_new_bucket_row(bucket):
 241         row = {key: value for key, value in six.iteritems(dummy_data)}
 242         row[sentry_function_alias] = bucket
 243         return row
 244 
 245     bucket_map = {r[sentry_function_alias]: r for r in results}
 246     new_results = []
 247     is_sorted, is_reversed = False, False
 248     if orderby:
 249         for o in orderby:
 250             if o.lstrip("-") == snuba_function_alias:
 251                 is_sorted = True
 252                 is_reversed = o.startswith("-")
 253                 break
 254 
 255     for i in range(num_buckets):
 256         bucket = bucket_offset + (bucket_size * i)
 257         if bucket not in bucket_map:
 258             bucket_map[bucket] = build_new_bucket_row(bucket)
 259 
 260     # If the list was sorted, pull results out in sorted order, else concat the results
 261     if is_sorted:
 262         i, diff, end = (0, 1, num_buckets) if not is_reversed else (num_buckets, -1, 0)
 263         while i <= end:
 264             bucket = bucket_offset + (bucket_size * i)
 265             if bucket in bucket_map:
 266                 new_results.append(bucket_map[bucket])
 267             i += diff
 268     else:
 269         new_results = results
 270         exists = set(r[sentry_function_alias] for r in results)
 271         for bucket in bucket_map:
 272             if bucket not in exists:
 273                 new_results.append(bucket_map[bucket])
 274 
 275     return new_results

 291 def zerofill(data, start, end, rollup, orderby):
 292     rv = []
 293     start = int(to_naive_timestamp(naiveify_datetime(start)) / rollup) * rollup
 294     end = (int(to_naive_timestamp(naiveify_datetime(end)) / rollup) * rollup) + rollup
 295     data_by_time = {}
 296 
 297     for obj in data:
 298         if obj["time"] in data_by_time:
 299             data_by_time[obj["time"]].append(obj)
 300         else:
 301             data_by_time[obj["time"]] = [obj]
 302 
 303     for key in six.moves.xrange(start, end, rollup):
 304         if key in data_by_time and len(data_by_time[key]) > 0:
 305             rv = rv + data_by_time[key]
 306             data_by_time[key] = []
 307         else:
 308             rv.append({"time": key})
 309 
 310     if "-time" in orderby:
 311         return list(reversed(rv))
 312 
 313     return rv

 316 def transform_results(result, translated_columns, snuba_filter, selected_columns=None):
 317     """
 318     Transform internal names back to the public schema ones.
 319 
 320     When getting timeseries results via rollup, this function will
 321     zerofill the output results.
 322     """
 323     if selected_columns is None:
 324         selected_columns = []
 325 
 326     # Determine user related fields to prune based on what wasn't selected.
 327     user_fields = FIELD_ALIASES["user"]["fields"]
 328     user_fields_to_remove = [field for field in user_fields if field not in selected_columns]
 329 
 330     # If the user field was selected update the meta data
 331     has_user = selected_columns and "user" in selected_columns
 332     meta = []
 333     for col in result["meta"]:
 334         # Translate back column names that were converted to snuba format
 335         col["name"] = translated_columns.get(col["name"], col["name"])
 336         # Remove user fields as they will be replaced by the alias.
 337         if has_user and col["name"] in user_fields_to_remove:
 338             continue
 339         meta.append(col)
 340     if has_user:
 341         meta.append({"name": "user", "type": "Nullable(String)"})
 342     result["meta"] = meta
 343 
 344     def get_row(row):
 345         transformed = {}
 346         for key, value in row.items():
 347             if isinstance(value, float) and math.isnan(value):
 348                 value = 0
 349             transformed[translated_columns.get(key, key)] = value
 350 
 351         if has_user:
 352             for field in user_fields:
 353                 if field in transformed and transformed[field]:
 354                     transformed["user"] = transformed[field]
 355                     break
 356             # Remove user component fields once the alias is resolved.
 357             for field in user_fields_to_remove:
 358                 del transformed[field]
 359         return transformed
 360 
 361     if len(translated_columns):
 362         result["data"] = [get_row(row) for row in result["data"]]
 363 
 364     rollup = snuba_filter.rollup
 365     if rollup and rollup > 0:
 366         with sentry_sdk.start_span(
 367             op="discover.discover", description="transform_results.zerofill"
 368         ) as span:
 369             span.set_data("result_count", len(result.get("data", [])))
 370             result["data"] = zerofill(
 371                 result["data"], snuba_filter.start, snuba_filter.end, rollup, snuba_filter.orderby
 372             )
 373 
 374     for col in result["meta"]:
 375         if col["name"].startswith("histogram"):
 376             # The column name here has been translated, we need the original name
 377             for snuba_name, sentry_name in six.iteritems(translated_columns):
 378                 if sentry_name == col["name"]:
 379                     with sentry_sdk.start_span(
 380                         op="discover.discover", description="transform_results.histogram_zerofill",
 381                     ) as span:
 382                         span.set_data("histogram_function", snuba_name)
 383                         result["data"] = zerofill_histogram(
 384                             result["data"],
 385                             result["meta"],
 386                             snuba_filter.orderby,
 387                             sentry_name,
 388                             snuba_name,
 389                         )
 390             break
 391 
 392     return result

 410 def transform_deprecated_functions_in_columns(columns):
 411     new_list = []
 412     translations = {}
 413     for column in columns:
 414         if column in OLD_FUNCTIONS_TO_NEW:
 415             new_column = OLD_FUNCTIONS_TO_NEW[column]
 416             translations[get_function_alias(new_column)] = column
 417             new_list.append(new_column)
 418         elif column.replace("()", "") in OLD_FUNCTIONS_TO_NEW:
 419             new_column = OLD_FUNCTIONS_TO_NEW[column.replace("()", "")]
 420             translations[get_function_alias(new_column)] = column.replace("()", "")
 421             new_list.append(new_column)
 422         else:
 423             new_list.append(column)
 424 
 425     return new_list, translations

 428 def transform_deprecated_functions_in_query(query):
 429     if query is None:
 430         return query
 431 
 432     for old_function in OLD_FUNCTIONS_TO_NEW:
 433         if old_function + "()" in query:
 434             replacement = OLD_FUNCTIONS_TO_NEW[old_function]
 435             query = query.replace(old_function + "()", replacement)
 436         elif old_function + ":" in query:
 437             replacement = OLD_FUNCTIONS_TO_NEW[old_function]
 438             query = query.replace(old_function, replacement)
 439 
 440     return query

 443 def query(
 444     selected_columns,
 445     query,
 446     params,
 447     orderby=None,
 448     offset=None,
 449     limit=50,
 450     reference_event=None,
 451     referrer=None,
 452     auto_fields=False,
 453     use_aggregate_conditions=False,
 454     conditions=None,
 455 ):
 456     """
 457     High-level API for doing arbitrary user queries against events.
 458 
 459     This function operates on the Discover public event schema and
 460     virtual fields/aggregate functions for selected columns and
 461     conditions are supported through this function.
 462 
 463     The resulting list will have all internal field names mapped
 464     back into their public schema names.
 465 
 466     selected_columns (Sequence[str]) List of public aliases to fetch.
 467     query (str) Filter query string to create conditions from.
 468     params (Dict[str, str]) Filtering parameters with start, end, project_id, environment
 469     orderby (None|str|Sequence[str]) The field to order results by.
 470     offset (None|int) The record offset to read.
 471     limit (int) The number of records to fetch.
 472     reference_event (ReferenceEvent) A reference event object. Used to generate additional
 473                     conditions based on the provided reference.
 474     referrer (str|None) A referrer string to help locate the origin of this query.
 475     auto_fields (bool) Set to true to have project + eventid fields automatically added.
 476     conditions (Sequence[any]) List of conditions that are passed directly to snuba without
 477                     any additional processing.
 478     """
 479     if not selected_columns:
 480         raise InvalidSearchQuery("No columns selected")
 481 
 482     with sentry_sdk.start_span(
 483         op="discover.discover", description="query.filter_transform"
 484     ) as span:
 485         span.set_data("query", query)
 486         # TODO(evanh): These can be removed once we migrate the frontend / saved queries
 487         # to use the new function values
 488         selected_columns, function_translations = transform_deprecated_functions_in_columns(
 489             selected_columns
 490         )
 491         query = transform_deprecated_functions_in_query(query)
 492 
 493         snuba_filter = get_filter(query, params)
 494         if not use_aggregate_conditions:
 495             snuba_filter.having = []
 496 
 497     # We need to run a separate query to be able to properly bucket the values for the histogram
 498     # Do that here, and format the bucket number in to the columns before passing it through
 499     # to event search.
 500     idx = 0
 501     for col in selected_columns:
 502         if col.startswith("histogram("):
 503             with sentry_sdk.start_span(
 504                 op="discover.discover", description="query.histogram_calculation"
 505             ) as span:
 506                 span.set_data("histogram", col)
 507                 histogram_column = find_histogram_buckets(col, params, snuba_filter.conditions)
 508                 selected_columns[idx] = histogram_column
 509                 function_translations[get_function_alias(histogram_column)] = get_function_alias(
 510                     col
 511                 )
 512 
 513             break
 514 
 515         idx += 1
 516 
 517     with sentry_sdk.start_span(op="discover.discover", description="query.field_translations"):
 518         # Check to see if we are ordering by any functions and convert the orderby to be the correct alias.
 519         if orderby:
 520             orderby = orderby if isinstance(orderby, (list, tuple)) else [orderby]
 521             new_orderby = []
 522             for ordering in orderby:
 523                 is_reversed = ordering.startswith("-")
 524                 ordering = ordering.lstrip("-")
 525                 for snuba_name, sentry_name in six.iteritems(function_translations):
 526                     if sentry_name == ordering:
 527                         ordering = snuba_name
 528                         break
 529 
 530                 ordering = "{}{}".format("-" if is_reversed else "", ordering)
 531                 new_orderby.append(ordering)
 532 
 533             snuba_filter.orderby = new_orderby
 534 
 535         snuba_filter.update_with(
 536             resolve_field_list(selected_columns, snuba_filter, auto_fields=auto_fields)
 537         )
 538 
 539         if reference_event:
 540             ref_conditions = create_reference_event_conditions(reference_event)
 541             if ref_conditions:
 542                 snuba_filter.conditions.extend(ref_conditions)
 543 
 544         # Resolve the public aliases into the discover dataset names.
 545         snuba_filter, translated_columns = resolve_discover_aliases(
 546             snuba_filter, function_translations
 547         )
 548 
 549         # Make sure that any aggregate conditions are also in the selected columns
 550         for having_clause in snuba_filter.having:
 551             found = any(
 552                 having_clause[0] == agg_clause[-1] for agg_clause in snuba_filter.aggregations
 553             )
 554             if not found:
 555                 raise InvalidSearchQuery(
 556                     u"Aggregate {} used in a condition but is not a selected column.".format(
 557                         having_clause[0]
 558                     )
 559                 )
 560 
 561         if conditions is not None:
 562             snuba_filter.conditions.extend(conditions)
 563 
 564     with sentry_sdk.start_span(op="discover.discover", description="query.snuba_query"):
 565         result = raw_query(
 566             start=snuba_filter.start,
 567             end=snuba_filter.end,
 568             groupby=snuba_filter.groupby,
 569             conditions=snuba_filter.conditions,
 570             aggregations=snuba_filter.aggregations,
 571             selected_columns=snuba_filter.selected_columns,
 572             filter_keys=snuba_filter.filter_keys,
 573             having=snuba_filter.having,
 574             orderby=snuba_filter.orderby,
 575             dataset=Dataset.Discover,
 576             limit=limit,
 577             offset=offset,
 578             referrer=referrer,
 579         )
 580 
 581     with sentry_sdk.start_span(
 582         op="discover.discover", description="query.transform_results"
 583     ) as span:
 584         span.set_data("result_count", len(result.get("data", [])))
 585         return transform_results(result, translated_columns, snuba_filter, selected_columns)

 588 def key_transaction_conditions(queryset):
 589     """
 590         The snuba query for transactions is of the form
 591         (transaction="1" AND project=1) OR (transaction="2" and project=2) ...
 592         which the schema intentionally doesn't support so we cannot do an AND in OR
 593         so here the "and" operator is being instead to do an AND in OR query
 594     """
 595     return [
 596         [
 597             # First layer is Ands
 598             [
 599                 # Second layer is Ors
 600                 [
 601                     "and",
 602                     [
 603                         [
 604                             "equals",
 605                             # Without the outer ' here, the transaction will be treated as another column
 606                             # instead of a string. This isn't an injection risk since snuba is smart enough to
 607                             # handle escaping for us.
 608                             ["transaction", u"'{}'".format(transaction.transaction)],
 609                         ],
 610                         ["equals", ["project_id", transaction.project.id]],
 611                     ],
 612                 ],
 613                 "=",
 614                 1,
 615             ]
 616             for transaction in queryset
 617         ]
 618     ]
 619

 625         params,
 626         orderby=orderby,
 627         referrer=referrer,
 628         conditions=key_transaction_conditions(queryset),
 629     )

 632 def get_timeseries_snuba_filter(selected_columns, query, params, rollup, reference_event=None):
 633     # TODO(evanh): These can be removed once we migrate the frontend / saved queries
 634     # to use the new function values
 635     selected_columns, _ = transform_deprecated_functions_in_columns(selected_columns)
 636     query = transform_deprecated_functions_in_query(query)
 637 
 638     snuba_filter = get_filter(query, params)
 639     if not snuba_filter.start and not snuba_filter.end:
 640         raise InvalidSearchQuery("Cannot get timeseries result without a start and end.")
 641 
 642     snuba_filter.update_with(resolve_field_list(selected_columns, snuba_filter, auto_fields=False))
 643     if reference_event:
 644         ref_conditions = create_reference_event_conditions(reference_event)
 645         if ref_conditions:
 646             snuba_filter.conditions.extend(ref_conditions)
 647 
 648     # Resolve the public aliases into the discover dataset names.
 649     snuba_filter, translated_columns = resolve_discover_aliases(snuba_filter)
 650     if not snuba_filter.aggregations:
 651         raise InvalidSearchQuery("Cannot get timeseries result with no aggregation.")
 652

 654     # with other parts of the timeseries endpoint expectations

 657 
 658     return snuba_filter, translated_columns

 763 def create_result_key(result_row, fields, issues):
 764     values = []
 765     for field in fields:
 766         if field == "issue.id":
 767             values.append(issues.get(result_row["issue.id"], "unknown"))
 768         else:
 769             value = result_row.get(field)
 770             if isinstance(value, list):
 771                 if len(value) > 0:
 772                     value = value[-1]
 773                 else:
 774                     value = ""
 775             values.append(six.text_type(value))
 776     return ",".join(values)

 779 def top_events_timeseries(
 780     timeseries_columns,
 781     selected_columns,
 782     user_query,
 783     params,
 784     orderby,
 785     rollup,
 786     limit,
 787     organization,
 788     referrer=None,
 789 ):
 790     """
 791     High-level API for doing arbitrary user timeseries queries for a limited number of top events
 792 
 793     Returns a dictionary of SnubaTSResult objects that have been zerofilled in
 794     case of gaps. Each value of the dictionary should match the result of a timeseries query
 795 
 796     timeseries_columns (Sequence[str]) List of public aliases to fetch for the timeseries query,
 797                         usually matches the y-axis of the graph
 798     selected_columns (Sequence[str]) List of public aliases to fetch for the events query,
 799                         this is to determine what the top events are
 800     user_query (str) Filter query string to create conditions from. needs to be user_query
 801                         to not conflict with the function query
 802     params (Dict[str, str]) Filtering parameters with start, end, project_id, environment,
 803     orderby (Sequence[str]) The fields to order results by.
 804     rollup (int) The bucket width in seconds
 805     limit (int) The number of events to get timeseries for
 806     organization (Organization) Used to map group ids to short ids
 807     referrer (str|None) A referrer string to help locate the origin of this query.
 808     """
 809     with sentry_sdk.start_span(op="discover.discover", description="top_events.fetch_events"):
 810         top_events = query(
 811             selected_columns,
 812             query=user_query,
 813             params=params,
 814             orderby=orderby,
 815             limit=limit,
 816             referrer=referrer,
 817             use_aggregate_conditions=True,
 818         )
 819 
 820     with sentry_sdk.start_span(
 821         op="discover.discover", description="top_events.filter_transform"
 822     ) as span:
 823         span.set_data("query", user_query)
 824         snuba_filter, translated_columns = get_timeseries_snuba_filter(
 825             timeseries_columns + selected_columns, user_query, params, rollup
 826         )
 827 
 828         user_fields = FIELD_ALIASES["user"]["fields"]
 829         for field in selected_columns:
 830             # project is handled by filter_keys already
 831             if field in ["project", "project.id"]:
 832                 continue
 833             if field == "issue":
 834                 field = FIELD_ALIASES["issue"]["column_alias"]
 835             # Note that because orderby shouldn't be an array field its not included in the values
 836             values = list(
 837                 {
 838                     event.get(field)
 839                     for event in top_events["data"]
 840                     if field in event and not isinstance(event.get(field), list)
 841                 }
 842             )
 843             if values:
 844                 # timestamp needs special handling, creating a big OR instead
 845                 if field == "timestamp":
 846                     snuba_filter.conditions.append([["timestamp", "=", value] for value in values])
 847                 # A user field can be any of its field aliases, do an OR across all the user fields
 848                 elif field == "user":
 849                     snuba_filter.conditions.append(
 850                         [
 851                             [resolve_discover_column(user_field), "IN", values]
 852                             for user_field in user_fields
 853                         ]
 854                     )
 855                 elif None in values:
 856                     non_none_values = [value for value in values if value is not None]
 857                     condition = [[["isNull", [resolve_discover_column(field)]], "=", 1]]
 858                     if non_none_values:
 859                         condition.append([resolve_discover_column(field), "IN", non_none_values])
 860                     snuba_filter.conditions.append(condition)
 861                 else:
 862                     snuba_filter.conditions.append([resolve_discover_column(field), "IN", values])
 863 
 864     with sentry_sdk.start_span(op="discover.discover", description="top_events.snuba_query"):
 865         result = raw_query(
 866             aggregations=snuba_filter.aggregations,
 867             conditions=snuba_filter.conditions,
 868             filter_keys=snuba_filter.filter_keys,
 869             start=snuba_filter.start,
 870             end=snuba_filter.end,
 871             rollup=rollup,
 872             orderby="time",
 873             groupby=["time"] + snuba_filter.groupby,
 874             dataset=Dataset.Discover,
 875             limit=10000,
 876             referrer=referrer,
 877         )
 878 
 879     with sentry_sdk.start_span(
 880         op="discover.discover", description="top_events.transform_results"
 881     ) as span:
 882         span.set_data("result_count", len(result.get("data", [])))
 883         result = transform_results(result, translated_columns, snuba_filter, selected_columns)
 884 
 885         translated_columns["project_id"] = "project"
 886         translated_groupby = [
 887             translated_columns.get(groupby, groupby) for groupby in snuba_filter.groupby
 888         ]
 889 
 890         if "user" in selected_columns:
 891             # Determine user related fields to prune based on what wasn't selected, since transform_results does the same
 892             for field in user_fields:
 893                 if field not in selected_columns:
 894                     translated_groupby.remove(field)
 895             translated_groupby.append("user")
 896         issues = {}
 897         if "issue" in selected_columns:
 898             issues = Group.issues_mapping(
 899                 set([event["issue.id"] for event in top_events["data"]]),
 900                 params["project_id"],
 901                 organization,
 902             )
 903         # so the result key is consistent
 904         translated_groupby.sort()
 905 
 906         results = {}
 907         # Using the top events add the order to the results
 908         for index, item in enumerate(top_events["data"]):
 909             result_key = create_result_key(item, translated_groupby, issues)
 910             results[result_key] = {
 911                 "order": index,
 912                 "data": [],
 913             }
 914         for row in result["data"]:
 915             result_key = create_result_key(row, translated_groupby, issues)
 916             if result_key in results:
 917                 results[result_key]["data"].append(row)
 918             else:
 919                 logger.warning(
 920                     "discover.top-events.timeseries.key-mismatch",
 921                     extra={"result_key": result_key, "top_event_keys": results.keys()},
 922                 )
 923         for key, item in six.iteritems(results):
 924             results[key] = SnubaTSResult(
 925                 {
 926                     "data": zerofill(
 927                         item["data"], snuba_filter.start, snuba_filter.end, rollup, "time"
 928                     ),
 929                     "order": item["order"],
 930                 },
 931                 snuba_filter.start,
 932                 snuba_filter.end,
 933                 rollup,
 934             )
 935 
 936     return results

 944 def get_pagination_ids(event, query, params, organization, reference_event=None, referrer=None):
 945     """
 946     High-level API for getting pagination data for an event + filter
 947 
 948     The provided event is used as a reference event to find events
 949     that are older and newer than the current one.
 950 
 951     event (Event) The event to find related events for.
 952     query (str) Filter query string to create conditions from.
 953     params (Dict[str, str]) Filtering parameters with start, end, project_id, environment,
 954     reference_event (ReferenceEvent) A reference event object. Used to generate additional
 955                                     conditions based on the provided reference.
 956     referrer (str|None) A referrer string to help locate the origin of this query.
 957     """
 958     # TODO(evanh): This can be removed once we migrate the frontend / saved queries
 959     # to use the new function values
 960     query = transform_deprecated_functions_in_query(query)
 961 
 962     snuba_filter = get_filter(query, params)
 963 
 964     if reference_event:
 965         ref_conditions = create_reference_event_conditions(reference_event)
 966         if ref_conditions:
 967             snuba_filter.conditions.extend(ref_conditions)
 968 
 969     result = {
 970         "next": eventstore.get_next_event_id(event, filter=snuba_filter),
 971         "previous": eventstore.get_prev_event_id(event, filter=snuba_filter),
 972         "latest": eventstore.get_latest_event_id(event, filter=snuba_filter),
 973         "oldest": eventstore.get_earliest_event_id(event, filter=snuba_filter),
 974     }
 975 
 976     # translate project ids to slugs
 977 
 978     project_ids = set([tuple[0] for tuple in result.values() if tuple])
 979 
 980     project_slugs = {}
 981     projects = Project.objects.filter(
 982         id__in=list(project_ids), organization=organization, status=ProjectStatus.VISIBLE
 983     ).values("id", "slug")
 984 
 985     for project in projects:
 986         project_slugs[project["id"]] = project["slug"]
 987 
 988     def into_pagination_record(project_slug_event_id):
 989 
 990         if not project_slug_event_id:
 991             return None
 992 
 993         project_id = int(project_slug_event_id[0])
 994 
 995         return "{}:{}".format(project_slugs[project_id], project_slug_event_id[1])
 996 
 997     for key, value in result.items():
 998         result[key] = into_pagination_record(value)
 999 
1000     return PaginationResult(**result)

1003 def get_facets(query, params, limit=10, referrer=None):
1004     """
1005     High-level API for getting 'facet map' results.
1006 
1007     Facets are high frequency tags and attribute results that
1008     can be used to further refine user queries. When many projects
1009     are requested sampling will be enabled to help keep response times low.
1010 
1011     query (str) Filter query string to create conditions from.
1012     params (Dict[str, str]) Filtering parameters with start, end, project_id, environment
1013     limit (int) The number of records to fetch.
1014     referrer (str|None) A referrer string to help locate the origin of this query.
1015 
1016     Returns Sequence[FacetResult]
1017     """
1018     with sentry_sdk.start_span(
1019         op="discover.discover", description="facets.filter_transform"
1020     ) as span:
1021         span.set_data("query", query)
1022         # TODO(evanh): This can be removed once we migrate the frontend / saved queries
1023         # to use the new function values
1024         query = transform_deprecated_functions_in_query(query)
1025 
1026         snuba_filter = get_filter(query, params)
1027 
1028         # Resolve the public aliases into the discover dataset names.
1029         snuba_filter, translated_columns = resolve_discover_aliases(snuba_filter)
1030 
1031     # Exclude tracing tags as they are noisy and generally not helpful.
1032     excluded_tags = ["tags_key", "NOT IN", ["trace", "trace.ctx", "trace.span", "project"]]
1033 
1034     # Sampling keys for multi-project results as we don't need accuracy
1035     # with that much data.
1036     sample = len(snuba_filter.filter_keys["project_id"]) > 2
1037 
1038     with sentry_sdk.start_span(op="discover.discover", description="facets.frequent_tags"):
1039         # Get the most frequent tag keys
1040         key_names = raw_query(
1041             aggregations=[["count", None, "count"]],
1042             start=snuba_filter.start,
1043             end=snuba_filter.end,
1044             conditions=snuba_filter.conditions,
1045             filter_keys=snuba_filter.filter_keys,
1046             orderby=["-count", "tags_key"],
1047             groupby="tags_key",
1048             having=[excluded_tags],
1049             dataset=Dataset.Discover,
1050             limit=limit,
1051             referrer=referrer,
1052             turbo=sample,
1053         )
1054         top_tags = [r["tags_key"] for r in key_names["data"]]
1055         if not top_tags:
1056             return []
1057 
1058     # TODO(mark) Make the sampling rate scale based on the result size and scaling factor in
1059     # sentry.options. To test the lowest acceptable sampling rate, we use 0.1 which
1060     # is equivalent to turbo. We don't use turbo though as we need to re-scale data, and
1061     # using turbo could cause results to be wrong if the value of turbo is changed in snuba.
1062     sample_rate = 0.1 if key_names["data"][0]["count"] > 10000 else None
1063     # Rescale the results if we're sampling
1064     multiplier = 1 / sample_rate if sample_rate is not None else 1
1065 
1066     fetch_projects = False
1067     if len(params.get("project_id", [])) > 1:
1068         if len(top_tags) == limit:
1069             top_tags.pop()
1070         fetch_projects = True
1071 
1072     results = []
1073     if fetch_projects:
1074         with sentry_sdk.start_span(op="discover.discover", description="facets.projects"):
1075             project_values = raw_query(
1076                 aggregations=[["count", None, "count"]],
1077                 start=snuba_filter.start,
1078                 end=snuba_filter.end,
1079                 conditions=snuba_filter.conditions,
1080                 filter_keys=snuba_filter.filter_keys,
1081                 groupby="project_id",
1082                 orderby="-count",
1083                 dataset=Dataset.Discover,
1084                 referrer=referrer,
1085                 sample=sample_rate,
1086                 # Ensures Snuba will not apply FINAL
1087                 turbo=sample_rate is not None,
1088             )
1089             results.extend(
1090                 [
1091                     FacetResult("project", r["project_id"], int(r["count"]) * multiplier)
1092                     for r in project_values["data"]
1093                 ]
1094             )
1095 
1096     # Get tag counts for our top tags. Fetching them individually
1097     # allows snuba to leverage promoted tags better and enables us to get
1098     # the value count we want.
1099     max_aggregate_tags = options.get("discover2.max_tags_to_combine")
1100     individual_tags = []
1101     aggregate_tags = []
1102     for i, tag in enumerate(top_tags):
1103         if tag == "environment":
1104             # Add here tags that you want to be individual
1105             individual_tags.append(tag)
1106         elif i >= len(top_tags) - max_aggregate_tags:
1107             aggregate_tags.append(tag)
1108         else:
1109             individual_tags.append(tag)
1110 
1111     with sentry_sdk.start_span(
1112         op="discover.discover", description="facets.individual_tags"
1113     ) as span:
1114         span.set_data("tag_count", len(individual_tags))
1115         for tag_name in individual_tags:
1116             tag = u"tags[{}]".format(tag_name)
1117             tag_values = raw_query(
1118                 aggregations=[["count", None, "count"]],
1119                 conditions=snuba_filter.conditions,
1120                 start=snuba_filter.start,
1121                 end=snuba_filter.end,
1122                 filter_keys=snuba_filter.filter_keys,
1123                 orderby=["-count"],
1124                 groupby=[tag],
1125                 limit=TOP_VALUES_DEFAULT_LIMIT,
1126                 dataset=Dataset.Discover,
1127                 referrer=referrer,
1128                 sample=sample_rate,
1129                 # Ensures Snuba will not apply FINAL
1130                 turbo=sample_rate is not None,
1131             )
1132             results.extend(
1133                 [
1134                     FacetResult(tag_name, r[tag], int(r["count"]) * multiplier)
1135                     for r in tag_values["data"]
1136                 ]
1137             )
1138 
1139     if aggregate_tags:
1140         with sentry_sdk.start_span(op="discover.discover", description="facets.aggregate_tags"):
1141             conditions = snuba_filter.conditions
1142             conditions.append(["tags_key", "IN", aggregate_tags])
1143             tag_values = raw_query(
1144                 aggregations=[["count", None, "count"]],
1145                 conditions=conditions,
1146                 start=snuba_filter.start,
1147                 end=snuba_filter.end,
1148                 filter_keys=snuba_filter.filter_keys,
1149                 orderby=["tags_key", "-count"],
1150                 groupby=["tags_key", "tags_value"],
1151                 dataset=Dataset.Discover,
1152                 referrer=referrer,
1153                 sample=sample_rate,
1154                 # Ensures Snuba will not apply FINAL
1155                 turbo=sample_rate is not None,
1156                 limitby=[TOP_VALUES_DEFAULT_LIMIT, "tags_key"],
1157             )
1158             results.extend(
1159                 [
1160                     FacetResult(r["tags_key"], r["tags_value"], int(r["count"]) * multiplier)
1161                     for r in tag_values["data"]
1162                 ]
1163             )
1164 
1165     return results
