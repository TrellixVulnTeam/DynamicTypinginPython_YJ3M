  59     )["data"]
  60     rv = {}
  61     for row in rows:

  82 def get_project_releases_by_stability(
  83     project_ids, offset, limit, scope, stats_period=None, environments=None
  84 ):
  85     """Given some project IDs returns adoption rates that should be updated
  86     on the postgres tables.
  87     """
  88     if stats_period is None:
  89         stats_period = "24h"
  90 
  91     # Special rule that we support sorting by the last 24h only.
  92     if scope.endswith("_24h"):
  93         scope = scope[:-4]
  94         stats_period = "24h"
  95 
  96     _, stats_start, _ = get_rollup_starts_and_buckets(stats_period)
  97 
  98     orderby = {
  99         "crash_free_sessions": [["divide", ["sessions_crashed", "sessions"]]],
 100         "crash_free_users": [["divide", ["users_crashed", "users"]]],
 101         "sessions": ["-sessions"],
 102         "users": ["-users"],
 103     }[scope]
 104 
 105     conditions = []
 106     if environments is not None:
 107         conditions.append(["environment", "IN", environments])
 108     filter_keys = {"project_id": project_ids}
 109     rv = []
 110 
 111     for x in raw_query(
 112         dataset=Dataset.Sessions,
 113         selected_columns=["project_id", "release"],
 114         groupby=["release", "project_id"],
 115         orderby=orderby,
 116         start=stats_start,
 117         offset=offset,
 118         limit=limit,
 119         conditions=conditions,
 120         filter_keys=filter_keys,
 121     )["data"]:
 122         rv.append((x["project_id"], x["release"]))
 123

 127 def _make_stats(start, rollup, buckets, default=0):
 128     rv = []
 129     start = int(to_timestamp(start) // rollup + 1) * rollup
 130     for x in range(buckets):
 131         rv.append([start, default])
 132         start += rollup
 133     return rv

 136 STATS_PERIODS = {
 137     "1h": (3600, 1),
 138     "24h": (3600, 24),
 139     "1d": (3600, 24),
 140     "48h": (3600, 48),
 141     "2d": (3600, 48),
 142     "7d": (86400, 7),
 143     "14d": (86400, 14),
 144     "30d": (86400, 30),
 145     "90d": (259200, 30),
 146 }

 149 def get_rollup_starts_and_buckets(period):
 150     if period is None:
 151         return None, None, None
 152     if period not in STATS_PERIODS:
 153         raise TypeError("Invalid stats period")
 154     seconds, buckets = STATS_PERIODS[period]
 155     start = datetime.now(pytz.utc) - timedelta(seconds=seconds * buckets)
 156     return seconds, start, buckets

 159 def get_release_adoption(project_releases, environments=None, now=None):
 160     """Get the adoption of the last 24 hours (or a difference reference timestamp)."""
 161     conditions, filter_keys = _get_conditions_and_filter_keys(project_releases, environments)
 162     if now is None:
 163         now = datetime.now(pytz.utc)
 164     start = now - timedelta(days=1)
 165 
 166     total_conditions = []
 167     if environments is not None:
 168         total_conditions.append(["environment", "IN", environments])
 169 
 170     total_users = {}
 171     for x in raw_query(
 172         dataset=Dataset.Sessions,
 173         selected_columns=["project_id", "users"],
 174         groupby=["project_id"],
 175         start=start,
 176         conditions=total_conditions,
 177         filter_keys=filter_keys,
 178     )["data"]:
 179         total_users[x["project_id"]] = x["users"]
 180 
 181     rv = {}
 182     for x in raw_query(
 183         dataset=Dataset.Sessions,
 184         selected_columns=["release", "project_id", "users", "sessions"],
 185         groupby=["release", "project_id"],
 186         start=start,
 187         conditions=conditions,
 188         filter_keys=filter_keys,
 189     )["data"]:
 190         total = total_users.get(x["project_id"])
 191         if not total:
 192             adoption = None
 193         else:
 194             adoption = float(x["users"]) / total * 100
 195         rv[x["project_id"], x["release"]] = {
 196             "adoption": adoption,
 197             "users_24h": x["users"],
 198             "sessions_24h": x["sessions"],
 199         }
 200 
 201     return rv

 204 def get_release_health_data_overview(
 205     project_releases,
 206     environments=None,
 207     summary_stats_period=None,
 208     health_stats_period=None,
 209     stat=None,
 210 ):
 211     """Checks quickly for which of the given project releases we have
 212     health data available.  The argument is a tuple of `(project_id, release_name)`
 213     tuples.  The return value is a set of all the project releases that have health
 214     data.
 215     """
 216     if stat is None:
 217         stat = "sessions"
 218     assert stat in ("sessions", "users")
 219 
 220     _, summary_start, _ = get_rollup_starts_and_buckets(summary_stats_period or "24h")
 221     conditions, filter_keys = _get_conditions_and_filter_keys(project_releases, environments)
 222 
 223     stats_rollup, stats_start, stats_buckets = get_rollup_starts_and_buckets(health_stats_period)
 224 
 225     missing_releases = set(project_releases)
 226     rv = {}
 227     for x in raw_query(
 228         dataset=Dataset.Sessions,
 229         selected_columns=[
 230             "release",
 231             "project_id",
 232             "duration_quantiles",
 233             "users",
 234             "sessions",
 235             "sessions_errored",
 236             "sessions_crashed",
 237             "sessions_abnormal",
 238             "users_crashed",
 239         ],
 240         groupby=["release", "project_id"],
 241         start=summary_start,
 242         conditions=conditions,
 243         filter_keys=filter_keys,
 244     )["data"]:
 245         rp = {
 246             "duration_p50": _convert_duration(x["duration_quantiles"][0]),
 247             "duration_p90": _convert_duration(x["duration_quantiles"][1]),
 248             "crash_free_users": (
 249                 100 - x["users_crashed"] / float(x["users"]) * 100 if x["users"] else None
 250             ),
 251             "crash_free_sessions": (
 252                 100 - x["sessions_crashed"] / float(x["sessions"]) * 100 if x["sessions"] else None
 253             ),
 254             "total_users": x["users"],
 255             "total_sessions": x["sessions"],
 256             "sessions_crashed": x["sessions_crashed"],
 257             "sessions_errored": max(
 258                 0, x["sessions_errored"] - x["sessions_crashed"] - x["sessions_abnormal"]
 259             ),
 260             "has_health_data": True,
 261         }
 262         if health_stats_period:
 263             rp["stats"] = {
 264                 health_stats_period: _make_stats(stats_start, stats_rollup, stats_buckets)
 265             }
 266         rv[x["project_id"], x["release"]] = rp
 267         missing_releases.discard((x["project_id"], x["release"]))
 268 
 269     # Add releases without data points
 270     if missing_releases:
 271         # If we're already looking at a 90 day horizont we don't need to
 272         # fire another query, we can already assume there is no data.
 273         if summary_stats_period != "90d":
 274             has_health_data = check_has_health_data(missing_releases)
 275         else:
 276             has_health_data = ()
 277         for key in missing_releases:
 278             rv[key] = {
 279                 "duration_p50": None,
 280                 "duration_p90": None,
 281                 "crash_free_users": None,
 282                 "crash_free_sessions": None,
 283                 "total_users": 0,
 284                 "total_sessions": 0,
 285                 "sessions_crashed": 0,
 286                 "sessions_errored": 0,
 287                 "has_health_data": key in has_health_data,
 288             }
 289             if health_stats_period:
 290                 rv[key]["stats"] = {
 291                     health_stats_period: _make_stats(stats_start, stats_rollup, stats_buckets)
 292                 }
 293 
 294     # Fill in release adoption
 295     release_adoption = get_release_adoption(project_releases, environments)
 296     for key in rv:
 297         adoption_info = release_adoption.get(key) or {}
 298         rv[key]["adoption"] = adoption_info.get("adoption")
 299         rv[key]["total_users_24h"] = adoption_info.get("users_24h")
 300         rv[key]["total_sessions_24h"] = adoption_info.get("sessions_24h")
 301 
 302     if health_stats_period:
 303         for x in raw_query(
 304             dataset=Dataset.Sessions,
 305             selected_columns=["release", "project_id", "bucketed_started", stat],
 306             groupby=["release", "project_id", "bucketed_started"],
 307             rollup=stats_rollup,
 308             start=stats_start,
 309             conditions=conditions,
 310             filter_keys=filter_keys,
 311         )["data"]:
 312             time_bucket = int(
 313                 (parse_snuba_datetime(x["bucketed_started"]) - stats_start).total_seconds()
 314                 / stats_rollup
 315             )
 316             rv[x["project_id"], x["release"]]["stats"][health_stats_period][time_bucket][1] = x[
 317                 stat
 318             ]
 319 
 320     return rv

 323 def get_crash_free_breakdown(project_id, release, start, environments=None):
 324     filter_keys = {"project_id": [project_id]}
 325     conditions = [["release", "=", release]]
 326     if environments is not None:
 327         conditions.append(["environment", "IN", environments])
 328 
 329     now = datetime.now(pytz.utc)
 330 
 331     def _query_stats(end):
 332         row = raw_query(
 333             dataset=Dataset.Sessions,
 334             selected_columns=["users", "users_crashed", "sessions", "sessions_crashed"],
 335             end=end,
 336             start=start,
 337             conditions=conditions,
 338             filter_keys=filter_keys,
 339         )["data"][0]
 340         return {
 341             "date": end,
 342             "total_users": row["users"],
 343             "crash_free_users": 100 - row["users_crashed"] / float(row["users"]) * 100
 344             if row["users"]
 345             else None,
 346             "total_sessions": row["sessions"],
 347             "crash_free_sessions": 100 - row["sessions_crashed"] / float(row["sessions"]) * 100
 348             if row["sessions"]
 349             else None,
 350         }
 351 
 352     last = None
 353     rv = []
 354     for offset in (
 355         timedelta(days=1),
 356         timedelta(days=2),
 357         timedelta(days=7),
 358         timedelta(days=14),
 359         timedelta(days=30),
 360     ):
 361         try:
 362             item_start = start + offset
 363             if item_start > now:
 364                 if last is None or (item_start - last).days > 1:
 365                     rv.append(_query_stats(now))
 366                 break
 367             rv.append(_query_stats(item_start))
 368             last = item_start
 369         except QueryOutsideRetentionError:
 370             # cannot query for these
 371             pass
 372 
 373     return rv

 376 def get_project_release_stats(project_id, release, stat, rollup, start, end, environments=None):
 377     assert stat in ("users", "sessions")
 378 
 379     # since snuba end queries are exclusive of the time and we're bucketing to
 380     # a full hour, we need to round to the next hour since snuba is exclusive
 381     # on the end.
 382     end = to_datetime((to_timestamp(end) // DATASET_BUCKET + 1) * DATASET_BUCKET)
 383 
 384     filter_keys = {"project_id": [project_id]}
 385     conditions = [["release", "=", release]]
 386     if environments is not None:
 387         conditions.append(["environment", "IN", environments])
 388 
 389     buckets = int((end - start).total_seconds() / rollup)
 390     stats = _make_stats(start, rollup, buckets, default=None)
 391 
 392     # Due to the nature of the probabilistic data structures some
 393     # subtractions can become negative.  As such we're making sure a number
 394     # never goes below zero to avoid confusion.
 395 
 396     totals = {
 397         stat: 0,
 398         stat + "_healthy": 0,
 399         stat + "_crashed": 0,
 400         stat + "_abnormal": 0,
 401         stat + "_errored": 0,
 402     }
 403 
 404     for rv in raw_query(
 405         dataset=Dataset.Sessions,
 406         selected_columns=[
 407             "bucketed_started",
 408             stat,
 409             stat + "_crashed",
 410             stat + "_abnormal",
 411             stat + "_errored",
 412             "duration_quantiles",
 413         ],
 414         groupby=["bucketed_started"],
 415         start=start,
 416         end=end,
 417         rollup=rollup,
 418         conditions=conditions,
 419         filter_keys=filter_keys,
 420     )["data"]:
 421         ts = parse_snuba_datetime(rv["bucketed_started"])
 422         bucket = int((ts - start).total_seconds() / rollup)
 423         stats[bucket][1] = {
 424             stat: rv[stat],
 425             stat + "_healthy": max(0, rv[stat] - rv[stat + "_errored"]),
 426             stat + "_crashed": rv[stat + "_crashed"],
 427             stat + "_abnormal": rv[stat + "_abnormal"],
 428             stat
 429             + "_errored": max(
 430                 0, rv[stat + "_errored"] - rv[stat + "_crashed"] - rv[stat + "_abnormal"]
 431             ),
 432             "duration_p50": _convert_duration(rv["duration_quantiles"][0]),
 433             "duration_p90": _convert_duration(rv["duration_quantiles"][1]),
 434         }
 435 
 436         # Session stats we can sum up directly without another query
 437         # as the data becomes available.
 438         if stat == "sessions":
 439             for k in totals:
 440                 totals[k] += stats[bucket][1][k]
 441 
 442     for idx, bucket in enumerate(stats):
 443         if bucket[1] is None:
 444             stats[idx][1] = {
 445                 stat: 0,
 446                 stat + "_healthy": 0,
 447                 stat + "_crashed": 0,
 448                 stat + "_abnormal": 0,
 449                 stat + "_errored": 0,
 450                 "duration_p50": None,
 451                 "duration_p90": None,
 452             }
 453 
 454     # For users we need a secondary query over the entire time range
 455     if stat == "users":
 456         rows = raw_query(
 457             dataset=Dataset.Sessions,
 458             selected_columns=["users", "users_crashed", "users_abnormal", "users_errored"],
 459             start=start,
 460             end=end,
 461             conditions=conditions,
 462             filter_keys=filter_keys,
 463         )["data"]
 464         if rows:
 465             rv = rows[0]
 466             totals = {
 467                 "users": rv["users"],
 468                 "users_healthy": max(0, rv["users"] - rv["users_errored"]),
 469                 "users_crashed": rv["users_crashed"],
 470                 "users_abnormal": rv["users_abnormal"],
 471                 "users_errored": max(
 472                     0, rv["users_errored"] - rv["users_crashed"] - rv["users_abnormal"]
 473                 ),
 474             }
 475 
 476     return stats, totals
