 111     def project_key_from_auth(self, auth):
 112         if not auth.public_key:
 113             raise APIUnauthorized("Invalid api key")
 114 
 115         # Make sure the key even looks valid first, since it's
 116         # possible to get some garbage input here causing further
 117         # issues trying to query it from cache or the database.
 118         if not ProjectKey.looks_like_api_key(auth.public_key):
 119             raise APIUnauthorized("Invalid api key")
 120 
 121         try:
 122             pk = ProjectKey.objects.get_from_cache(public_key=auth.public_key)
 123         except ProjectKey.DoesNotExist:
 124             raise APIUnauthorized("Invalid api key")
 125 
 126         # a secret key may not be present which will be validated elsewhere
 127         if not constant_time_compare(pk.secret_key, auth.secret_key or pk.secret_key):
 128             raise APIUnauthorized("Invalid api key")
 129 
 130         if not pk.is_active:
 131             raise APIUnauthorized("API key is disabled")
 132 
 133         if not pk.roles.store:
 134             raise APIUnauthorized("Key does not allow event storage access")
 135 
 136         return pk

 141     def insert_data_to_database(
 142         self, data, start_time=None, from_reprocessing=False, attachments=None
 143     ):
 144         if start_time is None:
 145             start_time = time()
 146 
 147         # we might be passed some subclasses of dict that fail dumping
 148         if isinstance(data, CANONICAL_TYPES):
 149             data = dict(data.items())
 150 
 151         cache_timeout = 3600
 152         cache_key = cache_key_for_event(data)
 153         default_cache.set(cache_key, data, cache_timeout)
 154 
 155         # Attachments will be empty or None if the "event-attachments" feature
 156         # is turned off. For native crash reports it will still contain the
 157         # crash dump (e.g. minidump) so we can load it during processing.
 158         if attachments is not None:
 159             attachment_cache.set(cache_key, attachments, cache_timeout)
 160 
 161         task = from_reprocessing and preprocess_event_from_reprocessing or preprocess_event
 162         task.delay(cache_key=cache_key, start_time=start_time, event_id=data["event_id"])

 176 class ClientAuthHelper(AbstractAuthHelper):
 177     @classmethod
 178     def auth_from_request(cls, request):
 179         result = {k: request.GET[k] for k in six.iterkeys(request.GET) if k[:7] == "sentry_"}
 180 
 181         if request.META.get("HTTP_X_SENTRY_AUTH", "")[:7].lower() == "sentry ":
 182             if result:
 183                 raise SuspiciousOperation("Multiple authentication payloads were detected.")
 184             result = parse_auth_header(request.META["HTTP_X_SENTRY_AUTH"])
 185         elif request.META.get("HTTP_AUTHORIZATION", "")[:7].lower() == "sentry ":
 186             if result:
 187                 raise SuspiciousOperation("Multiple authentication payloads were detected.")
 188             result = parse_auth_header(request.META["HTTP_AUTHORIZATION"])
 189 
 190         if not result:
 191             raise APIUnauthorized("Unable to find authentication information")
 192 
 193         origin = cls.origin_from_request(request)
 194         auth = Auth(
 195             client=result.get("sentry_client"),
 196             version=six.text_type(result.get("sentry_version")),
 197             secret_key=result.get("sentry_secret"),
 198             public_key=result.get("sentry_key"),
 199             is_public=bool(origin),
 200         )
 201         # default client to user agent
 202         if not auth.client:
 203             auth.client = request.META.get("HTTP_USER_AGENT")
 204             if isinstance(auth.client, bytes):
 205                 auth.client = auth.client.decode("latin1")
 206         return auth
 207 
 208     @classmethod
 209     def origin_from_request(cls, request):
 210         """
 211         Returns either the Origin or Referer value from the request headers.
 212         """
 213         if request.META.get("HTTP_ORIGIN") == "null":
 214             return "null"
 215         return origin_from_request(request)

 224     @classmethod
 225     def auth_from_request(cls, request):
 226         key = request.GET.get("sentry_key")
 227         if not key:
 228             raise APIUnauthorized("Unable to find authentication information")
 229 
 230         # Minidump requests are always "trusted".  We at this point only
 231         # use is_public to identify requests that have an origin set (via
 232         # CORS)
 233         auth = Auth(public_key=key, client="sentry-minidump", is_public=False)

 237 class SecurityAuthHelper(AbstractAuthHelper):
 238     @classmethod
 239     def origin_from_request(cls, request):
 240         # In the case of security reports, the origin is not available at the
 241         # dispatch() stage, as we need to parse it out of the request body, so
 242         # we do our own CORS check once we have parsed it.
 243         return None

 245     @classmethod
 246     def auth_from_request(cls, request):
 247         key = request.GET.get("sentry_key")
 248         if not key:
 249             raise APIUnauthorized("Unable to find authentication information")
 250 
 251         auth = Auth(public_key=key, is_public=True)
 252         auth.client = request.META.get("HTTP_USER_AGENT")
 253         return auth

 256 def decompress_deflate(encoded_data):
 257     try:
 258         return zlib.decompress(encoded_data).decode("utf-8")
 259     except Exception as e:
 260         # This error should be caught as it suggests that there's a
 261         # bug somewhere in the client's code.
 262         logger.debug(six.text_type(e), exc_info=True)
 263         raise APIError("Bad data decoding request (%s, %s)" % (type(e).__name__, e))

 266 def decompress_gzip(encoded_data):
 267     try:
 268         fp = BytesIO(encoded_data)
 269         try:
 270             f = GzipFile(fileobj=fp)
 271             return f.read().decode("utf-8")
 272         finally:
 273             f.close()
 274     except Exception as e:
 275         # This error should be caught as it suggests that there's a
 276         # bug somewhere in the client's code.
 277         logger.debug(six.text_type(e), exc_info=True)
 278         raise APIError("Bad data decoding request (%s, %s)" % (type(e).__name__, e))

 281 def decode_and_decompress_data(encoded_data):
 282     try:
 283         try:
 284             return decompress(encoded_data).decode("utf-8")
 285         except zlib.error:
 286             return base64.b64decode(encoded_data).decode("utf-8")
 287     except Exception as e:
 288         # This error should be caught as it suggests that there's a
 289         # bug somewhere in the client's code.
 290         logger.debug(six.text_type(e), exc_info=True)
 291         raise APIError("Bad data decoding request (%s, %s)" % (type(e).__name__, e))

 304 def safely_load_json_string(json_string):
 305     try:
 306         if isinstance(json_string, six.binary_type):
 307             json_string = json_string.decode("utf-8")
 308         obj = json.loads(json_string)
 309         assert isinstance(obj, dict)
 310     except Exception as e:
 311         # This error should be caught as it suggests that there's a
 312         # bug somewhere in the client's code.
 313         logger.debug(six.text_type(e), exc_info=True)
 314         raise APIError("Bad data reconstructing object (%s, %s)" % (type(e).__name__, e))
 315     return obj
