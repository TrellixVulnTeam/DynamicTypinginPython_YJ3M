  41 class IngestConsumerWorker(AbstractBatchWorker):
  42     def __init__(self, concurrency):
  43         self.pool = _multiprocessing.dummy.Pool(concurrency)
  44         atexit.register(self.pool.close)
  45 
  46     def process_message(self, message):
  47         message = msgpack.unpackb(message.value(), use_list=False)
  48         return message
  49 
  50     def flush_batch(self, batch):
  51         mark_scope_as_unsafe()
  52         with metrics.timer("ingest_consumer.flush_batch"):
  53             return self._flush_batch(batch)
  54 
  55     def _flush_batch(self, batch):
  56         attachment_chunks = []
  57         other_messages = []
  58         transactions = []
  59 
  60         projects_to_fetch = set()
  61 
  62         with metrics.timer("ingest_consumer.prepare_messages"):
  63             for message in batch:
  64                 message_type = message["type"]
  65                 projects_to_fetch.add(message["project_id"])
  66 
  67                 if message_type == "event":
  68                     other_messages.append((process_event, message))
  69                 elif message_type == "transaction":
  70                     transactions.append(message)
  71                 elif message_type == "attachment_chunk":
  72                     attachment_chunks.append(message)
  73                 elif message_type == "attachment":
  74                     other_messages.append((process_individual_attachment, message))
  75                 elif message_type == "user_report":
  76                     other_messages.append((process_userreport, message))
  77                 else:
  78                     raise ValueError("Unknown message type: {}".format(message_type))
  79                 metrics.incr(
  80                     "ingest_consumer.flush.messages_seen", tags={"message_type": message_type}
  81                 )
  82 
  83         with metrics.timer("ingest_consumer.fetch_projects"):
  84             projects = {p.id: p for p in Project.objects.get_many_from_cache(projects_to_fetch)}
  85 
  86         if attachment_chunks:
  87             # attachment_chunk messages need to be processed before attachment/event messages.
  88             with metrics.timer("ingest_consumer.process_attachment_chunk_batch"):
  89                 for _ in self.pool.imap_unordered(
  90                     lambda msg: process_attachment_chunk(msg, projects=projects),
  91                     attachment_chunks,
  92                     chunksize=100,
  93                 ):
  94                     pass
  95 
  96         if other_messages:
  97             with metrics.timer("ingest_consumer.process_other_messages_batch"):
  98                 for _ in self.pool.imap_unordered(
  99                     lambda args: args[0](args[1], projects=projects), other_messages, chunksize=100,
 100                 ):
 101                     pass
 102 
 103         if transactions:
 104             with metrics.timer("ingest_consumer.process_transactions"):
 105                 process_transactions_batch(transactions, projects)
 106 
 107     def shutdown(self):
 108         pass

 111 def trace_func(**span_kwargs):
 112     def wrapper(f):
 113         @functools.wraps(f)
 114         def inner(*args, **kwargs):
 115             span_kwargs["sampled"] = random.random() < getattr(
 116                 settings, "SENTRY_INGEST_CONSUMER_APM_SAMPLING", 0
 117             )
 118             with sentry_sdk.start_span(**span_kwargs):
 119                 return f(*args, **kwargs)
 120 
 121         return inner
 122 
 123     return wrapper

 128 def process_transactions_batch(messages, projects):
 129     if options.get("store.transactions-celery") is True:
 130         for message in messages:
 131             _do_process_event(message, projects)
 132         return
 133 
 134     jobs = []
 135     for message in messages:
 136         payload = message["payload"]
 137         project_id = int(message["project_id"])
 138         start_time = float(message["start_time"])
 139 
 140         if project_id not in projects:
 141             continue
 142 
 143         with metrics.timer("ingest_consumer.decode_transaction_json"):
 144             data = json.loads(payload)
 145         jobs.append({"data": data, "start_time": start_time})
 146 
 147     save_transaction_events(jobs, projects)

 150 @metrics.wraps("ingest_consumer.process_event")
 151 def _do_process_event(message, projects):
 152     payload = message["payload"]
 153     start_time = float(message["start_time"])
 154     event_id = message["event_id"]
 155     project_id = int(message["project_id"])
 156     remote_addr = message.get("remote_addr")
 157     attachments = message.get("attachments") or ()
 158 
 159     # check that we haven't already processed this event (a previous instance of the forwarder
 160     # died before it could commit the event queue offset)
 161     #
 162     # XXX(markus): I believe this code is extremely broken:
 163     #
 164     # * it practically uses memcached in prod which has no consistency
 165     #   guarantees (no idea how we don't run into issues there)
 166     #
 167     # * a TTL of 1h basically doesn't guarantee any deduplication at all. It
 168     #   just guarantees a good error message... for one hour.
 169     #
 170     # This code has been ripped from the old python store endpoint. We're
 171     # keeping it around because it does provide some protection against
 172     # reprocessing good events if a single consumer is in a restart loop.
 173     deduplication_key = "ev:{}:{}".format(project_id, event_id)
 174     if cache.get(deduplication_key) is not None:
 175         logger.warning(
 176             "pre-process-forwarder detected a duplicated event" " with id:%s for project:%s.",
 177             event_id,
 178             project_id,
 179         )
 180         return  # message already processed do not reprocess
 181 
 182     try:
 183         project = projects[project_id]
 184     except KeyError:
 185         logger.error("Project for ingested event does not exist: %s", project_id)
 186         return
 187 
 188     # Parse the JSON payload. This is required to compute the cache key and
 189     # call process_event. The payload will be put into Kafka raw, to avoid
 190     # serializing it again.
 191     # XXX: Do not use CanonicalKeyDict here. This may break preprocess_event
 192     # which assumes that data passed in is a raw dictionary.
 193     data = json.loads(payload)
 194 
 195     cache_key = event_processing_store.store(data)
 196 
 197     if attachments:
 198         attachment_objects = [
 199             CachedAttachment(type=attachment.pop("attachment_type"), **attachment)
 200             for attachment in attachments
 201         ]
 202 
 203         attachment_cache.set(cache_key, attachments=attachment_objects, timeout=CACHE_TIMEOUT)
 204 
 205     # Preprocess this event, which spawns either process_event or
 206     # save_event. Pass data explicitly to avoid fetching it again from the
 207     # cache.
 208     with sentry_sdk.start_span(op="ingest_consumer.process_event.preprocess_event"):
 209         preprocess_event(
 210             cache_key=cache_key,
 211             data=data,
 212             start_time=start_time,
 213             event_id=event_id,
 214             project=project,
 215         )
 216 
 217     # remember for an 1 hour that we saved this event (deduplication protection)
 218     cache.set(deduplication_key, "", CACHE_TIMEOUT)
 219 
 220     # emit event_accepted once everything is done
 221     event_accepted.send_robust(ip=remote_addr, data=data, project=project, sender=process_event)
 222

 243 @trace_func(transaction="ingest_consumer.process_individual_attachment")
 244 @metrics.wraps("ingest_consumer.process_individual_attachment")
 245 def process_individual_attachment(message, projects):
 246     event_id = message["event_id"]
 247     project_id = int(message["project_id"])
 248     cache_key = cache_key_for_event({"event_id": event_id, "project": project_id})
 249 
 250     try:
 251         project = projects[project_id]
 252     except KeyError:
 253         logger.error("Project for ingested event does not exist: %s", project_id)
 254         return
 255 
 256     if not features.has("organizations:event-attachments", project.organization, actor=None):
 257         logger.info("Organization has no event attachments: %s", project_id)
 258         return
 259 
 260     # Attachments may be uploaded for events that already exist. Fetch the
 261     # existing group_id, so that the attachment can be fetched by group-level
 262     # APIs. This is inherently racy.
 263     events = eventstore.get_unfetched_events(
 264         filter=eventstore.Filter(event_ids=[event_id], project_ids=[project.id]), limit=1
 265     )
 266 
 267     group_id = None
 268     if events:
 269         group_id = events[0].group_id
 270 
 271     attachment = message["attachment"]
 272     attachment = attachment_cache.get_from_chunks(
 273         key=cache_key, type=attachment.pop("attachment_type"), **attachment
 274     )
 275     if attachment.type != "event.attachment":
 276         logger.exception("invalid individual attachment type: %s", attachment.type)
 277         return
 278 
 279     file = File.objects.create(
 280         name=attachment.name,
 281         type=attachment.type,
 282         headers={"Content-Type": attachment.content_type},
 283     )
 284 
 285     try:
 286         data = attachment.data
 287     except MissingAttachmentChunks:
 288         logger.exception("Missing chunks for cache_key=%s", cache_key)
 289         return
 290 
 291     file.putfile(BytesIO(data))
 292     EventAttachment.objects.create(
 293         project_id=project.id, group_id=group_id, event_id=event_id, name=attachment.name, file=file
 294     )
 295 
 296     attachment.delete()

 301 def process_userreport(message, projects):
 302     project_id = int(message["project_id"])
 303     start_time = to_datetime(message["start_time"])
 304     feedback = json.loads(message["payload"])
 305 
 306     try:
 307         project = projects[project_id]
 308     except KeyError:
 309         logger.error("Project for ingested event does not exist: %s", project_id)
 310         return False
 311 
 312     try:
 313         save_userreport(project, feedback, start_time=start_time)
 314         return True
 315     except Conflict as e:
 316         logger.info("Invalid userreport: %s", e)
 317         return False

 320 def get_ingest_consumer(consumer_types, once=False, concurrency=None, **options):
 321     """
 322     Handles events coming via a kafka queue.
 323 
 324     The events should have already been processed (normalized... ) upstream (by Relay).
 325     """
 326     topic_names = set(
 327         ConsumerType.get_topic_name(consumer_type) for consumer_type in consumer_types
 328     )
 329     return create_batching_kafka_consumer(
 330         topic_names=topic_names, worker=IngestConsumerWorker(concurrency=concurrency), **options
 331     )
