  26     def get_metadata(self, data):
  27         exception = get_path(data, "exception", "values", -1)
  28         if not exception:
  29             return {}
  30 
  31         loc = get_crash_location(data)
  32         rv = {"value": trim(get_path(exception, "value", default=""), 1024)}
  33 
  34         # If the exception mechanism indicates a synthetic exception we do not
  35         # want to record the type and value into the metadata.
  36         if not get_path(exception, "mechanism", "synthetic"):
  37             rv["type"] = trim(get_path(exception, "type", default="Error"), 128)
  38 
  39         # Attach crash location if available
  40         if loc is not None:
  41             fn, func = loc
  42             if fn:
  43                 rv["filename"] = fn
  44             if func:
  45                 rv["function"] = func
  46 
  47         return rv

  49     def get_title(self, metadata):
  50         ty = metadata.get("type")
  51         if ty is None:
  52             return metadata.get("function") or "<unknown>"
  53         if not metadata.get("value"):
  54             return ty
  55         return u"{}: {}".format(ty, truncatechars(metadata["value"].splitlines()[0], 100))
