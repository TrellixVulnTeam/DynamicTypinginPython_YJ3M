  10 def convert_options_to_env(options):
  11     for k, v in six.iteritems(options):
  12         if v is None:
  13             continue
  14         key = "UWSGI_" + k.upper().replace("-", "_")
  15         if isinstance(v, six.string_types):
  16             value = v
  17         elif v is True:
  18             value = "true"
  19         elif v is False:
  20             value = "false"
  21         elif isinstance(v, six.integer_types):
  22             value = six.text_type(v)
  23         else:
  24             raise TypeError("Unknown option type: %r (%s)" % (k, type(v)))
  25         yield key, value

 132     def prepare_environment(self, env=None):
 133         if env is None:
 134             env = os.environ
 135 
 136         # Move all of the options into UWSGI_ env vars
 137         for k, v in convert_options_to_env(self.options):
 138             env.setdefault(k, v)
 139 
 140         # Signal that we're running within uwsgi
 141         env["SENTRY_RUNNING_UWSGI"] = "1"
 142 
 143         # This has already been validated inside __init__
 144         env["SENTRY_SKIP_BACKEND_VALIDATION"] = "1"
 145 
 146         # Look up the bin directory where `sentry` exists, which should be
 147         # sys.argv[0], then inject that to the front of our PATH so we can reliably
 148         # find the `uwsgi` that's installed when inside virtualenv.
 149         # This is so the virtualenv doesn't need to be sourced in, which effectively
 150         # does exactly this.
 151         virtualenv_path = os.path.dirname(os.path.abspath(sys.argv[0]))
 152         current_path = env.get("PATH", "")
 153         if virtualenv_path not in current_path:
 154             env["PATH"] = "%s:%s" % (virtualenv_path, current_path)
