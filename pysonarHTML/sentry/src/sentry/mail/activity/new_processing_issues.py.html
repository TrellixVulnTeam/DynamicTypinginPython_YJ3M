   9 def summarize_issues(issues):
  10     rv = []
  11     for issue in issues:
  12         extra_info = None
  13         msg_d = dict(issue["data"])
  14         msg_d["type"] = issue["type"]
  15 
  16         if "image_path" in issue["data"]:
  17             extra_info = issue["data"]["image_path"].rsplit("/", 1)[-1]
  18             if "image_arch" in issue["data"]:
  19                 extra_info = "%s (%s)" % (extra_info, issue["data"]["image_arch"])
  20 
  21         rv.append({"message": EventError(msg_d).message, "extra_info": extra_info})
  22     return rv
