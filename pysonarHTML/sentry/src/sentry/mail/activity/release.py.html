  34 class ReleaseActivityEmail(ActivityEmail):
  35     def __init__(self, activity):
  36         super(ReleaseActivityEmail, self).__init__(activity)
  37         self.organization = self.project.organization
  38         self.user_id_team_lookup = None
  39         self.email_list = {}
  40         self.user_ids = {}
  41 
  42         try:
  43             self.deploy = Deploy.objects.get(id=activity.data["deploy_id"])
  44         except Deploy.DoesNotExist:
  45             self.deploy = None
  46 
  47         try:
  48             self.release = Release.objects.get(
  49                 organization_id=self.project.organization_id, version=activity.data["version"]
  50             )
  51         except Release.DoesNotExist:
  52             self.release = None
  53             self.repos = []
  54             self.projects = []
  55         else:
  56             self.projects = list(self.release.projects.all())
  57             self.commit_list = [
  58                 rc.commit
  59                 for rc in ReleaseCommit.objects.filter(release=self.release).select_related(
  60                     "commit", "commit__author"
  61                 )
  62             ]
  63             repos = {
  64                 r_id: {"name": r_name, "commits": []}
  65                 for r_id, r_name in Repository.objects.filter(
  66                     organization_id=self.project.organization_id,
  67                     id__in={c.repository_id for c in self.commit_list},
  68                 ).values_list("id", "name")
  69             }
  70 
  71             self.email_list = set([c.author.email for c in self.commit_list if c.author])
  72             if self.email_list:
  73                 users = {
  74                     ue.email: ue.user
  75                     for ue in UserEmail.objects.filter(
  76                         in_iexact("email", self.email_list),
  77                         is_verified=True,
  78                         user__sentry_orgmember_set__organization=self.organization,
  79                     ).select_related("user")
  80                 }
  81                 self.user_ids = {u.id for u in six.itervalues(users)}
  82 
  83             else:
  84                 users = {}
  85 
  86             for commit in self.commit_list:
  87                 repos[commit.repository_id]["commits"].append(
  88                     (commit, users.get(commit.author.email) if commit.author_id else None)
  89                 )
  90 
  91             self.repos = repos.values()
  92 
  93             self.environment = (
  94                 Environment.objects.get(id=self.deploy.environment_id).name or "Default Environment"
  95             )
  96 
  97             self.group_counts_by_project = dict(
  98                 Group.objects.filter(
  99                     project__in=self.projects,
 100                     id__in=GroupLink.objects.filter(
 101                         linked_type=GroupLink.LinkedType.commit,
 102                         linked_id__in=ReleaseCommit.objects.filter(
 103                             release=self.release
 104                         ).values_list("commit_id", flat=True),
 105                     ).values_list("group_id", flat=True),
 106                 )
 107                 .values_list("project")
 108                 .annotate(num_groups=Count("id"))
 109             )

 111     def should_email(self):
 112         return bool(self.release and self.deploy)

 114     def get_participants(self):
 115         # collect all users with verified emails on a team in the related projects,
 116         users = list(
 117             User.objects.filter(
 118                 emails__is_verified=True,
 119                 sentry_orgmember_set__teams__in=Team.objects.filter(
 120                     id__in=ProjectTeam.objects.filter(project__in=self.projects).values_list(
 121                         "team_id", flat=True
 122                     )
 123                 ),
 124                 is_active=True,
 125             ).distinct()
 126         )
 127 
 128         # get all the involved users' settings for deploy-emails (user default
 129         # saved without org set)
 130         user_options = UserOption.objects.filter(
 131             Q(organization=self.organization) | Q(organization=None),
 132             user__in=users,
 133             key="deploy-emails",
 134         )
 135 
 136         options_by_user_id = defaultdict(dict)
 137         for uoption in user_options:
 138             key = "default" if uoption.organization is None else "org"
 139             options_by_user_id[uoption.user_id][key] = uoption.value
 140 
 141         # and couple them with the the users' setting value for deploy-emails
 142         # prioritize user/org specific, then user default, then product default
 143         users_with_options = []
 144         for user in users:
 145             options = options_by_user_id.get(user.id, {})
 146             users_with_options.append(
 147                 (
 148                     user,
 149                     options.get(
 150                         "org", options.get("default", UserOptionValue.committed_deploys_only)
 151                     ),
 152                 )
 153             )
 154 
 155         # filter down to members which have been seen in the commit log:
 156         participants_committed = {
 157             user: GroupSubscriptionReason.committed
 158             for user, option in users_with_options
 159             if option == UserOptionValue.committed_deploys_only and user.id in self.user_ids
 160         }
 161 
 162         # or who opt into all deploy emails:
 163         participants_opted = {
 164             user: GroupSubscriptionReason.deploy_setting
 165             for user, option in users_with_options
 166             if option == UserOptionValue.all_deploys
 167         }
 168 
 169         # merge the two type of participants
 170         return dict(chain(six.iteritems(participants_committed), six.iteritems(participants_opted)))

 172     def get_users_by_teams(self):
 173         if not self.user_id_team_lookup:
 174             user_teams = defaultdict(list)
 175             queryset = User.objects.filter(
 176                 sentry_orgmember_set__organization_id=self.organization.id
 177             ).values_list("id", "sentry_orgmember_set__teams")
 178             for user_id, team_id in queryset:
 179                 user_teams[user_id].append(team_id)
 180             self.user_id_team_lookup = user_teams
 181         return self.user_id_team_lookup

 183     def get_context(self):
 184         file_count = (
 185             CommitFileChange.objects.filter(
 186                 commit__in=self.commit_list, organization_id=self.organization.id
 187             )
 188             .values("filename")
 189             .distinct()
 190             .count()
 191         )
 192 
 193         return {
 194             "commit_count": len(self.commit_list),
 195             "author_count": len(self.email_list),
 196             "file_count": file_count,
 197             "repos": self.repos,
 198             "release": self.release,
 199             "deploy": self.deploy,
 200             "environment": self.environment,
 201             "setup_repo_link": absolute_uri(
 202                 u"/organizations/{}/repos/".format(self.organization.slug)
 203             ),
 204         }
 205 
 206     def get_user_context(self, user):
 207         if user.is_superuser or self.organization.flags.allow_joinleave:
 208             projects = self.projects
 209         else:
 210             teams = self.get_users_by_teams()[user.id]
 211             team_projects = set(
 212                 ProjectTeam.objects.filter(team_id__in=teams)
 213                 .values_list("project_id", flat=True)
 214                 .distinct()
 215             )
 216             projects = [p for p in self.projects if p.id in team_projects]
 217 
 218         release_links = [
 219             absolute_uri(
 220                 u"/organizations/{}/releases/{}/?project={}".format(
 221                     self.organization.slug, self.release.version, p.id
 222                 )
 223             )
 224             for p in projects
 225         ]
 226 
 227         resolved_issue_counts = [self.group_counts_by_project.get(p.id, 0) for p in projects]
 228         return {
 229             "projects": zip(projects, release_links, resolved_issue_counts),
 230             "project_count": len(projects),
 231         }
