 108     def get_context(self):
 109         description = self.get_description()
 110         try:
 111             description, params, html_params = description
 112         except ValueError:
 113             try:
 114                 description, params = description
 115                 html_params = params
 116             except ValueError:
 117                 params, html_params = {}, {}

 119         return {
 120             "activity_name": self.get_activity_name(),

 149     def avatar_as_html(self):
 150         user = self.activity.user
 151         if not user:
 152             return u'<img class="avatar" src="{}" width="20px" height="20px" />'.format(
 153                 escape(self._get_sentry_avatar_url())
 154             )
 155         avatar_type = user.get_avatar_type()
 156         if avatar_type == "upload":
 157             return u'<img class="avatar" src="{}" />'.format(
 158                 escape(self._get_user_avatar_url(user))
 159             )
 160         elif avatar_type == "letter_avatar":
 161             return get_email_avatar(user.get_display_name(), user.get_label(), 20, False)
 162         else:
 163             return get_email_avatar(user.get_display_name(), user.get_label(), 20, True)

 169     def _get_user_avatar_url(self, user, size=20):
 170         try:
 171             avatar = UserAvatar.objects.get(user=user)
 172         except UserAvatar.DoesNotExist:
 173             return ""
 174 
 175         url = reverse("sentry-user-avatar-url", args=[avatar.ident])
 176         if size:
 177             url = u"{}?s={}".format(url, int(size))
 178         return absolute_uri(url)

 180     def description_as_text(self, description, params):
 181         user = self.activity.user
 182         if user:
 183             name = user.name or user.email
 184         else:
 185             name = u"Sentry"
 186

 192     def description_as_html(self, description, params):
 193         user = self.activity.user
 194         if user:
 195             name = user.get_display_name()
 196         else:
 197             name = "Sentry"
 198 
 199         fmt = u'<span class="avatar-container">{}</span> <strong>{}</strong>'
 200 
 201         author = mark_safe(fmt.format(self.avatar_as_html(), escape(name)))
 202 
 203         an_issue = u'<a href="{}">an issue</a>'.format(escape(self.get_group_link()))
 204 
 205         context = {"author": author, "an issue": an_issue}
 206         context.update(params)
 207 
 208         return mark_safe(description.format(**context))

 210     def send(self):
 211         if not self.should_email():
 212             return
 213 
 214         participants = self.get_participants()
 215         if not participants:
 216             return
 217 
 218         activity = self.activity
 219         project = self.project
 220         group = self.group
 221 
 222         context = self.get_base_context()
 223         context.update(self.get_context())
 224 
 225         template = self.get_template()
 226         html_template = self.get_html_template()
 227         email_type = self.get_email_type()
 228         headers = self.get_headers()
 229 
 230         for user, reason in participants.items():
 231             if group:
 232                 context.update(
 233                     {
 234                         "reason": GroupSubscriptionReason.descriptions.get(
 235                             reason, "are subscribed to this issue"
 236                         ),
 237                         "unsubscribe_link": generate_signed_link(
 238                             user.id,
 239                             "sentry-account-email-unsubscribe-issue",
 240                             kwargs={"issue_id": group.id},
 241                         ),
 242                     }
 243                 )
 244             user_context = self.get_user_context(user)
 245             if user_context:
 246                 user_context.update(context)
 247             else:
 248                 user_context = context
 249 
 250             msg = MessageBuilder(
 251                 subject=self.get_subject_with_prefix(),
 252                 template=template,
 253                 html_template=html_template,
 254                 headers=headers,
 255                 type=email_type,
 256                 context=user_context,
 257                 reference=activity,
 258                 reply_reference=group,
 259             )
 260             msg.add_users([user.id], project=project)
 261             msg.send_async()
