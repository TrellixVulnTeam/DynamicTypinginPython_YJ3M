  19 class NotifyEmailForm(forms.Form):
  20     targetType = forms.ChoiceField(choices=CHOICES)

  22         required=False, help_text="Only required if 'Member' or 'Team' is selected"

  25     def __init__(self, project, *args, **kwargs):

  29     def clean_targetIdentifier(self):
  30         targetIdentifier = self.cleaned_data.get("targetIdentifier")
  31         # XXX: Clean up some bad data in the database
  32         if targetIdentifier == "None":
  33             targetIdentifier = None
  34         if targetIdentifier:
  35             try:
  36                 targetIdentifier = int(targetIdentifier)
  37             except ValueError:
  38                 raise forms.ValidationError("targetIdentifier must be an integer")
  39         return targetIdentifier
  40 
  41     def clean(self):
  42         cleaned_data = super(NotifyEmailForm, self).clean()
  43         try:
  44             targetType = ActionTargetType(cleaned_data.get("targetType"))
  45         except ValueError:
  46             msg = forms.ValidationError("Invalid targetType specified")
  47             self.add_error("targetType", msg)
  48             return
  49 
  50         targetIdentifier = cleaned_data.get("targetIdentifier")
  51 
  52         self.cleaned_data["targetType"] = targetType.value
  53         if targetType == ActionTargetType.ISSUE_OWNERS:
  54             return
  55 
  56         if targetIdentifier is None:
  57             msg = forms.ValidationError("You need to specify a Team or Member to send mail to.")
  58             self.add_error("targetIdentifier", msg)
  59             return
  60 
  61         if (
  62             targetType == ActionTargetType.TEAM
  63             and not Project.objects.filter(
  64                 teams__id=int(targetIdentifier), id=self.project.id
  65             ).exists()
  66         ):
  67             msg = forms.ValidationError("This team is not part of the project.")
  68             self.add_error("targetIdentifier", msg)
  69             return
  70 
  71         if (
  72             targetType == ActionTargetType.MEMBER
  73             and not User.objects.get_from_projects(self.project.organization.id, [self.project])
  74             .filter(id=int(targetIdentifier))
  75             .exists()
  76         ):
  77             msg = forms.ValidationError("This user is not part of the project.")
  78             self.add_error("targetIdentifier", msg)
  79             return
  80 
  81         self.cleaned_data["targetIdentifier"] = targetIdentifier

  84 class NotifyEmailAction(EventAction):
  85     form_cls = NotifyEmailForm
  86     label = "Send an email to {targetType}"
  87     prompt = "Send an email"
  88     metrics_slug = "EmailAction"
  89 
  90     def __init__(self, *args, **kwargs):
  91         super(NotifyEmailAction, self).__init__(*args, **kwargs)
  92         self.form_fields = {"targetType": {"type": "mailAction", "choices": CHOICES}}
  93 
  94     def after(self, event, state):
  95         extra = {"event_id": event.event_id}
  96         group = event.group
  97 
  98         if not mail_adapter.should_notify(group=group):
  99             extra["group_id"] = group.id
 100             self.logger.info("rule.fail.should_notify", extra=extra)
 101             return
 102 
 103         metrics.incr("notifications.sent", instance=self.metrics_slug, skip_internal=False)
 104         yield self.future(
 105             lambda event, futures: mail_adapter.rule_notify(
 106                 event,
 107                 futures,
 108                 ActionTargetType(self.data["targetType"]),
 109                 self.data.get("targetIdentifier", None),
 110             )
 111         )
 112 
 113     def get_form_instance(self):
 114         return self.form_cls(self.project, self.data)
