  32 class OptionsStore(object):
  33     """
  34     Abstraction for the Option storage logic that should be driven
  35     by the OptionsManager.
  36 
  37     OptionsStore is gooey and raw. It provides no protection over
  38     what goes into the store. It only knows that it's reading/writing
  39     to the right place. If using the OptionsStore directly, it's your
  40     job to do validation of the data. You should probably go through
  41     OptionsManager instead, unless you need raw access to something.
  42     """
  43 
  44     def __init__(self, cache=None, ttl=None):
  45         self.cache = cache
  46         self.ttl = ttl
  47         self.flush_local_cache()

  50     def model(self):
  51         from sentry.models.option import Option

  55     def make_key(self, name, default, type, flags, ttl, grace):
  56         return Key(name, default, type, flags, int(ttl), int(grace), _make_cache_key(name))

  58     def get(self, key, silent=False):
  59         """
  60         Fetches a value from the options store.
  61         """
  62         result = self.get_cache(key, silent=silent)
  63         if result is not None:
  64             return result
  65 
  66         result = self.get_store(key, silent=silent)
  67         if result is not None:
  68             return result
  69 
  70         # As a last ditch effort, let's hope we have a key
  71         # in local cache that's possibly stale
  72         return self.get_local_cache(key, force_grace=True)
  73 
  74     def get_cache(self, key, silent=False):
  75         """
  76         First check agaist our local in-process cache, falling
  77         back to the network cache.
  78         """
  79         value = self.get_local_cache(key)
  80         if value is not None:
  81             return value
  82 
  83         if self.cache is None:
  84             return None
  85 
  86         cache_key = key.cache_key
  87         try:
  88             value = self.cache.get(cache_key)
  89         except Exception:
  90             if not silent:
  91                 logger.warn(CACHE_FETCH_ERR, key.name, extra={"key": key.name}, exc_info=True)
  92             value = None
  93 
  94         if value is not None and key.ttl > 0:
  95             self._local_cache[cache_key] = _make_cache_value(key, value)
  96 
  97         return value

  99     def get_local_cache(self, key, force_grace=False):
 100         """
 101         Attempt to fetch a key out of the local cache.
 102 
 103         If the key exists, but is beyond expiration, we only
 104         return it if grace=True. This forces the key to be returned
 105         in a disaster scenario as long as we're still holding onto it.
 106         This allows the OptionStore to pave over potential network hiccups
 107         by returning a stale value.
 108         """
 109         try:
 110             value, expires, grace = self._local_cache[key.cache_key]
 111         except KeyError:
 112             return None
 113 
 114         now = int(time())
 115 
 116         # Key is within normal expiry window, so just return it
 117         if now < expires:
 118             return value
 119 
 120         # If we're able to accept within grace window, return it
 121         if force_grace and now < grace:
 122             return value
 123 
 124         # Let's clean up values if we're beyond grace.
 125         if now > grace:
 126             try:
 127                 del self._local_cache[key.cache_key]
 128             except KeyError:
 129                 # This could only exist in a race condition
 130                 # where another thread has already deleted this key,
 131                 # but we'll guard ourselves against it Justin Case.
 132                 # In this case, it's also possible that another thread
 133                 # has updated the value at this key, causing us to evict
 134                 # it prematurely. This isn't ideal, but not terrible
 135                 # since I don't want to introduce locking to prevent this.
 136                 # Even if it did happen, the consequence is just another
 137                 # network hop.
 138                 pass
 139 
 140         # If we're outside the grace window, even if we ask for it
 141         # in grace, too bad. The value is considered bad.
 142         return None
 143 
 144     def get_store(self, key, silent=False):
 145         """
 146         Attempt to fetch value from the database. If successful,
 147         also set it back in the cache.
 148 
 149         Returns None in both cases, if the key doesn't actually exist,
 150         or if we errored fetching it.
 151 
 152         NOTE: This behavior should probably be improved to differentiate
 153         between a miss vs error, but not worth it now since the value
 154         is limited at the moment.
 155         """
 156         try:
 157             value = self.model.objects.get(key=key.name).value
 158         except (self.model.DoesNotExist, ProgrammingError, OperationalError):
 159             value = None
 160         except Exception:
 161             if not silent:
 162                 logger.exception("option.failed-lookup", extra={"key": key.name})
 163             value = None
 164         else:
 165             # we only attempt to populate the cache if we were previously
 166             # able to successfully talk to the backend
 167             # NOTE: There is definitely a race condition here between updating
 168             # the store and the cache
 169             try:
 170                 self.set_cache(key, value)
 171             except Exception:
 172                 if not silent:
 173                     logger.warn(CACHE_UPDATE_ERR, key.name, extra={"key": key.name}, exc_info=True)
 174         return value
 175 
 176     def set(self, key, value):
 177         """
 178         Store a value in the option store. Value must get persisted to database first,
 179         then attempt caches. If it fails database, the entire operation blows up.
 180         If cache fails, we ignore silently since it'll get repaired later by sync_options.
 181         A boolean is returned to indicate if the network cache was set successfully.
 182         """
 183         assert self.cache is not None, "cache must be configured before mutating options"
 184 
 185         self.set_store(key, value)
 186         return self.set_cache(key, value)
 187 
 188     def set_store(self, key, value):
 189         from sentry.db.models.query import create_or_update
 190 
 191         create_or_update(
 192             model=self.model, key=key.name, values={"value": value, "last_updated": timezone.now()}
 193         )
 194 
 195     def set_cache(self, key, value):
 196         if self.cache is None:
 197             return None
 198 
 199         cache_key = key.cache_key
 200 
 201         if key.ttl > 0:
 202             self._local_cache[cache_key] = _make_cache_value(key, value)
 203 
 204         try:
 205             self.cache.set(cache_key, value, self.ttl)
 206             return True
 207         except Exception:
 208             logger.warn(CACHE_UPDATE_ERR, key.name, extra={"key": key.name}, exc_info=True)
 209             return False

 211     def delete(self, key):
 212         """
 213         Remove key out of option stores. This operation must succeed on the
 214         database first. If database fails, an exception is raised.
 215         If database succeeds, caches are then allowed to fail silently.
 216         A boolean is returned to indicate if the network deletion succeeds.
 217         """
 218         assert self.cache is not None, "cache must be configured before mutating options"
 219 
 220         self.delete_store(key)
 221         return self.delete_cache(key)
 222 
 223     def delete_store(self, key):
 224         self.model.objects.filter(key=key.name).delete()
 225 
 226     def delete_cache(self, key):
 227         cache_key = key.cache_key
 228         try:
 229             del self._local_cache[cache_key]
 230         except KeyError:
 231             pass
 232 
 233         try:
 234             self.cache.delete(cache_key)
 235             return True
 236         except Exception:
 237             logger.warn(CACHE_UPDATE_ERR, key.name, extra={"key": key.name}, exc_info=True)
 238             return False

 240     def clean_local_cache(self):
 241         """
 242         Iterate over our local cache items, and
 243         remove the keys that are beyond their grace time.
 244         """
 245         to_expire = []
 246         now = int(time())
 247 
 248         try:
 249             for k, (_, _, grace) in six.iteritems(self._local_cache):
 250                 if now > grace:
 251                     to_expire.append(k)
 252         except RuntimeError:
 253             # It's possible for the dictionary to be mutated in another thread
 254             # while iterating, but this case is rare, so instead of making a
 255             # copy and iterating that, it's more efficient to just let it fail
 256             # gracefully. It'll just get re-run later.
 257             return
 258 
 259         for k in to_expire:
 260             try:
 261                 del self._local_cache[k]
 262             except KeyError:
 263                 # This could only exist in a race condition
 264                 # where another thread has already deleted this key,
 265                 # but we'll guard ourselves against it Justin Case.
 266                 pass

 268     def flush_local_cache(self):
 269         """
 270         Empty store's local in-process cache.
 271         """
 272         self._local_cache = {}

 274     def maybe_clean_local_cache(self, **kwargs):
 275         # Periodically force an expire on the local cache.
 276         # This cleanup is purely to keep memory low and garbage collect
 277         # old values. It's not required to run to keep things consistent.
 278         # Internally, if an option is fetched and it's expired, it gets
 279         # evicted immediately. This is purely for options that haven't
 280         # been fetched since they've expired.
 281         if not self._local_cache:
 282             return
 283         if random() < 0.25:
 284             self.clean_local_cache()
