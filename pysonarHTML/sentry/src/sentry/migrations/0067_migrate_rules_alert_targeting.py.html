  26 def migrate_project_to_issue_alert_targeting(project, ProjectOption, Rule, User, UserOption):
  27     if project.flags.has_issue_alerts_targeting:
  28         # Migration has already been run.
  29         return
  30     with transaction.atomic():
  31         # Determine whether this project actually has mail enabled
  32         try:
  33             mail_enabled = ProjectOption.objects.get(project=project, key="mail:enabled").value
  34         except ProjectOption.DoesNotExist:
  35             mail_enabled = True
  36         for rule in Rule.objects.filter(project=project, status=0):
  37             migrate_legacy_rule(rule, mail_enabled)
  38 
  39         if not mail_enabled:
  40             # If mail disabled, then we want to disable mail options for all
  41             # users associated with this project so that they don't suddenly start
  42             # getting mail via the `MailAdapter`, since it will always be enabled.
  43             for user in User.objects.filter(
  44                 sentry_orgmember_set__teams__in=project.teams.all(), is_active=True
  45             ):
  46                 set_user_option(UserOption, user, "mail:alert", 0, project)
  47                 set_user_option(UserOption, user, "workflow:notifications", "2", project=project)
  48 
  49         # This marks the migration finished and shows the new UI
  50         project.flags.has_issue_alerts_targeting = True
  51         project.save()

  54 def migrate_legacy_rule(rule, mail_enabled):
  55     actions = rule.data.get("actions", [])
  56     new_actions = []
  57     has_mail_action = False
  58     for action in actions:
  59         action_id = action.get("id")
  60         if action_id == "sentry.rules.actions.notify_event.NotifyEventAction":
  61             # This is the "Send a notification (for all legacy integrations)" action.
  62             # When this action exists, we want to add the new `NotifyEmailAction` action
  63             # to the rule. We'll still leave `NotifyEventAction` in place, since it will
  64             # only notify non-mail plugins once we've migrated.
  65             new_actions.append(action)
  66             has_mail_action = True
  67         elif (
  68             action_id == "sentry.rules.actions.notify_event_service.NotifyEventServiceAction"
  69             and action.get("service") == "mail"
  70         ):
  71             # This is the "Send a notification via mail" action. When this action
  72             # exists, we want to add the new `NotifyEmailAction` action to the rule.
  73             # We'll drop this action from the rule, since all it does it send mail and
  74             # we don't want to double up.
  75             has_mail_action = True
  76         else:
  77             new_actions.append(action)
  78 
  79     # We only add the new action if the mail plugin is actually enabled, and there's an
  80     # action that sends by mail. We do this outside the loop to ensure we don't add it
  81     # more than once.
  82     if mail_enabled and has_mail_action:
  83         new_actions.append(mail_action)
  84 
  85     if actions != new_actions:
  86         rule.data["actions"] = new_actions
  87         rule.save()

  90 def migrate_to_issue_alert_targeting(apps, schema_editor):
  91     Project = apps.get_model("sentry", "Project")
  92     ProjectOption = apps.get_model("sentry", "ProjectOption")
  93     Organization = apps.get_model("sentry", "Organization")

  95     User = apps.get_model("sentry", "User")
  96     UserOption = apps.get_model("sentry", "UserOption")
  97 
  98     for org in RangeQuerySetWrapperWithProgressBar(Organization.objects.filter(status=0)):
  99         # We migrate a project at a time, but we prefer to group by org so that for the
 100         # most part an org will see the changes all at once.
 101         for project in Project.objects.filter(organization=org, status=0):
 102             try:
 103                 migrate_project_to_issue_alert_targeting(
 104                     project, ProjectOption, Rule, User, UserOption
 105                 )
 106             except Exception:
 107                 # If a project fails we'll just log and continue. We shouldn't see any
 108                 # failures, but if we do we can analyze them and re-run this migration,
 109                 # since it is idempotent.
 110                 logging.exception("Error migrating project {}".format(project.id))
