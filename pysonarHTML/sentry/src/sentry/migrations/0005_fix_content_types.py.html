   7 def fix_content_types(apps, schema_editor):
   8     # XXX: This is a gross hack. We missed removing this column a long time ago while
   9     # upgrading Django. Since different databases might be in different states depending
  10     # on which path they take to get to Django migrations, it's safest to just check
  11     # if the column exists for everyone, and remove it if so. This removal is safe,
  12     # since the column has been long removed from the Django model.
  13 
  14     c = schema_editor.connection.cursor()
  15     c.execute(
  16         """
  17         SELECT 1
  18         FROM information_schema.columns
  19         WHERE table_name='django_content_type' and column_name='name';
  20         """
  21     )
  22     results = c.fetchall()
  23     if len(results):
  24         c.execute('ALTER TABLE django_content_type DROP COLUMN "name";')
  25     c.close()
