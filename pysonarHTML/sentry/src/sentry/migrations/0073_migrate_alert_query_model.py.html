  24 def convert_alert_rule_to_snuba_query(alert_rule, SnubaQuery):
  25     if alert_rule.snuba_query:
  26         return
  27 
  28     with transaction.atomic():
  29         try:
  30             environment = alert_rule.environment.all()[0]
  31         except IndexError:
  32             environment = None
  33         snuba_query = SnubaQuery.objects.create(
  34             dataset=alert_rule.dataset,
  35             query=alert_rule.query,
  36             aggregate=aggregation_function_translations[QueryAggregations(alert_rule.aggregation)],
  37             time_window=int(timedelta(minutes=alert_rule.time_window).total_seconds()),
  38             resolution=int(timedelta(minutes=alert_rule.resolution).total_seconds()),
  39             environment=environment,
  40         )
  41         alert_rule.snuba_query = snuba_query
  42         alert_rule.save()
