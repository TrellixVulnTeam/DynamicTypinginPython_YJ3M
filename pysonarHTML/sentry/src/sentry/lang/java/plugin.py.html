  17 class JavaStacktraceProcessor(StacktraceProcessor):
  18     def __init__(self, *args, **kwargs):
  19         StacktraceProcessor.__init__(self, *args, **kwargs)
  20 
  21         self.images = set()
  22         self.available = False
  23 
  24         for image in get_path(self.data, "debug_meta", "images", filter=is_valid_image, default=()):
  25             self.available = True
  26             self.images.add(six.text_type(image["uuid"]).lower())

  28     def handles_frame(self, frame, stacktrace_info):
  29         platform = frame.get("platform") or self.data.get("platform")
  30         return platform == "java" and self.available and "function" in frame and "module" in frame

  32     def preprocess_step(self, processing_task):
  33         if not self.available:
  34             return False
  35 
  36         dif_paths = ProjectDebugFile.difcache.fetch_difs(
  37             self.project, self.images, features=["mapping"]
  38         )
  39         self.mapping_views = []
  40 
  41         for debug_id in self.images:
  42             error_type = None
  43 
  44             dif_path = dif_paths.get(debug_id)
  45             if dif_path is None:
  46                 error_type = EventError.PROGUARD_MISSING_MAPPING
  47             else:
  48                 view = ProguardMapper.open(dif_path)
  49                 if not view.has_line_info:
  50                     error_type = EventError.PROGUARD_MISSING_LINENO
  51                 else:
  52                     self.mapping_views.append(view)
  53 
  54             if error_type is None:
  55                 continue
  56 
  57             self.data.setdefault("_metrics", {})["flag.processing.error"] = True
  58 
  59             self.data.setdefault("errors", []).append(
  60                 {"type": error_type, "mapping_uuid": debug_id}
  61             )
  62 
  63             report_processing_issue(
  64                 self.data,
  65                 scope="proguard",
  66                 object="mapping:%s" % debug_id,
  67                 type=error_type,
  68                 data={"mapping_uuid": debug_id},
  69             )
  70 
  71         return True

  73     def process_exception(self, exception):
  74         ty = exception.get("type")
  75         mod = exception.get("module")
  76         if not ty or not mod:
  77             return False
  78 
  79         key = "%s.%s" % (mod, ty)
  80 
  81         for view in self.mapping_views:
  82             mapped = view.remap_class(key)
  83             if mapped:
  84                 new_module, new_cls = mapped.rsplit(".", 1)
  85                 exception["module"] = new_module
  86                 exception["type"] = new_cls
  87                 return True
  88 
  89         return False

  91     def process_frame(self, processable_frame, processing_task):
  92         frame = processable_frame.frame
  93         raw_frame = dict(frame)
  94 
  95         # first, try to remap complete frames
  96         for view in self.mapping_views:
  97             mapped = view.remap_frame(frame["module"], frame["function"], frame.get("lineno") or 0)
  98 
  99             if len(mapped) > 0:
 100                 new_frames = []
 101                 bottom_class = mapped[-1].class_name
 102 
 103                 # sentry expects stack traces in reverse order
 104                 for new_frame in reversed(mapped):
 105                     frame = dict(raw_frame)
 106                     frame["module"] = new_frame.class_name
 107                     frame["function"] = new_frame.method
 108                     frame["lineno"] = new_frame.line
 109 
 110                     # clear the filename for all *foreign* classes
 111                     if frame["module"] != bottom_class:
 112                         frame.pop("filename", None)
 113                         frame.pop("abs_path", None)
 114 
 115                     new_frames.append(frame)
 116 
 117                 return new_frames, [raw_frame], []
 118 
 119         # second, if that is not possible, try to re-map only the class-name
 120         for view in self.mapping_views:
 121             mapped = view.remap_class(frame["module"])
 122 
 123             if mapped:
 124                 new_frame = dict(raw_frame)
 125                 new_frame["module"] = mapped
 126                 return [new_frame], [raw_frame], []
 127 
 128         return
