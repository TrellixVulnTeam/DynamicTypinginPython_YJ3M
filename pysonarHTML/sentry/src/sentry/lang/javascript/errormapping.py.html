  31 def is_expired(ts):
  32     return ts > (time.time() - SOFT_TIMEOUT - random.random() * SOFT_TIMEOUT_FUZZINESS)

  35 class Processor(object):
  36     def __init__(self, vendor, mapping_url, regex, func):
  37         self.vendor = vendor
  38         self.mapping_url = mapping_url
  39         self.regex = re.compile(regex)
  40         self.func = func
  41 
  42     def load_mapping(self):
  43         key = "javascript.errormapping:%s" % self.vendor
  44         mapping = cache.get(key)
  45         cached_rv = None
  46         if mapping is not None:
  47             ts, cached_rv = json.loads(mapping)
  48             if not is_expired(ts):
  49                 return cached_rv
  50 
  51         try:
  52             http_session = http.build_session()
  53             response = http_session.get(
  54                 self.mapping_url, allow_redirects=True, timeout=settings.SENTRY_SOURCE_FETCH_TIMEOUT
  55             )
  56             # Make sure we only get a 2xx to prevent caching bad data
  57             response.raise_for_status()
  58             data = response.json()
  59             cache.set(key, json.dumps([time.time(), data]), HARD_TIMEOUT)
  60         except Exception:
  61             if cached_rv is None:
  62                 raise
  63             return cached_rv
  64         return data
  65 
  66     def try_process(self, exc):
  67         if not exc.get("value"):
  68             return False
  69         match = self.regex.search(exc["value"])
  70         if match is None:
  71             return False
  72         mapping = self.load_mapping()
  73         return self.func(exc, match, mapping)

  76 def minified_error(vendor, mapping_url, regex):
  77     def decorator(f):
  78         error_processors[vendor] = Processor(vendor, mapping_url, regex, f)
  79 
  80     return decorator

  88 def process_react_exception(exc, match, mapping):
  89     error_id, qs = match.groups()
  90     msg_format = mapping.get(error_id)
  91     if msg_format is None:
  92         return False
  93 
  94     arg_count = count_sprintf_parameters(msg_format)
  95     args = []
  96     for k, v in parse_qsl(qs, keep_blank_values=True):
  97         if k == "args[]":
  98             if isinstance(v, six.binary_type):
  99                 v = v.decode("utf-8", "replace")
 100             args.append(v)
 101 
 102     # Due to truncated error messages we sometimes might not be able to
 103     # get all arguments.  In that case we fill up missing parameters for
 104     # the format string with <redacted>.
 105     args = tuple(args + [u"<redacted>"] * (arg_count - len(args)))[:arg_count]
 106     exc["value"] = msg_format % args
 107 
 108     return True

 111 def rewrite_exception(data):
 112     """Rewrite an exception in an event if needed.  Updates the exception
 113     in place and returns `True` if a modification was performed or `False`
 114     otherwise.
 115     """
 116     rv = False
 117     for exc in get_path(data, "exception", "values", filter=True, default=()):
 118         for processor in six.itervalues(error_processors):
 119             try:
 120                 if processor.try_process(exc):
 121                     rv = True
 122                     break
 123             except Exception as e:
 124                 logger.error('Failed to run processor "%s": %s', processor.vendor, e, exc_info=True)
 125                 data.setdefault("_metrics", {})["flag.processing.error"] = True
 126 
 127     return rv
