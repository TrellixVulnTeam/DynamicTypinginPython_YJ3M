  10 LOCALES_DIR = os.path.join(os.path.dirname(__file__), "../../data/error-locale")
  11 TARGET_LOCALE = "en-US"
  12 
  13 translation_lookup_table = set()
  14 target_locale_lookup_table = dict()
  15 
  16 
  17 def populate_target_locale_lookup_table():
  18     for locale in os.listdir(LOCALES_DIR):
  19         fn = os.path.join(LOCALES_DIR, locale)
  20         if not os.path.isfile(fn):
  21             continue
  22 
  23         with io.open(fn, encoding="utf-8") as f:
  24             for line in f:
  25                 key, translation = line.split(",", 1)
  26                 translation = translation.strip()
  27 
  28                 if TARGET_LOCALE in locale:
  29                     target_locale_lookup_table[key] = translation
  30                 else:
  31                     translation_regexp = re.escape(translation)
  32                     translation_regexp = translation_regexp.replace(
  33                         "\%s", r"(?P<format_string_data>[a-zA-Z0-9-_\$]+)"
  34                     )
  35                     # Some errors are substrings of more detailed ones, so we need exact match
  36                     translation_regexp = re.compile("^" + translation_regexp + "$")
  37                     translation_lookup_table.add((translation_regexp, key))
  38

  40 def find_translation(message):
  41     if not target_locale_lookup_table:
  42         populate_target_locale_lookup_table()
  43 
  44     for translation in translation_lookup_table:
  45         translation_regexp, key = translation
  46         match = translation_regexp.search(message)
  47 
  48         if match is not None:
  49             format_string_data = match.groupdict().get("format_string_data")
  50 
  51             if format_string_data is None:
  52                 return [key, None]
  53             else:
  54                 return [key, format_string_data]
  55 
  56     return [None, None]

  59 def format_message(message, data):
  60     return message.replace("%s", data)
  61 
  62 
  63 message_type_regexp = re.compile("^(?P<type>[a-zA-Z]*Error): (?P<message>.*)")
  64 
  65 
  66 def translate_message(original_message):
  67     if not isinstance(original_message, six.string_types):
  68         return original_message
  69 
  70     type = None
  71     message = original_message.strip()
  72 
  73     # Handle both cases. Just a message and message preceded with error type
  74     # eg. `ReferenceError: foo`, `TypeError: bar`
  75     match = message_type_regexp.search(message)
  76 
  77     if match is not None:
  78         type = match.groupdict().get("type")
  79         message = match.groupdict().get("message")
  80 
  81     translation, format_string_data = find_translation(message)
  82 
  83     if translation is None:
  84         return original_message
  85     else:
  86         translated_message = target_locale_lookup_table.get(translation, original_message)
  87 
  88         if type is not None:
  89             translated_message = type + ": " + translated_message
  90 
  91         if format_string_data is None:
  92             return translated_message
  93         else:
  94             return format_message(translated_message, format_string_data)

  97 def translate_exception(data):
  98     message = get_path(data, "logentry", "message")
  99     if message:
 100         data["logentry"]["message"] = translate_message(message)
 101 
 102     formatted = get_path(data, "logentry", "formatted")
 103     if formatted:
 104         data["logentry"]["formatted"] = translate_message(formatted)
 105 
 106     for entry in get_path(data, "exception", "values", filter=True, default=()):
 107         if "value" in entry:
 108             entry["value"] = translate_message(entry["value"])
 109 
 110     return data
