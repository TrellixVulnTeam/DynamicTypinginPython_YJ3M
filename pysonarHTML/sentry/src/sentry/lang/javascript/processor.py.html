  82 def trim_line(line, column=0):
  83     """
  84     Trims a line down to a goal of 140 characters, with a little
  85     wiggle room to be sensible and tries to trim around the given
  86     `column`. So it tries to extract 60 characters before and after
  87     the provided `column` and yield a better context.
  88     """
  89     line = line.strip(u"\n")
  90     ll = len(line)
  91     if ll <= 150:
  92         return line
  93     if column > ll:
  94         column = ll
  95     start = max(column - 60, 0)
  96     # Round down if it brings us close to the edge
  97     if start < 5:
  98         start = 0
  99     end = min(start + 140, ll)
 100     # Round up to the end if it's close
 101     if end > ll - 5:
 102         end = ll
 103     # If we are bumped all the way to the end,
 104     # make sure we still get a full 140 characters in the line
 105     if end == ll:
 106         start = max(end - 140, 0)
 107     line = line[start:end]
 108     if end < ll:
 109         # we've snipped from the end
 110         line += u" {snip}"
 111     if start > 0:
 112         # we've snipped from the beginning
 113         line = u"{snip} " + line
 114     return line

 117 def get_source_context(source, lineno, colno, context=LINES_OF_CONTEXT):
 118     if not source:
 119         return None, None, None
 120 
 121     # lineno's in JS are 1-indexed
 122     # just in case. sometimes math is hard
 123     if lineno > 0:
 124         lineno -= 1
 125 
 126     lower_bound = max(0, lineno - context)
 127     upper_bound = min(lineno + 1 + context, len(source))
 128 
 129     try:
 130         pre_context = [trim_line(x) for x in source[lower_bound:lineno]]
 131     except IndexError:
 132         pre_context = []
 133 
 134     try:
 135         context_line = trim_line(source[lineno], colno)
 136     except IndexError:
 137         context_line = ""
 138 
 139     try:
 140         post_context = [trim_line(x) for x in source[(lineno + 1) : upper_bound]]
 141     except IndexError:
 142         post_context = []
 143 
 144     return pre_context or None, context_line, post_context or None

 147 def discover_sourcemap(result):
 148     """
 149     Given a UrlResult object, attempt to discover a sourcemap URL.
 150     """
 151     # When coercing the headers returned by urllib to a dict
 152     # all keys become lowercase so they're normalized
 153     sourcemap = result.headers.get("sourcemap", result.headers.get("x-sourcemap"))
 154 
 155     if not sourcemap:
 156         parsed_body = result.body.split("\n")
 157         # Source maps are only going to exist at either the top or bottom of the document.
 158         # Technically, there isn't anything indicating *where* it should exist, so we
 159         # are generous and assume it's somewhere either in the first or last 5 lines.
 160         # If it's somewhere else in the document, you're probably doing it wrong.
 161         if len(parsed_body) > 10:
 162             possibilities = parsed_body[:5] + parsed_body[-5:]
 163         else:
 164             possibilities = parsed_body
 165 
 166         # We want to scan each line sequentially, and the last one found wins
 167         # This behavior is undocumented, but matches what Chrome and Firefox do.
 168         for line in possibilities:
 169             if line[:21] in ("//# sourceMappingURL=", "//@ sourceMappingURL="):
 170                 # We want everything AFTER the indicator, which is 21 chars long
 171                 sourcemap = line[21:].rstrip()
 172 
 173         # If we still haven't found anything, check end of last line AFTER source code.
 174         # This is not the literal interpretation of the spec, but browsers support it.
 175         # e.g. {code}//# sourceMappingURL={url}
 176         if not sourcemap:
 177             # Only look at last 300 characters to keep search space reasonable (minified
 178             # JS on a single line could be tens of thousands of chars). This is a totally
 179             # arbitrary number / best guess; most sourceMappingURLs are relative and
 180             # not very long.
 181             search_space = possibilities[-1][-300:].rstrip()
 182             match = SOURCE_MAPPING_URL_RE.search(search_space)
 183             if match:
 184                 sourcemap = match.group(1)
 185 
 186     if sourcemap:
 187         # react-native shoves a comment at the end of the
 188         # sourceMappingURL line.
 189         # For example:
 190         #  sourceMappingURL=app.js.map/*ascii:...*/
 191         # This comment is completely out of spec and no browser
 192         # would support this, but we need to strip it to make
 193         # people happy.
 194         if "/*" in sourcemap and sourcemap[-2:] == "*/":
 195             index = sourcemap.index("/*")
 196             # comment definitely shouldn't be the first character,
 197             # so let's just make sure of that.
 198             if index == 0:
 199                 raise AssertionError(
 200                     "react-native comment found at bad location: %d, %r" % (index, sourcemap)
 201                 )
 202             sourcemap = sourcemap[:index]
 203         # fix url so its absolute
 204         sourcemap = non_standard_url_join(result.url, sourcemap)
 205 
 206     return sourcemap

 209 def fetch_release_file(filename, release, dist=None):
 210     """
 211     Attempt to retrieve a release artifact from the database.
 212 
 213     Caches the result of that attempt (whether successful or not).
 214     """
 215 
 216     dist_name = dist and dist.name or None
 217     cache_key = "releasefile:v1:%s:%s" % (release.id, ReleaseFile.get_ident(filename, dist_name))
 218 
 219     logger.debug("Checking cache for release artifact %r (release_id=%s)", filename, release.id)
 220     result = cache.get(cache_key)
 221 
 222     # not in the cache (meaning we haven't checked the database recently), so check the database
 223     if result is None:
 224         filename_choices = ReleaseFile.normalize(filename)
 225         filename_idents = [ReleaseFile.get_ident(f, dist_name) for f in filename_choices]
 226 
 227         logger.debug(
 228             "Checking database for release artifact %r (release_id=%s)", filename, release.id
 229         )
 230 
 231         possible_files = list(
 232             ReleaseFile.objects.filter(
 233                 release=release, dist=dist, ident__in=filename_idents
 234             ).select_related("file")
 235         )
 236 
 237         if len(possible_files) == 0:
 238             logger.debug(
 239                 "Release artifact %r not found in database (release_id=%s)", filename, release.id
 240             )
 241             cache.set(cache_key, -1, 60)
 242             return None
 243 
 244         elif len(possible_files) == 1:
 245             releasefile = possible_files[0]
 246 
 247         else:
 248             # Pick first one that matches in priority order.
 249             # This is O(N*M) but there are only ever at most 4 things here
 250             # so not really worth optimizing.
 251             releasefile = next(
 252                 (rf for ident in filename_idents for rf in possible_files if rf.ident == ident)
 253             )
 254 
 255         logger.debug(
 256             "Found release artifact %r (id=%s, release_id=%s)", filename, releasefile.id, release.id
 257         )
 258         try:
 259             with metrics.timer("sourcemaps.release_file_read"):
 260                 with ReleaseFile.cache.getfile(releasefile) as fp:
 261                     z_body, body = compress_file(fp)
 262         except Exception:
 263             logger.error("sourcemap.compress_read_failed", exc_info=sys.exc_info())
 264             result = None
 265         else:
 266             headers = {k.lower(): v for k, v in releasefile.file.headers.items()}
 267             encoding = get_encoding_from_headers(headers)
 268             result = http.UrlResult(filename, headers, body, 200, encoding)
 269             # This will implicitly skip too large payloads. Those will be cached
 270             # on the file system by `ReleaseFile.cache`, instead.
 271             cache.set(cache_key, (headers, z_body, 200, encoding), 3600)
 272 
 273     # in the cache as an unsuccessful attempt
 274     elif result == -1:
 275         result = None
 276 
 277     # in the cache as a successful attempt, including the zipped contents of the file
 278     else:
 279         # Previous caches would be a 3-tuple instead of a 4-tuple,
 280         # so this is being maintained for backwards compatibility
 281         try:
 282             encoding = result[3]
 283         except IndexError:
 284             encoding = None
 285         result = http.UrlResult(
 286             filename, result[0], zlib.decompress(result[1]), result[2], encoding
 287         )
 288 
 289     return result

 292 def fetch_file(url, project=None, release=None, dist=None, allow_scraping=True):
 293     """
 294     Pull down a URL, returning a UrlResult object.
 295 
 296     Attempts to fetch from the database first (assuming there's a release on the
 297     event), then the internet. Caches the result of each of those two attempts
 298     separately, whether or not those attempts are successful. Used for both
 299     source files and source maps.
 300     """
 301 
 302     # If our url has been truncated, it'd be impossible to fetch
 303     # so we check for this early and bail
 304     if url[-3:] == "...":
 305         raise http.CannotFetch({"type": EventError.JS_MISSING_SOURCE, "url": http.expose_url(url)})
 306 
 307     # if we've got a release to look on, try that first (incl associated cache)
 308     if release:
 309         with metrics.timer("sourcemaps.release_file"):
 310             result = fetch_release_file(url, release, dist)
 311     else:
 312         result = None
 313 
 314     # otherwise, try the web-scraping cache and then the web itself
 315 
 316     cache_key = "source:cache:v4:%s" % (md5_text(url).hexdigest(),)
 317 
 318     if result is None:
 319         if not allow_scraping or not url.startswith(("http:", "https:")):
 320             error = {"type": EventError.JS_MISSING_SOURCE, "url": http.expose_url(url)}
 321             raise http.CannotFetch(error)
 322 
 323         logger.debug("Checking cache for url %r", url)
 324         result = cache.get(cache_key)
 325         if result is not None:
 326             # Previous caches would be a 3-tuple instead of a 4-tuple,
 327             # so this is being maintained for backwards compatibility
 328             try:
 329                 encoding = result[4]
 330             except IndexError:
 331                 encoding = None
 332             # We got a cache hit, but the body is compressed, so we
 333             # need to decompress it before handing it off
 334             result = http.UrlResult(
 335                 result[0], result[1], zlib.decompress(result[2]), result[3], encoding
 336             )
 337 
 338     if result is None:
 339         headers = {}
 340         verify_ssl = False
 341         if project and is_valid_origin(url, project=project):
 342             verify_ssl = bool(project.get_option("sentry:verify_ssl", False))
 343             token = project.get_option("sentry:token")
 344             if token:
 345                 token_header = project.get_option("sentry:token_header") or "X-Sentry-Token"
 346                 headers[token_header] = token
 347 
 348         with metrics.timer("sourcemaps.fetch"):
 349             result = http.fetch_file(url, headers=headers, verify_ssl=verify_ssl)
 350             z_body = zlib.compress(result.body)
 351             cache.set(
 352                 cache_key,
 353                 (url, result.headers, z_body, result.status, result.encoding),
 354                 get_max_age(result.headers),
 355             )
 356 
 357     # If we did not get a 200 OK we just raise a cannot fetch here.
 358     if result.status != 200:
 359         raise http.CannotFetch(
 360             {
 361                 "type": EventError.FETCH_INVALID_HTTP_CODE,
 362                 "value": result.status,
 363                 "url": http.expose_url(url),
 364             }
 365         )
 366 
 367     # Make sure the file we're getting back is six.binary_type. The only
 368     # reason it'd not be binary would be from old cached blobs, so
 369     # for compatibility with current cached files, let's coerce back to
 370     # binary and say utf8 encoding.
 371     if not isinstance(result.body, six.binary_type):
 372         try:
 373             result = http.UrlResult(
 374                 result.url,
 375                 result.headers,
 376                 result.body.encode("utf8"),
 377                 result.status,
 378                 result.encoding,
 379             )
 380         except UnicodeEncodeError:
 381             error = {
 382                 "type": EventError.FETCH_INVALID_ENCODING,
 383                 "value": "utf8",
 384                 "url": http.expose_url(url),
 385             }
 386             raise http.CannotFetch(error)
 387 
 388     # For JavaScript files, check if content is something other than JavaScript/JSON (i.e. HTML)
 389     # NOTE: possible to have JS files that don't actually end w/ ".js", but
 390     # this should catch 99% of cases
 391     if url.endswith(".js"):
 392         # Check if response is HTML by looking if the first non-whitespace character is an open tag ('<').
 393         # This cannot parse as valid JS/JSON.
 394         # NOTE: not relying on Content-Type header because apps often don't set this correctly
 395         # Discard leading whitespace (often found before doctype)
 396         body_start = result.body[:20].lstrip()
 397 
 398         if body_start[:1] == u"<":
 399             error = {"type": EventError.JS_INVALID_CONTENT, "url": url}
 400             raise http.CannotFetch(error)
 401 
 402     return result
 403

 405 def get_max_age(headers):
 406     cache_control = headers.get("cache-control")
 407     max_age = CACHE_CONTROL_MIN
 408 
 409     if cache_control:
 410         match = CACHE_CONTROL_RE.search(cache_control)
 411         if match:
 412             max_age = max(CACHE_CONTROL_MIN, int(match.group(1)))
 413     return min(max_age, CACHE_CONTROL_MAX)

 416 def fetch_sourcemap(url, project=None, release=None, dist=None, allow_scraping=True):
 417     if is_data_uri(url):
 418         try:
 419             body = base64.b64decode(
 420                 url[BASE64_PREAMBLE_LENGTH:] + (b"=" * (-(len(url) - BASE64_PREAMBLE_LENGTH) % 4))
 421             )
 422         except TypeError as e:
 423             raise UnparseableSourcemap({"url": "<base64>", "reason": six.text_type(e)})
 424     else:
 425         # look in the database and, if not found, optionally try to scrape the web
 426         result = fetch_file(
 427             url, project=project, release=release, dist=dist, allow_scraping=allow_scraping
 428         )
 429         body = result.body
 430     try:
 431         return SourceMapView.from_json_bytes(body)
 432     except Exception as exc:
 433         # This is in debug because the product shows an error already.
 434         logger.debug(six.text_type(exc), exc_info=True)
 435         raise UnparseableSourcemap({"url": http.expose_url(url)})

 442 def generate_module(src):
 443     """
 444     Converts a url into a made-up module name by doing the following:
 445      * Extract just the path name ignoring querystrings
 446      * Trimming off the initial /
 447      * Trimming off the file extension
 448      * Removes off useless folder prefixes
 449 
 450     e.g. http://google.com/js/v1.0/foo/bar/baz.js -> foo/bar/baz
 451     """
 452     if not src:
 453         return UNKNOWN_MODULE
 454 
 455     filename, ext = splitext(urlsplit(src).path)
 456     if filename.endswith(".min"):
 457         filename = filename[:-4]
 458 
 459     # TODO(dcramer): replace CLEAN_MODULE_RE with tokenizer completely
 460     tokens = filename.split("/")
 461     for idx, token in enumerate(tokens):
 462         # a SHA
 463         if VERSION_RE.match(token):
 464             return "/".join(tokens[idx + 1 :])
 465 
 466     return CLEAN_MODULE_RE.sub("", filename) or UNKNOWN_MODULE

 473 class JavaScriptStacktraceProcessor(StacktraceProcessor):
 474     """
 475     Attempts to fetch source code for javascript frames.
 476 
 477     Frames must match the following requirements:
 478 
 479     - lineno >= 0
 480     - colno >= 0
 481     - abs_path is the HTTP URI to the source
 482     - context_line is empty
 483 
 484     Mutates the input ``data`` with expanded context if available.
 485     """
 486 
 487     def __init__(self, *args, **kwargs):
 488         StacktraceProcessor.__init__(self, *args, **kwargs)
 489 
 490         # Make sure we only fetch organization from cache
 491         # We don't need to persist it back since we don't want
 492         # to bloat the Event object.
 493         organization = getattr(self.project, "_organization_cache", None)
 494         if not organization:
 495             organization = Organization.objects.get_from_cache(id=self.project.organization_id)
 496 
 497         self.max_fetches = MAX_RESOURCE_FETCHES
 498         self.allow_scraping = organization.get_option(
 499             "sentry:scrape_javascript", True
 500         ) is not False and self.project.get_option("sentry:scrape_javascript", True)
 501         self.fetch_count = 0
 502         self.sourcemaps_touched = set()
 503 
 504         # cache holding mangled code, original code, and errors associated with
 505         # each abs_path in the stacktrace
 506         self.cache = SourceCache()
 507 
 508         # cache holding source URLs, corresponding source map URLs, and source map contents
 509         self.sourcemaps = SourceMapCache()
 510 
 511         self.release = None
 512         self.dist = None
 513 
 514     def get_stacktraces(self, data):
 515         exceptions = get_path(data, "exception", "values", filter=True, default=())
 516         stacktraces = [e["stacktrace"] for e in exceptions if e.get("stacktrace")]
 517 
 518         if "stacktrace" in data:
 519             stacktraces.append(data["stacktrace"])
 520 
 521         return [(s, Stacktrace.to_python(s)) for s in stacktraces]
 522 
 523     def get_valid_frames(self):
 524         # build list of frames that we can actually grab source for
 525         frames = []
 526         for info in self.stacktrace_infos:
 527             frames.extend(get_path(info.stacktrace, "frames", filter=is_valid_frame, default=()))
 528         return frames
 529 
 530     def preprocess_step(self, processing_task):
 531         frames = self.get_valid_frames()
 532         if not frames:
 533             logger.debug(
 534                 "Event %r has no frames with enough context to " "fetch remote source",
 535                 self.data["event_id"],
 536             )
 537             return False
 538 
 539         self.release = self.get_release(create=True)
 540         if self.data.get("dist") and self.release:
 541             self.dist = self.release.get_dist(self.data["dist"])
 542 
 543         self.populate_source_cache(frames)
 544         return True
 545 
 546     def handles_frame(self, frame, stacktrace_info):
 547         platform = frame.get("platform") or self.data.get("platform")
 548         return settings.SENTRY_SCRAPE_JAVASCRIPT_CONTEXT and platform in ("javascript", "node")
 549 
 550     def preprocess_frame(self, processable_frame):
 551         # Stores the resolved token.  This is used to cross refer to other
 552         # frames for function name resolution by call site.
 553         processable_frame.data = {"token": None}
 554 
 555     def process_frame(self, processable_frame, processing_task):
 556         """
 557         Attempt to demangle the given frame.
 558         """
 559 
 560         frame = processable_frame.frame
 561         token = None
 562 
 563         cache = self.cache
 564         sourcemaps = self.sourcemaps
 565         all_errors = []
 566         sourcemap_applied = False
 567 
 568         # can't demangle if there's no filename or line number present
 569         if not frame.get("abs_path") or not frame.get("lineno"):
 570             return
 571 
 572         # also can't demangle node's internal modules
 573         # therefore we only process user-land frames (starting with /)
 574         # or those created by bundle/webpack internals
 575         if self.data.get("platform") == "node" and not frame.get("abs_path").startswith(
 576             ("/", "app:", "webpack:")
 577         ):
 578             return
 579 
 580         errors = cache.get_errors(frame["abs_path"])
 581         if errors:
 582             all_errors.extend(errors)
 583 
 584         # This might fail but that's okay, we try with a different path a
 585         # bit later down the road.
 586         source = self.get_sourceview(frame["abs_path"])
 587 
 588         in_app = None
 589         new_frame = dict(frame)
 590         raw_frame = dict(frame)
 591 
 592         sourcemap_url, sourcemap_view = sourcemaps.get_link(frame["abs_path"])
 593         self.sourcemaps_touched.add(sourcemap_url)
 594         if sourcemap_view and frame.get("colno") is None:
 595             all_errors.append(
 596                 {"type": EventError.JS_NO_COLUMN, "url": http.expose_url(frame["abs_path"])}
 597             )
 598         elif sourcemap_view:
 599             if is_data_uri(sourcemap_url):
 600                 sourcemap_label = frame["abs_path"]
 601             else:
 602                 sourcemap_label = sourcemap_url
 603 
 604             sourcemap_label = http.expose_url(sourcemap_label)
 605 
 606             if frame.get("function"):
 607                 minified_function_name = frame["function"]
 608                 minified_source = self.get_sourceview(frame["abs_path"])
 609             else:
 610                 minified_function_name = minified_source = None
 611 
 612             try:
 613                 # Errors are 1-indexed in the frames, so we need to -1 to get
 614                 # zero-indexed value from tokens.
 615                 assert frame["lineno"] > 0, "line numbers are 1-indexed"
 616                 token = sourcemap_view.lookup(
 617                     frame["lineno"] - 1, frame["colno"] - 1, minified_function_name, minified_source
 618                 )
 619             except Exception:
 620                 token = None
 621                 all_errors.append(
 622                     {
 623                         "type": EventError.JS_INVALID_SOURCEMAP_LOCATION,
 624                         "column": frame.get("colno"),
 625                         "row": frame.get("lineno"),
 626                         "source": frame["abs_path"],
 627                         "sourcemap": sourcemap_label,
 628                     }
 629                 )
 630 
 631             # persist the token so that we can find it later
 632             processable_frame.data["token"] = token
 633 
 634             # Store original data in annotation
 635             new_frame["data"] = dict(frame.get("data") or {}, sourcemap=sourcemap_label)
 636 
 637             sourcemap_applied = True
 638 
 639             if token is not None:
 640                 abs_path = non_standard_url_join(sourcemap_url, token.src)
 641 
 642                 logger.debug(
 643                     "Mapping compressed source %r to mapping in %r", frame["abs_path"], abs_path
 644                 )
 645                 source = self.get_sourceview(abs_path)
 646 
 647             if source is None:
 648                 errors = cache.get_errors(abs_path)
 649                 if errors:
 650                     all_errors.extend(errors)
 651                 else:
 652                     all_errors.append(
 653                         {"type": EventError.JS_MISSING_SOURCE, "url": http.expose_url(abs_path)}
 654                     )
 655 
 656             if token is not None:
 657                 # the tokens are zero indexed, so offset correctly
 658                 new_frame["lineno"] = token.src_line + 1
 659                 new_frame["colno"] = token.src_col + 1
 660 
 661                 # Try to use the function name we got from symbolic
 662                 original_function_name = token.function_name
 663 
 664                 # In the ideal case we can use the function name from the
 665                 # frame and the location to resolve the original name
 666                 # through the heuristics in our sourcemap library.
 667                 if original_function_name is None:
 668                     last_token = None
 669 
 670                     # Find the previous token for function name handling as a
 671                     # fallback.
 672                     if (
 673                         processable_frame.previous_frame
 674                         and processable_frame.previous_frame.processor is self
 675                     ):
 676                         last_token = processable_frame.previous_frame.data.get("token")
 677                         if last_token:
 678                             original_function_name = last_token.name
 679 
 680                 if original_function_name is not None:
 681                     new_frame["function"] = original_function_name
 682 
 683                 filename = token.src
 684                 # special case webpack support
 685                 # abs_path will always be the full path with webpack:/// prefix.
 686                 # filename will be relative to that
 687                 if abs_path.startswith("webpack:"):
 688                     filename = abs_path
 689                     # webpack seems to use ~ to imply "relative to resolver root"
 690                     # which is generally seen for third party deps
 691                     # (i.e. node_modules)
 692                     if "/~/" in filename:
 693                         filename = "~/" + abs_path.split("/~/", 1)[-1]
 694                     else:
 695                         filename = filename.split("webpack:///", 1)[-1]
 696 
 697                     # As noted above:
 698                     # * [js/node] '~/' means they're coming from node_modules, so these are not app dependencies
 699                     # * [node] sames goes for `./node_modules/` and '../node_modules/', which is used when bundling node apps
 700                     # * [node] and webpack, which includes it's own code to bootstrap all modules and its internals
 701                     #   eg. webpack:///webpack/bootstrap, webpack:///external
 702                     if (
 703                         filename.startswith("~/")
 704                         or "/node_modules/" in filename
 705                         or not filename.startswith("./")
 706                     ):
 707                         in_app = False
 708                     # And conversely, local dependencies start with './'
 709                     elif filename.startswith("./"):
 710                         in_app = True
 711                     # We want to explicitly generate a webpack module name
 712                     new_frame["module"] = generate_module(filename)
 713 
 714                 # while you could technically use a subpath of 'node_modules' for your libraries,
 715                 # it would be an extremely complicated decision and we've not seen anyone do it
 716                 # so instead we assume if node_modules is in the path its part of the vendored code
 717                 elif "/node_modules/" in abs_path:
 718                     in_app = False
 719 
 720                 if abs_path.startswith("app:"):
 721                     if filename and NODE_MODULES_RE.search(filename):
 722                         in_app = False
 723                     else:
 724                         in_app = True
 725 
 726                 new_frame["abs_path"] = abs_path
 727                 new_frame["filename"] = filename
 728                 if not frame.get("module") and abs_path.startswith(
 729                     ("http:", "https:", "webpack:", "app:")
 730                 ):
 731                     new_frame["module"] = generate_module(abs_path)
 732 
 733         elif sourcemap_url:
 734             new_frame["data"] = dict(
 735                 new_frame.get("data") or {}, sourcemap=http.expose_url(sourcemap_url)
 736             )
 737 
 738         # TODO: theoretically a minified source could point to
 739         # another mapped, minified source
 740         changed_frame = self.expand_frame(new_frame, source=source)
 741 
 742         # If we did not manage to match but we do have a line or column
 743         # we want to report an error here.
 744         if not new_frame.get("context_line") and source and new_frame.get("colno") is not None:
 745             all_errors.append(
 746                 {
 747                     "type": EventError.JS_INVALID_SOURCEMAP_LOCATION,
 748                     "column": new_frame["colno"],
 749                     "row": new_frame["lineno"],
 750                     "source": new_frame["abs_path"],
 751                 }
 752             )
 753 
 754         changed_raw = sourcemap_applied and self.expand_frame(raw_frame)
 755 
 756         if sourcemap_applied or all_errors or changed_frame or changed_raw:
 757             # In case we are done processing, we iterate over all errors that we got
 758             # and we filter out all `JS_MISSING_SOURCE` errors since we consider if we have
 759             # a `context_line` we have a symbolicated frame and we don't need to show the error
 760             has_context_line = bool(new_frame.get("context_line"))
 761             if has_context_line:
 762                 all_errors[:] = [
 763                     x for x in all_errors if x.get("type") is not EventError.JS_MISSING_SOURCE
 764                 ]
 765 
 766             if in_app is not None:
 767                 new_frame["in_app"] = in_app
 768                 raw_frame["in_app"] = in_app
 769 
 770             new_frames = [new_frame]
 771             raw_frames = [raw_frame] if changed_raw else None
 772             return new_frames, raw_frames, all_errors
 773 
 774     def expand_frame(self, frame, source=None):
 775         """
 776         Mutate the given frame to include pre- and post-context lines.
 777         """
 778 
 779         if frame.get("lineno") is not None:
 780             if source is None:
 781                 source = self.get_sourceview(frame["abs_path"])
 782                 if source is None:
 783                     logger.debug("No source found for %s", frame["abs_path"])
 784                     return False
 785 
 786             frame["pre_context"], frame["context_line"], frame["post_context"] = get_source_context(
 787                 source=source, lineno=frame["lineno"], colno=frame.get("colno") or 0
 788             )
 789             return True
 790         return False
 791 
 792     def get_sourceview(self, filename):
 793         if filename not in self.cache:
 794             self.cache_source(filename)
 795         return self.cache.get(filename)
 796 
 797     def cache_source(self, filename):
 798         """
 799         Look for and (if found) cache a source file and its associated source
 800         map (if any).
 801         """
 802 
 803         sourcemaps = self.sourcemaps
 804         cache = self.cache
 805 
 806         self.fetch_count += 1
 807 
 808         if self.fetch_count > self.max_fetches:
 809             cache.add_error(filename, {"type": EventError.JS_TOO_MANY_REMOTE_SOURCES})
 810             return
 811 
 812         # TODO: respect cache-control/max-age headers to some extent
 813         logger.debug("Attempting to cache source %r", filename)
 814         try:
 815             # this both looks in the database and tries to scrape the internet
 816             result = fetch_file(
 817                 filename,
 818                 project=self.project,
 819                 release=self.release,
 820                 dist=self.dist,
 821                 allow_scraping=self.allow_scraping,
 822             )
 823         except http.BadSource as exc:
 824             # most people don't upload release artifacts for their third-party libraries,
 825             # so ignore missing node_modules files
 826             if exc.data["type"] == EventError.JS_MISSING_SOURCE and "node_modules" in filename:
 827                 pass
 828             else:
 829                 cache.add_error(filename, exc.data)
 830 
 831             # either way, there's no more for us to do here, since we don't have
 832             # a valid file to cache
 833             return
 834         cache.add(filename, result.body, result.encoding)
 835         cache.alias(result.url, filename)
 836 
 837         sourcemap_url = discover_sourcemap(result)
 838         if not sourcemap_url:
 839             return
 840 
 841         logger.debug(
 842             "Found sourcemap URL %r for minified script %r", sourcemap_url[:256], result.url
 843         )
 844         sourcemaps.link(filename, sourcemap_url)
 845         if sourcemap_url in sourcemaps:
 846             return
 847 
 848         # pull down sourcemap
 849         try:
 850             sourcemap_view = fetch_sourcemap(
 851                 sourcemap_url,
 852                 project=self.project,
 853                 release=self.release,
 854                 dist=self.dist,
 855                 allow_scraping=self.allow_scraping,
 856             )
 857         except http.BadSource as exc:
 858             # we don't perform the same check here as above, because if someone has
 859             # uploaded a node_modules file, which has a sourceMappingURL, they
 860             # presumably would like it mapped (and would like to know why it's not
 861             # working, if that's the case). If they're not looking for it to be
 862             # mapped, then they shouldn't be uploading the source file in the
 863             # first place.
 864             cache.add_error(filename, exc.data)
 865             return
 866 
 867         sourcemaps.add(sourcemap_url, sourcemap_view)
 868 
 869         # cache any inlined sources
 870         for src_id, source_name in sourcemap_view.iter_sources():
 871             source_view = sourcemap_view.get_sourceview(src_id)
 872             if source_view is not None:
 873                 self.cache.add(non_standard_url_join(sourcemap_url, source_name), source_view)
 874 
 875     def populate_source_cache(self, frames):
 876         """
 877         Fetch all sources that we know are required (being referenced directly
 878         in frames).
 879         """
 880         pending_file_list = set()
 881         for f in frames:
 882             # We can't even attempt to fetch source if abs_path is None
 883             if f.get("abs_path") is None:
 884                 continue
 885             # tbh not entirely sure how this happens, but raven-js allows this
 886             # to be caught. I think this comes from dev consoles and whatnot
 887             # where there is no page. This just bails early instead of exposing
 888             # a fetch error that may be confusing.
 889             if f["abs_path"] == "<anonymous>":
 890                 continue
 891             # we cannot fetch any other files than those uploaded by user
 892             if self.data.get("platform") == "node" and not f.get("abs_path").startswith("app:"):
 893                 continue
 894             pending_file_list.add(f["abs_path"])
 895 
 896         for idx, filename in enumerate(pending_file_list):
 897             self.cache_source(filename=filename)
 898 
 899     def close(self):
 900         StacktraceProcessor.close(self)
 901         if self.sourcemaps_touched:
 902             metrics.incr(
 903                 "sourcemaps.processed", amount=len(self.sourcemaps_touched), skip_internal=True
 904             )
