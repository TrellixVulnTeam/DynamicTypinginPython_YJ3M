  10 def is_utf8(codec):
  11     name = codec_lookup(codec).name
  12     return name in ("utf-8", "ascii")

  15 class SourceCache(object):
  16     def __init__(self):
  17         self._cache = {}
  18         self._errors = {}
  19         self._aliases = {}
  20 
  21     def __contains__(self, url):
  22         url = self._get_canonical_url(url)
  23         return url in self._cache
  24 
  25     def _get_canonical_url(self, url):
  26         if url in self._aliases:
  27             url = self._aliases[url]
  28         return url
  29 
  30     def get(self, url):
  31         return self._cache.get(self._get_canonical_url(url))
  32 
  33     def get_errors(self, url):
  34         url = self._get_canonical_url(url)
  35         return self._errors.get(url, [])
  36 
  37     def alias(self, alias, target):
  38         if alias != target:
  39             self._aliases[alias] = target
  40 
  41     def add(self, url, source, encoding=None):
  42         url = self._get_canonical_url(url)
  43 
  44         if not isinstance(source, SourceView):
  45             if isinstance(source, text_type):
  46                 source = source.encode("utf-8")
  47             # If an encoding is provided and it's not utf-8 compatible
  48             # we try to re-encoding the source and create a source view
  49             # from it.
  50             elif encoding is not None and not is_utf8(encoding):
  51                 try:
  52                     source = source.decode(encoding).encode("utf-8")
  53                 except UnicodeError:
  54                     pass
  55             source = SourceView.from_bytes(source)
  56         self._cache[url] = source
  57 
  58     def add_error(self, url, error):
  59         url = self._get_canonical_url(url)
  60         self._errors.setdefault(url, [])
  61         self._errors[url].append(error)

  64 class SourceMapCache(object):
  65     """
  66     Stores mappings between
  67 
  68         - the url of a file to be demangled and the url of its associated
  69           source map, and
  70         - a source map's url and the map's contents.
  71     """
  72 
  73     def __init__(self):
  74         self._cache = {}
  75         self._mapping = {}
  76 
  77     def __contains__(self, sourcemap_url):
  78         return sourcemap_url in self._cache
  79 
  80     def link(self, url, sourcemap_url):
  81         self._mapping[url] = sourcemap_url
  82 
  83     def add(self, sourcemap_url, sourcemap_view):
  84         self._cache[sourcemap_url] = sourcemap_view
  85 
  86     def get(self, sourcemap_url):
  87         return self._cache.get(sourcemap_url)
  88 
  89     def get_link(self, url):
  90         sourcemap_url = self._mapping.get(url)
  91         if sourcemap_url:
  92             sourcemap = self.get(sourcemap_url)
  93             return (sourcemap_url, sourcemap)
  94         return (None, None)
