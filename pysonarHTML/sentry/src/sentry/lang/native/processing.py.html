  35 def _merge_frame(new_frame, symbolicated):
  36     if symbolicated.get("function"):
  37         raw_func = trim(symbolicated["function"], 256)
  38         func = trim(trim_function_name(symbolicated["function"], "native"), 256)
  39 
  40         # if function and raw function match, we can get away without
  41         # storing a raw function
  42         if func == raw_func:
  43             new_frame["function"] = raw_func
  44         # otherwise we store both
  45         else:
  46             new_frame["raw_function"] = raw_func
  47             new_frame["function"] = func
  48     if symbolicated.get("instruction_addr"):
  49         new_frame["instruction_addr"] = symbolicated["instruction_addr"]
  50     if symbolicated.get("symbol"):
  51         new_frame["symbol"] = symbolicated["symbol"]
  52     if symbolicated.get("abs_path"):
  53         new_frame["abs_path"] = symbolicated["abs_path"]
  54         new_frame["filename"] = posixpath.basename(symbolicated["abs_path"])
  55     if symbolicated.get("filename"):
  56         new_frame["filename"] = symbolicated["filename"]
  57     if symbolicated.get("lineno"):
  58         new_frame["lineno"] = symbolicated["lineno"]
  59     if symbolicated.get("colno"):
  60         new_frame["colno"] = symbolicated["colno"]
  61     if symbolicated.get("package"):
  62         new_frame["package"] = symbolicated["package"]
  63     if symbolicated.get("trust"):
  64         new_frame["trust"] = symbolicated["trust"]
  65     if symbolicated.get("pre_context"):
  66         new_frame["pre_context"] = symbolicated["pre_context"]
  67     if symbolicated.get("context_line") is not None:
  68         new_frame["context_line"] = symbolicated["context_line"]
  69     if symbolicated.get("post_context"):
  70         new_frame["post_context"] = symbolicated["post_context"]
  71     if symbolicated.get("status"):
  72         frame_meta = new_frame.setdefault("data", {})
  73         frame_meta["symbolicator_status"] = symbolicated["status"]

  76 def _handle_image_status(status, image, sdk_info, data):
  77     if status in ("found", "unused"):
  78         return
  79     elif status == "missing":
  80         package = image.get("code_file")
  81         # TODO(mitsuhiko): This check seems wrong?  This call seems to
  82         # mirror the one in the ios symbol server support.  If we change
  83         # one we need to change the other.
  84         if not package or is_known_third_party(package, sdk_info=sdk_info):
  85             return
  86 
  87         if is_optional_package(package, sdk_info=sdk_info):
  88             error = SymbolicationFailed(type=EventError.NATIVE_MISSING_OPTIONALLY_BUNDLED_DSYM)
  89         else:
  90             error = SymbolicationFailed(type=EventError.NATIVE_MISSING_DSYM)
  91     elif status == "malformed":
  92         error = SymbolicationFailed(type=EventError.NATIVE_BAD_DSYM)
  93     elif status == "too_large":
  94         error = SymbolicationFailed(type=EventError.FETCH_TOO_LARGE)
  95     elif status == "fetching_failed":
  96         error = SymbolicationFailed(type=EventError.FETCH_GENERIC_ERROR)
  97     elif status == "other":
  98         error = SymbolicationFailed(type=EventError.UNKNOWN_ERROR)
  99     else:
 100         logger.error("Unknown status: %s", status)
 101         return
 102 
 103     error.image_arch = image.get("arch")
 104     error.image_path = image.get("code_file")
 105     error.image_name = image_name(image.get("code_file"))
 106     error.image_uuid = image.get("debug_id")
 107 
 108     write_error(error, data)

 111 def _merge_image(raw_image, complete_image, sdk_info, data):
 112     statuses = set()
 113 
 114     # Set image data from symbolicator as symbolicator might know more
 115     # than the SDK, especially for minidumps
 116     for k, v in six.iteritems(complete_image):
 117         if k in IMAGE_STATUS_FIELDS:
 118             statuses.add(v)
 119         if not (v is None or (v == "unknown" and k in ("arch", "type"))):
 120             raw_image[k] = v
 121 
 122     for status in set(statuses):
 123         _handle_image_status(status, raw_image, sdk_info, data)

 126 def _handle_response_status(event_data, response_json):
 127     if not response_json:
 128         error = SymbolicationFailed(type=EventError.NATIVE_INTERNAL_FAILURE)
 129     elif response_json["status"] == "completed":
 130         return True
 131     elif response_json["status"] == "failed":
 132         error = SymbolicationFailed(
 133             message=response_json.get("message") or None, type=EventError.NATIVE_SYMBOLICATOR_FAILED
 134         )
 135     else:
 136         logger.error("Unexpected symbolicator status: %s", response_json["status"])
 137         error = SymbolicationFailed(type=EventError.NATIVE_INTERNAL_FAILURE)
 138 
 139     write_error(error, event_data)

 142 def _merge_system_info(data, system_info):
 143     set_path(data, "contexts", "os", "type", value="os")  # Required by "get_sdk_from_event"
 144 
 145     os_name = system_info.get("os_name")
 146     os_version = system_info.get("os_version")
 147     os_build = system_info.get("os_build")
 148 
 149     if os_version:
 150         setdefault_path(data, "contexts", "os", "version", value=os_version)
 151     if os_build:
 152         setdefault_path(data, "contexts", "os", "build", value=os_build)
 153     if os_name and not os_version and not os_build:
 154         setdefault_path(data, "contexts", "os", "raw_description", value=os_name)
 155     elif os_name:
 156         setdefault_path(data, "contexts", "os", "name", value=os_name)
 157 
 158     set_path(data, "contexts", "device", "type", value="device")
 159     setdefault_path(data, "contexts", "device", "arch", value=system_info.get("cpu_arch"))
 160 
 161     device_model = system_info.get("device_model")
 162     if device_model:
 163         setdefault_path(data, "contexts", "device", "model", value=device_model)

 166 def _merge_full_response(data, response):
 167     data["platform"] = "native"
 168     if response.get("crashed") is not None:
 169         data["level"] = "fatal" if response["crashed"] else "info"
 170 
 171     validate_and_set_timestamp(data, response.get("timestamp"))
 172 
 173     if response.get("system_info"):
 174         _merge_system_info(data, response["system_info"])
 175 
 176     sdk_info = get_sdk_from_event(data)
 177 
 178     images = []
 179     set_path(data, "debug_meta", "images", value=images)
 180 
 181     for complete_image in response["modules"]:
 182         image = {}
 183         _merge_image(image, complete_image, sdk_info, data)
 184         images.append(image)
 185 
 186     # Extract the crash reason and infos
 187     data_exception = get_path(data, "exception", "values", 0)
 188     if response.get("assertion"):
 189         data_exception["value"] = "Assertion Error: %s" % (response["assertion"],)
 190     elif response.get("crash_details"):
 191         data_exception["value"] = response["crash_details"]
 192     elif response.get("crash_reason"):
 193         data_exception["value"] = "Fatal Error: %s" % (response["crash_reason"],)
 194     else:
 195         # We're merging a full response, so there was no initial payload
 196         # submitted. Assuming that this still contains the placeholder, remove
 197         # it rather than showing a default value.
 198         data_exception.pop("value", None)
 199 
 200     if response.get("crash_reason"):
 201         data_exception["type"] = response["crash_reason"]
 202 
 203     data_threads = []
 204     if response["stacktraces"]:
 205         data["threads"] = {"values": data_threads}
 206     else:
 207         error = SymbolicationFailed(
 208             message="minidump has no thread list", type=EventError.NATIVE_SYMBOLICATOR_FAILED
 209         )
 210         write_error(error, data)
 211 
 212     for complete_stacktrace in response["stacktraces"]:
 213         is_requesting = complete_stacktrace.get("is_requesting")
 214         thread_id = complete_stacktrace.get("thread_id")
 215 
 216         data_thread = {"id": thread_id, "crashed": is_requesting}
 217         data_threads.append(data_thread)
 218 
 219         if is_requesting:
 220             data_exception["thread_id"] = thread_id
 221             data_stacktrace = data_exception.setdefault("stacktrace", {})
 222             data_stacktrace["frames"] = []
 223         else:
 224             data_thread["stacktrace"] = data_stacktrace = {"frames": []}
 225 
 226         if complete_stacktrace.get("registers"):
 227             data_stacktrace["registers"] = complete_stacktrace["registers"]
 228 
 229         for complete_frame in reversed(complete_stacktrace["frames"]):
 230             new_frame = {}
 231             _merge_frame(new_frame, complete_frame)
 232             data_stacktrace["frames"].append(new_frame)

 235 def process_minidump(data):
 236     project = Project.objects.get_from_cache(id=data["project"])
 237 
 238     minidump = get_event_attachment(data, MINIDUMP_ATTACHMENT_TYPE)
 239     if not minidump:
 240         logger.error("Missing minidump for minidump event")
 241         return
 242 
 243     symbolicator = Symbolicator(project=project, event_id=data["event_id"])
 244 
 245     response = symbolicator.process_minidump(minidump.data)
 246 
 247     if _handle_response_status(data, response):
 248         _merge_full_response(data, response)
 249 
 250     return data

 271 def _handles_frame(data, frame):
 272     if not frame:
 273         return False
 274 
 275     if get_path(frame, "data", "symbolicator_status") is not None:
 276         return False
 277 
 278     # TODO: Consider ignoring platform
 279     platform = frame.get("platform") or data.get("platform")
 280     return is_native_platform(platform) and "instruction_addr" in frame

 283 def process_payload(data):
 284     project = Project.objects.get_from_cache(id=data["project"])
 285 
 286     symbolicator = Symbolicator(project=project, event_id=data["event_id"])
 287 
 288     stacktrace_infos = [
 289         stacktrace
 290         for stacktrace in find_stacktraces_in_data(data)
 291         if any(is_native_platform(x) for x in stacktrace.platforms)
 292     ]
 293 
 294     stacktraces = [
 295         {
 296             "registers": sinfo.stacktrace.get("registers") or {},
 297             "frames": [
 298                 f for f in reversed(sinfo.stacktrace.get("frames") or ()) if _handles_frame(data, f)
 299             ],
 300         }
 301         for sinfo in stacktrace_infos
 302     ]
 303 
 304     if not any(stacktrace["frames"] for stacktrace in stacktraces):
 305         return
 306 
 307     modules = native_images_from_data(data)
 308     signal = signal_from_data(data)
 309 
 310     response = symbolicator.process_payload(stacktraces=stacktraces, modules=modules, signal=signal)
 311 
 312     if not _handle_response_status(data, response):
 313         return data
 314 
 315     assert len(modules) == len(response["modules"]), (modules, response)
 316 
 317     sdk_info = get_sdk_from_event(data)
 318 
 319     for raw_image, complete_image in zip(modules, response["modules"]):
 320         _merge_image(raw_image, complete_image, sdk_info, data)
 321 
 322     assert len(stacktraces) == len(response["stacktraces"]), (stacktraces, response)
 323 
 324     for sinfo, complete_stacktrace in zip(stacktrace_infos, response["stacktraces"]):
 325         complete_frames_by_idx = {}
 326         for complete_frame in complete_stacktrace.get("frames") or ():
 327             complete_frames_by_idx.setdefault(complete_frame["original_index"], []).append(
 328                 complete_frame
 329             )
 330 
 331         new_frames = []
 332         native_frames_idx = 0
 333 
 334         for raw_frame in reversed(sinfo.stacktrace["frames"]):
 335             if not _handles_frame(data, raw_frame):
 336                 new_frames.append(raw_frame)
 337                 continue
 338 
 339             for complete_frame in complete_frames_by_idx.get(native_frames_idx) or ():
 340                 merged_frame = dict(raw_frame)
 341                 _merge_frame(merged_frame, complete_frame)
 342                 if merged_frame.get("package"):
 343                     raw_frame["package"] = merged_frame["package"]
 344                 new_frames.append(merged_frame)
 345 
 346             native_frames_idx += 1
 347 
 348         if sinfo.container is not None and native_frames_idx > 0:
 349             sinfo.container["raw_stacktrace"] = {
 350                 "frames": list(sinfo.stacktrace["frames"]),
 351                 "registers": sinfo.stacktrace.get("registers"),
 352             }
 353 
 354         new_frames.reverse()
 355         sinfo.stacktrace["frames"] = new_frames
 356 
 357     return data

 360 def get_symbolication_function(data):
 361     if is_minidump_event(data):
 362         return process_minidump
 363     elif is_applecrashreport_event(data):
 364         return process_applecrashreport
 365     elif is_native_event(data):
 366         return process_payload
