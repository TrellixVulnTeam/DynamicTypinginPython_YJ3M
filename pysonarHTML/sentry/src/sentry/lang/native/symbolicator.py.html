 104 class Symbolicator(object):
 105     def __init__(self, project, event_id):
 106         symbolicator_options = options.get("symbolicator.options")
 107         base_url = symbolicator_options["url"].rstrip("/")
 108         assert base_url
 109 
 110         self.sess = SymbolicatorSession(
 111             url=base_url,
 112             project_id=six.text_type(project.id),
 113             event_id=six.text_type(event_id),
 114             timeout=settings.SYMBOLICATOR_POLL_TIMEOUT,
 115             sources=get_sources_for_project(project),
 116         )
 117 
 118         self.task_id_cache_key = _task_id_cache_key_for_event(project.id, event_id)
 119 
 120     def _process(self, create_task):
 121         task_id = default_cache.get(self.task_id_cache_key)
 122         json_response = None
 123 
 124         with self.sess:
 125             try:
 126                 if task_id:
 127                     # Processing has already started and we need to poll
 128                     # symbolicator for an update. This in turn may put us back into
 129                     # the queue.
 130                     json_response = self.sess.query_task(task_id)
 131 
 132                 if json_response is None:
 133                     # This is a new task, so we compute all request parameters
 134                     # (potentially expensive if we need to pull minidumps), and then
 135                     # upload all information to symbolicator. It will likely not
 136                     # have a response ready immediately, so we start polling after
 137                     # some timeout.
 138                     json_response = create_task()
 139             except ServiceUnavailable:
 140                 # 503 can indicate that symbolicator is restarting. Wait for a
 141                 # reboot, then try again. This overrides the default behavior of
 142                 # retrying after just a second.
 143                 #
 144                 # If there is no response attached, it's a connection error.
 145                 raise RetrySymbolication(retry_after=10)
 146 
 147             metrics.incr(
 148                 "events.symbolicator.response",
 149                 tags={"response": json_response.get("status") or "null"},
 150             )
 151 
 152             # Symbolication is still in progress. Bail out and try again
 153             # after some timeout. Symbolicator keeps the response for the
 154             # first one to poll it.
 155             if json_response["status"] == "pending":
 156                 default_cache.set(
 157                     self.task_id_cache_key, json_response["request_id"], REQUEST_CACHE_TIMEOUT
 158                 )
 159                 raise RetrySymbolication(retry_after=json_response["retry_after"])
 160             else:
 161                 # Once we arrive here, we are done processing. Clean up the
 162                 # task id from the cache.
 163                 default_cache.delete(self.task_id_cache_key)
 164                 metrics.timing(
 165                     "events.symbolicator.response.completed.size", len(json.dumps(json_response))
 166                 )
 167                 return json_response
 168 
 169     def process_minidump(self, minidump):
 170         return self._process(lambda: self.sess.upload_minidump(minidump))
 171 
 172     def process_applecrashreport(self, report):
 173         return self._process(lambda: self.sess.upload_applecrashreport(report))
 174 
 175     def process_payload(self, stacktraces, modules, signal=None):
 176         return self._process(
 177             lambda: self.sess.symbolicate_stacktraces(
 178                 stacktraces=stacktraces, modules=modules, signal=signal
 179             )
 180         )

 195 def get_internal_source(project):
 196     """
 197     Returns the source configuration for a Sentry project.
 198     """
 199     internal_url_prefix = options.get("system.internal-url-prefix")
 200     if not internal_url_prefix:
 201         internal_url_prefix = options.get("system.url-prefix")
 202         if sys.platform == "darwin":
 203             internal_url_prefix = internal_url_prefix.replace(
 204                 "localhost", "host.docker.internal"
 205             ).replace("127.0.0.1", "host.docker.internal")
 206 
 207     assert internal_url_prefix
 208     sentry_source_url = "%s%s" % (
 209         internal_url_prefix.rstrip("/"),
 210         reverse(
 211             "sentry-api-0-dsym-files",
 212             kwargs={"organization_slug": project.organization.slug, "project_slug": project.slug},
 213         ),
 214     )
 215 
 216     return {
 217         "type": "sentry",
 218         "id": "sentry:project",
 219         "url": sentry_source_url,
 220         "token": get_system_token(),
 221     }

 224 def parse_sources(config):
 225     """
 226     Parses the given sources in the config string (from JSON).
 227     """
 228 
 229     if not config:
 230         return []
 231 
 232     try:
 233         sources = json.loads(config)
 234     except BaseException as e:
 235         raise InvalidSourcesError(six.text_type(e))
 236 
 237     try:
 238         jsonschema.validate(sources, SOURCES_SCHEMA)
 239     except jsonschema.ValidationError as e:
 240         raise InvalidSourcesError(e.message)
 241 
 242     ids = set()
 243     for source in sources:
 244         if source["id"].startswith("sentry"):
 245             raise InvalidSourcesError('Source ids must not start with "sentry:"')
 246         if source["id"] in ids:
 247             raise InvalidSourcesError("Duplicate source id: %s" % (source["id"],))
 248         ids.add(source["id"])
 249 
 250     return sources

 253 def get_sources_for_project(project):
 254     """
 255     Returns a list of symbol sources for this project.
 256     """
 257 
 258     sources = []
 259 
 260     if not getattr(project, "_organization_cache", False):
 261         with sentry_sdk.start_span(op="lang.native.symbolicator.organization.get_from_cache"):
 262             project._organization_cache = Organization.objects.get_from_cache(
 263                 id=project.organization_id
 264             )
 265 
 266     # The symbolicator evaluates sources in the order they are declared. Always
 267     # try to download symbols from Sentry first.
 268     project_source = get_internal_source(project)
 269     sources.append(project_source)
 270 
 271     # Check that the organization still has access to symbol sources. This
 272     # controls both builtin and external sources.
 273     organization = project.organization
 274     if not features.has("organizations:symbol-sources", organization):
 275         return sources
 276 
 277     # Custom sources have their own feature flag. Check them independently.
 278     if features.has("organizations:custom-symbol-sources", organization):
 279         sources_config = project.get_option("sentry:symbol_sources")
 280     else:
 281         sources_config = None
 282 
 283     if sources_config:
 284         try:
 285             custom_sources = parse_sources(sources_config)
 286             sources.extend(custom_sources)
 287         except InvalidSourcesError:
 288             # Source configs should be validated when they are saved. If this
 289             # did not happen, this indicates a bug. Record this, but do not stop
 290             # processing at this point.
 291             logger.error("Invalid symbolicator source config", exc_info=True)
 292 
 293     def resolve_alias(source):
 294         for key in source.get("sources") or ():
 295             other_source = settings.SENTRY_BUILTIN_SOURCES.get(key)
 296             if other_source:
 297                 if other_source.get("type") == "alias":
 298                     for item in resolve_alias(other_source):
 299                         yield item
 300                 else:
 301                     yield other_source
 302 
 303     # Add builtin sources last to ensure that custom sources have precedence
 304     # over our defaults.
 305     builtin_sources = project.get_option("sentry:builtin_symbol_sources")
 306     for key, source in six.iteritems(settings.SENTRY_BUILTIN_SOURCES):
 307         if key not in builtin_sources:
 308             continue
 309 
 310         # special internal alias type expands to more than one item.  This
 311         # is used to make `apple` expand to `ios`/`macos` and other
 312         # sources if configured as such.
 313         if source.get("type") == "alias":
 314             sources.extend(resolve_alias(source))
 315         else:
 316             sources.append(source)
 317 
 318     return sources

 321 class SymbolicatorSession(object):
 322     def __init__(self, url=None, sources=None, project_id=None, event_id=None, timeout=None):
 323         self.url = url
 324         self.project_id = project_id
 325         self.event_id = event_id
 326         self.sources = sources or []
 327         self.timeout = timeout
 328         self.session = None
 329 
 330     def __enter__(self):
 331         self.open()
 332         return self
 333 
 334     def __exit__(self, *args):
 335         self.close()
 336 
 337     def open(self):
 338         if self.session is None:
 339             self.session = Session()
 340 
 341     def close(self):
 342         if self.session is not None:
 343             self.session.close()
 344             self.session = None
 345 
 346     def _ensure_open(self):
 347         if not self.session:
 348             raise RuntimeError("Session not opened")
 349 
 350     def _request(self, method, path, **kwargs):
 351         self._ensure_open()
 352 
 353         url = urljoin(self.url, path)
 354 
 355         # required for load balancing
 356         kwargs.setdefault("headers", {})["x-sentry-project-id"] = self.project_id
 357         kwargs.setdefault("headers", {})["x-sentry-event-id"] = self.event_id
 358 
 359         attempts = 0
 360         wait = 0.5
 361 
 362         while True:
 363             try:
 364                 with metrics.timer(
 365                     "events.symbolicator.session.request", tags={"attempt": attempts}
 366                 ):
 367                     response = self.session.request(
 368                         method, url, timeout=settings.SYMBOLICATOR_POLL_TIMEOUT + 1, **kwargs
 369                     )
 370 
 371                 metrics.incr(
 372                     "events.symbolicator.status_code",
 373                     tags={"status_code": response.status_code, "project_id": self.project_id},
 374                 )
 375 
 376                 if (
 377                     method.lower() == "get"
 378                     and path.startswith("requests/")
 379                     and response.status_code == 404
 380                 ):
 381                     # The symbolicator does not know this task. This is
 382                     # expected to happen when we're currently deploying
 383                     # symbolicator (which will clear all of its state). Re-send
 384                     # the symbolication task.
 385                     return None
 386 
 387                 if response.status_code in (502, 503):
 388                     raise ServiceUnavailable()
 389 
 390                 if response.ok:
 391                     json = response.json()
 392                 else:
 393                     json = {"status": "failed", "message": "internal server error"}
 394 
 395                 return json
 396             except (IOError, RequestException) as e:
 397                 metrics.incr(
 398                     "events.symbolicator.request_error",
 399                     tags={
 400                         "exc": ".".join([e.__class__.__module__, e.__class__.__name__]),
 401                         "attempt": attempts,
 402                     },
 403                 )
 404 
 405                 attempts += 1
 406                 # Any server error needs to be treated as a failure. We can
 407                 # retry a couple of times, but ultimately need to bail out.
 408                 #
 409                 # This can happen for any network failure.
 410                 if attempts > MAX_ATTEMPTS:
 411                     logger.error("Failed to contact symbolicator", exc_info=True)
 412                     raise
 413 
 414                 time.sleep(wait)
 415                 wait *= 2.0
 416 
 417     def _create_task(self, path, **kwargs):
 418         params = {"timeout": self.timeout, "scope": self.project_id}
 419         with metrics.timer("events.symbolicator.create_task", tags={"path": path}):
 420             return self._request(method="post", path=path, params=params, **kwargs)
 421 
 422     def symbolicate_stacktraces(self, stacktraces, modules, signal=None):
 423         json = {"sources": self.sources, "stacktraces": stacktraces, "modules": modules}
 424 
 425         if signal:
 426             json["signal"] = signal
 427 
 428         return self._create_task("symbolicate", json=json)
 429 
 430     def upload_minidump(self, minidump):
 431         return self._create_task(
 432             path="minidump",
 433             data={"sources": json.dumps(self.sources)},
 434             files={"upload_file_minidump": minidump},
 435         )
 436 
 437     def upload_applecrashreport(self, report):
 438         return self._create_task(
 439             path="applecrashreport",
 440             data={"sources": json.dumps(self.sources)},
 441             files={"apple_crash_report": report},
 442         )
 443 
 444     def query_task(self, task_id):
 445         task_url = "requests/%s" % (task_id,)
 446 
 447         params = {
 448             "timeout": 0,  # Only wait when creating, but not when querying tasks
 449             "scope": self.project_id,
 450         }
 451 
 452         with metrics.timer("events.symbolicator.query_task"):
 453             return self._request("get", task_url, params=params)
 454 
 455     def healthcheck(self):
 456         return self._request("get", "healthcheck")
