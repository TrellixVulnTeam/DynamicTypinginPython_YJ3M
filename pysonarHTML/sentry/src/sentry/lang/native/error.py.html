  34 class SymbolicationFailed(Exception):
  35     message = None
  36 
  37     def __init__(self, message=None, type=None, obj=None):
  38         Exception.__init__(self)
  39         self.message = six.text_type(message)
  40         self.type = type
  41         self.image_name = None
  42         self.image_path = None
  43         if obj is not None:
  44             self.image_uuid = six.text_type(obj.debug_id)
  45             if obj.name:
  46                 self.image_path = obj.name
  47                 self.image_name = image_name(obj.name)
  48             self.image_arch = obj.arch
  49         else:
  50             self.image_uuid = None
  51             self.image_arch = None
  52 
  53     @property
  54     def is_user_fixable(self):
  55         """These are errors that a user can fix themselves."""
  56         return self.type in USER_FIXABLE_ERRORS
  57 
  58     @property
  59     def is_fatal(self):
  60         """If this is true then a processing issues has to be reported."""
  61         return self.type in FATAL_ERRORS
  62 
  63     @property
  64     def is_sdk_failure(self):
  65         """An error that most likely happened because of a bad SDK."""
  66         return self.type == EventError.NATIVE_UNKNOWN_IMAGE
  67 
  68     def get_data(self):
  69         """Returns the event data."""
  70         rv = {"message": self.message, "type": self.type}
  71         if self.image_path is not None:
  72             rv["image_path"] = self.image_path
  73         if self.image_uuid is not None:
  74             rv["image_uuid"] = self.image_uuid
  75         if self.image_arch is not None:
  76             rv["image_arch"] = self.image_arch
  77         return rv
  78 
  79     def __str__(self):
  80         rv = []
  81         if self.type is not None:
  82             rv.append(u"%s: " % self.type)
  83         rv.append(self.message or "no information available")
  84         if self.image_uuid is not None:
  85             rv.append(" image-uuid=%s" % self.image_uuid)
  86         if self.image_name is not None:
  87             rv.append(" image-name=%s" % self.image_name)
  88         return u"".join(rv)

  91 def write_error(e, data):
  92     # User fixable but fatal errors are reported as processing
  93     # issues. We skip this for minidumps, as reprocessing is not
  94     # possible without persisting minidumps.
  95     if e.is_user_fixable and e.is_fatal and not is_minidump_event(data):
  96         report_processing_issue(
  97             data, scope="native", object="dsym:%s" % e.image_uuid, type=e.type, data=e.get_data()
  98         )
  99 
 100     # This in many ways currently does not really do anything.
 101     # The reason is that once a processing issue is reported
 102     # the event will only be stored as a raw event and no
 103     # group will be generated.  As a result it also means that
 104     # we will not have any user facing event or error showing
 105     # up at all.  We want to keep this here though in case we
 106     # do not want to report some processing issues (eg:
 107     # optional difs)
 108     if e.is_user_fixable or e.is_sdk_failure:
 109         errors = data.setdefault("errors", [])
 110         errors.append(e.get_data())
 111     else:
 112         logger.debug("Failed to symbolicate with native backend", exc_info=True)
 113 
 114     if not e.is_user_fixable:
 115         data.setdefault("_metrics", {})["flag.processing.error"] = True
 116 
 117     if e.is_fatal:
 118         data.setdefault("_metrics", {})["flag.processing.fatal"] = True
