  56 def is_native_event(data):
  57     if is_native_platform(data.get("platform")):
  58         return True
  59 
  60     for stacktrace in find_stacktraces_in_data(data):
  61         if any(is_native_platform(x) for x in stacktrace.platforms):
  62             return True
  63 
  64     return False

  67 def image_name(pkg):
  68     if not pkg:
  69         return pkg
  70     split = "\\" if WINDOWS_PATH_RE.match(pkg) else "/"
  71     return pkg.rsplit(split, 1)[-1]

  74 def get_sdk_from_event(event):
  75     sdk_info = get_path(event, "debug_meta", "sdk_info")
  76     if sdk_info:
  77         return sdk_info
  78 
  79     os = get_path(event, "contexts", "os")
  80     if os and os.get("type") == "os":
  81         return get_sdk_from_os(os)

  84 def get_sdk_from_os(data):
  85     if data.get("name") is None or data.get("version") is None:
  86         return
  87 
  88     try:
  89         version = six.text_type(data["version"]).split("-", 1)[0] + ".0" * 3
  90         system_version = tuple(int(x) for x in version.split(".")[:3])
  91     except ValueError:
  92         return
  93 
  94     return {
  95         "sdk_name": data["name"],
  96         "version_major": system_version[0],
  97         "version_minor": system_version[1],
  98         "version_patchlevel": system_version[2],
  99         "build": data.get("build"),
 100     }

 103 def signal_from_data(data):
 104     exceptions = get_path(data, "exception", "values", filter=True)
 105     signal = get_path(exceptions, 0, "mechanism", "meta", "signal", "number")
 106     if signal is not None:
 107         return int(signal)
 108 
 109     return None

 118 def convert_crashreport_count(value):
 119     """
 120     Shim to read both legacy and new `sentry:store_crash_reports` project and
 121     organization options.
 122 
 123     The legacy format stored `True` for an unlimited number of crash reports,
 124     and `False` for no crash reports.
 125 
 126     The new format stores `-1` for unbounded storage, `0` for no crash reports,
 127     and a positive number for a bounded number per group.
 128 
 129     Defaults to `0` (no storage).
 130     """
 131     if value is True:
 132         return STORE_CRASH_REPORTS_ALL
 133     if value is None:
 134         return STORE_CRASH_REPORTS_DEFAULT
 135     return int(value)
