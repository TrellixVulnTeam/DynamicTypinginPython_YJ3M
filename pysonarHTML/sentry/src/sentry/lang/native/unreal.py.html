  12 def write_applecrashreport_placeholder(data):
  13     """
  14     Writes a placeholder to indicate that this event has an apple crash report.
  15 
  16     This will indicate to the ingestion pipeline that this event will need to be
  17     processed. The payload can be checked via ``is_applecrashreport_event``.
  18     """
  19     # Apple crash report events must be native platform for processing.
  20     data["platform"] = "native"
  21 
  22     # Assume that this minidump is the result of a crash and assign the fatal
  23     # level. Note that the use of `setdefault` here doesn't generally allow the
  24     # user to override the minidump's level as processing will overwrite it
  25     # later.
  26     setdefault_path(data, "level", value="fatal")
  27 
  28     # Create a placeholder exception. This signals normalization that this is an
  29     # error event and also serves as a placeholder if processing of the minidump
  30     # fails.
  31     exception = {
  32         "type": "AppleCrashReport",
  33         "value": "Invalid Apple Crash Report",
  34         "mechanism": {"type": "applecrashreport", "handled": False, "synthetic": True},
  35     }
  36     data["exception"] = {"values": [exception]}

  39 def is_applecrashreport_event(data):
  40     """
  41     Checks whether an event indicates that it has an apple crash report.
  42 
  43     This requires the event to have a special marker payload. It is written by
  44     ``write_applecrashreport_placeholder``.
  45     """
  46     exceptions = get_path(data, "exception", "values", filter=True)
  47     return get_path(exceptions, 0, "mechanism", "type") == "applecrashreport"

  50 def merge_unreal_user(event, user_id):
  51     """
  52     Merges user information from the unreal "UserId" into the event payload.
  53     """
  54 
  55     # https://github.com/EpicGames/UnrealEngine/blob/f509bb2d6c62806882d9a10476f3654cf1ee0634/Engine/Source/Programs/CrashReportClient/Private/CrashUpload.cpp#L769
  56     parts = user_id.split("|", 2)
  57     login_id, epic_account_id, machine_id = parts + [""] * (3 - len(parts))
  58     event["user"] = {"id": login_id if login_id else user_id}
  59     if epic_account_id:
  60         set_path(event, "tags", "epic_account_id", value=epic_account_id)
  61     if machine_id:
  62         set_path(event, "tags", "machine_id", value=machine_id)

  65 def unreal_attachment_type(unreal_file):
  66     """Returns the `attachment_type` for the
  67     unreal file type or None if not recognized"""
  68     if unreal_file.type == "minidump":
  69         return MINIDUMP_ATTACHMENT_TYPE
  70     if unreal_file.type == "applecrashreport":
  71         return APPLECRASHREPORT_ATTACHMENT_TYPE

  74 def merge_unreal_context_event(unreal_context, event, project):
  75     """Merges the context from an Unreal Engine 4 crash
  76     with the given event."""
  77     runtime_prop = unreal_context.get("runtime_properties")
  78     if runtime_prop is None:
  79         return
  80 
  81     message = runtime_prop.pop("error_message", None)
  82     if message is not None:
  83         event["message"] = message
  84 
  85     username = runtime_prop.pop("username", None)
  86     if username is not None:
  87         set_path(event, "user", "username", value=username)
  88 
  89     memory_physical = runtime_prop.pop("memory_stats_total_physical", None)
  90     if memory_physical is not None:
  91         set_path(event, "contexts", "device", "memory_size", value=memory_physical)
  92 
  93     # Likely overwritten by minidump processing
  94     os_major = runtime_prop.pop("misc_os_version_major", None)
  95     if os_major is not None:  # i.e: Windows 10
  96         set_path(event, "contexts", "os", "name", value=os_major)
  97 
  98     gpu_brand = runtime_prop.pop("misc_primary_cpu_brand", None)
  99     if gpu_brand is not None:
 100         set_path(event, "contexts", "gpu", "name", value=gpu_brand)
 101 
 102     user_desc = runtime_prop.pop("user_description", None)
 103     if user_desc is not None:
 104         feedback_user = "unknown"
 105         if username is not None:
 106             feedback_user = username
 107 
 108         UserReport.objects.create(
 109             project=project,
 110             event_id=event["event_id"],
 111             name=feedback_user,
 112             email="",
 113             comments=user_desc,
 114         )
 115 
 116     # drop modules. minidump processing adds 'images loaded'
 117     runtime_prop.pop("modules", None)
 118 
 119     # add everything else as extra
 120     set_path(event, "contexts", "unreal", "type", value="unreal")
 121     event["contexts"]["unreal"].update(**runtime_prop)
 122 
 123     # add sdk info
 124     event["sdk"] = {
 125         "name": "sentry.unreal.crashreporter",
 126         "version": runtime_prop.pop("crash_reporter_client_version", "0.0.0"),
 127     }

 130 def merge_unreal_logs_event(unreal_logs, event):
 131     setdefault_path(event, "breadcrumbs", "values", value=[])
 132     breadcrumbs = event["breadcrumbs"]["values"]
 133 
 134     for log in unreal_logs:
 135         message = log.get("message")
 136         if message:
 137             breadcrumbs.append(
 138                 {
 139                     "timestamp": log.get("timestamp"),
 140                     "category": log.get("component"),
 141                     "message": message,
 142                 }
 143             )
