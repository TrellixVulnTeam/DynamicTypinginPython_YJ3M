  19     def __init__(
  20         self, threads=None, context=None, debug_images=None, symbolicated=False, exceptions=None
  21     ):

  23         self.context = context
  24         self.debug_images = debug_images

  28     def __str__(self):
  29         rv = []
  30         rv.append(self._get_meta_header())
  31         rv.append(self._get_exception_info())
  32         rv.append(self.get_threads_apple_string())
  33         rv.append(self.get_binary_images_apple_string())
  34         return "\n\n".join(rv) + "\n\nEOF"

  36     def _get_meta_header(self):
  37         return "OS Version: %s %s (%s)\nReport Version: %s" % (
  38             get_path(self.context, "os", "name"),
  39             get_path(self.context, "os", "version"),
  40             get_path(self.context, "os", "build"),
  41             REPORT_VERSION,
  42         )

  44     def _get_exception_info(self):
  45         rv = []
  46 
  47         # We only have one exception at a time
  48         exception = get_path(self.exceptions, 0)
  49         if not exception:
  50             return ""
  51 
  52         mechanism = upgrade_legacy_mechanism(exception.get("mechanism")) or {}
  53         mechanism_meta = get_path(mechanism, "meta", default={})
  54 
  55         signal = get_path(mechanism_meta, "signal", "name")
  56         name = get_path(mechanism_meta, "mach_exception", "name")
  57 
  58         if name or signal:
  59             rv.append(
  60                 "Exception Type: %s%s" % (name or "Unknown", signal and (" (%s)" % signal) or "")
  61             )
  62 
  63         exc_name = get_path(mechanism_meta, "signal", "code_name")
  64         exc_addr = get_path(mechanism, "data", "relevant_address")
  65         if exc_name:
  66             rv.append(
  67                 "Exception Codes: %s%s"
  68                 % (exc_name, exc_addr is not None and (" at %s" % exc_addr) or "")
  69             )
  70 
  71         if exception.get("thread_id") is not None:
  72             rv.append("Crashed Thread: %s" % exception["thread_id"])
  73 
  74         if exception.get("value"):
  75             rv.append("\nApplication Specific Information:\n%s" % exception["value"])
  76 
  77         return "\n".join(rv)

  79     def get_threads_apple_string(self):
  80         rv = []
  81         exception = self.exceptions or []
  82         threads = self.threads or []
  83         for thread_info in exception + threads:
  84             thread_string = self.get_thread_apple_string(thread_info)
  85             if thread_string is not None:
  86                 rv.append(thread_string)
  87         return "\n\n".join(rv)
  88 
  89     def get_thread_apple_string(self, thread_info):
  90         rv = []
  91         stacktrace = get_path(thread_info, "stacktrace")
  92         if stacktrace is None:
  93             return None
  94 
  95         if stacktrace:
  96             frames = get_path(stacktrace, "frames", filter=True)
  97             if frames:
  98                 for i, frame in enumerate(reversed(frames)):
  99                     frame_string = self._convert_frame_to_apple_string(
 100                         frame=frame,
 101                         next=frames[len(frames) - i - 2] if i < len(frames) - 1 else None,
 102                         number=i,
 103                     )
 104                     if frame_string is not None:
 105                         rv.append(frame_string)
 106 
 107         if len(rv) == 0:
 108             return None  # No frames in thread, so we remove thread
 109 
 110         is_exception = bool(thread_info.get("mechanism"))
 111         thread_id = thread_info.get("id") or thread_info.get("thread_id") or "0"
 112         thread_name = thread_info.get("name")
 113         thread_name_string = " name: %s" % (thread_name) if thread_name else ""
 114         thread_crashed = thread_info.get("crashed") or is_exception
 115         thread_crashed_thread = " Crashed:" if thread_crashed else ""
 116         thread_string = "Thread %s%s%s\n" % (thread_id, thread_name_string, thread_crashed_thread)
 117         return thread_string + "\n".join(rv)

 119     def _convert_frame_to_apple_string(self, frame, next=None, number=0):
 120         if frame.get("instruction_addr") is None:
 121             return None
 122         slide_value = self._get_slide_value(frame.get("image_addr"))
 123         instruction_addr = slide_value + parse_addr(frame.get("instruction_addr"))
 124         image_addr = slide_value + parse_addr(frame.get("image_addr"))
 125         offset = ""
 126         if frame.get("image_addr") is not None and (
 127             not self.symbolicated
 128             or (frame.get("function") or NATIVE_UNKNOWN_STRING) == NATIVE_UNKNOWN_STRING
 129         ):
 130             offset = " + %s" % (
 131                 instruction_addr - slide_value - parse_addr(frame.get("symbol_addr"))
 132             )
 133         symbol = hex(image_addr)
 134         if self.symbolicated:
 135             file = ""
 136             if frame.get("filename") and frame.get("lineno"):
 137                 file = " (%s:%s)" % (
 138                     posixpath.basename(frame.get("filename") or NATIVE_UNKNOWN_STRING),
 139                     frame["lineno"],
 140                 )
 141             symbol = "%s%s" % (frame.get("function") or NATIVE_UNKNOWN_STRING, file)
 142             if next and parse_addr(frame.get("instruction_addr")) == parse_addr(
 143                 next.get("instruction_addr")
 144             ):
 145                 symbol = "[inlined] " + symbol
 146         return "%s%s%s%s%s" % (
 147             str(number).ljust(4, " "),
 148             image_name(frame.get("package") or NATIVE_UNKNOWN_STRING).ljust(32, " "),
 149             hex(instruction_addr).ljust(20, " "),
 150             symbol,
 151             offset,
 152         )
 153 
 154     def _get_slide_value(self, image_addr):
 155         if self.debug_images:
 156             for debug_image in self.debug_images:
 157                 if parse_addr(debug_image.get("image_addr")) == parse_addr(image_addr):
 158                     return parse_addr(debug_image.get("image_vmaddr", 0))
 159         return 0
 160 
 161     def get_binary_images_apple_string(self):
 162         # We dont need binary images on symbolicated crashreport
 163         if self.symbolicated or self.debug_images is None:
 164             return ""
 165         binary_images = map(
 166             lambda i: self._convert_debug_meta_to_binary_image_row(debug_image=i),
 167             sorted(self.debug_images, key=lambda i: parse_addr(i["image_addr"])),
 168         )
 169         return "Binary Images:\n" + "\n".join(binary_images)
 170 
 171     def _convert_debug_meta_to_binary_image_row(self, debug_image):
 172         slide_value = parse_addr(debug_image.get("image_vmaddr", 0))
 173         image_addr = parse_addr(debug_image["image_addr"]) + slide_value
 174         return "%s - %s %s %s  <%s> %s" % (
 175             hex(image_addr),
 176             hex(image_addr + debug_image["image_size"] - 1),
 177             image_name(debug_image.get("code_file") or NATIVE_UNKNOWN_STRING),
 178             self.context["device"]["arch"],
 179             debug_image.get("debug_id").replace("-", "").lower(),
 180             debug_image.get("code_file") or NATIVE_UNKNOWN_STRING,
 181         )
