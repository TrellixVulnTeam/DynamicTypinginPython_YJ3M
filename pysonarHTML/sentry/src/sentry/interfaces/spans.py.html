  21 class Span(Interface):
  22     """
  23     Holds timing spans related to APM and tracing.
  24 
  25     >>> {
  26     >>>   'trace_id': 'a0fa8803753e40fd8124b21eeb2986b5',
  27     >>>   'parent_span_id': '9c2a6db8c79068a2',
  28     >>>   'span_id': '8c931f4740435fb8',
  29     >>>   'start_timestamp': '2019-06-14T14:01:41Z',
  30     >>>   'same_process_as_parent': true,
  31     >>>   'description': 'http://httpbin.org/base64/aGVsbG8gd29ybGQK GET',
  32     >>>   'tags': { 'http.status_code': 200, 'error': false },
  33     >>>   'timestamp': '2019-06-14T14:01:41Z',
  34     >>>   'op': 'http',
  35     >>>   'data': {
  36     >>>     'url': 'http://httpbin.org/base64/aGVsbG8gd29ybGQK',
  37     >>>     'status_code': 200,
  38     >>>     'reason': 'OK',
  39     >>>     'method': 'GET'
  40     >>>   }
  41     >>> }
  42     """
  43 
  44     @classmethod
  45     def to_python(cls, data):
  46         for key in SPAN_KEYS:
  47             data.setdefault(key, None)
  48         return cls(**data)

  51 class Spans(Interface):
  52     """
  53     Contains a list of Span interfaces
  54     """
  55 
  56     display_score = 1950
  57     score = 1950
  58     path = "spans"
  59 
  60     @classmethod
  61     def to_python(cls, data):
  62         spans = [Span.to_python(span) for span in data]
  63         return cls(spans=spans)
  64 
  65     def __iter__(self):
  66         return iter(self.spans)
  67 
  68     def to_json(self):
  69         return [span.to_json() for span in self]
