  12 class User(Interface):
  13     """
  14     An interface which describes the authenticated User for a request.
  15 
  16     You should provide **at least** either an `id` (a unique identifier for
  17     an authenticated user) or `ip_address` (their IP address).
  18 
  19     All other attributes are optional.
  20 
  21     >>> {
  22     >>>     "id": "unique_id",
  23     >>>     "username": "my_user",
  24     >>>     "email": "foo@example.com"
  25     >>>     "ip_address": "127.0.0.1",
  26     >>>     "optional": "value"
  27     >>> }
  28     """
  29 
  30     score = 1
  31     display_score = 2020
  32 
  33     @classmethod
  34     def to_python(cls, data):
  35         data = data.copy()
  36         for key in ("id", "email", "username", "ip_address", "name", "geo", "data"):
  37             data.setdefault(key, None)
  38         if data["geo"] is not None:
  39             data["geo"] = Geo.to_python(data["geo"])
  40         return cls(**data)
  41 
  42     def to_json(self):
  43         return prune_empty_keys(
  44             {
  45                 "id": self.id,
  46                 "email": self.email,
  47                 "username": self.username,
  48                 "ip_address": self.ip_address,
  49                 "name": self.name,
  50                 "geo": self.geo.to_json() if self.geo is not None else None,
  51                 "data": self.data or None,
  52             }
  53         )
  54

  60             "ip_address": self.ip_address,
