  16 def upgrade_legacy_mechanism(data):
  17     """
  18     Conversion from mechanism objects sent by old sentry-cocoa SDKs. It assumes
  19     "type": "generic" and moves "posix_signal", "mach_exception" into "meta".
  20     All other keys are moved into "data".
  21 
  22     Example old payload:
  23     >>> {
  24     >>>     "posix_signal": {
  25     >>>         "name": "SIGSEGV",
  26     >>>         "code_name": "SEGV_NOOP",
  27     >>>         "signal": 11,
  28     >>>         "code": 0
  29     >>>     },
  30     >>>     "relevant_address": "0x1",
  31     >>>     "mach_exception": {
  32     >>>         "exception": 1,
  33     >>>         "exception_name": "EXC_BAD_ACCESS",
  34     >>>         "subcode": 8,
  35     >>>         "code": 1
  36     >>>     }
  37     >>> }
  38 
  39     Example normalization:
  40     >>> {
  41     >>>     "type": "generic",
  42     >>>     "data": {
  43     >>>         "relevant_address": "0x1"
  44     >>>     },
  45     >>>     "meta": {
  46     >>>         "mach_exception": {
  47     >>>             "exception": 1,
  48     >>>             "subcode": 8,
  49     >>>             "code": 1,
  50     >>>             "name": "EXC_BAD_ACCESS"
  51     >>>         },
  52     >>>         "signal": {
  53     >>>             "number": 11,
  54     >>>             "code": 0,
  55     >>>             "name": "SIGSEGV",
  56     >>>             "code_name": "SEGV_NOOP"
  57     >>>         }
  58     >>>     }
  59     >>> }
  60     """
  61 
  62     # Early exit for current protocol. We assume that when someone sends a
  63     # "type", we do not need to preprocess and can immediately validate
  64     if data is None or data.get("type") is not None:
  65         return data
  66 
  67     result = {"type": "generic"}
  68 
  69     # "posix_signal" and "mach_exception" were optional root-level objects,
  70     # which have now moved to special keys inside "meta". We only create "meta"
  71     # if there is actual data to add.
  72 
  73     posix_signal = data.pop("posix_signal", None)
  74     if posix_signal and posix_signal.get("signal"):
  75         result.setdefault("meta", {})["signal"] = prune_empty_keys(
  76             {
  77                 "number": posix_signal.get("signal"),
  78                 "code": posix_signal.get("code"),
  79                 "name": posix_signal.get("name"),
  80                 "code_name": posix_signal.get("code_name"),
  81             }
  82         )
  83 
  84     mach_exception = data.pop("mach_exception", None)
  85     if mach_exception:
  86         result.setdefault("meta", {})["mach_exception"] = prune_empty_keys(
  87             {
  88                 "exception": mach_exception.get("exception"),
  89                 "code": mach_exception.get("code"),
  90                 "subcode": mach_exception.get("subcode"),
  91                 "name": mach_exception.get("exception_name"),
  92             }
  93         )
  94 
  95     # All remaining data has to be moved to the "data" key. We assume that even
  96     # if someone accidentally sent a corret top-level key (such as "handled"),
  97     # it will not pass our interface validation and should be moved to "data"
  98     # instead.
  99     result.setdefault("data", {}).update(data)
 100     return result

 103 class Mechanism(Interface):
 104     """
 105     an optional field residing in the exception interface. It carries additional
 106     information about the way the exception was created on the target system.
 107     This includes general exception values obtained from operating system or
 108     runtime APIs, as well as mechanism-specific values.
 109 
 110     >>> {
 111     >>>     "type": "mach",
 112     >>>     "description": "EXC_BAD_ACCESS",
 113     >>>     "data": {
 114     >>>         "relevant_address": "0x1"
 115     >>>     },
 116     >>>     "handled": false,
 117     >>>     "synthetic": false,
 118     >>>     "help_link": "https://developer.apple.com/library/content/qa/qa1367/_index.html",
 119     >>>     "meta": {
 120     >>>         "mach_exception": {
 121     >>>              "exception": 1,
 122     >>>              "subcode": 8,
 123     >>>              "code": 1
 124     >>>         },
 125     >>>         "signal": {
 126     >>>             "number": 11
 127     >>>         }
 128     >>>     }
 129     >>> }
 130     """
 131 
 132     @classmethod
 133     def to_python(cls, data):
 134         for key in ("type", "synthetic", "description", "help_link", "handled", "data", "meta"):
 135             data.setdefault(key, None)
 136 
 137         return cls(**data)
 138 
 139     def to_json(self):
 140         return prune_empty_keys(
 141             {
 142                 "type": self.type,
 143                 "synthetic": self.synthetic,
 144                 "description": self.description,
 145                 "help_link": self.help_link,
 146                 "handled": self.handled,
 147                 "data": self.data or None,
 148                 "meta": prune_empty_keys(self.meta) or None,
 149             }
 150         )
 151 
 152     def iter_tags(self):
 153         yield (self.path, self.type)
 154 
 155         if self.handled is not None:
 156             yield ("handled", self.handled and "yes" or "no")

 159 def uncontribute_non_stacktrace_variants(variants):
 160     """If we have multiple variants and at least one has a stacktrace, we
 161     want to mark all non stacktrace variants non contributing.  The reason
 162     for this is that otherwise we end up in very generic grouping which has
 163     some negative consequences for the quality of the groups.
 164     """
 165     if len(variants) <= 1:
 166         return variants
 167     any_stacktrace_contributes = False
 168     non_contributing_components = []
 169     stacktrace_variants = set()
 170 
 171     # In case any of the variants has a contributing stacktrace, we want
 172     # to make all other variants non contributing.  Thr e
 173     for (key, component) in six.iteritems(variants):
 174         if any(
 175             s.contributes for s in component.iter_subcomponents(id="stacktrace", recursive=True)
 176         ):
 177             any_stacktrace_contributes = True
 178             stacktrace_variants.add(key)
 179         else:
 180             non_contributing_components.append(component)
 181 
 182     if any_stacktrace_contributes:
 183         if len(stacktrace_variants) == 1:
 184             hint_suffix = "but the %s variant does" % next(iter(stacktrace_variants))
 185         else:
 186             # this branch is basically dead because we only have two
 187             # variants right now, but this is so this does not break in
 188             # the future.
 189             hint_suffix = "others do"
 190         for component in non_contributing_components:
 191             component.update(
 192                 contributes=False,
 193                 hint="ignored because this variant does not contain a "
 194                 "stacktrace, but %s" % hint_suffix,
 195             )
 196 
 197     return variants

 200 class SingleException(Interface):
 201     """
 202     A standard exception with a ``type`` and value argument, and an optional
 203     ``module`` argument describing the exception class type and
 204     module namespace. Either ``type`` or ``value`` must be present.
 205 
 206     You can also optionally bind a stacktrace interface to an exception. The
 207     spec is identical to ``stacktrace``.
 208 
 209     >>> {
 210     >>>     "type": "ValueError",
 211     >>>     "value": "My exception value",
 212     >>>     "module": "__builtins__",
 213     >>>     "mechanism": {},
 214     >>>     "stacktrace": {
 215     >>>         # see stacktrace
 216     >>>     }
 217     >>> }
 218     """
 219 
 220     grouping_variants = ["system", "app"]
 221 
 222     @classmethod
 223     def to_python(cls, data):
 224         if get_path(data, "stacktrace", "frames", filter=True):
 225             stacktrace = Stacktrace.to_python(data["stacktrace"])
 226         else:
 227             stacktrace = None
 228 
 229         if get_path(data, "raw_stacktrace", "frames", filter=True):
 230             raw_stacktrace = Stacktrace.to_python(data["raw_stacktrace"], raw=True)
 231         else:
 232             raw_stacktrace = None
 233 
 234         type = data.get("type")
 235         value = data.get("value")
 236 
 237         if data.get("mechanism"):
 238             mechanism = Mechanism.to_python(data["mechanism"])
 239         else:
 240             mechanism = None
 241 
 242         kwargs = {
 243             "type": type,
 244             "value": value,
 245             "module": data.get("module"),
 246             "mechanism": mechanism,
 247             "stacktrace": stacktrace,
 248             "thread_id": data.get("thread_id"),
 249             "raw_stacktrace": raw_stacktrace,
 250         }
 251 
 252         return cls(**kwargs)
 253 
 254     def to_json(self):
 255         mechanism = (
 256             isinstance(self.mechanism, Mechanism)
 257             and self.mechanism.to_json()
 258             or self.mechanism
 259             or None
 260         )
 261 
 262         if self.stacktrace:
 263             stacktrace = self.stacktrace.to_json()
 264         else:
 265             stacktrace = None
 266 
 267         if self.raw_stacktrace:
 268             raw_stacktrace = self.raw_stacktrace.to_json()
 269         else:
 270             raw_stacktrace = None
 271 
 272         return prune_empty_keys(
 273             {
 274                 "type": self.type,
 275                 "value": self.value,
 276                 "mechanism": mechanism,
 277                 "module": self.module,
 278                 "stacktrace": stacktrace,
 279                 "thread_id": self.thread_id,
 280                 "raw_stacktrace": raw_stacktrace,
 281             }
 282         )
 283 
 284     def get_api_context(self, is_public=False, platform=None):
 285         mechanism = (
 286             isinstance(self.mechanism, Mechanism)
 287             and self.mechanism.get_api_context(is_public=is_public, platform=platform)
 288             or self.mechanism
 289             or None
 290         )
 291 
 292         if self.stacktrace:
 293             stacktrace = self.stacktrace.get_api_context(is_public=is_public, platform=platform)
 294         else:
 295             stacktrace = None
 296 
 297         if self.raw_stacktrace:
 298             raw_stacktrace = self.raw_stacktrace.get_api_context(
 299                 is_public=is_public, platform=platform
 300             )
 301         else:
 302             raw_stacktrace = None
 303 
 304         return {
 305             "type": self.type,
 306             "value": six.text_type(self.value) if self.value else None,
 307             "mechanism": mechanism,
 308             "threadId": self.thread_id,
 309             "module": self.module,
 310             "stacktrace": stacktrace,
 311             "rawStacktrace": raw_stacktrace,
 312         }
 313 
 314     def get_api_meta(self, meta, is_public=False, platform=None):
 315         mechanism_meta = (
 316             self.mechanism.get_api_meta(meta["mechanism"], is_public=is_public, platform=platform)
 317             if isinstance(self.mechanism, Mechanism) and meta.get("mechanism")
 318             else None
 319         )
 320 
 321         stacktrace_meta = (
 322             self.stacktrace.get_api_meta(meta, is_public=is_public, platform=platform)
 323             if self.stacktrace and meta.get("stacktrace")
 324             else None
 325         )
 326 
 327         return {
 328             "": meta.get(""),
 329             "type": meta.get("type"),
 330             "value": meta.get("value"),
 331             "mechanism": mechanism_meta,
 332             "threadId": meta.get("thread_id"),
 333             "module": meta.get("module"),
 334             "stacktrace": stacktrace_meta,
 335         }

 338 class Exception(Interface):
 339     """
 340     An exception consists of a list of values. In most cases, this list
 341     contains a single exception, with an optional stacktrace interface.
 342 
 343     Each exception has a mandatory ``value`` argument and optional ``type`` and
 344     ``module`` arguments describing the exception class type and module
 345     namespace.
 346 
 347     You can also optionally bind a stacktrace interface to an exception. The
 348     spec is identical to ``stacktrace``.
 349 
 350     >>> {
 351     >>>     "values": [{
 352     >>>         "type": "ValueError",
 353     >>>         "value": "My exception value",
 354     >>>         "module": "__builtins__",
 355     >>>         "mechanism": {
 356     >>>             # see sentry.interfaces.Mechanism
 357     >>>         },
 358     >>>         "stacktrace": {
 359     >>>             # see stacktrace
 360     >>>         }
 361     >>>     }]
 362     >>> }
 363 
 364     Values should be sent oldest to newest, this includes both the stacktrace
 365     and the exception itself.
 366 
 367     .. note:: This interface can be passed as the 'exception' key in addition
 368               to the full interface path.
 369     """
 370 
 371     score = 2000
 372     grouping_variants = ["system", "app"]
 373 
 374     def exceptions(self):
 375         return get_path(self.values, filter=True)
 376 
 377     def __getitem__(self, key):
 378         return self.exceptions()[key]
 379 
 380     def __iter__(self):
 381         return iter(self.exceptions())
 382 
 383     def __len__(self):
 384         return len(self.exceptions())
 385 
 386     @classmethod
 387     def to_python(cls, data):
 388         return cls(
 389             values=[
 390                 v and SingleException.to_python(v) for v in get_path(data, "values", default=[])
 391             ],
 392             exc_omitted=data.get("exc_omitted"),
 393         )
 394 
 395     # TODO(ja): Fix all following methods when to_python is refactored. All
 396     # methods below might throw if None exceptions are in ``values``.
 397 
 398     def to_json(self):
 399         return prune_empty_keys(
 400             {
 401                 "values": [v and v.to_json() for v in self.values] or None,
 402                 "exc_omitted": self.exc_omitted,
 403             }
 404         )
 405 
 406     def get_api_context(self, is_public=False, platform=None):
 407         return {
 408             "values": [
 409                 v.get_api_context(is_public=is_public, platform=platform) for v in self.values if v
 410             ],
 411             "hasSystemFrames": any(
 412                 v.stacktrace.get_has_system_frames() for v in self.values if v and v.stacktrace
 413             ),
 414             "excOmitted": self.exc_omitted,
 415         }
 416 
 417     def get_api_meta(self, meta, is_public=False, platform=None):
 418         if not meta:
 419             return meta
 420 
 421         result = {}
 422         values = meta.get("values", meta)
 423         for index, value in six.iteritems(values):
 424             exc = self.values[int(index)]
 425             if exc is not None:
 426                 result[index] = exc.get_api_meta(value, is_public=is_public, platform=platform)
 427 
 428         return {"values": result}
 429 
 430     def to_string(self, event, is_public=False, **kwargs):
 431         if not self.values:
 432             return ""
 433 
 434         output = []
 435         for exc in self.values:
 436             if not exc:
 437                 continue
 438 
 439             output.append(u"{0}: {1}\n".format(exc.type, exc.value))
 440             if exc.stacktrace:
 441                 output.append(
 442                     exc.stacktrace.get_stacktrace(
 443                         event, system_frames=False, max_frames=5, header=False
 444                     )
 445                     + "\n\n"
 446                 )
 447         return ("".join(output)).strip()
 448 
 449     def get_stacktrace(self, *args, **kwargs):
 450         exc = self.values[0]
 451         if exc.stacktrace:
 452             return exc.stacktrace.get_stacktrace(*args, **kwargs)
 453         return ""
 454 
 455     def iter_tags(self):
 456         if not self.values or not self.values[0]:
 457             return
 458 
 459         mechanism = self.values[0].mechanism
 460         if mechanism:
 461             for tag in mechanism.iter_tags():
 462                 yield tag
