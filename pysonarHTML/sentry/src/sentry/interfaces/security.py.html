 131 class Hpkp(SecurityReport):
 132     """
 133     A HTTP Public Key Pinning pin validation failure report.
 134 
 135     See also: https://tools.ietf.org/html/rfc7469#section-3
 136     >>> {
 137     >>>     "date-time": "2014-04-06T13:00:50Z",
 138     >>>     "hostname": "www.example.com",
 139     >>>     "port": 443,
 140     >>>     "effective-expiration-date": "2014-05-01T12:40:50Z",
 141     >>>     "include-subdomains": False,
 142     >>>     "served-certificate-chain": [],
 143     >>>     "validated-certificate-chain": [],
 144     >>>     "known-pins": [],
 145     >>> }
 146     """
 147 
 148     score = 1300
 149     display_score = 1300
 150 
 151     title = "HPKP Report"
 152 
 153     @classmethod
 154     def from_raw(cls, raw):
 155         # Validate the raw data against the input schema (raises on failure)
 156         schema = INPUT_SCHEMAS[cls.path]
 157         jsonschema.validate(raw, schema)
 158 
 159         # Trim values and convert keys to use underscores
 160         kwargs = {k.replace("-", "_"): trim(v, 1024) for k, v in six.iteritems(raw)}
 161 
 162         return cls.to_python(kwargs)
 163 
 164     def get_culprit(self):
 165         return None
 166 
 167     def get_message(self):
 168         return u"Public key pinning validation failed for '{self.hostname}'".format(self=self)
 169 
 170     def get_tags(self):
 171         return [
 172             ("port", six.text_type(self.port)),
 173             ("include-subdomains", json.dumps(self.include_subdomains)),
 174             ("hostname", self.hostname),
 175         ]
 176 
 177     def get_origin(self):
 178         # not quite origin, but the domain that failed pinning
 179         return self.hostname
 180 
 181     def get_referrer(self):
 182         return None
 183 
 184     def should_filter(self, project=None):
 185         return False

 304 class Csp(SecurityReport):
 305     """
 306     A CSP violation report.
 307 
 308     See also: http://www.w3.org/TR/CSP/#violation-reports
 309 
 310     >>> {
 311     >>>     "document_uri": "http://example.com/",
 312     >>>     "violated_directive": "style-src cdn.example.com",
 313     >>>     "blocked_uri": "http://example.com/style.css",
 314     >>>     "effective_directive": "style-src",
 315     >>> }
 316     """
 317 
 318     LOCAL = "'self'"
 319     score = 1300
 320     display_score = 1300
 321 
 322     title = "CSP Report"
 323 
 324     @classmethod
 325     def from_raw(cls, raw):
 326         # Firefox doesn't send effective-directive, so parse it from
 327         # violated-directive but prefer effective-directive when present
 328         #
 329         # refs: https://bugzil.la/1192684#c8
 330         try:
 331             report = raw["csp-report"]
 332             report["effective-directive"] = report.get(
 333                 "effective-directive", report["violated-directive"].split(None, 1)[0]
 334             )
 335         except (KeyError, IndexError):
 336             pass
 337 
 338         # Validate the raw data against the input schema (raises on failure)
 339         schema = INPUT_SCHEMAS[cls.path]
 340         jsonschema.validate(raw, schema)
 341 
 342         # For CSP, the values we want are nested under the 'csp-report' key.
 343         raw = raw["csp-report"]
 344         # Trim values and convert keys to use underscores
 345         kwargs = {k.replace("-", "_"): trim(v, 1024) for k, v in six.iteritems(raw)}
 346 
 347         return cls.to_python(kwargs)

 349     def get_message(self):
 350         templates = {
 351             "child-src": (u"Blocked 'child' from '{uri}'", "Blocked inline 'child'"),
 352             "connect-src": (u"Blocked 'connect' from '{uri}'", "Blocked inline 'connect'"),
 353             "font-src": (u"Blocked 'font' from '{uri}'", "Blocked inline 'font'"),
 354             "form-action": (u"Blocked 'form' action to '{uri}'",),  # no inline option
 355             "img-src": (u"Blocked 'image' from '{uri}'", "Blocked inline 'image'"),
 356             "manifest-src": (u"Blocked 'manifest' from '{uri}'", "Blocked inline 'manifest'"),
 357             "media-src": (u"Blocked 'media' from '{uri}'", "Blocked inline 'media'"),
 358             "object-src": (u"Blocked 'object' from '{uri}'", "Blocked inline 'object'"),
 359             "script-src": (
 360                 u"Blocked 'script' from '{uri}'",
 361                 "Blocked unsafe (eval() or inline) 'script'",
 362             ),
 363             "script-src-elem": (
 364                 u"Blocked 'script' from '{uri}'",
 365                 "Blocked unsafe 'script' element",
 366             ),
 367             "script-src-attr": (
 368                 u"Blocked inline script attribute from '{uri}'",
 369                 "Blocked inline script attribute",
 370             ),
 371             "style-src": (u"Blocked 'style' from '{uri}'", "Blocked inline 'style'"),
 372             "style-src-elem": (
 373                 u"Blocked 'style' from '{uri}'",
 374                 "Blocked 'style' or 'link' element",
 375             ),
 376             "style-src-attr": (u"Blocked style attribute from '{uri}'", "Blocked style attribute"),
 377             "unsafe-inline": (None, u"Blocked unsafe inline 'script'"),
 378             "unsafe-eval": (None, u"Blocked unsafe eval() 'script'"),
 379         }
 380         default_template = ("Blocked {directive!r} from {uri!r}", "Blocked inline {directive!r}")
 381 
 382         directive = self.local_script_violation_type or self.effective_directive
 383         uri = self.normalized_blocked_uri
 384         index = 1 if uri == self.LOCAL else 0
 385 
 386         try:
 387             tmpl = templates[directive][index]
 388         except (KeyError, IndexError):
 389             tmpl = default_template[index]
 390 
 391         return tmpl.format(directive=directive, uri=uri)

 419     def should_filter(self, project=None):
 420         disallowed = ()
 421         paths = ["blocked_uri", "source_file"]
 422         uris = [getattr(self, path) for path in paths if hasattr(self, path)]
 423 
 424         if project is None or bool(project.get_option("sentry:csp_ignored_sources_defaults", True)):
 425             disallowed += DEFAULT_DISALLOWED_SOURCES
 426         if project is not None:
 427             disallowed += tuple(project.get_option("sentry:csp_ignored_sources", []))
 428 
 429         if disallowed and any(is_valid_origin(uri, allowed=disallowed) for uri in uris):
 430             return True
 431 
 432         return False

 434     def _sanitized_blocked_uri(self):
 435         # HACK: This is 100% to work around Stripe urls
 436         # that will casually put extremely sensitive information
 437         # in querystrings. The real solution is to apply
 438         # data scrubbing to all tags generically
 439         # TODO this could be done in filter_csp
 440         # instead but that might only be run conditionally on the org/project settings
 441         # relevant code is @L191:
 442         #
 443         #   if netloc == 'api.stripe.com':
 444         #       query = ''
 445         #       fragment = ''
 446 
 447         uri = self.blocked_uri
 448         if uri.startswith("https://api.stripe.com/"):
 449             return urlunsplit(urlsplit(uri)[:3] + (None, None))
 450         return uri

 460     def _normalize_value(self, value):
 461         keywords = ("'none'", "'self'", "'unsafe-inline'", "'unsafe-eval'")
 462         all_schemes = ("data:", "mediastream:", "blob:", "filesystem:", "http:", "https:", "file:")
 463 
 464         # > If no scheme is specified, the same scheme as the one used to
 465         # > access the protected document is assumed.
 466         # Source: https://developer.mozilla.org/en-US/docs/Web/Security/CSP/CSP_policy_directives
 467         if value in keywords:
 468             return value
 469 
 470         # normalize a value down to 'self' if it matches the origin of document-uri
 471         # FireFox transforms a 'self' value into the spelled out origin, so we
 472         # want to reverse this and bring it back
 473         if value.startswith(all_schemes):
 474             if self._normalized_document_uri == self._normalize_uri(value):
 475                 return self.LOCAL
 476             # Their rule had an explicit scheme, so let's respect that
 477             return value
 478 
 479         # value doesn't have a scheme, but let's see if their
 480         # hostnames match at least, if so, they're the same
 481         if value == self._normalized_document_uri:
 482             return self.LOCAL
 483 
 484         # Now we need to stitch on a scheme to the value
 485         scheme = self.document_uri.split(":", 1)[0]
 486         # But let's not stitch on the boring values
 487         if scheme in ("http", "https"):
 488             return value
 489         return self._unsplit(scheme, value)
 490 
 491     @memoize
 492     def local_script_violation_type(self):
 493         """
 494         If this is a locally-sourced script-src error, gives the type.
 495         """
 496         if (
 497             self.violated_directive
 498             and self.effective_directive == "script-src"
 499             and self.normalized_blocked_uri == self.LOCAL
 500         ):
 501             if "'unsafe-inline'" in self.violated_directive:
 502                 return "unsafe-inline"
 503             elif "'unsafe-eval'" in self.violated_directive:
 504                 return "unsafe-eval"
 505         return None
 506 
 507     def _normalize_uri(self, value):
 508         if value in ("", self.LOCAL, self.LOCAL.strip("'")):
 509             return self.LOCAL
 510 
 511         # A lot of these values get reported as literally
 512         # just the scheme. So a value like 'data' or 'blob', which
 513         # are valid schemes, just not a uri. So we want to
 514         # normalize it into a uri.
 515         if ":" not in value:
 516             scheme, hostname = value, ""
 517         else:
 518             scheme, hostname = urlsplit(value)[:2]
 519             if scheme in ("http", "https"):
 520                 return hostname
 521         return self._unsplit(scheme, hostname)
 522 
 523     def _unsplit(self, scheme, hostname):
 524         return urlunsplit((scheme, hostname, "", None, None))
