  22 class Message(Interface):
  23     """
  24     A message consisting of either a ``formatted`` arg, or an optional
  25     ``message`` with a list of ``params``.
  26 
  27     - ``message`` and ``formatted`` are limited to 1000 characters.
  28 
  29     >>> {
  30     >>>     "message": "My raw message with interpreted strings like %s",
  31     >>>     "formatted": "My raw message with interpreted strings like this",
  32     >>>     "params": ["this"]
  33     >>> }
  34     """
  35 
  36     score = 0
  37     display_score = 2050
  38     path = "logentry"
  39     external_type = "message"
  40 
  41     @classmethod
  42     def to_python(cls, data):
  43         for key in ("message", "formatted", "params"):
  44             data.setdefault(key, None)
  45         return cls(**data)
  46 
  47     def to_json(self):
  48         return prune_empty_keys(
  49             {"message": self.message, "formatted": self.formatted, "params": self.params or None}
  50         )
  51 
  52     def to_string(self, event, is_public=False, **kwargs):
  53         return self.formatted or self.message
