 591         return None
 592     return jsonschema.Draft4Validator(
 593         INTERFACE_SCHEMAS[name],
 594         types={"array": (list, tuple)},

 599 def validate_and_default_interface(
 600     data, interface, name=None, meta=None, strip_nones=True, raise_on_invalid=False
 601 ):
 602     """
 603     Modify data to conform to named interface's schema.
 604 
 605     Takes the object in `data` and checks it against the schema for
 606     `interface`, removing or defaulting any keys that do not pass validation
 607     and adding defaults for any keys that are required by (and have a default
 608     value in) the schema.
 609 
 610     Returns whether the resulting modified data is valid against the schema and
 611     a list of any validation errors encountered in processing.
 612     """
 613     if meta is None:
 614         meta = Meta()
 615 
 616     is_valid = True
 617     needs_revalidation = False
 618     errors = []
 619 
 620     validator = validator_for_interface(interface)
 621     if validator is None:
 622         return (True, [])
 623     schema = validator.schema
 624 
 625     # Strip Nones so we don't have to take null into account for all schemas.
 626     if strip_nones and isinstance(data, dict):
 627         for k in data.keys():
 628             if data[k] is None:
 629                 del data[k]
 630 
 631     # Values that are missing entirely, but are required and should be defaulted
 632     if "properties" in schema and "required" in schema and isinstance(data, dict):
 633         for p in schema["required"]:
 634             if p not in data:
 635                 if p in schema["properties"] and "default" in schema["properties"][p]:
 636                     default = schema["properties"][p]["default"]
 637                     data[p] = default() if callable(default) else default
 638                 else:
 639                     meta.add_error(EventError.MISSING_ATTRIBUTE, data={"name": p})
 640                     errors.append({"type": EventError.MISSING_ATTRIBUTE, "name": p})
 641 
 642     validator_errors = list(validator.iter_errors(data))
 643     keyed_errors = [e for e in reversed(validator_errors) if len(e.path)]
 644     if len(validator_errors) > len(keyed_errors):
 645         needs_revalidation = True
 646 
 647     # Values that need to be defaulted or deleted because they are not valid.
 648     for key, group in groupby(keyed_errors, lambda e: e.path[0]):
 649         ve = six.next(group)
 650         is_max = ve.validator.startswith("max")
 651         if is_max:
 652             error_type = EventError.VALUE_TOO_LONG
 653         elif key == "environment":
 654             error_type = EventError.INVALID_ENVIRONMENT
 655         else:
 656             error_type = EventError.INVALID_DATA
 657         meta.enter(key).add_error(error_type, data[key])
 658         errors.append({"type": error_type, "name": name or key, "value": data[key]})
 659 
 660         if "default" in ve.schema:
 661             default = ve.schema["default"]
 662             data[key] = default() if callable(default) else default
 663         else:
 664             needs_revalidation = True
 665             del data[key]
 666 
 667     if needs_revalidation:
 668         is_valid = validator.is_valid(data)
 669 
 670     return is_valid, errors
