  25 def format_headers(value):
  26     if not value:
  27         return ()
  28 
  29     if isinstance(value, dict):
  30         value = value.items()
  31 
  32     result = []
  33     cookie_header = None
  34     for k, v in value:
  35         # If a header value is a list of header,
  36         # we want to normalize this into a comma separated list
  37         # This is how most other libraries handle this.
  38         # See: urllib3._collections:HTTPHeaderDict.itermerged
  39         if isinstance(v, list):
  40             v = ", ".join(v)
  41 
  42         if k.lower() == "cookie":
  43             cookie_header = v
  44         else:
  45             if not isinstance(v, six.string_types):
  46                 v = six.text_type(v)
  47             result.append((k.title(), v))
  48     return result, cookie_header

  51 def format_cookies(value):
  52     if not value:
  53         return ()
  54 
  55     if isinstance(value, six.string_types):
  56         value = parse_qsl(value, keep_blank_values=True)
  57 
  58     if isinstance(value, dict):
  59         value = value.items()
  60 
  61     return [map(fix_broken_encoding, (k.strip(), v)) for k, v in value]

  64 def fix_broken_encoding(value):
  65     """
  66     Strips broken characters that can't be represented at all
  67     in utf8. This prevents our parsers from breaking elsewhere.
  68     """
  69     if isinstance(value, six.text_type):
  70         value = value.encode("utf8", errors="replace")
  71     if isinstance(value, six.binary_type):
  72         value = value.decode("utf8", errors="replace")
  73     return value

  80 class Http(Interface):
  81     """
  82     The Request information is stored in the Http interface. Two arguments
  83     are required: ``url`` and ``method``.
  84 
  85     The ``env`` variable is a compounded dictionary of HTTP headers as well
  86     as environment information passed from the webserver. Sentry will explicitly
  87     look for ``REMOTE_ADDR`` in ``env`` for things which require an IP address.
  88 
  89     The ``data`` variable should only contain the request body (not the query
  90     string). It can either be a dictionary (for standard HTTP requests) or a
  91     raw request body.
  92 
  93     >>>  {
  94     >>>     "url": "http://absolute.uri/foo",
  95     >>>     "method": "POST",
  96     >>>     "data": "foo=bar",
  97     >>>     "query_string": "hello=world",
  98     >>>     "cookies": "foo=bar",
  99     >>>     "headers": [
 100     >>>         ["Content-Type", "text/html"]
 101     >>>     ],
 102     >>>     "env": {
 103     >>>         "REMOTE_ADDR": "192.168.0.1"
 104     >>>     }
 105     >>>  }
 106 
 107     .. note:: This interface can be passed as the 'request' key in addition
 108               to the full interface path.
 109     """

 118     def to_python(cls, data):
 119         data.setdefault("query_string", [])
 120         for key in (
 121             "method",
 122             "url",
 123             "fragment",
 124             "cookies",
 125             "headers",
 126             "data",
 127             "env",
 128             "inferred_content_type",
 129         ):
 130             data.setdefault(key, None)
 131         return cls(**data)

 148     @property
 149     def full_url(self):
 150         url = self.url
 151         if url:
 152             if self.query_string:
 153                 url = url + "?" + urlencode(get_path(self.query_string, filter=True))
 154             if self.fragment:
 155                 url = url + "#" + self.fragment
 156         return url

 174     def get_api_context(self, is_public=False, platform=None):
 175         if is_public:
 176             return {}
 177 
 178         cookies = self.cookies or ()
 179         if isinstance(cookies, dict):
 180             cookies = sorted(self.cookies.items())
 181 
 182         headers = self.headers or ()
 183         if isinstance(headers, dict):
 184             headers = sorted(self.headers.items())
 185 
 186         data = {
 187             "method": self.method,
 188             "url": self.url,
 189             "query": self.query_string,
 190             "fragment": self.fragment,
 191             "data": self.data,
 192             "headers": headers,
 193             "cookies": cookies,
 194             "env": self.env or None,
 195             "inferredContentType": self.inferred_content_type,

 199     def get_api_meta(self, meta, is_public=False, platform=None):
 200         if is_public:
 201             return None
 202 
 203         headers = meta.get("headers")
 204         if headers:
 205             headers_meta = headers.pop("", None)
 206             headers = {six.text_type(i): {"1": h[1]} for i, h in enumerate(sorted(headers.items()))}
 207             if headers_meta:
 208                 headers[""] = headers_meta
 209 
 210         cookies = meta.get("cookies")
 211         if cookies:
 212             cookies_meta = cookies.pop("", None)
 213             cookies = {six.text_type(i): {"1": h[1]} for i, h in enumerate(sorted(cookies.items()))}
 214             if cookies_meta:
 215                 cookies[""] = cookies_meta
 216 
 217         return {
 218             "": meta.get(""),
 219             "method": meta.get("method"),
 220             "url": meta.get("url"),
 221             "query": meta.get("query_string"),
 222             "data": meta.get("data"),
 223             "headers": headers,
 224             "cookies": cookies,
 225             "env": meta.get("env"),
 226         }
