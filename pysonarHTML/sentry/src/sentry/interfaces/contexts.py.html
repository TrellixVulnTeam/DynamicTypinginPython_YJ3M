  17 class _IndexFormatter(string.Formatter):
  18     def format_field(self, value, format_spec):
  19         if not format_spec and isinstance(value, bool):
  20             return value and "yes" or "no"
  21         return string.Formatter.format_field(self, value, format_spec)

  24 def format_index_expr(format_string, data):
  25     return six.text_type(_IndexFormatter().vformat(six.text_type(format_string), (), data).strip())

  33 class ContextType(object):
  34     indexed_fields = None
  35     type = None
  36 
  37     def __init__(self, alias, data):
  38         self.alias = alias
  39         ctx_data = {}
  40         for key, value in six.iteritems(trim(data)):
  41             # we use simple checks here, rather than ' in set()' to avoid
  42             # issues with maps/lists
  43             if value is not None and value != "":
  44                 ctx_data[force_text(key)] = value
  45         self.data = ctx_data
  46 
  47     def to_json(self):
  48         rv = dict(self.data)
  49         rv["type"] = self.type
  50         return prune_empty_keys(rv)
  51 
  52     @classmethod
  53     def values_for_data(cls, data):
  54         rv = []
  55         for context in six.itervalues(data.get("contexts") or {}):
  56             if context and context.get("type") == cls.type:
  57                 rv.append(context)
  58         return rv
  59 
  60     @classmethod
  61     def primary_value_for_data(cls, data):
  62         val = get_path(data, "contexts", cls.type)
  63         if val and val.get("type") == cls.type:
  64             return val
  65 
  66         rv = cls.values_for_data(data)
  67         if len(rv) == 1:
  68             return rv[0]
  69 
  70     def iter_tags(self):
  71         if self.indexed_fields:
  72             for field, f_string in six.iteritems(self.indexed_fields):
  73                 try:
  74                     value = format_index_expr(f_string, self.data)
  75                 except KeyError:
  76                     continue
  77                 if value:
  78                     if not field:
  79                         yield (self.alias, value)
  80                     else:
  81                         yield ("%s.%s" % (self.alias, field), value)

 141 class Contexts(Interface):
 142     """
 143     This interface stores context specific information.
 144     """
 145 
 146     display_score = 1100
 147     score = 800
 148 
 149     @classmethod
 150     def to_python(cls, data):
 151         rv = {}
 152         for alias, value in six.iteritems(data):
 153             # XXX(markus): The `None`-case should be handled in the UI and
 154             # other consumers of this interface
 155             if value is not None:
 156                 rv[alias] = cls.normalize_context(alias, value)
 157         return cls(**rv)
 158 
 159     @classmethod
 160     def normalize_context(cls, alias, data):
 161         ctx_type = data.get("type", alias)
 162         ctx_cls = context_types.get(ctx_type, DefaultContextType)
 163         return ctx_cls(alias, data)
 164 
 165     def iter_contexts(self):
 166         return six.itervalues(self._data)
 167 
 168     def to_json(self):
 169         rv = {}
 170         for alias, inst in six.iteritems(self._data):
 171             rv[alias] = inst.to_json()
 172         return rv
 173 
 174     def iter_tags(self):
 175         for inst in self.iter_contexts():
 176             for tag in inst.iter_tags():
 177                 yield tag
