   5 from sentry.interfaces.base import Interface
   6 from sentry.utils.json import prune_empty_keys

   8 
   9 class DebugMeta(Interface):
  10     """
  11     Holds debug meta information for processing stacktraces
  12     and similar things.  This information is deleted after event processing.
  13 
  14     Currently two attributes exist:
  15 
  16     ``sdk_info``:
  17         sets the SDK that is used for the system.  This affects the lookup
  18         for system symbols.  If not defined, system symbols are not looked up.
  19     ``images``:
  20         a list of debug images and their mappings.
  21     """
  22 
  23     ephemeral = False
  24     path = "debug_meta"
  25     external_type = "debugmeta"
  26 
  27     @classmethod
  28     def to_python(cls, data):
  29         return cls(
  30             images=data.get("images", None) or [],
  31             sdk_info=data.get("sdk_info"),
  32             is_debug_build=data.get("is_debug_build"),
  33         )
  34 
  35     def to_json(self):
  36         return prune_empty_keys(
  37             {
  38                 "images": self.images or None,
  39                 "sdk_info": self.sdk_info or None,
  40                 "is_debug_build": self.is_debug_build,
  41             }
  42         )
