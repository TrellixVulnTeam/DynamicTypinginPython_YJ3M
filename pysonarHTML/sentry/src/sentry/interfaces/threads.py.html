  11 def get_stacktrace(value, raw=False):
  12     # Special case: if the thread has no frames we set the
  13     # stacktrace to none.  Otherwise this will fail really
  14     # badly.
  15     if value and value.get("frames"):
  16         return Stacktrace.to_python(value, raw=raw)

  19 class Threads(Interface):
  20     score = 1900
  21     grouping_variants = ["system", "app"]

  23     @classmethod
  24     def to_python(cls, data):
  25         threads = []
  26 
  27         for thread in data.get("values") or ():
  28             if thread is None:
  29                 # XXX(markus): We should handle this in the UI and other
  30                 # consumers of this interface
  31                 continue
  32             threads.append(
  33                 {
  34                     "stacktrace": get_stacktrace(thread.get("stacktrace")),
  35                     "raw_stacktrace": get_stacktrace(thread.get("raw_stacktrace"), raw=True),
  36                     "id": trim(thread.get("id"), 40),
  37                     "crashed": bool(thread.get("crashed")),
  38                     "current": bool(thread.get("current")),
  39                     "name": trim(thread.get("name"), 200),
  40                 }
  41             )
  42 
  43         return cls(values=threads)

  45     def to_json(self):
  46         def export_thread(data):
  47             if data is None:
  48                 return None
  49 
  50             rv = {
  51                 "id": data["id"],
  52                 "current": data["current"],
  53                 "crashed": data["crashed"],
  54                 "name": data["name"],
  55                 "stacktrace": None,
  56             }
  57             if data["stacktrace"]:
  58                 rv["stacktrace"] = data["stacktrace"].to_json()
  59             if data["raw_stacktrace"]:
  60                 rv["raw_stacktrace"] = data["raw_stacktrace"].to_json()
  61             return prune_empty_keys(rv)
  62 
  63         return prune_empty_keys({"values": [export_thread(x) for x in self.values]})

  65     def get_api_context(self, is_public=False, platform=None):
  66         def export_thread(data):
  67             rv = {
  68                 "id": data["id"],
  69                 "current": data["current"],
  70                 "crashed": data["crashed"],
  71                 "name": data["name"],
  72                 "stacktrace": None,
  73                 "rawStacktrace": None,
  74             }
  75             if data["stacktrace"]:
  76                 rv["stacktrace"] = data["stacktrace"].get_api_context(is_public=is_public)
  77             if data["raw_stacktrace"]:
  78                 rv["rawStacktrace"] = data["raw_stacktrace"].get_api_context(is_public=is_public)
  79             return rv
  80 
  81         return {"values": [export_thread(x) for x in self.values]}
