  23 def get_interface(name):
  24     try:
  25         name = get_canonical_name(name)
  26         import_path = settings.SENTRY_INTERFACES[name]
  27     except KeyError:
  28         raise ValueError("Invalid interface name: %s" % (name,))
  29 
  30     try:
  31         interface = import_string(import_path)
  32     except Exception:
  33         raise ValueError("Unable to load interface: %s" % (name,))
  34 
  35     return interface

  38 def get_interfaces(data):
  39     result = []
  40     for key, data in six.iteritems(data):
  41         # Skip invalid interfaces that were nulled out during normalization
  42         if data is None:
  43             continue
  44 
  45         try:
  46             cls = get_interface(key)
  47         except ValueError:
  48             continue
  49 
  50         value = safe_execute(cls.to_python, data, _with_transaction=False)
  51         if not value:
  52             continue
  53 
  54         result.append((key, value))
  55 
  56     return OrderedDict(
  57         (k, v) for k, v in sorted(result, key=lambda x: x[1].get_score(), reverse=True)
  58     )

  65 class Interface(object):
  66     """
  67     An interface is a structured representation of data, which may
  68     render differently than the default ``extra`` metadata in an event.
  69     """
  70 
  71     _data = None
  72     score = 0
  73     display_score = None
  74     ephemeral = False
  75     grouping_variants = ["default"]
  76 
  77     def __init__(self, **data):
  78         self._data = data or {}
  79 
  80     @classproperty
  81     def path(cls):
  82         """The 'path' of the interface which is the root key in the data."""
  83         return cls.__name__.lower()
  84 
  85     @classproperty
  86     def external_type(cls):
  87         """The external name of the interface.  This is mostly the same as
  88         path with some small differences (message, debugmeta).
  89         """
  90         return cls.path
  91 
  92     def __eq__(self, other):
  93         if not isinstance(self, type(other)):
  94             return False
  95         return self._data == other._data
  96 
  97     def __getstate__(self):
  98         return {"_data": self._data}
  99 
 100     def __setstate__(self, state):
 101         self.__dict__.update(state)
 102         if not hasattr(self, "_data"):
 103             self._data = {}
 104 
 105     def __getattr__(self, name):
 106         return self._data[name]
 107 
 108     def __setattr__(self, name, value):
 109         if name == "_data":
 110             self.__dict__["_data"] = value
 111         else:
 112             self._data[name] = value
 113 
 114     @classmethod
 115     def to_python(cls, data):
 116         """Creates a python interface object from the given raw data.
 117 
 118         This function can assume fully normalized and valid data. It can create
 119         defaults where data is missing but does not need to handle interface
 120         validation.
 121         """
 122         return cls(**data) if data is not None else None
 123 
 124     def get_raw_data(self):
 125         """Returns the underlying raw data."""
 126         return self._data
 127 
 128     def get_api_context(self, is_public=False, platform=None):
 129         return self.to_json()
 130 
 131     def get_api_meta(self, meta, is_public=False, platform=None):
 132         return meta
 133 
 134     def to_json(self):
 135         return prune_empty_keys(self._data)
 136 
 137     def get_title(self):
 138         return _(type(self).__name__)
 139 
 140     def get_display_score(self):
 141         return self.display_score or self.score
 142 
 143     def get_score(self):
 144         return self.score
 145 
 146     def iter_tags(self):
 147         return iter(())
 148 
 149     def to_string(self, event, is_public=False, **kwargs):
 150         return ""

 152     def to_email_html(self, event, **kwargs):
 153         body = self.to_string(event)
 154         if not body:
 155             return ""
 156         return "<pre>%s</pre>" % (escape(body),)
