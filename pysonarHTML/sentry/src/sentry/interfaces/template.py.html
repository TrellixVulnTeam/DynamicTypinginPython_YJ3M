   9 class Template(Interface):
  10     """
  11     A rendered template (generally used like a single frame in a stacktrace).
  12 
  13     The attributes ``filename``, ``context_line``, and ``lineno`` are required.
  14 
  15     >>>  {
  16     >>>     "abs_path": "/real/file/name.html"
  17     >>>     "filename": "file/name.html",
  18     >>>     "pre_context": [
  19     >>>         "line1",
  20     >>>         "line2"
  21     >>>     ],
  22     >>>     "context_line": "line3",
  23     >>>     "lineno": 3,
  24     >>>     "post_context": [
  25     >>>         "line4",
  26     >>>         "line5"
  27     >>>     ],
  28     >>> }
  29 
  30     .. note:: This interface can be passed as the 'template' key in addition
  31               to the full interface path.
  32     """
  33 
  34     score = 1100
  35 
  36     @classmethod
  37     def to_python(cls, data):
  38         for key in (
  39             "abs_path",
  40             "filename",
  41             "context_line",
  42             "lineno",
  43             "pre_context",
  44             "post_context",
  45         ):
  46             data.setdefault(key, None)
  47         return cls(**data)
  48 
  49     def to_string(self, event, is_public=False, **kwargs):
  50         context = get_context(
  51             lineno=self.lineno,
  52             context_line=self.context_line,
  53             pre_context=self.pre_context,
  54             post_context=self.post_context,
  55         )
  56 
  57         result = ["Stacktrace (most recent call last):", "", self.get_traceback(event, context)]
  58 
  59         return "\n".join(result)
  60 
  61     def get_traceback(self, event, context):
  62         result = [event.message, "", 'File "%s", line %s' % (self.filename, self.lineno), ""]
  63         result.extend([n[1].strip("\n") if n[1] else "" for n in context])
  64 
  65         return "\n".join(result)
  66 
  67     def get_api_context(self, is_public=False, platform=None):
  68         return {
  69             "lineNo": self.lineno,
  70             "filename": self.filename,
  71             "context": get_context(
  72                 lineno=self.lineno,
  73                 context_line=self.context_line,
  74                 pre_context=self.pre_context,
  75                 post_context=self.post_context,
  76             ),
  77         }
  78 
  79     def get_api_meta(self, meta, is_public=False, platform=None):
  80         return {
  81             "": meta.get(""),
  82             "lineNo": meta.get("lineno"),
  83             "filename": meta.get("filename"),
  84             "context": get_context(
  85                 lineno=meta.get("lineno"),
  86                 context_line=meta.get("context_line"),
  87                 pre_context=meta.get("pre_context"),
  88                 post_context=meta.get("post_context"),
  89             ),
  90         }
