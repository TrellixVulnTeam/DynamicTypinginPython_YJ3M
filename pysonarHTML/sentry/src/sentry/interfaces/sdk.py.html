   9 class Sdk(Interface):
  10     """
  11     The SDK used to transmit this event.
  12 
  13     >>> {
  14     >>>     "name": "sentry.java",
  15     >>>     "version": "1.7.10",
  16     >>>     "integrations": ["log4j"],
  17     >>>     "packages": [
  18     >>>         {
  19     >>>             "name": "maven:io.sentry.sentry",
  20     >>>             "version": "1.7.10",
  21     >>>         }
  22     >>>     ]
  23     >>> }
  24     """
  25 
  26     @classmethod
  27     def to_python(cls, data):
  28         for key in ("name", "version", "integrations", "packages"):
  29             data.setdefault(key, None)
  30 
  31         return cls(**data)
  32 
  33     def to_json(self):
  34         return prune_empty_keys(
  35             {
  36                 "name": self.name,
  37                 "version": self.version,
  38                 "integrations": self.integrations or None,
  39                 "packages": self.packages or None,
  40             }
  41         )
  42 
  43     def get_api_context(self, is_public=False, platform=None):
  44         return {"name": self.name, "version": self.version}
  45 
  46     def get_api_meta(self, meta, is_public=False, platform=None):
