  16 def max_addr(cur, addr):
  17     if addr is None:
  18         return cur
  19     length = len(addr) - 2
  20     if length > cur:
  21         return length
  22     return cur

  25 def pad_hex_addr(addr, length):
  26     if length is None or addr is None:
  27         return addr
  28     return "0x" + addr[2:].rjust(length, "0")

  40 def to_hex_addr(addr):
  41     if addr is None:
  42         return None
  43     elif isinstance(addr, six.integer_types):
  44         rv = "0x%x" % addr
  45     elif isinstance(addr, six.string_types):
  46         if addr[:2] == "0x":
  47             addr = int(addr[2:], 16)
  48         rv = "0x%x" % int(addr)
  49     else:
  50         raise ValueError("Unsupported address format %r" % (addr,))
  51     if len(rv) > 24:
  52         raise ValueError("Address too long %r" % (rv,))
  53     return rv

  56 def get_context(lineno, context_line, pre_context=None, post_context=None):
  57     if lineno is None:
  58         return []
  59 
  60     if context_line is None and not (pre_context or post_context):
  61         return []
  62 
  63     lineno = int(lineno)
  64     context = []
  65     start_lineno = lineno - len(pre_context or [])
  66     if pre_context:
  67         start_lineno = lineno - len(pre_context)
  68         at_lineno = start_lineno
  69         for line in pre_context:
  70             context.append((at_lineno, line))
  71             at_lineno += 1
  72     else:
  73         start_lineno = lineno
  74         at_lineno = lineno
  75 
  76     if start_lineno < 0:
  77         start_lineno = 0
  78 
  79     context.append((at_lineno, context_line))
  80     at_lineno += 1
  81 
  82     if post_context:
  83         for line in post_context:
  84             context.append((at_lineno, line))
  85             at_lineno += 1
  86 
  87     return context

 117 def handle_nan(value):
 118     "Remove nan values that can't be json encoded"
 119     if isinstance(value, float):
 120         if value == float("inf"):
 121             return "<inf>"
 122         if value == float("-inf"):
 123             return "<-inf>"
 124         # lol checking for float('nan')
 125         if value != value:
 126             return "<nan>"
 127     return value

 263     def is_url(self):
 264         if not self.abs_path:
 265             return False
 266         # URLs can be generated such that they are:
 267         #   blob:http://example.com/7f7aaadf-a006-4217-9ed5-5fbf8585c6c0
 268         # https://developer.mozilla.org/en-US/docs/Web/API/URL/createObjectURL
 269         if self.abs_path.startswith("blob:"):
 270             return True
 271         return is_url(self.abs_path)

 279     def is_unhashable_module(self, platform):
 280         # Fix for the case where module is a partial copy of the URL
 281         # and should not be hashed
 282         if (
 283             platform == "javascript"
 284             and "/" in self.module
 285             and self.abs_path
 286             and self.abs_path.endswith(self.module)
 287         ):
 288             return True
 289         elif platform == "java" and "$$Lambda$" in self.module:
 290             return True
 291         return False

 320 class Stacktrace(Interface):
 321     """
 322     A stacktrace contains a list of frames, each with various bits (most optional)
 323     describing the context of that frame. Frames should be sorted from oldest
 324     to newest.
 325 
 326     The stacktrace contains an element, ``frames``, which is a list of hashes. Each
 327     hash must contain **at least** the ``filename`` attribute. The rest of the values
 328     are optional, but recommended.
 329 
 330     Additionally, if the list of frames is large, you can explicitly tell the
 331     system that you've omitted a range of frames. The ``frames_omitted`` must
 332     be a single tuple two values: start and end. For example, if you only
 333     removed the 8th frame, the value would be (8, 9), meaning it started at the
 334     8th frame, and went until the 9th (the number of frames omitted is
 335     end-start). The values should be based on a one-index.
 336 
 337     The list of frames should be ordered by the oldest call first.
 338 
 339     Each frame must contain the following attributes:
 340 
 341     ``filename``
 342       The relative filepath to the call
 343 
 344     OR
 345 
 346     ``function``
 347       The name of the function being called
 348 
 349     OR
 350 
 351     ``module``
 352       Platform-specific module path (e.g. stacktrace)
 353 
 354     The following additional attributes are supported:
 355 
 356     ``lineno``
 357       The line number of the call
 358     ``colno``
 359       The column number of the call
 360     ``abs_path``
 361       The absolute path to filename
 362     ``context_line``
 363       Source code in filename at lineno
 364     ``pre_context``
 365       A list of source code lines before context_line (in order) -- usually [lineno - 5:lineno]
 366     ``post_context``
 367       A list of source code lines after context_line (in order) -- usually [lineno + 1:lineno + 5]
 368     ``in_app``
 369       Signifies whether this frame is related to the execution of the relevant
 370       code in this stacktrace. For example, the frames that might power the
 371       framework's webserver of your app are probably not relevant, however calls
 372       to the framework's library once you start handling code likely are. See
 373       notes below on implicity ``in_app`` behavior.
 374     ``vars``
 375       A mapping of variables which were available within this frame (usually context-locals).
 376     ``package``
 377       Name of the package or object file that the frame is contained in.  This
 378       for instance can be the name of a DLL, .NET Assembly, jar file, object
 379       file etc.
 380 
 381     >>> {
 382     >>>     "frames": [{
 383     >>>         "abs_path": "/real/file/name.py"
 384     >>>         "filename": "file/name.py",
 385     >>>         "function": "myfunction",
 386     >>>         "vars": {
 387     >>>             "key": "value"
 388     >>>         },
 389     >>>         "pre_context": [
 390     >>>             "line1",
 391     >>>             "line2"
 392     >>>         ],
 393     >>>         "context_line": "line3",
 394     >>>         "lineno": 3,
 395     >>>         "in_app": true,
 396     >>>         "post_context": [
 397     >>>             "line4",
 398     >>>             "line5"
 399     >>>         ],
 400     >>>     }],
 401     >>>     "frames_omitted": [13, 56]
 402     >>> }
 403 
 404     Implicity ``in_app`` behavior exists when the value is not specified on all
 405     frames within a stacktrace (or collectively within an exception if this is
 406     part of a chain).
 407 
 408     If **any frame** is marked with ``in_app=True`` or ``in_app=False``:
 409 
 410     - Set ``in_app=False`` where ``in_app is None``
 411 
 412     If **all frames** are marked identical values for ``in_app``:
 413 
 414     - Set ``in_app=False`` on all frames
 415 
 416     .. note:: This interface can be passed as the 'stacktrace' key in addition
 417               to the full interface path.
 418     """

 499     def get_stacktrace(
 500         self, event, system_frames=True, newest_first=None, max_frames=None, header=True
 501     ):
 502         if newest_first is None:
 503             newest_first = is_newest_frame_first(event)
 504 
 505         result = []
 506         if header:
 507             if newest_first:
 508                 result.append(_("Stacktrace (most recent call first):"))
 509             else:
 510                 result.append(_("Stacktrace (most recent call last):"))
 511 
 512             result.append("")
 513 
 514         frames = self.frames
 515 
 516         num_frames = len(frames)
 517 
 518         if not system_frames:
 519             frames = [f for f in frames if f.in_app is not False]
 520             if not frames:
 521                 frames = self.frames
 522 
 523         if newest_first:
 524             frames = frames[::-1]
 525 
 526         if max_frames:
 527             visible_frames = max_frames
 528             if newest_first:
 529                 start, stop = None, max_frames
 530             else:
 531                 start, stop = -max_frames, None
 532 
 533         else:
 534             visible_frames = len(frames)
 535             start, stop = None, None
 536 
 537         if not newest_first and visible_frames < num_frames:
 538             result.extend(
 539                 (
 540                     "(%d additional frame(s) were not displayed)" % (num_frames - visible_frames,),
 541                     "...",
 542                 )
 543             )
 544 
 545         for frame in frames[start:stop]:
 546             result.append(frame.to_string(event))
 547 
 548         if newest_first and visible_frames < num_frames:
 549             result.extend(
 550                 (
 551                     "...",
 552                     "(%d additional frame(s) were not displayed)" % (num_frames - visible_frames,),
 553                 )
 554             )
 555 
 556         return "\n".join(result)
