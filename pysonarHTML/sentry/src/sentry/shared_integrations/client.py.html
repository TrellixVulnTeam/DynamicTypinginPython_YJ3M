  23 class BaseApiResponse(object):
  24     text = ""
  25 
  26     def __init__(self, headers=None, status_code=None):
  27         self.headers = headers
  28         self.status_code = status_code

  30     def __repr__(self):
  31         return u"<%s: code=%s, content_type=%s>" % (
  32             type(self).__name__,
  33             self.status_code,
  34             self.headers.get("Content-Type", "") if self.headers else "",
  35         )
  36 
  37     @cached_property
  38     def rel(self):
  39         if not self.headers:
  40             return {}
  41         link_header = self.headers.get("Link")
  42         if not link_header:
  43             return {}
  44         return {item["rel"]: item["url"] for item in requests.utils.parse_header_links(link_header)}

  46     @classmethod
  47     def from_response(self, response, allow_text=False):
  48         # XXX(dcramer): this doesnt handle leading spaces, but they're not common
  49         # paths so its ok
  50         if response.text.startswith(u"<?xml"):
  51             return XmlApiResponse(response.text, response.headers, response.status_code)
  52         elif response.text.startswith("<"):
  53             if not allow_text:
  54                 raise ValueError(u"Not a valid response type: {}".format(response.text[:128]))
  55             elif response.status_code < 200 or response.status_code >= 300:
  56                 raise ValueError(
  57                     u"Received unexpected plaintext response for code {}".format(
  58                         response.status_code
  59                     )
  60                 )
  61             return TextApiResponse(response.text, response.headers, response.status_code)
  62 
  63         # Some APIs will return JSON with an invalid content-type, so we try
  64         # to decode it anyways
  65         if "application/json" not in response.headers.get("Content-Type", ""):
  66             try:
  67                 data = json.loads(response.text, object_pairs_hook=OrderedDict)
  68             except (TypeError, ValueError):
  69                 if allow_text:
  70                     return TextApiResponse(response.text, response.headers, response.status_code)
  71                 raise UnsupportedResponseType(
  72                     response.headers.get("Content-Type", ""), response.status_code
  73                 )
  74         else:
  75             data = json.loads(response.text, object_pairs_hook=OrderedDict)
  76 
  77         if isinstance(data, dict):
  78             return MappingApiResponse(data, response.headers, response.status_code)
  79         elif isinstance(data, (list, tuple)):
  80             return SequenceApiResponse(data, response.headers, response.status_code)
  81         else:
  82             raise NotImplementedError

 132     def __init__(self, verify_ssl=True, logging_context=None):

 148     def get_cache_prefix(self):

 151     def track_response_data(self, code, span, error=None, resp=None):
 152         metrics.incr(
 153             u"%s.http_response" % (self.datadog_prefix),
 154             sample_rate=1.0,
 155             tags={self.integration_type: self.name, "status": code},
 156         )
 157 
 158         span.set_http_status(code)
 159         span.set_tag(self.integration_type, self.name)
 160 
 161         extra = {
 162             self.integration_type: self.name,
 163             "status_string": six.text_type(code),
 164             "error": six.text_type(error)[:256] if error else None,
 165         }
 166         extra.update(getattr(self, "logging_context", None) or {})
 167         self.logger.info(u"%s.http_response" % (self.integration_type), extra=extra)
 168 
 169     def build_url(self, path):
 170         if path.startswith("/"):
 171             if not self.base_url:
 172                 raise ValueError(u"Invalid URL: {}".format(path))
 173             return u"{}{}".format(self.base_url, path)
 174         return path

 176     def _request(
 177         self,
 178         method,
 179         path,
 180         headers=None,
 181         data=None,
 182         params=None,
 183         auth=None,
 184         json=True,
 185         allow_text=None,
 186         allow_redirects=None,
 187         timeout=None,
 188     ):
 189 
 190         if allow_text is None:
 191             allow_text = self.allow_text
 192 
 193         if allow_redirects is None:
 194             allow_redirects = self.allow_redirects
 195 
 196         if allow_redirects is None:  # is still None
 197             allow_redirects = method.upper() == "GET"
 198 
 199         if timeout is None:
 200             timeout = 30
 201 
 202         full_url = self.build_url(path)
 203         session = build_session()
 204 
 205         metrics.incr(
 206             u"%s.http_request" % self.datadog_prefix,
 207             sample_rate=1.0,
 208             tags={self.integration_type: self.name},
 209         )
 210 
 211         with sentry_sdk.start_span(
 212             op=u"{}.http".format(self.integration_type),
 213             transaction=u"{}.http_response.{}".format(self.integration_type, self.name),
 214         ) as span:
 215             try:
 216                 resp = getattr(session, method.lower())(
 217                     url=full_url,
 218                     headers=headers,
 219                     json=data if json else None,
 220                     data=data if not json else None,
 221                     params=params,
 222                     auth=auth,
 223                     verify=self.verify_ssl,
 224                     allow_redirects=allow_redirects,
 225                     timeout=timeout,
 226                 )
 227                 resp.raise_for_status()
 228             except ConnectionError as e:
 229                 self.track_response_data("connection_error", span, e)
 230                 raise ApiHostError.from_exception(e)
 231             except Timeout as e:
 232                 self.track_response_data("timeout", span, e)
 233                 raise ApiTimeoutError.from_exception(e)
 234             except HTTPError as e:
 235                 resp = e.response
 236                 if resp is None:
 237                     self.track_response_data("unknown", span, e)
 238                     self.logger.exception(
 239                         "request.error", extra={self.integration_type: self.name, "url": full_url}
 240                     )
 241                     raise ApiError("Internal Error", url=full_url)
 242                 self.track_response_data(resp.status_code, span, e)
 243                 raise ApiError.from_response(resp, url=full_url)
 244 
 245             self.track_response_data(resp.status_code, span, None, resp)
 246 
 247             if resp.status_code == 204:
 248                 return {}
 249 
 250             return BaseApiResponse.from_response(resp, allow_text=allow_text)

 259     def get_cached(self, path, *args, **kwargs):
 260         query = ""
 261         if kwargs.get("params", None):
 262             query = json.dumps(kwargs.get("params"), sort_keys=True)
 263         key = self.get_cache_prefix() + md5_text(self.build_url(path), query).hexdigest()
 264 
 265         result = cache.get(key)
 266         if result is None:
 267             result = self.request("GET", path, *args, **kwargs)
 268             cache.set(key, result, self.cache_time)
 269         return result
