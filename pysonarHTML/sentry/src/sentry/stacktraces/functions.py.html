  42 def replace_enclosed_string(s, start, end, replacement=None):
  43     if start not in s:
  44         return s
  45 
  46     depth = 0
  47 
  48     rv = []
  49     pair_start = None
  50     for idx, char in enumerate(s):
  51         if char == start:
  52             if depth == 0:
  53                 pair_start = idx
  54             depth += 1
  55         elif char == end:
  56             depth -= 1
  57             if depth == 0:
  58                 if replacement is not None:
  59                     if callable(replacement):
  60                         rv.append(replacement(s[pair_start + 1 : idx], pair_start))
  61                     else:
  62                         rv.append(replacement)
  63         elif depth == 0:
  64             rv.append(char)
  65 
  66     return "".join(rv)

  69 def split_func_tokens(s):
  70     buf = []
  71     rv = []
  72     stack = []
  73     end = 0
  74 
  75     for idx, char in enumerate(s):
  76         if char in PAIRS:
  77             stack.append(PAIRS[char])
  78         elif stack and char == stack[-1]:
  79             stack.pop()
  80             if not stack:
  81                 buf.append(s[end : idx + 1])
  82                 end = idx + 1
  83         elif not stack:
  84             if char.isspace():
  85                 if buf:
  86                     rv.append(buf)
  87                 buf = []
  88             else:
  89                 buf.append(s[end : idx + 1])
  90             end = idx + 1
  91 
  92     if buf:
  93         rv.append(buf)
  94 
  95     return ["".join(x) for x in rv]

  98 def trim_function_name(function, platform, normalize_lambdas=True):
  99     """Given a function value from the frame's function attribute this returns
 100     a trimmed version that can be stored in `function_name`.  This is only used
 101     if the client did not supply a value itself already.
 102     """
 103     if get_behavior_family_for_platform(platform) != "native":
 104         return function
 105     if function in ("<redacted>", "<unknown>"):
 106         return function
 107 
 108     original_function = function
 109     function = function.strip()
 110 
 111     # Ensure we don't operate on objc functions
 112     if function.startswith(("[", "+[", "-[")):
 113         return function
 114 
 115     # Chop off C++ trailers
 116     while True:
 117         match = _cpp_trailer_re.search(function)
 118         if match is None:
 119             break
 120         function = function[: match.start()].rstrip()
 121 
 122     # Because operator<< really screws with our balancing, so let's work
 123     # around that by replacing it with a character we do not observe in
 124     # `split_func_tokens` or `replace_enclosed_string`.
 125     function = (
 126         function.replace("operator<<", u"operator⟨⟨")
 127         .replace("operator<", u"operator⟨")
 128         .replace("operator()", u"operator◯")
 129         .replace(" -> ", u" ⟿ ")
 130         .replace("`anonymous namespace'", u"〔anonymousnamespace〕")
 131     )
 132 
 133     # normalize C++ lambdas.  This is necessary because different
 134     # compilers use different rules for now to name a lambda and they are
 135     # all quite inconsistent.  This does not give us perfect answers to
 136     # this problem but closer.  In particular msvc will call a lambda
 137     # something like `lambda_deadbeefeefffeeffeeff` whereas clang for
 138     # instance will name it `main::$_0` which will tell us in which outer
 139     # function it was declared.
 140     if normalize_lambdas:
 141         function = _lambda_re.sub("lambda", function)
 142 
 143     # Normalize MSVC anonymous namespaces from inline functions.  For inline
 144     # functions, the compiler inconsistently renders anonymous namespaces with
 145     # their hash.  For regular functions,  "`anonymous namespace'" is used.
 146     # The regular expression matches the trailing "::" to avoid accidental
 147     # replacement in mangled function names.
 148     if normalize_lambdas:
 149         function = _anon_namespace_re.sub(u"〔anonymousnamespace〕::", function)
 150 
 151     # Remove the arguments if there is one.
 152     def process_args(value, start):
 153         value = value.strip()
 154         if value in ("anonymous namespace", "operator"):
 155             return "(%s)" % value
 156         return ""
 157 
 158     function = replace_enclosed_string(function, "(", ")", process_args)
 159 
 160     # Resolve generic types, but special case rust which uses things like
 161     # <Foo as Bar>::baz to denote traits.
 162     def process_generics(value, start):
 163         # Special case for lambdas
 164         if value == "lambda" or _lambda_re.match(value):
 165             return "<%s>" % value
 166 
 167         if start > 0:
 168             return "<T>"
 169 
 170         # Rust special cases
 171         value = _rust_blanket_re.sub("", value)  # prefer trait for blanket impls
 172         value = replace_enclosed_string(value, "<", ">", process_generics)
 173         return value.split(" as ", 1)[0]
 174 
 175     function = replace_enclosed_string(function, "<", ">", process_generics)
 176 
 177     tokens = split_func_tokens(function)
 178 
 179     # MSVC demangles generic operator functions with a space between the
 180     # function name and the generics. Ensure that those two components both end
 181     # up in the function name.
 182     if len(tokens) > 1 and tokens[-1] == "<T>":
 183         tokens.pop()
 184         tokens[-1] += " <T>"
 185 
 186     # find the token which is the function name.  Since we chopped of C++
 187     # trailers there are only two cases we care about: the token left to
 188     # the -> return marker which is for instance used in Swift and if that
 189     # is not found, the last token in the last.
 190     #
 191     # ["unsigned", "int", "whatever"] -> whatever
 192     # ["@objc", "whatever", "->", "int"] -> whatever
 193     try:
 194         func_token = tokens[tokens.index(u"⟿") - 1]
 195     except ValueError:
 196         if tokens:
 197             func_token = tokens[-1]
 198         else:
 199             func_token = None
 200 
 201     if func_token:
 202         function = (
 203             func_token.replace(u"⟨", "<")
 204             .replace(u"◯", "()")
 205             .replace(u" ⟿ ", " -> ")
 206             .replace(u"〔anonymousnamespace〕", "`anonymous namespace'")
 207         )
 208 
 209     # This really should never happen
 210     else:
 211         function = original_function
 212 
 213     # trim off rust markers
 214     function = _rust_hash.sub("", function)
 215 
 216     # trim off windows decl markers
 217     return _windecl_hash.sub("\\1", function)

 220 def get_function_name_for_frame(frame, platform=None):
 221     """Given a frame object or dictionary this returns the actual function
 222     name trimmed.
 223     """
 224     if hasattr(frame, "get_raw_data"):
 225         frame = frame.get_raw_data()
 226 
 227     # if there is a raw function, prioritize the function unchanged
 228     if frame.get("raw_function"):
 229         return frame.get("function")
 230 
 231     # otherwise trim the function on demand
 232     rv = frame.get("function")
 233     if rv:
 234         return trim_function_name(rv, frame.get("platform") or platform)

 237 def set_in_app(frame, value):
 238     orig_in_app = frame.get("in_app")
 239     if orig_in_app == value:
 240         return
 241 
 242     orig_in_app = int(orig_in_app) if orig_in_app is not None else -1
 243     setdefault_path(frame, "data", "orig_in_app", value=orig_in_app)
 244     frame["in_app"] = value
