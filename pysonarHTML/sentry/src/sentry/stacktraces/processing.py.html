  29 class ProcessableFrame(object):
  30     def __init__(self, frame, idx, processor, stacktrace_info, processable_frames):
  31         self.frame = frame
  32         self.idx = idx
  33         self.processor = processor
  34         self.stacktrace_info = stacktrace_info
  35         self.data = None
  36         self.cache_key = None
  37         self.cache_value = None
  38         self.processable_frames = processable_frames
  39 
  40     def __repr__(self):
  41         return "<ProcessableFrame %r #%r at %r>" % (
  42             self.frame.get("function") or "unknown",
  43             self.idx,
  44             self.frame.get("instruction_addr"),
  45         )
  46 
  47     def __contains__(self, key):
  48         return key in self.frame
  49 
  50     def __getitem__(self, key):
  51         return self.frame[key]
  52 
  53     def get(self, key, default=None):
  54         return self.frame.get(key, default)
  55 
  56     def close(self):
  57         # manually break circular references
  58         self.closed = True
  59         self.processable_frames = None
  60         self.stacktrace_info = None
  61         self.processor = None
  62 
  63     @property
  64     def previous_frame(self):
  65         last_idx = len(self.processable_frames) - self.idx - 1 - 1
  66         if last_idx < 0:
  67             return
  68         return self.processable_frames[last_idx]
  69 
  70     def set_cache_value(self, value):
  71         if self.cache_key is not None:
  72             cache.set(self.cache_key, value, 3600)
  73             return True
  74         return False
  75 
  76     def set_cache_key_from_values(self, values):
  77         if values is None:
  78             self.cache_key = None
  79             return
  80 
  81         h = hash_values(values, seed=self.processor.__class__.__name__)
  82         self.cache_key = rv = "pf:%s" % h
  83         return rv

 119     def get_release(self, create=False):
 120         """Convenient helper to return the release for the current data
 121         and optionally creates the release if it's missing.  In case there
 122         is no release info it will return `None`.
 123         """
 124         release = self.data.get("release")
 125         if not release:
 126             return None
 127         if not create:
 128             return Release.get(project=self.project, version=self.data["release"])
 129         timestamp = self.data.get("timestamp")
 130         if timestamp is not None:
 131             date = datetime.fromtimestamp(timestamp).replace(tzinfo=timezone.utc)
 132         else:
 133             date = None
 134         return Release.get_or_create(
 135             project=self.project, version=self.data["release"], date_added=date
 136         )

 173 def find_stacktraces_in_data(data, include_raw=False, with_exceptions=False):
 174     """Finds all stracktraces in a given data blob and returns it
 175     together with some meta information.
 176 
 177     If `include_raw` is True, then also raw stacktraces are included.  If
 178     `with_exceptions` is set to `True` then stacktraces of the exception
 179     are always included and the `is_exception` flag is set on that stack
 180     info object.
 181     """
 182     rv = []
 183 
 184     def _report_stack(stacktrace, container, is_exception=False):
 185         if not is_exception and (not stacktrace or not get_path(stacktrace, "frames", filter=True)):
 186             return
 187 
 188         platforms = set(
 189             frame.get("platform") or data.get("platform")
 190             for frame in get_path(stacktrace, "frames", filter=True, default=())
 191         )
 192         rv.append(
 193             StacktraceInfo(
 194                 stacktrace=stacktrace,
 195                 container=container,
 196                 platforms=platforms,
 197                 is_exception=is_exception,
 198             )
 199         )
 200 
 201     for exc in get_path(data, "exception", "values", filter=True, default=()):
 202         _report_stack(exc.get("stacktrace"), exc, is_exception=with_exceptions)
 203 
 204     _report_stack(data.get("stacktrace"), None)
 205 
 206     for thread in get_path(data, "threads", "values", filter=True, default=()):
 207         _report_stack(thread.get("stacktrace"), thread)
 208 
 209     if include_raw:
 210         for info in rv[:]:
 211             if info.container is not None:
 212                 _report_stack(info.container.get("raw_stacktrace"), info.container)
 213 
 214     return rv

 217 def _has_system_frames(frames):
 218     """
 219     Determines whether there are any frames in the stacktrace with in_app=false.
 220     """
 221 
 222     system_frames = 0
 223     for frame in frames:
 224         if not frame.get("in_app"):
 225             system_frames += 1
 226     return bool(system_frames) and len(frames) != system_frames

 229 def _normalize_in_app(stacktrace, platform=None, sdk_info=None):
 230     """
 231     Ensures consistent values of in_app across a stacktrace.
 232     """
 233     has_system_frames = _has_system_frames(stacktrace)
 234     for frame in stacktrace:
 235         # If all frames are in_app, flip all of them. This is expected by the UI
 236         if not has_system_frames:
 237             set_in_app(frame, False)
 238 
 239         # Default to false in all cases where processors or grouping enhancers
 240         # have not yet set in_app.
 241         elif frame.get("in_app") is None:
 242             set_in_app(frame, False)

 245 def normalize_stacktraces_for_grouping(data, grouping_config=None):
 246     """
 247     Applies grouping enhancement rules and ensure in_app is set on all frames.
 248     This also trims functions if necessary.
 249     """
 250 
 251     stacktraces = []
 252 
 253     for stacktrace_info in find_stacktraces_in_data(data, include_raw=True):
 254         frames = get_path(stacktrace_info.stacktrace, "frames", filter=True, default=())
 255         if frames:
 256             stacktraces.append(frames)
 257 
 258     if not stacktraces:
 259         return
 260 
 261     platform = data.get("platform")
 262 
 263     # Put the trimmed function names into the frames.  We only do this if
 264     # the trimming produces a different function than the function we have
 265     # otherwise stored in `function` to not make the payload larger
 266     # unnecessarily.
 267     for frames in stacktraces:
 268         for frame in frames:
 269             # Restore the original in_app value before the first grouping
 270             # enhancers have been run. This allows to re-apply grouping
 271             # enhancers on the original frame data.
 272             orig_in_app = get_path(frame, "data", "orig_in_app")
 273             if orig_in_app is not None:
 274                 frame["in_app"] = None if orig_in_app == -1 else bool(orig_in_app)
 275 
 276             if frame.get("raw_function") is not None:
 277                 continue
 278             raw_func = frame.get("function")
 279             if not raw_func:
 280                 continue
 281             function_name = trim_function_name(raw_func, frame.get("platform") or platform)
 282             if function_name != raw_func:
 283                 frame["raw_function"] = raw_func
 284                 frame["function"] = function_name
 285 
 286     # If a grouping config is available, run grouping enhancers
 287     if grouping_config is not None:
 288         for frames in stacktraces:
 289             grouping_config.enhancements.apply_modifications_to_frame(frames, platform)
 290 
 291     # normalize in-app
 292     for stacktrace in stacktraces:
 293         _normalize_in_app(stacktrace, platform=platform)

 343 def get_processable_frames(stacktrace_info, processors):
 344     """Returns thin wrappers around the frames in a stacktrace associated
 345     with the processor for it.
 346     """
 347     frames = get_path(stacktrace_info.stacktrace, "frames", filter=True, default=())
 348     frame_count = len(frames)
 349     rv = []
 350     for idx, frame in enumerate(frames):
 351         processor = next((p for p in processors if p.handles_frame(frame, stacktrace_info)), None)
 352         if processor is not None:
 353             rv.append(
 354                 ProcessableFrame(frame, frame_count - idx - 1, processor, stacktrace_info, rv)
 355             )
 356     return rv

 359 def process_single_stacktrace(processing_task, stacktrace_info, processable_frames):
 360     # TODO: associate errors with the frames and processing issues
 361     changed_raw = False
 362     changed_processed = False
 363     raw_frames = []
 364     processed_frames = []
 365     all_errors = []
 366 
 367     bare_frames = get_path(stacktrace_info.stacktrace, "frames", filter=True, default=())
 368     frame_count = len(bare_frames)
 369     processable_frames = {frame.idx: frame for frame in processable_frames}
 370 
 371     for i, bare_frame in enumerate(bare_frames):
 372         idx = frame_count - i - 1
 373         rv = None
 374 
 375         if idx in processable_frames:
 376             processable_frame = processable_frames[idx]
 377             assert processable_frame.frame is bare_frame
 378             try:
 379                 rv = processable_frame.processor.process_frame(processable_frame, processing_task)
 380             except Exception:
 381                 logger.exception("Failed to process frame")
 382 
 383         expand_processed, expand_raw, errors = rv or (None, None, None)
 384 
 385         if expand_processed is not None:
 386             processed_frames.extend(expand_processed)
 387             changed_processed = True
 388         elif expand_raw:  # is not empty
 389             processed_frames.extend(expand_raw)
 390             changed_processed = True
 391         else:
 392             processed_frames.append(bare_frame)
 393 
 394         if expand_raw is not None:
 395             raw_frames.extend(expand_raw)
 396             changed_raw = True
 397         else:
 398             raw_frames.append(bare_frame)
 399         all_errors.extend(errors or ())
 400 
 401     return (
 402         processed_frames if changed_processed else None,
 403         raw_frames if changed_raw else None,
 404         all_errors,
 405     )

 408 def get_crash_frame_from_event_data(data, frame_filter=None):
 409     """
 410     Return the highest (closest to the crash) in-app frame in the top stacktrace
 411     which doesn't fail the given filter test.
 412 
 413     If no such frame is available, return the highest non-in-app frame which
 414     otherwise meets the same criteria.
 415 
 416     Return None if any of the following are true:
 417         - there are no frames
 418         - all frames fail the given filter test
 419         - we're unable to find any frames nested in either event.exception or
 420           event.stacktrace, and there's anything other than exactly one thread
 421           in the data
 422     """
 423 
 424     frames = get_path(data, "exception", "values", -1, "stacktrace", "frames") or get_path(
 425         data, "stacktrace", "frames"
 426     )
 427     if not frames:
 428         threads = get_path(data, "threads", "values")
 429         if threads and len(threads) == 1:
 430             frames = get_path(threads, 0, "stacktrace", "frames")
 431 
 432     default = None
 433     for frame in reversed(frames or ()):
 434         if frame is None:
 435             continue
 436         if frame_filter is not None:
 437             if not frame_filter(frame):
 438                 continue
 439         if frame.get("in_app"):
 440             return frame
 441         if default is None:
 442             default = frame
 443 
 444     if default:
 445         return default

 455 def get_stacktrace_processing_task(infos, processors):
 456     """Returns a list of all tasks for the processors.  This can skip over
 457     processors that seem to not handle any frames.
 458     """
 459     by_processor = {}
 460     to_lookup = {}
 461 
 462     # by_stacktrace_info requires stable sorting as it is used in
 463     # StacktraceProcessingTask.iter_processable_stacktraces. This is important
 464     # to guarantee reproducible symbolicator requests.
 465     by_stacktrace_info = OrderedDict()
 466 
 467     for info in infos:
 468         processable_frames = get_processable_frames(info, processors)
 469         for processable_frame in processable_frames:
 470             processable_frame.processor.preprocess_frame(processable_frame)
 471             by_processor.setdefault(processable_frame.processor, []).append(processable_frame)
 472             by_stacktrace_info.setdefault(processable_frame.stacktrace_info, []).append(
 473                 processable_frame
 474             )
 475             if processable_frame.cache_key is not None:
 476                 to_lookup[processable_frame.cache_key] = processable_frame
 477 
 478     frame_cache = lookup_frame_cache(to_lookup)
 479     for cache_key, processable_frame in six.iteritems(to_lookup):
 480         processable_frame.cache_value = frame_cache.get(cache_key)
 481 
 482     return StacktraceProcessingTask(
 483         processable_stacktraces=by_stacktrace_info, processors=by_processor
 484     )

 487 def dedup_errors(errors):
 488     # This operation scales bad but we do not expect that many items to
 489     # end up in rv, so that should be okay enough to do.
 490     rv = []
 491     for error in errors:
 492         if error not in rv:
 493             rv.append(error)
 494     return rv

 497 def process_stacktraces(data, make_processors=None, set_raw_stacktrace=True):
 498     infos = find_stacktraces_in_data(data, with_exceptions=True)
 499     if make_processors is None:
 500         processors = get_processors_for_stacktraces(data, infos)
 501     else:
 502         processors = make_processors(data, infos)
 503 
 504     # Early out if we have no processors.  We don't want to record a timer
 505     # in that case.
 506     if not processors:
 507         return
 508 
 509     changed = False
 510 
 511     # Build a new processing task
 512     processing_task = get_stacktrace_processing_task(infos, processors)
 513     try:
 514 
 515         # Preprocess step
 516         for processor in processing_task.iter_processors():
 517             with sentry_sdk.start_span(
 518                 op="stacktraces.processing.process_stacktraces.preprocess_step"
 519             ) as span:
 520                 span.set_data("processor", processor.__class__.__name__)
 521                 if processor.preprocess_step(processing_task):
 522                     changed = True
 523                     span.set_data("data_changed", True)
 524 
 525         # Process all stacktraces
 526         for stacktrace_info, processable_frames in processing_task.iter_processable_stacktraces():
 527             # Let the stacktrace processors touch the exception
 528             if stacktrace_info.is_exception and stacktrace_info.container:
 529                 for processor in processing_task.iter_processors():
 530                     with sentry_sdk.start_span(
 531                         op="stacktraces.processing.process_stacktraces.process_exception"
 532                     ) as span:
 533                         span.set_data("processor", processor.__class__.__name__)
 534                         if processor.process_exception(stacktrace_info.container):
 535                             changed = True
 536                             span.set_data("data_changed", True)
 537 
 538             # If the stacktrace is empty we skip it for processing
 539             if not stacktrace_info.stacktrace:
 540                 continue
 541             with sentry_sdk.start_span(
 542                 op="stacktraces.processing.process_stacktraces.process_single_stacktrace"
 543             ) as span:
 544                 new_frames, new_raw_frames, errors = process_single_stacktrace(
 545                     processing_task, stacktrace_info, processable_frames
 546                 )
 547                 if new_frames is not None:
 548                     stacktrace_info.stacktrace["frames"] = new_frames
 549                     changed = True
 550                     span.set_data("data_changed", True)
 551             if (
 552                 set_raw_stacktrace
 553                 and new_raw_frames is not None
 554                 and stacktrace_info.container is not None
 555             ):
 556                 stacktrace_info.container["raw_stacktrace"] = dict(
 557                     stacktrace_info.stacktrace, frames=new_raw_frames
 558                 )
 559                 changed = True
 560             if errors:
 561                 data.setdefault("errors", []).extend(dedup_errors(errors))
 562                 data.setdefault("_metrics", {})["flag.processing.error"] = True
 563                 changed = True
 564 
 565     except Exception:
 566         logger.exception("stacktraces.processing.crash")
 567         data.setdefault("_metrics", {})["flag.processing.fatal"] = True
 568         data.setdefault("_metrics", {})["flag.processing.error"] = True
 569         changed = True
 570     finally:
 571         for processor in processors:
 572             processor.close()
 573         processing_task.close()
 574 
 575     if changed:
 576         return data
