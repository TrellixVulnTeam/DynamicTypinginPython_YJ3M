  24 class QuotaConfig(object):
  25     """
  26     Abstract configuration for a quota.
  27 
  28     Sentry applies multiple quotas to an event before accepting it, some of
  29     which can be configured by the user depending on plan. An event will be
  30     counted against all quotas that it matches with based on the ``category``.
  31 
  32     The `QuotaConfig` object is not persisted, but is the contract between
  33     Sentry and Relay. Most importantly, a `QuotaConfig` instance does not
  34     contain information about how many events can still be accepted, it only
  35     represents settings that should be applied. The actual counts are in the
  36     rate limiter (e.g. implemented via Redis caches).
  37 
  38     :param id:          The unique identifier for counting this quota. Required
  39                         except for quotas with ``limit=0``, since they are
  40                         statically enforced.
  41     :param categories:  A set of data categories that this quota applies to. If
  42                         missing or empty, this quota applies to all data.
  43     :param scope:       A scope for this quota. This quota is enforced
  44                         separately within each instance of this scope (e.g. for
  45                         each project key separately). Defaults to ORGANIZATION.
  46     :param scope_id:    Identifier of the scope to apply to. If set, then this
  47                         quota will only apply to the specified scope instance
  48                         (e.g. a project key). Requires ``scope`` to be set
  49                         explicitly.
  50     :param limit:       Maximum number of matching events allowed. Can be ``0``
  51                         to reject all events, ``None`` for an unlimited counted
  52                         quota, or a positive number for enforcement. Requires
  53                         ``window`` if the limit is not ``0``.
  54     :param window:      The time window in seconds to enforce this quota in.
  55                         Required in all cases except ``limit=0``, since those
  56                         quotas are not measured.
  57     :param reason_code: A machine readable reason returned when this quota is
  58                         exceeded. Required in all cases except ``limit=None``,
  59                         since unlimited quotas can never be exceeded.
  60     """
  61 
  62     __slots__ = ["id", "categories", "scope", "scope_id", "limit", "window", "reason_code"]
  63 
  64     def __init__(
  65         self,
  66         id=None,
  67         categories=None,
  68         scope=None,
  69         scope_id=None,
  70         limit=None,
  71         window=None,
  72         reason_code=None,
  73     ):
  74         if limit is not None:
  75             assert reason_code, "reason code required for fallible quotas"
  76 
  77         if limit == 0:
  78             assert id is None, "reject-all quotas cannot be tracked"
  79             assert window is None, "tracked quotas must specify a window"
  80         else:
  81             assert id, "measured quotas require an identifier"
  82             assert window and window > 0, "window cannot be zero"
  83 
  84         if scope_id is not None:
  85             assert scope, "scope must be declared explicitly when scope_id is given"
  86         elif scope is None:
  87             scope = QuotaScope.ORGANIZATION
  88 
  89         self.id = id
  90         self.scope = scope
  91         self.scope_id = six.text_type(scope_id) if scope_id is not None else None
  92         self.categories = set(categories or [])
  93         # NOTE: Use `quotas.base._limit_from_settings` to map from settings
  94         self.limit = limit
  95         self.window = window
  96         self.reason_code = reason_code
  97 
  98     @property
  99     def should_track(self):
 100         """
 101         Whether the quotas service should track this quota.
 102         """
 103 
 104         return self.id is not None and self.window is not None
 105 
 106     def to_json_legacy(self):
 107         data = {
 108             "prefix": six.text_type(self.id) if self.id is not None else None,
 109             "subscope": six.text_type(self.scope_id) if self.scope_id is not None else None,
 110             "limit": self.limit,
 111             "window": self.window,
 112             "reasonCode": self.reason_code,
 113         }
 114 
 115         if self.scope != QuotaScope.ORGANIZATION and self.scope_id is not None:
 116             data["subscope"] = self.scope_id
 117 
 118         return prune_empty_keys(data)
 119 
 120     def to_json(self):
 121         categories = None
 122         if self.categories:
 123             categories = [c.api_name() for c in self.categories]
 124 
 125         data = {
 126             "id": six.text_type(self.id) if self.id is not None else None,
 127             "scope": self.scope.api_name(),
 128             "scopeId": self.scope_id,
 129             "categories": categories,
 130             "limit": self.limit,
 131             "window": self.window,
 132             "reasonCode": self.reason_code,
 133         }
 134 
 135         return prune_empty_keys(data)

 138 class RateLimit(object):
 139     """
 140     Return value of ``quotas.is_rate_limited``.
 141     """
 142 
 143     __slots__ = ["is_limited", "retry_after", "reason", "reason_code"]
 144 
 145     def __init__(self, is_limited, retry_after=None, reason=None, reason_code=None):
 146         self.is_limited = is_limited
 147         # delta of seconds in the future to retry
 148         self.retry_after = retry_after
 149         # human readable description
 150         self.reason = reason
 151         # machine readable description
 152         self.reason_code = reason_code
 153 
 154     def to_dict(self):
 155         """
 156         Converts the object into a plain dictionary
 157         :return: a dict containing the non None elm of the RateLimit
 158 
 159         >>> x = RateLimit(is_limited = False, retry_after = 33)
 160         >>> x.to_dict() == {'is_limited': False, 'retry_after': 33}
 161         True
 162 
 163         """
 164         return {
 165             name: getattr(self, name, None)
 166             for name in self.__slots__
 167             if getattr(self, name, None) is not None
 168         }

 172     def __init__(self, **kwargs):
 173         super(NotRateLimited, self).__init__(False, **kwargs)
 174 
 175 
 176 class RateLimited(RateLimit):
 177     def __init__(self, **kwargs):
 178         super(RateLimited, self).__init__(True, **kwargs)

 181 def _limit_from_settings(x):
 182     """
 183     limit=0 (or any falsy value) in database means "no limit". Convert that to
 184     limit=None as limit=0 in code means "reject all".
 185     """
 186 
 187     return int(x or 0) or None

 190 class Quota(Service):
 191     """
 192     Quotas handle tracking a project's usage and respond whether or not a
 193     project has been configured to throttle incoming data if they go beyond the
 194     specified quota.
 195 
 196     Quotas can specify a window to be tracked in, such as per minute or per
 197     hour. Additionally, quotas allow to specify the data categories they apply
 198     to, for example error events or attachments. For more information on quota
 199     parameters, see ``QuotaConfig``.
 200 
 201     To retrieve a list of active quotas, use ``quotas.get_quotas``. Also, to
 202     check the current status of quota usage, call ``quotas.get_usage``.
 203     """
 204 
 205     __all__ = (
 206         "get_maximum_quota",
 207         "get_organization_quota",
 208         "get_project_quota",
 209         "is_rate_limited",
 210         "validate",
 211         "refund",
 212         "get_event_retention",
 213         "get_quotas",
 214     )
 215 
 216     def __init__(self, **options):
 217         pass
 218 
 219     def get_quotas(self, project, key=None, keys=None):
 220         """
 221         Returns a quotas for the given project and its organization.
 222 
 223         The return values are instances of ``QuotaConfig``. See its
 224         documentation for more information about the values.
 225 
 226         :param project: The project instance that is used to determine quotas.
 227         :param key:     A project project key to obtain quotas for. If omitted,
 228                         only project and organization quotas are used.
 229         :param keys:    Similar to ``key``, except for multiple keys.
 230         """
 231         return []
 232 
 233     def is_rate_limited(self, project, key=None):
 234         """
 235         Checks whether any of the quotas in effect for the given project and
 236         project key has been exceeded and records consumption of the quota.
 237 
 238         By invoking this method, the caller signals that data is being ingested
 239         and needs to be counted against the quota. This increment happens
 240         atomically if none of the quotas have been exceeded. Otherwise, a rate
 241         limit is returned and data is not counted against the quotas.
 242 
 243         When an event or any other data is dropped after ``is_rate_limited`` has
 244         been called, use ``quotas.refund``.
 245 
 246         If no key is specified, then only organization-wide and project-wide
 247         quotas are checked. If a key is specified, then key-quotas are also
 248         checked.
 249 
 250         The return value is a subclass of ``RateLimit``:
 251 
 252          - ``RateLimited``, if at least one quota has been exceeded. The event
 253            should not be ingested by the caller, and none of the quotas have
 254            been counted.
 255 
 256          - ``NotRateLimited``, if consumption is within all quotas. Data must be
 257            ingested by the caller, and the counters for all counters have been
 258            incremented.
 259 
 260         :param project: The project instance that is used to determine quotas.
 261         :param key:     A project key to obtain quotas for. If omitted, only
 262                         project and organization quotas are used.
 263         """
 264         return NotRateLimited()
 265 
 266     def refund(self, project, key=None, timestamp=None):
 267         """
 268         Signals event rejection after ``quotas.is_rate_limited`` has been called
 269         successfully, and refunds the previously consumed quota.
 270 
 271         :param project:   The project that the dropped data belonged to.
 272         :param key:       The project key that was used to ingest the data. If
 273                           omitted, then only project and organization quotas are
 274                           refunded.
 275         :param timestamp: The timestamp at which data was ingested. This is used
 276                           to determine the correct quota window to refund the
 277                           previously consumed data to.
 278         """
 279         pass

 296     def _translate_quota(self, quota, parent_quota):
 297         if six.text_type(quota).endswith("%"):
 298             pct = int(quota[:-1])
 299             quota = int(parent_quota or 0) * pct / 100
 300 
 301         return _limit_from_settings(quota or parent_quota)
 302 
 303     def get_key_quota(self, key):
 304         from sentry import features
 305 
 306         # XXX(epurkhiser): Avoid excessive feature manager checks (which can be
 307         # expensive depending on feature handlers) for project rate limits.
 308         # This happens on /store.
 309         cache_key = u"project:{}:features:rate-limits".format(key.project.id)
 310 
 311         has_rate_limits = cache.get(cache_key)
 312         if has_rate_limits is None:
 313             has_rate_limits = features.has("projects:rate-limits", key.project)
 314             cache.set(cache_key, has_rate_limits, 600)
 315 
 316         if not has_rate_limits:
 317             return (None, None)
 318 
 319         limit, window = key.rate_limit
 320         return _limit_from_settings(limit), window
 321 
 322     def get_project_quota(self, project):
 323         from sentry.models import Organization, OrganizationOption
 324 
 325         org = getattr(project, "_organization_cache", None)
 326         if not org:
 327             org = Organization.objects.get_from_cache(id=project.organization_id)
 328             project._organization_cache = org
 329 
 330         max_quota_share = int(
 331             OrganizationOption.objects.get_value(org, "sentry:project-rate-limit", 100)
 332         )
 333 
 334         org_quota, window = self.get_organization_quota(org)
 335 
 336         if max_quota_share != 100 and org_quota:
 337             quota = self._translate_quota(u"{}%".format(max_quota_share), org_quota)
 338         else:
 339             quota = None
 340 
 341         return (quota, window)
 342 
 343     def get_organization_quota(self, organization):
 344         from sentry.models import OrganizationOption
 345 
 346         account_limit = _limit_from_settings(
 347             OrganizationOption.objects.get_value(
 348                 organization=organization, key="sentry:account-rate-limit", default=0
 349             )
 350         )
 351 
 352         system_limit = _limit_from_settings(options.get("system.rate-limit"))
 353 
 354         # If there is only a single org, this one org should
 355         # be allowed to consume the entire quota.
 356         if settings.SENTRY_SINGLE_ORGANIZATION or account_limit:
 357             if system_limit and (not account_limit or system_limit < account_limit / 60):
 358                 return (system_limit, 60)
 359             # an account limit is enforced, which is set as a fixed value and cannot
 360             # utilize percentage based limits
 361             return (account_limit, 3600)
 362 
 363         default_limit = self._translate_quota(
 364             settings.SENTRY_DEFAULT_MAX_EVENTS_PER_MINUTE, system_limit
 365         )
 366         return (default_limit, 60)
 367 
 368     def get_maximum_quota(self, organization):
 369         """
 370         Return the maximum capable rate for an organization.
 371         """
 372         return (_limit_from_settings(options.get("system.rate-limit")), 60)
