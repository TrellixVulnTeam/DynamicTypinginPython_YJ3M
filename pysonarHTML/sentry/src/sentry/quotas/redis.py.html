  22 class RedisQuota(Quota):
  23     #: The ``grace`` period allows accommodating for clock drift in TTL
  24     #: calculation since the clock on the Redis instance used to store quota
  25     #: metrics may not be in sync with the computer running this code.
  26     grace = 60

  28     def __init__(self, **options):
  29         self.is_redis_cluster, self.cluster, options = get_dynamic_cluster_from_options(
  30             "SENTRY_QUOTA_OPTIONS", options
  31         )
  32 
  33         # Based on the `is_redis_cluster` flag, self.cluster is set two one of
  34         # the following two objects:
  35         #  - false: `cluster` is a `RBCluster`. Call `get_local_client_for_key`
  36         #    on the cluster to resolve a client to run a script or query a key.
  37         #  - true: `cluster` is a `RedisCluster`. It automatically dispatches to
  38         #    the correct node and can be used as a client directly.
  39 
  40         super(RedisQuota, self).__init__(**options)
  41         self.namespace = "quota"

  43     def validate(self):
  44         validate_dynamic_cluster(self.is_redis_cluster, self.cluster)

  46     def __get_redis_client(self, routing_key):
  47         if self.is_redis_cluster:
  48             return self.cluster
  49         else:
  50             return self.cluster.get_local_client_for_key(routing_key)

  52     def __get_redis_key(self, quota, timestamp, shift, organization_id):
  53         if self.is_redis_cluster:
  54             scope_id = quota.scope_id or "" if quota.scope != QuotaScope.ORGANIZATION else ""
  55             # new style redis cluster format which always has the organization id in
  56             local_key = "%s{%s}%s" % (quota.id, organization_id, scope_id)
  57         else:
  58             # legacy key format
  59             local_key = "%s:%s" % (quota.id, quota.scope_id or organization_id)
  60 
  61         interval = quota.window
  62         return u"{}:{}:{}".format(self.namespace, local_key, int((timestamp - shift) // interval))

  95         if oquota[0] is not None:

  97                 QuotaConfig(
  98                     id="o",
  99                     scope=QuotaScope.ORGANIZATION,

 103                     window=oquota[1],
 104                     reason_code="org_quota",
 105                 )

 130     def get_usage(self, organization_id, quotas, timestamp=None):
 131         if timestamp is None:
 132             timestamp = time()
 133 
 134         def get_usage_for_quota(client, quota):
 135             if not quota.should_track:
 136                 return (None, None)
 137 
 138             key = self.__get_redis_key(
 139                 quota, timestamp, organization_id % quota.window, organization_id
 140             )
 141             refund_key = self.get_refunded_quota_key(key)
 142 
 143             return (client.get(key), client.get(refund_key))
 144 
 145         def get_value_for_result(result, refund_result):
 146             if result is None:
 147                 return None
 148 
 149             return int(result.value or 0) - int(refund_result.value or 0)
 150 
 151         if self.is_redis_cluster:
 152             results = map(functools.partial(get_usage_for_quota, self.cluster), quotas)
 153         else:
 154             with self.cluster.fanout() as client:
 155                 results = map(
 156                     functools.partial(
 157                         get_usage_for_quota, client.target_key(six.text_type(organization_id))
 158                     ),
 159                     quotas,
 160                 )
 161 
 162         return [get_value_for_result(*r) for r in results]
 163 
 164     def get_refunded_quota_key(self, key):
 165         return u"r:{}".format(key)

 167     def refund(self, project, key=None, timestamp=None):
 168         if timestamp is None:
 169             timestamp = time()
 170 
 171         quotas = [quota for quota in self.get_quotas(project, key=key) if quota.should_track]
 172 
 173         if not quotas:
 174             return
 175 
 176         client = self.__get_redis_client(six.text_type(project.organization_id))
 177         pipe = client.pipeline()
 178 
 179         for quota in quotas:
 180             shift = project.organization_id % quota.window
 181             # kind of arbitrary, but seems like we don't want this to expire til we're
 182             # sure the window is over?
 183             expiry = self.get_next_period_start(quota.window, shift, timestamp) + self.grace
 184             return_key = self.get_refunded_quota_key(
 185                 self.__get_redis_key(quota, timestamp, shift, project.organization_id)
 186             )
 187             pipe.incr(return_key, 1)
 188             pipe.expireat(return_key, int(expiry))
 189 
 190         pipe.execute()
 191 
 192     def get_next_period_start(self, interval, shift, timestamp):
 193         """Return the timestamp when the next rate limit period begins for an interval."""
 194         return (((timestamp - shift) // interval) + 1) * interval + shift

 196     def is_rate_limited(self, project, key=None, timestamp=None):
 197         # XXX: This is effectively deprecated and scheduled for removal. Event
 198         # ingestion quotas are now enforced in Relay. This function will be
 199         # deleted once the Python store endpoints are removed.
 200 
 201         if timestamp is None:
 202             timestamp = time()
 203 
 204         # Relay supports separate rate limiting per data category and and can
 205         # handle scopes explicitly. This function implements a simplified logic
 206         # that treats all events the same and ignores transaction rate limits.
 207         # Thus, we filter for (1) no categories, which implies this quota
 208         # affects all data, and (2) quotas that specify `error` events.
 209         quotas = [
 210             q
 211             for q in self.get_quotas(project, key=key)
 212             if not q.categories or DataCategory.ERROR in q.categories
 213         ]
 214 
 215         # If there are no quotas to actually check, skip the trip to the database.
 216         if not quotas:
 217             return NotRateLimited()
 218 
 219         keys = []
 220         args = []
 221         for quota in quotas:
 222             if quota.limit == 0:
 223                 # A zero-sized quota is the absolute worst-case. Do not call
 224                 # into Redis at all, and do not increment any keys, as one
 225                 # quota has reached capacity (this is how regular quotas behave
 226                 # as well).
 227                 assert quota.window is None
 228                 assert not quota.should_track
 229                 return RateLimited(retry_after=None, reason_code=quota.reason_code)
 230 
 231             assert quota.should_track
 232 
 233             shift = project.organization_id % quota.window
 234             key = self.__get_redis_key(quota, timestamp, shift, project.organization_id)
 235             return_key = self.get_refunded_quota_key(key)
 236             keys.extend((key, return_key))
 237             expiry = self.get_next_period_start(quota.window, shift, timestamp) + self.grace
 238 
 239             # limit=None is represented as limit=-1 in lua
 240             lua_quota = quota.limit if quota.limit is not None else -1
 241             args.extend((lua_quota, int(expiry)))
 242 
 243         if not keys or not args:
 244             return NotRateLimited()
 245 
 246         client = self.__get_redis_client(six.text_type(project.organization_id))
 247         rejections = is_rate_limited(client, keys, args)
 248 
 249         if not any(rejections):
 250             return NotRateLimited()
 251 
 252         worst_case = (0, None)
 253         for quota, rejected in zip(quotas, rejections):
 254             if not rejected:
 255                 continue
 256 
 257             shift = project.organization_id % quota.window
 258             delay = self.get_next_period_start(quota.window, shift, timestamp) - timestamp
 259             if delay > worst_case[0]:
 260                 worst_case = (delay, quota.reason_code)
 261 
 262         return RateLimited(retry_after=worst_case[0], reason_code=worst_case[1])
