  67     def on_delete_repository(self, repo):
  68         """Clean up the attached webhook"""
  69         installation = self.get_installation(repo.integration_id, repo.organization_id)
  70         client = installation.get_client()
  71         try:
  72             client.delete_project_webhook(repo.config["project_id"], repo.config["webhook_id"])
  73         except ApiError as e:
  74             if e.code == 404:
  75                 return
  76             installation.raise_error(e)

  78     def compare_commits(self, repo, start_sha, end_sha):
  79         """Fetch the commit list and diffed files between two shas"""
  80         installation = self.get_installation(repo.integration_id, repo.organization_id)
  81         client = installation.get_client()
  82         try:
  83             if start_sha is None:
  84                 res = client.get_last_commits(repo.config["project_id"], end_sha)
  85                 return self._format_commits(client, repo, res)
  86             else:
  87                 res = client.compare_commits(repo.config["project_id"], start_sha, end_sha)
  88                 return self._format_commits(client, repo, res["commits"])
  89         except Exception as e:
  90             installation.raise_error(e)

  92     def _format_commits(self, client, repo, commit_list):
  93         """Convert GitLab commits into our internal format
  94         """
  95         return [
  96             {
  97                 "id": c["id"],
  98                 "repository": repo.name,
  99                 "author_email": c["author_email"],
 100                 "author_name": c["author_name"],
 101                 "message": c["title"],
 102                 "timestamp": self.format_date(c["created_at"]),
 103                 "patch_set": self._get_patchset(client, repo, c["id"]),
 104             }
 105             for c in commit_list
 106         ]
 107 
 108     def _get_patchset(self, client, repo, sha):
 109         """GitLab commit lists don't come with diffs so we have
 110         to make additional round trips.
 111         """
 112         diffs = client.get_diff(repo.config["project_id"], sha)
 113         return self._transform_patchset(diffs)
 114 
 115     def _transform_patchset(self, patch_set):
 116         file_changes = []
 117         for changed_file in patch_set:
 118             if changed_file["new_file"]:
 119                 file_changes.append({"path": changed_file["new_path"], "type": "A"})
 120             elif changed_file["deleted_file"]:
 121                 file_changes.append({"path": changed_file["old_path"], "type": "D"})
 122             elif changed_file["renamed_file"]:
 123                 file_changes.append({"path": changed_file["old_path"], "type": "D"})
 124                 file_changes.append({"path": changed_file["new_path"], "type": "A"})
 125             else:
 126                 file_changes.append({"path": changed_file["new_path"], "type": "M"})
 127 
 128         return file_changes
