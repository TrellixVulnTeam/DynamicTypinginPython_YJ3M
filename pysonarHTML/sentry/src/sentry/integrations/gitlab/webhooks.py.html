  29     def get_repo(self, integration, organization, event):
  30         """
  31         Given a webhook payload, get the associated Repository record.
  32 
  33         Assumes a 'project' key in event payload.
  34         """
  35         try:
  36             project_id = event["project"]["id"]
  37         except KeyError:
  38             logger.info(
  39                 "gitlab.webhook.missing-projectid", extra={"integration_id": integration.id}
  40             )
  41             raise Http404()
  42 
  43         external_id = u"{}:{}".format(integration.metadata["instance"], project_id)
  44         try:
  45             repo = Repository.objects.get(
  46                 organization_id=organization.id, provider=PROVIDER_NAME, external_id=external_id
  47             )
  48         except Repository.DoesNotExist:
  49             return None
  50         return repo

  52     def update_repo_data(self, repo, event):
  53         """
  54         # TODO(kmclb): the name w namespace bit is currently causing false positives
  55         # for people with three or more levels in their repo names. Need to research
  56         # exactly how gitlab handles these cases (what's the name, what's the namespace,
  57         # does the url reliably have the info we need?) and then fix this for those
  58         # cases.
  59 
  60         Given a webhook payload, update stored repo data if needed.
  61 
  62         Assumes a 'project' key in event payload, with certain subkeys. Rework
  63         this if that stops being a safe assumption.
  64         """
  65 
  66         project = event["project"]
  67 
  68         name_from_event = u"{} / {}".format(project["namespace"], project["name"])
  69         url_from_event = project["web_url"]
  70         path_from_event = project["path_with_namespace"]
  71 
  72         if (
  73             repo.name != name_from_event
  74             or repo.url != url_from_event
  75             or repo.config.get("path") != path_from_event
  76         ):
  77             repo.update(
  78                 name=name_from_event,
  79                 url=url_from_event,
  80                 config=dict(repo.config, path=path_from_event),
  81             )

  84 class MergeEventWebhook(Webhook):
  85     """
  86     Handle Merge Request Hook
  87 
  88     See https://docs.gitlab.com/ee/user/project/integrations/webhooks.html#merge-request-events
  89     """
  90 
  91     def __call__(self, integration, organization, event):
  92         repo = self.get_repo(integration, organization, event)
  93         if repo is None:
  94             return
  95 
  96         # TODO(kmclb): turn this back on once tri-level repo problem has been solved
  97         # while we're here, make sure repo data is up to date
  98         # self.update_repo_data(repo, event)
  99 
 100         try:
 101             number = event["object_attributes"]["iid"]
 102             title = event["object_attributes"]["title"]
 103             body = event["object_attributes"]["description"]
 104             created_at = event["object_attributes"]["created_at"]
 105             merge_commit_sha = event["object_attributes"]["merge_commit_sha"]
 106 
 107             last_commit = event["object_attributes"]["last_commit"]
 108             author_email = None
 109             author_name = None
 110             if last_commit:
 111                 author_email = last_commit["author"]["email"]
 112                 author_name = last_commit["author"]["name"]
 113         except KeyError as e:
 114             logger.info(
 115                 "gitlab.webhook.invalid-merge-data",
 116                 extra={"integration_id": integration.id, "error": six.string_type(e)},
 117             )
 118 
 119         if not author_email:
 120             raise Http404()
 121 
 122         author = CommitAuthor.objects.get_or_create(
 123             organization_id=organization.id, email=author_email, defaults={"name": author_name}
 124         )[0]
 125 
 126         try:
 127             PullRequest.create_or_save(
 128                 organization_id=organization.id,
 129                 repository_id=repo.id,
 130                 key=number,
 131                 values={
 132                     "title": title,
 133                     "author": author,
 134                     "message": body,
 135                     "merge_commit_sha": merge_commit_sha,
 136                     "date_added": dateutil.parser.parse(created_at).astimezone(timezone.utc),
 137                 },
 138             )
 139         except IntegrityError:
 140             pass

 143 class PushEventWebhook(Webhook):
 144     """
 145     Handle push hook
 146 
 147     See https://docs.gitlab.com/ee/user/project/integrations/webhooks.html#push-events
 148     """
 149 
 150     def __call__(self, integration, organization, event):
 151         repo = self.get_repo(integration, organization, event)
 152         if repo is None:
 153             return
 154 
 155         # TODO(kmclb): turn this back on once tri-level repo problem has been solved
 156         # while we're here, make sure repo data is up to date
 157         # self.update_repo_data(repo, event)
 158 
 159         authors = {}
 160 
 161         # TODO gitlab only sends a max of 20 commits. If a push contains
 162         # more commits they provide a total count and require additional API
 163         # requests to fetch the commit details
 164         for commit in event.get("commits", []):
 165             if IntegrationRepositoryProvider.should_ignore_commit(commit["message"]):
 166                 continue
 167 
 168             author_email = commit["author"]["email"]
 169 
 170             # TODO(dcramer): we need to deal with bad values here, but since
 171             # its optional, lets just throw it out for now
 172             if author_email is None or len(author_email) > 75:
 173                 author = None
 174             elif author_email not in authors:
 175                 authors[author_email] = author = CommitAuthor.objects.get_or_create(
 176                     organization_id=organization.id,
 177                     email=author_email,
 178                     defaults={"name": commit["author"]["name"]},
 179                 )[0]
 180             else:
 181                 author = authors[author_email]
 182             try:
 183                 with transaction.atomic():
 184                     Commit.objects.create(
 185                         repository_id=repo.id,
 186                         organization_id=organization.id,
 187                         key=commit["id"],
 188                         message=commit["message"],
 189                         author=author,
 190                         date_added=dateutil.parser.parse(commit["timestamp"]).astimezone(
 191                             timezone.utc
 192                         ),
 193                     )
 194             except IntegrityError:
 195                 pass

 198 class GitlabWebhookEndpoint(View):
 199     provider = "gitlab"
 200 
 201     _handlers = {"Push Hook": PushEventWebhook, "Merge Request Hook": MergeEventWebhook}
 202 
 203     @method_decorator(csrf_exempt)
 204     def dispatch(self, request, *args, **kwargs):
 205         if request.method != "POST":
 206             return HttpResponse(status=405)
 207 
 208         return super(GitlabWebhookEndpoint, self).dispatch(request, *args, **kwargs)
 209 
 210     def post(self, request):
 211         token = "<unknown>"
 212         try:
 213             # Munge the token to extract the integration external_id.
 214             # gitlab hook payloads don't give us enough unique context
 215             # to find data on our side so we embed one in the token.
 216             token = request.META["HTTP_X_GITLAB_TOKEN"]
 217             instance, group_path, secret = token.split(":")
 218             external_id = u"{}:{}".format(instance, group_path)
 219         except Exception:
 220             logger.info("gitlab.webhook.invalid-token", extra={"token": token})
 221             return HttpResponse(status=400)
 222 
 223         try:
 224             integration = (
 225                 Integration.objects.filter(provider=self.provider, external_id=external_id)
 226                 .prefetch_related("organizations")
 227                 .get()
 228             )
 229         except Integration.DoesNotExist:
 230             logger.info(
 231                 "gitlab.webhook.invalid-organization",
 232                 extra={"external_id": request.META["HTTP_X_GITLAB_TOKEN"]},
 233             )
 234             return HttpResponse(status=400)
 235 
 236         if not constant_time_compare(secret, integration.metadata["webhook_secret"]):
 237             logger.info(
 238                 "gitlab.webhook.invalid-token-secret", extra={"integration_id": integration.id}
 239             )
 240             return HttpResponse(status=400)
 241 
 242         try:
 243             event = json.loads(request.body.decode("utf-8"))
 244         except JSONDecodeError:
 245             logger.info(
 246                 "gitlab.webhook.invalid-json", extra={"external_id": integration.external_id}
 247             )
 248             return HttpResponse(status=400)
 249 
 250         try:
 251             handler = self._handlers[request.META["HTTP_X_GITLAB_EVENT"]]
 252         except KeyError:
 253             logger.info(
 254                 "gitlab.webhook.missing-event", extra={"event": request.META["HTTP_X_GITLAB_EVENT"]}
 255             )
 256             return HttpResponse(status=400)
 257 
 258         for organization in integration.organizations.all():
 259             handler()(integration, organization, event)
 260         return HttpResponse(status=204)
