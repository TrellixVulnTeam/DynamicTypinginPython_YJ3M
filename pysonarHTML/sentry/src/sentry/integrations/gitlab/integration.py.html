  73 class GitlabIntegration(IntegrationInstallation, GitlabIssueBasic, RepositoryMixin):
  74     repo_search = True
  75 
  76     def __init__(self, *args, **kwargs):
  77         super(GitlabIntegration, self).__init__(*args, **kwargs)
  78         self.default_identity = None
  79 
  80     def get_group_id(self):
  81         return self.model.metadata["group_id"]
  82 
  83     def get_client(self):
  84         if self.default_identity is None:
  85             self.default_identity = self.get_default_identity()
  86 
  87         return GitLabApiClient(self)
  88 
  89     def get_repositories(self, query=None):
  90         # Note: gitlab projects are the same things as repos everywhere else
  91         group = self.get_group_id()
  92         resp = self.get_client().search_group_projects(group, query)
  93         return [{"identifier": repo["id"], "name": repo["name_with_namespace"]} for repo in resp]
  94 
  95     def search_projects(self, query):
  96         client = self.get_client()
  97         group_id = self.get_group_id()
  98         return client.search_group_projects(group_id, query)
  99 
 100     def search_issues(self, project_id, query, iids):
 101         client = self.get_client()
 102         return client.search_project_issues(project_id, query, iids)

 104     def error_message_from_json(self, data):
 105         """
 106         Extract error messages from gitlab API errors.
 107         Generic errors come in the `error` key while validation errors
 108         are generally in `message`.
 109 
 110         See https://docs.gitlab.com/ee/api/#data-validation-and-error-reporting
 111         """
 112         if "message" in data:
 113             return data["message"]
 114         if "error" in data:
 115             return data["error"]

 176 class InstallationConfigView(PipelineView):
 177     def dispatch(self, request, pipeline):
 178         if "goback" in request.GET:
 179             pipeline.state.step_index = 0
 180             return pipeline.current_step()
 181 
 182         if request.method == "POST":
 183             form = InstallationForm(request.POST)
 184             if form.is_valid():
 185                 form_data = form.cleaned_data
 186 
 187                 pipeline.bind_state("installation_data", form_data)
 188 
 189                 pipeline.bind_state(
 190                     "oauth_config_information",
 191                     {
 192                         "access_token_url": u"{}/oauth/token".format(form_data.get("url")),
 193                         "authorize_url": u"{}/oauth/authorize".format(form_data.get("url")),
 194                         "client_id": form_data.get("client_id"),
 195                         "client_secret": form_data.get("client_secret"),
 196                         "verify_ssl": form_data.get("verify_ssl"),
 197                     },
 198                 )
 199                 pipeline.get_logger().info(
 200                     "gitlab.setup.installation-config-view.success",
 201                     extra={
 202                         "base_url": form_data.get("url"),
 203                         "client_id": form_data.get("client_id"),
 204                         "verify_ssl": form_data.get("verify_ssl"),
 205                     },
 206                 )
 207                 return pipeline.next_step()
 208         else:
 209             form = InstallationForm()
 210 
 211         return render_to_response(
 212             template="sentry/integrations/gitlab-config.html",
 213             context={"form": form},
 214             request=request,
 215         )

 237 class GitlabIntegrationProvider(IntegrationProvider):
 238     key = "gitlab"
 239     name = "GitLab"
 240     metadata = metadata
 241     integration_cls = GitlabIntegration
 242 
 243     needs_default_identity = True
 244 
 245     features = frozenset([IntegrationFeatures.ISSUE_BASIC, IntegrationFeatures.COMMITS])
 246 
 247     setup_dialog_config = {"width": 1030, "height": 1000}

 249     def _make_identity_pipeline_view(self):
 250         """
 251         Make the nested identity provider view. It is important that this view is
 252         not constructed until we reach this step and the
 253         ``oauth_config_information`` is available in the pipeline state. This
 254         method should be late bound into the pipeline vies.
 255         """
 256         identity_pipeline_config = dict(
 257             oauth_scopes=sorted(GitlabIdentityProvider.oauth_scopes),
 258             redirect_url=absolute_uri("/extensions/gitlab/setup/"),
 259             **self.pipeline.fetch_state("oauth_config_information")
 260         )
 261 
 262         return NestedPipelineView(
 263             bind_key="identity",
 264             provider_key="gitlab",
 265             pipeline_cls=IdentityProviderPipeline,
 266             config=identity_pipeline_config,
 267         )

 269     def get_group_info(self, access_token, installation_data):
 270         client = GitLabSetupClient(
 271             installation_data["url"], access_token, installation_data["verify_ssl"]
 272         )
 273         try:
 274             resp = client.get_group(installation_data["group"])
 275             return resp.json
 276         except ApiError as e:
 277             self.get_logger().info(
 278                 "gitlab.installation.get-group-info-failure",
 279                 extra={
 280                     "base_url": installation_data["url"],
 281                     "verify_ssl": installation_data["verify_ssl"],
 282                     "group": installation_data["group"],
 283                     "include_subgroups": installation_data["include_subgroups"],
 284                     "error_message": six.text_type(e),
 285                     "error_status": e.code,
 286                 },
 287             )
 288             raise IntegrationError("The requested GitLab group could not be found.")
 289 
 290     def get_pipeline_views(self):
 291         return [
 292             InstallationGuideView(),
 293             InstallationConfigView(),
 294             lambda: self._make_identity_pipeline_view(),
 295         ]

 297     def build_integration(self, state):
 298         data = state["identity"]["data"]
 299         oauth_data = get_oauth_data(data)
 300         user = get_user_info(data["access_token"], state["installation_data"])
 301         group = self.get_group_info(data["access_token"], state["installation_data"])
 302         include_subgroups = state["installation_data"]["include_subgroups"]
 303         scopes = sorted(GitlabIdentityProvider.oauth_scopes)
 304         base_url = state["installation_data"]["url"]
 305 
 306         hostname = urlparse(base_url).netloc
 307         verify_ssl = state["installation_data"]["verify_ssl"]
 308 
 309         # Generate a hash to prevent stray hooks from being accepted
 310         # use a consistent hash so that reinstalls/shared integrations don't
 311         # rotate secrets.
 312         secret = sha1_text("".join([hostname, state["installation_data"]["client_id"]]))
 313 
 314         integration = {
 315             "name": group["full_name"],
 316             # Splice the gitlab host and project together to
 317             # act as unique link between a gitlab instance, group + sentry.
 318             # This value is embedded then in the webook token that we
 319             # give to gitlab to allow us to find the integration a hook came
 320             # from.
 321             "external_id": u"{}:{}".format(hostname, group["id"]),
 322             "metadata": {
 323                 "icon": group["avatar_url"],
 324                 "instance": hostname,
 325                 "domain_name": u"{}/{}".format(hostname, group["full_path"]),
 326                 "scopes": scopes,
 327                 "verify_ssl": verify_ssl,
 328                 "base_url": base_url,
 329                 "webhook_secret": secret.hexdigest(),
 330                 "group_id": group["id"],
 331                 "include_subgroups": include_subgroups,
 332             },
 333             "user_identity": {
 334                 "type": "gitlab",
 335                 "external_id": u"{}:{}".format(hostname, user["id"]),
 336                 "scopes": scopes,
 337                 "data": oauth_data,
 338             },
 339         }
 340         return integration
