  10 ISSUE_EXTERNAL_KEY_FORMAT = re.compile(r".+:(.+)#(.+)")
  11

  13 class GitlabIssueBasic(IssueBasicMixin):
  14     def make_external_key(self, data):
  15         return u"{}:{}".format(self.model.metadata["domain_name"], data["key"])
  16 
  17     def get_issue_url(self, key):
  18         match = ISSUE_EXTERNAL_KEY_FORMAT.match(key)
  19         project, issue_id = match.group(1), match.group(2)
  20         return u"{}/{}/issues/{}".format(self.model.metadata["base_url"], project, issue_id)
  21 
  22     def get_persisted_default_config_fields(self):
  23         return ["project"]
  24 
  25     def get_projects_and_default(self, group, **kwargs):
  26         params = kwargs.get("params", {})
  27         defaults = self.get_project_defaults(group.project_id)
  28         kwargs["repo"] = params.get("project", defaults.get("project"))
  29 
  30         # In GitLab Repositories are called Projects
  31         default_project, project_choices = self.get_repository_choices(group, **kwargs)
  32         return default_project, project_choices
  33 
  34     def create_default_repo_choice(self, default_repo):
  35         client = self.get_client()
  36         try:
  37             # default_repo should be the project_id
  38             project = client.get_project(default_repo)
  39         except (ApiError, ApiUnauthorized):
  40             return ("", "")
  41         return (project["id"], project["name_with_namespace"])

  43     def get_create_issue_config(self, group, **kwargs):
  44         default_project, project_choices = self.get_projects_and_default(group, **kwargs)
  45         kwargs["link_referrer"] = "gitlab_integration"
  46         fields = super(GitlabIssueBasic, self).get_create_issue_config(group, **kwargs)
  47 
  48         org = group.organization
  49         autocomplete_url = reverse(
  50             "sentry-extensions-gitlab-search", args=[org.slug, self.model.id]
  51         )
  52 
  53         return [
  54             {
  55                 "name": "project",
  56                 "label": "GitLab Project",
  57                 "type": "select",
  58                 "url": autocomplete_url,
  59                 "choices": project_choices,
  60                 "defaultValue": default_project,
  61                 "required": True,
  62             }
  63         ] + fields

  65     def create_issue(self, data, **kwargs):
  66         client = self.get_client()
  67 
  68         project_id = data.get("project")
  69 
  70         if not project_id:
  71             raise IntegrationError("project kwarg must be provided")
  72 
  73         try:
  74             issue = client.create_issue(
  75                 project=project_id,
  76                 data={"title": data["title"], "description": data["description"]},
  77             )
  78             project = client.get_project(project_id)
  79         except ApiError as e:
  80             raise IntegrationError(self.message_from_error(e))
  81 
  82         project_and_issue_iid = "%s#%s" % (project["path_with_namespace"], issue["iid"])
  83         return {
  84             "key": project_and_issue_iid,
  85             "title": issue["title"],
  86             "description": issue["description"],
  87             "url": issue["web_url"],
  88             "project": project_id,
  89             "metadata": {"display_name": project_and_issue_iid},
  90         }

  92     def after_link_issue(self, external_issue, **kwargs):
  93         data = kwargs["data"]
  94         project_id, issue_id = data.get("externalIssue", "").split("#")
  95         if not (project_id and issue_id):
  96             raise IntegrationError("Project and Issue id must be provided")
  97 
  98         client = self.get_client()
  99         comment = data.get("comment")
 100         if not comment:
 101             return
 102 
 103         try:
 104             client.create_issue_comment(
 105                 project_id=project_id, issue_id=issue_id, data={"body": comment}
 106             )
 107         except ApiError as e:
 108             raise IntegrationError(self.message_from_error(e))

 152     def get_issue(self, issue_id, **kwargs):
 153         project_id, issue_num = issue_id.split("#")
 154         client = self.get_client()
 155 
 156         if not project_id:
 157             raise IntegrationError("project must be provided")
 158 
 159         if not issue_num:
 160             raise IntegrationError("issue must be provided")
 161 
 162         try:
 163             issue = client.get_issue(project_id, issue_num)
 164             project = client.get_project(project_id)
 165         except ApiError as e:
 166             raise IntegrationError(self.message_from_error(e))
 167 
 168         project_and_issue_iid = "%s#%s" % (project["path_with_namespace"], issue["iid"])
 169         return {
 170             "key": project_and_issue_iid,
 171             "title": issue["title"],
 172             "description": issue["description"],
 173             "url": issue["web_url"],
 174             "project": project_id,
 175             "metadata": {"display_name": project_and_issue_iid},
 176         }
