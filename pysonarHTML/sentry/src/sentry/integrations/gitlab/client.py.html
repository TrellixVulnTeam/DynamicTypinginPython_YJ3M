  37 class GitLabSetupClient(ApiClient):
  38     """
  39     API Client that doesn't require an installation.
  40     This client is used during integration setup to fetch data
  41     needed to build installation metadata
  42     """
  43 
  44     integration_name = "gitlab_setup"
  45 
  46     def __init__(self, base_url, access_token, verify_ssl):
  47         self.base_url = base_url
  48         self.token = access_token
  49         self.verify_ssl = verify_ssl
  50 
  51     def get_group(self, group):
  52         """Get a group based on `path` which is a slug.
  53 
  54         We need to URL quote because subgroups use `/` in their
  55         `id` and GitLab requires slugs to be URL encoded.
  56         """
  57         group = quote(group, safe="")
  58         path = GitLabApiClientPath.group.format(group=group)
  59         return self.get(path)
  60 
  61     def request(self, method, path, data=None, params=None):
  62         headers = {"Authorization": u"Bearer {}".format(self.token)}
  63         return self._request(
  64             method,
  65             GitLabApiClientPath.build_api_url(self.base_url, path),
  66             headers=headers,
  67             data=data,
  68             params=params,
  69         )

  72 class GitLabApiClient(ApiClient):
  73     integration_name = "gitlab"
  74 
  75     def __init__(self, installation):
  76         self.installation = installation
  77         verify_ssl = self.metadata["verify_ssl"]
  78         self.is_refreshing_token = False
  79         super(GitLabApiClient, self).__init__(verify_ssl)
  80 
  81     @property
  82     def identity(self):
  83         return self.installation.default_identity
  84 
  85     @property
  86     def metadata(self):
  87         return self.installation.model.metadata
  88 
  89     def request(self, method, path, data=None, params=None):
  90         access_token = self.identity.data["access_token"]
  91         headers = {"Authorization": u"Bearer {}".format(access_token)}
  92         url = GitLabApiClientPath.build_api_url(self.metadata["base_url"], path)
  93         try:
  94             return self._request(method, url, headers=headers, data=data, params=params)
  95         except ApiUnauthorized as e:
  96             if self.is_refreshing_token:
  97                 raise e
  98             self.is_refreshing_token = True
  99             self.refresh_auth()
 100             resp = self._request(method, url, headers=headers, data=data, params=params)
 101             self.is_refreshing_token = False
 102             return resp
 103 
 104     def refresh_auth(self):
 105         """
 106         Modeled after Doorkeeper's docs
 107         where Doorkeeper is a dependency for GitLab that handles OAuth
 108 
 109         https://github.com/doorkeeper-gem/doorkeeper/wiki/Enable-Refresh-Token-Credentials#testing-with-oauth2-gem
 110         """
 111         self.identity.get_provider().refresh_identity(
 112             self.identity,
 113             refresh_token_url="%s%s" % (self.metadata["base_url"], GitLabApiClientPath.oauth_token),
 114         )

 123     def search_group_projects(self, group, query=None, simple=True):
 124         """Get projects for a group
 125 
 126         See https://docs.gitlab.com/ee/api/groups.html#list-a-group-s-projects
 127         """
 128         # simple param returns limited fields for the project.
 129         # Really useful, because we often don't need most of the project information
 130         return self.get(
 131             GitLabApiClientPath.group_projects.format(group=group),
 132             params={
 133                 "search": query,
 134                 "simple": simple,
 135                 "include_subgroups": self.metadata.get("include_subgroups", False),
 136             },
 137         )

 181     def create_project_webhook(self, project_id):
 182         """Create a webhook on a project
 183 
 184         See https://docs.gitlab.com/ee/api/projects.html#add-project-hook
 185         """
 186         path = GitLabApiClientPath.project_hooks.format(project=project_id)
 187         hook_uri = reverse("sentry-extensions-gitlab-webhook")
 188         model = self.installation.model
 189         data = {
 190             "url": absolute_uri(hook_uri),
 191             "token": u"{}:{}".format(model.external_id, model.metadata["webhook_secret"]),
 192             "merge_requests_events": True,
 193             "push_events": True,
 194             "enable_ssl_verification": model.metadata["verify_ssl"],
 195         }
 196         resp = self.post(path, data)
 197 
 198         return resp["id"]

 208     def get_last_commits(self, project_id, end_sha):
 209         """Get the last set of commits ending at end_sha
 210 
 211         Gitlab doesn't give us a good way to do this, so we fetch the end_sha
 212         and use its date to find the block of commits. We only fetch one page
 213         of commits to match other implementations (github, bitbucket)
 214 
 215         See https://docs.gitlab.com/ee/api/commits.html#get-a-single-commit and
 216         https://docs.gitlab.com/ee/api/commits.html#list-repository-commits
 217         """
 218         path = GitLabApiClientPath.commit.format(project=project_id, sha=end_sha)
 219         commit = self.get(path)
 220         if not commit:
 221             return []
 222         end_date = commit["created_at"]
 223 
 224         path = GitLabApiClientPath.commits.format(project=project_id)
 225         return self.get(path, params={"until": end_date})
