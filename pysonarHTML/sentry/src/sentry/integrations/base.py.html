  60 class IntegrationMetadata(IntegrationMetadata):
  61     @staticmethod
  62     def feature_flag_name(f):
  63         """
  64         FeatureDescriptions are set using the IntegrationFeatures constants,
  65         however we expose them here as mappings to organization feature flags, thus
  66         we prefix them with `integration`.
  67         """
  68         if f is not None:
  69             return u"integrations-{}".format(f)
  70 
  71     def _asdict(self):
  72         metadata = super(IntegrationMetadata, self)._asdict()
  73         metadata["features"] = [
  74             {
  75                 "description": f.description.strip(),
  76                 "featureGate": self.feature_flag_name(f.featureGate.value),
  77             }
  78             for f in metadata["features"]
  79         ]
  80         return metadata

 106 class IntegrationProvider(PipelineProvider):
 107     """
 108     An integration provider describes a third party that can be registered within Sentry.
 109 
 110     The core behavior is simply how to add the integration (the setup
 111     pipeline), which will likely use a nested pipeline for identity
 112     authentication, and what kind of configuration is stored.
 113 
 114     This is similar to Sentry's legacy 'plugin' information, except that an
 115     integration is lives as an instance in the database, and the ``IntegrationProvider``
 116     class is just a descriptor for how that object functions, and what behavior
 117     it provides (such as extensions provided).
 118     """
 119 
 120     # a unique identifier (e.g. 'slack').
 121     # Used to lookup sibling classes and the ``key`` used when creating
 122     # Integration objects.
 123     key = None
 124 
 125     # a unique identifier to use when creating the ``Integration`` object.
 126     # Only needed when you want to create the above object with something other
 127     # than ``key``. See: VstsExtensionIntegrationProvider.
 128     _integration_key = None
 129 
 130     # Whether this integration should show up in the list on the Organization
 131     # Integrations page.
 132     visible = True
 133 
 134     # a human readable name (e.g. 'Slack')
 135     name = None
 136 
 137     # an IntegrationMetadata object, used to provide extra details in the
 138     # configuration interface of the integration.
 139     metadata = None
 140 
 141     # an Integration class that will manage the functionality once installed
 142     integration_cls = None
 143 
 144     # configuration for the setup dialog
 145     setup_dialog_config = {"width": 600, "height": 600}
 146 
 147     # whether or not the integration installation be initiated from Sentry
 148     can_add = True
 149 
 150     # can the integration be disabled ?
 151     can_disable = False
 152 
 153     # if the integration has no application-style access token, associate
 154     # the installer's identity to the organization integration
 155     needs_default_identity = False
 156 
 157     # can be any number of IntegrationFeatures
 158     features = frozenset()
 159 
 160     # if this is hidden without the feature flag
 161     requires_feature_flag = False
 162 
 163     @classmethod
 164     def get_installation(cls, model, organization_id, **kwargs):
 165         if cls.integration_cls is None:
 166             raise NotImplementedError
 167 
 168         return cls.integration_cls(model, organization_id, **kwargs)
 169 
 170     @property
 171     def integration_key(self):
 172         return self._integration_key or self.key
 173 
 174     def get_logger(self):
 175         return logging.getLogger("sentry.integration.%s" % (self.key,))
 176 
 177     def post_install(self, integration, organization, extra=None):
 178         pass

 180     def create_audit_log_entry(self, integration, organization, request, action, extra=None):
 181         """
 182         Creates an audit log entry for the newly installed integration.
 183         """
 184         if action == "install":
 185             create_audit_entry(
 186                 request=request,
 187                 organization=organization,
 188                 target_object=integration.id,
 189                 event=AuditLogEntryEvent.INTEGRATION_ADD,
 190                 data={"provider": integration.provider, "name": integration.name},
 191             )
 192 
 193     def get_pipeline_views(self):
 194         """
 195         Return a list of ``View`` instances describing this integration's
 196         configuration pipeline.
 197 
 198         >>> def get_pipeline_views(self):
 199         >>>    return []
 200         """
 201         raise NotImplementedError

 203     def build_integration(self, state):
 204         """
 205         Given state captured during the setup pipeline, return a dictionary
 206         of configuration and metadata to store with this integration.
 207 
 208         This data **must not** be specific to an organization, as the
 209         integration may be shared among multiple organizations.
 210 
 211         This is the ideal place to store metadata like the 'name' or 'url' to
 212         the relevant entity, or shared API keys.
 213 
 214         This **must** return an 'external_id' attribute.
 215 
 216         This **should** return a 'name' attribute.
 217 
 218         >>> def build_integration(self, state):
 219         >>>     return {
 220         >>>         'external_id': state['id'],
 221         >>>         'name': state['name'],
 222         >>>         'metadata': {url': state['url']},
 223         >>>     }
 224 
 225         This can return the 'expect_exists' flag, and this method  will expect
 226         that the passed 'external_id' exists and will not attempt to recreate
 227         or update the integration.
 228 
 229         >>> def build_integration(self, state):
 230         >>>    return {
 231         >>>        'external_id': state['id'],
 232         >>>        'expect_exists': True,
 233         >>>    }
 234 
 235         """
 236         raise NotImplementedError
 237 
 238     def setup(self):
 239         """
 240         Executed once Sentry has been initialized at runtime.
 241 
 242         >>> def setup(self):
 243         >>>     bindings.add('repository.provider', GitHubRepositoryProvider, key='github')
 244         """
 245 
 246     def has_feature(self, feature):
 247         return feature in self.features

 250 class IntegrationInstallation(object):
 251     """
 252     An IntegrationInstallation represents an installed integration and manages the
 253     core functionality of the integration.
 254     """
 255 
 256     logger = logging.getLogger("sentry.integrations")
 257 
 258     def __init__(self, model, organization_id):
 259         self.model = model
 260         self.organization_id = organization_id
 261         self._org_integration = None
 262 
 263     @property
 264     def org_integration(self):
 265         if self._org_integration is None:
 266             self._org_integration = OrganizationIntegration.objects.get(
 267                 organization_id=self.organization_id, integration_id=self.model.id
 268             )
 269         return self._org_integration
 270 
 271     def get_organization_config(self):
 272         """
 273         Returns a list of JSONForm configuration object descriptors used to
 274         configure the integration per-organization. This simply represents the
 275         configuration structure.
 276 
 277         See the JSONForm react component for structure details.
 278         """
 279         return []
 280 
 281     def update_organization_config(self, data):
 282         """
 283         Update the configuration field for an organization integration.
 284         """
 285         config = self.org_integration.config
 286         config.update(data)
 287         self.org_integration.update(config=config)
 288 
 289     def get_config_data(self):
 290         return self.org_integration.config
 291 
 292     def get_client(self):
 293         # Return the api client for a given provider
 294         raise NotImplementedError
 295 
 296     def get_default_identity(self):
 297         """
 298         For Integrations that rely solely on user auth for authentication
 299         """
 300 
 301         identity = Identity.objects.get(id=self.org_integration.default_auth_id)
 302         return identity
 303 
 304     def error_message_from_json(self, data):
 305         return data.get("message", "unknown error")
 306 
 307     def error_fields_from_json(self, data):
 308         """
 309         If we can determine error fields from the response JSON this should
 310         format and return them, allowing an IntegrationFormError to be raised.
 311         Return None if no form errors are present.
 312 
 313         Error fields should be in the format: {field: [message]}
 314         """
 315         return None
 316 
 317     def message_from_error(self, exc):
 318         if isinstance(exc, ApiUnauthorized):
 319             return ERR_UNAUTHORIZED
 320         elif isinstance(exc, ApiHostError):
 321             return exc.text
 322         elif isinstance(exc, UnsupportedResponseType):
 323             return ERR_UNSUPPORTED_RESPONSE_TYPE.format(content_type=exc.content_type)
 324         elif isinstance(exc, ApiError):
 325             if exc.json:
 326                 msg = self.error_message_from_json(exc.json) or "unknown error"
 327             else:
 328                 msg = "unknown error"
 329             return "Error Communicating with %s (HTTP %s): %s" % (
 330                 self.model.get_provider().name,
 331                 exc.code,
 332                 msg,
 333             )
 334         else:
 335             return ERR_INTERNAL
 336 
 337     def raise_error(self, exc, identity=None):
 338         if isinstance(exc, ApiUnauthorized):
 339             six.reraise(
 340                 InvalidIdentity,
 341                 InvalidIdentity(self.message_from_error(exc), identity=identity),
 342                 sys.exc_info()[2],
 343             )
 344         elif isinstance(exc, ApiError):
 345             if exc.json:
 346                 error_fields = self.error_fields_from_json(exc.json)
 347                 if error_fields is not None:
 348                     six.reraise(
 349                         IntegrationFormError, IntegrationFormError(error_fields), sys.exc_info()[2]
 350                     )
 351 
 352             six.reraise(
 353                 IntegrationError, IntegrationError(self.message_from_error(exc)), sys.exc_info()[2]
 354             )
 355         elif isinstance(exc, IntegrationError):
 356             raise
 357         else:
 358             self.logger.exception(six.text_type(exc))
 359             six.reraise(
 360                 IntegrationError, IntegrationError(self.message_from_error(exc)), sys.exc_info()[2]
 361             )
