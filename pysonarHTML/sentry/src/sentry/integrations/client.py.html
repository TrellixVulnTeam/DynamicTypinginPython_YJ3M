  20 class OAuth2RefreshMixin(object):

  22         """
  23         Checks if auth is expired and if so refreshes it
  24         """
  25         time_expires = self.identity.data.get("expires")

  29             self.identity.get_provider().refresh_identity(self.identity, *args, **kwargs)
