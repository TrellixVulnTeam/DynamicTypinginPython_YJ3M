  39 class BitbucketApiClient(ApiClient):
  40     """
  41     The API Client for the Bitbucket Integration
  42 
  43     NOTE: repo is the fully qualified slug containing 'username/repo_slug'
  44     """
  45 
  46     integration_name = "bitbucket"
  47 
  48     def __init__(self, base_url, shared_secret, subject, *args, **kwargs):
  49         # subject is probably the clientKey
  50         super(BitbucketApiClient, self).__init__(*args, **kwargs)
  51         self.base_url = base_url
  52         self.shared_secret = shared_secret
  53         self.subject = subject
  54 
  55     def request(self, method, path, data=None, params=None, **kwargs):
  56         jwt_payload = {
  57             "iss": BITBUCKET_KEY,
  58             "iat": datetime.datetime.utcnow(),
  59             "exp": datetime.datetime.utcnow() + datetime.timedelta(seconds=5 * 60),
  60             "qsh": get_query_hash(path, method.upper(), params),
  61             "sub": self.subject,
  62         }
  63         encoded_jwt = jwt.encode(jwt_payload, self.shared_secret)
  64         headers = {"Authorization": "JWT %s" % encoded_jwt}
  65         return self._request(method, path, data=data, params=params, headers=headers, **kwargs)

 109     def transform_patchset(self, patch_set):
 110         file_changes = []
 111         for patched_file in patch_set.added_files:
 112             file_changes.append({"path": patched_file.path, "type": "A"})
 113 
 114         for patched_file in patch_set.removed_files:
 115             file_changes.append({"path": patched_file.path, "type": "D"})
 116 
 117         for patched_file in patch_set.modified_files:
 118             file_changes.append({"path": patched_file.path, "type": "M"})
 119 
 120         return file_changes

 122     def get_commit_filechanges(self, repo, sha):
 123         resp = self.get(
 124             BitbucketAPIPath.repository_diff.format(repo=repo, spec=sha), allow_text=True
 125         )
 126         diff_file = resp.text
 127         ps = PatchSet.from_string(diff_file)
 128         return self.transform_patchset(ps)
 129 
 130     def zip_commit_data(self, repo, commit_list):
 131         for commit in commit_list:
 132             commit.update({"patch_set": self.get_commit_filechanges(repo, commit["hash"])})
 133         return commit_list

 135     def get_last_commits(self, repo, end_sha):
 136         # return api request that fetches last ~30 commits
 137         # see https://developer.atlassian.com/bitbucket/api/2/reference/resource/repositories/%7Busername%7D/%7Brepo_slug%7D/commits/%7Brevision%7D
 138         # using end_sha as parameter
 139         data = self.get(BitbucketAPIPath.repository_commits.format(repo=repo, revision=end_sha))
 140         return self.zip_commit_data(repo, data["values"])

 142     def compare_commits(self, repo, start_sha, end_sha):
 143         # where start_sha is oldest and end_sha is most recent
 144         # see
 145         # https://developer.atlassian.com/bitbucket/api/2/reference/resource/repositories/%7Busername%7D/%7Brepo_slug%7D/commits/%7Brevision%7D
 146         commits = []
 147         done = False
 148 
 149         url = BitbucketAPIPath.repository_commits.format(repo=repo, revision=end_sha)
 150 
 151         while not done and len(commits) < 90:
 152             data = self.get(url)
 153 
 154             for commit in data["values"]:
 155                 if commit["hash"] == start_sha:
 156                     done = True
 157                     break
 158                 commits.append(commit)
 159 
 160             # move page forward
 161             try:
 162                 url = data["next"]
 163             except KeyError:
 164                 break
 165 
 166         return self.zip_commit_data(repo, commits)
