  37 def parse_raw_user_email(raw):
  38     # captures content between angle brackets
  39     match = re.search("(?<=<).*(?=>$)", raw)
  40     if match is None:
  41         return
  42     return match.group(0)

  45 def parse_raw_user_name(raw):
  46     # captures content before angle bracket
  47     return raw.split("<")[0].strip()

  50 class Webhook(object):
  51     def __call__(self, organization, event):
  52         raise NotImplementedError
  53 
  54     def update_repo_data(self, repo, event):
  55         """
  56         Given a webhook payload, update stored repo data if needed.
  57 
  58         NB: Assumes event['repository']['full_name'] is defined. Rework this if
  59         that stops being a safe assumption.
  60         """
  61 
  62         name_from_event = event["repository"]["full_name"]
  63         # build the URL manually since it doesn't come back from the API in
  64         # the form that we need
  65         # see https://confluence.atlassian.com/bitbucket/event-payloads-740262817.html#EventPayloads-entity_repository
  66         # and click on 'Repository property' underneath the table for example data
  67         # (all entries are from the `api` subdomain, rather than bitbucket.org)
  68         url_from_event = u"https://bitbucket.org/{}".format(name_from_event)
  69 
  70         if (
  71             repo.name != name_from_event
  72             or repo.config.get("name") != name_from_event
  73             or repo.url != url_from_event
  74         ):
  75             repo.update(
  76                 name=name_from_event,
  77                 url=url_from_event,
  78                 config=dict(repo.config, name=name_from_event),
  79             )

  82 class PushEventWebhook(Webhook):
  83     # https://confluence.atlassian.com/bitbucket/event-payloads-740262817.html#EventPayloads-Push
  84     def __call__(self, organization, event):
  85         authors = {}
  86 
  87         try:
  88             repo = Repository.objects.get(
  89                 organization_id=organization.id,
  90                 provider=PROVIDER_NAME,
  91                 external_id=six.text_type(event["repository"]["uuid"]),
  92             )
  93         except Repository.DoesNotExist:
  94             raise Http404()
  95 
  96         # while we're here, make sure repo data is up to date
  97         self.update_repo_data(repo, event)
  98 
  99         for change in event["push"]["changes"]:
 100             for commit in change.get("commits", []):
 101                 if IntegrationRepositoryProvider.should_ignore_commit(commit["message"]):
 102                     continue
 103 
 104                 author_email = parse_raw_user_email(commit["author"]["raw"])
 105 
 106                 # TODO(dcramer): we need to deal with bad values here, but since
 107                 # its optional, lets just throw it out for now
 108                 if author_email is None or len(author_email) > 75:
 109                     author = None
 110                 elif author_email not in authors:
 111                     authors[author_email] = author = CommitAuthor.objects.get_or_create(
 112                         organization_id=organization.id,
 113                         email=author_email,
 114                         defaults={"name": commit["author"]["raw"].split("<")[0].strip()},
 115                     )[0]
 116                 else:
 117                     author = authors[author_email]
 118                 try:
 119                     with transaction.atomic():
 120 
 121                         Commit.objects.create(
 122                             repository_id=repo.id,
 123                             organization_id=organization.id,
 124                             key=commit["hash"],
 125                             message=commit["message"],
 126                             author=author,
 127                             date_added=dateutil.parser.parse(commit["date"]).astimezone(
 128                                 timezone.utc
 129                             ),
 130                         )
 131 
 132                 except IntegrityError:
 133                     pass

 136 class BitbucketWebhookEndpoint(View):
 137     _handlers = {"repo:push": PushEventWebhook}
 138 
 139     def get_handler(self, event_type):
 140         return self._handlers.get(event_type)
 141 
 142     @method_decorator(csrf_exempt)
 143     def dispatch(self, request, *args, **kwargs):
 144         if request.method != "POST":
 145             return HttpResponse(status=405)
 146 
 147         return super(BitbucketWebhookEndpoint, self).dispatch(request, *args, **kwargs)
 148 
 149     def post(self, request, organization_id):
 150         try:
 151             organization = Organization.objects.get_from_cache(id=organization_id)
 152         except Organization.DoesNotExist:
 153             logger.error(
 154                 PROVIDER_NAME + ".webhook.invalid-organization",
 155                 extra={"organization_id": organization_id},
 156             )
 157             return HttpResponse(status=400)
 158 
 159         body = six.binary_type(request.body)
 160         if not body:
 161             logger.error(
 162                 PROVIDER_NAME + ".webhook.missing-body", extra={"organization_id": organization.id}
 163             )
 164             return HttpResponse(status=400)
 165 
 166         try:
 167             handler = self.get_handler(request.META["HTTP_X_EVENT_KEY"])
 168         except KeyError:
 169             logger.error(
 170                 PROVIDER_NAME + ".webhook.missing-event", extra={"organization_id": organization.id}
 171             )
 172             return HttpResponse(status=400)
 173 
 174         if not handler:
 175             return HttpResponse(status=204)
 176 
 177         address_string = six.text_type(request.META["REMOTE_ADDR"])
 178         ip = ipaddress.ip_address(address_string)
 179         valid_ip = False
 180         for ip_range in BITBUCKET_IP_RANGES:
 181             if ip in ip_range:
 182                 valid_ip = True
 183                 break
 184 
 185         if not valid_ip and address_string not in BITBUCKET_IPS:
 186             logger.error(
 187                 PROVIDER_NAME + ".webhook.invalid-ip-range",
 188                 extra={"organization_id": organization.id},
 189             )
 190             return HttpResponse(status=401)
 191 
 192         try:
 193             event = json.loads(body.decode("utf-8"))
 194         except JSONDecodeError:
 195             logger.error(
 196                 PROVIDER_NAME + ".webhook.invalid-json",
 197                 extra={"organization_id": organization.id},
 198                 exc_info=True,
 199             )
 200             return HttpResponse(status=400)
 201 
 202         handler()(organization, event)
 203         return HttpResponse(status=204)
