  19     def get_installation(self, integration_id, organization_id):
  20         if integration_id is None:
  21             raise IntegrationError("Bitbucket requires an integration id.")
  22         integration_model = Integration.objects.get(
  23             id=integration_id, organizations=organization_id, provider="bitbucket"
  24         )
  25 
  26         return integration_model.get_installation(organization_id)

  28     def get_repository_data(self, organization, config):
  29         installation = self.get_installation(config.get("installation"), organization.id)
  30         client = installation.get_client()
  31         try:
  32             repo = client.get_repo(config["identifier"])
  33         except Exception as e:
  34             installation.raise_error(e)
  35         else:
  36             config["external_id"] = six.text_type(repo["uuid"])
  37             config["name"] = repo["full_name"]
  38         return config

  40     def get_webhook_secret(self, organization):
  41         # TODO(LB): Revisit whether Integrations V3 should be using OrganizationOption for storage
  42         lock = locks.get(u"bitbucket:webhook-secret:{}".format(organization.id), duration=60)
  43         with lock.acquire():
  44             secret = OrganizationOption.objects.get_value(
  45                 organization=organization, key="bitbucket:webhook_secret"
  46             )
  47             if secret is None:
  48                 secret = generate_token()
  49                 OrganizationOption.objects.set_value(
  50                     organization=organization, key="bitbucket:webhook_secret", value=secret
  51                 )
  52         return secret

  54     def build_repository_config(self, organization, data):
  55         installation = self.get_installation(data.get("installation"), organization.id)
  56         client = installation.get_client()
  57         try:
  58             resp = client.create_hook(
  59                 data["identifier"],
  60                 {
  61                     "description": "sentry-bitbucket-repo-hook",
  62                     "url": absolute_uri(
  63                         u"/extensions/bitbucket/organizations/{}/webhook/".format(organization.id)
  64                     ),
  65                     "active": True,
  66                     "events": ["repo:push", "pullrequest:fulfilled"],
  67                 },
  68             )
  69         except Exception as e:
  70             installation.raise_error(e)
  71         else:
  72             return {
  73                 "name": data["identifier"],
  74                 "external_id": data["external_id"],
  75                 "url": u"https://bitbucket.org/{}".format(data["name"]),
  76                 "config": {"name": data["name"], "webhook_id": resp["uuid"]},
  77                 "integration_id": data["installation"],

  80     def on_delete_repository(self, repo):
  81         installation = self.get_installation(repo.integration_id, repo.organization_id)
  82         client = installation.get_client()
  83 
  84         try:
  85             client.delete_hook(repo.config["name"], repo.config["webhook_id"])

  91     def _format_commits(self, repo, commit_list):
  92         return [
  93             {
  94                 "id": c["hash"],
  95                 "repository": repo.name,
  96                 "author_email": parse_raw_user_email(c["author"]["raw"]),
  97                 "author_name": parse_raw_user_name(c["author"]["raw"]),
  98                 "message": c["message"],
  99                 "timestamp": self.format_date(c["date"]),
 100                 "patch_set": c.get("patch_set"),
 101             }
 102             for c in commit_list
 103         ]

 105     def compare_commits(self, repo, start_sha, end_sha):
 106         installation = self.get_installation(repo.integration_id, repo.organization_id)
 107         client = installation.get_client()
 108         # use config name because that is kept in sync via webhooks
 109         name = repo.config["name"]
 110         if start_sha is None:
 111             try:
 112                 res = client.get_last_commits(name, end_sha)
 113             except Exception as e:
 114                 installation.raise_error(e)
 115             else:
 116                 return self._format_commits(repo, res[:10])
 117         else:
 118             try:
 119                 res = client.compare_commits(name, start_sha, end_sha)
 120             except Exception as e:
 121                 installation.raise_error(e)
 122             else:
 123                 return self._format_commits(repo, res)
