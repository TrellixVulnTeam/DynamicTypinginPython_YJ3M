 112     def create_issue(self, data, **kwargs):
 113         client = self.get_client()
 114         if not data.get("repo"):
 115             raise IntegrationFormError({"repo": ["Repository is required"]})
 116 
 117         data["content"] = {"raw": data["description"]}
 118         del data["description"]
 119 
 120         try:
 121             issue = client.create_issue(data.get("repo"), data)
 122         except ApiError as e:
 123             self.raise_error(e)
 124 
 125         return {
 126             "key": issue["id"],
 127             "title": issue["title"],
 128             "description": issue["content"]["html"],  # users content rendered as html
 129             "repo": data.get("repo"),
 130         }
 131 
 132     def get_issue(self, issue_id, **kwargs):
 133         client = self.get_client()
 134         repo = kwargs["data"].get("repo")
 135         issue = client.get_issue(repo, issue_id)
 136         return {
 137             "key": issue["id"],
 138             "title": issue["title"],
 139             "description": issue["content"]["html"],  # users content rendered as html
 140             "repo": repo,
 141         }

 146     def after_link_issue(self, external_issue, **kwargs):
 147         data = kwargs["data"]
 148         client = self.get_client()
 149 
 150         repo, issue_num = external_issue.key.split("#")
 151 
 152         if not repo:
 153             raise IntegrationFormError({"repo": "Repository is required"})
 154         if not issue_num:
 155             raise IntegrationFormError({"externalIssue": "Issue ID is required"})
 156 
 157         comment = data.get("comment")
 158         if comment:
 159             try:
 160                 client.create_comment(
 161                     repo=repo, issue_id=issue_num, data={"content": {"raw": comment}}
 162                 )
 163             except ApiError as e:
 164                 self.raise_error(e)
