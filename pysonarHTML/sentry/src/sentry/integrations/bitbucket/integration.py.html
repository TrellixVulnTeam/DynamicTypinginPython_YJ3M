  76 class BitbucketIntegration(IntegrationInstallation, BitbucketIssueBasicMixin, RepositoryMixin):
  77     repo_search = True
  78 
  79     def get_client(self):
  80         return BitbucketApiClient(
  81             self.model.metadata["base_url"],
  82             self.model.metadata["shared_secret"],
  83             self.model.external_id,
  84         )
  85 
  86     @property
  87     def username(self):
  88         return self.model.name
  89 
  90     def error_message_from_json(self, data):
  91         return data.get("error", {}).get("message", "unknown error")
  92 
  93     def get_repositories(self, query=None):
  94         if not query:
  95             resp = self.get_client().get_repos(self.username)
  96             return [
  97                 {"identifier": repo["full_name"], "name": repo["full_name"]}
  98                 for repo in resp.get("values", [])
  99             ]
 100 
 101         exact_query = (u'name="%s"' % (query)).encode("utf-8")
 102         fuzzy_query = (u'name~"%s"' % (query)).encode("utf-8")
 103 
 104         exact_search_resp = self.get_client().search_repositories(self.username, exact_query)
 105         fuzzy_search_resp = self.get_client().search_repositories(self.username, fuzzy_query)
 106 
 107         result = OrderedSet()
 108 
 109         for j in exact_search_resp.get("values", []):
 110             result.add(j["full_name"])
 111 
 112         for i in fuzzy_search_resp.get("values", []):
 113             result.add(i["full_name"])
 114 
 115         return [{"identifier": full_name, "name": full_name} for full_name in result]
 116 
 117     def has_repo_access(self, repo):
 118         client = self.get_client()
 119         try:
 120             client.get_hooks(repo.config["name"])
 121         except ApiError:
 122             return False
 123         return True
 124 
 125     def get_unmigratable_repositories(self):
 126         repos = Repository.objects.filter(
 127             organization_id=self.organization_id, provider="bitbucket"
 128         )
 129 
 130         accessible_repos = [r["identifier"] for r in self.get_repositories()]
 131 
 132         return [repo for repo in repos if repo.name not in accessible_repos]
 133 
 134     def reinstall(self):
 135         self.reinstall_repositories()

 138 class BitbucketIntegrationProvider(IntegrationProvider):
 139     key = "bitbucket"
 140     name = "Bitbucket"
 141     metadata = metadata
 142     scopes = scopes
 143     integration_cls = BitbucketIntegration
 144     features = frozenset([IntegrationFeatures.ISSUE_BASIC, IntegrationFeatures.COMMITS])
 145 
 146     def get_pipeline_views(self):
 147         identity_pipeline_config = {"redirect_url": absolute_uri("/extensions/bitbucket/setup/")}
 148         identity_pipeline_view = NestedPipelineView(
 149             bind_key="identity",
 150             provider_key="bitbucket",
 151             pipeline_cls=IdentityProviderPipeline,
 152             config=identity_pipeline_config,
 153         )
 154         return [identity_pipeline_view, VerifyInstallation()]
 155 
 156     def post_install(self, integration, organization, extra=None):
 157         repo_ids = Repository.objects.filter(
 158             organization_id=organization.id,
 159             provider__in=["bitbucket", "integrations:bitbucket"],
 160             integration_id__isnull=True,
 161         ).values_list("id", flat=True)
 162 
 163         for repo_id in repo_ids:
 164             migrate_repo.apply_async(
 165                 kwargs={
 166                     "repo_id": repo_id,
 167                     "integration_id": integration.id,
 168                     "organization_id": organization.id,
 169                 }
 170             )
 171 
 172     def build_integration(self, state):
 173         if state.get("publicKey"):
 174             principal_data = state["principal"]
 175 
 176             domain = principal_data["links"]["html"]["href"].replace("https://", "").rstrip("/")
 177 
 178             return {
 179                 "provider": self.key,
 180                 "external_id": state["clientKey"],
 181                 "name": principal_data.get("username", principal_data["uuid"]),
 182                 "metadata": {
 183                     "public_key": state["publicKey"],
 184                     "shared_secret": state["sharedSecret"],
 185                     "base_url": state["baseApiUrl"],
 186                     "domain_name": domain,
 187                     "icon": principal_data["links"]["avatar"]["href"],
 188                     "scopes": self.scopes,
 189                     "uuid": principal_data["uuid"],
 190                     "type": principal_data["type"],  # team or user account
 191                 },
 192             }
 193         else:
 194             return {
 195                 "provider": self.key,
 196                 "external_id": state["external_id"],
 197                 "expect_exists": True,
 198             }
 199 
 200     def setup(self):
 201         from sentry.plugins.base import bindings
 202 
 203         bindings.add(
 204             "integration-repository.provider",
 205             BitbucketRepositoryProvider,
 206             id="integrations:%s" % self.key,
 207         )

 210 class VerifyInstallation(PipelineView):
 211     def dispatch(self, request, pipeline):
 212         try:
 213             integration = get_integration_from_request(request, BitbucketIntegrationProvider.key)
 214         except AtlassianConnectValidationError:
 215             return pipeline.error("Unable to verify installation.")
 216         pipeline.bind_state("external_id", integration.external_id)
 217         return pipeline.next_step()
