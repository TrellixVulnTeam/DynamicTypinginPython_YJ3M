  28 def get_installation_metadata(event, host):
  29     if not host:
  30         return
  31     try:
  32         integration = Integration.objects.get(
  33             external_id=u"{}:{}".format(host, event["installation"]["id"]),
  34             provider="github_enterprise",
  35         )
  36     except Integration.DoesNotExist:
  37         return
  38     return integration.metadata["installation"]

  89     def is_valid_signature(self, method, body, secret, signature):
  90         if method != "sha1":
  91             raise NotImplementedError("signature method %s is not supported" % (method,))
  92         expected = hmac.new(
  93             key=secret.encode("utf-8"), msg=body, digestmod=hashlib.sha1
  94         ).hexdigest()
  95         return constant_time_compare(expected, signature)

 114     def handle(self, request):
 115         body = six.binary_type(request.body)
 116         if not body:
 117             logger.warning("github_enterprise.webhook.missing-body", extra=self.get_logging_data())
 118             return HttpResponse(status=400)
 119 
 120         try:
 121             handler = self.get_handler(request.META["HTTP_X_GITHUB_EVENT"])
 122         except KeyError:
 123             logger.warning("github_enterprise.webhook.missing-event", extra=self.get_logging_data())
 124             return HttpResponse(status=400)
 125 
 126         if not handler:
 127             return HttpResponse(status=204)
 128 
 129         try:
 130             event = json.loads(body.decode("utf-8"))
 131         except JSONDecodeError:
 132             logger.warning(
 133                 "github_enterprise.webhook.invalid-json",
 134                 extra=self.get_logging_data(),
 135                 exc_info=True,
 136             )
 137             return HttpResponse(status=400)
 138 
 139         try:
 140             host = request.META["HTTP_X_GITHUB_ENTERPRISE_HOST"]
 141         except KeyError:
 142             logger.warning("github_enterprise.webhook.missing-enterprise-host")
 143             return HttpResponse(status=400)
 144 
 145         secret = self.get_secret(event, host)
 146         if not secret:
 147             logger.warning("github_enterprise.webhook.missing-integration", extra={"host": host})
 148             return HttpResponse(status=400)
 149 
 150         try:
 151             # Attempt to validate the signature. Older versions of
 152             # GitHub Enterprise do not send the signature so this is an optional step.
 153             method, signature = request.META["HTTP_X_HUB_SIGNATURE"].split("=", 1)
 154             if not self.is_valid_signature(method, body, secret, signature):
 155                 logger.warning(
 156                     "github_enterprise.webhook.invalid-signature", extra=self.get_logging_data()
 157                 )
 158                 return HttpResponse(status=401)
 159         except (KeyError, IndexError) as e:
 160             logger.info(
 161                 "github_enterprise.webhook.missing-signature",
 162                 extra={"host": host, "error": six.text_type(e)},
 163             )
 164         handler()(event, host)
 165         return HttpResponse(status=204)
