 104 class GitHubEnterpriseIntegration(IntegrationInstallation, GitHubIssueBasic, RepositoryMixin):
 105     repo_search = True
 106 
 107     def get_client(self):
 108         base_url = self.model.metadata["domain_name"].split("/")[0]
 109         return GitHubEnterpriseAppsClient(
 110             base_url=base_url,
 111             integration=self.model,
 112             private_key=self.model.metadata["installation"]["private_key"],
 113             app_id=self.model.metadata["installation"]["id"],
 114             verify_ssl=self.model.metadata["installation"]["verify_ssl"],
 115         )
 116 
 117     def get_repositories(self, query=None):
 118         if not query:
 119             return [
 120                 {"name": i["name"], "identifier": i["full_name"]}
 121                 for i in self.get_client().get_repositories()
 122             ]
 123 
 124         full_query = build_repository_query(self.model.metadata, self.model.name, query)
 125         response = self.get_client().search_repositories(full_query)
 126         return [
 127             {"name": i["name"], "identifier": i["full_name"]} for i in response.get("items", [])
 128         ]
 129 
 130     def search_issues(self, query):
 131         return self.get_client().search_issues(query)
 132 
 133     def reinstall(self):
 134         installation_id = self.model.external_id.split(":")[1]
 135         metadata = self.model.metadata
 136         metadata["installation_id"] = installation_id
 137         self.model.update(metadata=metadata)
 138         self.reinstall_repositories()
 139 
 140     def message_from_error(self, exc):
 141         if isinstance(exc, ApiError):
 142             message = API_ERRORS.get(exc.code)
 143             if message is None:
 144                 message = exc.json.get("message", "unknown error") if exc.json else "unknown error"
 145             return "Error Communicating with GitHub Enterprise (HTTP %s): %s" % (exc.code, message)
 146         else:
 147             return ERR_INTERNAL

 218 class InstallationConfigView(PipelineView):
 219     def dispatch(self, request, pipeline):
 220         if request.method == "POST":
 221             form = InstallationForm(request.POST)
 222             if form.is_valid():
 223                 form_data = form.cleaned_data
 224                 form_data["url"] = urlparse(form_data["url"]).netloc
 225 
 226                 pipeline.bind_state("installation_data", form_data)
 227 
 228                 pipeline.bind_state(
 229                     "oauth_config_information",
 230                     {
 231                         "access_token_url": u"https://{}/login/oauth/access_token".format(
 232                             form_data.get("url")
 233                         ),
 234                         "authorize_url": u"https://{}/login/oauth/authorize".format(
 235                             form_data.get("url")
 236                         ),
 237                         "client_id": form_data.get("client_id"),
 238                         "client_secret": form_data.get("client_secret"),
 239                         "verify_ssl": form_data.get("verify_ssl"),
 240                     },
 241                 )
 242 
 243                 return pipeline.next_step()
 244         else:
 245             form = InstallationForm()
 246 
 247         return render_to_response(
 248             template="sentry/integrations/github-enterprise-config.html",
 249             context={"form": form},
 250             request=request,
 251         )

 254 class GitHubEnterpriseIntegrationProvider(GitHubIntegrationProvider):
 255     key = "github_enterprise"

 260     def _make_identity_pipeline_view(self):
 261         """
 262         Make the nested identity provider view. It is important that this view is
 263         not constructed until we reach this step and the
 264         ``oauth_config_information`` is available in the pipeline state. This
 265         method should be late bound into the pipeline vies.
 266         """
 267         identity_pipeline_config = dict(
 268             oauth_scopes=(),
 269             redirect_url=absolute_uri("/extensions/github-enterprise/setup/"),
 270             **self.pipeline.fetch_state("oauth_config_information")
 271         )
 272 
 273         return NestedPipelineView(
 274             bind_key="identity",
 275             provider_key="github_enterprise",
 276             pipeline_cls=IdentityProviderPipeline,
 277             config=identity_pipeline_config,
 278         )
 279 
 280     def get_pipeline_views(self):
 281         return [
 282             InstallationConfigView(),
 283             GitHubEnterpriseInstallationRedirect(),
 284             # The identity provider pipeline should be constructed at execution
 285             # time, this allows for the oauth configuration parameters to be made
 286             # available from the installation config view.
 287             lambda: self._make_identity_pipeline_view(),
 288         ]

 293     def get_installation_info(self, installation_data, access_token, installation_id):
 294         session = http.build_session()
 295         resp = session.get(
 296             u"https://{}/api/v3/app/installations/{}".format(
 297                 installation_data["url"], installation_id
 298             ),
 299             headers={
 300                 "Authorization": "Bearer %s"
 301                 % get_jwt(
 302                     github_id=installation_data["id"],
 303                     github_private_key=installation_data["private_key"],
 304                 ),
 305                 "Accept": "application/vnd.github.machine-man-preview+json",
 306             },
 307             verify=installation_data["verify_ssl"],
 308         )
 309         resp.raise_for_status()
 310         installation_resp = resp.json()
 311 
 312         resp = session.get(
 313             u"https://{}/api/v3/user/installations".format(installation_data["url"]),
 314             params={"access_token": access_token},
 315             headers={"Accept": "application/vnd.github.machine-man-preview+json"},
 316             verify=installation_data["verify_ssl"],
 317         )
 318         resp.raise_for_status()
 319         user_installations_resp = resp.json()
 320 
 321         # verify that user actually has access to the installation
 322         for installation in user_installations_resp["installations"]:
 323             if installation["id"] == installation_resp["id"]:
 324                 return installation_resp
 325 
 326         return None

 328     def build_integration(self, state):
 329         identity = state["identity"]["data"]
 330         installation_data = state["installation_data"]
 331         user = get_user_info(installation_data["url"], identity["access_token"])
 332         installation = self.get_installation_info(
 333             installation_data, identity["access_token"], state["installation_id"]
 334         )
 335 
 336         domain = urlparse(installation["account"]["html_url"]).netloc
 337         integration = {
 338             "name": installation["account"]["login"],
 339             # installation id is not enough to be unique for self-hosted GH
 340             "external_id": u"{}:{}".format(domain, installation["id"]),
 341             # GitHub identity is associated directly to the application, *not*
 342             # to the installation itself.
 343             # app id is not enough to be unique for self-hosted GH
 344             "idp_external_id": u"{}:{}".format(domain, installation["app_id"]),
 345             "metadata": {
 346                 # The access token will be populated upon API usage
 347                 "access_token": None,
 348                 "expires_at": None,
 349                 "icon": installation["account"]["avatar_url"],
 350                 "domain_name": installation["account"]["html_url"].replace("https://", ""),
 351                 "account_type": installation["account"]["type"],
 352                 "installation_id": installation["id"],
 353                 "installation": installation_data,
 354             },
 355             "user_identity": {
 356                 "type": "github_enterprise",
 357                 "external_id": user["id"],
 358                 "scopes": [],  # GitHub apps do not have user scopes
 359                 "data": {"access_token": identity["access_token"]},
 360             },
 361             "idp_config": state["oauth_config_information"],
 362         }
 363 
 364         if state.get("reinstall_id"):
 365             integration["reinstall_id"] = state["reinstall_id"]
 366 
 367         return integration

 379 class GitHubEnterpriseInstallationRedirect(PipelineView):
 380     def get_app_url(self, installation_data):
 381         url = installation_data.get("url")
 382         name = installation_data.get("name")
 383         return u"https://{}/github-apps/{}".format(url, name)
 384 
 385     def dispatch(self, request, pipeline):
 386         installation_data = pipeline.fetch_state(key="installation_data")
 387         if "reinstall_id" in request.GET:
 388             pipeline.bind_state("reinstall_id", request.GET["reinstall_id"])
 389 
 390         if "installation_id" in request.GET:
 391             pipeline.bind_state("installation_id", request.GET["installation_id"])
 392             return pipeline.next_step()
 393 
 394         return self.redirect(self.get_app_url(installation_data))
