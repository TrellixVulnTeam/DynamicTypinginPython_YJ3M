  39     def post(self, request, *args, **kwargs):
  40         data = request.data
  41         try:
  42             event_type = data["eventType"]
  43             external_id = data["resourceContainers"]["collection"]["id"]
  44         except KeyError as e:
  45             logger.info("vsts.invalid-webhook-payload", extra={"error": six.text_type(e)})
  46 
  47         # https://docs.microsoft.com/en-us/azure/devops/service-hooks/events?view=azure-devops#workitem.updated
  48         if event_type == "workitem.updated":
  49             try:
  50                 integration = Integration.objects.get(
  51                     provider=PROVIDER_KEY, external_id=external_id
  52                 )
  53             except Integration.DoesNotExist:
  54                 logger.info(
  55                     "vsts.integration-in-webhook-payload-does-not-exist",
  56                     extra={"external_id": external_id, "event_type": event_type},
  57                 )
  58 
  59             try:
  60                 self.check_webhook_secret(request, integration)
  61                 logger.info(
  62                     "vsts.valid-webhook-secret",
  63                     extra={"event_type": event_type, "integration_id": integration.id},
  64                 )
  65             except AssertionError:
  66                 logger.info(
  67                     "vsts.invalid-webhook-secret",
  68                     extra={"event_type": event_type, "integration_id": integration.id},
  69                 )
  70                 return self.respond(status=401)
  71             self.handle_updated_workitem(data, integration)
  72         return self.respond()

  74     def check_webhook_secret(self, request, integration):
  75         try:
  76             integration_secret = integration.metadata["subscription"]["secret"]
  77             webhook_payload_secret = request.META["HTTP_SHARED_SECRET"]
  78             # TODO(Steve): remove
  79             logger.info(
  80                 "vsts.special-webhook-secret",
  81                 extra={
  82                     "integration_id": integration.id,
  83                     "integration_secret": six.text_type(integration_secret)[:6],
  84                     "webhook_payload_secret": six.text_type(webhook_payload_secret)[:6],
  85                 },
  86             )
  87         except KeyError as e:
  88             logger.info(
  89                 "vsts.missing-webhook-secret",
  90                 extra={"error": six.text_type(e), "integration_id": integration.id},
  91             )
  92 
  93         assert constant_time_compare(integration_secret, webhook_payload_secret)

  95     def handle_updated_workitem(self, data, integration):
  96         project = None
  97         try:
  98             external_issue_key = data["resource"]["workItemId"]
  99             project = data["resourceContainers"]["project"]["id"]
 100         except KeyError as e:
 101             logger.info(
 102                 "vsts.updating-workitem-does-not-have-necessary-information",
 103                 extra={"error": six.text_type(e), "integration_id": integration.id},
 104             )
 105 
 106         try:
 107             assigned_to = data["resource"]["fields"].get("System.AssignedTo")
 108             status_change = data["resource"]["fields"].get("System.State")
 109         except KeyError as e:
 110             logger.info(
 111                 "vsts.updated-workitem-fields-not-passed",
 112                 extra={
 113                     "error": six.text_type(e),
 114                     "workItemId": data["resource"]["workItemId"],
 115                     "integration_id": integration.id,
 116                     "azure_project_id": project,
 117                 },
 118             )
 119             return  # In the case that there are no fields sent, no syncing can be done
 120         logger.info(
 121             "vsts.updated-workitem-fields-correct",
 122             extra={
 123                 "workItemId": data["resource"]["workItemId"],
 124                 "integration_id": integration.id,
 125                 "azure_project_id": project,
 126             },
 127         )
 128         self.handle_assign_to(integration, external_issue_key, assigned_to)
 129         self.handle_status_change(integration, external_issue_key, status_change, project)

 131     def handle_assign_to(self, integration, external_issue_key, assigned_to):
 132         if not assigned_to:
 133             return
 134         new_value = assigned_to.get("newValue")
 135         if new_value is not None:
 136             try:
 137                 email = self.parse_email(new_value)
 138             except AttributeError as e:
 139                 logger.info(
 140                     "vsts.failed-to-parse-email-in-handle-assign-to",
 141                     extra={
 142                         "error": six.text_type(e),
 143                         "integration_id": integration.id,
 144                         "assigned_to_values": assigned_to,
 145                         "external_issue_key": external_issue_key,
 146                     },
 147                 )
 148                 return  # TODO(lb): return if cannot parse email?
 149             assign = True
 150         else:
 151             email = None
 152             assign = False
 153 
 154         sync_group_assignee_inbound(
 155             integration=integration,
 156             email=email,
 157             external_issue_key=external_issue_key,
 158             assign=assign,
 159         )

 161     def handle_status_change(self, integration, external_issue_key, status_change, project):
 162         if status_change is None:
 163             return
 164 
 165         organization_ids = OrganizationIntegration.objects.filter(
 166             integration_id=integration.id
 167         ).values_list("organization_id", flat=True)
 168 
 169         for organization_id in organization_ids:
 170             installation = integration.get_installation(organization_id)
 171             data = {
 172                 "new_state": status_change["newValue"],
 173                 # old_state is None when the issue is New
 174                 "old_state": status_change.get("oldValue"),
 175                 "project": project,
 176             }
 177 
 178             installation.sync_status_inbound(external_issue_key, data)
