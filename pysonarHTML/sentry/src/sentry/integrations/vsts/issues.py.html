  31     def get_project_choices(self, group, **kwargs):
  32         client = self.get_client()
  33         try:
  34             projects = client.get_projects(self.instance)["value"]
  35         except (ApiError, ApiUnauthorized, KeyError) as e:
  36             self.raise_error(e)
  37 
  38         project_choices = [(project["id"], project["name"]) for project in projects]
  39 
  40         params = kwargs.get("params", {})
  41         defaults = self.get_project_defaults(group.project_id)
  42         try:
  43             default_project = params.get(
  44                 "project", defaults.get("project") or project_choices[0][0]
  45             )
  46         except IndexError:
  47             return None, project_choices
  48 
  49         # If a project has been selected outside of the default list of
  50         # projects, stick it onto the front of the list so that it can be
  51         # selected.
  52         try:
  53             next(True for r in project_choices if r[0] == default_project)
  54         except StopIteration:
  55             try:
  56                 project_choices.insert(0, self.create_default_repo_choice(default_project))
  57             except (ApiError, ApiUnauthorized):
  58                 return None, project_choices
  59 
  60         return default_project, project_choices

  62     def get_create_issue_config(self, group, **kwargs):
  63         kwargs["link_referrer"] = "vsts_integration"
  64         fields = super(VstsIssueSync, self).get_create_issue_config(group, **kwargs)
  65         # Azure/VSTS has BOTH projects and repositories. A project can have many repositories.
  66         # Workitems (issues) are associated with the project not the repository.
  67         default_project, project_choices = self.get_project_choices(group, **kwargs)
  68 
  69         return [
  70             {
  71                 "name": "project",
  72                 "required": True,
  73                 "type": "choice",
  74                 "choices": project_choices,
  75                 "defaultValue": default_project,
  76                 "label": _("Project"),
  77                 "placeholder": default_project or _("MyProject"),
  78             }
  79         ] + fields

  81     def get_link_issue_config(self, group, **kwargs):
  82         fields = super(VstsIssueSync, self).get_link_issue_config(group, **kwargs)
  83         org = group.organization
  84         autocomplete_url = reverse("sentry-extensions-vsts-search", args=[org.slug, self.model.id])
  85         for field in fields:
  86             if field["name"] == "externalIssue":
  87                 field["url"] = autocomplete_url
  88                 field["type"] = "select"
  89         return fields

  94     def create_issue(self, data, **kwargs):
  95         """
  96         Creates the issue on the remote service and returns an issue ID.
  97         """
  98         project_id = data.get("project")
  99         if project_id is None:
 100             raise ValueError("Azure DevOps expects project")
 101 
 102         client = self.get_client()
 103 
 104         title = data["title"]
 105         description = data["description"]
 106 
 107         try:
 108             created_item = client.create_work_item(
 109                 instance=self.instance,
 110                 project=project_id,
 111                 title=title,
 112                 # Descriptions cannot easily be seen. So, a comment will be added as well.
 113                 description=markdown(description),
 114                 comment=markdown(description),
 115             )
 116         except Exception as e:
 117             self.raise_error(e)
 118 
 119         project_name = created_item["fields"]["System.AreaPath"]
 120         return {
 121             "key": six.text_type(created_item["id"]),
 122             "title": title,
 123             "description": description,
 124             "metadata": {"display_name": "%s#%s" % (project_name, created_item["id"])},
 125         }

 139     def sync_assignee_outbound(self, external_issue, user, assign=True, **kwargs):
 140         client = self.get_client()
 141         assignee = None
 142 
 143         if assign is True:
 144             sentry_emails = [email.email.lower() for email in user.get_verified_emails()]
 145             continuation_token = None
 146             while True:
 147                 vsts_users = client.get_users(self.model.name, continuation_token)
 148                 continuation_token = vsts_users.headers.get("X-MS-ContinuationToken")
 149                 for vsts_user in vsts_users["value"]:
 150                     vsts_email = vsts_user.get(u"mailAddress")
 151                     if vsts_email and vsts_email.lower() in sentry_emails:
 152                         assignee = vsts_user["mailAddress"]
 153                         break
 154 
 155                 if not continuation_token:
 156                     break
 157 
 158             if assignee is None:
 159                 # TODO(lb): Email people when this happens
 160                 self.logger.info(
 161                     "vsts.assignee-not-found",
 162                     extra={
 163                         "integration_id": external_issue.integration_id,
 164                         "user_id": user.id,
 165                         "issue_key": external_issue.key,
 166                     },
 167                 )
 168                 return
 169 
 170         try:
 171             client.update_work_item(self.instance, external_issue.key, assigned_to=assignee)
 172         except (ApiUnauthorized, ApiError):
 173             self.logger.info(
 174                 "vsts.failed-to-assign",
 175                 extra={
 176                     "integration_id": external_issue.integration_id,
 177                     "user_id": user.id,
 178                     "issue_key": external_issue.key,
 179                 },
 180             )

 182     def sync_status_outbound(self, external_issue, is_resolved, project_id, **kwargs):
 183         client = self.get_client()
 184         work_item = client.get_work_item(self.instance, external_issue.key)
 185 
 186         # For some reason, vsts doesn't include the project id
 187         # in the work item response.
 188         # TODO(jess): figure out if there's a better way to do this
 189         vsts_project_name = work_item["fields"]["System.TeamProject"]
 190 
 191         vsts_projects = client.get_projects(self.instance)["value"]
 192 
 193         vsts_project_id = None
 194         for p in vsts_projects:
 195             if p["name"] == vsts_project_name:
 196                 vsts_project_id = p["id"]
 197                 break
 198 
 199         try:
 200             external_project = IntegrationExternalProject.objects.get(
 201                 external_id=vsts_project_id,
 202                 organization_integration_id__in=OrganizationIntegration.objects.filter(
 203                     organization_id=external_issue.organization_id,
 204                     integration_id=external_issue.integration_id,
 205                 ),
 206             )
 207         except IntegrationExternalProject.DoesNotExist:
 208             self.logger.info(
 209                 "vsts.external-project-not-found",
 210                 extra={
 211                     "integration_id": external_issue.integration_id,
 212                     "is_resolved": is_resolved,
 213                     "issue_key": external_issue.key,
 214                 },
 215             )
 216             return
 217 
 218         status = (
 219             external_project.resolved_status if is_resolved else external_project.unresolved_status
 220         )
 221 
 222         try:
 223             client.update_work_item(self.instance, external_issue.key, state=status)
 224         except (ApiUnauthorized, ApiError) as error:
 225             self.logger.info(
 226                 "vsts.failed-to-change-status",
 227                 extra={
 228                     "integration_id": external_issue.integration_id,
 229                     "is_resolved": is_resolved,
 230                     "issue_key": external_issue.key,
 231                     "exception": error,
 232                 },
 233             )

 235     def should_unresolve(self, data):
 236         done_states = self.get_done_states(data["project"])
 237         return (
 238             data["old_state"] in done_states
 239             or data["old_state"] is None
 240             and not data["new_state"] in done_states
 241         )
 242 
 243     def should_resolve(self, data):
 244         done_states = self.get_done_states(data["project"])
 245         return not data["old_state"] in done_states and data["new_state"] in done_states

 247     def get_done_states(self, project):
 248         client = self.get_client()
 249         try:
 250             all_states = client.get_work_item_states(self.instance, project)["value"]
 251         except ApiError as err:
 252             self.logger.info(
 253                 "vsts.get-done-states.failed",
 254                 extra={"integration_id": self.model.id, "exception": err},
 255             )
 256             return []
 257 
 258         done_states = [
 259             state["name"] for state in all_states if state["category"] in self.done_categories
 260         ]
 261         return done_states

 268     def create_comment(self, issue_id, user_id, group_note):
 269         comment = group_note.data["text"]
 270         quoted_comment = self.create_comment_attribution(user_id, comment)
 271         self.get_client().update_work_item(self.instance, issue_id, comment=quoted_comment)

 273     def create_comment_attribution(self, user_id, comment_text):
 274         # VSTS uses markdown or xml
 275         # https://docs.microsoft.com/en-us/microsoftteams/platform/concepts/bots/bots-text-formats
 276         user = User.objects.get(id=user_id)
 277         attribution = "%s wrote:\n\n" % user.name
 278         quoted_comment = "%s<blockquote>%s</blockquote>" % (attribution, comment_text)
 279         return quoted_comment
