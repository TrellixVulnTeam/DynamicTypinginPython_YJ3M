  41 class VstsApiClient(ApiClient, OAuth2RefreshMixin):
  42     api_version = "4.1"
  43     api_version_preview = "-preview.1"
  44     integration_name = "vsts"
  45 
  46     def __init__(self, identity, oauth_redirect_url, *args, **kwargs):
  47         super(VstsApiClient, self).__init__(*args, **kwargs)
  48         self.identity = identity
  49         self.oauth_redirect_url = oauth_redirect_url
  50         if "access_token" not in self.identity.data:
  51             raise ValueError("Vsts Identity missing access token")
  52 
  53     def request(self, method, path, data=None, params=None, api_preview=False, timeout=None):
  54         self.check_auth(redirect_url=self.oauth_redirect_url)
  55         headers = {
  56             "Accept": u"application/json; api-version={}{}".format(
  57                 self.api_version, self.api_version_preview if api_preview else ""
  58             ),
  59             "Content-Type": "application/json-patch+json"
  60             if method == "PATCH"
  61             else "application/json",
  62             "X-HTTP-Method-Override": method,
  63             "X-TFS-FedAuthRedirect": "Suppress",
  64             "Authorization": u"Bearer {}".format(self.identity.data["access_token"]),
  65         }
  66         return self._request(
  67             method, path, headers=headers, data=data, params=params, timeout=timeout
  68         )

  96     def update_work_item(
  97         self,
  98         instance,
  99         id,
 100         title=UNSET,
 101         description=UNSET,
 102         link=UNSET,
 103         comment=UNSET,
 104         assigned_to=UNSET,
 105         state=UNSET,
 106     ):
 107         data = []
 108 
 109         for f_name, f_value in (
 110             ("title", title),
 111             ("description", description),
 112             ("link", link),
 113             ("assigned_to", assigned_to),
 114             ("state", state),
 115         ):
 116             if f_name == "link":
 117                 # XXX: Link is not yet used, as we can't explicitly bind it to Sentry.
 118                 continue
 119             elif f_value is None:
 120                 data.append({"op": "remove", "path": FIELD_MAP[f_name]})
 121             elif f_value is not UNSET:
 122                 data.append(
 123                     {
 124                         # TODO(dcramer): this is problematic when the link already exists
 125                         "op": "replace" if f_name != "link" else "add",
 126                         "path": FIELD_MAP[f_name],
 127                         "value": {"rel": "Hyperlink", "url": f_value}
 128                         if f_name == "link"
 129                         else f_value,
 130                     }
 131                 )
 132 
 133         if comment is not UNSET and comment:
 134             data.append({"op": "add", "path": FIELD_MAP["comment"], "value": comment})
 135 
 136         return self.patch(VstsApiPath.work_items.format(instance=instance, id=id), data=data)
