 114         try:
 115             repos = self.get_client().get_repos(self.instance)

 118         data = []

 121                 {

 253     def update_organization_config(self, data):
 254         if "sync_status_forward" in data:
 255             project_ids_and_statuses = data.pop("sync_status_forward")
 256             if any(
 257                 not mapping["on_unresolve"] or not mapping["on_resolve"]
 258                 for mapping in project_ids_and_statuses.values()
 259             ):
 260                 raise IntegrationError("Resolve and unresolve status are required.")
 261 
 262             data["sync_status_forward"] = bool(project_ids_and_statuses)
 263 
 264             IntegrationExternalProject.objects.filter(
 265                 organization_integration_id=self.org_integration.id
 266             ).delete()
 267 
 268             for project_id, statuses in project_ids_and_statuses.items():
 269                 IntegrationExternalProject.objects.create(
 270                     organization_integration_id=self.org_integration.id,
 271                     external_id=project_id,
 272                     resolved_status=statuses["on_resolve"],
 273                     unresolved_status=statuses["on_unresolve"],
 274                 )
 275 
 276         config = self.org_integration.config
 277         config.update(data)
 278         self.org_integration.update(config=config)

 280     def get_config_data(self):
 281         config = self.org_integration.config
 282         project_mappings = IntegrationExternalProject.objects.filter(
 283             organization_integration_id=self.org_integration.id
 284         )
 285         sync_status_forward = {}
 286         for pm in project_mappings:
 287             sync_status_forward[pm.external_id] = {
 288                 "on_unresolve": pm.unresolved_status,
 289                 "on_resolve": pm.resolved_status,
 290             }
 291         config["sync_status_forward"] = sync_status_forward
 292         return config

 355     def build_integration(self, state):
 356         data = state["identity"]["data"]
 357         oauth_data = self.get_oauth_data(data)
 358         account = state["account"]
 359         user = get_user_info(data["access_token"])
 360         scopes = sorted(VSTSIdentityProvider.oauth_scopes)
 361         base_url = self.get_base_url(data["access_token"], account["accountId"])
 362 
 363         integration = {
 364             "name": account["accountName"],
 365             "external_id": account["accountId"],
 366             "metadata": {"domain_name": base_url, "scopes": scopes},
 367             "user_identity": {
 368                 "type": "vsts",
 369                 "external_id": user["id"],
 370                 "scopes": scopes,
 371                 "data": oauth_data,
 372             },
 373         }
 374 
 375         try:
 376             integration_model = IntegrationModel.objects.get(
 377                 provider="vsts", external_id=account["accountId"], status=ObjectStatus.VISIBLE
 378             )
 379             # preserve previously created subscription information
 380             integration["metadata"]["subscription"] = integration_model.metadata["subscription"]
 381 
 382             assert OrganizationIntegration.objects.filter(
 383                 integration_id=integration_model.id, status=ObjectStatus.VISIBLE
 384             ).exists()
 385 
 386         except (IntegrationModel.DoesNotExist, AssertionError, KeyError):
 387             subscription_id, subscription_secret = self.create_subscription(base_url, oauth_data)
 388             integration["metadata"]["subscription"] = {
 389                 "id": subscription_id,
 390                 "secret": subscription_secret,
 391             }
 392 
 393         return integration

 395     def create_subscription(self, instance, oauth_data):
 396         webhook = WorkItemWebhook()
 397         try:
 398             subscription, shared_secret = webhook.create_subscription(
 399                 instance, oauth_data, self.oauth_redirect_url
 400             )
 401         except ApiError as e:
 402             auth_codes = (400, 401, 403)
 403             permission_error = "permission" in six.text_type(
 404                 e
 405             ) or "not authorized" in six.text_type(e)
 406             if e.code in auth_codes or permission_error:
 407                 raise IntegrationError(
 408                     "You do not have sufficient account access to create webhooks "
 409                     "on the selected Azure DevOps organization.\n"
 410                     "Please check with the owner of this Azure DevOps account."
 411                 )
 412             raise e
 413 
 414         subscription_id = subscription["id"]
 415         return subscription_id, shared_secret

 417     def get_oauth_data(self, payload):
 418         data = {"access_token": payload["access_token"]}
 419 
 420         if "expires_in" in payload:
 421             data["expires"] = int(time()) + int(payload["expires_in"])
 422         if "refresh_token" in payload:
 423             data["refresh_token"] = payload["refresh_token"]
 424         if "token_type" in payload:
 425             data["token_type"] = payload["token_type"]
 426 
 427         return data

 429     @classmethod
 430     def get_base_url(cls, access_token, account_id):
 431         session = http.build_session()
 432         url = VstsIntegrationProvider.VSTS_ACCOUNT_LOOKUP_URL % account_id
 433         response = session.get(
 434             url,
 435             headers={
 436                 "Content-Type": "application/json",
 437                 "Authorization": "Bearer %s" % access_token,
 438             },
 439         )
 440         if response.status_code == 200:
 441             return response.json()["locationUrl"]
 442         return None

 452 class AccountConfigView(PipelineView):
 453     def dispatch(self, request, pipeline):
 454         if "account" in request.POST:
 455             account_id = request.POST.get("account")
 456             accounts = pipeline.fetch_state(key="accounts")
 457             account = self.get_account_from_id(account_id, accounts)
 458             if account is not None:
 459                 state = pipeline.fetch_state(key="identity")
 460                 access_token = state["data"]["access_token"]
 461                 pipeline.bind_state("account", account)
 462                 return pipeline.next_step()
 463 
 464         state = pipeline.fetch_state(key="identity")
 465         access_token = state["data"]["access_token"]
 466         user = get_user_info(access_token)
 467 
 468         accounts = self.get_accounts(access_token, user["uuid"])
 469         logger.info(
 470             "vsts.get_accounts",
 471             extra={
 472                 "organization_id": pipeline.organization.id,
 473                 "user_id": request.user.id,
 474                 "accounts": accounts,
 475             },
 476         )
 477         if not accounts or not accounts.get("value"):
 478             return render_to_response(
 479                 template="sentry/integrations/vsts-config.html",
 480                 context={"no_accounts": True},
 481                 request=request,
 482             )
 483         accounts = accounts["value"]
 484         pipeline.bind_state("accounts", accounts)
 485         account_form = AccountForm(accounts)
 486         return render_to_response(
 487             template="sentry/integrations/vsts-config.html",
 488             context={"form": account_form, "no_accounts": False},
 489             request=request,
 490         )

 492     def get_account_from_id(self, account_id, accounts):
 493         for account in accounts:
 494             if account["accountId"] == account_id:
 495                 return account
 496         return None

 498     def get_accounts(self, access_token, user_id):
 499         session = http.build_session()
 500         url = (
 501             "https://app.vssps.visualstudio.com/_apis/accounts?ownerId=%s&api-version=4.1" % user_id
 502         )
 503         response = session.get(
 504             url,
 505             headers={
 506                 "Content-Type": "application/json",
 507                 "Authorization": "Bearer %s" % access_token,
 508             },
 509         )
 510         if response.status_code == 200:
 511             return response.json()
