  25     def get_repository_data(self, organization, config):
  26         installation = self.get_installation(config.get("installation"), organization.id)
  27         client = installation.get_client()
  28         instance = installation.instance
  29 
  30         repo_id = config["identifier"]
  31 
  32         try:
  33             repo = client.get_repo(instance, repo_id)
  34         except Exception as e:

  37             {
  38                 "instance": instance,
  39                 "project": repo["project"]["name"],
  40                 "name": repo["name"],
  41                 "external_id": six.text_type(repo["id"]),
  42                 "url": repo["_links"]["web"]["href"],
  43             }

  60     def transform_changes(self, patch_set):
  61         type_mapping = {"add": "A", "delete": "D", "edit": "M"}
  62         file_changes = []
  63         # https://docs.microsoft.com/en-us/rest/api/vsts/git/commits/get%20changes#versioncontrolchangetype
  64         for change in patch_set:
  65             change_type = type_mapping.get(change["changeType"])
  66 
  67             if change_type and change.get("item") and change["item"]["gitObjectType"] == "blob":
  68                 file_changes.append({"path": change["item"]["path"], "type": change_type})
  69 
  70         return file_changes

  72     def zip_commit_data(self, repo, commit_list, organization_id):
  73         installation = self.get_installation(repo.integration_id, organization_id)
  74         client = installation.get_client()
  75         n = 0
  76         for commit in commit_list:
  77             # Azure will truncate commit comments to only the first line.
  78             # We need to make an additional API call to get the full commit message.
  79             # This is important because issue refs could be anywhere in the commit
  80             # message.
  81             if commit.get("commentTruncated", False):
  82                 full_commit = client.get_commit(
  83                     repo.config["instance"], repo.external_id, commit["commitId"]
  84                 )
  85                 commit["comment"] = full_commit["comment"]
  86 
  87             commit["patch_set"] = self.transform_changes(
  88                 client.get_commit_filechanges(
  89                     repo.config["instance"], repo.external_id, commit["commitId"]
  90                 )
  91             )
  92             # We only fetch patch data for 90 commits.
  93             n += 1
  94             if n > MAX_COMMIT_DATA_REQUESTS:
  95                 break
  96 
  97         return commit_list

  99     def compare_commits(self, repo, start_sha, end_sha):
 100         installation = self.get_installation(repo.integration_id, repo.organization_id)
 101         client = installation.get_client()
 102         instance = repo.config["instance"]
 103 
 104         try:
 105             if start_sha is None:
 106                 res = client.get_commits(instance, repo.external_id, commit=end_sha, limit=10)
 107             else:
 108                 res = client.get_commit_range(instance, repo.external_id, start_sha, end_sha)
 109         except Exception as e:
 110             installation.raise_error(e)
 111 
 112         commits = self.zip_commit_data(repo, res["value"], repo.organization_id)
 113         return self._format_commits(repo, commits)

 115     def _format_commits(self, repo, commit_list):
 116         return [
 117             {
 118                 "id": c["commitId"],
 119                 "repository": repo.name,
 120                 "author_email": c["author"]["email"],
 121                 "author_name": c["author"]["name"],
 122                 "message": c["comment"],
 123                 "patch_set": c.get("patch_set"),
 124                 "timestamp": self.format_date(c["author"]["date"]),
 125             }
 126             for c in commit_list
 127         ]
