  52 class VercelIntegration(IntegrationInstallation):
  53     def get_client(self):
  54         access_token = self.model.metadata["access_token"]
  55         if self.model.metadata["installation_type"] == "team":
  56             return VercelClient(access_token, self.model.external_id)
  57 
  58         return VercelClient(access_token)
  59 
  60     def get_organization_config(self):
  61         vercel_client = self.get_client()
  62         # TODO: add try/catch if we get API failure
  63         vercel_projects = [
  64             {"value": p["id"], "label": p["name"]} for p in vercel_client.get_projects()
  65         ]
  66 
  67         proj_fields = ["id", "platform", "name", "slug"]
  68         sentry_projects = map(
  69             lambda proj: {key: proj[key] for key in proj_fields},
  70             (
  71                 Project.objects.filter(organization_id=self.organization_id)
  72                 .order_by("slug")
  73                 .values(*proj_fields)
  74             ),
  75         )
  76 
  77         fields = [
  78             {
  79                 "name": "project_mappings",
  80                 "type": "project_mapper",
  81                 "mappedDropdown": {
  82                     "items": vercel_projects,
  83                     "placeholder": "Select a Vercel Project",  # TOOD: add translation
  84                 },
  85                 "sentryProjects": sentry_projects,
  86             }
  87         ]
  88 
  89         return fields

  92 class VercelIntegrationProvider(IntegrationProvider):
  93     key = "vercel"
  94     name = "Vercel"
  95     requires_feature_flag = True
  96     metadata = metadata
  97     integration_cls = VercelIntegration
  98     features = frozenset([IntegrationFeatures.DEPLOYMENT])
  99     oauth_redirect_url = "/extensions/vercel/configure/"
 100 
 101     def get_pipeline_views(self):
 102         identity_pipeline_config = {"redirect_url": absolute_uri(self.oauth_redirect_url)}
 103 
 104         identity_pipeline_view = NestedPipelineView(
 105             bind_key="identity",
 106             provider_key=self.key,
 107             pipeline_cls=IdentityProviderPipeline,
 108             config=identity_pipeline_config,
 109         )
 110 
 111         return [identity_pipeline_view]
 112 
 113     def build_integration(self, state):
 114         data = state["identity"]["data"]
 115         access_token = data["access_token"]
 116         team_id = data.get("team_id")
 117         client = VercelClient(access_token, team_id)
 118 
 119         if team_id:
 120             external_id = team_id
 121             installation_type = "team"
 122             team = client.get_team()
 123             name = team["name"]
 124         else:
 125             external_id = data["user_id"]
 126             installation_type = "user"
 127             user = client.get_user()
 128             name = user["name"]
 129 
 130         try:
 131             webhook = client.create_deploy_webhook()
 132         except ApiError as err:
 133             logger.info(
 134                 "vercel.create_webhook.failed",
 135                 extra={"error": six.text_type(err), "external_id": external_id},
 136             )
 137             try:
 138                 details = err.json["messages"][0].values().pop()
 139             except Exception:
 140                 details = "Unknown Error"
 141             message = u"Could not create deployment webhook in Vercel: {}".format(details)
 142             raise IntegrationError(message)
 143 
 144         integration = {
 145             "name": name,
 146             "external_id": external_id,
 147             "metadata": {
 148                 "access_token": access_token,
 149                 "installation_id": data["installation_id"],
 150                 "installation_type": installation_type,
 151                 "webhook_id": webhook["id"],
 152             },
 153         }
 154 
 155         return integration
