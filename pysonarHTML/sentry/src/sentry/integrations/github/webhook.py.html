  43     def __call__(self, event, host=None):
  44         external_id = event["installation"]["id"]
  45         if host:
  46             external_id = u"{}:{}".format(host, event["installation"]["id"])
  47 
  48         try:
  49             integration = Integration.objects.get(external_id=external_id, provider=self.provider)
  50         except Integration.DoesNotExist:
  51             # It seems possible for the GH or GHE app to be installed on their

  54             # didn't get created in the first place)

  60                     "external_id": six.text_type(external_id),

  63             return
  64

  66

  68 
  69             repos = Repository.objects.filter(
  70                 organization_id__in=orgs.keys(),
  71                 provider="integrations:%s" % self.provider,
  72                 external_id=six.text_type(event["repository"]["id"]),
  73             )

  79         Given a webhook payload, update stored repo data if needed.

  83         """

  85         name_from_event = event["repository"]["full_name"]

 100 class InstallationEventWebhook(Webhook):
 101     # https://developer.github.com/v3/activity/events/types/#installationevent
 102     def __call__(self, event, host=None):
 103         installation = event["installation"]
 104         if installation and event["action"] == "deleted":
 105             external_id = event["installation"]["id"]
 106             if host:
 107                 external_id = u"{}:{}".format(host, event["installation"]["id"])
 108             try:
 109                 integration = Integration.objects.get(
 110                     external_id=external_id, provider=self.provider
 111                 )
 112                 self._handle_delete(event, integration)
 113             except Integration.DoesNotExist:
 114                 # It seems possible for the GH or GHE app to be installed on their
 115                 # end, but the integration to not exist. Possibly from deleting in
 116                 # Sentry first or from a failed install flow (where the integration
 117                 # didn't get created in the first place)
 118                 logger.info(
 119                     "github.deletion-missing-integration",
 120                     extra={
 121                         "action": event["action"],
 122                         "installation_name": event["account"]["login"],
 123                         "external_id": six.text_type(external_id),
 124                     },
 125                 )

 127     def _handle_delete(self, event, integration):
 128 
 129         organizations = integration.organizations.all()
 130 
 131         logger.info(
 132             "InstallationEventWebhook._handle_delete",
 133             extra={
 134                 "external_id": event["installation"]["id"],
 135                 "integration_id": integration.id,
 136                 "organization_id_list": organizations.values_list("id", flat=True),
 137             },
 138         )

 140         integration.update(status=ObjectStatus.DISABLED)
 141 
 142         Repository.objects.filter(
 143             organization_id__in=organizations.values_list("id", flat=True),
 144             provider="integrations:%s" % self.provider,

 155 class PushEventWebhook(Webhook):
 156     # https://developer.github.com/v3/activity/events/types/#pushevent
 157 
 158     def is_anonymous_email(self, email):
 159         return email[-25:] == "@users.noreply.github.com"
 160 
 161     def get_external_id(self, username):
 162         return "github:%s" % username
 163 
 164     def get_idp_external_id(self, integration, host=None):
 165         return options.get("github-app.id")
 166 
 167     def should_ignore_commit(self, commit):
 168         return GitHubRepositoryProvider.should_ignore_commit(commit["message"])
 169 
 170     def _handle(self, integration, event, organization, repo, host=None):
 171         # while we're here, make sure repo data is up to date
 172         self.update_repo_data(repo, event)
 173 
 174         authors = {}
 175         client = integration.get_installation(organization_id=organization.id).get_client()
 176         gh_username_cache = {}
 177 
 178         for commit in event["commits"]:
 179             if not commit["distinct"]:
 180                 continue
 181 
 182             if self.should_ignore_commit(commit):
 183                 continue
 184 
 185             author_email = commit["author"]["email"]
 186             if "@" not in author_email:
 187                 author_email = u"{}@localhost".format(author_email[:65])
 188             # try to figure out who anonymous emails are
 189             elif self.is_anonymous_email(author_email):
 190                 gh_username = commit["author"].get("username")
 191                 # bot users don't have usernames
 192                 if gh_username:
 193                     external_id = self.get_external_id(gh_username)
 194                     if gh_username in gh_username_cache:
 195                         author_email = gh_username_cache[gh_username] or author_email
 196                     else:
 197                         try:
 198                             commit_author = CommitAuthor.objects.get(
 199                                 external_id=external_id, organization_id=organization.id
 200                             )
 201                         except CommitAuthor.DoesNotExist:
 202                             commit_author = None
 203 
 204                         if commit_author is not None and not self.is_anonymous_email(
 205                             commit_author.email
 206                         ):
 207                             author_email = commit_author.email
 208                             gh_username_cache[gh_username] = author_email
 209                         else:
 210                             try:
 211                                 gh_user = client.get_user(gh_username)
 212                             except ApiError as exc:
 213                                 logger.exception(six.text_type(exc))
 214                             else:
 215                                 # even if we can't find a user, set to none so we
 216                                 # don't re-query
 217                                 gh_username_cache[gh_username] = None
 218                                 try:
 219                                     identity = Identity.objects.get(
 220                                         external_id=gh_user["id"],
 221                                         idp__type=self.provider,
 222                                         idp__external_id=self.get_idp_external_id(
 223                                             integration, host
 224                                         ),
 225                                     )
 226                                 except Identity.DoesNotExist:
 227                                     pass
 228                                 else:
 229                                     author_email = identity.user.email
 230                                     gh_username_cache[gh_username] = author_email
 231                                     if commit_author is not None:
 232                                         try:
 233                                             with transaction.atomic():
 234                                                 commit_author.update(
 235                                                     email=author_email, external_id=external_id
 236                                                 )
 237                                         except IntegrityError:
 238                                             pass
 239 
 240                         if commit_author is not None:
 241                             authors[author_email] = commit_author
 242 
 243             # TODO(dcramer): we need to deal with bad values here, but since
 244             # its optional, lets just throw it out for now
 245             if len(author_email) > 75:
 246                 author = None
 247             elif author_email not in authors:
 248                 authors[author_email] = author = CommitAuthor.objects.get_or_create(
 249                     organization_id=organization.id,
 250                     email=author_email,
 251                     defaults={"name": commit["author"]["name"][:128]},
 252                 )[0]
 253 
 254                 update_kwargs = {}
 255 
 256                 if author.name != commit["author"]["name"][:128]:
 257                     update_kwargs["name"] = commit["author"]["name"][:128]
 258 
 259                 gh_username = commit["author"].get("username")
 260                 if gh_username:
 261                     external_id = self.get_external_id(gh_username)
 262                     if author.external_id != external_id and not self.is_anonymous_email(
 263                         author.email
 264                     ):
 265                         update_kwargs["external_id"] = external_id
 266 
 267                 if update_kwargs:
 268                     try:
 269                         with transaction.atomic():
 270                             author.update(**update_kwargs)
 271                     except IntegrityError:
 272                         pass
 273             else:
 274                 author = authors[author_email]
 275 
 276             try:
 277                 with transaction.atomic():
 278                     c = Commit.objects.create(
 279                         repository_id=repo.id,
 280                         organization_id=organization.id,
 281                         key=commit["id"],
 282                         message=commit["message"],
 283                         author=author,
 284                         date_added=dateutil.parser.parse(commit["timestamp"]).astimezone(
 285                             timezone.utc
 286                         ),
 287                     )
 288                     for fname in commit["added"]:
 289                         CommitFileChange.objects.create(
 290                             organization_id=organization.id, commit=c, filename=fname, type="A"
 291                         )
 292                     for fname in commit["removed"]:
 293                         CommitFileChange.objects.create(
 294                             organization_id=organization.id, commit=c, filename=fname, type="D"
 295                         )
 296                     for fname in commit["modified"]:
 297                         CommitFileChange.objects.create(
 298                             organization_id=organization.id, commit=c, filename=fname, type="M"
 299                         )
 300             except IntegrityError:
 301                 pass

 304 class PullRequestEventWebhook(Webhook):
 305     # https://developer.github.com/v3/activity/events/types/#pullrequestevent
 306     def is_anonymous_email(self, email):
 307         return email[-25:] == "@users.noreply.github.com"
 308 
 309     def get_external_id(self, username):
 310         return "github:%s" % username
 311 
 312     def get_idp_external_id(self, integration, host=None):
 313         return options.get("github-app.id")
 314 
 315     def _handle(self, integration, event, organization, repo, host=None):
 316         # while we're here, make sure repo data is up to date
 317         self.update_repo_data(repo, event)
 318 
 319         pull_request = event["pull_request"]
 320         number = pull_request["number"]
 321         title = pull_request["title"]
 322         body = pull_request["body"]
 323         user = pull_request["user"]
 324 
 325         # The value of the merge_commit_sha attribute changes depending on the state of the pull request. Before a pull request is merged, the merge_commit_sha attribute holds the SHA of the test merge commit. After a pull request is merged, the attribute changes depending on how the pull request was merged:
 326         # - If the pull request was merged as a merge commit, the attribute represents the SHA of the merge commit.
 327         # - If the pull request was merged via a squash, the attribute represents the SHA of the squashed commit on the base branch.
 328         # - If the pull request was rebased, the attribute represents the commit that the base branch was updated to.
 329         # https://developer.github.com/v3/pulls/#get-a-single-pull-request
 330         merge_commit_sha = pull_request["merge_commit_sha"] if pull_request["merged"] else None
 331 
 332         author_email = u"{}@localhost".format(user["login"][:65])
 333         try:
 334             commit_author = CommitAuthor.objects.get(
 335                 external_id=self.get_external_id(user["login"]), organization_id=organization.id
 336             )
 337             author_email = commit_author.email
 338         except CommitAuthor.DoesNotExist:
 339             try:
 340                 identity = Identity.objects.get(
 341                     external_id=user["id"],
 342                     idp__type=self.provider,
 343                     idp__external_id=self.get_idp_external_id(integration, host),
 344                 )
 345             except Identity.DoesNotExist:
 346                 pass
 347             else:
 348                 author_email = identity.user.email
 349 
 350         try:
 351             author = CommitAuthor.objects.get(
 352                 organization_id=organization.id, external_id=self.get_external_id(user["login"])
 353             )
 354         except CommitAuthor.DoesNotExist:
 355             try:
 356                 author = CommitAuthor.objects.get(
 357                     organization_id=organization.id, email=author_email
 358                 )
 359             except CommitAuthor.DoesNotExist:
 360                 author = CommitAuthor.objects.create(
 361                     organization_id=organization.id,
 362                     email=author_email,
 363                     external_id=self.get_external_id(user["login"]),
 364                     name=user["login"][:128],
 365                 )
 366 
 367         try:
 368             PullRequest.create_or_save(
 369                 organization_id=organization.id,
 370                 repository_id=repo.id,
 371                 key=number,
 372                 values={
 373                     "organization_id": organization.id,
 374                     "title": title,
 375                     "author": author,
 376                     "message": body,
 377                     "merge_commit_sha": merge_commit_sha,
 378                 },
 379             )
 380         except IntegrityError:
 381             pass

 384 class GitHubWebhookBase(View):
 385     # https://developer.github.com/webhooks/
 386     def get_handler(self, event_type):

 389     def is_valid_signature(self, method, body, secret, signature):
 390         if method == "sha1":
 391             mod = hashlib.sha1
 392         else:
 393             raise NotImplementedError("signature method %s is not supported" % (method,))
 394         expected = hmac.new(key=secret.encode("utf-8"), msg=body, digestmod=mod).hexdigest()
 395         return constant_time_compare(expected, signature)

 400             return HttpResponse(status=405)
 401 
 402         return super(GitHubWebhookBase, self).dispatch(request, *args, **kwargs)

 405         pass

 407     def get_secret(self):
 408         raise NotImplementedError
 409 
 410     def handle(self, request):
 411         secret = self.get_secret()
 412 
 413         if secret is None:
 414             logger.error("github.webhook.missing-secret", extra=self.get_logging_data())
 415             return HttpResponse(status=401)
 416 
 417         body = six.binary_type(request.body)
 418         if not body:
 419             logger.error("github.webhook.missing-body", extra=self.get_logging_data())
 420             return HttpResponse(status=400)
 421 
 422         try:
 423             handler = self.get_handler(request.META["HTTP_X_GITHUB_EVENT"])
 424         except KeyError:
 425             logger.error("github.webhook.missing-event", extra=self.get_logging_data())
 426             return HttpResponse(status=400)
 427 
 428         if not handler:
 429             return HttpResponse(status=204)
 430 
 431         try:
 432             method, signature = request.META["HTTP_X_HUB_SIGNATURE"].split("=", 1)
 433         except (KeyError, IndexError):
 434             logger.error("github.webhook.missing-signature", extra=self.get_logging_data())
 435             return HttpResponse(status=400)
 436 
 437         if not self.is_valid_signature(method, body, self.get_secret(), signature):
 438             logger.error("github.webhook.invalid-signature", extra=self.get_logging_data())
 439             return HttpResponse(status=401)
 440 
 441         try:
 442             event = json.loads(body.decode("utf-8"))
 443         except JSONDecodeError:
 444             logger.error(
 445                 "github.webhook.invalid-json", extra=self.get_logging_data(), exc_info=True
 446             )
 447             return HttpResponse(status=400)
 448 
 449         handler()(event)
 450         return HttpResponse(status=204)
