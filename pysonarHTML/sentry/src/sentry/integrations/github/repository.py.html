  18     def _validate_repo(self, client, installation, repo):
  19         try:
  20             repo_data = client.get_repo(repo)
  21         except Exception as e:
  22             installation.raise_error(e)
  23 
  24         try:
  25             # make sure installation has access to this specific repo
  26             # use hooks endpoint since we explicitly ask for those permissions
  27             # when installing the app (commits can be accessed for public repos)
  28             # https://developer.github.com/v3/repos/hooks/#list-hooks
  29             client.repo_hooks(repo)
  30         except ApiError:
  31             raise IntegrationError(u"You must grant Sentry access to {}".format(repo))
  32 
  33         return repo_data

  55     def compare_commits(self, repo, start_sha, end_sha):
  56         def eval_commits(client):
  57             # use config name because that is kept in sync via webhooks
  58             name = repo.config["name"]
  59             if start_sha is None:
  60                 res = client.get_last_commits(name, end_sha)
  61                 return self._format_commits(client, name, res[:20])
  62             else:
  63                 res = client.compare_commits(name, start_sha, end_sha)
  64                 return self._format_commits(client, name, res["commits"])
  65 
  66         integration_id = repo.integration_id
  67         if integration_id is None:
  68             raise NotImplementedError("GitHub apps requires an integration id to fetch commits")
  69         integration = Integration.objects.get(id=integration_id)
  70         installation = integration.get_installation(repo.organization_id)
  71         client = installation.get_client()
  72 
  73         try:
  74             return eval_commits(client)
  75         except ApiError as e:
  76             if e.code == 404:
  77                 try:
  78                     client.get_token(force_refresh=True)
  79                     return eval_commits(client)
  80                 except Exception as e:
  81                     installation.raise_error(e)
  82             installation.raise_error(e)
  83         except Exception as e:
  84             installation.raise_error(e)

  86     def _format_commits(self, client, repo_name, commit_list):
  87         """Convert GitHub commits into our internal format
  88 
  89         For each commit in the list we have to fetch patch data, as the
  90         compare API gives us all of the files changed in the commit
  91         range but not which files changed in each commit. Without this
  92         we cannot know which specific commit changed a given file.
  93 
  94         See sentry.models.Release.set_commits
  95         """
  96         return [
  97             {
  98                 "id": c["sha"],
  99                 "repository": repo_name,
 100                 "author_email": c["commit"]["author"].get("email"),
 101                 "author_name": c["commit"]["author"].get("name"),
 102                 "message": c["commit"]["message"],
 103                 "timestamp": self.format_date(c["commit"]["author"].get("date")),
 104                 "patch_set": self._get_patchset(client, repo_name, c["sha"]),
 105             }
 106             for c in commit_list
 107         ]

 109     def _get_patchset(self, client, repo_name, sha):
 110         """Get the modified files for a commit
 111         """
 112         commit = client.get_commit(repo_name, sha)
 113         return self._transform_patchset(commit["files"])
 114 
 115     def _transform_patchset(self, diff):
 116         """Convert the patch data from GitHub into our internal format
 117 
 118         See sentry.models.Release.set_commits
 119         """
 120         changes = []
 121         for change in diff:
 122             if change["status"] == "modified":
 123                 changes.append({"path": change["filename"], "type": "M"})
 124             if change["status"] == "added":
 125                 changes.append({"path": change["filename"], "type": "A"})
 126             if change["status"] == "removed":
 127                 changes.append({"path": change["filename"], "type": "D"})
 128             if change["status"] == "renamed":
 129                 changes.append({"path": change["previous_filename"], "type": "D"})
 130                 changes.append({"path": change["filename"], "type": "A"})
 131         return changes
