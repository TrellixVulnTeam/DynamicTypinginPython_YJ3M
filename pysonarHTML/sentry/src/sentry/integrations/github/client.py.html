  18     def get_last_commits(self, repo, end_sha):
  19         # return api request that fetches last ~30 commits
  20         # see https://developer.github.com/v3/repos/commits/#list-commits-on-a-repository
  21         # using end_sha as parameter

  81     def get_token(self, force_refresh=False):
  82         """
  83         Get token retrieves the active access token from the integration model.
  84         Should the token have expired, a new token will be generated and
  85         automatically persisted into the integration.
  86         """
  87         token = self.integration.metadata.get("access_token")
  88         expires_at = self.integration.metadata.get("expires_at")
  89 
  90         if expires_at is not None:
  91             expires_at = datetime.strptime(expires_at, "%Y-%m-%dT%H:%M:%S")
  92 
  93         if not token or expires_at < datetime.utcnow() or force_refresh:
  94             res = self.create_token()
  95             token = res["token"]
  96             expires_at = datetime.strptime(res["expires_at"], "%Y-%m-%dT%H:%M:%SZ")
  97 
  98             self.integration.metadata.update(
  99                 {"access_token": token, "expires_at": expires_at.isoformat()}
 100             )
 101             self.integration.save()
 102 
 103         return token
