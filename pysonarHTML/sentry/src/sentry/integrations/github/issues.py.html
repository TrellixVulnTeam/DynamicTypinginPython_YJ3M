  80     def create_issue(self, data, **kwargs):
  81         client = self.get_client()
  82 
  83         repo = data.get("repo")
  84 
  85         if not repo:
  86             raise IntegrationError("repo kwarg must be provided")
  87 
  88         try:
  89             issue = client.create_issue(
  90                 repo=repo,
  91                 data={
  92                     "title": data["title"],
  93                     "body": data["description"],
  94                     "assignee": data.get("assignee"),
  95                 },
  96             )
  97         except ApiError as e:
  98             raise IntegrationError(self.message_from_error(e))
  99 
 100         return {
 101             "key": issue["number"],
 102             "title": issue["title"],
 103             "description": issue["body"],
 104             "url": issue["html_url"],
 105             "repo": repo,

 151     def get_issue(self, issue_id, **kwargs):
 152         data = kwargs["data"]
 153         repo = data.get("repo")
 154         issue_num = data.get("externalIssue")
 155         client = self.get_client()
 156 
 157         if not repo:
 158             raise IntegrationError("repo must be provided")
 159 
 160         if not issue_num:
 161             raise IntegrationError("issue must be provided")
 162 
 163         try:
 164             issue = client.get_issue(repo, issue_num)
 165         except ApiError as e:
 166             raise IntegrationError(self.message_from_error(e))
 167 
 168         return {
 169             "key": issue["number"],
 170             "title": issue["title"],
 171             "description": issue["body"],
 172             "url": issue["html_url"],
 173             "repo": repo,
 174         }

 176     def get_allowed_assignees(self, repo):
 177         client = self.get_client()
 178         try:
 179             response = client.get_assignees(repo)
 180         except Exception as e:
 181             self.raise_error(e)
 182 
 183         users = tuple((u["login"], u["login"]) for u in response)
 184 
 185         return (("", "Unassigned"),) + users

 187     def get_repo_issues(self, repo):
 188         client = self.get_client()
 189         try:
 190             response = client.get_issues(repo)
 191         except Exception as e:
 192             self.raise_error(e)
 193 
 194         issues = tuple((i["number"], u"#{} {}".format(i["number"], i["title"])) for i in response)
 195 
 196         return issues
