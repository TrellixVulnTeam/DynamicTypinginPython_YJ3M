  88 def build_repository_query(metadata, name, query):
  89     account_type = "user" if metadata["account_type"] == "User" else "org"
  90     return (u"%s:%s %s" % (account_type, name, query)).encode("utf-8")

  93 class GitHubIntegration(IntegrationInstallation, GitHubIssueBasic, RepositoryMixin):
  94     repo_search = True
  95 
  96     def get_client(self):
  97         return GitHubAppsClient(integration=self.model)
  98 
  99     def get_repositories(self, query=None):
 100         if not query:
 101             return [
 102                 {"name": i["name"], "identifier": i["full_name"]}
 103                 for i in self.get_client().get_repositories()
 104             ]
 105 
 106         full_query = build_repository_query(self.model.metadata, self.model.name, query)
 107         response = self.get_client().search_repositories(full_query)
 108         return [
 109             {"name": i["name"], "identifier": i["full_name"]} for i in response.get("items", [])
 110         ]
 111 
 112     def search_issues(self, query):
 113         return self.get_client().search_issues(query)
 114 
 115     def get_unmigratable_repositories(self):
 116         accessible_repos = self.get_repositories()
 117         accessible_repo_names = [r["identifier"] for r in accessible_repos]
 118 
 119         existing_repos = Repository.objects.filter(
 120             organization_id=self.organization_id, provider="github"
 121         )
 122 
 123         return [repo for repo in existing_repos if repo.name not in accessible_repo_names]
 124 
 125     def reinstall(self):
 126         self.reinstall_repositories()
 127 
 128     def message_from_error(self, exc):
 129         if isinstance(exc, ApiError):
 130             message = API_ERRORS.get(exc.code)
 131             if exc.code == 404 and re.search(r"/repos/.*/(compare|commits)", exc.url):
 132                 message += (
 133                     " Please also confirm that the commits associated with the following URL have been pushed to GitHub: %s"
 134                     % exc.url
 135                 )
 136 
 137             if message is None:
 138                 message = exc.json.get("message", "unknown error") if exc.json else "unknown error"
 139             return "Error Communicating with GitHub (HTTP %s): %s" % (exc.code, message)
 140         else:
 141             return ERR_INTERNAL
 142 
 143     def has_repo_access(self, repo):
 144         client = self.get_client()
 145         try:
 146             # make sure installation has access to this specific repo
 147             # use hooks endpoint since we explicitly ask for those permissions
 148             # when installing the app (commits can be accessed for public repos)
 149             # https://developer.github.com/v3/repos/hooks/#list-hooks
 150             client.repo_hooks(repo.config["name"])
 151         except ApiError:
 152             return False
 153         return True

 156 class GitHubIntegrationProvider(IntegrationProvider):
 157     key = "github"
 158     name = "GitHub"
 159     metadata = metadata
 160     integration_cls = GitHubIntegration
 161     features = frozenset([IntegrationFeatures.COMMITS, IntegrationFeatures.ISSUE_BASIC])
 162 
 163     can_disable = True
 164 
 165     setup_dialog_config = {"width": 1030, "height": 1000}
 166 
 167     def post_install(self, integration, organization, extra=None):
 168         repo_ids = Repository.objects.filter(
 169             organization_id=organization.id,
 170             provider__in=["github", "integrations:github"],
 171             integration_id__isnull=True,
 172         ).values_list("id", flat=True)
 173 
 174         for repo_id in repo_ids:
 175             migrate_repo.apply_async(
 176                 kwargs={
 177                     "repo_id": repo_id,
 178                     "integration_id": integration.id,
 179                     "organization_id": organization.id,
 180                 }
 181             )
 182 
 183     def get_pipeline_views(self):
 184         return [GitHubInstallationRedirect()]
 185 
 186     def get_installation_info(self, installation_id):
 187         session = http.build_session()
 188         resp = session.get(
 189             "https://api.github.com/app/installations/%s" % installation_id,
 190             headers={
 191                 "Authorization": "Bearer %s" % get_jwt(),
 192                 "Accept": "application/vnd.github.machine-man-preview+json",
 193             },
 194         )
 195         resp.raise_for_status()
 196         installation_resp = resp.json()
 197 
 198         return installation_resp
 199 
 200     def build_integration(self, state):
 201         installation = self.get_installation_info(state["installation_id"])
 202 
 203         integration = {
 204             "name": installation["account"]["login"],
 205             # TODO(adhiraj): This should be a constant representing the entire github cloud.
 206             "external_id": installation["id"],
 207             # GitHub identity is associated directly to the application, *not*
 208             # to the installation itself.
 209             "idp_external_id": installation["app_id"],
 210             "metadata": {
 211                 # The access token will be populated upon API usage
 212                 "access_token": None,
 213                 "expires_at": None,
 214                 "icon": installation["account"]["avatar_url"],
 215                 "domain_name": installation["account"]["html_url"].replace("https://", ""),
 216                 "account_type": installation["account"]["type"],
 217             },
 218         }
 219 
 220         if state.get("reinstall_id"):
 221             integration["reinstall_id"] = state["reinstall_id"]
 222 
 223         return integration
 224 
 225     def setup(self):
 226         from sentry.plugins.base import bindings
 227 
 228         bindings.add(
 229             "integration-repository.provider", GitHubRepositoryProvider, id="integrations:github"
 230         )

 233 class GitHubInstallationRedirect(PipelineView):
 234     def get_app_url(self):
 235         name = options.get("github-app.name")
 236         return "https://github.com/apps/%s" % slugify(name)
 237 
 238     def dispatch(self, request, pipeline):
 239         if "reinstall_id" in request.GET:
 240             pipeline.bind_state("reinstall_id", request.GET["reinstall_id"])
 241 
 242         if "installation_id" in request.GET:
 243             pipeline.bind_state("installation_id", request.GET["installation_id"])
 244             return pipeline.next_step()
 245 
 246         return self.redirect(self.get_app_url())
