  20 class JiraServerSetupClient(ApiClient):
  21     """
  22     Client for making requests to JiraServer to follow OAuth1 flow.
  23 
  24     Jira OAuth1 docs: https://developer.atlassian.com/server/jira/platform/oauth/
  25     """
  26 
  27     request_token_url = u"{}/plugins/servlet/oauth/request-token"
  28     access_token_url = u"{}/plugins/servlet/oauth/access-token"
  29     authorize_url = u"{}/plugins/servlet/oauth/authorize?oauth_token={}"
  30     integration_name = "jira_server_setup"
  31 
  32     def __init__(self, base_url, consumer_key, private_key, verify_ssl=True):
  33         self.base_url = base_url
  34         self.consumer_key = consumer_key
  35         self.private_key = private_key
  36         self.verify_ssl = verify_ssl
  37 
  38     def get_request_token(self):
  39         """
  40         Step 1 of the oauth flow.
  41         Get a request token that we can have the user verify.
  42         """
  43         url = self.request_token_url.format(self.base_url)
  44         resp = self.post(url, allow_text=True)
  45         return dict(parse_qsl(resp.text))
  46 
  47     def get_authorize_url(self, request_token):
  48         """
  49         Step 2 of the oauth flow.
  50         Get a URL that the user can verify our request token at.
  51         """
  52         return self.authorize_url.format(self.base_url, request_token["oauth_token"])
  53 
  54     def get_access_token(self, request_token, verifier):
  55         """
  56         Step 3 of the oauth flow.
  57         Use the verifier and request token from step 1 to get an access token.
  58         """
  59         if not verifier:
  60             raise ApiError("Missing OAuth token verifier")
  61         auth = OAuth1(
  62             client_key=self.consumer_key,
  63             resource_owner_key=request_token["oauth_token"],
  64             resource_owner_secret=request_token["oauth_token_secret"],
  65             verifier=verifier,
  66             rsa_key=self.private_key,
  67             signature_method=SIGNATURE_RSA,
  68             signature_type="auth_header",
  69         )
  70         url = self.access_token_url.format(self.base_url)
  71         resp = self.post(url, auth=auth, allow_text=True)
  72         return dict(parse_qsl(resp.text))
  73 
  74     def create_issue_webhook(self, external_id, secret, credentials):
  75         auth = OAuth1(
  76             client_key=credentials["consumer_key"],
  77             rsa_key=credentials["private_key"],
  78             resource_owner_key=credentials["access_token"],
  79             resource_owner_secret=credentials["access_token_secret"],
  80             signature_method=SIGNATURE_RSA,
  81             signature_type="auth_header",
  82         )
  83 
  84         # Create a JWT token that we can add to the webhook URL
  85         # so we can locate the matching integration later.
  86         token = jwt.encode({"id": external_id}, secret)
  87         path = reverse("sentry-extensions-jiraserver-issue-updated", kwargs={"token": token})
  88         data = {
  89             "name": "Sentry Issue Sync",
  90             "url": absolute_uri(path),
  91             "events": ["jira:issue_created", "jira:issue_updated"],
  92         }
  93         return self.post("/rest/webhooks/1.0/webhook", auth=auth, data=data)
  94 
  95     def request(self, *args, **kwargs):
  96         """
  97         Add OAuth1 RSA signatures.
  98         """
  99         if "auth" not in kwargs:
 100             kwargs["auth"] = OAuth1(
 101                 client_key=self.consumer_key,
 102                 rsa_key=self.private_key,
 103                 signature_method=SIGNATURE_RSA,
 104                 signature_type="auth_header",
 105             )
 106         return self._request(*args, **kwargs)

 109 class JiraServer(object):
 110     """
 111     Contains the jira-server specifics that a JiraClient needs
 112     in order to communicate with jira
 113 
 114     You can find JIRA REST API docs here:
 115 
 116     https://developer.atlassian.com/server/jira/platform/rest-apis/
 117     """
 118 
 119     def __init__(self, credentials):
 120         self.credentials = credentials
 121 
 122     @property
 123     def cache_prefix(self):
 124         return "sentry-jira-server:"
 125 
 126     def request_hook(self, method, path, data, params, **kwargs):
 127         """
 128         Used by Jira Client to apply the jira-server authentication
 129         Which is RSA signed OAuth1
 130         """
 131         if "auth" not in kwargs:
 132             kwargs["auth"] = OAuth1(
 133                 client_key=self.credentials["consumer_key"],
 134                 rsa_key=self.credentials["private_key"],
 135                 resource_owner_key=self.credentials["access_token"],
 136                 resource_owner_secret=self.credentials["access_token_secret"],
 137                 signature_method=SIGNATURE_RSA,
 138                 signature_type="auth_header",
 139             )
 140 
 141         request_spec = kwargs.copy()
 142         request_spec.update(dict(method=method, path=path, data=data, params=params))
 143         return request_spec
 144 
 145     def user_id_field(self):
 146         """
 147         Jira-Server doesn't require GDPR compliant API usage so we can use `name`
 148         """
 149         return "name"
 150 
 151     def query_field(self):
 152         """
 153         Jira-Server doesn't require GDPR compliant API usage so we can use `username`
 154         """
 155         return "username"
