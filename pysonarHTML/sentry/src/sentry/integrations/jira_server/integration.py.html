  95         widget=forms.CheckboxInput(),
  96         required=False,
  97         initial=True,
  98     )
  99     consumer_key = forms.CharField(
 100         label=_("Jira Consumer Key"),
 101         widget=forms.TextInput(attrs={"placeholder": "sentry-consumer-key"}),
 102     )
 103     private_key = forms.CharField(
 104         label=_("Jira Consumer Private Key"),

 106             attrs={
 107                 "placeholder": "-----BEGIN RSA PRIVATE KEY-----\n...\n-----END RSA PRIVATE KEY-----"
 108             }

 112     def clean_url(self):
 113         """Strip off trailing / as they cause invalid URLs downstream"""
 114         return self.cleaned_data["url"].rstrip("/")
 115 
 116     def clean_private_key(self):
 117         data = self.cleaned_data["private_key"]
 118 
 119         try:
 120             load_pem_private_key(data.encode("utf-8"), None, default_backend())
 121         except Exception:
 122             raise forms.ValidationError(
 123                 "Private key must be a valid SSH private key encoded in a PEM format."
 124             )
 125         return data
 126 
 127     def clean_consumer_key(self):
 128         data = self.cleaned_data["consumer_key"]
 129         if len(data) > 200:
 130             raise forms.ValidationError("Consumer key is limited to 200 characters.")
 131         return data

 134 class InstallationConfigView(PipelineView):
 135     """
 136     Collect the OAuth client credentials from the user.
 137     """
 138 
 139     def dispatch(self, request, pipeline):
 140         if request.method == "POST":
 141             form = InstallationForm(request.POST)
 142             if form.is_valid():
 143                 form_data = form.cleaned_data
 144 
 145                 pipeline.bind_state("installation_data", form_data)
 146                 return pipeline.next_step()
 147         else:
 148             form = InstallationForm()
 149 
 150         return render_to_response(
 151             template="sentry/integrations/jira-server-config.html",
 152             context={"form": form},
 153             request=request,
 154         )

 157 class OAuthLoginView(PipelineView):
 158     """
 159     Start the OAuth dance by creating a request token
 160     and redirecting the user to approve it.
 161     """
 162 
 163     @csrf_exempt
 164     def dispatch(self, request, pipeline):
 165         if "oauth_token" in request.GET:
 166             return pipeline.next_step()
 167 
 168         config = pipeline.fetch_state("installation_data")
 169         client = JiraServerSetupClient(
 170             config.get("url"),
 171             config.get("consumer_key"),
 172             config.get("private_key"),
 173             config.get("verify_ssl"),
 174         )
 175         try:
 176             request_token = client.get_request_token()
 177             pipeline.bind_state("request_token", request_token)
 178             authorize_url = client.get_authorize_url(request_token)
 179 
 180             return self.redirect(authorize_url)
 181         except ApiError as error:
 182             logger.info(
 183                 "identity.jira-server.request-token",
 184                 extra={"url": config.get("url"), "error": error},
 185             )
 186             return pipeline.error("Could not fetch a request token from Jira. %s" % error)

 189 class OAuthCallbackView(PipelineView):
 190     """
 191     Complete the OAuth dance by exchanging our request token
 192     into an access token.
 193     """
 194 
 195     @csrf_exempt
 196     def dispatch(self, request, pipeline):
 197         config = pipeline.fetch_state("installation_data")
 198         client = JiraServerSetupClient(
 199             config.get("url"),
 200             config.get("consumer_key"),
 201             config.get("private_key"),
 202             config.get("verify_ssl"),
 203         )
 204 
 205         try:
 206             access_token = client.get_access_token(
 207                 pipeline.fetch_state("request_token"), request.GET["oauth_token"]
 208             )
 209             pipeline.bind_state("access_token", access_token)
 210 
 211             return pipeline.next_step()
 212         except ApiError as error:
 213             logger.info("identity.jira-server.access-token", extra={"error": error})
 214             return pipeline.error("Could not fetch an access token from Jira")

 217 class JiraServerIntegration(JiraIntegration):
 218     """
 219     IntegrationInstallation implementation for Jira-Server
 220     """
 221 
 222     default_identity = None
 223 
 224     @classproperty
 225     def use_email_scope(cls):
 226         # jira server doesn't need the email scope since it's not restricted by GDPR
 227         return False
 228 
 229     def get_client(self):
 230         if self.default_identity is None:
 231             self.default_identity = self.get_default_identity()
 232 
 233         return JiraServerClient(
 234             self.model.metadata["base_url"],
 235             JiraServer(self.default_identity.data),
 236             self.model.metadata["verify_ssl"],
 237         )
 238 
 239     def get_link_issue_config(self, group, **kwargs):
 240         fields = super(JiraIntegration, self).get_link_issue_config(group, **kwargs)
 241         org = group.organization
 242         autocomplete_url = reverse(
 243             "sentry-extensions-jiraserver-search", args=[org.slug, self.model.id]
 244         )
 245         for field in fields:
 246             if field["name"] == "externalIssue":
 247                 field["url"] = autocomplete_url
 248                 field["type"] = "select"
 249         return fields
 250 
 251     def search_url(self, org_slug):
 252         return reverse("sentry-extensions-jiraserver-search", args=[org_slug, self.model.id])

 255 class JiraServerIntegrationProvider(IntegrationProvider):
 256     key = "jira_server"
 257     name = "Jira Server"
 258     metadata = metadata
 259     integration_cls = JiraServerIntegration
 260 
 261     needs_default_identity = True
 262 
 263     can_add = True
 264 
 265     features = frozenset([IntegrationFeatures.ISSUE_BASIC, IntegrationFeatures.ISSUE_SYNC])
 266 
 267     setup_dialog_config = {"width": 1030, "height": 1000}
 268 
 269     def get_pipeline_views(self):
 270         return [InstallationConfigView(), OAuthLoginView(), OAuthCallbackView()]
 271 
 272     def build_integration(self, state):
 273         install = state["installation_data"]
 274         access_token = state["access_token"]
 275 
 276         webhook_secret = sha1_text(install["private_key"]).hexdigest()
 277 
 278         hostname = urlparse(install["url"]).netloc
 279         external_id = u"{}:{}".format(hostname, install["consumer_key"])[:64]
 280 
 281         credentials = {
 282             "consumer_key": install["consumer_key"],
 283             "private_key": install["private_key"],
 284             "access_token": access_token["oauth_token"],
 285             "access_token_secret": access_token["oauth_token_secret"],
 286         }
 287         # Create the webhook before the integration record exists
 288         # so that if it fails we don't persist a broken integration.
 289         self.create_webhook(external_id, webhook_secret, install, credentials)
 290 
 291         return {
 292             "name": install["consumer_key"],
 293             "provider": "jira_server",
 294             "external_id": external_id,
 295             "metadata": {
 296                 "base_url": install["url"],
 297                 "domain_name": hostname,
 298                 "verify_ssl": install["verify_ssl"],
 299                 "webhook_secret": webhook_secret,
 300             },
 301             "user_identity": {
 302                 "type": "jira_server",
 303                 "external_id": external_id,
 304                 "scopes": [],
 305                 "data": credentials,
 306             },
 307         }
 308 
 309     def create_webhook(self, external_id, webhook_secret, install, credentials):
 310         client = JiraServerSetupClient(
 311             install["url"], install["consumer_key"], install["private_key"], install["verify_ssl"]
 312         )
 313         try:
 314             client.create_issue_webhook(external_id, webhook_secret, credentials)
 315         except ApiError as err:
 316             logger.info(
 317                 "jira-server.webhook.failed",
 318                 extra={"error": six.text_type(err), "external_id": external_id},
 319             )
 320             try:
 321                 details = err.json["messages"][0].values().pop()
 322             except Exception:
 323                 details = ""
 324             message = u"Could not create issue webhook in Jira. {}".format(details)
 325             raise IntegrationError(message)
