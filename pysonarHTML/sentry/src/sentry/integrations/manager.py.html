   7 from sentry.exceptions import NotRegistered

  10 # Ideally this and PluginManager abstracted from the same base, but
  11 # InstanceManager has become convoluted and wasteful
  12 class IntegrationManager(object):
  13     def __init__(self):
  14         self.__values = {}
  15 
  16     def __iter__(self):
  17         return iter(self.all())
  18 
  19     def all(self):
  20         for key in six.iterkeys(self.__values):
  21             integration = self.get(key)
  22             if integration.visible:
  23                 yield integration
  24 
  25     def get(self, key, **kwargs):
  26         try:
  27             cls = self.__values[key]
  28         except KeyError:
  29             raise NotRegistered(key)
  30         return cls(**kwargs)
  31 
  32     def exists(self, key):
  33         return key in self.__values
  34 
  35     def register(self, cls):
  36         self.__values[cls.key] = cls
  37 
  38     def unregister(self, cls):
  39         try:
  40             if self.__values[cls.key] != cls:
  41                 # dont allow unregistering of arbitrary provider
  42                 raise NotRegistered(cls.key)
  43         except KeyError:
  44             # we gracefully handle a missing provider
  45             return
  46         del self.__values[cls.key]
