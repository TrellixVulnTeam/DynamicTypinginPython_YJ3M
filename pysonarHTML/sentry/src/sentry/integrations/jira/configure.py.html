  36 class JiraConfigureView(View):
  37     def get_response(self, context):
  38         context["ac_js_src"] = "https://connect-cdn.atl-paas.net/all.js"
  39         res = render_to_response("sentry/integrations/jira-config.html", context, self.request)
  40         res["X-Frame-Options"] = "ALLOW-FROM %s" % self.request.GET["xdm_e"]
  41         return res

  43     def get(self, request, *args, **kwargs):
  44         return self.handle(request)
  45 
  46     def post(self, request, *args, **kwargs):

  49     def handle(self, request):
  50         try:
  51             integration = get_integration_from_request(request, "jira")
  52         except AtlassianConnectValidationError:
  53             return self.get_response({"error_message": "Unable to verify installation."})
  54         except ExpiredSignatureError:
  55             return self.get_response({"refresh_required": True})
  56 
  57         if not request.user.is_authenticated():
  58             parsed_user_agent = user_agent_parser.ParseUserAgent(
  59                 request.META.get("HTTP_USER_AGENT", "")
  60             )
  61             # not enabling cross site cookies seems to be a common problem with Safari
  62             # as a result, there is a Safari specific link to instructions when is_safari=true
  63             is_safari = parsed_user_agent.get("family") == "Safari"
  64             return self.get_response(
  65                 {
  66                     "login_required": True,
  67                     "is_safari": is_safari,
  68                     "login_url": absolute_uri(reverse("sentry-login")),
  69                 }
  70             )
  71 
  72         organizations = list(
  73             request.user.get_orgs().filter(
  74                 id__in=OrganizationMember.objects.filter(
  75                     role__in=[r.id for r in roles.get_all() if r.is_global], user=request.user
  76                 ).values("organization")
  77             )
  78         )
  79 
  80         form = JiraConfigForm(organizations, request.POST)
  81 
  82         if request.method == "GET" or not form.is_valid():
  83             active_orgs = OrganizationIntegration.objects.filter(
  84                 integration__provider="jira",
  85                 integration=integration,
  86                 organization__in=organizations,
  87             ).values_list("organization_id", flat=True)
  88 
  89             form = JiraConfigForm(organizations, initial={"organizations": active_orgs})
  90             return self.get_response({"form": form, "organizations": organizations})
  91 
  92         enabled_orgs = [o for o in organizations if o.id in form.cleaned_data["organizations"]]
  93         disabled_orgs = list(set(organizations) - set(enabled_orgs))
  94 
  95         # Remove Jira integrations not in the set of enabled organizations
  96         OrganizationIntegration.objects.filter(
  97             integration__provider="jira", integration=integration, organization__in=disabled_orgs
  98         ).delete()
  99 
 100         # Ensure all enabled integrations.
 101         for org in enabled_orgs:
 102             integration.add_organization(org, request.user)
 103 
 104         return self.get_response({"form": form, "completed": True})
