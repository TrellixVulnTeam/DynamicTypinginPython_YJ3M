  21 class JiraCloud(object):
  22     """
  23     Contains the jira-cloud specifics that a JiraClient needs
  24     in order to communicate with jira
  25     """
  26 
  27     def __init__(self, shared_secret):
  28         self.shared_secret = shared_secret
  29 
  30     @property
  31     def cache_prefix(self):
  32         return "sentry-jira-2:"
  33 
  34     def request_hook(self, method, path, data, params, **kwargs):
  35         """
  36         Used by Jira Client to apply the jira-cloud authentication
  37         """
  38         # handle params that are already part of the path
  39         url_params = dict(parse_qs(urlsplit(path).query))
  40         url_params.update(params or {})
  41         path = path.split("?")[0]
  42 
  43         jwt_payload = {
  44             "iss": JIRA_KEY,
  45             "iat": datetime.datetime.utcnow(),
  46             "exp": datetime.datetime.utcnow() + datetime.timedelta(seconds=5 * 60),
  47             "qsh": get_query_hash(path, method.upper(), url_params),
  48         }
  49         encoded_jwt = jwt.encode(jwt_payload, self.shared_secret)
  50         params = dict(jwt=encoded_jwt, **(url_params or {}))
  51         request_spec = kwargs.copy()
  52         request_spec.update(dict(method=method, path=path, data=data, params=params))
  53         return request_spec
  54 
  55     def user_id_field(self):
  56         """
  57         Jira-Cloud requires GDPR compliant API usage so we have to use accountId
  58         """
  59         return "accountId"
  60 
  61     def query_field(self):
  62         """
  63         Jira-Cloud requires GDPR compliant API usage so we have to use query
  64         """
  65         return "query"

  68 class JiraApiClient(ApiClient):
  69     # TODO: Update to v3 endpoints

  71     COMMENT_URL = "/rest/api/2/issue/%s/comment/%s"

  73     CREATE_URL = "/rest/api/2/issue"
  74     ISSUE_URL = "/rest/api/2/issue/%s"
  75     META_URL = "/rest/api/2/issue/createmeta"
  76     PRIORITIES_URL = "/rest/api/2/priority"
  77     PROJECT_URL = "/rest/api/2/project"
  78     SEARCH_URL = "/rest/api/2/search/"
  79     VERSIONS_URL = "/rest/api/2/project/%s/versions"
  80     USERS_URL = "/rest/api/2/user/assignable/search"
  81     SERVER_INFO_URL = "/rest/api/2/serverInfo"
  82     ASSIGN_URL = "/rest/api/2/issue/%s/assignee"
  83     TRANSITION_URL = "/rest/api/2/issue/%s/transitions"
  84     EMAIL_URL = "/rest/api/3/user/email"
  85 
  86     integration_name = "jira"
  87 
  88     # This timeout is completely arbitrary. Jira doesn't give us any
  89     # caching headers to work with. Ideally we want a duration that
  90     # lets the user make their second jira issue with cached data.
  91     cache_time = 240

  93     def __init__(self, base_url, jira_style, verify_ssl, logging_context=None):
  94         self.base_url = base_url
  95         # `jira_style` encapsulates differences between jira server & jira cloud.
  96         # We only support one API version for Jira, but server/cloud require different
  97         # authentication mechanisms and caching.
  98         self.jira_style = jira_style
  99         super(JiraApiClient, self).__init__(verify_ssl, logging_context)
 100 
 101     def get_cache_prefix(self):
 102         return self.jira_style.cache_prefix
 103 
 104     def request(self, method, path, data=None, params=None, **kwargs):
 105         """
 106         Use the request_hook method for our specific style of Jira to
 107         add authentication data and transform parameters.
 108         """
 109         request_spec = self.jira_style.request_hook(method, path, data, params, **kwargs)
 110         if "headers" not in request_spec:
 111             request_spec["headers"] = {}
 112 
 113         # Force adherence to the GDPR compliant API conventions.
 114         # See
 115         # https://developer.atlassian.com/cloud/jira/platform/deprecation-notice-user-privacy-api-migration-guide
 116         request_spec["headers"]["x-atlassian-force-account-id"] = "true"
 117         return self._request(**request_spec)
 118 
 119     def user_id_field(self):
 120         return self.jira_style.user_id_field()
 121 
 122     def query_field(self):
 123         return self.jira_style.query_field()
 124 
 125     def get_issue(self, issue_id):
 126         return self.get(self.ISSUE_URL % (issue_id,))
 127 
 128     def search_issues(self, query):
 129         # check if it looks like an issue id
 130         if ISSUE_KEY_RE.match(query):
 131             jql = 'id="%s"' % query.replace('"', '\\"')
 132         else:
 133             jql = 'text ~ "%s"' % query.replace('"', '\\"')
 134         return self.get(self.SEARCH_URL, params={"jql": jql})
 135 
 136     def create_comment(self, issue_key, comment):
 137         return self.post(self.COMMENTS_URL % issue_key, data={"body": comment})
 138 
 139     def update_comment(self, issue_key, comment_id, comment):
 140         return self.put(self.COMMENT_URL % (issue_key, comment_id), data={"body": comment})
 141 
 142     def get_projects_list(self):
 143         return self.get_cached(self.PROJECT_URL)
 144 
 145     def get_project_key_for_id(self, project_id):
 146         if not project_id:
 147             return ""
 148         projects = self.get_projects_list()
 149         for project in projects:
 150             if project["id"] == project_id:
 151                 return project["key"].encode("utf-8")
 152         return ""
 153 
 154     def get_create_meta_for_project(self, project):
 155         params = {"expand": "projects.issuetypes.fields", "projectIds": project}
 156         metas = self.get_cached(self.META_URL, params=params)
 157         # We saw an empty JSON response come back from the API :(
 158         if not metas:
 159             logger.info(
 160                 "jira.get-create-meta.empty-response",
 161                 extra={"base_url": self.base_url, "project": project},
 162             )
 163             return None
 164 
 165         # XXX(dcramer): document how this is possible, if it even is
 166         if len(metas["projects"]) > 1:
 167             raise ApiError(u"More than one project found matching {}.".format(project))
 168 
 169         try:
 170             return metas["projects"][0]
 171         except IndexError:
 172             logger.info(
 173                 "jira.get-create-meta.key-error",
 174                 extra={"base_url": self.base_url, "project": project},
 175             )
 176             return None

 184     def get_users_for_project(self, project):
 185         # Jira Server wants a project key, while cloud is indifferent.
 186         project_key = self.get_project_key_for_id(project)
 187         return self.get_cached(self.USERS_URL, params={"project": project_key})
 188 
 189     def search_users_for_project(self, project, username):
 190         # Jira Server wants a project key, while cloud is indifferent.
 191         project_key = self.get_project_key_for_id(project)
 192         return self.get_cached(
 193             self.USERS_URL, params={"project": project_key, self.query_field(): username}
 194         )
 195 
 196     def search_users_for_issue(self, issue_key, email):
 197         return self.get_cached(
 198             self.USERS_URL, params={"issueKey": issue_key, self.query_field(): email}
 199         )

 217     def assign_issue(self, key, name_or_account_id):
