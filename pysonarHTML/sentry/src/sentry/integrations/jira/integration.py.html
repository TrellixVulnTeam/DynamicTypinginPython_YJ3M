 194     def update_organization_config(self, data):
 195         """
 196         Update the configuration field for an organization integration.
 197         """
 198         config = self.org_integration.config
 199 
 200         if "sync_status_forward" in data:
 201             project_mappings = data.pop("sync_status_forward")
 202 
 203             if any(
 204                 not mapping["on_unresolve"] or not mapping["on_resolve"]
 205                 for mapping in project_mappings.values()
 206             ):
 207                 raise IntegrationError("Resolve and unresolve status are required.")
 208 
 209             data["sync_status_forward"] = bool(project_mappings)
 210 
 211             IntegrationExternalProject.objects.filter(
 212                 organization_integration_id=self.org_integration.id
 213             ).delete()
 214 
 215             for project_id, statuses in project_mappings.items():
 216                 IntegrationExternalProject.objects.create(
 217                     organization_integration_id=self.org_integration.id,
 218                     external_id=project_id,
 219                     resolved_status=statuses["on_resolve"],
 220                     unresolved_status=statuses["on_unresolve"],
 221                 )
 222 
 223         config.update(data)
 224         self.org_integration.update(config=config)

 332     def make_choices(self, values):
 333         if not values:
 334             return []
 335         results = []
 336         for item in values:
 337             key = item.get("id", None)
 338             if "name" in item:
 339                 value = item["name"]
 340             elif "value" in item:
 341                 # Value based options prefer the value on submit.
 342                 key = item["value"]
 343                 value = item["value"]
 344             elif "label" in item:
 345                 # Label based options prefer the value on submit.
 346                 key = item["label"]
 347                 value = item["label"]
 348             else:
 349                 continue
 350             results.append((key, value))
 351         return results

 376     def build_dynamic_field(self, group, field_meta):
 377         """
 378         Builds a field based on Jira's meta field information
 379         """
 380         schema = field_meta["schema"]
 381 
 382         # set up some defaults for form fields
 383         fieldtype = "text"
 384         fkwargs = {"label": field_meta["name"], "required": field_meta["required"]}
 385         # override defaults based on field configuration
 386         if (
 387             schema["type"] in ["securitylevel", "priority"]
 388             or schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES["select"]
 389         ):
 390             fieldtype = "select"
 391             fkwargs["choices"] = self.make_choices(field_meta.get("allowedValues"))
 392         elif field_meta.get("autoCompleteUrl") and (
 393             schema.get("items") == "user" or schema["type"] == "user"
 394         ):
 395             fieldtype = "select"
 396             fkwargs["url"] = self.search_url(group.organization.slug)
 397             fkwargs["choices"] = []
 398         elif schema["type"] in ["timetracking"]:
 399             # TODO: Implement timetracking (currently unsupported altogether)
 400             return None
 401         elif schema.get("items") in ["worklog", "attachment"]:
 402             # TODO: Implement worklogs and attachments someday
 403             return None
 404         elif schema["type"] == "array" and schema["items"] != "string":
 405             fieldtype = "select"
 406             fkwargs.update(
 407                 {
 408                     "multiple": True,
 409                     "choices": self.make_choices(field_meta.get("allowedValues")),
 410                     "default": "",
 411                 }
 412             )
 413         elif schema["type"] == "option" and len(field_meta.get("allowedValues", [])):
 414             fieldtype = "select"
 415             fkwargs.update(
 416                 {"choices": self.make_choices(field_meta.get("allowedValues")), "default": ""}
 417             )
 418 
 419         # break this out, since multiple field types could additionally
 420         # be configured to use a custom property instead of a default.
 421         if schema.get("custom"):
 422             if schema["custom"] == JIRA_CUSTOM_FIELD_TYPES["textarea"]:
 423                 fieldtype = "textarea"
 424 
 425         fkwargs["type"] = fieldtype
 426         return fkwargs

 428     def get_issue_type_meta(self, issue_type, meta):
 429         issue_types = meta["issuetypes"]
 430         issue_type_meta = None
 431         if issue_type:
 432             matching_type = [t for t in issue_types if t["id"] == issue_type]
 433             issue_type_meta = matching_type[0] if len(matching_type) > 0 else None
 434 
 435         # still no issue type? just use the first one.
 436         if not issue_type_meta:

 441     def get_issue_create_meta(self, client, project_id, jira_projects):
 442         if project_id:
 443             return self.fetch_issue_create_meta(client, project_id)
 444 
 445         # If we don't have a jira projectid, iterate all projects and find
 446         # the first project that has metadata. We only want one project as getting
 447         # all project metadata is expensive and wasteful. In the first run experience,
 448         # the user won't have a 'last used' project id so we need to iterate available
 449         # projects until we find one that we can get metadata for.
 450         attempts = 0
 451         if len(jira_projects):
 452             for fallback in jira_projects:
 453                 attempts += 1
 454                 meta = self.fetch_issue_create_meta(client, fallback["id"])
 455                 if meta:
 456                     logger.info(
 457                         "jira.get-issue-create-meta.attempts",
 458                         extra={"organization_id": self.organization_id, "attempts": attempts},
 459                     )
 460                     return meta
 461 
 462         jira_project_ids = "no projects"
 463         if len(jira_projects):
 464             jira_project_ids = ",".join([project["key"] for project in jira_projects])
 465 
 466         logger.info(
 467             "jira.get-issue-create-meta.no-metadata",
 468             extra={
 469                 "organization_id": self.organization_id,
 470                 "attempts": attempts,
 471                 "jira_projects": jira_project_ids,
 472             },
 473         )
 474         raise IntegrationError(
 475             "Could not get issue create metadata for any Jira projects. "
 476             "Ensure that your project permissions are correct."
 477         )

 506     def get_create_issue_config(self, group, **kwargs):
 507         kwargs["link_referrer"] = "jira_integration"
 508         fields = super(JiraIntegration, self).get_create_issue_config(group, **kwargs)
 509         params = kwargs.get("params", {})
 510 
 511         defaults = self.get_project_defaults(group.project_id)
 512         project_id = params.get("project", defaults.get("project"))
 513         client = self.get_client()
 514         try:
 515             jira_projects = client.get_projects_list()
 516         except ApiError as e:
 517             logger.info(
 518                 "jira.get-create-issue-config.no-projects",
 519                 extra={
 520                     "integration_id": self.model.id,
 521                     "organization_id": self.organization_id,
 522                     "error": six.text_type(e),
 523                 },
 524             )
 525             raise IntegrationError(
 526                 "Could not fetch project list from Jira"
 527                 "Ensure that jira is available and your account is still active."
 528             )
 529 
 530         meta = self.get_issue_create_meta(client, project_id, jira_projects)
 531         if not meta:
 532             raise IntegrationError(
 533                 "Could not fetch issue create metadata from Jira. "
 534                 "Ensure that the integration user has access to the requested project."
 535             )
 536 
 537         # check if the issuetype was passed as a parameter
 538         issue_type = params.get("issuetype", defaults.get("issuetype"))
 539         issue_type_meta = self.get_issue_type_meta(issue_type, meta)
 540         issue_type_choices = self.make_choices(meta["issuetypes"])
 541 
 542         # make sure default issue type is actually
 543         # one that is allowed for project
 544         if issue_type:
 545             if not any((c for c in issue_type_choices if c[0] == issue_type)):
 546                 issue_type = issue_type_meta["id"]
 547 
 548         fields = (
 549             [
 550                 {
 551                     "name": "project",
 552                     "label": "Jira Project",
 553                     "choices": [(p["id"], p["key"]) for p in jira_projects],
 554                     "default": meta["id"],
 555                     "type": "select",
 556                     "updatesForm": True,
 557                 }
 558             ]
 559             + fields
 560             + [
 561                 {
 562                     "name": "issuetype",
 563                     "label": "Issue Type",
 564                     "default": issue_type or issue_type_meta["id"],
 565                     "type": "select",
 566                     "choices": issue_type_choices,
 567                     "updatesForm": True,
 568                 }
 569             ]
 570         )
 571 
 572         # title is renamed to summary before sending to Jira
 573         standard_fields = [f["name"] for f in fields] + ["summary"]
 574 
 575         # TODO(jess): are we going to allow ignored fields?
 576         # ignored_fields = (self.get_option('ignored_fields', group.project) or '').split(',')
 577         ignored_fields = set(
 578             k
 579             for k, v in six.iteritems(issue_type_meta["fields"])
 580             if v["name"] in HIDDEN_ISSUE_FIELDS["names"]
 581         )
 582         ignored_fields.update(HIDDEN_ISSUE_FIELDS["keys"])
 583 
 584         # apply ordering to fields based on some known built-in Jira fields.
 585         # otherwise weird ordering occurs.
 586         anti_gravity = {"priority": -150, "fixVersions": -125, "components": -100, "security": -50}
 587 
 588         dynamic_fields = issue_type_meta["fields"].keys()
 589         dynamic_fields.sort(key=lambda f: anti_gravity.get(f) or 0)
 590 
 591         # build up some dynamic fields based on required shit.
 592         for field in dynamic_fields:
 593             if field in standard_fields or field in [x.strip() for x in ignored_fields]:
 594                 # don't overwrite the fixed fields for the form.
 595                 continue
 596             mb_field = self.build_dynamic_field(group, issue_type_meta["fields"][field])
 597             if mb_field:
 598                 mb_field["name"] = field
 599                 fields.append(mb_field)
 600 
 601         for field in fields:
 602             if field["name"] == "priority":
 603                 # whenever priorities are available, put the available ones in the list.
 604                 # allowedValues for some reason doesn't pass enough info.
 605                 field["choices"] = self.make_choices(client.get_priorities())
 606                 field["default"] = defaults.get("priority", "")
 607             elif field["name"] == "fixVersions":
 608                 field["choices"] = self.make_choices(client.get_versions(meta["key"]))
 609             elif field["name"] == "labels":
 610                 field["default"] = defaults.get("labels", "")
 611 
 612         return fields

 614     def create_issue(self, data, **kwargs):
 615         client = self.get_client()
 616         cleaned_data = {}
 617         # protect against mis-configured integration submitting a form without an
 618         # issuetype assigned.
 619         if not data.get("issuetype"):
 620             raise IntegrationFormError({"issuetype": ["Issue type is required."]})
 621 
 622         jira_project = data.get("project")
 623         if not jira_project:
 624             raise IntegrationFormError({"project": ["Jira project is required"]})
 625 
 626         meta = client.get_create_meta_for_project(jira_project)
 627         if not meta:
 628             raise IntegrationError("Could not fetch issue create configuration from Jira.")
 629 
 630         issue_type_meta = self.get_issue_type_meta(data["issuetype"], meta)
 631         user_id_field = client.user_id_field()
 632 
 633         fs = issue_type_meta["fields"]
 634         for field in fs.keys():
 635             f = fs[field]
 636             if field == "description":
 637                 cleaned_data[field] = data[field]
 638                 continue
 639             elif field == "summary":
 640                 cleaned_data["summary"] = data["title"]
 641                 continue
 642             elif field == "labels" and "labels" in data:
 643                 labels = [label.strip() for label in data["labels"].split(",") if label.strip()]
 644                 cleaned_data["labels"] = labels
 645                 continue
 646             if field in data.keys():
 647                 v = data.get(field)
 648                 if not v:
 649                     continue
 650 
 651                 schema = f.get("schema")
 652                 if schema:
 653                     if schema.get("type") == "string" and not schema.get("custom"):
 654                         cleaned_data[field] = v
 655                         continue
 656                     if schema["type"] == "user" or schema.get("items") == "user":
 657                         v = {user_id_field: v}
 658                     elif schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES.get("multiuserpicker"):
 659                         # custom multi-picker
 660                         v = [{user_id_field: v}]
 661                     elif schema["type"] == "array" and schema.get("items") == "option":
 662                         v = [{"value": vx} for vx in v]
 663                     elif schema["type"] == "array" and schema.get("items") == "string":
 664                         v = [v]
 665                     elif schema["type"] == "array" and schema.get("items") != "string":
 666                         v = [{"id": vx} for vx in v]
 667                     elif schema["type"] == "option":
 668                         v = {"value": v}
 669                     elif schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES.get("textarea"):
 670                         v = v
 671                     elif (
 672                         schema["type"] == "number"
 673                         or schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES["tempo_account"]
 674                     ):
 675                         try:
 676                             if "." in v:
 677                                 v = float(v)
 678                             else:
 679                                 v = int(v)
 680                         except ValueError:
 681                             pass
 682                     elif (
 683                         schema.get("type") != "string"
 684                         or (schema.get("items") and schema.get("items") != "string")
 685                         or schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES.get("select")
 686                     ):
 687                         v = {"id": v}
 688                 cleaned_data[field] = v
 689 
 690         if not (isinstance(cleaned_data["issuetype"], dict) and "id" in cleaned_data["issuetype"]):
 691             # something fishy is going on with this field, working on some Jira
 692             # instances, and some not.
 693             # testing against 5.1.5 and 5.1.4 does not convert (perhaps is no longer included
 694             # in the projectmeta API call, and would normally be converted in the
 695             # above clean method.)
 696             cleaned_data["issuetype"] = {"id": cleaned_data["issuetype"]}
 697 
 698         try:
 699             response = client.create_issue(cleaned_data)
 700         except Exception as e:
 701             self.raise_error(e)
 702 
 703         issue_key = response.get("key")
 704         if not issue_key:
 705             raise IntegrationError("There was an error creating the issue.")
 706 
 707         issue = client.get_issue(issue_key)
 708 
 709         return {
 710             "title": issue["fields"]["summary"],
 711             "description": issue["fields"]["description"],
 712             "key": issue_key,
 713         }

 715     def sync_assignee_outbound(self, external_issue, user, assign=True, **kwargs):
 716         """
 717         Propagate a sentry issue's assignee to a jira issue's assignee
 718         """
 719         client = self.get_client()
 720 
 721         jira_user = None
 722         if assign:
 723             for ue in user.emails.filter(is_verified=True):
 724                 try:
 725                     possible_users = client.search_users_for_issue(external_issue.key, ue.email)
 726                 except (ApiUnauthorized, ApiError):
 727                     continue
 728                 for possible_user in possible_users:
 729                     email = possible_user.get("emailAddress")
 730                     # pull email from API if we can use it
 731                     if not email and self.use_email_scope:
 732                         account_id = possible_user.get("accountId")
 733                         email = client.get_email(account_id)
 734                     # match on lowercase email
 735                     # TODO(steve): add check against display name when JIRA_USE_EMAIL_SCOPE is false
 736                     if email and email.lower() == ue.email.lower():
 737                         jira_user = possible_user
 738                         break
 739             if jira_user is None:
 740                 # TODO(jess): do we want to email people about these types of failures?
 741                 logger.info(
 742                     "jira.assignee-not-found",
 743                     extra={
 744                         "integration_id": external_issue.integration_id,
 745                         "user_id": user.id,
 746                         "issue_key": external_issue.key,
 747                     },
 748                 )
 749                 return
 750 
 751         try:
 752             id_field = client.user_id_field()
 753             client.assign_issue(external_issue.key, jira_user and jira_user.get(id_field))
 754         except (ApiUnauthorized, ApiError):
 755             # TODO(jess): do we want to email people about these types of failures?
 756             logger.info(
 757                 "jira.failed-to-assign",
 758                 extra={
 759                     "organization_id": external_issue.organization_id,
 760                     "integration_id": external_issue.integration_id,
 761                     "user_id": user.id,
 762                     "issue_key": external_issue.key,
 763                 },
 764             )

 766     def sync_status_outbound(self, external_issue, is_resolved, project_id, **kwargs):
 767         """
 768         Propagate a sentry issue's status to a linked issue's status.
 769         """
 770         client = self.get_client()
 771         jira_issue = client.get_issue(external_issue.key)
 772         jira_project = jira_issue["fields"]["project"]
 773 
 774         try:
 775             external_project = IntegrationExternalProject.objects.get(
 776                 external_id=jira_project["id"],
 777                 organization_integration_id__in=OrganizationIntegration.objects.filter(
 778                     organization_id=external_issue.organization_id,
 779                     integration_id=external_issue.integration_id,
 780                 ),
 781             )
 782         except IntegrationExternalProject.DoesNotExist:
 783             return
 784 
 785         jira_status = (
 786             external_project.resolved_status if is_resolved else external_project.unresolved_status
 787         )
 788 
 789         # don't bother updating if it's already the status we'd change it to
 790         if jira_issue["fields"]["status"]["id"] == jira_status:
 791             return
 792 
 793         transitions = client.get_transitions(external_issue.key)
 794 
 795         try:
 796             transition = [t for t in transitions if t["to"]["id"] == jira_status][0]
 797         except IndexError:
 798             # TODO(jess): Email for failure
 799             logger.warning(
 800                 "jira.status-sync-fail",
 801                 extra={
 802                     "organization_id": external_issue.organization_id,
 803                     "integration_id": external_issue.integration_id,
 804                     "issue_key": external_issue.key,
 805                 },
 806             )
 807             return
 808 
 809         client.transition_issue(external_issue.key, transition["id"])
