  21     def post(self, request, *args, **kwargs):
  22         state = request.data
  23         data = JiraIntegrationProvider().build_integration(state)
  24         integration = ensure_integration("jira", data)
  25 
  26         # Sync integration metadata from Jira. This msut be executed *after*
  27         # the integration has been installed on Jira as the access tokens will
  28         # not work until then.
  29         sync_metadata.apply_async(kwargs={"integration_id": integration.id}, countdown=10)
  30 
  31         return self.respond()
