  19 def handle_assignee_change(integration, data, use_email_scope=False):
  20     assignee_changed = any(
  21         item for item in data["changelog"]["items"] if item["field"] == "assignee"
  22     )
  23     if not assignee_changed:
  24         return
  25 
  26     fields = data["issue"]["fields"]
  27 
  28     # if no assignee, assume it was unassigned
  29     assignee = fields.get("assignee")
  30     issue_key = data["issue"]["key"]
  31 
  32     if assignee is None:
  33         sync_group_assignee_inbound(integration, None, issue_key, assign=False)
  34         return
  35     email = assignee.get("emailAddress")
  36     # pull email from API if we can use it
  37     if not email and use_email_scope:
  38         account_id = assignee.get("accountId")
  39         client = JiraApiClient(
  40             integration.metadata["base_url"],
  41             JiraCloud(integration.metadata["shared_secret"]),
  42             verify_ssl=True,
  43         )
  44         email = client.get_email(account_id)
  45 
  46     # TODO(steve) check display name
  47     if not email:
  48         logger.info(
  49             "missing-assignee-email",
  50             extra={"issue_key": issue_key, "integration_id": integration.id},
  51         )
  52         return
  53 
  54     sync_group_assignee_inbound(integration, email, issue_key, assign=True)

  57 def handle_status_change(integration, data):
  58     status_changed = any(item for item in data["changelog"]["items"] if item["field"] == "status")
  59     if not status_changed:
  60         return
  61 
  62     issue_key = data["issue"]["key"]
  63 
  64     try:
  65         changelog = next(item for item in data["changelog"]["items"] if item["field"] == "status")
  66     except StopIteration:
  67         logger.info(
  68             "missing-changelog-status",
  69             extra={"issue_key": issue_key, "integration_id": integration.id},
  70         )
  71         return
  72 
  73     for org_id in integration.organizations.values_list("id", flat=True):
  74         installation = integration.get_installation(org_id)
  75 
  76         installation.sync_status_inbound(
  77             issue_key, {"changelog": changelog, "issue": data["issue"]}
  78         )

  81 class JiraIssueUpdatedWebhook(Endpoint):
  82     authentication_classes = ()
  83     permission_classes = ()
  84 
  85     @csrf_exempt
  86     def dispatch(self, request, *args, **kwargs):
  87         return super(JiraIssueUpdatedWebhook, self).dispatch(request, *args, **kwargs)
  88 
  89     def post(self, request, *args, **kwargs):
  90         try:
  91             token = request.META["HTTP_AUTHORIZATION"].split(" ", 1)[1]
  92         except (KeyError, IndexError):
  93             return self.respond(status=400)
  94 
  95         try:
  96             integration = get_integration_from_jwt(
  97                 token, request.path, "jira", request.GET, method="POST"
  98             )
  99         except AtlassianConnectValidationError:
 100             return self.respond(status=400)
 101 
 102         data = request.data
 103 
 104         if not data.get("changelog"):
 105             logger.info("missing-changelog", extra={"integration_id": integration.id})
 106             return self.respond()
 107 
 108         handle_assignee_change(integration, data, use_email_scope=settings.JIRA_USE_EMAIL_SCOPE)
 109         handle_status_change(integration, data)
 110 
 111         return self.respond()
