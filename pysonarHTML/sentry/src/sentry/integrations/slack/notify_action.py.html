  17 class SlackNotifyServiceForm(forms.Form):
  18     workspace = forms.ChoiceField(choices=(), widget=forms.Select())
  19     channel = forms.CharField(widget=forms.TextInput())
  20     channel_id = forms.HiddenInput()
  21     tags = forms.CharField(required=False, widget=forms.TextInput())
  22 
  23     def __init__(self, *args, **kwargs):
  24         # NOTE: Workspace maps directly to the integration ID
  25         workspace_list = [(i.id, i.name) for i in kwargs.pop("integrations")]
  26         self.channel_transformer = kwargs.pop("channel_transformer")
  27 
  28         super(SlackNotifyServiceForm, self).__init__(*args, **kwargs)
  29 
  30         if workspace_list:
  31             self.fields["workspace"].initial = workspace_list[0][0]
  32 
  33         self.fields["workspace"].choices = workspace_list
  34         self.fields["workspace"].widget.choices = self.fields["workspace"].choices
  35 
  36         # XXX(meredith): When this gets set to True, it lets the RuleSerializer
  37         # know to only save if and when we have the channel_id. The rule will get saved
  38         # in the task (integrations/slack/tasks.py) if the channel_id is found.
  39         self._pending_save = False
  40 
  41     def clean(self):
  42         cleaned_data = super(SlackNotifyServiceForm, self).clean()
  43 
  44         workspace = cleaned_data.get("workspace")
  45         channel = cleaned_data.get("channel", "")
  46 
  47         try:
  48             channel_prefix, channel_id, timed_out = self.channel_transformer(workspace, channel)
  49         except DuplicateDisplayNameError as e:
  50             integration = Integration.objects.get(id=workspace)
  51             domain = integration.metadata["domain_name"]
  52 
  53             params = {"channel": e.message, "domain": domain}
  54 
  55             raise forms.ValidationError(
  56                 _(
  57                     'Multiple users were found with display name "%(channel)s". Please use your username, found at %(domain)s/account/settings.',
  58                 ),
  59                 code="invalid",
  60                 params=params,
  61             )
  62 
  63         channel = strip_channel_name(channel)
  64 
  65         if channel_id is None and timed_out:
  66             cleaned_data["channel"] = channel_prefix + channel
  67             self._pending_save = True
  68             return cleaned_data
  69 
  70         if channel_id is None and workspace is not None:
  71             params = {
  72                 "channel": channel,
  73                 "workspace": dict(self.fields["workspace"].choices).get(int(workspace)),
  74             }
  75 
  76             raise forms.ValidationError(
  77                 _(
  78                     'The slack resource "%(channel)s" does not exist or has not been granted access in the %(workspace)s Slack workspace.'
  79                 ),
  80                 code="invalid",
  81                 params=params,
  82             )
  83 
  84         cleaned_data["channel"] = channel_prefix + channel
  85         cleaned_data["channel_id"] = channel_id
  86 
  87         return cleaned_data

  90 class SlackNotifyServiceAction(EventAction):
  91     form_cls = SlackNotifyServiceForm
  92     label = u"Send a notification to the {workspace} Slack workspace to {channel} and show tags {tags} in notification"
  93     prompt = "Send a Slack notification"
  94 
  95     def __init__(self, *args, **kwargs):
  96         super(SlackNotifyServiceAction, self).__init__(*args, **kwargs)
  97         self.form_fields = {
  98             "workspace": {
  99                 "type": "choice",
 100                 "choices": [(i.id, i.name) for i in self.get_integrations()],
 101             },
 102             "channel": {"type": "string", "placeholder": "i.e #critical"},
 103             "tags": {"type": "string", "placeholder": "i.e environment,user,my_tag"},
 104         }
 105 
 106     def is_enabled(self):
 107         return self.get_integrations().exists()
 108 
 109     def after(self, event, state):
 110         integration_id = self.get_option("workspace")
 111         channel = self.get_option("channel_id")
 112         tags = set(self.get_tags_list())
 113 
 114         try:
 115             integration = Integration.objects.get(
 116                 provider="slack", organizations=self.project.organization, id=integration_id
 117             )
 118         except Integration.DoesNotExist:
 119             # Integration removed, rule still active.
 120             return
 121 
 122         def send_notification(event, futures):
 123             rules = [f.rule for f in futures]
 124             attachment = build_group_attachment(event.group, event=event, tags=tags, rules=rules)
 125 
 126             payload = {
 127                 "token": integration.metadata["access_token"],
 128                 "channel": channel,
 129                 "link_names": 1,
 130                 "attachments": json.dumps([attachment]),
 131             }
 132 
 133             client = SlackClient()
 134             try:
 135                 client.post("/chat.postMessage", data=payload, timeout=5)
 136             except ApiError as e:
 137                 self.logger.info(
 138                     "rule.fail.slack_post",
 139                     extra={
 140                         "error": six.text_type(e),
 141                         "project_id": event.project_id,
 142                         "event_id": event.event_id,
 143                         "channel_name": self.get_option("channel"),
 144                     },
 145                 )
 146 
 147         key = u"slack:{}:{}".format(integration_id, channel)
 148 
 149         metrics.incr("notifications.sent", instance="slack.notification", skip_internal=False)
 150         yield self.future(send_notification, key=key)
 151 
 152     def render_label(self):
 153         try:
 154             integration_name = Integration.objects.get(
 155                 provider="slack",
 156                 organizations=self.project.organization,
 157                 id=self.get_option("workspace"),
 158             ).name
 159         except Integration.DoesNotExist:
 160             integration_name = "[removed]"
 161 
 162         tags = self.get_tags_list()
 163 
 164         return self.label.format(
 165             workspace=integration_name,
 166             channel=self.get_option("channel"),
 167             tags=u"[{}]".format(", ".join(tags)),
 168         )
 169 
 170     def get_tags_list(self):
 171         return [s.strip() for s in self.get_option("tags", "").split(",")]
 172 
 173     def get_integrations(self):
 174         return Integration.objects.filter(provider="slack", organizations=self.project.organization)
 175 
 176     def get_form_instance(self):
 177         return self.form_cls(
 178             self.data, integrations=self.get_integrations(), channel_transformer=self.get_channel_id
 179         )
 180 
 181     def get_channel_id(self, integration_id, name):
 182         return get_channel_id(self.project.organization, integration_id, name)
