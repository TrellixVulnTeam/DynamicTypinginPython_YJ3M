  39     def api_error(self, error, action_type, logging_data):
  40         logging_data = logging_data.copy()
  41         logging_data["response"] = six.text_type(error.body)
  42         logging_data["action_type"] = action_type
  43         logger.info("slack.action.api-error-pre-message: %s" % six.text_type(logging_data))
  44         logger.info("slack.action.api-error", extra=logging_data)
  45         return self.respond(
  46             {
  47                 "response_type": "ephemeral",
  48                 "replace_original": False,
  49                 "text": "Sentry can't perform that action right now on your behalf!",
  50             }
  51         )

  53     def on_assign(self, request, identity, group, action):
  54         assignee = action["selected_options"][0]["value"]
  55 
  56         if assignee == "none":
  57             assignee = None
  58 
  59         self.update_group(group, identity, {"assignedTo": assignee})
  60         analytics.record("integrations.slack.assign", actor_id=identity.user_id)
  61 
  62     def on_status(self, request, identity, group, action, data, integration):
  63         status = action["value"]
  64 
  65         status_data = status.split(":", 1)
  66         status = {"status": status_data[0]}
  67 
  68         resolve_type = status_data[-1]
  69 
  70         if resolve_type == "inNextRelease":
  71             status.update({"statusDetails": {"inNextRelease": True}})
  72         elif resolve_type == "inCurrentRelease":
  73             status.update({"statusDetails": {"inRelease": "latest"}})
  74 
  75         self.update_group(group, identity, status)
  76 
  77         analytics.record(
  78             "integrations.slack.status",
  79             status=status["status"],
  80             resolve_type=resolve_type,
  81             actor_id=identity.user_id,
  82         )

  84     def update_group(self, group, identity, data):
  85         event_write_key = ApiKey(
  86             organization=group.project.organization, scope_list=["event:write"]
  87         )
  88 
  89         return client.put(
  90             path=u"/projects/{}/{}/issues/".format(
  91                 group.project.organization.slug, group.project.slug
  92             ),
  93             params={"id": group.id},
  94             data=data,
  95             user=identity.user,
  96             auth=event_write_key,
  97         )

  99     def open_resolve_dialog(self, data, group, integration):
 100         # XXX(epurkhiser): In order to update the original message we have to
 101         # keep track of the response_url in the callback_id. Definitely hacky,
 102         # but seems like there's no other solutions [1]:
 103         #
 104         # [1]: https://stackoverflow.com/questions/46629852/update-a-bot-message-after-responding-to-a-slack-dialog#comment80795670_46629852
 105         callback_id = json.dumps(
 106             {
 107                 "issue": group.id,
 108                 "orig_response_url": data["response_url"],
 109                 "is_message": self.is_message(data),
 110             }
 111         )
 112 
 113         dialog = {
 114             "callback_id": callback_id,
 115             "title": u"Resolve Issue",
 116             "submit_label": "Resolve",
 117             "elements": [RESOLVE_SELECTOR],
 118         }
 119 
 120         payload = {
 121             "dialog": json.dumps(dialog),
 122             "trigger_id": data["trigger_id"],
 123             "token": integration.metadata["access_token"],
 124         }
 125 
 126         slack_client = SlackClient()
 127         try:
 128             slack_client.post("/dialog.open", data=payload)
 129         except ApiError as e:
 130             logger.error("slack.action.response-error", extra={"error": six.text_type(e)})
 131 
 132     def construct_reply(self, attachment, is_message=False):
 133         # XXX(epurkhiser): Slack is inconsistent about it's expected responses
 134         # for interactive action requests.
 135         #
 136         #  * For _unfurled_ action responses, slack expects the entire
 137         #    attachment body used to replace the unfurled attachment to be at
 138         #    the top level of the json response body.
 139         #
 140         #  * For _bot posted message_ action responses, slack expects the
 141         #    attachment body used to replace the attachment to be within an
 142         #    `attachments` array.
 143         if is_message:
 144             attachment = {"attachments": [attachment]}
 145 
 146         return attachment
 147 
 148     def is_message(self, data):
 149         # XXX(epurkhsier): Used in coordination with construct_reply. Bot
 150         # posted messages will not have the type at all.
 151         return data.get("original_message", {}).get("type") == "message"

 153     @transaction_start("SlackActionEndpoint")
 154     def post(self, request):
 155         logging_data = {}
 156 
 157         try:
 158             slack_request = SlackActionRequest(request)
 159             slack_request.validate()
 160         except SlackRequestError as e:
 161             return self.respond(status=e.status)
 162 
 163         data = slack_request.data
 164 
 165         channel_id = data.get("channel", {}).get("id")
 166         user_id = data.get("user", {}).get("id")
 167 
 168         logging_data["channel_id"] = channel_id
 169         logging_data["slack_user_id"] = user_id
 170 
 171         integration = slack_request.integration
 172         logging_data["integration_id"] = integration.id
 173 
 174         # Determine the issue group action is being taken on
 175         group_id = slack_request.callback_data["issue"]
 176         logging_data["group_id"] = group_id
 177 
 178         # Actions list may be empty when receiving a dialog response
 179         action_list = data.get("actions", [])
 180 
 181         try:
 182             group = Group.objects.select_related("project__organization").get(
 183                 project__in=Project.objects.filter(
 184                     organization__in=integration.organizations.all()
 185                 ),
 186                 id=group_id,
 187             )
 188         except Group.DoesNotExist:
 189             logger.error("slack.action.invalid-issue", extra=logging_data)
 190             return self.respond(status=403)
 191 
 192         logging_data["organization_id"] = group.organization.id
 193 
 194         # Determine the acting user by slack identity
 195         try:
 196             idp = IdentityProvider.objects.get(type="slack", external_id=slack_request.team_id)
 197         except IdentityProvider.DoesNotExist:
 198             logger.error("slack.action.invalid-team-id", extra=logging_data)
 199             return self.respond(status=403)
 200 
 201         try:
 202             identity = Identity.objects.get(idp=idp, external_id=user_id)
 203         except Identity.DoesNotExist:
 204             associate_url = build_linking_url(
 205                 integration, group.organization, user_id, channel_id, data.get("response_url")
 206             )
 207 
 208             return self.respond(
 209                 {
 210                     "response_type": "ephemeral",
 211                     "replace_original": False,
 212                     "text": LINK_IDENTITY_MESSAGE.format(associate_url=associate_url),
 213                 }
 214             )
 215 
 216         # Handle status dialog submission
 217         if slack_request.type == "dialog_submission" and "resolve_type" in data["submission"]:
 218             # Masquerade a status action
 219             action = {"name": "status", "value": data["submission"]["resolve_type"]}
 220 
 221             try:
 222                 self.on_status(request, identity, group, action, data, integration)
 223             except client.ApiError as e:
 224                 return self.api_error(e, "status_dialog", logging_data)
 225 
 226             group = Group.objects.get(id=group.id)
 227             attachment = build_group_attachment(group, identity=identity, actions=[action])
 228 
 229             body = self.construct_reply(
 230                 attachment, is_message=slack_request.callback_data["is_message"]
 231             )
 232 
 233             # use the original response_url to update the link attachment
 234             slack_client = SlackClient()
 235             try:
 236                 slack_client.post(
 237                     slack_request.callback_data["orig_response_url"], data=body, json=True
 238                 )
 239             except ApiError as e:
 240                 logger.error("slack.action.response-error", extra={"error": six.text_type(e)})
 241 
 242             return self.respond()
 243 
 244         # Usually we'll want to respond with the updated attachment including
 245         # the list of actions taken. However, when opening a dialog we do not
 246         # have anything to update the message with and will use the
 247         # response_url later to update it.
 248         defer_attachment_update = False
 249 
 250         # Handle interaction actions
 251         action_type = None
 252         try:
 253             for action in action_list:
 254                 action_type = action["name"]
 255 
 256                 if action_type == "status":
 257                     self.on_status(request, identity, group, action, data, integration)
 258                 elif action_type == "assign":
 259                     self.on_assign(request, identity, group, action)
 260                 elif action_type == "resolve_dialog":
 261                     self.open_resolve_dialog(data, group, integration)
 262                     defer_attachment_update = True
 263         except client.ApiError as e:
 264             return self.api_error(e, action_type, logging_data)
 265 
 266         if defer_attachment_update:
 267             return self.respond()
 268 
 269         # Reload group as it may have been mutated by the action
 270         group = Group.objects.get(id=group.id)
 271 
 272         attachment = build_group_attachment(group, identity=identity, actions=action_list)
 273         body = self.construct_reply(attachment, is_message=self.is_message(data))
 274 
 275         return self.respond(body)
