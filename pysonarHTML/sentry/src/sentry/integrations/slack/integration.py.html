 112     def get_pipeline_views(self):
 113         identity_pipeline_config = {
 114             "oauth_scopes": self.identity_oauth_scopes,
 115             "user_scopes": frozenset(["links:read"]),
 116             "redirect_url": absolute_uri("/extensions/slack/setup/"),

 126         return [SlackReAuthIntro(), SlackReAuthChannels(), identity_pipeline_view]

 128     def get_team_info(self, access_token):
 129         payload = {"token": access_token}
 130 
 131         client = SlackClient()
 132         try:
 133             resp = client.get("/team.info", params=payload)
 134         except ApiError as e:
 135             logger.error("slack.team-info.response-error", extra={"error": six.text_type(e)})
 136             raise IntegrationError("Could not retrieve Slack team information.")
 137 
 138         return resp["team"]

 140     def build_integration(self, state):
 141         data = state["identity"]["data"]
 142         assert data["ok"]
 143 
 144         access_token = data["access_token"]
 145         # bot apps have a different response format
 146         # see: https://api.slack.com/authentication/quickstart#installing
 147         user_id_slack = data["authed_user"]["id"]
 148         team_name = data["team"]["name"]
 149         team_id = data["team"]["id"]
 150 
 151         scopes = sorted(self.identity_oauth_scopes)
 152         team_data = self.get_team_info(access_token)
 153 
 154         metadata = {
 155             "access_token": access_token,
 156             "scopes": scopes,
 157             "icon": team_data["icon"]["image_132"],
 158             "domain_name": team_data["domain"] + ".slack.com",
 159             "installation_type": "born_as_bot",
 160         }
 161 
 162         integration = {
 163             "name": team_name,
 164             "external_id": team_id,
 165             "metadata": metadata,
 166             "user_identity": {
 167                 "type": "slack",
 168                 "external_id": user_id_slack,
 169                 "scopes": [],
 170                 "data": {},
 171             },
 172         }
 173 
 174         # if we have the integration_id then we need to set the
 175         # information for the migration, the user_id and channels
 176         # are using in post_install to send messages to slack
 177         if state.get("integration_id"):
 178             metadata["installation_type"] = "migrated_to_bot"
 179             metadata["migrated_at"] = timezone.now()
 180 
 181             post_install_data = {
 182                 "user_id": state["user_id"],
 183                 "private_channels": state["private_channels"],
 184                 "missing_channels": state["missing_channels"],
 185                 "extra_orgs": state["extra_orgs"],
 186             }
 187 
 188             integration["integration_id"] = state.get("integration_id")
 189             integration["post_install_data"] = post_install_data
 190 
 191         return integration

 219     def post_install(self, integration, organization, extra=None):
 220         # normal installtions don't have extra, quit immediately
 221         if extra is None:
 222             return
 223 
 224         private_channels = extra.get("private_channels")
 225         missing_channels = extra.get("missing_channels")
 226         run_args = {
 227             "integration_id": integration.id,
 228             "organization_id": organization.id,
 229             "user_id": extra.get("user_id"),
 230             "private_channels": private_channels,
 231             "missing_channels": missing_channels,
 232         }
 233         if private_channels or missing_channels:
 234             logger.info("slack.integration.run_post_migration", extra=run_args)
 235             post_migration.run_post_migration.apply_async(kwargs=run_args)
 236         else:
 237             # if we don't have channels, log it so we know we skipped this
 238             logger.info("slack.integration.skipped_post_migration", extra=run_args)

 241 class SlackReAuthIntro(PipelineView):
 242     """
 243         This pipeline step handles rendering the migration
 244         intro with context about the migration.
 245 
 246         If the `integration_id` is not present in the request
 247         then we can fast forward through the pipeline to move
 248         on to installing the integration as normal.
 249 
 250     """
 251 
 252     def dispatch(self, request, pipeline):
 253         integration_id = request.GET.get("integration_id")
 254 
 255         if integration_id:
 256             pipeline.bind_state("integration_id", integration_id)
 257             pipeline.bind_state("user_id", request.user.id)
 258 
 259             try:
 260                 integration = Integration.objects.get(id=integration_id, provider="slack",)
 261             except Integration.DoesNotExist:
 262                 return pipeline.error(IntegrationError("Could not find Slack integration."))
 263 
 264             # We check if there are any other orgs tied to the integration to let the
 265             # user know those organizations will be affected by the migration
 266             extra_orgs = map(
 267                 lambda x: x.slug, integration.organizations.filter(~Q(id=pipeline.organization.id))
 268             )
 269 
 270             pipeline.bind_state("extra_orgs", extra_orgs)
 271 
 272             try:
 273                 all_channels = _get_channels_from_rules(pipeline.organization, integration)
 274             except IntegrationError as error:
 275                 return pipeline.error(error)
 276 
 277             pipeline.bind_state("all_channels", all_channels)
 278 
 279             next_param = "?show_verification_results"
 280 
 281             return render_to_response(
 282                 template="sentry/integrations/slack-reauth-introduction.html",
 283                 context={
 284                     "next_url": "%s%s" % (absolute_uri("/extensions/slack/setup/"), next_param),
 285                     "workspace": integration.name,
 286                     "extra_orgs": extra_orgs,
 287                 },
 288                 request=request,
 289             )
 290 
 291         if "show_verification_results" in request.GET:
 292             return pipeline.next_step()
 293 
 294         # if we dont have the integration_id we dont care about the
 295         # migration path, skip straight to install
 296         return pipeline.next_step(step_size=2)

 299 class SlackReAuthChannels(PipelineView):
 300     """
 301         This pipeline step handles making requests to Slack and
 302         displaying the channels (if any) that are problematic:
 303 
 304         1. private
 305         2. removed
 306         3. unauthorized
 307 
 308         Any private channels in alert rules will also be binded
 309         to the pipeline state to be used later.
 310 
 311     """
 312 
 313     def dispatch(self, request, pipeline):
 314         if "start_migration" in request.GET:
 315             return pipeline.next_step()
 316 
 317         next_url_param = "?start_migration"
 318 
 319         try:
 320             channels = _request_channel_info(pipeline)
 321         except IntegrationError as error:
 322             return pipeline.error(error)
 323 
 324         return render_to_response(
 325             template="sentry/integrations/slack-reauth-details.html",
 326             context={
 327                 "next_url": "%s%s" % (absolute_uri("/extensions/slack/setup/"), next_url_param),
 328                 "private": channels["private"],
 329                 "no_permission": channels["no_permission"],
 330                 "not_found": channels["channel_not_found"],
 331             },
 332             request=request,
 333         )

 336 def _request_channel_info(pipeline):
 337     channels = pipeline.fetch_state("all_channels")
 338     integration_id = pipeline.fetch_state("integration_id")
 339 
 340     try:
 341         integration = Integration.objects.get(id=integration_id, provider="slack",)
 342     except Integration.DoesNotExist:
 343         raise IntegrationError("Could not find Slack integration.")
 344 
 345     channel_responses = defaultdict(lambda: set())
 346     for channel in channels:
 347         payload = {
 348             "token": integration.metadata["access_token"],
 349             "channel": channel["id"],
 350         }
 351         client = SlackClient()
 352         try:
 353             resp = client.post("/conversations.info", data=payload)
 354             # TODO(meredith): subclass the ApiError and make a SlackApiError so we can
 355             # reraise the other errors
 356         except ApiError as e:
 357             logger.info(
 358                 "slack.request_channel_info.response-error",
 359                 extra={
 360                     "error": six.text_type(e),
 361                     "channel": channel["id"],
 362                     "integration_id": integration.id,
 363                 },
 364             )
 365             # regardless of the type of error, we are going to group the channels together
 366             # when showing this to the user to make things a bit simpler
 367             channel_responses["not_found"].add(Channel(channel["name"], channel["id"]))
 368             continue
 369 
 370         if resp["channel"]["is_private"]:
 371             channel_responses["private"].add(Channel(channel["name"], channel["id"]))
 372 
 373     pipeline.bind_state("private_channels", channel_responses["private"])
 374     pipeline.bind_state("missing_channels", channel_responses["not_found"])
 375     return channel_responses

 378 def _get_channels_from_rules(organization, integration):
 379     rules = Rule.objects.filter(
 380         project__in=organization.project_set.all(), status=RuleStatus.ACTIVE,
 381     )
 382 
 383     channels = set()
 384     for rule in rules:
 385         # try and see if its used for slack
 386         for rule_action in rule.data["actions"]:
 387             rule_integration_id = rule_action.get("workspace")
 388             if rule_integration_id and rule_integration_id == six.text_type(integration.id):
 389 
 390                 channel_id = rule_action.get("channel_id")
 391                 channel_name = rule_action["channel"]
 392 
 393                 # skip if its a user or if somehow channel_id is missing
 394                 if channel_name[0] == "@" or not channel_id:
 395                     continue
 396 
 397                 channels.add(Channel(channel_name, channel_id))
 398 
 399     return channels
