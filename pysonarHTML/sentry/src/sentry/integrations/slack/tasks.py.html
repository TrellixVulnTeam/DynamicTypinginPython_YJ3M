  17 class RedisRuleStatus(object):
  18     def __init__(self, uuid=None):
  19         self._uuid = uuid or self._generate_uuid()
  20 
  21         cluster_id = getattr(settings, "SENTRY_RULE_TASK_REDIS_CLUSTER", "default")
  22         self.client = redis_clusters.get(cluster_id)
  23         self._set_inital_value()
  24 
  25     @property
  26     def uuid(self):
  27         return self._uuid
  28 
  29     def set_value(self, status, rule_id=None):
  30         value = self._format_value(status, rule_id)
  31         self.client.set(self._get_redis_key(), u"{}".format(value), ex=60 * 60)
  32 
  33     def get_value(self):
  34         key = self._get_redis_key()
  35         value = self.client.get(key)
  36         return json.loads(value)
  37 
  38     def _generate_uuid(self):
  39         return uuid4().hex
  40 
  41     def _set_inital_value(self):
  42         value = json.dumps({"status": "pending"})
  43         self.client.set(self._get_redis_key(), u"{}".format(value), ex=60 * 60, nx=True)
  44 
  45     def _get_redis_key(self):
  46         return u"slack-channel-task:1:{}".format(self.uuid)
  47 
  48     def _format_value(self, status, rule_id):
  49         value = {"status": status}
  50         if rule_id:
  51             value["rule_id"] = six.text_type(rule_id)
  52         if status == "failed":
  53             value[
  54                 "error"
  55             ] = "The slack resource does not exist or has not been granted access in that workspace."
  56 
  57         return json.dumps(value)

  60 @instrumented_task(name="sentry.integrations.slack.search_channel_id", queue="integrations")
  61 def find_channel_id_for_rule(
  62     name, environment, project, action_match, conditions, actions, frequency, uuid, rule_id=None
  63 ):
  64     redis_rule_status = RedisRuleStatus(uuid)
  65 
  66     try:
  67         project = Project.objects.get(id=project.id)
  68     except Project.DoesNotExist:
  69         redis_rule_status.set_value("failed")
  70         return
  71 
  72     organization = project.organization
  73     integration_id = None
  74     channel_name = None
  75 
  76     for action in actions:
  77         if action.get("workspace") and action.get("channel"):
  78             integration_id = action["workspace"]
  79             # we need to strip the prefix when searching on the channel name
  80             channel_name = strip_channel_name(action["channel"])
  81             break
  82 
  83     try:
  84         integration = Integration.objects.get(
  85             provider="slack", organizations=organization, id=integration_id
  86         )
  87     except Integration.DoesNotExist:
  88         redis_rule_status.set_value("failed")
  89         return
  90 
  91     # we dont' know exactly how long it will take to paginate through all of the slack
  92     # endpoints but need some time limit imposed. 3 minutes should be more than enough time,
  93     # we can always update later
  94     try:
  95         (prefix, item_id, _timed_out) = get_channel_id_with_timeout(
  96             integration, channel_name, 3 * 60
  97         )
  98     except DuplicateDisplayNameError:
  99         # if we find a duplicate display name and nothing else, we
 100         # want to set the status to failed. This just lets us skip
 101         # over the next block and hit the failed status at the end.
 102         item_id = None
 103 
 104     if item_id:
 105         for action in actions:
 106             # need to make sure we are adding back the right prefix and also the channel_id
 107             if action.get("channel") and strip_channel_name(action.get("channel")) == channel_name:
 108                 action["channel"] = prefix + channel_name
 109                 action["channel_id"] = item_id
 110                 break
 111 
 112         kwargs = {
 113             "name": name,
 114             "environment": environment,
 115             "project": project,
 116             "action_match": action_match,
 117             "conditions": conditions,
 118             "actions": actions,
 119             "frequency": frequency,
 120         }
 121 
 122         if rule_id:
 123             rule = Rule.objects.get(id=rule_id)
 124             rule = project_rules.Updater.run(rule=rule, pending_save=False, **kwargs)
 125         else:
 126             rule = project_rules.Creator.run(pending_save=False, **kwargs)
 127 
 128         redis_rule_status.set_value("success", rule.id)
 129         return
 130     # if we never find the channel name we failed :(
 131     redis_rule_status.set_value("failed")
