  66 def get_member_assignees(group):
  67     queryset = (
  68         OrganizationMember.objects.filter(
  69             user__is_active=True,
  70             organization=group.organization,
  71             teams__in=group.project.teams.all(),
  72         )
  73         .distinct()
  74         .select_related("user")
  75     )
  76 
  77     members = sorted(queryset, key=lambda u: u.user.get_display_name())
  78 
  79     return [format_actor_option(u.user) for u in members]

  98 def build_attachment_title(obj):
  99     ev_metadata = obj.get_event_metadata()
 100     ev_type = obj.get_event_type()
 101 
 102     if ev_type == "error" and "type" in ev_metadata:
 103         return ev_metadata["type"]
 104     elif ev_type == "csp":
 105         return u"{} - {}".format(ev_metadata["directive"], ev_metadata["uri"])
 106     else:
 107         return obj.title

 110 def build_attachment_text(group, event=None):
 111     # Group and Event both implement get_event_{type,metadata}
 112     obj = event if event is not None else group
 113     ev_metadata = obj.get_event_metadata()
 114     ev_type = obj.get_event_type()
 115 
 116     if ev_type == "error":
 117         return ev_metadata.get("value") or ev_metadata.get("function")
 118     else:
 119         return None

 122 def build_assigned_text(group, identity, assignee):
 123     actor = Actor.from_actor_identifier(assignee)
 124 
 125     try:
 126         assigned_actor = actor.resolve()
 127     except actor.type.DoesNotExist:
 128         return
 129 
 130     if actor.type == Team:
 131         assignee_text = u"#{}".format(assigned_actor.slug)
 132     elif actor.type == User:
 133         try:
 134             assignee_ident = Identity.objects.get(
 135                 user=assigned_actor, idp__type="slack", idp__external_id=identity.idp.external_id
 136             )
 137             assignee_text = u"<@{}>".format(assignee_ident.external_id)
 138         except Identity.DoesNotExist:
 139             assignee_text = assigned_actor.get_display_name()
 140     else:
 141         raise NotImplementedError
 142 
 143     return u"*Issue assigned to {assignee_text} by <@{user_id}>*".format(
 144         assignee_text=assignee_text, user_id=identity.external_id
 145     )

 148 def build_action_text(group, identity, action):
 149     if action["name"] == "assign":
 150         return build_assigned_text(group, identity, action["selected_options"][0]["value"])
 151 
 152     statuses = {"resolved": "resolved", "ignored": "ignored", "unresolved": "re-opened"}
 153 
 154     # Resolve actions have additional 'parameters' after ':'
 155     status = action["value"].split(":", 1)[0]
 156 
 157     # Action has no valid action text, ignore
 158     if status not in statuses:
 159         return
 160 
 161     return u"*Issue {status} by <@{user_id}>*".format(
 162         status=statuses[status], user_id=identity.external_id
 163     )

 173 def build_group_attachment(group, event=None, tags=None, identity=None, actions=None, rules=None):
 174     # XXX(dcramer): options are limited to 100 choices, even when nested
 175     status = group.get_status()
 176 
 177     members = get_member_assignees(group)
 178     teams = get_team_assignees(group)
 179 
 180     logo_url = absolute_uri(get_asset_url("sentry", "images/sentry-email-avatar.png"))
 181     color = (
 182         LEVEL_TO_COLOR.get(event.get_tag("level"), "error") if event else LEVEL_TO_COLOR["error"]
 183     )
 184 
 185     text = build_attachment_text(group, event) or ""
 186 
 187     if actions is None:
 188         actions = []
 189 
 190     assignee = get_assignee(group)
 191 
 192     resolve_button = {
 193         "name": "resolve_dialog",
 194         "value": "resolve_dialog",
 195         "type": "button",
 196         "text": "Resolve...",
 197     }
 198 
 199     ignore_button = {"name": "status", "value": "ignored", "type": "button", "text": "Ignore"}
 200 
 201     project = Project.objects.get_from_cache(id=group.project_id)
 202 
 203     cache_key = "has_releases:2:%s" % (project.id)
 204     has_releases = cache.get(cache_key)
 205     if has_releases is None:
 206         has_releases = ReleaseProject.objects.filter(project_id=project.id).exists()
 207         if has_releases:
 208             cache.set(cache_key, True, 3600)
 209         else:
 210             cache.set(cache_key, False, 60)
 211 
 212     if not has_releases:
 213         resolve_button.update({"name": "status", "text": "Resolve", "value": "resolved"})
 214 
 215     if status == GroupStatus.RESOLVED:
 216         resolve_button.update({"name": "status", "text": "Unresolve", "value": "unresolved"})
 217 
 218     if status == GroupStatus.IGNORED:
 219         ignore_button.update({"text": "Stop Ignoring", "value": "unresolved"})
 220 
 221     option_groups = []
 222 
 223     if teams:
 224         option_groups.append({"text": "Teams", "options": teams})
 225 
 226     if members:
 227         option_groups.append({"text": "People", "options": members})
 228 
 229     payload_actions = [
 230         resolve_button,
 231         ignore_button,
 232         {
 233             "name": "assign",
 234             "text": "Select Assignee...",
 235             "type": "select",
 236             "selected_options": [assignee],
 237             "option_groups": option_groups,
 238         },
 239     ]
 240 
 241     fields = []
 242 
 243     if tags:
 244         event_tags = event.tags if event else group.get_latest_event().tags
 245 
 246         for key, value in event_tags:
 247             std_key = tagstore.get_standardized_key(key)
 248             if std_key not in tags:
 249                 continue
 250 
 251             labeled_value = tagstore.get_tag_value_label(key, value)
 252             fields.append(
 253                 {
 254                     "title": std_key.encode("utf-8"),
 255                     "value": labeled_value.encode("utf-8"),
 256                     "short": True,
 257                 }
 258             )
 259 
 260     if actions:
 261         action_texts = [_f for _f in [build_action_text(group, identity, a) for a in actions] if _f]
 262         text += "\n" + "\n".join(action_texts)
 263 
 264         color = ACTIONED_ISSUE_COLOR
 265         payload_actions = []
 266 
 267     ts = group.last_seen
 268 
 269     if event:
 270         event_ts = event.datetime
 271         ts = max(ts, event_ts)
 272 
 273     footer = u"{}".format(group.qualified_short_id)
 274 
 275     if rules:
 276         rule_url = build_rule_url(rules[0], group, project)
 277         footer += u" via <{}|{}>".format(rule_url, rules[0].label)
 278 
 279         if len(rules) > 1:
 280             footer += u" (+{} other)".format(len(rules) - 1)
 281 
 282     obj = event if event is not None else group
 283     return {
 284         "fallback": u"[{}] {}".format(project.slug, obj.title),
 285         "title": build_attachment_title(obj),
 286         "title_link": group.get_absolute_url(params={"referrer": "slack"}),
 287         "text": text,
 288         "fields": fields,
 289         "mrkdwn_in": ["text"],
 290         "callback_id": json.dumps({"issue": group.id}),
 291         "footer_icon": logo_url,
 292         "footer": footer,
 293         "ts": to_timestamp(ts),
 294         "color": color,
 295         "actions": payload_actions,
 296     }

 299 def build_incident_attachment(incident, metric_value=None):
 300     """
 301     Builds an incident attachment for slack unfurling
 302     :param incident: The `Incident` to build the attachment for
 303     :param metric_value: The value of the metric that triggered this alert to fire. If
 304     not provided we'll attempt to calculate this ourselves.
 305     :return:
 306     """
 307     logo_url = absolute_uri(get_asset_url("sentry", "images/sentry-email-avatar.png"))
 308     alert_rule = incident.alert_rule
 309 
 310     incident_trigger = (
 311         IncidentTrigger.objects.filter(incident=incident).order_by("-date_modified").first()
 312     )
 313     if incident_trigger:
 314         alert_rule_trigger = incident_trigger.alert_rule_trigger
 315         # TODO: If we're relying on this and expecting possible delays between a trigger fired and this function running,
 316         # then this could actually be incorrect if they changed the trigger's time window in this time period. Should we store it?
 317         start = incident_trigger.date_modified - timedelta(
 318             seconds=alert_rule_trigger.alert_rule.snuba_query.time_window
 319         )
 320         end = incident_trigger.date_modified
 321     else:
 322         start, end = None, None
 323 
 324     if incident.status == IncidentStatus.CLOSED.value:
 325         status = "Resolved"
 326         color = RESOLVED_COLOR
 327     elif incident.status == IncidentStatus.WARNING.value:
 328         status = "Warning"
 329         color = LEVEL_TO_COLOR["warning"]
 330     elif incident.status == IncidentStatus.CRITICAL.value:
 331         status = "Critical"
 332         color = LEVEL_TO_COLOR["fatal"]
 333 
 334     agg_text = QUERY_AGGREGATION_DISPLAY.get(
 335         alert_rule.snuba_query.aggregate, alert_rule.snuba_query.aggregate
 336     )
 337     if metric_value is None:
 338         metric_value = get_incident_aggregates(incident, start, end, use_alert_aggregate=True)[
 339             "count"
 340         ]
 341     time_window = alert_rule.snuba_query.time_window / 60
 342 
 343     text = "{} {} in the last {} minutes".format(metric_value, agg_text, time_window)
 344 
 345     if alert_rule.snuba_query.query != "":
 346         text = text + "\nFilter: {}".format(alert_rule.snuba_query.query)
 347 
 348     ts = incident.date_started
 349 
 350     title = u"{}: {}".format(status, alert_rule.name)
 351 
 352     return {
 353         "fallback": title,
 354         "title": title,
 355         "title_link": absolute_uri(
 356             reverse(
 357                 "sentry-metric-alert",
 358                 kwargs={
 359                     "organization_slug": incident.organization.slug,
 360                     "incident_id": incident.identifier,
 361                 },
 362             )
 363         ),
 364         "text": text,
 365         "fields": [],
 366         "mrkdwn_in": ["text"],
 367         "footer_icon": logo_url,
 368         "footer": "Sentry Incident",
 369         "ts": to_timestamp(ts),
 370         "color": color,
 371         "actions": [],
 372     }

 375 # Different list types in slack that we'll use to resolve a channel name. Format is
 376 # (<list_name>, <result_name>, <prefix>).
 377 LEGACY_LIST_TYPES = [
 378     ("channels", "channels", CHANNEL_PREFIX),
 379     ("groups", "groups", CHANNEL_PREFIX),
 380     ("users", "members", MEMBER_PREFIX),
 381 ]
 382 LIST_TYPES = [("conversations", "channels", CHANNEL_PREFIX), ("users", "members", MEMBER_PREFIX)]

 389 def get_channel_id(organization, integration_id, name):
 390     name = strip_channel_name(name)
 391     try:
 392         integration = Integration.objects.get(
 393             provider="slack", organizations=organization, id=integration_id
 394         )
 395     except Integration.DoesNotExist:
 396         return None
 397 
 398     # XXX(meredith): For large accounts that have many, many channels it's
 399     # possible for us to timeout while attempting to paginate through to find the channel id
 400     # This means some users are unable to create/update alert rules. To avoid this, we attempt
 401     # to find the channel id asynchronously if it takes longer than a certain amount of time,
 402     # which I have set as the SLACK_DEFAULT_TIMEOUT - arbitrarily - to 10 seconds.
 403     return get_channel_id_with_timeout(integration, name, SLACK_DEFAULT_TIMEOUT)

 406 def get_channel_id_with_timeout(integration, name, timeout):
 407     """
 408     Fetches the internal slack id of a channel.
 409     :param organization: The organization that is using this integration
 410     :param integration_id: The integration id of this slack integration
 411     :param name: The name of the channel
 412     :return: a tuple of three values
 413         1. prefix: string (`"#"` or `"@"`)
 414         2. channel_id: string or `None`
 415         3. timed_out: boolean (whether we hit our self-imposed time limit)
 416     """
 417 
 418     token_payload = {"token": integration.metadata["access_token"]}
 419 
 420     # Look for channel ID
 421     payload = dict(token_payload, **{"exclude_archived": False, "exclude_members": True})
 422 
 423     if options.get("slack.legacy-app") is True:
 424         list_types = LEGACY_LIST_TYPES
 425     else:
 426         list_types = LIST_TYPES
 427         payload = dict(payload, **{"types": "public_channel,private_channel"})
 428 
 429     time_to_quit = time.time() + timeout
 430 
 431     client = SlackClient()
 432     id_data = None
 433     found_duplicate = False
 434     for list_type, result_name, prefix in list_types:
 435         cursor = ""
 436         while True:
 437             endpoint = "/%s.list" % list_type
 438             try:
 439                 # Slack limits the response of `<list_type>.list` to 1000 channels
 440                 items = client.get(endpoint, params=dict(payload, cursor=cursor, limit=1000))
 441             except ApiError as e:
 442                 logger.info(
 443                     "rule.slack.%s_list_failed" % list_type, extra={"error": six.text_type(e)}
 444                 )
 445                 return (prefix, None, False)
 446 
 447             for c in items[result_name]:
 448                 # The "name" field is unique (this is the username for users)
 449                 # so we return immediately if we find a match.
 450                 if c["name"] == name:
 451                     return (prefix, c["id"], False)
 452                 # If we don't get a match on a unique identifier, we look through
 453                 # the users' display names, and error if there is a repeat.
 454                 if list_type == "users":
 455                     profile = c.get("profile")
 456                     if profile and profile.get("display_name") == name:
 457                         if id_data:
 458                             found_duplicate = True
 459                         else:
 460                             id_data = (prefix, c["id"], False)
 461 
 462             cursor = items.get("response_metadata", {}).get("next_cursor", None)
 463             if time.time() > time_to_quit:
 464                 return (prefix, None, True)
 465 
 466             if not cursor:
 467                 break
 468         if found_duplicate:
 469             raise DuplicateDisplayNameError(name)
 470         elif id_data:
 471             return id_data
 472 
 473     return (prefix, None, False)

 476 def send_incident_alert_notification(action, incident, metric_value):
 477     channel = action.target_identifier
 478     integration = action.integration
 479     attachment = build_incident_attachment(incident, metric_value)
 480     payload = {
 481         "token": integration.metadata["access_token"],
 482         "channel": channel,
 483         "attachments": json.dumps([attachment]),
 484     }
 485 
 486     client = SlackClient()
 487     try:
 488         client.post("/chat.postMessage", data=payload, timeout=5)
 489     except ApiError as e:
 490         logger.info("rule.fail.slack_post", extra={"error": six.text_type(e)})
