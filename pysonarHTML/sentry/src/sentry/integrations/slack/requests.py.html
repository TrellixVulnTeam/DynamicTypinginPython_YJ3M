  26 class SlackRequest(object):
  27     """
  28     Encapsulation of a request from Slack.
  29 
  30     Action and Event requests share much of the same validation needs and data
  31     access characteristics.
  32 
  33     Raises ``SlackRequestError`` if the request in invalid in some way (the
  34     payload missing, it not being JSON, etc.) ``SlackRequestError`` will also
  35     have the appropriate response code the endpoint should respond with, for
  36     the error that was raised.
  37     """
  38 
  39     def __init__(self, request):
  40         self.request = request
  41         self.integration = None
  42         self._data = {}
  43         self._log_request()
  44 
  45     def validate(self):
  46         """
  47         Ensure everything is present to properly process this request
  48         """
  49         self._authorize()
  50         self._validate_data()
  51         self._validate_integration()
  52 
  53     def is_challenge(self):
  54         return False
  55 
  56     @property
  57     def type(self):
  58         # Found in different places, so this is implemented in each request's
  59         # specific object (``SlackEventRequest`` and ``SlackActionRequest``).
  60         raise NotImplementedError
  61 
  62     @property
  63     def team_id(self):
  64         """
  65         Provide a normalized interface to ``team_id``, which Action and Event
  66         requests provide in different places.
  67         """
  68         return self.data.get("team_id") or self.data.get("team", {}).get("id")
  69 
  70     @property
  71     def data(self):
  72         if not self._data:
  73             self._validate_data()
  74         return self._data
  75 
  76     @property
  77     def logging_data(self):
  78         data = {
  79             "slack_team_id": self.team_id,
  80             "slack_channel_id": self.data.get("channel", {}).get("id"),
  81             "slack_user_id": self.data.get("user", {}).get("id"),
  82             "slack_event_id": self.data.get("event_id"),
  83             "slack_callback_id": self.data.get("callback_id"),
  84             "slack_api_app_id": self.data.get("api_app_id"),
  85         }
  86 
  87         if self.integration:
  88             data["integration_id"] = self.integration.id
  89 
  90         return dict((k, v) for k, v in data.items() if v)
  91 
  92     def _validate_data(self):
  93         try:
  94             self._data = self.request.data
  95         except (ValueError, TypeError):
  96             raise SlackRequestError(status=400)
  97 
  98     def _authorize(self):
  99         # check v1 then v2
 100         signing_secret = options.get("slack.signing-secret")
 101         verification_token = options.get("slack.verification-token")
 102         # for v1, only check the verification_token if we don't have a signing_secret
 103         if signing_secret:
 104             if self._check_signing_secret(signing_secret):
 105                 return
 106         elif verification_token and self._check_verification_token(verification_token):
 107             return
 108         # for v2, only check signing secret
 109         signing_secret = options.get("slack-v2.signing-secret")
 110         if signing_secret and self._check_signing_secret(signing_secret):
 111             return
 112         # unfortunately, we can't know which auth was supposed to succeed
 113         self._error("slack.action.auth")
 114         raise SlackRequestError(status=401)
 115 
 116     def _check_signing_secret(self, signing_secret):
 117         # Taken from: https://github.com/slackapi/python-slack-events-api/blob/master/slackeventsapi/server.py#L47
 118         # Slack docs on this here: https://api.slack.com/authentication/verifying-requests-from-slack#about
 119         signature = self.request.META["HTTP_X_SLACK_SIGNATURE"]
 120         timestamp = self.request.META["HTTP_X_SLACK_REQUEST_TIMESTAMP"]
 121 
 122         req = six.binary_type("v0:%s:%s" % (timestamp, self.request.body))
 123         request_hash = "v0=" + hmac.new(six.binary_type(signing_secret), req, sha256).hexdigest()
 124         return hmac.compare_digest(six.binary_type(request_hash), six.binary_type(signature))
 125 
 126     def _check_verification_token(self, verification_token):
 127         return self.data.get("token") == verification_token
 128 
 129     def _validate_integration(self):
 130         try:
 131             self.integration = Integration.objects.get(provider="slack", external_id=self.team_id)
 132         except Integration.DoesNotExist:
 133             self._error("slack.action.invalid-team-id")
 134             raise SlackRequestError(status=403)
 135 
 136     def _log_request(self):
 137         self._info("slack.request")
 138 
 139     def _error(self, key):
 140         logger.error(key, extra=self.logging_data)
 141 
 142     def _info(self, key):
 143         logger.info(key, extra=self.logging_data)

 146 class SlackEventRequest(SlackRequest):
 147     """
 148     An Event request sent from Slack.
 149 
 150     These requests require the same Data and Token validation as all other
 151     requests from Slack, but also event data validation.
 152 
 153     Challenge Requests
 154     ------------------
 155     Slack Event requests first start with a "challenge request". This is just a
 156     request Sentry needs to verifying using it's shared key.
 157 
 158     Challenge requests will have a ``type`` of ``url_verification``.
 159     """
 160 
 161     def validate(self):
 162         if self.is_challenge():
 163             # Challenge requests only include the Token and data to verify the
 164             # request, so only validate those.
 165             self._authorize()
 166             self._validate_data()
 167         else:
 168             # Non-Challenge requests need to validate everything plus the data
 169             # about the event.
 170             super(SlackEventRequest, self).validate()
 171             self._validate_event()
 172 
 173     def is_challenge(self):
 174         return self.data.get("type") == "url_verification"
 175 
 176     @property
 177     def type(self):
 178         return self.data.get("event", {}).get("type")
 179 
 180     def _validate_event(self):
 181         if not self.data.get("event"):
 182             self._error("slack.event.invalid-event-data")
 183             raise SlackRequestError(status=400)
 184 
 185         if not self.data.get("event", {}).get("type"):
 186             self._error("slack.event.invalid-event-type")
 187             raise SlackRequestError(status=400)
 188 
 189     def _log_request(self):
 190         self._info(u"slack.event.{}".format(self.type))

 193 class SlackActionRequest(SlackRequest):
 194     """
 195     An Action request sent from Slack.
 196 
 197     Action requests nest their data inside of a ``payload`` key in the request
 198     body, for some reason. Therefor they require an extra bit of data
 199     validation.
 200     """
 201 
 202     def __init__(self, request):
 203         super(SlackActionRequest, self).__init__(request)
 204         self._callback_data = None
 205 
 206     @property
 207     def type(self):
 208         return self.data.get("type")
 209 
 210     @memoize
 211     def callback_data(self):
 212         """
 213         We store certain data in ``callback_id`` as JSON. It's a bit hacky, but
 214         it's the simplest way to store state without saving it on the Sentry
 215         side.
 216 
 217         Data included in this field:
 218             - issue: the ID of the corresponding Issue
 219             - orig_response_url: URL from the original message we received
 220             - is_message: did the original message have a 'message' type
 221         """
 222         return json.loads(self.data.get("callback_id"))
 223 
 224     def _validate_data(self):
 225         """
 226         Action requests provide the body of the request differently than Event
 227         requests (nested in a ``payload`` attribute), so there's extra
 228         validation needed.
 229         """
 230         super(SlackActionRequest, self)._validate_data()
 231 
 232         if "payload" not in self.request.data:
 233             raise SlackRequestError(status=400)
 234 
 235         try:
 236             self._data = json.loads(self.data["payload"])
 237         except (KeyError, IndexError, TypeError, ValueError):
 238             raise SlackRequestError(status=400)
 239 
 240     def _log_request(self):
 241         self._info("slack.action")
