  20 # XXX(dcramer): this could be more tightly bound to our configured domain,
  21 # but slack limits what we can unfurl anyways so its probably safe
  22 _link_regexp = re.compile(r"^https?\://[^/]+/[^/]+/[^/]+/(issues|incidents)/(\d+)")
  23 _org_slug_regexp = re.compile(r"^https?\://[^/]+/organizations/([^/]+)/")

  26 def unfurl_issues(integration, issue_map):
  27     results = {
  28         g.id: g
  29         for g in Group.objects.filter(
  30             id__in=set(issue_map.keys()),
  31             project__in=Project.objects.filter(organization__in=integration.organizations.all()),
  32         )
  33     }
  34     if not results:
  35         return {}
  36 
  37     return {
  38         v: build_group_attachment(results[k]) for k, v in six.iteritems(issue_map) if k in results
  39     }

  42 def unfurl_incidents(integration, incident_map):
  43     filter_query = Q()
  44     # Since we don't have real ids here, we have to also extract the org slug
  45     # from the url so that we can make sure the identifiers correspond to the
  46     # correct organization.
  47     for identifier, url in six.iteritems(incident_map):
  48         org_slug = _org_slug_regexp.match(url).group(1)
  49         filter_query |= Q(identifier=identifier, organization__slug=org_slug)
  50 
  51     results = {
  52         i.identifier: i
  53         for i in Incident.objects.filter(
  54             filter_query,
  55             # Filter by integration organization here as well to make sure that
  56             # we have permission to access these incidents.
  57             organization__in=integration.organizations.all(),
  58         )
  59     }
  60     if not results:
  61         return {}
  62 
  63     return {
  64         v: build_incident_attachment(results[k])
  65         for k, v in six.iteritems(incident_map)
  66         if k in results
  67     }

  72 class SlackEventEndpoint(Endpoint):
  73     event_handlers = {"issues": unfurl_issues, "incidents": unfurl_incidents}
  74 
  75     authentication_classes = ()
  76     permission_classes = ()

  78     def _parse_url(self, link):
  79         """
  80         Extracts event type and id from a url.
  81         :param link: Url to parse to information from
  82         :return: If successful, a tuple containing the event_type and id. If we
  83         were unsuccessful at matching, a tuple containing two None values
  84         """
  85         match = _link_regexp.match(link)
  86         if not match:
  87             return None, None
  88         try:
  89             return match.group(1), int(match.group(2))
  90         except (TypeError, ValueError):
  91             return None, None
  92 
  93     def on_url_verification(self, request, data):
  94         return self.respond({"challenge": data["challenge"]})
  95 
  96     def on_link_shared(self, request, integration, token, data):
  97         parsed_events = defaultdict(dict)
  98         for item in data["links"]:
  99             event_type, instance_id = self._parse_url(item["url"])
 100             if not instance_id:
 101                 continue
 102             parsed_events[event_type][instance_id] = item["url"]
 103 
 104         if not parsed_events:
 105             return
 106 
 107         results = {}
 108         for event_type, instance_map in parsed_events.items():
 109             results.update(self.event_handlers[event_type](integration, instance_map))
 110 
 111         if not results:
 112             return
 113 
 114         # the classic bot tokens must use the user auth token for URL unfurling
 115         # we stored the user_access_token there
 116         # but for workspace apps and new slack bot tokens, we can just use access_token
 117         access_token = integration.metadata.get("user_access_token")
 118         if not access_token:
 119             access_token = integration.metadata["access_token"]
 120 
 121         payload = {
 122             "token": access_token,
 123             "channel": data["channel"],
 124             "ts": data["message_ts"],
 125             "unfurls": json.dumps(results),
 126         }
 127 
 128         client = SlackClient()
 129         try:
 130             client.post("/chat.unfurl", data=payload)
 131         except ApiError as e:
 132             logger.error("slack.event.unfurl-error", extra={"error": six.text_type(e)})
 133 
 134         return self.respond()

 136     # TODO(dcramer): implement app_uninstalled and tokens_revoked
 137     @transaction_start("SlackEventEndpoint")
 138     def post(self, request):
 139         try:
 140             slack_request = SlackEventRequest(request)
 141             slack_request.validate()
 142         except SlackRequestError as e:
 143             return self.respond(status=e.status)
 144 
 145         if slack_request.is_challenge():
 146             return self.on_url_verification(request, slack_request.data)
 147 
 148         if slack_request.type == "link_shared":
 149             resp = self.on_link_shared(
 150                 request,
 151                 slack_request.integration,
 152                 slack_request.data.get("token"),
 153                 slack_request.data.get("event"),
 154             )
 155 
 156             if resp:
 157                 return resp
 158 
 159         return self.respond()
