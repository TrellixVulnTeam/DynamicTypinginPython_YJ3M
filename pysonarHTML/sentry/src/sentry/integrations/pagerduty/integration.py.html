  63 class PagerDutyIntegration(IntegrationInstallation):
  64     def get_client(self, integration_key):
  65         return PagerDutyClient(integration_key=integration_key)
  66 
  67     def get_organization_config(self):
  68         fields = [
  69             {
  70                 "name": "service_table",
  71                 "type": "table",
  72                 "label": "PagerDuty services with the Sentry integration enabled",
  73                 "help": "If services need to be updated, deleted, or added manually please do so here. Alert rules will need to be individually updated for any additions or deletions of services.",
  74                 "addButtonText": "",
  75                 "columnLabels": {"service": "Service", "integration_key": "Integration Key"},
  76                 "columnKeys": ["service", "integration_key"],
  77                 "confirmDeleteMessage": "Any alert rules associated with this service will stop working. The rules will still exist but will show a `removed` service.",
  78             }
  79         ]
  80 
  81         return fields
  82 
  83     def update_organization_config(self, data):
  84         if "service_table" in data:
  85             service_rows = data["service_table"]
  86             # validate fields
  87             bad_rows = filter(lambda x: not x["service"] or not x["integration_key"], service_rows)
  88             if bad_rows:
  89                 raise IntegrationError("Name and key are required")
  90 
  91             with transaction.atomic():
  92                 exising_service_items = PagerDutyService.objects.filter(
  93                     organization_integration=self.org_integration
  94                 )
  95 
  96                 for service_item in exising_service_items:
  97                     # find the matching row from the input
  98                     matched_rows = filter(lambda x: x["id"] == service_item.id, service_rows)
  99                     if matched_rows:
 100                         matched_row = matched_rows[0]
 101                         service_item.integration_key = matched_row["integration_key"]
 102                         service_item.service_name = matched_row["service"]
 103                         service_item.save()
 104                     else:
 105                         service_item.delete()
 106 
 107                 # new rows don't have an id
 108                 new_rows = filter(lambda x: not x["id"], service_rows)
 109                 for row in new_rows:
 110                     service_name = row["service"]
 111                     key = row["integration_key"]
 112                     PagerDutyService.objects.create(
 113                         organization_integration=self.org_integration,
 114                         service_name=service_name,
 115                         integration_key=key,
 116                     )
 117 
 118     def get_config_data(self):
 119         service_list = []
 120         for s in self.services:
 121             service_list.append(
 122                 {"service": s.service_name, "integration_key": s.integration_key, "id": s.id}
 123             )
 124         return {"service_table": service_list}
 125 
 126     @property
 127     def services(self):
 128         services = PagerDutyService.objects.filter(organization_integration=self.org_integration)
 129 
 130         return services

 133 class PagerDutyIntegrationProvider(IntegrationProvider):
 134     key = "pagerduty"
 135     name = "PagerDuty"
 136     metadata = metadata
 137     features = frozenset([IntegrationFeatures.ALERT_RULE])
 138     integration_cls = PagerDutyIntegration
 139 
 140     setup_dialog_config = {"width": 600, "height": 900}
 141 
 142     def get_pipeline_views(self):
 143         return [PagerDutyInstallationRedirect()]
 144 
 145     def post_install(self, integration, organization, extra=None):
 146         services = integration.metadata["services"]
 147         try:
 148             org_integration = OrganizationIntegration.objects.get(
 149                 integration=integration, organization=organization
 150             )
 151         except OrganizationIntegration.DoesNotExist:
 152             return
 153 
 154         with transaction.atomic():
 155             for service in services:
 156                 PagerDutyService.objects.create_or_update(
 157                     organization_integration=org_integration,
 158                     integration_key=service["integration_key"],
 159                     service_name=service["name"],
 160                 )
 161 
 162     def build_integration(self, state):
 163         config = json.loads(state.get("config"))
 164         account = config["account"]
 165         # PagerDuty gives us integration keys for various things, some of which
 166         # are not services. For now we only care about services.
 167         services = [x for x in config["integration_keys"] if x["type"] == "service"]
 168 
 169         return {
 170             "name": account["name"],
 171             "external_id": account["subdomain"],
 172             "metadata": {"services": services, "domain_name": account["subdomain"]},
 173         }

 176 class PagerDutyInstallationRedirect(PipelineView):
 177     def get_app_url(self, account_name=None):
 178         if not account_name:
 179             account_name = "app"
 180 
 181         app_id = options.get("pagerduty.app-id")
 182         setup_url = absolute_uri("/extensions/pagerduty/setup/")
 183 
 184         return (
 185             u"https://%s.pagerduty.com/install/integration?app_id=%s&redirect_url=%s&version=2"
 186             % (account_name, app_id, setup_url)
 187         )
 188 
 189     def dispatch(self, request, pipeline):
 190         if "config" in request.GET:
 191             pipeline.bind_state("config", request.GET["config"])
 192             return pipeline.next_step()
 193 
 194         account_name = request.GET.get("account", None)
 195 
 196         return self.redirect(self.get_app_url(account_name))
