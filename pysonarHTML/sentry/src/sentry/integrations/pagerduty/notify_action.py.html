  17 class PagerDutyNotifyServiceForm(forms.Form):
  18     account = forms.ChoiceField(choices=(), widget=forms.Select())
  19     service = forms.ChoiceField(required=False, choices=(), widget=forms.Select())
  20 
  21     def __init__(self, *args, **kwargs):
  22         integrations = [(i.id, i.name) for i in kwargs.pop("integrations")]
  23         services = kwargs.pop("services")
  24 
  25         super(PagerDutyNotifyServiceForm, self).__init__(*args, **kwargs)
  26         if integrations:
  27             self.fields["account"].initial = integrations[0][0]
  28 
  29         self.fields["account"].choices = integrations
  30         self.fields["account"].widget.choices = self.fields["account"].choices
  31 
  32         if services:
  33             self.fields["service"].initial = services[0][0]
  34 
  35         self.fields["service"].choices = services
  36         self.fields["service"].widget.choices = self.fields["service"].choices
  37 
  38     def clean(self):
  39         cleaned_data = super(PagerDutyNotifyServiceForm, self).clean()
  40 
  41         integration_id = cleaned_data.get("account")
  42         service_id = cleaned_data.get("service")
  43 
  44         service = PagerDutyService.objects.get(id=service_id)
  45 
  46         # need to make sure that the service actually belongs to that integration - meaning
  47         # that it belongs under the appropriate account in PagerDuty
  48         if not service.organization_integration.integration_id == int(integration_id):
  49             params = {
  50                 "account": dict(self.fields["account"].choices).get(int(integration_id)),
  51                 "service": dict(self.fields["service"].choices).get(int(service_id)),
  52             }
  53 
  54             raise forms.ValidationError(
  55                 _(
  56                     'The service "%(service)s" does not exist or has not been granted access in the %(account)s Pagerduty account.'
  57                 ),
  58                 code="invalid",
  59                 params=params,
  60             )
  61 
  62         return cleaned_data

  65 class PagerDutyNotifyServiceAction(EventAction):
  66     form_cls = PagerDutyNotifyServiceForm
  67     label = "Send a notification to PagerDuty account {account} and service {service}"
  68     prompt = "Send a PagerDuty notification"
  69 
  70     def __init__(self, *args, **kwargs):
  71         super(PagerDutyNotifyServiceAction, self).__init__(*args, **kwargs)
  72         self.form_fields = {
  73             "account": {
  74                 "type": "choice",
  75                 "choices": [(i.id, i.name) for i in self.get_integrations()],
  76             },
  77             "service": {"type": "choice", "choices": self.get_services()},
  78         }
  79 
  80     def is_enabled(self):
  81         return self.get_integrations().exists()
  82 
  83     def after(self, event, state):
  84         integration_id = self.get_option("account")
  85         service_id = self.get_option("service")
  86 
  87         try:
  88             integration = Integration.objects.get(
  89                 id=integration_id,
  90                 provider="pagerduty",
  91                 organizations=self.project.organization,
  92                 status=ObjectStatus.VISIBLE,
  93             )
  94         except Integration.DoesNotExist:
  95             # integration removed but rule still exists
  96             return
  97 
  98         try:
  99             service = PagerDutyService.objects.get(pk=service_id)
 100         except PagerDutyService.DoesNotExist:
 101             return
 102 
 103         def send_notification(event, futures):
 104             client = PagerDutyClient(integration_key=service.integration_key)
 105             try:
 106                 resp = client.send_trigger(event)
 107             except ApiError as e:
 108                 self.logger.info(
 109                     "rule.fail.pagerduty_trigger",
 110                     extra={
 111                         "error": six.text_type(e),
 112                         "service_name": service.service_name,
 113                         "service_id": service.id,
 114                     },
 115                 )
 116                 raise e
 117 
 118             # TODO(meredith): Maybe have a generic success log statements for
 119             # first-party integrations similar to plugin `notification.dispatched`
 120             self.logger.info(
 121                 "rule.success.pagerduty_trigger",
 122                 extra={
 123                     "status_code": resp.status_code,
 124                     "project_id": event.project_id,
 125                     "event_id": event.event_id,
 126                     "service_id": service.id,
 127                 },
 128             )
 129 
 130         key = u"pagerduty:{}".format(integration.id)
 131         yield self.future(send_notification, key=key)
 132 
 133     def get_integrations(self):
 134         integrations = Integration.objects.filter(
 135             provider="pagerduty",
 136             organizations=self.project.organization,
 137             status=ObjectStatus.VISIBLE,
 138         )
 139 
 140         return integrations
 141 
 142     def get_services(self):
 143         organization = self.project.organization
 144         integrations = Integration.objects.filter(
 145             provider="pagerduty", organizations=organization, status=ObjectStatus.VISIBLE
 146         )
 147         services = []
 148         for integration in integrations:
 149             service_list = PagerDutyService.objects.filter(
 150                 organization_integration_id=OrganizationIntegration.objects.get(
 151                     organization=organization, integration=integration
 152                 )
 153             ).values_list("id", "service_name")
 154             services += service_list
 155         return services
 156 
 157     def render_label(self):
 158         try:
 159             integration_name = Integration.objects.get(
 160                 provider="pagerduty",
 161                 organizations=self.project.organization,
 162                 id=self.get_option("account"),
 163             ).name
 164         except Integration.DoesNotExist:
 165             integration_name = "[removed]"
 166 
 167         try:
 168             service_name = PagerDutyService.objects.get(id=self.get_option("service")).service_name
 169         except PagerDutyService.DoesNotExist:
 170             service_name = "[removed]"
 171 
 172         return self.label.format(account=integration_name, service=service_name)
 173 
 174     def get_form_instance(self):
 175         return self.form_cls(
 176             self.data, integrations=self.get_integrations(), services=self.get_services()
 177         )
