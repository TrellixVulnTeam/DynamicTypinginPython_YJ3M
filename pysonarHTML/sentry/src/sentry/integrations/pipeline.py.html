  20 def ensure_integration(key, data):
  21     defaults = {
  22         "metadata": data.get("metadata", {}),
  23         "name": data.get("name", data["external_id"]),
  24         "status": ObjectStatus.VISIBLE,
  25     }
  26     integration, created = Integration.objects.get_or_create(
  27         provider=key, external_id=data["external_id"], defaults=defaults
  28     )
  29     if not created:
  30         integration.update(**defaults)
  31 
  32     return integration

  35 class IntegrationPipeline(Pipeline):
  36     pipeline_name = "integration_pipeline"
  37     provider_manager = default_manager
  38 
  39     def finish_pipeline(self):
  40         try:
  41             data = self.provider.build_integration(self.state.data)
  42         except IntegrationError as e:
  43             self.get_logger().info(
  44                 "build-integration.failure",
  45                 extra={
  46                     "error_message": six.text_type(e),
  47                     "error_status": getattr(e, "code", None),
  48                     "provider_key": self.provider.key,
  49                 },
  50             )
  51             return self.error(six.text_type(e))
  52 
  53         response = self._finish_pipeline(data)
  54 
  55         extra = data.get("post_install_data")
  56         action = "upgrade" if extra else "install"
  57         # to Slack
  58         self.provider.create_audit_log_entry(
  59             self.integration, self.organization, self.request, action, extra=extra
  60         )
  61         self.provider.post_install(self.integration, self.organization, extra=extra)
  62         self.clear_session()
  63         return response
  64 
  65     def _finish_pipeline(self, data):
  66         if "reinstall_id" in data:
  67             self.integration = Integration.objects.get(
  68                 provider=self.provider.integration_key, id=data["reinstall_id"]
  69             )
  70             self.integration.update(external_id=data["external_id"], status=ObjectStatus.VISIBLE)
  71             self.integration.get_installation(self.organization.id).reinstall()
  72         if "integration_id" in data:
  73             self.integration = Integration.objects.get(
  74                 provider=self.provider.integration_key, id=data["integration_id"]
  75             )
  76             self.integration.reauthorize(data)
  77         elif "expect_exists" in data:
  78             self.integration = Integration.objects.get(
  79                 provider=self.provider.integration_key, external_id=data["external_id"]
  80             )
  81         else:
  82             self.integration = ensure_integration(self.provider.integration_key, data)
  83 
  84         # Does this integration provide a user identity for the user setting up
  85         # the integration?
  86         identity = data.get("user_identity")
  87 
  88         if identity:
  89             # Some identity providers may not be directly associated to the
  90             # external integration. Integrations may specify the external_id to
  91             # be used for the idp.
  92             idp_external_id = data.get("idp_external_id", data["external_id"])
  93             idp_config = data.get("idp_config", {})
  94 
  95             # Create identity provider for this integration if necessary
  96             idp, created = IdentityProvider.objects.get_or_create(
  97                 external_id=idp_external_id, type=identity["type"], defaults={"config": idp_config}
  98             )
  99             if not created:
 100                 idp.update(config=idp_config)
 101 
 102             identity_data = {
 103                 "status": IdentityStatus.VALID,
 104                 "scopes": identity["scopes"],
 105                 "data": identity["data"],
 106                 "date_verified": timezone.now(),
 107             }
 108 
 109             try:
 110                 identity_model, created = Identity.objects.get_or_create(
 111                     idp=idp,
 112                     user=self.request.user,
 113                     external_id=identity["external_id"],
 114                     defaults=identity_data,
 115                 )
 116                 if not created:
 117                     identity_model.update(**identity_data)
 118             except IntegrityError:
 119                 # If the external_id is already used for a different user or
 120                 # the user already has a different external_id remove those
 121                 # identities and recreate it, except in the case of Identities
 122                 # being used for login.
 123                 if idp.type in ("github", "vsts", "google"):
 124                     try:
 125                         other_identity = Identity.objects.get(
 126                             idp=idp, external_id=identity["external_id"]
 127                         )
 128                     except Identity.DoesNotExist:
 129                         # The user is linked to a different external_id. It's ok to relink
 130                         # here because they'll still be able to log in with the new external_id.
 131                         pass
 132                     else:
 133                         # The external_id is linked to a different user. If that user doesn't
 134                         # have a password, we don't delete the link as it may lock them out.
 135                         if not other_identity.user.has_usable_password():
 136                             proper_name = idp.get_provider().name
 137                             return self._dialog_response(
 138                                 {
 139                                     "error": _(
 140                                         "The provided %(proper_name)s account is linked to a different Sentry user. "
 141                                         "To continue linking the current Sentry user, please use a different %(proper_name)s account."
 142                                     )
 143                                     % ({"proper_name": proper_name})
 144                                 },
 145                                 False,
 146                             )
 147                 identity_model = Identity.reattach(
 148                     idp, identity["external_id"], self.request.user, identity_data
 149                 )
 150 
 151         default_auth_id = None
 152         if self.provider.needs_default_identity:
 153             if not (identity and identity_model):
 154                 raise NotImplementedError("Integration requires an identity")
 155             default_auth_id = identity_model.id
 156 
 157         org_integration = self.integration.add_organization(
 158             self.organization, self.request.user, default_auth_id=default_auth_id
 159         )
 160         return self._dialog_success(org_integration)
 161 
 162     def _dialog_success(self, org_integration):
 163         return self._dialog_response(serialize(org_integration, self.request.user), True)
 164 
 165     def _dialog_response(self, data, success):
 166         context = {"payload": {"success": success, "data": data}}
 167         return render_to_response("sentry/integrations/dialog-complete.html", context, self.request)
