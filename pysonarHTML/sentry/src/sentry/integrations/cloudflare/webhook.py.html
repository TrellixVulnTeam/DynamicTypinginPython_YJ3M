  20 def requires_auth(func):
  21     @wraps(func)
  22     def wrapped(self, request, *args, **kwargs):
  23         if not request.user.is_authenticated():
  24             return Response({"proceed": False}, 401)
  25         return func(self, request, *args, **kwargs)
  26 
  27     return wrapped

  30 class CloudflareTokenAuthentication(TokenAuthentication):
  31     def authenticate(self, request):
  32         # XXX(dcramer): Hack around CF needing a token in the JSON body,
  33         # but us additionally needing to verify the signature of the payload.
  34         # This technically lets a user brute force a token before we actually
  35         # verify the signature, HOWEVER, they could do that either way so we
  36         # are ok with it.
  37         request.body = six.text_type(request.body)
  38         try:
  39             token = request.data["authentications"]["account"]["token"]["token"]
  40         except KeyError:
  41             return None
  42         return self.authenticate_credentials(request, token)

  49     def verify(self, payload, key, signature):
  50         return constant_time_compare(
  51             signature,
  52             hmac.new(
  53                 key=key.encode("utf-8"), msg=payload.encode("utf-8"), digestmod=sha256
  54             ).hexdigest(),
  55         )

  93     def on_account_change(self, request, data, is_test):
  94         organizations = sorted(
  95             Organization.objects.get_for_user(request.user, scope="project:write"),
  96             key=lambda x: x.slug,
  97         )
  98 
  99         enum_choices = [six.text_type(o.id) for o in organizations]
 100 
 101         data["install"]["schema"]["properties"]["organization"] = {
 102             "type": "string",
 103             "title": "Sentry Organization",
 104             "order": 1,
 105             "enum": enum_choices,
 106             "enumNames": {six.text_type(o.id): o.slug for o in organizations},
 107             "required": True,
 108         }
 109         if not enum_choices:
 110             return self.on_organization_clear(request, data, is_test)
 111         else:
 112             if data["install"]["options"].get("organization") not in enum_choices:
 113                 data["install"]["options"]["organization"] = enum_choices[0]
 114             return self.on_organization_change(request, data, is_test)
 115 
 116         return Response({"install": data["install"], "proceed": True})

 194     def post(self, request):
 195         signature = request.META.get("HTTP_X_SIGNATURE_HMAC_SHA256_HEX")
 196         variant = request.META.get("HTTP_X_SIGNATURE_KEY_VARIANT")
 197         logging_data = {
 198             "user_id": request.user.id if request.user.is_authenticated() else None,
 199             "signature": signature,
 200             "variant": variant,
 201         }
 202 
 203         payload = request.body
 204         try:
 205             data = request.data
 206         except (ValueError, TypeError):
 207             logger.error("cloudflare.webhook.invalid-json", extra=logging_data)
 208             return Response(status=400)
 209 
 210         event = data.get("event")
 211         logger.info(u"cloudflare.webhook.{}".format(event), extra=logging_data)
 212         if not signature:
 213             logger.error("cloudflare.webhook.invalid-signature", extra=logging_data)
 214             return Response(status=400)
 215         if not variant:
 216             logger.error("cloudflare.webhook.invalid-variant", extra=logging_data)
 217             return Response(status=400)
 218 
 219         if variant == "test":
 220             key = "test-key"
 221         elif variant == "1":
 222             key = options.get("cloudflare.secret-key")
 223         else:
 224             logger.error("cloudflare.webhook.invalid-variant", extra=logging_data)
 225             return Response(status=400)
 226 
 227         app_id = data.get("app", {}).get("id")
 228         if app_id not in ("local", "") and variant == "test":
 229             logger.error("cloudflare.webhook.invalid-variant", extra=logging_data)
 230             return Response(status=400)
 231 
 232         if not self.verify(payload, key, signature):
 233             logger.error(u"cloudflare.webhook.invalid-signature".format(event), extra=logging_data)
 234             return Response(status=400)
 235 
 236         if event == "option-change:account":
 237             return self.on_account_change(request, data, is_test=variant == "test")
 238         if event == "option-change:organization":
 239             return self.on_organization_change(request, data, is_test=variant == "test")
 240         if event == "option-change:project":
 241             return self.on_project_change(request, data, is_test=variant == "test")
 242         elif event == "preview":
 243             return self.on_preview(request, data, is_test=variant == "test")
 244         return Response({"install": data["install"], "proceed": True})
