  30     def get_group_body(self, group, event, **kwargs):
  31         result = []
  32         for interface in six.itervalues(event.interfaces):
  33             output = safe_execute(interface.to_string, event, _with_transaction=False)
  34             if output:
  35                 result.append(output)
  36         return "\n\n".join(result)

  38     def get_group_description(self, group, event, **kwargs):
  39         params = {}
  40         if kwargs.get("link_referrer"):
  41             params["referrer"] = kwargs.get("link_referrer")
  42         output = [
  43             u"Sentry Issue: [{}]({})".format(
  44                 group.qualified_short_id, absolute_uri(group.get_absolute_url(params=params))
  45             )
  46         ]
  47         body = self.get_group_body(group, event)
  48         if body:
  49             output.extend(["", "```", body, "```"])
  50         return "\n".join(output)

  97     def store_issue_last_defaults(self, project_id, data):
  98         """
  99         Stores the last used field defaults on a per-project basis. This
 100         accepts a dict of values that will be filtered to keys returned by
 101         ``get_persisted_default_config_fields`` which will automatically be
 102         merged into the associated field config object as the default.
 103 
 104         >>> integ.store_issue_last_defaults(1, {'externalProject': 2})
 105 
 106         When the integration is serialized these values will automatically be
 107         merged into the field configuration objects.
 108 
 109         NOTE: These are currently stored for both link and create issue, no
 110               differentiation is made between the two field configs.
 111         """
 112         persisted_fields = self.get_persisted_default_config_fields()
 113         if not persisted_fields:
 114             return
 115 
 116         defaults = {k: v for k, v in six.iteritems(data) if k in persisted_fields}
 117 
 118         self.org_integration.config.update({"project_issue_defaults": {project_id: defaults}})
 119         self.org_integration.save()

 186     def get_repository_choices(self, group, **kwargs):
 187         """
 188         Returns the default repository and a set/subset of repositories of associated with the installation
 189         """
 190         try:
 191             repos = self.get_repositories()
 192         except ApiError:
 193             raise IntegrationError("Unable to retrive repositories. Please try again later.")
 194         else:
 195             repo_choices = [(repo["identifier"], repo["name"]) for repo in repos]
 196 
 197         repo = kwargs.get("repo")
 198         if not repo:
 199             params = kwargs.get("params", {})
 200             defaults = self.get_project_defaults(group.project_id)
 201             repo = params.get("repo", defaults.get("repo"))
 202 
 203         try:
 204             default_repo = repo or repo_choices[0][0]
 205         except IndexError:
 206             return "", repo_choices
 207 
 208         # If a repo has been selected outside of the default list of
 209         # repos, stick it onto the front of the list so that it can be
 210         # selected.
 211         try:
 212             next(True for r in repo_choices if r[0] == default_repo)
 213         except StopIteration:
 214             repo_choices.insert(0, self.create_default_repo_choice(default_repo))
 215 
 216         return default_repo, repo_choices
 217

 225     def get_annotations_for_group_list(self, group_list):
 226         group_links = GroupLink.objects.filter(
 227             group_id__in=[group.id for group in group_list],
 228             project_id__in=list(set(group.project.id for group in group_list)),
 229             linked_type=GroupLink.LinkedType.issue,
 230             relationship=GroupLink.Relationship.references,
 231         )
 232 
 233         external_issues = ExternalIssue.objects.filter(
 234             id__in=[group_link.linked_id for group_link in group_links],
 235             integration_id=self.model.id,
 236         )
 237 
 238         # group annotations by group id
 239         annotations_by_group_id = defaultdict(list)
 240         for group_link in group_links:
 241             issues_for_group = filter(lambda x: x.id == group_link.linked_id, external_issues)
 242             annotations = self.map_external_issues_to_annotations(issues_for_group)
 243             annotations_by_group_id[group_link.group_id].extend(annotations)
 244 
 245         return annotations_by_group_id

 247     def map_external_issues_to_annotations(self, external_issues):
 248         annotations = []
 249         for ei in external_issues:
 250             link = self.get_issue_url(ei.key)
 251             label = self.get_issue_display_name(ei) or ei.key
 252             annotations.append('<a href="%s">%s</a>' % (link, label))
 253 
 254         return annotations

 327     def update_group_status(self, groups, status, activity_type):
 328         updated = (
 329             Group.objects.filter(id__in=[g.id for g in groups])
 330             .exclude(status=status)
 331             .update(status=status)
 332         )
 333         if updated:
 334             for group in groups:
 335                 activity = Activity.objects.create(
 336                     project=group.project, group=group, type=activity_type
 337                 )
 338                 activity.send_notification()

 340     def sync_status_inbound(self, issue_key, data):
 341         if not self.should_sync("inbound_status"):
 342             return
 343 
 344         organization = Organization.objects.get(id=self.organization_id)
 345         has_issue_sync = features.has("organizations:integrations-issue-sync", organization)
 346 
 347         if not has_issue_sync:
 348             return
 349 
 350         affected_groups = list(
 351             Group.objects.get_groups_by_external_issue(self.model, issue_key)
 352             .filter(project__organization_id=self.organization_id)
 353             .select_related("project")
 354         )
 355 
 356         groups_to_resolve = []
 357         groups_to_unresolve = []
 358 
 359         should_resolve = self.should_resolve(data)
 360         should_unresolve = self.should_unresolve(data)
 361 
 362         for group in affected_groups:
 363 
 364             # this probably shouldn't be possible unless there
 365             # is a bug in one of those methods
 366             if should_resolve is True and should_unresolve is True:
 367                 logger.warning(
 368                     "sync-config-conflict",
 369                     extra={
 370                         "organization_id": group.project.organization_id,
 371                         "integration_id": self.model.id,
 372                         "provider": self.model.get_provider(),
 373                     },
 374                 )
 375                 continue
 376 
 377             if should_unresolve:
 378                 groups_to_unresolve.append(group)
 379             elif should_resolve:
 380                 groups_to_resolve.append(group)
 381 
 382         if groups_to_resolve:
 383             self.update_group_status(groups_to_resolve, GroupStatus.RESOLVED, Activity.SET_RESOLVED)
 384 
 385         if groups_to_unresolve:
 386             self.update_group_status(
 387                 groups_to_unresolve, GroupStatus.UNRESOLVED, Activity.SET_UNRESOLVED
 388             )
