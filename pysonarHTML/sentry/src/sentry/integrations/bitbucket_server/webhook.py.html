  24 class Webhook(object):
  25     def __call__(self, organization, integration_id, event):

  28     def update_repo_data(self, repo, event):
  29         """
  30         Given a webhook payload, update stored repo data if needed.
  31         """
  32 
  33         name_from_event = event["repository"]["project"]["key"] + "/" + event["repository"]["slug"]
  34         # build the URL manually since it doesn't come back from the API in
  35         # the form that we need
  36         url_from_event = event["repository"]["links"]["self"][0]["href"]
  37 
  38         if (
  39             repo.name != name_from_event
  40             or repo.config.get("name") != name_from_event
  41             or repo.url != url_from_event
  42         ):
  43             repo.update(name=name_from_event, config=dict(repo.config, name=name_from_event))

  46 class PushEventWebhook(Webhook):
  47     def __call__(self, organization, integration_id, event):
  48         authors = {}
  49 
  50         try:
  51             repo = Repository.objects.get(
  52                 organization_id=organization.id,
  53                 provider=PROVIDER_NAME,
  54                 external_id=six.text_type(event["repository"]["id"]),
  55             )
  56         except Repository.DoesNotExist:
  57             raise Http404()
  58 
  59         client = repo.get_provider().get_installation(integration_id, organization.id).get_client()
  60 
  61         # while we're here, make sure repo data is up to date
  62         self.update_repo_data(repo, event)
  63 
  64         [project_name, repo_name] = repo.name.split("/")
  65 
  66         for change in event["changes"]:
  67             from_hash = None if change.get("fromHash") == "0" * 40 else change.get("fromHash")
  68             for commit in client.get_commits(
  69                 project_name, repo_name, from_hash, change.get("toHash")
  70             )["values"]:
  71 
  72                 if IntegrationRepositoryProvider.should_ignore_commit(commit["message"]):
  73                     continue
  74 
  75                 author_email = commit["author"]["emailAddress"]
  76 
  77                 # its optional, lets just throw it out for now
  78                 if author_email is None or len(author_email) > 75:
  79                     author = None
  80                 elif author_email not in authors:
  81                     authors[author_email] = author = CommitAuthor.objects.get_or_create(
  82                         organization_id=organization.id,
  83                         email=author_email,
  84                         defaults={"name": commit["author"]["name"]},
  85                     )[0]
  86                 else:
  87                     author = authors[author_email]
  88                 try:
  89                     with transaction.atomic():
  90 
  91                         Commit.objects.create(
  92                             repository_id=repo.id,
  93                             organization_id=organization.id,
  94                             key=commit["id"],
  95                             message=commit["message"],
  96                             author=author,
  97                             date_added=datetime.fromtimestamp(
  98                                 commit["authorTimestamp"] / 1000, timezone.utc
  99                             ),
 100                         )
 101 
 102                 except IntegrityError:
 103                     pass

 106 class BitbucketServerWebhookEndpoint(View):
 107     _handlers = {"repo:refs_changed": PushEventWebhook}
 108 
 109     def get_handler(self, event_type):
 110         return self._handlers.get(event_type)
 111 
 112     @method_decorator(csrf_exempt)
 113     def dispatch(self, request, *args, **kwargs):
 114         if request.method != "POST":
 115             return HttpResponse(status=405)
 116 
 117         return super(BitbucketServerWebhookEndpoint, self).dispatch(request, *args, **kwargs)
 118 
 119     def post(self, request, organization_id, integration_id):
 120         try:
 121             organization = Organization.objects.get_from_cache(id=organization_id)
 122         except Organization.DoesNotExist:
 123             logger.error(
 124                 PROVIDER_NAME + ".webhook.invalid-organization",
 125                 extra={"organization_id": organization_id, "integration_id": integration_id},
 126             )
 127             return HttpResponse(status=400)
 128 
 129         body = six.binary_type(request.body)
 130         if not body:
 131             logger.error(
 132                 PROVIDER_NAME + ".webhook.missing-body", extra={"organization_id": organization.id}
 133             )
 134             return HttpResponse(status=400)
 135 
 136         try:
 137             handler = self.get_handler(request.META["HTTP_X_EVENT_KEY"])
 138         except KeyError:
 139             logger.error(
 140                 PROVIDER_NAME + ".webhook.missing-event",
 141                 extra={"organization_id": organization.id, "integration_id": integration_id},
 142             )
 143             return HttpResponse(status=400)
 144 
 145         if not handler:
 146             return HttpResponse(status=204)
 147 
 148         try:
 149             event = json.loads(body.decode("utf-8"))
 150         except JSONDecodeError:
 151             logger.error(
 152                 PROVIDER_NAME + ".webhook.invalid-json",
 153                 extra={"organization_id": organization.id, "integration_id": integration_id},
 154                 exc_info=True,
 155             )
 156             return HttpResponse(status=400)
 157 
 158         handler()(organization, integration_id, event)
 159         return HttpResponse(status=204)
