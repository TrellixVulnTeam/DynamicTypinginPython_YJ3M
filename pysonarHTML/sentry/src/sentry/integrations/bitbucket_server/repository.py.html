  30         client = installation.get_client()
  31         try:
  32             project, repo = config["identifier"].split("/", 1)
  33             repo = client.get_repo(project, repo)
  34         except Exception as e:

  98     def _format_commits(self, client, repo, commit_list):
  99         return [
 100             {
 101                 "id": c["id"],
 102                 "repository": repo.name,
 103                 "author_email": c["author"]["emailAddress"],
 104                 "author_name": c["author"].get("displayName", c["author"]["name"]),
 105                 "message": c["message"],
 106                 "timestamp": datetime.fromtimestamp(c["authorTimestamp"] / 1000, timezone.utc),
 107                 "patch_set": self._get_patchset(
 108                     client, repo.config["project"], repo.config["repo"], c["id"]
 109                 ),
 110             }
 111             for c in commit_list["values"]
 112         ]

 114     def compare_commits(self, repo, start_sha, end_sha):
 115         installation = self.get_installation(repo.integration_id, repo.organization_id)
 116         client = installation.get_client()
 117 
 118         try:
 119             if "0" * 40 == start_sha or start_sha is None:
 120                 res = client.get_last_commits(repo.config["project"], repo.config["repo"])
 121             else:
 122                 res = client.get_commits(
 123                     repo.config["project"], repo.config["repo"], start_sha, end_sha
 124                 )
 125             return self._format_commits(client, repo, res)
 126         except Exception as e:
 127             installation.raise_error(e)

 132     def _get_patchset(self, client, project, repo, sha):
 133         """
 134             Get the modified files for a commit
 135         """
 136 
 137         key = u"get_changelist:{}:{}".format(md5_text(project + repo).hexdigest(), sha)
 138         commit_files = cache.get(key)
 139         if commit_files is None:
 140             commit_files = client.get_commit_filechanges(project, repo, sha)
 141             cache.set(key, commit_files, 900)
 142 
 143         return self._transform_patchset(commit_files)
 144 
 145     def _transform_patchset(self, diff):
 146         """Convert the patch data from Bitbucket into our internal format
 147 
 148         See sentry.models.Release.set_commits
 149         """
 150         changes = []
 151         for change in diff["values"]:
 152             if change["type"] == "MODIFY":
 153                 changes.append({"path": change["path"]["toString"], "type": "M"})
 154             if change["type"] == "ADD":
 155                 changes.append({"path": change["path"]["toString"], "type": "A"})
 156             if change["type"] == "DELETE":
 157                 changes.append({"path": change["path"]["toString"], "type": "D"})
 158             if change["type"] == "MOVE":
 159                 changes.append({"path": change["srcPath"]["toString"], "type": "D"})
 160                 changes.append({"path": change["path"]["toString"], "type": "A"})
 161         return changes
