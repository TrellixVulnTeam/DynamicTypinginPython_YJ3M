  24 class BitbucketServerSetupClient(ApiClient):
  25     """
  26     Client for making requests to Bitbucket Server to follow OAuth1 flow.
  27     """
  28 
  29     request_token_url = u"{}/plugins/servlet/oauth/request-token"
  30     access_token_url = u"{}/plugins/servlet/oauth/access-token"
  31     authorize_url = u"{}/plugins/servlet/oauth/authorize?oauth_token={}"
  32     integration_name = "bitbucket_server_setup"
  33 
  34     def __init__(self, base_url, consumer_key, private_key, verify_ssl=True):
  35         self.base_url = base_url
  36         self.consumer_key = consumer_key
  37         self.private_key = private_key
  38         self.verify_ssl = verify_ssl
  39 
  40     def get_request_token(self):
  41         """
  42         Step 1 of the oauth flow.
  43         Get a request token that we can have the user verify.
  44         """
  45         url = self.request_token_url.format(self.base_url)
  46         resp = self.post(url, allow_text=True)
  47         return dict(parse_qsl(resp.text))
  48 
  49     def get_authorize_url(self, request_token):
  50         """
  51         Step 2 of the oauth flow.
  52         Get a URL that the user can verify our request token at.
  53         """
  54         return self.authorize_url.format(self.base_url, request_token["oauth_token"])
  55 
  56     def get_access_token(self, request_token, verifier):
  57         """
  58         Step 3 of the oauth flow.
  59         Use the verifier and request token from step 1 to get an access token.
  60         """
  61         if not verifier:
  62             raise ApiError("Missing OAuth token verifier")
  63         auth = OAuth1(
  64             client_key=self.consumer_key,
  65             resource_owner_key=request_token["oauth_token"],
  66             resource_owner_secret=request_token["oauth_token_secret"],
  67             verifier=verifier,
  68             rsa_key=self.private_key,
  69             signature_method=SIGNATURE_RSA,
  70             signature_type="auth_header",
  71         )
  72         url = self.access_token_url.format(self.base_url)
  73         resp = self.post(url, auth=auth, allow_text=True)
  74         return dict(parse_qsl(resp.text))
  75 
  76     def request(self, *args, **kwargs):
  77         """
  78         Add OAuth1 RSA signatures.
  79         """
  80         if "auth" not in kwargs:
  81             kwargs["auth"] = OAuth1(
  82                 client_key=self.consumer_key,
  83                 rsa_key=self.private_key,
  84                 signature_method=SIGNATURE_RSA,
  85                 signature_type="auth_header",
  86             )
  87         return self._request(*args, **kwargs)

  90 class BitbucketServer(ApiClient):
  91     """
  92     Contains the BitBucket Server specifics in order to communicate with bitbucket
  93 
  94     You can find BitBucket REST API docs here:
  95 
  96     https://developer.atlassian.com/server/bitbucket/reference/rest-api/
  97     """
  98 
  99     integration_name = "bitbucket_server"
 100 
 101     def __init__(self, base_url, credentials, verify_ssl):
 102         super(BitbucketServer, self).__init__(verify_ssl)
 103 
 104         self.base_url = base_url
 105         self.credentials = credentials
 106 
 107     def get_repos(self):
 108         return self.get(
 109             BitbucketServerAPIPath.repositories,
 110             auth=self.get_auth(),
 111             params={"limit": 250, "permission": "REPO_ADMIN"},
 112         )
 113 
 114     def search_repositories(self, query_string):
 115         return self.get(
 116             BitbucketServerAPIPath.repositories,
 117             auth=self.get_auth(),
 118             params={"limit": 250, "permission": "REPO_ADMIN", "name": query_string},
 119         )
 120 
 121     def get_repo(self, project, repo):
 122         return self.get(
 123             BitbucketServerAPIPath.repository.format(project=project, repo=repo),
 124             auth=self.get_auth(),
 125         )
 126 
 127     def create_hook(self, project, repo, data):
 128         return self.post(
 129             BitbucketServerAPIPath.repository_hooks.format(project=project, repo=repo),
 130             auth=self.get_auth(),
 131             data=data,
 132         )
 133 
 134     def delete_hook(self, project, repo, webhook_id):
 135         return self.delete(
 136             BitbucketServerAPIPath.repository_hook.format(
 137                 project=project, repo=repo, id=webhook_id
 138             ),
 139             auth=self.get_auth(),
 140         )
 141 
 142     def get_commits(self, project, repo, fromHash, toHash):
 143         return self.get(
 144             BitbucketServerAPIPath.repository_commits.format(project=project, repo=repo),
 145             auth=self.get_auth(),
 146             params={"since": fromHash, "until": toHash, "merges": "exclude"},
 147         )
 148 
 149     def get_last_commits(self, project, repo, limit=10):
 150         return self.get(
 151             BitbucketServerAPIPath.repository_commits.format(project=project, repo=repo),
 152             auth=self.get_auth(),
 153             params={"merges": "exclude", "limit": limit},
 154         )
 155 
 156     def get_commit_filechanges(self, project, repo, commit):
 157         return self.get(
 158             BitbucketServerAPIPath.commit_changes.format(project=project, repo=repo, commit=commit),
 159             auth=self.get_auth(),
 160         )
 161 
 162     def get_auth(self):
 163         return OAuth1(
 164             client_key=self.credentials["consumer_key"],
 165             rsa_key=self.credentials["private_key"],
 166             resource_owner_key=self.credentials["access_token"],
 167             resource_owner_secret=self.credentials["access_token_secret"],
 168             signature_method=SIGNATURE_RSA,
 169             signature_type="auth_header",
 170         )
