  73 class InstallationForm(forms.Form):

  87         ),
  88         widget=forms.CheckboxInput(),

  90         initial=True,
  91     )
  92     consumer_key = forms.CharField(
  93         label=_("Bitbucket Consumer Key"),
  94         widget=forms.TextInput(attrs={"placeholder": "sentry-consumer-key"}),
  95     )
  96     private_key = forms.CharField(
  97         label=_("Bitbucket Consumer Private Key"),
  98         widget=forms.Textarea(
  99             attrs={
 100                 "placeholder": "-----BEGIN RSA PRIVATE KEY-----\n...\n-----END RSA PRIVATE KEY-----"
 101             }
 102         ),

 105     def clean_url(self):
 106         """Strip off trailing / as they cause invalid URLs downstream"""
 107         return self.cleaned_data["url"].rstrip("/")
 108 
 109     def clean_private_key(self):
 110         data = self.cleaned_data["private_key"]
 111 
 112         try:
 113             load_pem_private_key(data.encode("utf-8"), None, default_backend())
 114         except Exception:
 115             raise forms.ValidationError(
 116                 "Private key must be a valid SSH private key encoded in a PEM format."
 117             )
 118         return data
 119 
 120     def clean_consumer_key(self):
 121         data = self.cleaned_data["consumer_key"]
 122         if len(data) > 200:
 123             raise forms.ValidationError("Consumer key is limited to 200 characters.")
 124         return data

 127 class InstallationConfigView(PipelineView):
 128     """
 129     Collect the OAuth client credentials from the user.
 130     """
 131 
 132     def dispatch(self, request, pipeline):
 133         if request.method == "POST":
 134             form = InstallationForm(request.POST)
 135             if form.is_valid():
 136                 form_data = form.cleaned_data
 137 
 138                 pipeline.bind_state("installation_data", form_data)
 139                 return pipeline.next_step()
 140         else:
 141             form = InstallationForm()
 142 
 143         return render_to_response(
 144             template="sentry/integrations/bitbucket-server-config.html",
 145             context={"form": form},
 146             request=request,
 147         )

 150 class OAuthLoginView(PipelineView):
 151     """
 152     Start the OAuth dance by creating a request token
 153     and redirecting the user to approve it.
 154     """
 155 
 156     @csrf_exempt
 157     def dispatch(self, request, pipeline):
 158         if "oauth_token" in request.GET:
 159             return pipeline.next_step()
 160 
 161         config = pipeline.fetch_state("installation_data")
 162         client = BitbucketServerSetupClient(
 163             config.get("url"),
 164             config.get("consumer_key"),
 165             config.get("private_key"),
 166             config.get("verify_ssl"),
 167         )
 168 
 169         try:
 170             request_token = client.get_request_token()
 171             pipeline.bind_state("request_token", request_token)
 172             authorize_url = client.get_authorize_url(request_token)
 173 
 174             return self.redirect(authorize_url)
 175         except ApiError as error:
 176             logger.info(
 177                 "identity.bitbucket-server.request-token",
 178                 extra={"url": config.get("url"), "error": error},
 179             )
 180             return pipeline.error("Could not fetch a request token from Bitbucket. %s" % error)
 181

 183 class OAuthCallbackView(PipelineView):
 184     """
 185     Complete the OAuth dance by exchanging our request token
 186     into an access token.
 187     """
 188 
 189     @csrf_exempt
 190     def dispatch(self, request, pipeline):
 191         config = pipeline.fetch_state("installation_data")
 192         client = BitbucketServerSetupClient(
 193             config.get("url"),
 194             config.get("consumer_key"),
 195             config.get("private_key"),
 196             config.get("verify_ssl"),
 197         )
 198 
 199         try:
 200             access_token = client.get_access_token(
 201                 pipeline.fetch_state("request_token"), request.GET["oauth_token"]
 202             )
 203 
 204             pipeline.bind_state("access_token", access_token)
 205 
 206             return pipeline.next_step()
 207         except ApiError as error:
 208             logger.info("identity.bitbucket-server.access-token", extra={"error": error})
 209             return pipeline.error(
 210                 u"Could not fetch an access token from Bitbucket. %s" % six.text_type(error)
 211             )

 214 class BitbucketServerIntegration(IntegrationInstallation, RepositoryMixin):
 215     """
 216         IntegrationInstallation implementation for Bitbucket Server
 217     """
 218 
 219     repo_search = True
 220 
 221     default_identity = None
 222 
 223     def get_client(self):
 224         if self.default_identity is None:
 225             self.default_identity = self.get_default_identity()
 226 
 227         return BitbucketServer(
 228             self.model.metadata["base_url"],
 229             self.default_identity.data,
 230             self.model.metadata["verify_ssl"],
 231         )
 232 
 233     @property
 234     def username(self):
 235         return self.model.name
 236 
 237     def error_message_from_json(self, data):
 238         return data.get("error", {}).get("message", "unknown error")
 239 
 240     def get_repositories(self, query=None):
 241         if not query:
 242             resp = self.get_client().get_repos()
 243 
 244             return [
 245                 {
 246                     "identifier": repo["project"]["key"] + "/" + repo["slug"],
 247                     "project": repo["project"]["key"],
 248                     "repo": repo["slug"],
 249                     "name": repo["project"]["name"] + "/" + repo["name"],
 250                 }
 251                 for repo in resp.get("values", [])
 252             ]
 253 
 254         full_query = (query).encode("utf-8")
 255         resp = self.get_client().search_repositories(full_query)
 256 
 257         return [
 258             {
 259                 "identifier": repo["project"]["key"] + "/" + repo["slug"],
 260                 "project": repo["project"]["key"],
 261                 "repo": repo["slug"],
 262                 "name": repo["project"]["name"] + "/" + repo["name"],
 263             }
 264             for repo in resp.get("values", [])
 265         ]
 266 
 267     def has_repo_access(self, repo):
 268         """
 269         We can assume user always has repo access, since the Bitbucket API is limiting the results based on the REPO_ADMIN permission
 270         """
 271 
 272         return True
 273 
 274     def get_unmigratable_repositories(self):
 275         repos = Repository.objects.filter(
 276             organization_id=self.organization_id, provider="bitbucket_server"
 277         )
 278 
 279         accessible_repos = [r["identifier"] for r in self.get_repositories()]
 280 
 281         return filter(lambda repo: repo.name not in accessible_repos, repos)
 282 
 283     def reinstall(self):
 284         self.reinstall_repositories()

 287 class BitbucketServerIntegrationProvider(IntegrationProvider):
 288     key = "bitbucket_server"
 289     name = "Bitbucket Server"
 290     metadata = metadata
 291     integration_cls = BitbucketServerIntegration
 292     needs_default_identity = True
 293     can_add = True
 294     features = frozenset([IntegrationFeatures.COMMITS])
 295     setup_dialog_config = {"width": 1030, "height": 1000}
 296 
 297     def get_pipeline_views(self):
 298         return [InstallationConfigView(), OAuthLoginView(), OAuthCallbackView()]
 299 
 300     def post_install(self, integration, organization, extra=None):
 301         repo_ids = Repository.objects.filter(
 302             organization_id=organization.id,
 303             provider__in=["bitbucket_server", "integrations:bitbucket_server"],
 304             integration_id__isnull=True,
 305         ).values_list("id", flat=True)
 306 
 307         for repo_id in repo_ids:
 308             migrate_repo.apply_async(
 309                 kwargs={
 310                     "repo_id": repo_id,
 311                     "integration_id": integration.id,
 312                     "organization_id": organization.id,
 313                 }
 314             )
 315 
 316     def build_integration(self, state):
 317         install = state["installation_data"]
 318         access_token = state["access_token"]
 319 
 320         hostname = urlparse(install["url"]).netloc
 321         external_id = u"{}:{}".format(hostname, install["consumer_key"])[:64]
 322 
 323         credentials = {
 324             "consumer_key": install["consumer_key"],
 325             "private_key": install["private_key"],
 326             "access_token": access_token["oauth_token"],
 327             "access_token_secret": access_token["oauth_token_secret"],
 328         }
 329 
 330         return {
 331             "name": install["consumer_key"],
 332             "provider": self.key,
 333             "external_id": external_id,
 334             "metadata": {
 335                 "base_url": install["url"],
 336                 "domain_name": hostname,
 337                 "verify_ssl": install["verify_ssl"],
 338             },
 339             "user_identity": {
 340                 "type": self.key,
 341                 "external_id": external_id,
 342                 "data": credentials,
 343                 "scopes": [],
 344             },
 345         }
 346 
 347     def setup(self):
 348         from sentry.plugins.base import bindings
 349 
 350         bindings.add(
 351             "integration-repository.provider",
 352             BitbucketServerRepositoryProvider,
 353             id="integrations:%s" % self.key,
 354         )
