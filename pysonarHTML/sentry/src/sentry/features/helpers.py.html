   8 def requires_feature(feature, any_org=None):
   9     """
  10     Require a feature flag to access an endpoint.
  11 
  12     If ``any_org`` is ``True``, this will check all of the request User's
  13     Organizations for the flag. If any are flagged in, the endpoint is accessible.
  14 
  15     Without ``any_org=True``, the endpoint must resolve an Organization via
  16     ``convert_args`` (and therefor be in ``kwargs``). The resolved Org must have
  17     the passed feature.
  18 
  19     If any failure case, the API returns a 404.
  20 
  21     Example:
  22         >>> @requires_feature('organizations:internal-catchall')
  23         >>> def get(self, request, organization):
  24         >>>     return Response()
  25     """
  26 
  27     def decorator(func):
  28         def wrapped(self, request, *args, **kwargs):
  29             # The endpoint is accessible if any of the User's Orgs have the feature
  30             # flag enabled.
  31             if any_org:
  32                 if not any(
  33                     features.has(feature, org, actor=request.user)
  34                     for org in request.user.get_orgs()
  35                 ):
  36                     return Response(status=404)
  37 
  38                 return func(self, request, *args, **kwargs)
  39             # The Org in scope for the request must have the feature flag enabled.
  40             else:
  41                 if "organization" not in kwargs:
  42                     return Response(status=404)
  43 
  44                 if not features.has(feature, kwargs["organization"], actor=request.user):
  45                     return Response(status=404)
  46 
  47                 return func(self, request, *args, **kwargs)
  48 
  49         return wrapped
  50 
  51     return decorator
