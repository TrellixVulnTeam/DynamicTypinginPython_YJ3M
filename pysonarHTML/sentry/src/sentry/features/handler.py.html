   6 class FeatureHandler(object):
   7     features = set()

   9     def __call__(self, feature, actor):
  10         if feature.name not in self.features:
  11             return None
  12 
  13         return self.has(feature, actor)
  14 
  15     def has(self, feature, actor):
  16         raise NotImplementedError

  18     def has_for_batch(self, batch):
  19         # If not overridden, iterate over objects in the batch individually.
  20         return {
  21             obj: self.has(feature, batch.actor)
  22             for (obj, feature) in batch.get_feature_objects().items()
  23         }
