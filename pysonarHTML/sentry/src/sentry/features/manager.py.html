  14 class FeatureManager(object):
  15     def __init__(self):
  16         self._feature_registry = {}
  17         self._handler_registry = defaultdict(list)
  18 
  19     def all(self, feature_type=Feature):
  20         """
  21         Get a mapping of feature name -> feature class, optionally specific to a
  22         particular feature type.
  23         """
  24         return {k: v for k, v in self._feature_registry.items() if v == feature_type}
  25 
  26     def add(self, name, cls=Feature):
  27         """
  28         Register a feature.
  29 
  30         The passed class is a Feature container object, this object can be used
  31         to encapsulate the context associated to a feature.
  32 
  33         >>> FeatureManager.has('my:feature', actor=request.user)
  34         """
  35         self._feature_registry[name] = cls
  36 
  37     def _get_feature_class(self, name):
  38         try:
  39             return self._feature_registry[name]
  40         except KeyError:
  41             raise FeatureNotRegistered(name)
  42 
  43     def get(self, name, *args, **kwargs):
  44         """
  45         Lookup a registered feature handler given the feature name.
  46 
  47         >>> FeatureManager.get('my:feature', actor=request.user)
  48         """
  49         cls = self._get_feature_class(name)
  50         return cls(name, *args, **kwargs)
  51 
  52     def add_handler(self, handler):
  53         """
  54         Register a feature handler.
  55 
  56         The passed object is a FeatureHandler that is associated with all
  57         features defined in the ``handler.features`` property.
  58         """
  59         for feature_name in handler.features:
  60             self._handler_registry[feature_name].append(handler)
  61 
  62     def has(self, name, *args, **kwargs):
  63         """
  64         Determine if a feature is enabled.
  65 
  66         Features are checked in the following order:
  67 
  68         1. Execute registered feature handlers. Any
  69            ``feature.handler.FeatureHandler`` objects that have been registered
  70            with ``add_handler` will be executed in the order they are declared.
  71 
  72            When each handler is executed, should the handler return None
  73            instead of True or False (feature enabled / disabled), the
  74            next registered feature handler will be executed.
  75 
  76         2. The default configuration of the feature. This can be located in
  77            sentry.conf.server.SENTRY_FEATURES.
  78 
  79         Depending on the Feature class, additional arguments may need to be
  80         provided to assign organization or project context to the feature.
  81 
  82         >>> FeatureManager.has('organizations:feature', organization, actor=request.user)
  83 
  84         """
  85         actor = kwargs.pop("actor", None)
  86         feature = self.get(name, *args, **kwargs)
  87 
  88         # Check registered feature handlers
  89         rv = self._get_handler(feature, actor)
  90         if rv is not None:
  91             return rv
  92 
  93         rv = settings.SENTRY_FEATURES.get(feature.name, False)
  94         if rv is not None:
  95             return rv
  96 
  97         # Features are by default disabled if no plugin or default enables them
  98         return False
  99 
 100     def _get_handler(self, feature, actor):
 101         for handler in self._handler_registry[feature.name]:
 102             rv = handler(feature, actor)
 103             if rv is not None:
 104                 return rv
 105         return None

 107     def has_for_batch(self, name, organization, objects, actor=None):
 108         """
 109         Determine in a batch if a feature is enabled.
 110 
 111         This applies the same procedure as ``FeatureManager.has``, but with a
 112         performance benefit where the objects being checked all belong to the
 113         same organization. The objects are entities (e.g., projects) with the
 114         common parent organization, as would be passed individually to ``has``.
 115 
 116         Feature handlers that depend only on organization attributes, and not
 117         on attributes of the individual objects being checked, will generally
 118         perform faster if this method is used in preference to ``has``.
 119 
 120         The return value is a dictionary with the objects as keys. Each value
 121         is what would be returned if the key were passed to ``has``.
 122 
 123         >>> FeatureManager.has_for_batch('projects:feature', organization, [project1, project2], actor=request.user)
 124         """
 125 
 126         result = dict()
 127         remaining = set(objects)
 128 
 129         handlers = self._handler_registry[name]
 130         for handler in handlers:
 131             if not remaining:
 132                 break
 133 
 134             with sentry_sdk.start_span(
 135                 op="feature.has_for_batch.handler",
 136                 description="{0} ({1})".format(type(handler).__name__, name),
 137             ) as span:
 138                 batch_size = len(remaining)
 139                 span.set_data("Batch Size", batch_size)
 140                 span.set_data("Feature Name", name)
 141                 span.set_data("Handler Type", type(handler).__name__)
 142 
 143                 batch = FeatureCheckBatch(self, name, organization, remaining, actor)
 144                 handler_result = handler.has_for_batch(batch)
 145                 for (obj, flag) in handler_result.items():
 146                     if flag is not None:
 147                         remaining.remove(obj)
 148                         result[obj] = flag
 149                 span.set_data("Flags Found", batch_size - len(remaining))
 150 
 151         default_flag = settings.SENTRY_FEATURES.get(name, False)
 152         for obj in remaining:
 153             result[obj] = default_flag
 154 
 155         return result

 158 class FeatureCheckBatch(object):
 159     """
 160     A batch of objects to be checked for a feature flag.
 161 
 162     An instance of this class encapsulates a call to
 163     ``FeatureManager.has_for_batch``. The objects (such as projects) have a
 164     common parent organization.
 165     """
 166 
 167     def __init__(self, manager, name, organization, objects, actor):
 168         self._manager = manager
 169         self.feature_name = name
 170         self.organization = organization
 171         self.objects = objects
 172         self.actor = actor
 173 
 174     def get_feature_objects(self):
 175         """
 176         Iterate over individual Feature objects.
 177 
 178         This is a fallback mode for applying a FeatureHandler that doesn't
 179         support checking the entire batch at once.
 180         """
 181 
 182         cls = self._manager._get_feature_class(self.feature_name)
 183         return {obj: cls(self.feature_name, obj) for obj in self.objects}
