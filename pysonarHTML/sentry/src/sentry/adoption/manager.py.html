   3 from collections import defaultdict, namedtuple

   5 FEATURE_LOCATION = {"language", "integration", "code", "web", "admin", "api"}
   6 
   7 Feature = namedtuple("Feature", ["id", "slug", "name", "location", "prerequisite"])
   8 
   9 
  10 class UnknownFeature(KeyError):
  11     pass

  14 class AdoptionManager(object):
  15     def __init__(self):
  16         self._id_registry = {}
  17         self._slug_registry = {}
  18         self._integration_slugs = defaultdict(set)
  19         self._location_slugs = defaultdict(set)
  20         self._slugs = set()

  23     def add(self, id, slug, name, location, prerequisite=None):
  24         assert location in FEATURE_LOCATION
  25 
  26         feature = Feature(id, slug, name, location, prerequisite)
  27         self._id_registry[id] = feature
  28         self._slug_registry[slug] = feature
  29         self._slugs.add(slug)
  30         self._ids.add(id)
  31         self._location_slugs[location].add(slug)
  32 
  33         if location == "integration":
  34             self._integration_slugs[prerequisite[0]].add(slug)
  35 
  36     def get_by_id(self, id):
  37         try:
  38             return self._id_registry[id]
  39         except KeyError:
  40             raise UnknownFeature(id)
  41 
  42     def get_by_slug(self, slug):
  43         try:
  44             return self._slug_registry[slug]
  45         except KeyError:
  46             raise UnknownFeature(slug)
  47 
  48     def all(self):
  49         return [self._id_registry[k] for k in self._id_registry]
  50 
  51     def location_slugs(self, location):
  52         return self._location_slugs[location]
  53 
  54     def integration_slugs(self, language):
  55         return self._integration_slugs[language]
