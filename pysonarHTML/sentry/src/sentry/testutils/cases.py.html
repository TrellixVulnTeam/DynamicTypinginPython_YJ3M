 389 class _AssertQueriesContext(CaptureQueriesContext):
 390     def __init__(self, test_case, queries, debug, connection):
 391         self.test_case = test_case
 392         self.queries = queries
 393         self.debug = debug
 394         super(_AssertQueriesContext, self).__init__(connection)
 395 
 396     def __exit__(self, exc_type, exc_value, traceback):
 397         super(_AssertQueriesContext, self).__exit__(exc_type, exc_value, traceback)
 398         if exc_type is not None:
 399             return
 400 
 401         parsed_queries = parse_queries(self.captured_queries)
 402 
 403         if self.debug:
 404             import pprint
 405 
 406             pprint.pprint("====================== Raw Queries ======================")
 407             pprint.pprint(self.captured_queries)
 408             pprint.pprint("====================== Table writes ======================")
 409             pprint.pprint(parsed_queries)
 410 
 411         for table, num in parsed_queries.items():
 412             expected = self.queries.get(table, 0)
 413             if expected == 0:
 414                 import pprint
 415 
 416                 pprint.pprint(
 417                     "WARNING: no query against %s emitted, add debug=True to see all the queries"
 418                     % (table)
 419                 )
 420             else:
 421                 self.test_case.assertTrue(
 422                     num == expected,
 423                     "%d write queries expected on `%s`, got %d, add debug=True to see all the queries"
 424                     % (expected, table, num),
 425                 )
 426 
 427         for table, num in self.queries.items():
 428             executed = parsed_queries.get(table, None)
 429             self.test_case.assertFalse(
 430                 executed is None,
 431                 "no query against %s emitted, add debug=True to see all the queries" % (table),
 432             )

 445     endpoint = None
 446     method = "get"

 448     def get_response(self, *args, **params):
 449         if self.endpoint is None:
 450             raise Exception("Implement self.endpoint to use this method.")
 451 
 452         url = reverse(self.endpoint, args=args)
 453         # In some cases we want to pass querystring params to put/post, handle
 454         # this here.
 455         if "qs_params" in params:
 456             query_string = urlencode(params.pop("qs_params"), doseq=True)
 457             url = u"{}?{}".format(url, query_string)
 458 
 459         method = params.pop("method", self.method)
 460 
 461         return getattr(self.client, method)(url, format="json", data=params)
 462 
 463     def get_valid_response(self, *args, **params):
 464         status_code = params.pop("status_code", 200)
 465         resp = self.get_response(*args, **params)
 466         assert resp.status_code == status_code, (resp.status_code, resp.content)
 467         return resp
