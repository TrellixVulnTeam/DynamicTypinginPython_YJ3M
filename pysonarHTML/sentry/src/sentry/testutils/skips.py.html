  13 def snuba_is_available():
  14     if "snuba" in _service_status:
  15         return _service_status["snuba"]
  16     try:
  17         parsed = urlparse(settings.SENTRY_SNUBA)
  18         socket.create_connection((parsed.hostname, parsed.port), 1.0)
  19     except socket.error:
  20         _service_status["snuba"] = False
  21     else:
  22         _service_status["snuba"] = True

  40 def skip_for_relay_store(reason):
  41     """
  42     Decorator factory will skip marked tests if Relay is enabled.
  43     A test decorated with @skip_for_relay_store("this test has been moved in relay")
  44     Will not be executed when the settings SENTRY_USE_RELAY = True
  45     :param reason: the reason the test should be skipped
  46 
  47     Note: Eventually, when Relay becomes compulsory, tests marked with this decorator will be deleted.
  48     """
  49 
  50     def decorator(function):
  51         return pytest.mark.skipif(settings.SENTRY_USE_RELAY, reason=reason)(function)
  52 
  53     return decorator
