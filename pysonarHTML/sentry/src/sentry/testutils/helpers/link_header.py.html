  24 from __future__ import absolute_import
  25 
  26 __all__ = ("parse_link_header",)
  27 
  28 import re
  29 
  30 TOKEN = r'(?:[^\(\)<>@,;:\\"/\[\]\?={} \t]+?)'
  31 QUOTED_STRING = r'(?:"(?:\\"|[^"])*")'
  32 PARAMETER = r"(?:%(TOKEN)s(?:=(?:%(TOKEN)s|%(QUOTED_STRING)s))?)" % locals()
  33 LINK = r"<[^>]*>\s*(?:;\s*%(PARAMETER)s?\s*)*" % locals()
  34 COMMA = r"(?:\s*(?:,\s*)+)"
  35 LINK_SPLIT = r"%s(?=%s|\s*$)" % (LINK, COMMA)
  36 
  37 link_splitter = re.compile(LINK_SPLIT)
  38 
  39 
  40 def _unquotestring(instr):
  41     if instr[0] == instr[-1] == '"':
  42         instr = instr[1:-1]
  43         instr = re.sub(r"\\(.)", r"\1", instr)
  44     return instr
  45 
  46 
  47 def _splitstring(instr, item, split):
  48     if not instr:
  49         return []
  50     return [h.strip() for h in re.findall(r"%s(?=%s|\s*$)" % (item, split), instr)]
  51 
  52 
  53 def parse_link_header(instr):
  54     """
  55     Given a link-value (i.e., after separating the header-value on commas),
  56     return a dictionary whose keys are link URLs and values are dictionaries
  57     of the parameters for their associated links.
  58 
  59     Note that internationalised parameters (e.g., title*) are
  60     NOT percent-decoded.
  61 
  62     Also, only the last instance of a given parameter will be included.
  63 
  64     For example,
  65 
  66     >>> parse_link_value('</foo>; rel="self"; title*=utf-8\'de\'letztes%20Kapitel')
  67     {'/foo': {'title*': "utf-8'de'letztes%20Kapitel", 'rel': 'self'}}
  68 
  69     """
  70     out = {}
  71     if not instr:
  72         return out
  73 
  74     for link in [h.strip() for h in link_splitter.findall(instr)]:
  75         url, params = link.split(">", 1)
  76         url = url[1:]
  77         param_dict = {}
  78         for param in _splitstring(params, PARAMETER, "\s*;\s*"):
  79             try:
  80                 a, v = param.split("=", 1)
  81                 param_dict[a.lower()] = _unquotestring(v)
  82             except ValueError:
  83                 param_dict[param.lower()] = None
  84         out[url] = param_dict
  85     return out
