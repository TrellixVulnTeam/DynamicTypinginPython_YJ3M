  11 @contextmanager
  12 def Feature(names):
  13     """
  14     Control whether a feature is enabled.
  15 
  16     A single feature may be conveniently enabled with
  17 
  18     >>> with Feature('feature-1'):
  19     >>>   # Executes with feature-1 enabled
  20 
  21     More advanced enabling / disabling can be done using a dict
  22 
  23     >>> with Feature({'feature-1': True, 'feature-2': False}):
  24     >>>   # Executes with feature-1 enabled and feature-2 disabled
  25 
  26     The following two invocations are equivalent:
  27 
  28     >>> with Feature(['feature-1', 'feature-2']):
  29     >>>   # execute with both features enabled
  30     >>> with Feature({'feature-1': True, 'feature-2': True}):
  31     >>>   # execute with both features enabled
  32     """
  33     if isinstance(names, six.string_types):
  34         names = {names: True}
  35 
  36     elif not isinstance(names, collections.Mapping):
  37         names = {k: True for k in names}
  38 
  39     with patch("sentry.features.has") as features_has:
  40         features_has.side_effect = lambda x, *a, **k: names.get(x, False)
  41         yield
