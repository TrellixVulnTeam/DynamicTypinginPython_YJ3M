  11 class Faux(object):
  12     """
  13     Convenience functions for testing, and asserting, with ``unittest.mock``
  14     objects.
  15 
  16     Usage:
  17         >>> with patch('module.ClassName.func') as mock:
  18         >>>     ClassName.func('foo', extra={'bar': {'baz': 1}})
  19         >>>
  20         >>>     assert faux(mock).args == ('foo',)
  21         >>>     assert faux(mock).args_contain('foo')
  22         >>>     assert faux(mock).kwargs_contain('extra')
  23         >>>     assert faux(mock).kwargs_contain('extra.bar')
  24         >>>     assert faux(mock).kwarg_equals('extra.bar.baz', 1)
  25 
  26     Dot Notation:
  27         Functions that deal with ``kwargs`` will accept a dot notation
  28         shorthand for dealing with deeply nested ``dict`` keys.
  29 
  30         >>> with patch('module.ClassName.func') as mock:
  31         >>>     ClassName.func('foo', extra={'bar': {'baz': 1}})
  32         >>>
  33         >>>     assert faux(mock).kwargs_contain('extra.foo.baz')
  34         >>>     assert faux(mock).kwarg_equals('extra.foo.baz', 1)
  35 
  36     Multiple Calls
  37         By default, ``faux`` will select the last call to a ``mock``. You can
  38         specify which call you want by passing the index as a second parameter.
  39 
  40         >>> with patch('module.ClassName.func') as mock:
  41         >>>     ClassName.func('foo', extra={'bar': {'baz': 1}})
  42         >>>     ClassName.func(None)
  43         >>>
  44         >>>     assert faux(mock).args == (None,)
  45         >>>     assert faux(mock, 0).args == ('foo',)
  46     """
  47 
  48     def __init__(self, call):
  49         self.call = call
  50 
  51     @property
  52     def args(self):
  53         return self.call[1]
  54 
  55     @property
  56     def kwargs(self):
  57         return self.call[2]
  58 
  59     def called_with(self, *args, **kwargs):
  60         if self.args == tuple(args) and self.kwargs == kwargs:
  61             return True
  62 
  63         raise AssertionError(
  64             u"Expected to be called with {}. Received {}.".format(
  65                 self._invocation_to_s(*args, **kwargs),
  66                 self._invocation_to_s(*self.args, **self.kwargs),
  67             )
  68         )
  69 
  70     def kwargs_contain(self, key, **kwargs):
  71         if self._kwarg_exists(key, **kwargs):
  72             return True
  73 
  74         raise AssertionError(
  75             u"Expected kwargs to contain key '{}'. Received ({}).".format(
  76                 key, self._kwargs_to_s(**self.kwargs)
  77             )
  78         )
  79 
  80     def kwarg_equals(self, key, expected, **kwargs):
  81         actual = self._kwarg_value(key, **kwargs)
  82 
  83         if actual == expected:
  84             return True
  85 
  86         raise AssertionError(
  87             u"Expected kwargs[{}] to equal {!r}. Received {!r}.".format(key, expected, actual)
  88         )
  89 
  90     def args_contain(self, value, **kwargs):
  91         if value in self.args:
  92             return True
  93 
  94         raise AssertionError(
  95             u"Expected args to contain {!r}. Received ({}).".format(
  96                 value, self._args_to_s(*self.args)
  97             )
  98         )
  99 
 100     def args_equals(self, *args, **kwargs):
 101         if self.args == tuple(args):
 102             return True
 103 
 104         raise AssertionError(
 105             u"Expected args to equal ({}). Received ({}).".format(
 106                 self._args_to_s(*args), self._args_to_s(*self.args)
 107             )
 108         )
 109 
 110     def _kwarg_exists(self, key):
 111         try:
 112             self._kwarg_value(key)
 113             return True
 114         except (KeyError, TypeError):
 115             return False
 116 
 117     def _kwarg_value(self, key, **kwargs):
 118         """
 119         Support a dot notation shortcut for deeply nested dicts or just look
 120         up the value if passed a normal key.
 121 
 122         >>> self.kwargs = {'foo': {'bar': {'baz': 1}}}
 123         >>> self._kwarg_value('foo.bar.baz')
 124         1
 125         >>> self._kwarg_value('foo')
 126         {'bar': {'baz': 1}}
 127         """
 128         if "." in key:
 129             keys = deque(key.split("."))
 130         else:
 131             kwarg = self.kwargs[key]
 132 
 133             if kwargs.get("format") == "json":
 134                 return json.loads(kwarg)
 135 
 136             return kwarg
 137 
 138         kwarg = dict(self.kwargs)
 139 
 140         if kwargs.get("format") == "json":
 141             kwarg = json.loads(kwarg[keys.popleft()])
 142 
 143         while keys:
 144             kwarg = kwarg[keys.popleft()]
 145 
 146         return kwarg
 147 
 148     def _invocation_to_s(self, *args, **kwargs):
 149         """
 150         Convert a function invocation into a pretty printable string.
 151         """
 152         return u"({})".format(
 153             ", ".join(compact([self._args_to_s(*args), self._kwargs_to_s(**kwargs)]))
 154         )
 155 
 156     def _args_to_s(self, *args):
 157         if not len(args):
 158             return None
 159         return ", ".join(u"{!r}".format(arg) for arg in args)
 160 
 161     def _kwargs_to_s(self, **kwargs):
 162         if not len(kwargs):
 163             return None
 164         return ", ".join(u"{}={!r}".format(k, v) for k, v in six.iteritems(kwargs))

 167 class Mock(object):
 168     def __init__(self, *args, **kwargs):
 169         for k, v in six.iteritems(kwargs):
 170             setattr(self, k, v)

 173 def faux(mock, call=None):
 174     if call is not None:
 175         return Faux(mock.mock_calls[call])
 176     else:
 177         return Faux(mock.mock_calls[-1])
