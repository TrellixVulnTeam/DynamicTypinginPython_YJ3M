 215     manifest["release"] = release
 216     if project:

 515     @staticmethod
 516     def store_event(data, project_id, assert_no_errors=True, sent_at=None):
 517         # Like `create_event`, but closer to how events are actually
 518         # ingested. Prefer to use this method over `create_event`
 519         manager = EventManager(data, sent_at=sent_at)
 520         manager.normalize()
 521         if assert_no_errors:
 522             errors = manager.get_data().get("errors")
 523             assert not errors, errors
 524 
 525         event = manager.save(project_id)
 526         if event.group:
 527             event.group.save()
 528         return event
