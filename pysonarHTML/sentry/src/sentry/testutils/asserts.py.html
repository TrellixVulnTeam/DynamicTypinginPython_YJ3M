   5 def assert_mock_called_once_with_partial(mock, *args, **kwargs):
   6     """
   7     Similar to ``mock.assert_called_once_with()``, but we dont require all
   8     args and kwargs to be specified.
   9     """
  10     assert len(mock.mock_calls) == 1
  11     m_args, m_kwargs = mock.call_args
  12     for i, arg in enumerate(args):
  13         assert m_args[i] == arg
  14     for kwarg in kwargs:
  15         assert m_kwargs[kwarg] == kwargs[kwarg]
