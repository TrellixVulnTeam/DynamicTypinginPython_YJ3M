  13 # Grammar is defined in EBNF syntax.
  14 ownership_grammar = Grammar(
  15     r"""
  16 
  17 ownership = line+
  18 
  19 line = _ (comment / rule / empty) newline?
  20 
  21 rule = _ matcher owners
  22 
  23 matcher      = _ matcher_tag any_identifier
  24 matcher_tag  = (matcher_type sep)?
  25 matcher_type = "url" / "path" / event_tag
  26 
  27 event_tag   = ~r"tags.[^:]+"
  28 
  29 owners       = _ owner+
  30 owner        = _ team_prefix identifier
  31 team_prefix  = "#"?
  32 
  33 comment = ~r"#[^\r\n]*"
  34 
  35 # TODO: make more specific
  36 any_identifier = quoted_identifier / identifier
  37 identifier = ~r"\S+"
  38 quoted_identifier = ~r'"([^"\\]*(?:\\.[^"\\]*)*)"'
  39 
  40 sep     = ":"
  41 space   = " "
  42 empty   = ""
  43 newline = ~r"[\r\n]"
  44 _       = space*
  45 
  46 """
  47 )

  50 class Rule(namedtuple("Rule", "matcher owners")):
  51     """
  52     A Rule represents a single line in an Ownership file.
  53     This line contains a Matcher and a list of Owners.
  54     """
  55 
  56     def dump(self):
  57         return {"matcher": self.matcher.dump(), "owners": [o.dump() for o in self.owners]}
  58 
  59     @classmethod
  60     def load(cls, data):
  61         return cls(Matcher.load(data["matcher"]), [Owner.load(o) for o in data["owners"]])
  62 
  63     def test(self, data):
  64         return self.matcher.test(data)

  67 class Matcher(namedtuple("Matcher", "type pattern")):
  68     """
  69     A Matcher represents a type:pattern pairing for use in
  70     comparing with an Event.
  71 
  72     type is either `path` or `url` at this point.
  73 
  74     TODO(mattrobenolt): pattern needs to be parsed into a regex
  75 
  76     Examples:
  77         url:example.com
  78         path:src/*
  79         src/*
  80     """
  81 
  82     def dump(self):
  83         return {"type": self.type, "pattern": self.pattern}
  84 
  85     @classmethod
  86     def load(cls, data):
  87         return cls(data["type"], data["pattern"])
  88 
  89     def test(self, data):
  90         if self.type == "url":
  91             return self.test_url(data)
  92         elif self.type == "path":
  93             return self.test_path(data)
  94         elif self.type.startswith("tags."):
  95             return self.test_tag(data)
  96         return False
  97 
  98     def test_url(self, data):
  99         try:
 100             url = data["request"]["url"]
 101         except KeyError:
 102             return False
 103         return url and glob_match(url, self.pattern, ignorecase=True)
 104 
 105     def test_path(self, data):
 106         for frame in _iter_frames(data):
 107             filename = frame.get("filename") or frame.get("abs_path")
 108 
 109             if not filename:
 110                 continue
 111 
 112             if glob_match(filename, self.pattern, ignorecase=True, path_normalize=True):
 113                 return True
 114 
 115         return False
 116 
 117     def test_tag(self, data):
 118         tag = self.type[5:]
 119         for k, v in data.get("tags"):
 120             if k == tag and glob_match(v, self.pattern):
 121                 return True
 122         return False

 125 class Owner(namedtuple("Owner", "type identifier")):
 126     """
 127     An Owner represents a User or Team who owns this Rule.
 128 
 129     type is either `user` or `team`.
 130 
 131     Examples:
 132         foo@example.com
 133         #team
 134     """
 135 
 136     def dump(self):
 137         return {"type": self.type, "identifier": self.identifier}
 138 
 139     @classmethod
 140     def load(cls, data):
 141         return cls(data["type"], data["identifier"])

 144 class OwnershipVisitor(NodeVisitor):
 145     visit_comment = visit_empty = lambda *a: None
 146 
 147     def visit_ownership(self, node, children):
 148         return [_f for _f in children if _f]
 149 
 150     def visit_line(self, node, children):
 151         _, line, _ = children
 152         comment_or_rule_or_empty = line[0]
 153         if comment_or_rule_or_empty:
 154             return comment_or_rule_or_empty
 155 
 156     def visit_rule(self, node, children):
 157         _, matcher, owners = children
 158         return Rule(matcher, owners)
 159 
 160     def visit_matcher(self, node, children):
 161         _, tag, identifier = children
 162         return Matcher(tag, identifier)
 163 
 164     def visit_matcher_tag(self, node, children):
 165         if not children:
 166             return "path"
 167         (tag,) = children
 168         type, _ = tag
 169         return type[0].text
 170 
 171     def visit_owners(self, node, children):
 172         _, owners = children
 173         return owners
 174 
 175     def visit_owner(self, node, children):
 176         _, is_team, pattern = children
 177         type = "team" if is_team else "user"
 178         # User emails are case insensitive, so coerce them
 179         # to lowercase, so they can be de-duped, etc.
 180         if type == "user":
 181             pattern = pattern.lower()
 182         return Owner(type, pattern)
 183 
 184     def visit_team_prefix(self, node, children):
 185         return bool(children)
 186 
 187     def visit_any_identifier(self, node, children):
 188         return children[0]
 189 
 190     def visit_identifier(self, node, children):
 191         return node.text
 192 
 193     def visit_quoted_identifier(self, node, children):
 194         return node.text[1:-1].encode("ascii", "backslashreplace").decode("unicode-escape")
 195 
 196     def generic_visit(self, node, children):
 197         return children or node

 200 def _iter_frames(data):
 201     try:
 202         for frame in get_path(data, "stacktrace", "frames", filter=True) or ():
 203             yield frame
 204     except KeyError:
 205         pass
 206 
 207     try:
 208         values = get_path(data, "exception", "values", filter=True) or ()
 209     except KeyError:
 210         return
 211 
 212     for value in values:
 213         try:
 214             for frame in get_path(value, "stacktrace", "frames", filter=True) or ():
 215                 yield frame
 216         except KeyError:
 217             continue

 220 def parse_rules(data):
 221     """Convert a raw text input into a Rule tree"""
 222     tree = ownership_grammar.parse(data)
 223     return OwnershipVisitor().visit(tree)

 226 def dump_schema(rules):
 227     """Convert a Rule tree into a JSON schema"""
 228     return {"$version": VERSION, "rules": [r.dump() for r in rules]}
 229 
 230 
 231 def load_schema(schema):
 232     """Convert a JSON schema into a Rule tree"""
 233     if schema["$version"] != VERSION:
 234         raise RuntimeError("Invalid schema $version: %r" % schema["$version"])
 235     return [Rule.load(r) for r in schema["rules"]]
