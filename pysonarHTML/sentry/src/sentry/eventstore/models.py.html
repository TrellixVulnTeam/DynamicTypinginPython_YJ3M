 113     @property
 114     def tags(self):
 115         """
 116         Tags property uses tags from snuba if loaded otherwise falls back to
 117         nodestore.
 118         """
 119         tags_key_column = self.__get_column_name(Columns.TAGS_KEY)
 120         tags_value_column = self.__get_column_name(Columns.TAGS_VALUE)
 121 
 122         if tags_key_column in self._snuba_data and tags_value_column in self._snuba_data:
 123             keys = self._snuba_data[tags_key_column]
 124             values = self._snuba_data[tags_value_column]
 125             if keys and values and len(keys) == len(values):
 126                 return sorted(zip(keys, values))
 127             else:
 128                 return []
 129         # Nodestore implementation
 130         try:
 131             rv = sorted(
 132                 [
 133                     (t, v)
 134                     for t, v in get_path(self.data, "tags", filter=True) or ()
 135                     if t is not None and v is not None
 136                 ]
 137             )
 138             return rv
 139         except ValueError:
 140             # at one point Sentry allowed invalid tag sets such as (foo, bar)
 141             # vs ((tag, foo), (tag, bar))
 142             return []

 173     def get_minimal_user(self):
 174         """
 175         A minimal 'User' interface object that gives us enough information
 176         to render a user badge.
 177         """
 178         from sentry.interfaces.user import User
 179 
 180         user_id_column = self.__get_column_name(Columns.USER_ID)
 181         user_email_column = self.__get_column_name(Columns.USER_EMAIL)
 182         user_username_column = self.__get_column_name(Columns.USER_USERNAME)
 183         user_ip_address_column = self.__get_column_name(Columns.USER_IP_ADDRESS)
 184 
 185         if all(
 186             key in self._snuba_data
 187             for key in [
 188                 user_id_column,
 189                 user_email_column,
 190                 user_username_column,
 191                 user_ip_address_column,

 193         ):
 194             user_id = self._snuba_data[user_id_column]
 195             email = self._snuba_data[user_email_column]
 196             username = self._snuba_data[user_username_column]
 197             ip_address = self._snuba_data[user_ip_address_column]
 198 
 199             return User.to_python(
 200                 {"id": user_id, "email": email, "username": username, "ip_address": ip_address}

 257     def generate_node_id(cls, project_id, event_id):
 258         """
 259         Returns a deterministic node_id for this event based on the project_id
 260         and event_id which together are globally unique. The event body should
 261         be saved under this key in nodestore so it can be retrieved using the
 262         same generated id when we only have project_id and event_id.
 263         """
 264         return md5(u"{}:{}".format(project_id, event_id).encode("utf-8")).hexdigest()

 328     def get_hashes(self, force_config=None):
 329         """
 330         Returns the calculated hashes for the event.  This uses the stored
 331         information if available.  Grouping hashes will take into account
 332         fingerprinting and checksums.
 333         """
 334         # If we have hashes stored in the data we use them, otherwise we
 335         # fall back to generating new ones from the data.  We can only use
 336         # this if we do not force a different config.
 337         if force_config is None:
 338             hashes = self.data.get("hashes")
 339             if hashes is not None:
 340                 return hashes
 341 
 342         return [
 343             _f
 344             for _f in [x.get_hash() for x in self.get_grouping_variants(force_config).values()]
 345             if _f
 346         ]

 348     def get_grouping_variants(self, force_config=None, normalize_stacktraces=False):
 349         """
 350         This is similar to `get_hashes` but will instead return the
 351         grouping components for each variant in a dictionary.
 352 
 353         If `normalize_stacktraces` is set to `True` then the event data will be
 354         modified for `in_app` in addition to event variants being created.  This
 355         means that after calling that function the event data has been modified
 356         in place.
 357         """
 358         from sentry.grouping.api import get_grouping_variants_for_event, load_grouping_config
 359         from sentry.stacktraces.processing import normalize_stacktraces_for_grouping
 360 
 361         # Forcing configs has two separate modes.  One is where just the
 362         # config ID is given in which case it's merged with the stored or
 363         # default config dictionary
 364         if force_config is not None:
 365             if isinstance(force_config, six.string_types):
 366                 stored_config = self.get_grouping_config()
 367                 config = dict(stored_config)
 368                 config["id"] = force_config
 369             else:
 370                 config = force_config
 371 
 372         # Otherwise we just use the same grouping config as stored.  if
 373         # this is None the `get_grouping_variants_for_event` will fill in
 374         # the default.
 375         else:
 376             config = self.data.get("grouping_config")
 377 
 378         config = load_grouping_config(config)
 379         if normalize_stacktraces:
 380             normalize_stacktraces_for_grouping(self.data, config)
 381 
 382         return get_grouping_variants_for_event(self, config)

 414     def as_dict(self):
 415         """Returns the data in normalized form for external consumers."""
 416         # We use a OrderedDict to keep elements ordered for a potential JSON serializer
 417         data = OrderedDict()
 418         data["event_id"] = self.event_id
 419         data["project"] = self.project_id
 420         data["release"] = self.release
 421         data["dist"] = self.dist
 422         data["platform"] = self.platform
 423         data["message"] = self.message
 424         data["datetime"] = self.datetime
 425         data["tags"] = [(k.split("sentry:", 1)[-1], v) for (k, v) in self.tags]
 426         for k, v in sorted(six.iteritems(self.data)):
 427             if k in data:
 428                 continue
 429             if k == "sdk":
 430                 v = {v_k: v_v for v_k, v_v in six.iteritems(v) if v_k != "client_ip"}
 431             data[k] = v
 432 
 433         # for a long time culprit was not persisted.  In those cases put
 434         # the culprit in from the group.
 435         if data.get("culprit") is None and self.group_id:
 436             data["culprit"] = self.group.culprit
 437 
 438         # Override title and location with dynamically generated data
 439         data["title"] = self.title
 440         data["location"] = self.location
 441 
 442         return data

 444     @memoize
 445     def search_message(self):
 446         """
 447         The internal search_message attribute is only used for search purposes.
 448         It adds a bunch of data from the metadata and the culprit.
 449         """
 450         data = self.data
 451         culprit = self.culprit
 452 
 453         event_metadata = self.get_event_metadata()
 454 
 455         if event_metadata is None:
 456             event_metadata = eventtypes.get(self.get_event_type())().get_metadata(self.data)
 457 
 458         message = ""
 459 
 460         if data.get("logentry"):
 461             message += data["logentry"].get("formatted") or data["logentry"].get("message") or ""
 462 
 463         if event_metadata:
 464             for value in six.itervalues(event_metadata):
 465                 value_u = force_text(value, errors="replace")
 466                 if value_u not in message:
 467                     message = u"{} {}".format(message, value_u)
 468 
 469         if culprit and culprit not in message:
 470             culprit_u = force_text(culprit, errors="replace")
 471             message = u"{} {}".format(message, culprit_u)
 472 
 473         return trim(message.strip(), settings.SENTRY_MAX_MESSAGE_LENGTH)

 484 class EventSubjectTemplateData(object):
 485     tag_aliases = {"release": "sentry:release", "dist": "sentry:dist", "user": "sentry:user"}
 486 
 487     def __init__(self, event):
 488         self.event = event
 489 
 490     def __getitem__(self, name):
 491         if name.startswith("tag:"):
 492             name = name[4:]
 493             value = self.event.get_tag(self.tag_aliases.get(name, name))
 494             if value is None:
 495                 raise KeyError
 496             return six.text_type(value)
 497         elif name == "project":
 498             return self.event.project.get_full_name()
 499         elif name == "projectID":
 500             return self.event.project.slug
 501         elif name == "shortID" and self.event.group_id:
 502             return self.event.group.qualified_short_id
 503         elif name == "orgID":
 504             return self.event.organization.slug
 505         elif name == "title":
 506             return self.event.title
 507         raise KeyError
