  91     def __get_events(
  92         self,
  93         filter,  # NOQA
  94         orderby=None,
  95         limit=DEFAULT_LIMIT,
  96         offset=DEFAULT_OFFSET,
  97         referrer=None,
  98         should_bind_nodes=False,
  99     ):
 100         assert filter, "You must provide a filter"  # NOQA
 101         cols = self.__get_columns()
 102         orderby = orderby or DESC_ORDERING
 103 
 104         # This is an optimization for the Group.filter_by_event_id query where we
 105         # have a single event ID and want to check all accessible projects for a
 106         # direct hit. In this case it's usually faster to go to nodestore first.
 107         if (
 108             filter.event_ids
 109             and filter.project_ids
 110             and len(filter.event_ids) * len(filter.project_ids) < min(limit, NODESTORE_LIMIT)
 111             and offset == 0
 112             and should_bind_nodes
 113         ):
 114             event_list = [
 115                 Event(project_id=project_id, event_id=event_id)
 116                 for event_id in filter.event_ids
 117                 for project_id in filter.project_ids
 118             ]
 119             self.bind_nodes(event_list)
 120 
 121             nodestore_events = [event for event in event_list if len(event.data)]
 122 
 123             if nodestore_events:
 124                 event_ids = {event.event_id for event in nodestore_events}
 125                 project_ids = {event.project_id for event in nodestore_events}
 126                 start = min(event.datetime for event in nodestore_events)
 127                 end = max(event.datetime for event in nodestore_events) + timedelta(seconds=1)
 128 
 129                 result = snuba.aliased_query(
 130                     selected_columns=cols,
 131                     start=start,
 132                     end=end,
 133                     conditions=filter.conditions,
 134                     filter_keys={"project_id": project_ids, "event_id": event_ids},
 135                     orderby=orderby,
 136                     limit=len(nodestore_events),
 137                     offset=DEFAULT_OFFSET,
 138                     referrer=referrer,
 139                     dataset=snuba.Dataset.Events,
 140                 )
 141 
 142                 if "error" not in result:
 143                     events = [self.__make_event(evt) for evt in result["data"]]
 144 
 145                     # Bind previously fetched node data
 146                     nodestore_dict = {
 147                         (e.event_id, e.project_id): e.data.data for e in nodestore_events
 148                     }
 149                     for event in events:
 150                         node_data = nodestore_dict[(event.event_id, event.project_id)]
 151                         event.data.bind_data(node_data)
 152                     return events
 153 
 154             return []
 155 
 156         result = snuba.aliased_query(
 157             selected_columns=cols,
 158             start=filter.start,
 159             end=filter.end,
 160             conditions=filter.conditions,
 161             filter_keys=filter.filter_keys,
 162             orderby=orderby,
 163             limit=limit,
 164             offset=offset,
 165             referrer=referrer,
 166             dataset=snuba.Dataset.Events,
 167         )
 168 
 169         if "error" not in result:
 170             events = [self.__make_event(evt) for evt in result["data"]]
 171             if should_bind_nodes:
 172                 self.bind_nodes(events)
 173             return events
 174 
 175         return []

 177     def get_event_by_id(self, project_id, event_id):
 178         """
 179         Get an event given a project ID and event ID
 180         Returns None if an event cannot be found
 181         """
 182         event_id = normalize_event_id(event_id)
 183 
 184         if not event_id:
 185             return None
 186 
 187         event = Event(project_id=project_id, event_id=event_id)
 188 
 189         # Return None if there was no data in nodestore
 190         if len(event.data) == 0:
 191             return None
 192 
 193         # Load group_id from Snuba if not a transaction
 194         if event.get_event_type() != "transaction":
 195             result = snuba.raw_query(
 196                 selected_columns=["group_id"],
 197                 start=event.datetime,
 198                 end=event.datetime + timedelta(seconds=1),
 199                 filter_keys={"project_id": [project_id], "event_id": [event_id]},
 200                 limit=1,
 201                 referrer="eventstore.get_event_by_id_nodestore",
 202             )
 203 
 204             # Return None if the event from Nodestore was not yet written to Snuba
 205             if len(result["data"]) != 1:
 206                 logger.warning(
 207                     "eventstore.missing-snuba-event",
 208                     extra={"project_id": project_id, "event_id": event_id},
 209                 )
 210                 return None
 211 
 212             event.group_id = result["data"][0]["group_id"]
 213 
 214         return event

 269     def __get_event_id_from_filter(self, filter=None, orderby=None):  # NOQA
 270         columns = [Columns.EVENT_ID.value.alias, Columns.PROJECT_ID.value.alias]
 271 
 272         try:
 273             # This query uses the discover dataset to enable
 274             # getting events across both errors and transactions, which is
 275             # required when doing pagination in discover
 276             result = snuba.aliased_query(
 277                 selected_columns=columns,
 278                 conditions=filter.conditions,
 279                 filter_keys=filter.filter_keys,
 280                 start=filter.start,
 281                 end=filter.end,
 282                 limit=1,
 283                 referrer="eventstore.get_next_or_prev_event_id",
 284                 orderby=orderby,
 285                 dataset=snuba.Dataset.Discover,
 286             )
 287         except (snuba.QueryOutsideRetentionError, snuba.QueryOutsideGroupActivityError):
 288             # This can happen when the date conditions for paging
 289             # and the current event generate impossible conditions.
 290             return None
 291 
 292         if "error" in result or len(result["data"]) == 0:
 293             return None
 294 
 295         row = result["data"][0]
 296 
 297         return (six.text_type(row["project_id"]), six.text_type(row["event_id"]))

 299     def __make_event(self, snuba_data):
 300         event_id = snuba_data[Columns.EVENT_ID.value.event_name]
 301         group_id = snuba_data[Columns.GROUP_ID.value.event_name]
 302         project_id = snuba_data[Columns.PROJECT_ID.value.event_name]
 303 
 304         return Event(
 305             event_id=event_id, group_id=group_id, project_id=project_id, snuba_data=snuba_data
