  14 class Filter(object):
  15     """
  16     A set of conditions, start/end times and project, group and event ID sets
  17     used to restrict the results of a Snuba query.
  18 
  19     start (DateTime): Start datetime - default None
  20     end (DateTime): Start datetime - default None
  21     conditions (Sequence[Sequence[str, str, Any]]): List of conditions to fetch - default None
  22     having (Sequence[str, str, Any]]): List of having conditions to filter by - default None
  23     project_ids (Sequence[int]): List of project IDs to fetch - default None
  24     group_ids (Sequence[int]): List of group IDs to fetch - default None
  25     event_ids (Sequence[int]): List of event IDs to fetch - default None
  26 
  27     selected_columns (Sequence[str]): List of columns to select
  28     aggregations (Sequence[Any, str|None, str]): Aggregate functions to fetch.
  29     groupby (Sequence[str]): List of columns to group results by
  30     """
  31 
  32     def __init__(
  33         self,
  34         start=None,
  35         end=None,
  36         conditions=None,
  37         having=None,
  38         project_ids=None,
  39         group_ids=None,
  40         event_ids=None,
  41         selected_columns=None,
  42         aggregations=None,
  43         rollup=None,
  44         groupby=None,
  45         orderby=None,
  46     ):
  47         self.start = start
  48         self.end = end
  49         self.conditions = conditions
  50         self.having = having
  51         self.project_ids = project_ids
  52         self.group_ids = group_ids
  53         self.event_ids = event_ids
  54 
  55         self.rollup = rollup
  56         self.selected_columns = selected_columns if selected_columns is not None else []
  57         self.aggregations = aggregations if aggregations is not None else []
  58         self.groupby = groupby
  59         self.orderby = orderby
  60 
  61     @property
  62     def filter_keys(self):
  63         """
  64         Get filter_keys value required for raw snuba query
  65         """
  66         filter_keys = {}
  67 
  68         if self.project_ids:
  69             filter_keys["project_id"] = self.project_ids
  70 
  71         if self.group_ids:
  72             filter_keys["group_id"] = self.group_ids
  73 
  74         if self.event_ids:
  75             filter_keys["event_id"] = self.event_ids
  76 
  77         return filter_keys
  78 
  79     @property
  80     def date_params(self):
  81         """
  82         Get the datetime parameters as a dictionary
  83         """
  84         return {"start": self.start, "end": self.end}
  85 
  86     def update_with(self, updates):
  87         keys = ("selected_columns", "aggregations", "conditions", "orderby", "groupby")
  88         for key in keys:
  89             if key in updates:
  90                 setattr(self, key, updates[key])

  92     def clone(self):
  93         return deepcopy(self)

 222     def bind_nodes(self, object_list, node_name="data"):
 223         """
 224         For a list of Event objects, and a property name where we might find an
 225         (unfetched) NodeData on those objects, fetch all the data blobs for
 226         those NodeDatas with a single multi-get command to nodestore, and bind
 227         the returned blobs to the NodeDatas.
 228 
 229         It's not necessary to bind a single Event object since data will be lazily
 230         fetched on any attempt to access a property.
 231         """
 232         with sentry_sdk.start_span(op="eventstore.base.bind_nodes"):
 233             object_node_list = [
 234                 (i, getattr(i, node_name)) for i in object_list if getattr(i, node_name).id
 235             ]
 236 
 237             # Remove duplicates from the list of nodes to be fetched
 238             node_ids = list({n.id for _, n in object_node_list})
 239             if not node_ids:
 240                 return
 241 
 242             node_results = nodestore.get_multi(node_ids)
 243 
 244             for item, node in object_node_list:
 245                 data = node_results.get(node.id) or {}
 246                 node.bind_data(data, ref=node.get_ref(item))
