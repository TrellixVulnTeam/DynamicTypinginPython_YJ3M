  30 def validate_phone(phone):
  31     try:
  32         p = phonenumbers.parse(phone, DEFAULT_REGION)
  33     except phonenumbers.NumberParseException:
  34         return False
  35     if not phonenumbers.is_possible_number(p):
  36         return False
  37     if not phonenumbers.is_valid_number(p):
  38         return False
  39     return True
  40 
  41 
  42 def clean_phone(phone):
  43     # This could raise, but should have been checked with validate_phone first
  44     return phonenumbers.format_number(
  45         phonenumbers.parse(phone, DEFAULT_REGION), phonenumbers.PhoneNumberFormat.E164
  46     )

  49 # XXX: can likely remove the dedupe here after notify_users has test coverage;
  50 #      in theory only cleaned data would make it to the plugin via the form,
  51 #      and cleaned numbers are deduped already.
  52 def split_sms_to(data):
  53     return set(filter(bool, re.split(r"\s*,\s*|\s+", data)))

  73         required=True,

  78     def clean_sms_from(self):
  79         data = self.cleaned_data["sms_from"]
  80         if not validate_phone(data):
  81             raise forms.ValidationError("{0} is not a valid phone number.".format(data))
  82         return clean_phone(data)
  83 
  84     def clean_sms_to(self):
  85         data = self.cleaned_data["sms_to"]
  86         phones = split_sms_to(data)
  87         if len(phones) > 10:
  88             raise forms.ValidationError(
  89                 "Max of 10 phone numbers, {0} were given.".format(len(phones))
  90             )
  91         for phone in phones:
  92             if not validate_phone(phone):
  93                 raise forms.ValidationError("{0} is not a valid phone number.".format(phone))
  94         return ",".join(sorted(set(map(clean_phone, phones))))
  95 
  96     def clean(self):
  97         # TODO: Ping Twilio and check credentials (?)
  98         return self.cleaned_data

 152     def error_message_from_json(self, data):
 153         code = data.get("code")
 154         message = data.get("message")
 155         more_info = data.get("more_info")
 156         error_message = "%s - %s %s" % (code, message, more_info)
 157         if message:
 158             return error_message
 159         return None
 160 
 161     def notify_users(self, group, event, **kwargs):
 162         if not self.is_configured(group.project):
 163             return
 164         project = group.project
 165 
 166         body = "Sentry [{0}] {1}: {2}".format(
 167             project.name.encode("utf-8"),
 168             event.group.get_level_display().upper().encode("utf-8"),
 169             event.title.encode("utf-8").splitlines()[0],
 170         )
 171         body = body[:MAX_SMS_LENGTH]
 172 
 173         client = self.get_client(group.project)
 174 
 175         payload = {"From": client.sms_from, "Body": body}
 176 
 177         errors = []
 178 
 179         for phone in client.sms_to:
 180             if not phone:
 181                 continue
 182             try:
 183                 # TODO: Use API client with raise_error
 184                 phone = clean_phone(phone)
 185                 payload = payload.copy()
 186                 payload["To"] = phone
 187                 client.request(payload)
 188             except Exception as e:
 189                 errors.append(e)
 190 
 191         if errors:
 192             self.raise_error(errors[0])
 193 
 194     def get_client(self, project):
 195         account_sid = self.get_option("account_sid", project)
 196         auth_token = self.get_option("auth_token", project)
 197         sms_from = clean_phone(self.get_option("sms_from", project))
 198         sms_to = self.get_option("sms_to", project)
 199         sms_to = split_sms_to(sms_to)
 200         return TwilioApiClient(account_sid, auth_token, sms_from, sms_to)
