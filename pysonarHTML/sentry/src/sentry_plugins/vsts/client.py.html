  19     def request(self, method, path, data=None, params=None):
  20         headers = {
  21             "Accept": "application/json; api-version={}".format(self.api_version),
  22             "Content-Type": "application/json-patch+json"
  23             if method == "PATCH"
  24             else "application/json",
  25             "X-HTTP-Method-Override": method,
  26             "X-TFS-FedAuthRedirect": "Suppress",
  27         }
  28         return self._request(method, path, headers=headers, data=data, params=params)

  55     def update_work_item(
  56         self, instance, id, title=UNSET, description=UNSET, link=UNSET, comment=UNSET
  57     ):
  58         data = []
  59 
  60         for f_name, f_value in (("title", title), ("description", description), ("link", link)):
  61             if f_name == "link":
  62                 # XXX: Link is not yet used, as we can't explicitly bind it to Sentry.
  63                 continue
  64             elif f_value is None:
  65                 data.append({"op": "remove", "path": FIELD_MAP[f_name]})
  66             elif f_value is not UNSET:
  67                 data.append(
  68                     {
  69                         # TODO(dcramer): this is problematic when the link already exists
  70                         "op": "replace" if f_name != "link" else "add",
  71                         "path": FIELD_MAP[f_name],
  72                         "value": {"rel": "Hyperlink", "url": f_value}
  73                         if f_name == "link"
  74                         else f_value,
  75                     }
  76                 )
  77 
  78         if comment is not UNSET and comment:
  79             data.append({"op": "add", "path": FIELD_MAP["comment"], "value": comment})
  80 
  81         return self.patch(
  82             "https://{}/DefaultCollection/_apis/wit/workitems/{}".format(instance, id), data=data
  83         )

  95     def get_commits(self, instance, repo_id, commit, limit=100):
  96         return self.get(
  97             "https://{}/DefaultCollection/_apis/git/repositories/{}/commits".format(
  98                 instance, repo_id
  99             ),
 100             params={"commit": commit, "$top": limit},

 103     def get_commit_filechanges(self, instance, repo_id, commit):
 104 
 105         resp = self.get(
 106             "https://{}/DefaultCollection/_apis/git/repositories/{}/commits/{}/changes".format(
 107                 instance, repo_id, commit
 108             )
 109         )
 110         changes = resp["changes"]

 112 
 113     def get_commit_range(self, instance, repo_id, start_sha, end_sha):
 114         return self.post(
 115             "https://{}/DefaultCollection/_apis/git/repositories/{}/commitsBatch".format(
 116                 instance, repo_id
 117             ),
 118             data={
 119                 "itemVersion": {"versionType": "commit", "version": start_sha},
 120                 "compareVersion": {"versionType": "commit", "version": end_sha},
 121             },
 122         )
 123 
 124     def get_projects(self, instance):
