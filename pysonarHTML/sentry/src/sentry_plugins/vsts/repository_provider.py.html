  36     def validate_config(self, organization, config, actor=None):
  37         if config.get("url"):
  38             client = self.get_client(actor)
  39             # parse out the repo name and the instance
  40             parts = urlparse(config["url"])
  41             instance = parts.netloc
  42             name = parts.path.rsplit("_git/", 1)[-1]
  43             project = config.get("project") or name
  44 
  45             try:
  46                 repo = client.get_repo(instance, name, project)
  47             except Exception as e:
  48                 self.raise_error(e, identity=client.auth)
  49             config.update(
  50                 {
  51                     "instance": instance,
  52                     "project": project,
  53                     "name": repo["name"],
  54                     "external_id": six.text_type(repo["id"]),
  55                     "url": repo["_links"]["web"]["href"],
  56                 }
  57             )
  58         return config

  78     def transform_changes(self, patch_set):
  79         type_mapping = {"add": "A", "delete": "D", "edit": "M"}
  80         file_changes = []
  81         # https://docs.microsoft.com/en-us/rest/api/vsts/git/commits/get%20changes#versioncontrolchangetype
  82         for change in patch_set:
  83             change_type = type_mapping.get(change["changeType"])
  84 
  85             if change_type and change.get("item") and change["item"]["gitObjectType"] == "blob":
  86                 file_changes.append({"path": change["item"]["path"], "type": change_type})
  87 
  88         return file_changes
  89 
  90     def zip_commit_data(self, repo, commit_list, actor):
  91         client = self.get_client(actor)
  92         n = 0
  93         for commit in commit_list:
  94             commit.update(
  95                 {
  96                     "patch_set": self.transform_changes(
  97                         client.get_commit_filechanges(
  98                             repo.config["instance"], repo.external_id, commit["commitId"]
  99                         )
 100                     )
 101                 }
 102             )
 103 
 104             n += 1
 105             if n > MAX_COMMIT_DATA_REQUESTS:
 106                 break
 107 
 108         return commit_list

 110     def compare_commits(self, repo, start_sha, end_sha, actor=None):
 111         if actor is None:
 112             raise NotImplementedError("Cannot fetch commits anonymously")
 113 
 114         client = self.get_client(actor)
 115         instance = repo.config["instance"]
 116 
 117         try:
 118             if start_sha is None:
 119                 res = client.get_commits(instance, repo.external_id, commit=end_sha, limit=10)
 120             else:
 121                 res = client.get_commit_range(instance, repo.external_id, start_sha, end_sha)
 122         except Exception as e:
 123             self.raise_error(e, identity=client.auth)
 124 
 125         commits = self.zip_commit_data(repo, res["value"], actor)
 126         return self._format_commits(repo, commits)
 127 
 128     def _format_commits(self, repo, commit_list):
 129         return [
 130             {
 131                 "id": c["commitId"],
 132                 "repository": repo.name,
 133                 "author_email": c["author"]["email"],
 134                 "author_name": c["author"]["name"],
 135                 "message": c["comment"],
 136                 "patch_set": c.get("patch_set"),
 137             }
 138             for c in commit_list
 139         ]
