  67     def is_configured(self, request, project, **kwargs):
  68         for o in ("instance",):
  69             if not bool(self.get_option(o, project)):
  70                 return False
  71         return True

  73     def get_issue_label(self, group, issue, **kwargs):
  74         return "Bug {}".format(issue["id"])
  75 
  76     def get_issue_url(self, group, issue, **kwargs):
  77         return issue["url"]

  79     def get_new_issue_fields(self, request, group, event, **kwargs):
  80         fields = super(VstsPlugin, self).get_new_issue_fields(request, group, event, **kwargs)
  81         client = self.get_client(request.user)
  82         instance = self.get_option("instance", group.project)
  83 
  84         try:
  85             projects = client.get_projects(instance)
  86         except Exception as e:
  87             self.raise_error(e, identity=client.auth)
  88 
  89         return [
  90             {
  91                 "name": "project",
  92                 "label": "Project",
  93                 "default": self.get_option("default_project", group.project),
  94                 "type": "text",
  95                 "choices": [i["name"] for i in projects["value"]],
  96                 "required": True,
  97             }
  98         ] + fields

 115     def create_issue(self, request, group, form_data, **kwargs):
 116         """
 117         Creates the issue on the remote service and returns an issue ID.
 118         """
 119         instance = self.get_option("instance", group.project)
 120         project = form_data.get("project") or self.get_option("default_project", group.project)
 121 
 122         client = self.get_client(request.user)
 123 
 124         title = form_data["title"]
 125         description = form_data["description"]
 126         link = absolute_uri(group.get_absolute_url(params={"referrer": "vsts_plugin"}))
 127         try:
 128             created_item = client.create_work_item(
 129                 instance=instance,
 130                 project=project,
 131                 title=title,
 132                 comment=markdown(description),
 133                 link=link,
 134             )
 135         except Exception as e:
 136             self.raise_error(e, identity=client.auth)
 137 
 138         return {
 139             "id": created_item["id"],
 140             "url": created_item["_links"]["html"]["href"],
 141             "title": title,
 142         }
 143 
 144     def link_issue(self, request, group, form_data, **kwargs):
 145         client = self.get_client(request.user)
 146         instance = self.get_option("instance", group.project)
 147         if form_data.get("comment"):
 148             try:
 149                 work_item = client.update_work_item(
 150                     instance=instance,
 151                     id=form_data["item_id"],
 152                     link=absolute_uri(group.get_absolute_url(params={"referrer": "vsts_plugin"})),
 153                     comment=markdown(form_data["comment"]) if form_data.get("comment") else None,
 154                 )
 155             except Exception as e:
 156                 self.raise_error(e, identity=client.auth)
 157         else:
 158             try:
 159                 work_item = client.get_work_item(instance=instance, id=form_data["item_id"])
 160             except Exception as e:
 161                 self.raise_error(e, identity=client.auth)
 162 
 163         return {
 164             "id": work_item["id"],
 165             "url": work_item["_links"]["html"]["href"],
 166             "title": work_item["fields"]["System.Title"],
 167         }
 168
