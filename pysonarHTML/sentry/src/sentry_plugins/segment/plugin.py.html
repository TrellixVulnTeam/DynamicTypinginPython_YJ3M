  10 DESCRIPTION = """
  11 Send Sentry events to Segment. This integration allows you to collect all your client-side data
  12 for Sentry automatically without the need to install the Sentry client library.
  13 Enable Sentry in your Segment settings to asynchronously load Raven.js onto your page without

  51     def get_event_props(self, event):
  52         props = {
  53             "eventId": event.event_id,
  54             "transaction": event.get_tag("transaction") or "",
  55             "release": event.get_tag("sentry:release") or "",
  56             "environment": event.get_tag("environment") or "",
  57         }
  58         if "sentry.interfaces.Http" in event.interfaces:
  59             http = event.interfaces["sentry.interfaces.Http"]
  60             headers = http.headers
  61             if not isinstance(headers, dict):
  62                 headers = dict(headers or ())
  63 
  64             props.update(
  65                 {
  66                     "requestUrl": http.url,
  67                     "requestMethod": http.method,
  68                     "requestReferer": headers.get("Referer", ""),
  69                 }
  70             )
  71         if "sentry.interfaces.Exception" in event.interfaces:
  72             exc = event.interfaces["sentry.interfaces.Exception"].values[0]
  73             props.update({"exceptionType": exc.type})
  74         return props

  76     # https://segment.com/docs/spec/track/
  77     def get_event_payload(self, event):
  78         context = {"library": {"name": "sentry", "version": self.version}}
  79 
  80         props = {
  81             "eventId": event.event_id,
  82             "transaction": event.get_tag("transaction") or "",
  83             "release": event.get_tag("sentry:release") or "",
  84             "environment": event.get_tag("environment") or "",
  85         }
  86 
  87         if "sentry.interfaces.User" in event.interfaces:
  88             user = event.interfaces["sentry.interfaces.User"]
  89             if user.ip_address:
  90                 context["ip"] = user.ip_address
  91             user_id = user.id
  92         else:
  93             user_id = None
  94 
  95         if "sentry.interfaces.Http" in event.interfaces:
  96             http = event.interfaces["sentry.interfaces.Http"]
  97             headers = http.headers
  98             if not isinstance(headers, dict):
  99                 headers = dict(headers or ())
 100 
 101             context.update(
 102                 {
 103                     "userAgent": headers.get("User-Agent", ""),
 104                     "page": {
 105                         "url": http.url,
 106                         "method": http.method,
 107                         "search": http.query_string or "",
 108                         "referrer": headers.get("Referer", ""),
 109                     },
 110                 }
 111             )
 112 
 113         if "sentry.interfaces.Exception" in event.interfaces:
 114             exc = event.interfaces["sentry.interfaces.Exception"].values[0]
 115             props.update({"exceptionType": exc.type})
 116 
 117         return {
 118             "context": context,
 119             "userId": user_id,
 120             "event": "Error Captured",
 121             "properties": props,
 122             "integration": {"name": "sentry", "version": self.version},
 123             "timestamp": event.datetime.isoformat() + "Z",
 124         }
 125 
 126     def forward_event(self, event, payload, **kwargs):
 127         # TODO(dcramer): we currently only support authenticated events, as the
 128         # value of anonymous errors/crashes/etc is much less meaningful in the
 129         # context of Segment
 130 
 131         # we currently only support errors
 132         if event.get_event_type() != "error":
 133             return
 134 
 135         # we avoid instantiating interfaces here as they're only going to be
 136         # used if there's a User present
 137         user_interface = event.data.get("sentry.interfaces.User")
 138         if not user_interface:
 139             return
 140 
 141         user_id = user_interface.get("id")
 142 
 143         if not user_id:
 144             return
 145 
 146         write_key = self.get_option("write_key", event.project)
 147         if not write_key:
 148             return
 149 
 150         session = http.build_session()
 151         session.post(self.endpoint, json=payload, auth=(write_key, ""))
