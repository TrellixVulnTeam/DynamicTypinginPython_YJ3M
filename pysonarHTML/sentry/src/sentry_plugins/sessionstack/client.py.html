  31         self.player_url = remove_trailing_slashes(player_url)
  32

  35             "Content-Type": "application/json",

  38     def validate_api_access(self):
  39         website_endpoint = WEBSITES_ENDPOINT.format(self.website_id)
  40 
  41         try:
  42             response = self._make_request(website_endpoint, "GET")
  43         except requests.exceptions.ConnectionError:
  44             raise InvalidApiUrlError
  45 
  46         if response.status_code == requests.codes.UNAUTHORIZED:
  47             raise UnauthorizedError
  48         elif response.status_code == requests.codes.BAD_REQUEST:
  49             raise InvalidWebsiteIdError
  50         elif response.status_code == requests.codes.NOT_FOUND:
  51             raise InvalidApiUrlError
  52 
  53         response.raise_for_status()

  55     def get_session_url(self, session_id, event_timestamp):
  56 
  57         player_url = self.player_url + SESSION_URL_PATH + session_id
  58         query_params = {}
  59 
  60         access_token = self._get_access_token(session_id)
  61         if access_token is not None:
  62             query_params["access_token"] = access_token
  63 
  64         if event_timestamp is not None:
  65             start_timestamp = self._get_session_start_timestamp(session_id)
  66             if start_timestamp is not None:
  67                 pause_at = event_timestamp - start_timestamp
  68                 play_from = pause_at - MILLISECONDS_BEFORE_EVENT
  69 
  70                 query_params["pause_at"] = pause_at
  71                 query_params["play_from"] = play_from
  72 
  73         return add_query_params(player_url, query_params)

  75     def _get_access_token(self, session_id):
  76         access_token = self._create_access_token(session_id)
  77         if not access_token:
  78             access_token = self._get_existing_access_token(session_id)
  79 
  80         return access_token

  82     def _get_existing_access_token(self, session_id):
  83         response = self._make_access_tokens_request(session_id, "GET")
  84 
  85         if response.status_code != requests.codes.OK:
  86             return None
  87 
  88         access_tokens = json.loads(response.content).get("data")
  89         for token in access_tokens:
  90             token_name = token.get("name")
  91             if token_name == ACCESS_TOKEN_NAME:
  92                 return token.get("access_token")
  93 
  94         return None
  95 
  96     def _create_access_token(self, session_id):
  97         response = self._make_access_tokens_request(
  98             session_id=session_id, method="POST", body={"name": ACCESS_TOKEN_NAME}
  99         )
 100 
 101         if response.status_code != requests.codes.OK:
 102             return None
 103 
 104         return json.loads(response.content).get("access_token")

 106     def _make_access_tokens_request(self, session_id, method, **kwargs):
 107         access_tokens_endpoint = self._get_access_tokens_endpoint(session_id)
 108         return self._make_request(access_tokens_endpoint, method, **kwargs)
 109 
 110     def _get_access_tokens_endpoint(self, session_id):
 111         return ACCESS_TOKENS_ENDPOINT.format(self.website_id, session_id)
 112 
 113     def _get_session_start_timestamp(self, session_id):
 114         endpoint = SESSION_ENDPOINT.format(self.website_id, session_id)
 115         response = self._make_request(endpoint, "GET")
 116 
 117         if response.status_code == requests.codes.OK:
 118             return json.loads(response.content).get("client_start")

 120     def _make_request(self, endpoint, method, **kwargs):
 121         url = self.api_url + endpoint
 122 
 123         request_kwargs = {"method": method, "headers": self.request_headers}
 124 
 125         body = kwargs.get("body")
 126         if body:
 127             request_kwargs["json"] = body
 128 
 129         return safe_urlopen(url, **request_kwargs)
