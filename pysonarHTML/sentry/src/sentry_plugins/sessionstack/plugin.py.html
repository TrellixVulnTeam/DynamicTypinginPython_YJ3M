  72     def validate_config(self, project, config, actor=None):
  73         sessionstack_client = SessionStackClient(
  74             account_email=config.get("account_email"),
  75             api_token=config.get("api_token"),
  76             website_id=config.get("website_id"),
  77             api_url=config.get("api_url"),
  78             player_url=config.get("player_url"),
  79         )
  80 
  81         try:
  82             sessionstack_client.validate_api_access()
  83         except UnauthorizedError:
  84             raise PluginError(UNAUTHORIZED_ERROR)
  85         except InvalidApiUrlError:
  86             raise PluginError(INVALID_API_URL_ERROR)
  87         except InvalidWebsiteIdError:

  90             raise PluginError(UNEXPECTED_ERROR)

  92         return config

 154     def get_event_preprocessors(self, data, **kwargs):
 155         context = SessionStackContextType.primary_value_for_data(data)
 156         if not context:
 157             return []
 158 
 159         session_id = context.get("session_id")
 160         if not session_id:
 161             return []
 162 
 163         project = Project.objects.get_from_cache(id=data.get("project"))
 164         if not self.is_enabled(project):
 165             return []
 166 
 167         def preprocess_event(event):
 168             sessionstack_client = SessionStackClient(
 169                 account_email=self.get_option("account_email", project),
 170                 api_token=self.get_option("api_token", project),
 171                 website_id=self.get_option("website_id", project),
 172                 api_url=self.get_option("api_url", project),
 173                 player_url=self.get_option("player_url", project),
 174             )
 175 
 176             session_url = sessionstack_client.get_session_url(
 177                 session_id=session_id, event_timestamp=context.get("timestamp")
 178             )
 179 
 180             context["session_url"] = session_url
 181 
 182             contexts = event.get("contexts") or {}
 183             contexts["sessionstack"] = context
 184             event["contexts"] = contexts
 185 
 186             return event
 187 
 188         return [preprocess_event]
