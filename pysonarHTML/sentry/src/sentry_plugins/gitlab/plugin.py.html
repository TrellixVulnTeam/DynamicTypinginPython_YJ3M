 111     def get_allowed_assignees(self, request, group):
 112         repo = self.get_option("gitlab_repo", group.project)
 113         client = self.get_client(group.project)
 114         try:
 115             response = client.list_project_members(repo)
 116         except ApiError as e:
 117             self.raise_error(e)
 118         users = tuple((u["id"], u["username"]) for u in response)
 119 
 120         return (("", "(Unassigned)"),) + users

 125     def get_client(self, project):
 126         url = self.get_option("gitlab_url", project).rstrip("/")
 127         token = self.get_option("gitlab_token", project)
 128 
 129         return GitLabClient(url, token)

 131     def create_issue(self, request, group, form_data, **kwargs):
 132         repo = self.get_option("gitlab_repo", group.project)
 133 
 134         client = self.get_client(group.project)
 135 
 136         try:
 137             response = client.create_issue(
 138                 repo,
 139                 {
 140                     "title": form_data["title"],
 141                     "description": form_data["description"],
 142                     "labels": form_data.get("labels"),
 143                     "assignee_id": form_data.get("assignee"),
 144                 },
 145             )
 146         except Exception as e:
 147             self.raise_error(e)
 148 
 149         return response["iid"]
 150 
 151     def link_issue(self, request, group, form_data, **kwargs):
 152         client = self.get_client(group.project)
 153         repo = self.get_option("gitlab_repo", group.project)
 154         try:
 155             issue = client.get_issue(repo=repo, issue_id=form_data["issue_id"])
 156         except Exception as e:
 157             self.raise_error(e)
 158 
 159         comment = form_data.get("comment")
 160         if comment:
 161             try:
 162                 client.create_note(repo=repo, issue_iid=issue["iid"], data={"body": comment})
 163             except Exception as e:
 164                 self.raise_error(e)
 165 
 166         return {"title": issue["title"]}
 167

 171     def get_issue_url(self, group, issue_iid, **kwargs):
 172         url = self.get_option("gitlab_url", group.project).rstrip("/")

 174 
 175         return "{}/{}/issues/{}".format(url, repo, issue_iid)
