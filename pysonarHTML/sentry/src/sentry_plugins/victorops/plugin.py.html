   1 from __future__ import absolute_import

   3 import six
   4 
   5 from sentry.plugins.bases.notify import NotifyPlugin

   7 from sentry_plugins.base import CorePluginMixin
   8 from sentry.shared_integrations.exceptions import ApiError
   9 from sentry_plugins.utils import get_secret_field_config
  10 from sentry.integrations import FeatureDescription, IntegrationFeatures

  14 ENHANCED_PRIVACY_BODY = """
  15 Details about this issue are not shown in this notification since enhanced
  16 privacy controls are enabled. For more details about this issue, view this
  17 issue on Sentry.
  18 """.strip()
  19 
  20 DESCRIPTION = """
  21 Trigger alerts in VictorOps from Sentry.
  22 
  23 VictorOps is incident response software purpose-built for teams powering the
  24 evolution of software. With on-call basics, cross-team collaboration, and
  25 streamlined visibility, we champion the engineers powering innovation and uptime.
  26 """
  27 
  28 
  29 class VictorOpsPlugin(CorePluginMixin, NotifyPlugin):
  30     description = DESCRIPTION
  31     slug = "victorops"
  32     title = "VictorOps"
  33     conf_key = slug
  34     conf_title = title
  35     required_field = "api_key"
  36     feature_descriptions = [
  37         FeatureDescription(
  38             """
  39             Manage incidents and outages by sending Sentry notifications to VictorOps.
  40             """,
  41             IntegrationFeatures.INCIDENT_MANAGEMENT,
  42         ),
  43         FeatureDescription(
  44             """
  45             Configure Sentry rules to trigger notifications based on conditions you set.
  46             """,
  47             IntegrationFeatures.ALERT_RULE,
  48         ),
  49     ]
  50 
  51     def is_configured(self, project, **kwargs):
  52         return bool(self.get_option("api_key", project))
  53 
  54     def get_config(self, **kwargs):
  55         return [
  56             get_secret_field_config(
  57                 name="api_key",
  58                 label="API Key",
  59                 secret=self.get_option("api_key", kwargs["project"]),
  60                 help_text="VictorOps's Sentry API Key",
  61                 include_prefix=True,
  62             ),
  63             {
  64                 "name": "routing_key",
  65                 "label": "Routing Key",
  66                 "type": "string",
  67                 "default": "everyone",
  68                 "required": False,
  69             },
  70         ]
  71 
  72     def get_client(self, project):
  73         return VictorOpsClient(
  74             api_key=self.get_option("api_key", project),
  75             routing_key=self.get_option("routing_key", project),
  76         )
  77 
  78     def build_description(self, event):
  79         enhanced_privacy = event.organization.flags.enhanced_privacy
  80         if enhanced_privacy:
  81             return ENHANCED_PRIVACY_BODY
  82 
  83         interface_list = []
  84         for interface in six.itervalues(event.interfaces):
  85             body = interface.to_string(event)
  86             if not body:
  87                 continue
  88             interface_list.append((interface.get_title(), body))
  89 
  90         return u"\n\n".join((u"{}\n-----------\n\n{}".format(k, v) for k, v in interface_list))
  91 
  92     def notify_users(self, group, event, fail_silently=False, **kwargs):
  93         if not self.is_configured(group.project):
  94             return
  95 
  96         level = event.get_tag("level")
  97         if level in ("info", "debug"):
  98             message_type = "INFO"
  99         if level == "warning":
 100             message_type = "WARNING"
 101         else:
 102             message_type = "CRITICAL"
 103 
 104         client = self.get_client(group.project)
 105         try:
 106             response = client.trigger_incident(
 107                 message_type=message_type,
 108                 entity_id=group.id,
 109                 entity_display_name=event.title,
 110                 state_message=self.build_description(event),
 111                 timestamp=int(event.datetime.strftime("%s")),
 112                 issue_url=group.get_absolute_url(),
 113                 issue_id=group.id,
 114                 project_id=group.project.id,
 115             )
 116         except ApiError as e:
 117             self.raise_error(e)
 118 
 119         assert response["result"] == "success"
