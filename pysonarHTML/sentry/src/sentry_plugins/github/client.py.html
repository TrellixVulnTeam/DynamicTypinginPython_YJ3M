  19     def get_last_commits(self, repo, end_sha):

  82 class GitHubAppsClient(GitHubClientMixin, ApiClient):
  83     def __init__(self, integration):
  84         self.integration = integration
  85         self.token = None
  86         self.expires_at = None
  87         super(GitHubAppsClient, self).__init__()
  88 
  89     def get_token(self):
  90         if not self.token or self.expires_at < datetime.datetime.utcnow():
  91             res = self.create_token()
  92             self.token = res["token"]
  93             self.expires_at = datetime.datetime.strptime(res["expires_at"], "%Y-%m-%dT%H:%M:%SZ")
  94 
  95         return self.token
  96 
  97     def get_jwt(self):
  98         exp = datetime.datetime.utcnow() + datetime.timedelta(minutes=10)
  99         exp = calendar.timegm(exp.timetuple())
 100         # Generate the JWT
 101         payload = {
 102             # issued at time
 103             "iat": int(time.time()),
 104             # JWT expiration time (10 minute maximum)
 105             "exp": exp,
 106             # Integration's GitHub identifier
 107             "iss": options.get("github.integration-app-id"),
 108         }
 109 
 110         return jwt.encode(payload, options.get("github.integration-private-key"), algorithm="RS256")
 111 
 112     def request(self, method, path, headers=None, data=None, params=None):
 113         if headers is None:
 114             headers = {
 115                 "Authorization": "token %s" % self.get_token(),
 116                 # TODO(jess): remove this whenever it's out of preview
 117                 "Accept": "application/vnd.github.machine-man-preview+json",
 118             }
 119         return self._request(method, path, headers=headers, data=data, params=params)
 120 
 121     def create_token(self):
 122         return self.post(
 123             "/installations/{}/access_tokens".format(self.integration.external_id),
 124             headers={
 125                 "Authorization": "Bearer %s" % self.get_jwt(),
 126                 # TODO(jess): remove this whenever it's out of preview
 127                 "Accept": "application/vnd.github.machine-man-preview+json",
 128             },
 129         )
 130

 132         return self.get("/installation/repositories")
