  41 class GitHubMixin(CorePluginMixin):
  42     def message_from_error(self, exc):
  43         if isinstance(exc, ApiError):
  44             message = API_ERRORS.get(exc.code)
  45             if message:
  46                 return message
  47             return "Error Communicating with GitHub (HTTP %s): %s" % (
  48                 exc.code,
  49                 exc.json.get("message", "unknown error") if exc.json else "unknown error",
  50             )
  51         else:
  52             return ERR_INTERNAL
  53 
  54     def get_client(self, user):
  55         auth = self.get_auth(user=user)
  56         if auth is None:
  57             raise PluginError(API_ERRORS[401])
  58         return GitHubClient(auth=auth)

 187     def link_issue(self, request, group, form_data, **kwargs):
 188         client = self.get_client(request.user)
 189         repo = self.get_option("repo", group.project)
 190         try:
 191             issue = client.get_issue(repo=repo, issue_id=form_data["issue_id"])
 192         except Exception as e:
 193             self.raise_error(e)
 194 
 195         comment = form_data.get("comment")
 196         if comment:
 197             try:
 198                 client.create_comment(repo=repo, issue_id=issue["number"], data={"body": comment})
 199             except Exception as e:
 200                 self.raise_error(e)
 201 
 202         return {"title": issue["title"]}

 213     def view_autocomplete(self, request, group, **kwargs):
 214         field = request.GET.get("autocomplete_field")
 215         query = request.GET.get("autocomplete_query")
 216         if field != "issue_id" or not query:
 217             return Response({"issue_id": []})
 218 
 219         repo = self.get_option("repo", group.project)
 220         client = self.get_client(request.user)
 221 
 222         try:
 223             response = client.search_issues(query=(u"repo:%s %s" % (repo, query)).encode("utf-8"))
 224         except Exception as e:
 225             return self.handle_api_error(e)
 226 
 227         issues = [
 228             {"text": "(#%s) %s" % (i["number"], i["title"]), "id": i["number"]}
 229             for i in response.get("items", [])
 230         ]
 231 
 232         return Response({field: issues})

 285     def validate_config(self, organization, config, actor=None):
 286         """
 287         ```
 288         if config['foo'] and not config['bar']:
 289             raise PluginError('You cannot configure foo with bar')
 290         return config
 291         ```
 292         """
 293         if config.get("name"):
 294             client = self.get_client(actor)
 295             try:
 296                 repo = client.get_repo(config["name"])
 297             except Exception as e:
 298                 self.raise_error(e)
 299             else:
 300                 config["external_id"] = six.text_type(repo["id"])
 301         return config

 303     def get_webhook_secret(self, organization):
 304         lock = locks.get("github:webhook-secret:{}".format(organization.id), duration=60)
 305         with lock.acquire():
 306             # TODO(dcramer): get_or_create would be a useful native solution
 307             secret = OrganizationOption.objects.get_value(
 308                 organization=organization, key="github:webhook_secret"
 309             )
 310             if secret is None:
 311                 secret = uuid4().hex + uuid4().hex
 312                 OrganizationOption.objects.set_value(
 313                     organization=organization, key="github:webhook_secret", value=secret
 314                 )
 315         return secret

 337     def create_repository(self, organization, data, actor=None):
 338         if actor is None:
 339             raise NotImplementedError("Cannot create a repository anonymously")
 340 
 341         client = self.get_client(actor)
 342 
 343         try:
 344             resp = self._create_webhook(client, organization, data["name"])
 345         except Exception as e:
 346             self.logger.exception(
 347                 "github.webhook.create-failure",
 348                 extra={
 349                     "organization_id": organization.id,
 350                     "repository": data["name"],
 351                     "status_code": getattr(e, "code", None),
 352                 },
 353             )
 354 
 355             self.raise_error(e)
 356         else:
 357             return {
 358                 "name": data["name"],
 359                 "external_id": data["external_id"],
 360                 "url": "https://github.com/{}".format(data["name"]),
 361                 "config": {
 362                     "name": data["name"],
 363                     "webhook_id": resp["id"],
 364                     "webhook_events": resp["events"],
 365                 },
 366             }

 368     # TODO(dcramer): let's make this core functionality and move the actual database
 369     # updates into Sentry core
 370     def update_repository(self, repo, actor=None):
 371         if actor is None:
 372             raise NotImplementedError("Cannot update a repository anonymously")
 373 
 374         client = self.get_client(actor)
 375         org = Organization.objects.get(id=repo.organization_id)
 376         webhook_id = repo.config.get("webhook_id")
 377         if not webhook_id:
 378             resp = self._create_webhook(client, org, repo.config["name"])
 379         else:
 380             resp = self._update_webhook(client, org, repo.config["name"], repo.config["webhook_id"])
 381         repo.config.update({"webhook_id": resp["id"], "webhook_events": resp["events"]})
 382         repo.update(config=repo.config)

 384     def delete_repository(self, repo, actor=None):
 385         if actor is None:
 386             raise NotImplementedError("Cannot delete a repository anonymously")
 387 
 388         if "webhook_id" in repo.config:
 389             client = self.get_client(actor)
 390             try:
 391                 client.delete_hook(repo.config["name"], repo.config["webhook_id"])
 392             except ApiError as exc:
 393                 if exc.code == 404:
 394                     return
 395                 raise

 397     def _format_commits(self, repo, commit_list):
 398         return [

 404                 "message": c["commit"]["message"],
 405             }
 406             for c in commit_list

 409     def compare_commits(self, repo, start_sha, end_sha, actor=None):
 410         if actor is None:
 411             raise NotImplementedError("Cannot fetch commits anonymously")
 412         client = self.get_client(actor)
 413 
 414         # use config name because that is kept in sync via webhooks
 415         name = repo.config["name"]
 416         if start_sha is None:
 417             try:
 418                 res = client.get_last_commits(name, end_sha)
 419             except Exception as e:
 420                 self.raise_error(e)
 421             else:
 422                 return self._format_commits(repo, res[:10])
 423         else:
 424             try:
 425                 res = client.compare_commits(name, start_sha, end_sha)
 426             except Exception as e:
 427                 self.raise_error(e)
 428             else:
 429                 return self._format_commits(repo, res["commits"])
 430

 440                 res = client.get_pr_commits(name, number)

 442                 self.raise_error(e)
 443             else:
 444                 return self._format_commits(repo, res)

 455     def get_available_auths(self, user, organization, integrations, social_auths, **kwargs):
 456         allowed_gh_installations = set(self.get_installations(user))
 457 
 458         linked_integrations = {i.id for i in integrations}
 459 
 460         _integrations = list(Integration.objects.filter(external_id__in=allowed_gh_installations))
 461 
 462         # add in integrations that might have been set up for org
 463         # by users w diff permissions
 464         _integrations.extend(
 465             [i for i in integrations if i.external_id not in allowed_gh_installations]
 466         )
 467 
 468         return [
 469             {
 470                 "defaultAuthId": None,
 471                 "user": None,
 472                 "externalId": i.external_id,
 473                 "integrationId": six.text_type(i.id),
 474                 "linked": i.id in linked_integrations,
 475             }
 476             for i in _integrations
 477         ]

 479     def link_auth(self, user, organization, data):
 480         integration_id = data["integration_id"]
 481 
 482         try:
 483             integration = Integration.objects.get(provider=self.auth_provider, id=integration_id)
 484         except Integration.DoesNotExist:
 485             raise PluginError("Invalid integration id")
 486 
 487         # check that user actually has access to add
 488         allowed_gh_installations = set(self.get_installations(user))
 489         if int(integration.external_id) not in allowed_gh_installations:
 490             raise PluginError("You do not have access to that integration")
 491 
 492         integration.add_organization(organization)
 493 
 494         for repo in self.get_repositories(integration):
 495             # TODO(jess): figure out way to migrate from github --> github apps
 496             Repository.objects.create_or_update(
 497                 organization_id=organization.id,
 498                 name=repo["name"],
 499                 external_id=repo["external_id"],
 500                 provider="github_apps",
 501                 values={
 502                     "integration_id": integration.id,
 503                     "url": repo["url"],
 504                     "config": repo["config"],
 505                 },
 506             )

 508     def delete_repository(self, repo, actor=None):
 509         if actor is None:
 510             raise NotImplementedError("Cannot delete a repository anonymously")

 512         # there isn't a webhook to delete for integrations
 513         if not repo.config.get("webhook_id") and repo.integration_id is not None:
 514             return
 515 
 516         return super(GitHubAppsRepositoryProvider, self).delete_repository(repo, actor=actor)

 542     def get_installations(self, actor):
 543         if not actor.is_authenticated():
 544             raise PluginError(API_ERRORS[401])
 545 
 546         auth = UserSocialAuth.objects.filter(user=actor, provider="github_apps").first()
 547 
 548         if not auth:
 549             self.logger.warn("get_installations.no-linked-auth")
 550             return []
 551 
 552         client = GitHubClient(auth=auth)
 553 
 554         res = client.get_installations()
 555 
 556         return [install["id"] for install in res["installations"]]
