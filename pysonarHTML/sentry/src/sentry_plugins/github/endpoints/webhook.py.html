 104 class PushEventWebhook(Webhook):
 105     def _handle(self, event, organization, is_apps):
 106         authors = {}
 107 
 108         client = GitHubClient()
 109         gh_username_cache = {}
 110 
 111         try:
 112             repo = Repository.objects.get(
 113                 organization_id=organization.id,
 114                 provider="github_apps" if is_apps else "github",
 115                 external_id=six.text_type(event["repository"]["id"]),
 116             )
 117         except Repository.DoesNotExist:
 118             raise Http404()
 119 
 120         # We need to track GitHub's "full_name" which is the repository slug.
 121         # This is needed to access the API since `external_id` isn't sufficient.
 122         if repo.config.get("name") != event["repository"]["full_name"]:
 123             repo.config["name"] = event["repository"]["full_name"]
 124             repo.save()
 125 
 126         for commit in event["commits"]:
 127             if not commit["distinct"]:
 128                 continue
 129 
 130             if RepositoryProvider.should_ignore_commit(commit["message"]):
 131                 continue
 132 
 133             author_email = commit["author"]["email"]
 134             if "@" not in author_email:
 135                 author_email = u"{}@localhost".format(author_email[:65])
 136             # try to figure out who anonymous emails are
 137             elif is_anonymous_email(author_email):
 138                 gh_username = commit["author"].get("username")
 139                 # bot users don't have usernames
 140                 if gh_username:
 141                     external_id = get_external_id(gh_username)
 142                     if gh_username in gh_username_cache:
 143                         author_email = gh_username_cache[gh_username] or author_email
 144                     else:
 145                         try:
 146                             commit_author = CommitAuthor.objects.get(
 147                                 external_id=external_id, organization_id=organization.id
 148                             )
 149                         except CommitAuthor.DoesNotExist:
 150                             commit_author = None
 151 
 152                         if commit_author is not None and not is_anonymous_email(
 153                             commit_author.email
 154                         ):
 155                             author_email = commit_author.email
 156                             gh_username_cache[gh_username] = author_email
 157                         else:
 158                             try:
 159                                 gh_user = client.request_no_auth("GET", "/users/%s" % gh_username)
 160                             except ApiError as exc:
 161                                 logger.exception(six.text_type(exc))
 162                             else:
 163                                 # even if we can't find a user, set to none so we
 164                                 # don't re-query
 165                                 gh_username_cache[gh_username] = None
 166                                 try:
 167                                     user = User.objects.filter(
 168                                         social_auth__provider="github",
 169                                         social_auth__uid=gh_user["id"],
 170                                         org_memberships=organization,
 171                                     )[0]
 172                                 except IndexError:
 173                                     pass
 174                                 else:
 175                                     author_email = user.email
 176                                     gh_username_cache[gh_username] = author_email
 177                                     if commit_author is not None:
 178                                         try:
 179                                             with transaction.atomic():
 180                                                 commit_author.update(
 181                                                     email=author_email, external_id=external_id
 182                                                 )
 183                                         except IntegrityError:
 184                                             pass
 185 
 186                         if commit_author is not None:
 187                             authors[author_email] = commit_author
 188 
 189             # TODO(dcramer): we need to deal with bad values here, but since
 190             # its optional, lets just throw it out for now
 191             if len(author_email) > 75:
 192                 author = None
 193             elif author_email not in authors:
 194                 authors[author_email] = author = CommitAuthor.objects.get_or_create(
 195                     organization_id=organization.id,
 196                     email=author_email,
 197                     defaults={"name": commit["author"]["name"][:128]},
 198                 )[0]
 199 
 200                 update_kwargs = {}
 201 
 202                 if author.name != commit["author"]["name"][:128]:
 203                     update_kwargs["name"] = commit["author"]["name"][:128]
 204 
 205                 gh_username = commit["author"].get("username")
 206                 if gh_username:
 207                     external_id = get_external_id(gh_username)
 208                     if author.external_id != external_id and not is_anonymous_email(author.email):
 209                         update_kwargs["external_id"] = external_id
 210 
 211                 if update_kwargs:
 212                     try:
 213                         with transaction.atomic():
 214                             author.update(**update_kwargs)
 215                     except IntegrityError:
 216                         pass
 217             else:
 218                 author = authors[author_email]
 219 
 220             try:
 221                 with transaction.atomic():
 222                     c = Commit.objects.create(
 223                         repository_id=repo.id,
 224                         organization_id=organization.id,
 225                         key=commit["id"],
 226                         message=commit["message"],
 227                         author=author,
 228                         date_added=dateutil.parser.parse(commit["timestamp"]).astimezone(
 229                             timezone.utc
 230                         ),
 231                     )
 232                     for fname in commit["added"]:
 233                         CommitFileChange.objects.create(
 234                             organization_id=organization.id, commit=c, filename=fname, type="A"
 235                         )
 236                     for fname in commit["removed"]:
 237                         CommitFileChange.objects.create(
 238                             organization_id=organization.id, commit=c, filename=fname, type="D"
 239                         )
 240                     for fname in commit["modified"]:
 241                         CommitFileChange.objects.create(
 242                             organization_id=organization.id, commit=c, filename=fname, type="M"
 243                         )
 244             except IntegrityError:
 245                 pass

 248     def __call__(self, event, organization=None):
 249         is_apps = "installation" in event
 250         if organization is None:
 251             if "installation" not in event:
 252                 return
 253 
 254             integration = Integration.objects.get(
 255                 external_id=event["installation"]["id"], provider="github_apps"
 256             )
 257             organizations = list(integration.organizations.all())
 258         else:
 259             organizations = [organization]
 260

 262             self._handle(event, org, is_apps)

 265 class PullRequestEventWebhook(Webhook):
 266     # https://developer.github.com/v3/activity/events/types/#pullrequestevent
 267     def __call__(self, event, organization):
 268         # TODO(maxbittker) handle is_apps correctly (What does this comment mean?)
 269         is_apps = "installation" in event
 270         try:
 271             repo = Repository.objects.get(
 272                 organization_id=organization.id,
 273                 provider="github_apps" if is_apps else "github",
 274                 external_id=six.text_type(event["repository"]["id"]),
 275             )
 276 
 277         except Repository.DoesNotExist:
 278             raise Http404()
 279 
 280         # We need to track GitHub's "full_name" which is the repository slug.
 281         # This is needed to access the API since `external_id` isn't sufficient.
 282         if repo.config.get("name") != event["repository"]["full_name"]:
 283             repo.config["name"] = event["repository"]["full_name"]
 284             repo.save()
 285 
 286         pull_request = event["pull_request"]
 287         number = pull_request["number"]
 288         title = pull_request["title"]
 289         body = pull_request["body"]
 290         user = pull_request["user"]
 291         # The value of the merge_commit_sha attribute changes depending on the state of the pull request. Before a pull request is merged, the merge_commit_sha attribute holds the SHA of the test merge commit. After a pull request is merged, the attribute changes depending on how the pull request was merged:
 292         # - If the pull request was merged as a merge commit, the attribute represents the SHA of the merge commit.
 293         # - If the pull request was merged via a squash, the attribute represents the SHA of the squashed commit on the base branch.
 294         # - If the pull request was rebased, the attribute represents the commit that the base branch was updated to.
 295         # https://developer.github.com/v3/pulls/#get-a-single-pull-request
 296         merge_commit_sha = pull_request["merge_commit_sha"] if pull_request["merged"] else None
 297 
 298         author_email = u"{}@localhost".format(user["login"][:65])
 299         try:
 300             commit_author = CommitAuthor.objects.get(
 301                 external_id=get_external_id(user["login"]), organization_id=organization.id
 302             )
 303             author_email = commit_author.email
 304         except CommitAuthor.DoesNotExist:
 305             try:
 306                 user_model = User.objects.filter(
 307                     social_auth__provider="github",
 308                     social_auth__uid=user["id"],
 309                     org_memberships=organization,
 310                 )[0]
 311             except IndexError:
 312                 pass
 313             else:
 314                 author_email = user_model.email
 315 
 316         try:
 317             author = CommitAuthor.objects.get(
 318                 organization_id=organization.id, external_id=get_external_id(user["login"])
 319             )
 320         except CommitAuthor.DoesNotExist:
 321             try:
 322                 author = CommitAuthor.objects.get(
 323                     organization_id=organization.id, email=author_email
 324                 )
 325             except CommitAuthor.DoesNotExist:
 326                 author = CommitAuthor.objects.create(
 327                     organization_id=organization.id,
 328                     email=author_email,
 329                     external_id=get_external_id(user["login"]),
 330                     name=user["login"][:128],
 331                 )
 332 
 333         try:
 334             PullRequest.objects.create_or_update(
 335                 repository_id=repo.id,
 336                 key=number,
 337                 values={
 338                     "organization_id": organization.id,
 339                     "title": title,
 340                     "author": author,
 341                     "message": body,
 342                     "merge_commit_sha": merge_commit_sha,
 343                 },
 344             )
 345         except IntegrityError:
 346             pass

 349 class GithubWebhookBase(View):
 350     _handlers = {"push": PushEventWebhook, "pull_request": PullRequestEventWebhook}
 351 
 352     # https://developer.github.com/webhooks/
 353     def get_handler(self, event_type):
 354         return self._handlers.get(event_type)
 355 
 356     def is_valid_signature(self, method, body, secret, signature):
 357         if method == "sha1":
 358             mod = hashlib.sha1
 359         else:
 360             raise NotImplementedError("signature method %s is not supported" % (method,))
 361         expected = hmac.new(key=secret.encode("utf-8"), msg=body, digestmod=mod).hexdigest()
 362         return constant_time_compare(expected, signature)
 363 
 364     @method_decorator(csrf_exempt)
 365     def dispatch(self, request, *args, **kwargs):
 366         if request.method != "POST":
 367             return HttpResponse(status=405)
 368 
 369         return super(GithubWebhookBase, self).dispatch(request, *args, **kwargs)
 370 
 371     def get_logging_data(self, organization):
 372         pass
 373 
 374     def get_secret(self, organization):
 375         raise NotImplementedError
 376 
 377     def handle(self, request, organization=None):
 378         secret = self.get_secret(organization)
 379 
 380         if secret is None:
 381             logger.error("github.webhook.missing-secret", extra=self.get_logging_data(organization))
 382             return HttpResponse(status=401)
 383 
 384         body = six.binary_type(request.body)
 385         if not body:
 386             logger.error("github.webhook.missing-body", extra=self.get_logging_data(organization))
 387             return HttpResponse(status=400)
 388 
 389         try:
 390             handler = self.get_handler(request.META["HTTP_X_GITHUB_EVENT"])
 391         except KeyError:
 392             logger.error("github.webhook.missing-event", extra=self.get_logging_data(organization))
 393             return HttpResponse(status=400)
 394 
 395         if not handler:
 396             return HttpResponse(status=204)
 397 
 398         try:
 399             method, signature = request.META["HTTP_X_HUB_SIGNATURE"].split("=", 1)
 400         except (KeyError, IndexError):
 401             logger.error(
 402                 "github.webhook.missing-signature", extra=self.get_logging_data(organization)
 403             )
 404             return HttpResponse(status=400)
 405 
 406         if not self.is_valid_signature(method, body, self.get_secret(organization), signature):
 407             logger.error(
 408                 "github.webhook.invalid-signature", extra=self.get_logging_data(organization)
 409             )
 410             return HttpResponse(status=401)
 411 
 412         try:
 413             event = json.loads(body.decode("utf-8"))
 414         except JSONDecodeError:
 415             logger.error(
 416                 "github.webhook.invalid-json",
 417                 extra=self.get_logging_data(organization),
 418                 exc_info=True,
 419             )
 420             return HttpResponse(status=400)
 421 
 422         handler()(event, organization=organization)
 423         return HttpResponse(status=204)

 426 # non-integration version
 427 class GithubWebhookEndpoint(GithubWebhookBase):
 428     def get_logging_data(self, organization):
 429         return {"organization_id": organization.id}
 430 
 431     def get_secret(self, organization):
 432         return OrganizationOption.objects.get_value(
 433             organization=organization, key="github:webhook_secret"
 434         )
 435 
 436     def post(self, request, organization_id):
 437         try:
 438             organization = Organization.objects.get_from_cache(id=organization_id)
 439         except Organization.DoesNotExist:
 440             logger.error(
 441                 "github.webhook.invalid-organization", extra={"organization_id": organization_id}
 442             )
 443             return HttpResponse(status=400)
 444 
 445         return self.handle(request, organization=organization)
