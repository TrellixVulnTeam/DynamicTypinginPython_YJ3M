  33 class TeamworkTaskForm(NewIssueForm):
  34     title = forms.CharField(
  35         label=_("Title"), max_length=200, widget=forms.TextInput(attrs={"class": "span9"})
  36     )
  37     description = forms.CharField(
  38         label=_("Description"), widget=forms.Textarea(attrs={"class": "span9"})
  39     )
  40     project = forms.ChoiceField(label=_("Project"), choices=())
  41     tasklist = forms.ChoiceField(label=_("Task List"), choices=())
  42 
  43     create_issue_template = "sentry_teamwork/create_issue.html"
  44 
  45     def __init__(self, client, data=None, initial=None):
  46         super(TeamworkTaskForm, self).__init__(data=data, initial=initial)
  47 
  48         try:
  49             project_list = client.list_projects()
  50         except RequestException as e:
  51             raise forms.ValidationError(_("Error contacting Teamwork API: %s") % six.text_type(e))
  52 
  53         self.fields["project"].choices = [(six.text_type(i["id"]), i["name"]) for i in project_list]
  54         self.fields["project"].widget.choices = self.fields["project"].choices
  55 
  56         if self.data.get("project"):
  57             try:
  58                 tasklist_list = client.list_tasklists(data["project"])
  59             except RequestException as e:
  60                 raise forms.ValidationError(
  61                     _("Error contacting Teamwork API: %s") % six.text_type(e)
  62                 )
  63             self.fields["tasklist"].choices = [
  64                 (six.text_type(i["id"]), i["name"]) for i in tasklist_list
  65             ]

 101     def _get_group_description(self, request, group, event):
 102         """
 103         Return group description in markdown-compatible format.
 104 
 105         This overrides an internal method to IssuePlugin.
 106         """
 107         output = [absolute_uri(group.get_absolute_url())]
 108         body = self._get_group_body(request, group, event)
 109         if body:
 110             output.extend(["", "\n".join("    " + line for line in body.splitlines())])
 111         return "\n".join(output)
 112 
 113     def is_configured(self, request, project, **kwargs):
 114         return all((self.get_option(key, project) for key in ("url", "token")))
 115 
 116     def get_client(self, project):
 117         return TeamworkClient(
 118             base_url=self.get_option("url", project), token=self.get_option("token", project)
 119         )
 120 
 121     def get_new_issue_form(self, request, group, event, **kwargs):
 122         """
 123         Return a Form for the "Create new issue" page.
 124         """
 125         return self.new_issue_form(
 126             client=self.get_client(group.project),
 127             data=request.POST or None,
 128             initial=self.get_initial_form_data(request, group, event),
 129         )
 130 
 131     def get_issue_url(self, group, issue_id, **kwargs):
 132         url = self.get_option("url", group.project)
 133         return "%s/tasks/%s" % (url.rstrip("/"), issue_id)
 134 
 135     def get_new_issue_title(self, **kwargs):
 136         return _("Create Teamwork Task")
 137 
 138     def create_issue(self, request, group, form_data, **kwargs):
 139         client = self.get_client(group.project)
 140         try:
 141             task_id = client.create_task(
 142                 content=form_data["title"],
 143                 description=form_data["description"],
 144                 tasklist_id=form_data["tasklist"],
 145             )
 146         except RequestException as e:
 147             raise forms.ValidationError(_("Error creating Teamwork task: %s") % six.text_type(e))
 148 
 149         return task_id
 150 
 151     def view(self, request, group, **kwargs):
 152         op = request.GET.get("op")
 153         # TODO(dcramer): add caching
 154         if op == "getTaskLists":
 155             project_id = request.GET.get("pid")
 156             if not project_id:
 157                 return HttpResponse(status=400)
 158 
 159             client = self.get_client(group.project)
 160             task_list = client.list_tasklists(project_id)
 161             return JSONResponse([{"id": i["id"], "text": i["name"]} for i in task_list])
 162 
 163         return super(TeamworkPlugin, self).view(request, group, **kwargs)
