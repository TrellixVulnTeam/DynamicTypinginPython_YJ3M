  52     def get_webhook_secret(self, organization):
  53         lock = locks.get("bitbucket:webhook-secret:{}".format(organization.id), duration=60)
  54         with lock.acquire():
  55             secret = OrganizationOption.objects.get_value(
  56                 organization=organization, key="bitbucket:webhook_secret"
  57             )
  58             if secret is None:
  59                 secret = uuid4().hex + uuid4().hex
  60                 OrganizationOption.objects.set_value(
  61                     organization=organization, key="bitbucket:webhook_secret", value=secret
  62                 )
  63         return secret

  65     def create_repository(self, organization, data, actor=None):
  66         if actor is None:
  67             raise NotImplementedError("Cannot create a repository anonymously")
  68 
  69         client = self.get_client(actor)
  70         try:
  71             resp = client.create_hook(
  72                 data["name"],
  73                 {
  74                     "description": "sentry-bitbucket-repo-hook",
  75                     "url": absolute_uri(
  76                         "/plugins/bitbucket/organizations/{}/webhook/".format(organization.id)
  77                     ),
  78                     "active": True,
  79                     "events": ["repo:push"],
  80                 },
  81             )
  82         except Exception as e:
  83             self.raise_error(e, identity=client.auth)
  84         else:
  85             return {
  86                 "name": data["name"],
  87                 "external_id": data["external_id"],
  88                 "url": "https://bitbucket.org/{}".format(data["name"]),
  89                 "config": {"name": data["name"], "webhook_id": resp["uuid"]},

  92     def delete_repository(self, repo, actor=None):
  93         if actor is None:
  94             raise NotImplementedError("Cannot delete a repository anonymously")
  95 
  96         client = self.get_client(actor)
  97         try:
  98             client.delete_hook(repo.config["name"], repo.config["webhook_id"])
  99         except ApiError as exc:
 100             if exc.code == 404:
 101                 return
 102             raise

 104     def _format_commits(self, repo, commit_list):
 105         return [
 106             {
 107                 "id": c["hash"],
 108                 "repository": repo.name,
 109                 "author_email": parse_raw_user_email(c["author"]["raw"]),
 110                 "author_name": parse_raw_user_name(c["author"]["raw"]),
 111                 "message": c["message"],
 112                 "patch_set": c.get("patch_set"),
 113             }
 114             for c in commit_list
 115         ]

 117     def compare_commits(self, repo, start_sha, end_sha, actor=None):
 118         if actor is None:
 119             raise NotImplementedError("Cannot fetch commits anonymously")
 120 
 121         client = self.get_client(actor)
 122         # use config name because that is kept in sync via webhooks
 123         name = repo.config["name"]
 124         if start_sha is None:
 125             try:
 126                 res = client.get_last_commits(name, end_sha)
 127             except Exception as e:
 128                 self.raise_error(e, identity=client.auth)
 129             else:
 130                 return self._format_commits(repo, res[:10])
 131         else:
 132             try:
 133                 res = client.compare_commits(name, start_sha, end_sha)
 134             except Exception as e:
 135                 self.raise_error(e, identity=client.auth)
 136             else:
 137                 return self._format_commits(repo, res)
