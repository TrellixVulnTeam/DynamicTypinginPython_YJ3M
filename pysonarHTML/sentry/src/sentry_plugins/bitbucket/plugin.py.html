 135     def message_from_error(self, exc):
 136         if isinstance(exc, ApiError) and exc.code == 404:
 137             return ERR_404
 138         return super(BitbucketPlugin, self).message_from_error(exc)

 140     def create_issue(self, request, group, form_data, **kwargs):
 141         client = self.get_client(request.user)
 142 
 143         try:
 144             response = client.create_issue(
 145                 repo=self.get_option("repo", group.project), data=form_data
 146             )
 147         except Exception as e:
 148             self.raise_error(e, identity=client.auth)
 149 
 150         return response["local_id"]
 151 
 152     def link_issue(self, request, group, form_data, **kwargs):
 153         client = self.get_client(request.user)
 154         repo = self.get_option("repo", group.project)
 155         try:
 156             issue = client.get_issue(repo=repo, issue_id=form_data["issue_id"])
 157         except Exception as e:
 158             self.raise_error(e, identity=client.auth)
 159 
 160         comment = form_data.get("comment")
 161         if comment:
 162             try:
 163                 client.create_comment(repo, issue["local_id"], {"content": comment})
 164             except Exception as e:
 165                 self.raise_error(e, identity=client.auth)
 166 
 167         return {"title": issue["title"]}
 168 
 169     def get_issue_label(self, group, issue_id, **kwargs):

 172     def get_issue_url(self, group, issue_id, **kwargs):
 173         repo = self.get_option("repo", group.project)
 174         return "https://bitbucket.org/%s/issue/%s/" % (repo, issue_id)

 176     def view_autocomplete(self, request, group, **kwargs):
 177         field = request.GET.get("autocomplete_field")
 178         query = request.GET.get("autocomplete_query")
 179         if field != "issue_id" or not query:
 180             return Response({"issue_id": []})
 181 
 182         repo = self.get_option("repo", group.project)
 183         client = self.get_client(request.user)
 184 
 185         try:
 186             response = client.search_issues(repo, query.encode("utf-8"))
 187         except Exception as e:
 188             return Response(
 189                 {"error_type": "validation", "errors": [{"__all__": self.message_from_error(e)}]},
 190                 status=400,
 191             )
 192 
 193         issues = [
 194             {"text": "(#%s) %s" % (i["local_id"], i["title"]), "id": i["local_id"]}
 195             for i in response.get("issues", [])
 196         ]
 197 
 198         return Response({field: issues})
