  43 def parse_raw_user_email(raw):
  44     # captures content between angle brackets
  45     match = re.search("(?<=<).*(?=>$)", raw)
  46     if match is None:
  47         return
  48     return match.group(0)
  49 
  50 
  51 def parse_raw_user_name(raw):
  52     # captures content before angle bracket
  53     return raw.split("<")[0].strip()

  56 class PushEventWebhook(Webhook):
  57     # https://confluence.atlassian.com/bitbucket/event-payloads-740262817.html#EventPayloads-Push
  58     def __call__(self, organization, event):
  59         authors = {}
  60 
  61         try:
  62             repo = Repository.objects.get(
  63                 organization_id=organization.id,
  64                 provider="bitbucket",
  65                 external_id=six.text_type(event["repository"]["uuid"]),
  66             )
  67         except Repository.DoesNotExist:
  68             raise Http404()
  69 
  70         if repo.config.get("name") != event["repository"]["full_name"]:
  71             repo.config["name"] = event["repository"]["full_name"]
  72             repo.save()
  73 
  74         for change in event["push"]["changes"]:
  75             for commit in change.get("commits", []):
  76                 if RepositoryProvider.should_ignore_commit(commit["message"]):
  77                     continue
  78 
  79                 author_email = parse_raw_user_email(commit["author"]["raw"])
  80 
  81                 # TODO(dcramer): we need to deal with bad values here, but since
  82                 # its optional, lets just throw it out for now
  83                 if author_email is None or len(author_email) > 75:
  84                     author = None
  85                 elif author_email not in authors:
  86                     authors[author_email] = author = CommitAuthor.objects.get_or_create(
  87                         organization_id=organization.id,
  88                         email=author_email,
  89                         defaults={"name": commit["author"]["raw"].split("<")[0].strip()},
  90                     )[0]
  91                 else:
  92                     author = authors[author_email]
  93                 try:
  94                     with transaction.atomic():
  95 
  96                         Commit.objects.create(
  97                             repository_id=repo.id,
  98                             organization_id=organization.id,
  99                             key=commit["hash"],
 100                             message=commit["message"],
 101                             author=author,
 102                             date_added=dateutil.parser.parse(commit["date"]).astimezone(
 103                                 timezone.utc
 104                             ),
 105                         )
 106 
 107                 except IntegrityError:
 108                     pass

 111 class BitbucketWebhookEndpoint(View):
 112     _handlers = {"repo:push": PushEventWebhook}
 113 
 114     def get_handler(self, event_type):
 115         return self._handlers.get(event_type)
 116 
 117     @method_decorator(csrf_exempt)
 118     def dispatch(self, request, *args, **kwargs):
 119         if request.method != "POST":
 120             return HttpResponse(status=405)
 121 
 122         return super(BitbucketWebhookEndpoint, self).dispatch(request, *args, **kwargs)
 123 
 124     def post(self, request, organization_id):
 125         try:
 126             organization = Organization.objects.get_from_cache(id=organization_id)
 127         except Organization.DoesNotExist:
 128             logger.error(
 129                 "bitbucket.webhook.invalid-organization", extra={"organization_id": organization_id}
 130             )
 131             return HttpResponse(status=400)
 132 
 133         body = six.binary_type(request.body)
 134         if not body:
 135             logger.error(
 136                 "bitbucket.webhook.missing-body", extra={"organization_id": organization.id}
 137             )
 138             return HttpResponse(status=400)
 139 
 140         try:
 141             handler = self.get_handler(request.META["HTTP_X_EVENT_KEY"])
 142         except KeyError:
 143             logger.error(
 144                 "bitbucket.webhook.missing-event", extra={"organization_id": organization.id}
 145             )
 146             return HttpResponse(status=400)
 147 
 148         if not handler:
 149             return HttpResponse(status=204)
 150 
 151         address_string = six.text_type(request.META["REMOTE_ADDR"])
 152         ip = ipaddress.ip_address(address_string)
 153         valid_ip = False
 154         for ip_range in BITBUCKET_IP_RANGES:
 155             if ip in ip_range:
 156                 valid_ip = True
 157                 break
 158         if not valid_ip and address_string not in BITBUCKET_IPS:
 159             logger.error(
 160                 "bitbucket.webhook.invalid-ip-range", extra={"organization_id": organization.id}
 161             )
 162             return HttpResponse(status=401)
 163 
 164         try:
 165             event = json.loads(body.decode("utf-8"))
 166         except JSONDecodeError:
 167             logger.error(
 168                 "bitbucket.webhook.invalid-json",
 169                 extra={"organization_id": organization.id},
 170                 exc_info=True,
 171             )
 172             return HttpResponse(status=400)
 173 
 174         handler()(organization, event)
 175         return HttpResponse(status=204)
