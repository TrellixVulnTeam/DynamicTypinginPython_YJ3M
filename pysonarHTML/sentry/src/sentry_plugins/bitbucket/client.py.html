  33     def get_issue(self, repo, issue_id):

  62     def transform_patchset(self, patch_set):
  63         file_changes = []
  64         for patched_file in patch_set.added_files:
  65             file_changes.append({"path": patched_file.path, "type": "A"})
  66 
  67         for patched_file in patch_set.removed_files:
  68             file_changes.append({"path": patched_file.path, "type": "D"})
  69 
  70         for patched_file in patch_set.modified_files:
  71             file_changes.append({"path": patched_file.path, "type": "M"})
  72 
  73         return file_changes

  75     def get_commit_filechanges(self, repo, sha):
  76         # returns unidiff file
  77 
  78         resp = self.get("/2.0/repositories/{}/diff/{}".format(repo, sha), allow_text=True)
  79 
  80         diff_file = resp.text
  81         ps = PatchSet.from_string(diff_file)
  82         return self.transform_patchset(ps)

  84     def zip_commit_data(self, repo, commit_list):
  85         for commit in commit_list:
  86             commit.update({"patch_set": self.get_commit_filechanges(repo, commit["hash"])})
  87         return commit_list

  89     def get_last_commits(self, repo, end_sha):
  90         # return api request that fetches last ~30 commits
  91         # see https://developer.atlassian.com/bitbucket/api/2/reference/resource/repositories/%7Busername%7D/%7Brepo_slug%7D/commits/%7Brevision%7D
  92         # using end_sha as parameter
  93         data = self.get("/2.0/repositories/{}/commits/{}".format(repo, end_sha))
  94 
  95         return self.zip_commit_data(repo, data["values"])

  97     def compare_commits(self, repo, start_sha, end_sha):
  98         # where start_sha is oldest and end_sha is most recent
  99         # see
 100         # https://developer.atlassian.com/bitbucket/api/2/reference/resource/repositories/%7Busername%7D/%7Brepo_slug%7D/commits/%7Brevision%7D
 101         commits = []
 102         done = False
 103 
 104         url = "/2.0/repositories/{}/commits/{}".format(repo, end_sha)
 105 
 106         while not done and len(commits) < 90:
 107             data = self.get(url)
 108 
 109             for commit in data["values"]:
 110                 if commit["hash"].startswith(start_sha):
 111                     done = True
 112                     break
 113                 commits.append(commit)
 114 
 115             # move page forward
 116             try:
 117                 url = data["next"]
 118             except KeyError:
 119                 break
 120 
 121         return self.zip_commit_data(repo, commits)
