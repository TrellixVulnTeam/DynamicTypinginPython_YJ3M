  97     def validate_config(self, project, config, actor):
  98         if int(config["priority"]) == 2 and config["retry"] < 30:
  99             retry = six.binary_type(config["retry"])
 100             self.logger.exception(six.text_type("Retry not 30 or higher. It is {}.".format(retry)))
 101             raise PluginError("Retry must be 30 or higher. It is {}.".format(retry))
 102         return config
 103 
 104     def get_client(self, project):
 105         return PushoverClient(
 106             apikey=self.get_option("apikey", project), userkey=self.get_option("userkey", project)
 107         )
 108 
 109     def error_message_from_json(self, data):
 110         errors = data.get("errors")
 111         if errors:
 112             return " ".join(errors)
 113         return "unknown error"
 114 
 115     def notify(self, notification, **kwargs):
 116         event = notification.event
 117         group = event.group
 118         project = group.project
 119         priority = int(self.get_option("priority", project) or 0)
 120         retry = int(self.get_option("retry", project) or 30)
 121         expire = int(self.get_option("expire", project) or 90)
 122 
 123         title = "%s: %s" % (project.name, group.title)
 124         link = group.get_absolute_url(params={"referrer": "pushover_plugin"})
 125 
 126         message = event.title[:256]
 127 
 128         tags = event.tags
 129         if tags:
 130             message += "\n\nTags: %s" % (", ".join("%s=%s" % (k, v) for (k, v) in tags))
 131 
 132         client = self.get_client(project)
 133         try:
 134             response = client.send_message(
 135                 {
 136                     "message": message[:1024],
 137                     "title": title[:250],
 138                     "url": link,
 139                     "url_title": "Issue Details",
 140                     "priority": priority,
 141                     "retry": retry,
 142                     "expire": expire,
 143                 }
 144             )
 145         except Exception as e:
 146             self.raise_error(e)
 147         assert response["status"]
