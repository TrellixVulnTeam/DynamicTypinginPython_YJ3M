 135     def color_for_event(self, event):
 136         return "#" + LEVEL_TO_COLOR.get(event.get_tag("level"), "error")
 137 
 138     def _get_tags(self, event):
 139         tag_list = event.tags
 140         if not tag_list:
 141             return ()
 142 
 143         return (
 144             (tagstore.get_tag_key_label(k), tagstore.get_tag_value_label(k, v)) for k, v in tag_list
 145         )
 146 
 147     def get_tag_list(self, name, project):
 148         option = self.get_option(name, project)
 149         if not option:
 150             return None
 151         return set(tag.strip().lower() for tag in option.split(","))
 152 
 153     def notify(self, notification, raise_exception=False):
 154         event = notification.event
 155         group = event.group
 156         project = group.project
 157 
 158         if not self.is_configured(project):
 159             return
 160 
 161         title = event.title.encode("utf-8")
 162         # TODO(dcramer): we'd like this to be the event culprit, but Sentry
 163         # does not currently retain it
 164         if group.culprit:
 165             culprit = group.culprit.encode("utf-8")
 166         else:
 167             culprit = None
 168         project_name = project.get_full_name().encode("utf-8")
 169 
 170         fields = []
 171 
 172         # They can be the same if there is no culprit
 173         # So we set culprit to an empty string instead of duplicating the text
 174         if not self.get_option("exclude_culprit", project) and culprit and title != culprit:
 175             fields.append({"title": "Culprit", "value": culprit, "short": False})
 176         if not self.get_option("exclude_project", project):
 177             fields.append({"title": "Project", "value": project_name, "short": True})
 178 
 179         if self.get_option("custom_message", project):
 180             fields.append(
 181                 {
 182                     "title": "Custom message",
 183                     "value": self.get_option("custom_message", project),
 184                     "short": False,
 185                 }
 186             )
 187 
 188         if self.get_option("include_rules", project):
 189             rules = []
 190             for rule in notification.rules:
 191                 rule_link = "/%s/%s/settings/alerts/rules/%s/" % (
 192                     group.organization.slug,
 193                     project.slug,
 194                     rule.id,
 195                 )
 196 
 197                 # Make sure it's an absolute uri since we're sending this
 198                 # outside of Sentry into Slack
 199                 rule_link = absolute_uri(rule_link)
 200                 rules.append((rule_link, rule.label))
 201 
 202             if rules:
 203                 value = u", ".join(u"<{} | {}>".format(*r) for r in rules)
 204 
 205                 fields.append(
 206                     {"title": "Triggered By", "value": value.encode("utf-8"), "short": False}
 207                 )
 208 
 209         if self.get_option("include_tags", project):
 210             included_tags = set(self.get_tag_list("included_tag_keys", project) or [])
 211             excluded_tags = set(self.get_tag_list("excluded_tag_keys", project) or [])
 212             for tag_key, tag_value in self._get_tags(event):
 213                 key = tag_key.lower()
 214                 std_key = tagstore.get_standardized_key(key)
 215                 if included_tags and key not in included_tags and std_key not in included_tags:
 216                     continue
 217                 if excluded_tags and (key in excluded_tags or std_key in excluded_tags):
 218                     continue
 219                 fields.append(
 220                     {
 221                         "title": tag_key.encode("utf-8"),
 222                         "value": tag_value.encode("utf-8"),
 223                         "short": True,
 224                     }
 225                 )
 226         payload = {
 227             "attachments": [
 228                 {
 229                     "fallback": "[%s] %s" % (project_name, title),
 230                     "title": title,
 231                     "title_link": group.get_absolute_url(params={"referrer": "slack"}),
 232                     "color": self.color_for_event(event),
 233                     "fields": fields,
 234                 }
 235             ]
 236         }
 237         client = self.get_client(project)
 238 
 239         if client.username:
 240             payload["username"] = client.username.encode("utf-8")
 241 
 242         if client.channel:
 243             payload["channel"] = client.channel
 244 
 245         if client.icon_url:
 246             payload["icon_url"] = client.icon_url
 247 
 248         values = {"payload": json.dumps(payload)}
 249         client.request(values)
 250 
 251     def get_client(self, project):
 252         webhook = self.get_option("webhook", project).strip()
 253         # Apparently we've stored some bad data from before we used `URLField`.
 254         username = (self.get_option("username", project) or "Sentry").strip()
 255         icon_url = self.get_option("icon_url", project)
 256         channel = (self.get_option("channel", project) or "").strip()
 257 
 258         return SlackApiClient(webhook, username, icon_url, channel)
