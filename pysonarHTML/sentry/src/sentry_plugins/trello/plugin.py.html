  53     def get_config(self, project, **kwargs):
  54         """
  55         Return the configuration of the plugin.
  56         Pull the value out of our the arguments to this function or from the DB
  57         """
  58 
  59         def get_value(field):
  60             initial_values = kwargs.get("initial", {})
  61             return initial_values.get(field) or self.get_option(field, project)
  62 
  63         token_config = {
  64             "name": "token",
  65             "type": "secret",
  66             "label": "Trello API Token",
  67             "default": None,
  68         }
  69 
  70         token_val = get_value("token")
  71         if token_val:
  72             # The token is sensitive so we should mask the value by only sending back the first 5 characters
  73             token_config["required"] = False
  74             token_config["prefix"] = token_val[:5]
  75             token_config["has_saved_value"] = True
  76         else:
  77             token_config["required"] = True
  78 
  79         api_key = get_value("key")
  80 
  81         key_config = {
  82             "name": "key",
  83             "type": "text",
  84             "required": True,
  85             "label": "Trello API Key",
  86             "default": api_key,
  87         }
  88 
  89         config = [key_config, token_config]
  90         org_value = get_value("organization")
  91         include_org = kwargs.get("add_additial_fields", org_value)
  92         if api_key and token_val and include_org:
  93             trello_client = TrelloApiClient(api_key, token_val)
  94             try:
  95                 org_options = trello_client.get_organization_options()
  96                 config.append(
  97                     {
  98                         "name": "organization",
  99                         "label": "Trello Organization",
 100                         "choices": org_options,
 101                         "type": "select",
 102                         "required": False,
 103                         "default": org_value,
 104                     }
 105                 )
 106             except Exception as e:
 107                 self.raise_error(e)
 108         return config

 110     def validate_config(self, project, config, actor=None):
 111         """
 112         Make sure the configuration is valid by trying to query for the organizations with the auth
 113         """
 114         trello_client = TrelloApiClient(config["key"], config["token"])
 115         try:
 116             trello_client.get_organization_options()
 117         except Exception as e:
 118             self.raise_error(e)
 119         return config

 202     def error_message_from_json(self, data):
 203         errors = data.get("errors")
 204         if errors:
 205             return " ".join(e["message"] for e in errors)
 206         return "unknown error"

 208     def create_issue(self, request, group, form_data, **kwargs):
 209         client = self.get_client(group.project)
 210 
 211         try:
 212             response = client.new_card(
 213                 id_list=form_data["list"], name=form_data["title"], desc=form_data["description"]
 214             )
 215         except Exception as e:
 216             self.raise_error(e)
 217 
 218         return response["shortLink"]

 220     def link_issue(self, request, group, form_data, **kwargs):
 221         client = self.get_client(group.project)
 222 
 223         try:
 224             card = client.get_card(form_data["issue_id"])
 225         except Exception as e:
 226             self.raise_error(e)
 227 
 228         comment = form_data.get("comment")
 229         if comment:
 230             try:
 231                 client.create_comment(card["shortLink"], comment)
 232             except Exception as e:
 233                 self.raise_error(e)
 234 
 235         return {"title": card["name"], "id": card["shortLink"]}

 237     def get_issue_label(self, group, issue, **kwargs):
 238         """
 239         Return label of the linked issue we show in the UI from the issue string
 240         """
 241         # the old version of the plugin stores the url in the issue
 242         if LABLEX_REGEX.search(issue):
 243             short_issue = issue.split("/", 1)[0]
 244             return "Trello-%s" % short_issue
 245         return "Trello-%s" % issue
 246 
 247     def get_issue_url(self, group, issue, **kwargs):
 248         """
 249         Return label of the url of card in Trello based off the issue object or issue ID
 250         """
 251         # TODO(Steve): figure out why we sometimes get a string and sometimes a dict
 252         if isinstance(issue, dict):
 253             issue = issue["id"]
 254         # the old version of the plugin stores the url in the issue
 255         if LABLEX_REGEX.search(issue):
 256             return issue.split("/", 1)[1]
 257         return "https://trello.com/c/%s" % issue

 259     def view_options(self, request, group, **kwargs):
 260         """
 261         Return the lists on a given Trello board
 262         """
 263         field = request.GET.get("option_field")
 264         board = request.GET.get("board")
 265 
 266         results = []
 267         if field == "list" and board:
 268             client = self.get_client(group.project)
 269             try:
 270                 response = client.get_lists_of_board(board)
 271             except Exception as e:
 272                 return Response(
 273                     {
 274                         "error_type": "validation",
 275                         "errors": [{"__all__": self.message_from_error(e)}],
 276                     },
 277                     status=400,
 278                 )
 279             else:
 280                 results = self.map_to_options(response)
 281 
 282         return Response({field: results})

 284     def view_autocomplete(self, request, group, **kwargs):
 285         """
 286         Return the cards matching a given query and the organization of the configuration
 287         """
 288         field = request.GET.get("autocomplete_field")
 289         query = request.GET.get("autocomplete_query")
 290 
 291         output = []
 292         if field == "issue_id" and query:
 293             organization = self.get_option("organization", group.project)
 294 
 295             client = self.get_client(group.project)
 296             cards = client.get_cards(query, organization)
 297             output = [
 298                 {"text": "(#%s) %s" % (card["idShort"], card["name"]), "id": card["shortLink"]}
 299                 for card in cards
 300             ]
 301 
 302         return Response({field: output})
