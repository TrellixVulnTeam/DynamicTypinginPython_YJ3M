  23     def __init__(self, api_key, token=None, **kwargs):

  28     def request(self, method="GET", path="", data=None, params=None, **kwargs):
  29         params = {} if params is None else params.copy()

  34     def get_organization_boards(self, org_id_or_name, fields=None):

  40     def get_member_boards(self, fields=None):

  46     def get_boards(self, org=None):

  74     def get_organization_options(self):
  75         """

  81     def get_cards(self, query, org_id=None):
  82         """
  83         Return the cards matching a query, limited to an org if passed in
  84         """
  85         params = {
  86             "query": query,
  87             "modelTypes": "cards",
  88             "cards_limit": 100,
  89             "partial": "true",
  90             "card_fields": CARD_FIELDS,
  91         }
  92         if org_id:
  93             params["idOrganizations"] = org_id
  94         response = self.request(path=SEARCH_PATH, params=params)
  95         return response["cards"]
