  82     def create_issue(self, request, group, form_data, **kwargs):
  83         client = self.get_client(group.project)
  84 
  85         try:
  86             response = client.create_story(
  87                 project=self.get_option("project", group.project), data=form_data
  88             )
  89         except Exception as e:
  90             self.raise_error(e)
  91 
  92         return {"id": response["id"], "title": response["name"], "url": response["app_url"]}

 100     def validate_config(self, project, config, actor):
 101         try:
 102             config["project"] = int(config["project"])
 103         except ValueError as exc:
 104             self.logger.exception(six.text_type(exc))
 105             raise PluginError(
 106                 "Invalid Project ID. "
 107                 "Project IDs are numbers-only, and can be found on the Project's page"
 108             )
 109         return config

 140     # Handle the incoming search terms, make requests and build responses
 141     def view_autocomplete(self, request, group, **kwargs):
 142         field = request.GET.get("autocomplete_field")
 143         query = request.GET.get("autocomplete_query")
 144         if field != "issue_id" or not query:
 145             return Response({"issue_id": []})
 146 
 147         project = self.get_option("project", group.project)
 148 
 149         client = self.get_client(group.project)
 150 
 151         # TODO: Something about the search API won't allow an explicit number search.
 152         # Should it switch the search mechanism from search_stories(text) to get_story(id)?
 153         try:
 154             response = client.search_stories(
 155                 query=(u"project:%s %s" % (project, query)).encode("utf-8")
 156             )
 157         except Exception as e:
 158             return self.handle_api_error(e)
 159 
 160         issues = [
 161             {"text": "(#%s) %s" % (i["id"], i["name"]), "id": i["id"]}
 162             for i in response.get("data", [])
 163         ]
 164 
 165         return Response({field: issues})

 167     def link_issue(self, request, group, form_data, **kwargs):
 168         client = self.get_client(group.project)
 169 
 170         try:
 171             story = client.get_story(story_id=form_data["issue_id"])
 172         except Exception as e:
 173             self.raise_error(e)
 174 
 175         comment = form_data.get("comment")
 176         if comment:
 177             try:
 178                 client.add_comment(story_id=story["id"], comment=comment)
 179             except Exception as e:
 180                 self.raise_error(e)
 181 
 182         return {"id": story["id"], "title": story["name"], "url": story["app_url"]}
