  61     def has_workspace_access(self, workspace, choices):
  62         for c, _ in choices:
  63             if workspace == c:
  64                 return True
  65         return False

  67     def get_workspace_choices(self, workspaces):
  68         return [(w["gid"], w["name"]) for w in workspaces["data"]]

  70     def get_new_issue_fields(self, request, group, event, **kwargs):
  71         fields = super(AsanaPlugin, self).get_new_issue_fields(request, group, event, **kwargs)
  72         client = self.get_client(request.user)
  73         workspaces = client.get_workspaces()
  74         workspace_choices = self.get_workspace_choices(workspaces)
  75         workspace = self.get_option("workspace", group.project)
  76         if workspace and not self.has_workspace_access(workspace, workspace_choices):
  77             workspace_choices.append((workspace, workspace))
  78 
  79         # use labels that are more applicable to asana
  80         for field in fields:
  81             if field["name"] == "title":
  82                 field["label"] = "Name"
  83             if field["name"] == "description":
  84                 field["label"] = "Notes"
  85                 field["required"] = False
  86 
  87         return (
  88             [
  89                 {
  90                     "name": "workspace",
  91                     "label": "Asana Workspace",
  92                     "default": workspace,
  93                     "type": "select",
  94                     "choices": workspace_choices,
  95                     "readonly": True,
  96                 }
  97             ]
  98             + fields
  99             + [

 103                     "type": "select",
 104                     "has_autocomplete": True,
 105                     "required": False,

 140     def get_client(self, user):
 141         auth = self.get_auth_for_user(user=user)
 142         if auth is None:
 143             raise PluginIdentityRequired(ERR_AUTH_NOT_CONFIGURED)
 144         return AsanaClient(auth=auth)

 146     def error_message_from_json(self, data):
 147         errors = data.get("errors")
 148         if errors:
 149             return " ".join([e["message"] for e in errors])
 150         return "unknown error"

 152     def create_issue(self, request, group, form_data, **kwargs):
 153         client = self.get_client(request.user)
 154 
 155         try:
 156             response = client.create_issue(
 157                 workspace=self.get_option("workspace", group.project), data=form_data
 158             )
 159         except Exception as e:
 160             self.raise_error(e, identity=client.auth)
 161 
 162         return response["data"]["gid"]

 164     def link_issue(self, request, group, form_data, **kwargs):
 165         client = self.get_client(request.user)
 166         try:
 167             issue = client.get_issue(issue_id=form_data["issue_id"])["data"]
 168         except Exception as e:
 169             self.raise_error(e, identity=client.auth)
 170 
 171         comment = form_data.get("comment")
 172         if comment:
 173             try:
 174                 client.create_comment(issue["gid"], {"text": comment})
 175             except Exception as e:
 176                 self.raise_error(e, identity=client.auth)
 177 
 178         return {"title": issue["name"]}

 186     def validate_config(self, project, config, actor):
 187         """
 188         ```
 189         if config['foo'] and not config['bar']:
 190             raise PluginError('You cannot configure foo with bar')
 191         return config
 192         ```
 193         """
 194         try:
 195             int(config["workspace"])
 196         except ValueError as exc:
 197             self.logger.exception(six.text_type(exc))
 198             raise PluginError("Non-numeric workspace value")
 199         return config

 201     def get_config(self, *args, **kwargs):
 202         user = kwargs["user"]
 203         try:
 204             client = self.get_client(user)
 205         except PluginIdentityRequired as e:
 206             self.raise_error(e)
 207         workspaces = client.get_workspaces()
 208         workspace_choices = self.get_workspace_choices(workspaces)
 209         workspace = self.get_option("workspace", kwargs["project"])
 210         # check to make sure the current user has access to the workspace
 211         helptext = None
 212         if workspace and not self.has_workspace_access(workspace, workspace_choices):
 213             workspace_choices.append((workspace, workspace))
 214             helptext = (
 215                 "This plugin has been configured for an Asana workspace "
 216                 "that either you don't have access to or doesn't "
 217                 "exist. You can edit the configuration, but you will not "
 218                 "be able to change it back to the current configuration "
 219                 "unless a teammate grants you access to the workspace in Asana."
 220             )
 221         return [
 222             {
 223                 "name": "workspace",
 224                 "label": "Workspace",
 225                 "type": "select",
 226                 "choices": workspace_choices,
 227                 "default": workspace or workspaces["data"][0]["gid"],
 228                 "help": helptext,
 229             }
 230         ]

 232     def view_autocomplete(self, request, group, **kwargs):
 233         field = request.GET.get("autocomplete_field")
 234         query = request.GET.get("autocomplete_query")
 235 
 236         client = self.get_client(request.user)
 237         workspace = self.get_option("workspace", group.project)
 238         results = []
 239         field_name = field
 240         if field == "issue_id":
 241             field_name = "task"
 242         elif field == "assignee":
 243             field_name = "user"
 244         try:
 245             response = client.search(workspace, field_name, query.encode("utf-8"))
 246         except Exception as e:
 247             return Response(
 248                 {"error_type": "validation", "errors": [{"__all__": self.message_from_error(e)}]},
 249                 status=400,
 250             )
 251         else:
 252             results = [
 253                 {"text": "(#%s) %s" % (i["gid"], i["name"]), "id": i["gid"]}
 254                 for i in response.get("data", [])
 255             ]
 256 
 257         return Response({field: results})
