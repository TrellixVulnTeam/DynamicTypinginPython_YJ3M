  85     def handle_api_error(self, error):
  86         msg = u"Error communicating with Pivotal Tracker"
  87         status = 400 if isinstance(error, PluginError) else 502
  88         return Response({"error_type": "validation", "errors": {"__all__": msg}}, status=status)

  90     def view_autocomplete(self, request, group, **kwargs):
  91         field = request.GET.get("autocomplete_field")
  92         query = request.GET.get("autocomplete_query")
  93         if field != "issue_id" or not query:
  94             return Response({"issue_id": []})
  95         query = query.encode("utf-8")
  96         _url = "%s?%s" % (self.build_api_url(group, "search"), urlencode({"query": query}))
  97         try:
  98             req = self.make_api_request(group.project, _url)
  99             body = safe_urlread(req)
 100         except (requests.RequestException, PluginError) as e:
 101             return self.handle_api_error(e)
 102 
 103         try:
 104             json_resp = json.loads(body)
 105 
 106         except ValueError as e:
 107             return self.handle_api_error(e)
 108 
 109         resp = json_resp.get("stories", {})
 110         stories = resp.get("stories", [])
 111         issues = [{"text": "(#%s) %s" % (i["id"], i["name"]), "id": i["id"]} for i in stories]
 112 
 113         return Response({field: issues})

 115     def link_issue(self, request, group, form_data, **kwargs):
 116         comment = form_data.get("comment")
 117         if not comment:
 118             return
 119         _url = "%s/%s/comments" % (self.build_api_url(group, "stories"), form_data["issue_id"])
 120         try:
 121             req = self.make_api_request(group.project, _url, json_data={"text": comment})
 122             body = safe_urlread(req)
 123         except requests.RequestException as e:
 124             msg = six.text_type(e)
 125             raise PluginError("Error communicating with Pivotal: %s" % (msg,))
 126 
 127         try:
 128             json_resp = json.loads(body)
 129         except ValueError as e:
 130             msg = six.text_type(e)
 131             raise PluginError("Error communicating with Pivotal: %s" % (msg,))
 132 
 133         if req.status_code > 399:
 134             raise PluginError(json_resp["error"])

 136     def build_api_url(self, group, pivotal_api=None):
 137         project = self.get_option("project", group.project)
 138 
 139         _url = "https://www.pivotaltracker.com/services/v5/projects/%s/%s" % (project, pivotal_api)

 143     def make_api_request(self, project, _url, json_data=None):
 144         req_headers = {
 145             "X-TrackerToken": six.text_type(self.get_option("token", project)),
 146             "Content-Type": "application/json",
 147         }
 148         return safe_urlopen(_url, json=json_data, headers=req_headers, allow_redirects=True)
 149 
 150     def create_issue(self, request, group, form_data, **kwargs):
 151         json_data = {
 152             "story_type": "bug",
 153             "name": force_text(form_data["title"], encoding="utf-8", errors="replace"),
 154             "description": force_text(form_data["description"], encoding="utf-8", errors="replace"),
 155             "labels": ["sentry"],
 156         }
 157 
 158         try:
 159             _url = self.build_api_url(group, "stories")
 160             req = self.make_api_request(group.project, _url, json_data=json_data)
 161             body = safe_urlread(req)
 162         except requests.RequestException as e:
 163             msg = six.text_type(e)
 164             raise PluginError("Error communicating with Pivotal: %s" % (msg,))
 165 
 166         try:
 167             json_resp = json.loads(body)
 168         except ValueError as e:
 169             msg = six.text_type(e)
 170             raise PluginError("Error communicating with Pivotal: %s" % (msg,))
 171 
 172         if req.status_code > 399:
 173             raise PluginError(json_resp["error"])
 174 
 175         return json_resp["id"]
