  18 class HerokuReleaseHook(ReleaseHook):
  19     def handle(self, request):
  20         email = None
  21         if "user" in request.POST:
  22             email = request.POST["user"]
  23         elif "actor" in request.POST:
  24             email = request.POST["actor"].get("email")
  25         try:
  26             user = User.objects.get(
  27                 email__iexact=email, sentry_orgmember_set__organization__project=self.project
  28             )
  29         except (User.DoesNotExist, User.MultipleObjectsReturned):
  30             user = None
  31             logger.info(
  32                 "owner.missing",
  33                 extra={
  34                     "organization_id": self.project.organization_id,
  35                     "project_id": self.project.id,
  36                     "email": email,
  37                 },
  38             )
  39         self.finish_release(version=request.POST["head_long"], url=request.POST["url"], owner=user)
  40 
  41     def set_refs(self, release, **values):
  42         if not values.get("owner", None):
  43             return
  44         # check if user exists, and then try to get refs based on version
  45         repo_project_option = ProjectOption.objects.get_value(
  46             project=self.project, key="heroku:repository"
  47         )
  48         deploy_project_option = (
  49             ProjectOption.objects.get_value(
  50                 project=self.project, key="heroku:environment", default="production"
  51             )
  52             or "production"
  53         )
  54         if repo_project_option:
  55             try:
  56                 repository = Repository.objects.get(
  57                     organization_id=self.project.organization_id, name=repo_project_option
  58                 )
  59             except Repository.DoesNotExist:
  60                 logger.info(
  61                     "repository.missing",
  62                     extra={
  63                         "organization_id": self.project.organization_id,
  64                         "project_id": self.project.id,
  65                         "repository": repo_project_option,
  66                     },
  67                 )
  68             else:
  69                 release.set_refs(
  70                     refs=[{"commit": release.version, "repository": repository.name}],
  71                     user=values["owner"],
  72                     fetch=True,
  73                 )
  74         # create deploy associated with release via ReleaseDeploysEndpoint
  75         endpoint = "/organizations/{}/releases/{}/deploys/".format(
  76             self.project.organization.slug, release.version
  77         )
  78         auth = ApiKey(organization=self.project.organization, scope_list=["project:write"])
  79         client.post(endpoint, data={"environment": deploy_project_option}, auth=auth)

  82 class HerokuPlugin(CorePluginMixin, ReleaseTrackingPlugin):
  83     author = "Sentry Team"
  84     author_url = "https://github.com/getsentry"
  85     title = "Heroku"
  86     slug = "heroku"
  87     description = "Integrate Heroku release tracking."
  88     required_field = "repository"
  89     feature_descriptions = [
  90         FeatureDescription(
  91             """
  92             Integrate Heroku release tracking.
  93             """,
  94             IntegrationFeatures.DEPLOYMENT,
  95         )
  96     ]
  97 
  98     def configure(self, project, request):
  99         return react_plugin_config(self, project, request)
 100 
 101     def can_enable_for_projects(self):
 102         return True
 103 
 104     def can_configure_for_project(self, project):
 105         return True
 106 
 107     def has_project_conf(self):
 108         return True
 109 
 110     def get_conf_key(self):
 111         return "heroku"
 112 
 113     def get_config(self, project, **kwargs):
 114         repo_list = list(Repository.objects.filter(organization_id=project.organization_id))
 115         if not ProjectOption.objects.get_value(project=project, key="heroku:repository"):
 116             choices = [("", "select a repo")]
 117         else:
 118             choices = []
 119         choices.extend([(repo.name, repo.name) for repo in repo_list])
 120         return [
 121             {
 122                 "name": "repository",
 123                 "label": "Respository",
 124                 "type": "select",
 125                 "required": True,
 126                 "choices": choices,
 127                 "help": "Select which repository you would like to be associated with this project",
 128             },
 129             {
 130                 "name": "environment",
 131                 "label": "Deploy Environment",
 132                 "type": "text",
 133                 "required": False,
 134                 "default": "production",
 135                 "help": "Specify an environment name for your Heroku deploys",
 136             },
 137         ]
 138 
 139     def get_release_doc_html(self, hook_url):
 140         return """
 141         <p>Add Sentry as a deploy hook to automatically track new releases.</p>
 142         <pre class="clippy">heroku addons:create deployhooks:http --url={hook_url}</pre>
 143         """.format(
 144             hook_url=hook_url
 145         )

 147     def get_release_hook(self):
 148         return HerokuReleaseHook
