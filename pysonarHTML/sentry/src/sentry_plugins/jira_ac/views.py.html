  26 class BaseJiraWidgetView(View):
  27     jira_auth = None
  28 
  29     def get_jira_auth(self):
  30         if self.jira_auth is None:
  31             self.jira_auth = get_jira_auth_from_request(self.request)
  32         return self.jira_auth
  33 
  34     def get_context(self):
  35         return {
  36             "ac_js_src": "https://connect-cdn.atl-paas.net/all.js",
  37             "login_url": absolute_uri(reverse("sentry-login")),
  38             "body_class": "",
  39         }
  40 
  41     def get_response(self, template, context=None):
  42         context = context or self.get_context()
  43         res = render_to_response(template, context, self.request)
  44         res["X-Frame-Options"] = "ALLOW-FROM %s" % self.request.GET["xdm_e"]
  45         return res

  48 class JiraUIWidgetView(BaseJiraWidgetView):
  49     def get(self, request, *args, **kwargs):
  50         try:
  51             # make sure this exists and is valid
  52             jira_auth = self.get_jira_auth()
  53         except (ApiError, JiraTenant.DoesNotExist, ExpiredSignatureError):
  54             return self.get_response("error.html")
  55 
  56         if request.user.is_anonymous():
  57             return self.get_response("signin.html")
  58 
  59         org = jira_auth.organization
  60         context = self.get_context()
  61         if org is None:
  62             context.update(
  63                 {
  64                     "error_message": (
  65                         "You still need to configure this plugin, which "
  66                         "can be done from the Manage Add-ons page."
  67                     )
  68                 }
  69             )
  70             return self.get_response("error.html", context)
  71 
  72         context.update(
  73             {
  74                 "sentry_api_url": absolute_uri(
  75                     "/api/0/organizations/%s/users/issues/" % (org.slug,)
  76                 ),
  77                 "issue_key": self.request.GET.get("issueKey"),
  78             }
  79         )
  80 
  81         return self.get_response("widget.html", context)

  84 class JiraConfigView(BaseJiraWidgetView):
  85     def get_context(self):
  86         context = super(JiraConfigView, self).get_context()
  87         context["body_class"] = "aui-page-focused aui-page-size-medium"
  88         return context
  89 
  90     def get(self, request, *args, **kwargs):
  91         try:
  92             jira_auth = self.get_jira_auth()
  93         except (ApiError, JiraTenant.DoesNotExist):
  94             return self.get_response("error.html")
  95 
  96         if request.user.is_anonymous():
  97             return self.get_response("signin.html")
  98 
  99         org = jira_auth.organization
 100         form_context = None
 101         if org:
 102             form_context = {"organization": org.id}
 103 
 104         form = JiraConfigForm([(o.id, o.name) for o in request.user.get_orgs()], form_context)
 105         context = self.get_context()
 106         context.update({"is_configured": jira_auth.is_configured(), "form": form})
 107 
 108         return self.get_response("config.html", context)
 109 
 110     def post(self, request, *args, **kwargs):
 111         try:
 112             jira_auth = get_jira_auth_from_request(request)
 113         except (ApiError, JiraTenant.DoesNotExist):
 114             self.get_response("error.html")
 115 
 116         if request.user.is_anonymous():
 117             return self.get_response("signin.html")
 118 
 119         orgs = self.request.user.get_orgs()
 120 
 121         form = JiraConfigForm([(o.id, o.name) for o in orgs], self.request.POST)
 122 
 123         if form.is_valid():
 124             try:
 125                 org = orgs.get(id=form.cleaned_data["organization"])
 126             except Organization.DoesNotExist:
 127                 errors = form._errors.setdefault("organization", ErrorList())
 128                 errors.append("Invalid organization")
 129             else:
 130                 jira_auth.update(organization=org)
 131 
 132         context = self.get_context()
 133         context.update({"is_configured": jira_auth.is_configured(), "form": form})
 134 
 135         return self.get_response("config.html", context)

 173 class JiraInstalledCallback(View):
 174     @method_decorator(csrf_exempt)
 175     def dispatch(self, request, *args, **kwargs):
 176         return super(JiraInstalledCallback, self).dispatch(request, *args, **kwargs)
 177 
 178     @method_decorator(csrf_exempt)
 179     def post(self, request, *args, **kwargs):
 180         registration_info = json.loads(request.body)
 181         JiraTenant.objects.create_or_update(
 182             client_key=registration_info["clientKey"],
 183             values={
 184                 "secret": registration_info["sharedSecret"],
 185                 "base_url": registration_info["baseUrl"],
 186                 "public_key": registration_info["publicKey"],
 187             },
 188         )
 189         return HttpResponse(json.dumps({}), content_type="application/json")
