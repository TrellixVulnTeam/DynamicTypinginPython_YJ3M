  20 class JiraClient(ApiClient):
  21     """
  22     The JIRA API Client, so you don't have to.
  23     """

  25     PROJECT_URL = "/rest/api/2/project"
  26     META_URL = "/rest/api/2/issue/createmeta"
  27     CREATE_URL = "/rest/api/2/issue"
  28     PRIORITIES_URL = "/rest/api/2/priority"
  29     VERSIONS_URL = "/rest/api/2/project/%s/versions"
  30     USERS_URL = "/rest/api/2/user/assignable/search"
  31     ISSUE_URL = "/rest/api/2/issue/%s"
  32     SEARCH_URL = "/rest/api/2/search/"

  37     def __init__(self, instance_uri, username, password):
  38         self.base_url = instance_uri.rstrip("/")
  39         self.username = username
  40         self.password = password
  41         super(JiraClient, self).__init__(verify_ssl=False)
  42 
  43     def request(self, method, path, data=None, params=None):
  44         if self.username and self.password:
  45             auth = self.username.encode("utf8"), self.password.encode("utf8")
  46         else:
  47             auth = None
  48         return self._request(method, path, data=data, params=params, auth=auth)

  50     def get_projects_list(self):
  51         return self.get_cached(self.PROJECT_URL)

  53     def get_create_meta(self, project):
  54         return self.get(
  55             self.META_URL, params={"projectKeys": project, "expand": "projects.issuetypes.fields"}
  56         )
  57 
  58     def get_create_meta_for_project(self, project):
  59         metas = self.get_create_meta(project)
  60         # We saw an empty JSON response come back from the API :(
  61         if not metas:
  62             return None
  63 
  64         # XXX(dcramer): document how this is possible, if it even is
  65         if len(metas["projects"]) > 1:
  66             raise ApiError("More than one project found.")
  67 
  68         try:
  69             return metas["projects"][0]
  70         except IndexError:
  71             return None

  95     def search_issues(self, project, query):
  96         # check if it looks like an issue id
  97         if re.search(r"^[A-Za-z]+-\d+$", query) and project.lower() in query.lower():
  98             jql = 'id="%s"' % query.replace('"', '\\"')
  99         else:
 100             jql = 'text ~ "%s"' % query.replace('"', '\\"')
 101         jql = 'project="%s" AND %s' % (project, jql)
 102         return self.get(self.SEARCH_URL, params={"jql": jql})

 104     # Steve(XXX): Might consider moving this method to the base plugin API client
 105     def get_cached(self, full_url):
 106         """
 107         Basic Caching mechanism for requests and responses. It only caches responses
 108         based on URL
 109         TODO: Implement GET attr in cache as well. (see self.create_meta for example)
 110         """
 111         key = "sentry-jira:" + md5(full_url, self.base_url).hexdigest()
 112         cached_result = cache.get(key)
 113         if not cached_result:
 114             cached_result = self.get(full_url)
 115             cache.set(key, cached_result, 60)
 116         return cached_result
