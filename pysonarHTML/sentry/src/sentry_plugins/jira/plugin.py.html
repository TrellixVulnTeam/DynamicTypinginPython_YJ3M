  77     def build_dynamic_field(self, group, field_meta):
  78         """
  79         Builds a field based on JIRA's meta field information
  80         """
  81         schema = field_meta["schema"]
  82 
  83         # set up some defaults for form fields
  84         fieldtype = "text"
  85         fkwargs = {"label": field_meta["name"], "required": field_meta["required"]}
  86         # override defaults based on field configuration
  87         if (
  88             schema["type"] in ["securitylevel", "priority"]
  89             or schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES["select"]
  90         ):
  91             fieldtype = "select"
  92             fkwargs["choices"] = self.make_choices(field_meta.get("allowedValues"))
  93         elif field_meta.get("autoCompleteUrl") and (
  94             schema.get("items") == "user" or schema["type"] == "user"
  95         ):
  96             fieldtype = "select"
  97             sentry_url = "/api/0/issues/%s/plugins/%s/autocomplete" % (group.id, self.slug)
  98             fkwargs["url"] = "%s?jira_url=%s" % (
  99                 sentry_url,
 100                 quote_plus(field_meta["autoCompleteUrl"]),
 101             )
 102             fkwargs["has_autocomplete"] = True
 103             fkwargs["placeholder"] = "Start typing to search for a user"
 104         elif schema["type"] in ["timetracking"]:
 105             # TODO: Implement timetracking (currently unsupported alltogether)
 106             return None
 107         elif schema.get("items") in ["worklog", "attachment"]:
 108             # TODO: Implement worklogs and attachments someday
 109             return None
 110         elif schema["type"] == "array" and schema["items"] != "string":
 111             fieldtype = "select"
 112             fkwargs.update(
 113                 {
 114                     "multiple": True,
 115                     "choices": self.make_choices(field_meta.get("allowedValues")),
 116                     "default": [],
 117                 }
 118             )
 119 
 120         # break this out, since multiple field types could additionally
 121         # be configured to use a custom property instead of a default.
 122         if schema.get("custom"):
 123             if schema["custom"] == JIRA_CUSTOM_FIELD_TYPES["textarea"]:
 124                 fieldtype = "textarea"
 125 
 126         fkwargs["type"] = fieldtype
 127         return fkwargs

 129     def get_issue_type_meta(self, issue_type, meta):
 130         issue_types = meta["issuetypes"]
 131         issue_type_meta = None
 132         if issue_type:
 133             matching_type = [t for t in issue_types if t["id"] == issue_type]
 134             issue_type_meta = matching_type[0] if len(matching_type) > 0 else None
 135 
 136         # still no issue type? just use the first one.
 137         if not issue_type_meta:
 138             issue_type_meta = issue_types[0]
 139 
 140         return issue_type_meta

 142     def get_new_issue_fields(self, request, group, event, **kwargs):
 143         fields = super(JiraPlugin, self).get_new_issue_fields(request, group, event, **kwargs)
 144 
 145         jira_project_key = self.get_option("default_project", group.project)
 146 
 147         client = self.get_jira_client(group.project)
 148         try:
 149             meta = client.get_create_meta_for_project(jira_project_key)
 150         except ApiError as e:
 151             raise PluginError(
 152                 u"JIRA responded with an error. We received a status code of {}".format(e.code)
 153             )
 154         except ApiUnauthorized:
 155             raise PluginError(
 156                 "JIRA returned: Unauthorized. "
 157                 "Please check your username, password, "
 158                 "instance and project in your configuration settings."
 159             )
 160 
 161         if not meta:
 162             raise PluginError(
 163                 "Error in JIRA configuration, no projects " "found for user %s." % client.username
 164             )
 165 
 166         # check if the issuetype was passed as a GET parameter
 167         issue_type = None
 168         if request is not None:
 169             if request.method == "POST":
 170                 issue_type = request.data.get("issuetype")
 171             else:
 172                 issue_type = request.GET.get("issuetype")
 173 
 174         if issue_type is None:
 175             issue_type = self.get_option("default_issue_type", group.project)
 176 
 177         issue_type_meta = self.get_issue_type_meta(issue_type, meta)
 178 
 179         issue_type_choices = self.make_choices(meta["issuetypes"])
 180 
 181         # make sure default issue type is actually
 182         # one that is allowed for project
 183         if issue_type:
 184             if not any((c for c in issue_type_choices if c[0] == issue_type)):
 185                 issue_type = issue_type_meta["id"]
 186 
 187         fields = (
 188             [
 189                 {
 190                     "name": "project",
 191                     "label": "Jira Project",
 192                     "choices": ((meta["id"], jira_project_key),),
 193                     "default": meta["id"],
 194                     "type": "select",
 195                     "readonly": True,
 196                 }
 197             ]
 198             + fields
 199             + [
 200                 {
 201                     "name": "issuetype",
 202                     "label": "Issue Type",
 203                     "default": issue_type or issue_type_meta["id"],
 204                     "type": "select",
 205                     "choices": issue_type_choices,
 206                 }
 207             ]
 208         )
 209 
 210         # title is renamed to summary before sending to JIRA
 211         standard_fields = [f["name"] for f in fields] + ["summary"]
 212         ignored_fields = (self.get_option("ignored_fields", group.project) or "").split(",")
 213 
 214         # apply ordering to fields based on some known built-in JIRA fields.
 215         # otherwise weird ordering occurs.
 216         anti_gravity = {"priority": -150, "fixVersions": -125, "components": -100, "security": -50}
 217 
 218         dynamic_fields = issue_type_meta.get("fields").keys()
 219         dynamic_fields.sort(key=lambda f: anti_gravity.get(f) or 0)
 220         # build up some dynamic fields based on required shit.
 221         for field in dynamic_fields:
 222             if field in standard_fields or field in [x.strip() for x in ignored_fields]:
 223                 # don't overwrite the fixed fields for the form.
 224                 continue
 225             mb_field = self.build_dynamic_field(group, issue_type_meta["fields"][field])
 226             if mb_field:
 227                 mb_field["name"] = field
 228                 fields.append(mb_field)
 229 
 230         for field in fields:
 231             if field["name"] == "priority":
 232                 # whenever priorities are available, put the available ones in the list.
 233                 # allowedValues for some reason doesn't pass enough info.
 234                 field["choices"] = self.make_choices(client.get_priorities())
 235                 field["default"] = self.get_option("default_priority", group.project) or ""
 236             elif field["name"] == "fixVersions":
 237                 field["choices"] = self.make_choices(client.get_versions(jira_project_key))
 238 
 239         return fields

 260     def link_issue(self, request, group, form_data, **kwargs):
 261         client = self.get_jira_client(group.project)
 262         try:
 263             issue = client.get_issue(form_data["issue_id"])
 264         except Exception as e:
 265             self.raise_error(e)
 266 
 267         comment = form_data.get("comment")
 268         if comment:
 269             try:
 270                 client.create_comment(issue["key"], comment)
 271             except Exception as e:
 272                 self.raise_error(e)
 273 
 274         return {"title": issue["fields"]["summary"]}

 291     def view_autocomplete(self, request, group, **kwargs):
 292         query = request.GET.get("autocomplete_query")
 293         field = request.GET.get("autocomplete_field")
 294         project = self.get_option("default_project", group.project)
 295 
 296         if field == "issue_id":
 297             client = self.get_jira_client(group.project)
 298             try:
 299                 response = client.search_issues(project, query)
 300             except ApiError as e:
 301                 return Response(
 302                     {
 303                         "error_type": "validation",
 304                         "errors": [{"__all__": self.message_from_error(e)}],
 305                     },
 306                     status=400,
 307                 )
 308             else:
 309                 issues = [
 310                     {"text": "(%s) %s" % (i["key"], i["fields"]["summary"]), "id": i["key"]}
 311                     for i in response.get("issues", [])
 312                 ]
 313                 return Response({field: issues})
 314 
 315         jira_url = request.GET.get("jira_url")
 316         if jira_url:
 317             jira_url = unquote_plus(jira_url)
 318             parsed = list(urlsplit(jira_url))
 319             jira_query = parse_qs(parsed[3])
 320 
 321             jira_client = self.get_jira_client(group.project)
 322 
 323             is_user_api = re.search("/rest/api/(latest|[0-9])/user/", jira_url)
 324 
 325             is_user_picker = "/rest/api/1.0/users/picker" in jira_url
 326 
 327             if is_user_api:  # its the JSON version of the autocompleter
 328                 is_xml = False
 329                 jira_query["username"] = query.encode("utf8")
 330                 jira_query.pop(
 331                     "issueKey", False
 332                 )  # some reason JIRA complains if this key is in the URL.
 333                 jira_query["project"] = project.encode("utf8")
 334             elif is_user_picker:
 335                 is_xml = False
 336                 # for whatever reason, the create meta api returns an
 337                 # invalid path, so let's just use the correct, documented one here:
 338                 # https://docs.atlassian.com/jira/REST/cloud/#api/2/user
 339                 # also, only pass path so saved instance url will be used
 340                 parsed[0] = ""
 341                 parsed[1] = ""
 342                 parsed[2] = "/rest/api/2/user/picker"
 343                 jira_query["query"] = query.encode("utf8")
 344             else:  # its the stupid XML version of the API.
 345                 is_xml = True
 346                 jira_query["query"] = query.encode("utf8")
 347                 if jira_query.get("fieldName"):
 348                     # for some reason its a list.
 349                     jira_query["fieldName"] = jira_query["fieldName"][0]
 350 
 351             parsed[3] = urlencode(jira_query)
 352             final_url = urlunsplit(parsed)
 353 
 354             autocomplete_response = jira_client.get_cached(final_url)
 355 
 356             if is_user_picker:
 357                 autocomplete_response = autocomplete_response["users"]
 358 
 359             users = []
 360 
 361             if is_xml:
 362                 for userxml in autocomplete_response.xml.findAll("users"):
 363                     users.append(
 364                         {"id": userxml.find("name").text, "text": userxml.find("html").text}
 365                     )
 366             else:
 367                 for user in autocomplete_response:
 368                     if user.get("name"):
 369                         users.append(self._get_formatted_user(user))
 370 
 371             # if JIRA user doesn't have proper permission for user api,
 372             # try the assignee api instead
 373             if not users and is_user_api:
 374                 try:
 375                     autocomplete_response = jira_client.search_users_for_project(
 376                         jira_query.get("project"), jira_query.get("username")
 377                     )
 378                 except (ApiUnauthorized, ApiError) as e:
 379                     return Response(
 380                         {
 381                             "error_type": "validation",
 382                             "errors": [{"__all__": self.message_from_error(e)}],
 383                         },
 384                         status=400,
 385                     )
 386 
 387                 for user in autocomplete_response:
 388                     if user.get("name"):
 389                         users.append(self._get_formatted_user(user))
 390 
 391             return Response({field: users})

 398     def error_message_from_json(self, data):
 399         message = ""
 400         if data.get("errorMessages"):
 401             message = " ".join(data["errorMessages"])
 402         if data.get("errors"):
 403             if message:
 404                 message += " "
 405             message += " ".join(["%s: %s" % (k, v) for k, v in data.get("errors").items()])
 406         return message

 408     def create_issue(self, request, group, form_data, **kwargs):
 409         cleaned_data = {}
 410 
 411         # protect against mis-configured plugin submitting a form without an
 412         # issuetype assigned.
 413         if not form_data.get("issuetype"):
 414             raise PluginError("Issue Type is required.")
 415 
 416         jira_project_key = self.get_option("default_project", group.project)
 417         client = self.get_jira_client(group.project)
 418         meta = client.get_create_meta_for_project(jira_project_key)
 419 
 420         if not meta:
 421             raise PluginError("Something went wrong. Check your plugin configuration.")
 422 
 423         issue_type_meta = self.get_issue_type_meta(form_data["issuetype"], meta)
 424 
 425         fs = issue_type_meta["fields"]
 426         for field in fs.keys():
 427             f = fs[field]
 428             if field == "description":
 429                 cleaned_data[field] = form_data[field]
 430                 continue
 431             elif field == "summary":
 432                 cleaned_data["summary"] = form_data["title"]
 433                 continue
 434             if field in form_data.keys():
 435                 v = form_data.get(field)
 436                 if v:
 437                     schema = f["schema"]
 438                     if schema.get("type") == "string" and not schema.get("custom"):
 439                         cleaned_data[field] = v
 440                         continue
 441                     if schema["type"] == "user" or schema.get("items") == "user":
 442                         v = {"name": v}
 443                     elif schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES.get("multiuserpicker"):
 444                         # custom multi-picker
 445                         v = [{"name": v}]
 446                     elif schema["type"] == "array" and schema.get("items") != "string":
 447                         v = [{"id": vx} for vx in v]
 448                     elif schema["type"] == "array" and schema.get("items") == "string":
 449                         v = [v]
 450                     elif schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES.get("textarea"):
 451                         v = v
 452                     elif (
 453                         schema["type"] == "number"
 454                         or schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES["tempo_account"]
 455                     ):
 456                         try:
 457                             if "." in v:
 458                                 v = float(v)
 459                             else:
 460                                 v = int(v)
 461                         except ValueError:
 462                             pass
 463                     elif (
 464                         schema.get("type") != "string"
 465                         or (schema.get("items") and schema.get("items") != "string")
 466                         or schema.get("custom") == JIRA_CUSTOM_FIELD_TYPES.get("select")
 467                     ):
 468                         v = {"id": v}
 469                     cleaned_data[field] = v
 470 
 471         if not (isinstance(cleaned_data["issuetype"], dict) and "id" in cleaned_data["issuetype"]):
 472             # something fishy is going on with this field, working on some JIRA
 473             # instances, and some not.
 474             # testing against 5.1.5 and 5.1.4 does not convert (perhaps is no longer included
 475             # in the projectmeta API call, and would normally be converted in the
 476             # above clean method.)
 477             cleaned_data["issuetype"] = {"id": cleaned_data["issuetype"]}
 478 
 479         try:
 480             response = client.create_issue(cleaned_data)
 481         except Exception as e:
 482             self.raise_error(e)
 483 
 484         return response.get("key")

 621                 "name": "auto_create",

 623                 "default": self.get_option("auto_create", project) or False,
 624                 "type": "bool",
 625                 "required": False,
 626                 "help": "Automatically create a JIRA ticket for EVERY new issue",
 627             },
 628         ]
 629 
 630     def should_create(self, group, event, is_new):
 631         if not is_new:
 632             return False
 633 
 634         if not self.get_option("auto_create", group.project):
 635             return False
 636 
 637         # XXX(dcramer): Sentry doesn't expect GroupMeta referenced here so we
 638         # need to populate the cache
 639         GroupMeta.objects.populate_cache([group])
 640         if GroupMeta.objects.get_value(group, "%s:tid" % self.get_conf_key(), None):
 641             return False
 642 
 643         return True
 644 
 645     def post_process(self, group, event, is_new, **kwargs):
 646         if not self.should_create(group, event, is_new):
 647             return
 648 
 649         fields = self.get_new_issue_fields(None, group, event, **kwargs)
 650 
 651         post_data = {}
 652         included_fields = set(["priority", "issuetype", "title", "description", "project"])
 653         for field in fields:
 654             name = field["name"]
 655             if name in included_fields:
 656                 post_data[name] = field.get("default")
 657 
 658         if not (
 659             post_data.get("priority") and post_data.get("issuetype") and post_data.get("project")
 660         ):
 661             return
 662 
 663         interface = event.interfaces.get("sentry.interfaces.Exception")
 664 
 665         if interface:
 666             post_data["description"] += "\n{code}%s{code}" % interface.get_stacktrace(
 667                 event, system_frames=False, max_frames=settings.SENTRY_MAX_STACKTRACE_FRAMES
 668             )
 669 
 670         try:
 671             issue_id = self.create_issue(request={}, group=group, form_data=post_data)
 672         except PluginError as e:
 673             logging.exception("Error creating JIRA ticket: %s", e)
 674         else:
 675             prefix = self.get_conf_key()
 676             GroupMeta.objects.set_value(group, "%s:tid" % prefix, issue_id)
