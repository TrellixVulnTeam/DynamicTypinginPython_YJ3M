  15 class PagerDutyPlugin(CorePluginMixin, NotifyPlugin):
  16     description = "Send alerts to PagerDuty."
  17     slug = "pagerduty"
  18     title = "PagerDuty"
  19     conf_key = slug
  20     conf_title = title
  21     required_field = "service_key"
  22     feature_descriptions = [
  23         FeatureDescription(
  24             """
  25             Manage incidents and outages by sending Sentry notifications to PagerDuty.
  26             """,
  27             IntegrationFeatures.INCIDENT_MANAGEMENT,
  28         ),
  29         FeatureDescription(
  30             """
  31             Configure rule based PagerDuty alerts to automatically be triggered in a specific
  32             service - or in multiple services!
  33             """,
  34             IntegrationFeatures.ALERT_RULE,
  35         ),
  36     ]

  38     def error_message_from_json(self, data):
  39         message = data.get("message", "unknown error")
  40         errors = data.get("errors", None)
  41         if errors:
  42             return "%s: %s" % (message, " ".join(errors))
  43 
  44         return message
  45 
  46     def is_configured(self, project, **kwargs):
  47         return bool(self.get_option("service_key", project))
  48 
  49     def get_config(self, **kwargs):
  50         service_key = self.get_option("service_key", kwargs["project"])
  51         secret_field = get_secret_field_config(
  52             service_key, "PagerDuty's Sentry service Integration Key", include_prefix=True
  53         )
  54         secret_field.update({"name": "service_key", "label": "Service Key"})
  55         return [
  56             secret_field,
  57             {
  58                 "name": "routes",
  59                 "label": "Tag routing",
  60                 "type": "textarea",
  61                 "placeholder": "environment,production,KEY1\ndevice,Other,KEY2",
  62                 "required": False,
  63                 "help": (
  64                     "Route each event to a different PagerDuty service key based "
  65                     "on the event's tags. Each line should contain a tag, "
  66                     "value, and service key separated by commas. The first "
  67                     "line that matches a tag's key and value will send to that "
  68                     "integration key instead of the main service key above."
  69                 ),
  70             },
  71         ]

  73     def notify_users(self, group, event, fail_silently=False, **kwargs):
  74         if not self.is_configured(group.project):
  75             return
  76 
  77         # TODO: This should eventually just be event.title in line with other plugins.
  78         # However, we should notify users first, since PD alert routing may be
  79         # based off the message field. We default to the title now because it's
  80         # possible for `event.message` to be "" and the description
  81         # is a required field for the PD payload.
  82         description = (event.message or event.title)[:1024]
  83 
  84         tags = dict(event.tags)
  85         details = {
  86             "event_id": event.event_id,
  87             "project": group.project.name,
  88             "release": event.get_tag("sentry:release"),
  89             "platform": event.platform,
  90             "culprit": event.culprit,
  91             "datetime": event.datetime.strftime("%Y-%m-%dT%H:%M:%S.%fZ"),
  92             "tags": tags,
  93             "url": group.get_absolute_url(params={"referrer": "pagerduty_plugin"}),
  94         }
  95 
  96         service_key = self.get_option("service_key", group.project)
  97 
  98         routes = self.get_option("routes", group.project) or ""
  99         for route in (r.strip() for r in routes.split("\n")):
 100             fields = [f.strip() for f in route.split(",")]
 101             if len(fields) != 3:
 102                 continue
 103             tag_key, tag_value, route_service_key = fields
 104             if tag_key in tags and tags[tag_key] == tag_value:
 105                 service_key = route_service_key
 106                 break
 107 
 108         client = PagerDutyClient(service_key=service_key)
 109         try:
 110             response = client.trigger_incident(
 111                 description=description,
 112                 event_type="trigger",
 113                 incident_key=six.text_type(group.id),
 114                 details=details,
 115                 contexts=[
 116                     {
 117                         "type": "link",
 118                         "href": absolute_uri(
 119                             group.get_absolute_url(params={"referrer": "pagerduty_plugin"})
 120                         ),
 121                         "text": "Issue Details",
 122                     }
 123                 ],
 124             )
 125             assert response["status"] == "success"
 126         except Exception as e:
 127             self.raise_error(e)
