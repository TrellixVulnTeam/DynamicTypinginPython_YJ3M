  19 class FreightReleaseHook(ReleaseHook):
  20     def handle(self, request):
  21         data = json.loads(request.body)
  22         if data["event"] == "started":
  23             self.start_release(version=data["sha"], ref=data["ref"], url=data["link"])
  24         elif data["event"] == "finished":
  25             self.finish_release(version=data["sha"], ref=data["ref"], url=data["link"])
  26         else:
  27             raise ValueError(data["event"])
