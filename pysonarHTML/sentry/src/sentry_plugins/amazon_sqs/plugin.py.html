  26 def get_regions():
  27     public_region_list = boto3.session.Session().get_available_regions("sqs")
  28     cn_region_list = boto3.session.Session().get_available_regions("sqs", partition_name="aws-cn")
  29     return public_region_list + cn_region_list

  32 def track_response_metric(fn):
  33     # https://boto3.amazonaws.com/v1/documentation/api/latest/reference/services/sqs.html#SQS.Queue.send_message
  34     # boto3's send_message doesn't return success/fail or http codes
  35     # success is a boolean based on whether there was an exception or not
  36     def wrapper(*args, **kwargs):
  37         try:
  38             success = fn(*args, **kwargs)
  39             metrics.incr(
  40                 "data-forwarding.http_response", tags={"plugin": "amazon-sqs", "success": success}
  41             )
  42         except Exception:
  43             metrics.incr(
  44                 "data-forwarding.http_response", tags={"plugin": "amazon-sqs", "success": False}
  45             )
  46             raise
  47         return success
  48 
  49     return wrapper

  52 class AmazonSQSPlugin(CorePluginMixin, DataForwardingPlugin):
  53     title = "Amazon SQS"
  54     slug = "amazon-sqs"
  55     description = DESCRIPTION
  56     conf_key = "amazon-sqs"
  57     required_field = "queue_url"
  58     feature_descriptions = [
  59         FeatureDescription(
  60             """
  61             Forward Sentry errors and events to Amazon SQS.
  62             """,
  63             IntegrationFeatures.DATA_FORWARDING,
  64         )
  65     ]
  66 
  67     def get_config(self, project, **kwargs):
  68         return [
  69             {
  70                 "name": "queue_url",
  71                 "label": "Queue URL",
  72                 "type": "url",
  73                 "placeholder": "https://sqs-us-east-1.amazonaws.com/12345678/myqueue",
  74             },
  75             {
  76                 "name": "region",
  77                 "label": "Region",
  78                 "type": "select",
  79                 "choices": tuple((z, z) for z in get_regions()),
  80             },
  81             get_secret_field_config(
  82                 name="access_key", label="Access Key", secret=self.get_option("access_key", project)
  83             ),
  84             get_secret_field_config(
  85                 name="secret_key", label="Secret Key", secret=self.get_option("secret_key", project)
  86             ),
  87             {
  88                 "name": "message_group_id",
  89                 "label": "Message Group ID",
  90                 "type": "text",
  91                 "required": False,
  92                 "placeholder": "Required for FIFO queues, exclude for standard queues",
  93             },
  94         ]
  95 
  96     @track_response_metric
  97     def forward_event(self, event, payload):
  98         queue_url = self.get_option("queue_url", event.project)
  99         access_key = self.get_option("access_key", event.project)
 100         secret_key = self.get_option("secret_key", event.project)
 101         region = self.get_option("region", event.project)
 102         message_group_id = self.get_option("message_group_id", event.project)
 103 
 104         if not all((queue_url, access_key, secret_key, region)):
 105             return
 106 
 107         # TODO(dcramer): Amazon doesnt support payloads larger than 256kb
 108         # We could support this by simply trimming it and allowing upload
 109         # to S3
 110         message = json.dumps(payload)
 111         if len(message) > 256 * 1024:
 112             return False
 113 
 114         try:
 115             client = boto3.client(
 116                 service_name="sqs",
 117                 aws_access_key_id=access_key,
 118                 aws_secret_access_key=secret_key,
 119                 region_name=region,
 120             )
 121 
 122             message = {"QueueUrl": queue_url, "MessageBody": message}
 123             # need a MessageGroupId for FIFO queues
 124             # note that if MessageGroupId is specified for non-FIFO, this will fail
 125             if message_group_id:
 126                 from uuid import uuid4
 127 
 128                 message["MessageGroupId"] = message_group_id
 129                 # if content based de-duplication is not enabled, we need to provide a
 130                 # MessageDeduplicationId
 131                 message["MessageDeduplicationId"] = uuid4().hex
 132             client.send_message(**message)
 133         except ClientError as e:
 134             if six.text_type(e).startswith(
 135                 "An error occurred (InvalidClientTokenId)"
 136             ) or six.text_type(e).startswith("An error occurred (AccessDenied)"):
 137                 # If there's an issue with the user's token then we can't do
 138                 # anything to recover. Just log and continue.
 139                 metrics_name = "sentry_plugins.amazon_sqs.access_token_invalid"
 140                 logger.info(
 141                     metrics_name,
 142                     extra={
 143                         "project_id": event.project.id,
 144                         "organization_id": event.project.organization_id,
 145                     },
 146                 )
 147 
 148                 metrics.incr(
 149                     metrics_name,
 150                     tags={
 151                         "project_id": event.project_id,
 152                         "organization_id": event.project.organization_id,
 153                     },
 154                 )
 155                 return False
 156             elif six.text_type(e).endswith("must contain the parameter MessageGroupId."):
 157                 metrics_name = "sentry_plugins.amazon_sqs.missing_message_group_id"
 158                 logger.info(
 159                     metrics_name,
 160                     extra={
 161                         "project_id": event.project.id,
 162                         "organization_id": event.project.organization_id,
 163                     },
 164                 )
 165                 metrics.incr(
 166                     metrics_name,
 167                     tags={
 168                         "project_id": event.project_id,
 169                         "organization_id": event.project.organization_id,
 170                     },
 171                 )
 172                 return False
 173             raise
 174         return True
