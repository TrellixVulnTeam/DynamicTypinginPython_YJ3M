  45 class SplunkError(Exception):
  46     def __init__(self, status_code, code=0, text="unknown error"):
  47         self.status_code = status_code
  48         self.code = code
  49         self.text = text
  50         super(SplunkError, self).__init__(text)
  51 
  52     @classmethod
  53     def from_response(cls, response):
  54         try:
  55             body = response.json()
  56         except Exception:
  57             return cls(
  58                 status_code=response.status_code, code=0, text="Unable to parse response body"
  59             )
  60 
  61         code = body.get("code")
  62         if code in SplunkInvalidToken.KNOWN_CODES:
  63             cls = SplunkInvalidToken
  64         elif code in SplunkServerBusy.KNOWN_CODES:
  65             cls = SplunkInvalidToken
  66         elif code in SplunkConfigError.KNOWN_CODES:
  67             cls = SplunkConfigError
  68         return cls(status_code=response.status_code, code=code, text=body.get("text"))
  69 
  70     def __repr__(self):
  71         return "<%s: status_code=%s, code=%s, text=%s>" % (
  72             type(self).__name__,
  73             self.status_code,
  74             self.code,
  75             self.text,
  76         )

 153     def get_host_for_splunk(self, event):
 154         host = event.get_tag("server_name")
 155         if host:
 156             return host
 157 
 158         user_interface = event.interfaces.get("sentry.interfaces.User")
 159         if user_interface:
 160             host = user_interface.ip_address
 161             if host:
 162                 return host
 163 
 164         return None

 166     def get_event_payload(self, event):
 167         props = {
 168             "event_id": event.event_id,
 169             "issue_id": event.group_id,
 170             "project_id": event.project.slug,
 171             "transaction": event.get_tag("transaction") or "",
 172             "release": event.get_tag("sentry:release") or "",
 173             "environment": event.get_tag("environment") or "",
 174             "type": event.get_event_type(),
 175         }
 176         props["tags"] = [[k.format(tagstore.get_standardized_key(k)), v] for k, v in event.tags]
 177         for key, value in six.iteritems(event.interfaces):
 178             if key == "request":
 179                 headers = value.headers
 180                 if not isinstance(headers, dict):
 181                     headers = dict(headers or ())
 182 
 183                 props.update(
 184                     {
 185                         "request_url": value.url,
 186                         "request_method": value.method,
 187                         "request_referer": headers.get("Referer", ""),
 188                     }
 189                 )
 190             elif key == "exception":
 191                 exc = value.values[0]
 192                 props.update({"exception_type": exc.type, "exception_value": exc.value})
 193             elif key == "logentry":
 194                 props.update({"message": value.formatted or value.message})
 195             elif key in ("csp", "expectct", "expectstable", "hpkp"):
 196                 props.update(
 197                     {
 198                         "{}_{}".format(key.rsplit(".", 1)[-1].lower(), k): v
 199                         for k, v in six.iteritems(value.to_json())
 200                     }
 201                 )
 202             elif key == "user":
 203                 user_payload = {}
 204                 if value.id:
 205                     user_payload["user_id"] = value.id
 206                 if value.email:
 207                     user_payload["user_email_hash"] = md5_text(value.email).hexdigest()
 208                 if value.ip_address:
 209                     user_payload["user_ip_trunc"] = anonymize_ip(value.ip_address)
 210                 if user_payload:
 211                     props.update(user_payload)
 212         return props

 214     # http://dev.splunk.com/view/event-collector/SP-CAAAE6M
 215     def post_process(self, event, **kwargs):
 216         token = self.get_option("token", event.project)
 217         index = self.get_option("index", event.project)
 218         instance = self.get_option("instance", event.project)
 219         if not (token and index and instance):
 220             metrics.incr(
 221                 "integrations.splunk.forward-event.unconfigured",
 222                 tags={
 223                     "project_id": event.project_id,
 224                     "organization_id": event.project.organization_id,
 225                     "event_type": event.get_event_type(),
 226                 },
 227             )
 228             return
 229 
 230         if not instance.endswith("/services/collector"):
 231             instance = instance.rstrip("/") + "/services/collector"
 232 
 233         source = self.get_option("source", event.project) or "sentry"
 234 
 235         rl_key = "splunk:{}".format(md5_text(token).hexdigest())
 236         # limit splunk to 50 requests/second
 237         if ratelimiter.is_limited(rl_key, limit=1000, window=1):
 238             metrics.incr(
 239                 "integrations.splunk.forward-event.rate-limited",
 240                 tags={
 241                     "project_id": event.project_id,
 242                     "organization_id": event.project.organization_id,
 243                     "event_type": event.get_event_type(),
 244                 },
 245             )
 246             return
 247 
 248         payload = {
 249             "time": int(event.datetime.strftime("%s")),
 250             "source": source,
 251             "index": index,
 252             "event": self.get_event_payload(event),
 253         }
 254         host = self.get_host_for_splunk(event)
 255         if host:
 256             payload["host"] = host
 257 
 258         session = http.build_session()
 259         try:
 260             # https://docs.splunk.com/Documentation/Splunk/7.2.3/Data/TroubleshootHTTPEventCollector
 261             resp = session.post(
 262                 instance,
 263                 json=payload,
 264                 # Splunk cloud instances certifcates dont play nicely
 265                 verify=False,
 266                 headers={"Authorization": "Splunk {}".format(token)},
 267                 timeout=5,
 268             )
 269             if resp.status_code != 200:
 270                 raise SplunkError.from_response(resp)
 271         except Exception as exc:
 272             metric = "integrations.splunk.forward-event.error"
 273             metrics.incr(
 274                 metric,
 275                 tags={
 276                     "project_id": event.project_id,
 277                     "organization_id": event.project.organization_id,
 278                     "event_type": event.get_event_type(),
 279                     "error_code": getattr(exc, "code", None),
 280                 },
 281             )
 282             logger.info(
 283                 metric,
 284                 extra={
 285                     "instance": instance,
 286                     "project_id": event.project_id,
 287                     "organization_id": event.project.organization_id,
 288                 },
 289             )
 290 
 291             if isinstance(exc, ReadTimeout):
 292                 # If we get a ReadTimeout we don't need to raise an error here.
 293                 # Just log and return.
 294                 return
 295             raise
 296 
 297         metrics.incr(
 298             "integrations.splunk.forward-event.success",
 299             tags={
 300                 "project_id": event.project_id,
 301                 "organization_id": event.project.organization_id,
 302                 "event_type": event.get_event_type(),
 303             },
 304         )
