  75     def _get_group_description(self, request, group, event):
  76         output = [absolute_uri(group.get_absolute_url())]
  77         body = self._get_group_body(request, group, event)
  78         if body:
  79             output.extend(["", "<pre>", body, "</pre>"])
  80         return "\n".join(output)

  87     def create_issue(self, group, form_data, **kwargs):
  88         """
  89         Create a Redmine issue
  90         """
  91         client = self.get_client(group.project)
  92         default_priority = self.get_option("default_priority", group.project)
  93         if default_priority is None:
  94             default_priority = 4
  95 
  96         issue_dict = {
  97             "project_id": self.get_option("project_id", group.project),
  98             "tracker_id": self.get_option("tracker_id", group.project),
  99             "priority_id": default_priority,
 100             "subject": form_data["title"].encode("utf-8"),
 101             "description": form_data["description"].encode("utf-8"),
 102         }
 103 
 104         extra_fields_str = self.get_option("extra_fields", group.project)
 105         if extra_fields_str:
 106             extra_fields = json.loads(extra_fields_str)
 107         else:
 108             extra_fields = {}
 109         issue_dict.update(extra_fields)
 110 
 111         response = client.create_issue(issue_dict)
 112         return response["issue"]["id"]

 114     def get_issue_url(self, group, issue_id, **kwargs):

 163     def add_choices(self, field_name, choices, default):
 164         for field in self.fields:
 165             if field_name == field["name"]:
 166                 field["choices"] = choices
 167                 field["default"] = default
 168                 return
 169 
 170     def remove_field(self, field_name):
 171         for field in self.fields:
 172             if field["name"] == field_name:
 173                 self.fields.remove(field)
 174                 return

 176     def build_initial(self, inital_args, project):
 177         initial = {}
 178         fields = ["host", "key", "project_id", "tracker_id", "default_priority", "extra_fields"]
 179         for field in fields:
 180             value = inital_args.get(field) or self.get_option(field, project)
 181             if value is not None:
 182                 initial[field] = value
 183         return initial
 184 
 185     def get_config(self, project, **kwargs):
 186         self.client_errors = []
 187         self.fields = self.build_config()
 188         initial_args = kwargs.get("initial") or {}
 189         initial = self.build_initial(initial_args, project)
 190 
 191         has_credentials = all(initial.get(k) for k in ("host", "key"))
 192         if has_credentials:
 193             client = RedmineClient(initial["host"], initial["key"])
 194             try:
 195                 projects = client.get_projects()
 196             except Exception:
 197                 has_credentials = False
 198                 self.client_errors.append("There was an issue authenticating with Redmine")
 199             else:
 200                 choices_value = self.get_option("project_id", project)
 201                 project_choices = [("", "--")] if not choices_value else []
 202                 project_choices += [
 203                     (p["id"], "%s (%s)" % (p["name"], p["identifier"]))
 204                     for p in projects["projects"]
 205                 ]
 206                 self.add_choices("project_id", project_choices, choices_value)
 207 
 208         if has_credentials:
 209             try:
 210                 trackers = client.get_trackers()
 211             except Exception:
 212                 self.remove_field("tracker_id")
 213             else:
 214                 choices_value = self.get_option("tracker_id", project)
 215                 tracker_choices = [("", "--")] if not choices_value else []
 216                 tracker_choices += [(p["id"], p["name"]) for p in trackers["trackers"]]
 217                 self.add_choices("tracker_id", tracker_choices, choices_value)
 218 
 219             try:
 220                 priorities = client.get_priorities()
 221             except Exception:
 222                 self.remove_field("default_priority")
 223             else:
 224                 choices_value = self.get_option("default_priority", project)
 225                 tracker_choices = [("", "--")] if not choices_value else []
 226                 tracker_choices += [(p["id"], p["name"]) for p in priorities["issue_priorities"]]
 227                 self.add_choices("default_priority", tracker_choices, choices_value)
 228 
 229         if not has_credentials:
 230             for field_name in ["project_id", "tracker_id", "default_priority", "extra_fields"]:
 231                 self.remove_field(field_name)
 232 
 233         return self.fields

 235     def validate_config(self, project, config, actor):
 236         super(RedminePlugin, self).validate_config(project, config, actor)
 237         self.client_errors = []
 238 
 239         for field in self.fields:
 240             if field["name"] in ["project_id", "tracker_id", "default_priority"]:
 241                 if not config[field["name"]]:
 242                     self.logger.exception(six.text_type("{} required.".format(field["name"])))
 243                     self.client_errors.append(field["name"])
 244 
 245         if self.client_errors:
 246             raise PluginError(", ".join(self.client_errors) + " required.")
 247         return config
