  10 class RedmineOptionsForm(forms.Form):
  11     host = forms.URLField(help_text=_("e.g. http://bugs.redmine.org"))
  12     key = forms.CharField(
  13         widget=forms.TextInput(attrs={"class": "span9"}),
  14         help_text="Your API key is available on your account page after enabling the Rest API (Administration -> Settings -> Authentication)",
  15     )
  16     project_id = forms.TypedChoiceField(label="Project", coerce=int)
  17     tracker_id = forms.TypedChoiceField(label="Tracker", coerce=int)
  18     default_priority = forms.TypedChoiceField(label="Default Priority", coerce=int)
  19     extra_fields = forms.CharField(
  20         widget=forms.Textarea(attrs={"rows": 5, "class": "span9"}),
  21         help_text="Extra attributes (custom fields, status id, etc.) in JSON format",
  22         label="Extra Fields",
  23         required=False,
  24     )

  26     def __init__(self, data=None, *args, **kwargs):
  27         super(RedmineOptionsForm, self).__init__(data=data, *args, **kwargs)
  28 
  29         initial = kwargs.get("initial") or {}
  30         for key, value in self.data.items():
  31             initial[key.lstrip(self.prefix or "")] = value
  32 
  33         has_credentials = all(initial.get(k) for k in ("host", "key"))
  34         if has_credentials:
  35             client = RedmineClient(initial["host"], initial["key"])
  36             try:
  37                 projects = client.get_projects()
  38             except Exception:
  39                 has_credentials = False
  40             else:
  41                 project_choices = [
  42                     (p["id"], "%s (%s)" % (p["name"], p["identifier"]))
  43                     for p in projects["projects"]
  44                 ]
  45                 self.fields["project_id"].choices = project_choices
  46 
  47         if has_credentials:
  48             try:
  49                 trackers = client.get_trackers()
  50             except Exception:
  51                 del self.fields["tracker_id"]
  52             else:
  53                 tracker_choices = [(p["id"], p["name"]) for p in trackers["trackers"]]
  54                 self.fields["tracker_id"].choices = tracker_choices
  55 
  56             try:
  57                 priorities = client.get_priorities()
  58             except Exception:
  59                 del self.fields["default_priority"]
  60             else:
  61                 tracker_choices = [(p["id"], p["name"]) for p in priorities["issue_priorities"]]
  62                 self.fields["default_priority"].choices = tracker_choices
  63 
  64         if not has_credentials:
  65             del self.fields["project_id"]
  66             del self.fields["tracker_id"]
  67             del self.fields["default_priority"]

  69     def clean(self):
  70         cd = self.cleaned_data
  71         if cd.get("host") and cd.get("key"):
  72             client = RedmineClient(cd["host"], cd["key"])
  73             try:
  74                 client.get_projects()
  75             except Exception:
  76                 raise forms.ValidationError("There was an issue authenticating with Redmine")
  77         return cd

  79     def clean_host(self):
  80         """
  81         Strip forward slashes off any url passed through the form.
  82         """
  83         url = self.cleaned_data.get("host")
  84         if url:
  85             return url.rstrip("/")
  86         return url

  88     def clean_extra_fields(self):
  89         """
  90         Ensure that the value provided is either a valid JSON dictionary,
  91         or the empty string.
  92         """
  93         extra_fields_json = self.cleaned_data.get("extra_fields").strip()
  94         if not extra_fields_json:
  95             return ""
  96 
  97         try:
  98             extra_fields_dict = json.loads(extra_fields_json)
  99         except ValueError:
 100             raise forms.ValidationError("Invalid JSON specified")
 101 
 102         if not isinstance(extra_fields_dict, dict):
 103             raise forms.ValidationError("JSON dictionary must be specified")
 104         return json.dumps(extra_fields_dict, indent=4)
