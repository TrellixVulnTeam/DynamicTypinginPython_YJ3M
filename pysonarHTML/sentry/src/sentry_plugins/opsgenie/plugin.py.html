  47 class OpsGeniePlugin(CorePluginMixin, notify.NotificationPlugin):
  48     author = "Sentry Team"
  49     author_url = "https://github.com/getsentry"
  50     title = "OpsGenie"
  51     slug = "opsgenie"
  52     description = DESCRIPTION
  53     conf_key = "opsgenie"
  54     version = sentry.VERSION
  55     project_conf_form = OpsGenieOptionsForm
  56     required_field = "api_key"
  57     feature_descriptions = [
  58         FeatureDescription(
  59             """
  60             Manage incidents and outages by sending Sentry notifications to OpsGenie.
  61             """,
  62             IntegrationFeatures.INCIDENT_MANAGEMENT,
  63         ),
  64         FeatureDescription(
  65             """
  66             Configure Sentry rules to trigger notifications based on conditions you set.
  67             """,
  68             IntegrationFeatures.ALERT_RULE,
  69         ),
  70     ]
  71 
  72     logger = logging.getLogger("sentry.plugins.opsgenie")
  73 
  74     def is_configured(self, project):
  75         return all((self.get_option(k, project) for k in ("api_key", "alert_url")))
  76 
  77     def get_form_initial(self, project=None):
  78         return {"alert_url": "https://api.opsgenie.com/v2/alerts"}
  79 
  80     def build_payload(self, group, event, triggering_rules):
  81         payload = {
  82             "message": event.message or event.title,
  83             "alias": "sentry: %d" % group.id,
  84             "source": "Sentry",
  85             "details": {
  86                 "Sentry ID": six.text_type(group.id),
  87                 "Sentry Group": getattr(group, "title", group.message).encode("utf-8"),
  88                 "Project ID": group.project.slug,
  89                 "Project Name": group.project.name,
  90                 "Logger": group.logger,
  91                 "Level": group.get_level_display(),
  92                 "URL": group.get_absolute_url(),
  93                 "Triggering Rules": json.dumps(triggering_rules),
  94             },
  95             "entity": group.culprit,
  96         }
  97 
  98         payload["tags"] = [
  99             "%s:%s" % (six.text_type(x).replace(",", ""), six.text_type(y).replace(",", ""))
 100             for x, y in event.tags
 101         ]
 102 
 103         return payload
 104 
 105     def notify_users(self, group, event, fail_silently=False, triggering_rules=None, **kwargs):
 106         if not self.is_configured(group.project):
 107             return
 108 
 109         client = self.get_client(group.project)
 110         payload = self.build_payload(group, event, triggering_rules)
 111         try:
 112             client.trigger_incident(payload)
 113         except Exception as e:
 114             self.raise_error(e)
 115 
 116     def get_client(self, project):
 117         api_key = self.get_option("api_key", project)
 118         alert_url = self.get_option("alert_url", project)
 119         recipients = self.get_option("recipients", project)
 120         return OpsGenieApiClient(api_key, alert_url, recipients)
