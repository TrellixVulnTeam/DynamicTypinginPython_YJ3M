  29 def query_to_result(field, result):
  30     if field == "issue_id":
  31         return u"T{}: {}".format(result["id"], result["fields"]["name"])
  32 
  33     if field == "assignee":
  34         return u"{} ({})".format(result["fields"]["realName"], result["fields"]["username"])
  35 
  36     return result["fields"]["name"]

 159     def validate_config(self, project, config, actor):
 160         projectPHIDs = config.get("projectPHIDs")
 161         if projectPHIDs:
 162             try:
 163                 json.loads(projectPHIDs)
 164             except ValueError:
 165                 raise PluginError("projectPHIDs field must be a valid JSON if present")
 166         if config.get("host") and (
 167             (config.get("username") and config.get("certificate")) or config.get("token")
 168         ):
 169             api = phabricator.Phabricator(
 170                 host=urljoin(config["host"], "api/"),
 171                 username=config.get("username"),
 172                 certificate=config.get("certificate"),
 173                 token=config.get("token"),
 174             )
 175             try:
 176                 api.user.whoami()
 177             except phabricator.APIError as e:
 178                 raise PluginError("%s %s" % (e.code, e))
 179             except HTTPException as e:
 180                 raise PluginError("Unable to reach Phabricator host: %s" % (e,))
 181             except Exception as e:
 182                 raise PluginError("Unhandled error from Phabricator: %s" % (e,))
 183         return config

 185     def is_configured(self, request, project, **kwargs):
 186         if not self.get_option("host", project):
 187             return False
 188         if self.get_option("token", project):
 189             return True
 190         if self.get_option("username", project) and self.get_option("certificate", project):
 191             return True
 192         return False

 204     def view_autocomplete(self, request, group, **kwargs):
 205         field = request.GET.get("autocomplete_field")
 206         query = request.GET.get("autocomplete_query")
 207 
 208         try:
 209             api = self.get_api(group.project)
 210             if field == "tags":
 211                 response = api.project.search(constraints={"name": query})
 212             elif field == "issue_id":
 213                 response = api.maniphest.search(constraints={"query": query})
 214             elif field == "assignee":
 215                 response = api.user.search(constraints={"nameLike": query})
 216 
 217         except Exception as e:
 218             return self.handle_api_error(e)
 219 
 220         results = [
 221             {"text": query_to_result(field, i), "id": i["phid"]} for i in response.get("data", [])
 222         ]
 223 
 224         return Response({field: results})

 226     def create_issue(self, request, group, form_data, **kwargs):
 227         api = self.get_api(group.project)
 228         try:
 229             data = api.maniphest.createtask(
 230                 title=form_data["title"].encode("utf-8"),
 231                 description=form_data["description"].encode("utf-8"),
 232                 ownerPHID=form_data.get("assignee"),
 233                 projectPHIDs=form_data.get("tags"),
 234             )
 235         except phabricator.APIError as e:
 236             raise PluginError("%s %s" % (e.code, e))
 237         except HTTPException as e:
 238             raise PluginError("Unable to reach Phabricator host: %s" % e)
 239 
 240         return data["id"]
 241 
 242     def link_issue(self, request, group, form_data, **kwargs):
 243         api = self.get_api(group.project)
 244 
 245         try:
 246             results = api.maniphest.search(constraints={"phids": [form_data["issue_id"]]})
 247         except Exception as e:
 248             self.raise_error(e)
 249 
 250         task = results["data"][0]
 251 
 252         comment = form_data.get("comment")
 253         if comment:
 254             try:
 255                 api.maniphest.edit(
 256                     objectIdentifier=form_data["issue_id"],
 257                     transactions=[{"type": "comment", "value": comment}],
 258                 )
 259             except Exception as e:
 260                 self.raise_error(e)
 261 
 262         return {
 263             "id": task["id"],
 264             "title": task["fields"]["name"],
 265             "url": self.get_issue_url(group, task["id"]),
 266         }
