  54 def deep_merge(defaults, overrides):
  55     """
  56     Deep merges two dictionaries.
  57 
  58     If the value is None in `overrides`, that key-value pair will not show up in the final result.
  59     """
  60     merged = {}
  61     for key in defaults:
  62         if isinstance(defaults[key], dict):
  63             if key not in overrides:
  64                 merged[key] = defaults[key]
  65             elif overrides[key] is None:
  66                 continue
  67             elif isinstance(overrides[key], dict):
  68                 merged[key] = deep_merge(defaults[key], overrides[key])
  69             else:
  70                 raise Exception("Types must match")
  71         elif key in overrides and overrides[key] is None:
  72             continue
  73         elif key in overrides:
  74             merged[key] = overrides[key]
  75         else:
  76             merged[key] = defaults[key]
  77     return merged

  80 @contextmanager
  81 def apidoc_containers():
  82     network = get_or_create(client, "network", namespace)
  83 
  84     containers = deep_merge(devservices_settings, apidoc_containers_overrides)
  85 
  86     # Massage our list into some shared settings instead of repeating
  87     # it for each definition.
  88     for name, options in containers.items():
  89         options["network"] = namespace
  90         options["detach"] = True
  91         options["name"] = namespace + "_" + name
  92         containers[name] = options
  93 
  94     # Pull all of our unique images once.
  95     pulled = set()
  96 
  97     for name, options in containers.items():
  98         if options["image"] not in pulled:
  99             click.secho("> Pulling image '%s'" % options["image"], err=True, fg="green")
 100             client.images.pull(options["image"])
 101             pulled.add(options["image"])
 102 
 103     # Run each of our containers, if found running already, delete first
 104     # and create new. We never want to reuse.
 105     for name, options in containers.items():
 106         try:
 107             container = client.containers.get(options["name"])
 108         except docker.errors.NotFound:
 109             pass
 110         else:
 111             container.stop()
 112             container.remove()
 113 
 114         click.secho("> Creating '%s' container" % options["name"], err=True, fg="yellow")
 115         client.containers.run(**options)
 116 
 117     yield
 118 
 119     # Delete all of our containers now. If it's not running, do nothing.
 120     for name, options in containers.items():
 121         try:
 122             container = client.containers.get(options["name"])
 123         except docker.errors.NotFound:
 124             pass
 125         else:
 126             click.secho("> Removing '%s' container" % container.name, err=True, fg="red")
 127             container.stop()
 128             container.remove()
 129 
 130     # Remove our network that we created.
 131     click.secho("> Removing '%s' network" % network.name, err=True, fg="red")
 132     network.remove()

 135 def color_for_string(s):
 136     colors = ("red", "green", "yellow", "blue", "cyan", "magenta")
 137     return colors[zlib.crc32(s) % len(colors)]

 273 def output_markdown(sections, scenarios, section_mapping):
 274     report("docs", "Generating markdown documents")
 275 
 276     # With nested URLs, we can have groups of URLs that are nested under multiple base URLs. We only want
 277     # them to show up once in the index.md. So, keep a set of endpoints we have already processed
 278     # to avoid duplication.
 279     processed_endpoints = set()
 280 
 281     for section, title in sections.items():
 282         i = 0
 283         links = []
 284         for endpoint in section_mapping.get(section, []):
 285             i += 1
 286             path = u"{}/{}.md".format(section, endpoint["endpoint_name"])
 287 
 288             auth = ""
 289             if len(endpoint["params"].get("auth", [])):
 290                 auth = endpoint["params"]["auth"][0]["description"]
 291             payload = dict(
 292                 title=endpoint["title"],
 293                 sidebar_order=i,
 294                 description="\n".join(endpoint["text"]).strip(),
 295                 warning=endpoint["warning"],
 296                 method=endpoint["method"],
 297                 api_path=endpoint["path"],
 298                 query_parameters=endpoint["params"].get("query"),
 299                 path_parameters=endpoint["params"].get("path"),
 300                 parameters=endpoint["params"].get("param"),
 301                 authentication=auth,
 302                 example_request=format_request(endpoint, scenarios),
 303                 example_response=format_response(endpoint, scenarios),
 304             )
 305             dump_markdown(path, payload)
 306 
 307             if path not in processed_endpoints:
 308                 links.append({"title": endpoint["title"], "path": path})
 309                 processed_endpoints.add(path)
 310         dump_index_markdown(section, title, links)

 354 def find_first_scenario(endpoint, scenario_map):
 355     for scene in endpoint["scenarios"]:
 356         if scene not in scenario_map:
 357             continue
 358         try:
 359             return scenario_map[scene]["requests"][0]
 360         except IndexError:
 361             return None
 362     return None

 365 def format_request(endpoint, scenario_map):
 366     scene = find_first_scenario(endpoint, scenario_map)
 367     if not scene:
 368         return ""
 369     request = scene["request"]
 370     lines = [
 371         u"{} {} HTTP/1.1".format(request["method"], request["path"]),
 372         "Host: sentry.io",
 373         "Authorization: Bearer <token>",
 374     ]
 375     lines.extend(format_headers(request["headers"]))
 376     if request["data"]:
 377         lines.append("")
 378         lines.append(json.dumps(request["data"], sort_keys=True, indent=2))
 379     return "\n".join(lines)
 380

 382 def format_response(endpoint, scenario_map):
 383     scene = find_first_scenario(endpoint, scenario_map)
 384     if not scene:
 385         return ""
 386     response = scene["response"]
 387     lines = [u"HTTP/1.1 {} {}".format(response["status"], response["reason"])]
 388     lines.extend(format_headers(response["headers"]))
 389     if response["data"]:
 390         lines.append("")
 391         lines.append(json.dumps(response["data"], sort_keys=True, indent=2))
 392     return "\n".join(lines)

 395 def format_headers(headers):
 396     """Format headers into a list."""
 397     return [u"{}: {}".format(key, value) for key, value in headers.items()]
