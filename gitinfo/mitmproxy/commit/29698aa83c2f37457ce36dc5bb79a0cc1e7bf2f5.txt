+	@echo "Build finished. The XML files are in $(BUILDDIR)/xml."

+	@echo "Build finished. The pseudo-XML files are in $(BUILDDIR)/pseudoxml."

+livehtml:
+	sphinx-autobuild -b html $(ALLSPHINXOPTS) $(BUILDDIR)/html
