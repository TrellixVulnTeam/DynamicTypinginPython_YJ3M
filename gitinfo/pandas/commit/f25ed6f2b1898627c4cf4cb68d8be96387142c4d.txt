+    def test_join_multiindex_leftright(self):
+        # GH 10741
+        df1 = pd.DataFrame(
+            [
+                ["a", "x", 0.471780],
+                ["a", "y", 0.774908],
+                ["a", "z", 0.563634],

+                ["b", "y", 0.368062],
+                ["b", "z", -1.721840],
+                ["c", "x", 1],

+        df2 = pd.DataFrame(

+                [0.774908, 10],

+                [0.368062, 20],

+                [1.000000, np.nan],
+                [2.000000, np.nan],

+            columns=["value1", "value2"],
