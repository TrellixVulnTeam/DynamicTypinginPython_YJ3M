+def translateGrouping(how):
+    if how == 'olhc':
+        return {'open' : lambda arr: arr[0],
+                'low' : lambda arr: arr.min(),
+                'high' : lambda arr: arr.max(),
+                'close' : lambda arr: arr[-1]}
+
+    if how == 'last':
+        return lambda arr: arr[-1]
+
+    raise ValueError("Unrecognized method: %s" % how)
