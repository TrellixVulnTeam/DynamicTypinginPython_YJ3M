+    def is_full(self):
+        """
+        Returns True if there are any missing periods from start to end
+        """
+        if len(self) == 0:
+            return True
+        if not self.is_monotonic:
+            raise ValueError('Index is not monotonic')
+        values = self.values
+        return ((values[1:] - values[:-1]) < 2).all()
