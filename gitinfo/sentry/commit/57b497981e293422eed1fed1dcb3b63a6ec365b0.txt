+

+    [RavenClient clientWithDSN:@"{% if dsn %}{{ dsn }}{% else %}<strong>SENTRY_DSN</strong>{% endif %}"];
+
+    // Install the global error handler
+    [[RavenClient sharedClient] setupExceptionHandler];
+
+    return YES;
+}
+</pre>
