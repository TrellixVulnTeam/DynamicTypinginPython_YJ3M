+def cookiejar_from_dict(cookie_dict, domain=None):
+    """Returns a CookieJar from a key/value dictoinary."""
+
+    # create cookiejar
+    cj = cookielib.CookieJar()
+
+    for k, v in cookie_dict.items():
+
+        # create cookie
+        ck = Cookie.SimpleCookie()
+        ck.name = v
+        ck.expires = 0
+        ck.path = '/'
+        ck.domain = domain
+
+        # add cookie to cookiejar
+        cj.set_cookie(ck)
+
+    return cj
