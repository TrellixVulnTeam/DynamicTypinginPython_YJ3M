+        ])
+
+        @app.callback(

+             Input('button-2', 'n_clicks'),
+             Input('button-2', 'n_clicks_previous')])
+        def update_output(*args):
+            call_count.value += 1
+            return '{}, {}, {}, {}'.format(*args)
