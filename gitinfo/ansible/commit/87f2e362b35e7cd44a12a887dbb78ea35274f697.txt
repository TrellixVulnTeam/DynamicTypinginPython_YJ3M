+- name: run a command while in check mode
+  shell: echo "running"
+  always_run: yes
+  register: result
+
+- name: assert that the command was run
+  assert:
+    that:
+    - "result.changed == true"
+    - "result.stdout == 'running'"
+    - "result.rc == 0"
