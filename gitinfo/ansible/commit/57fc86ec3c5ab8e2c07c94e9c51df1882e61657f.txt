+# UNINSTALL
+- name: uninstall sos
+  yum: name=sos state=removed
+  register: yum_result
+
+- name: check sos with rpm
+  shell: rpm -q sos
+  failed_when: False
+  register: rpm_result
+
+- debug: var=yum_result
+- debug: var=rpm_result
+
+- name: verify uninstalltion of sos
+  assert:
+    that: 
+        - "yum_result.rc == 0"
+        - "rpm_result.rc == 1"
+
+# INSTALL
+- name: install sos
+  yum: name=sos state=present
+  register: yum_result
+
+- name: check sos with rpm
+  shell: rpm -q sos
+  failed_when: False
+  register: rpm_result
+
+- debug: var=yum_result
+- debug: var=rpm_result
+
+- name: verify installation of sos
+  assert:
+    that: 
+        - "yum_result.rc == 0"
+        - "yum_result.changed"
+        - "rpm_result.rc == 0"
+
