+---
+module: net_static_route
+version_added: "2.4"
+author: "Ricardo Carrillo Cruz (@rcarrillocruz)"
+short_description: Manage static IP routes on network devices
+description:
+  - This module provides declarative management of static
+    IP routes on network devices.
+options:
+  prefix:
+    description:
+      - Network prefix of the static route.
+    required: true
+  mask:
+    description:
+      - Network prefix mask of the static route.
+    required: true
+  next_hop:
+    description:
+      - Next hop IP of the static route.
+    required: true
+  admin_distance:
+    description:
+      - Admin distance of the static route.
+  collection:
+    description: List of static route definitions
+  purge:
+    description:
+      - Purge static routes not defined in the collections parameter.
+    default: no
+  state:
+    description:
+      - State of the static route configuration.
+    default: present
+    choices: ['present', 'absent']
+"""
+
+EXAMPLES = """
+- name: configure static route
+  net_static_route:
+    prefix: 192.168.2.0
+    mask: 255.255.255.0
+    next_hop: 10.0.0.1
+
+- name: remove configuration
+  net_static_route:
+    prefix: 192.168.2.0
+    mask: 255.255.255.0
+    next_hop: 10.0.0.1
+    state: absent
+
+- name: configure collections of static routes
+  net_static_route:
+    collection:
+      - { prefix: 192.168.2.0, mask 255.255.255.0, next_hop: 10.0.0.1 }
+      - { prefix: 192.168.3.0, mask 255.255.255.0, next_hop: 10.0.2.1 }
+"""

+  returned: always
+  type: list
+  sample:

+"""
