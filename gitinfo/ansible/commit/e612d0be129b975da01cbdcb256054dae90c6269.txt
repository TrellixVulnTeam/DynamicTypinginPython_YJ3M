+  trunk_vlans:

+  native_vlan:

+      - Native VLAN to be configured in trunk port.
+  trunk_allowed_vlans:
+    description:
+      - List of allowed VLAN's in a given trunk port.
+  state:
+    description:
+      - State of the Layer-2 Interface configuration.
+    default: present
+    choices: ['present', 'absent',]
